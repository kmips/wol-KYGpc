يࣵلَّ مُسَلْ نَ نُوحٍ
ݧُ جُڪّࣹي ڪࣷ ݖِ تَوْرࣹيتُ مُوسَىٰ
نجࣵلْبࣺينْ گَ، سَارْ 9.6-17.9
اَسْڪَنُ نُوحٍ
‏1 9 اَسْڪَنُ نُوحٍ مࣷويْ لِي؞
نُوحٍ نِتْ ڪُ جُبْ لَ وࣷونْ،
دِ ڪُ مَتْ ݖِ بِيرْ نِتْ ݧِ مُ بࣷڪَّلْ جَمࣷنࣷ،
تࣹ دِ ڪُيْ تࣷݒّ يࣵلَّ؞
‏2 10 نُوحٍ نَگْ اَمࣷونْ نَ ݧࣹتِّ دࣷومْ يُ گٗورْ:
سࣹينِ تُرْ دِ سࣹمْ،
خَمْ اَڪْ يَفࣹتْ؞
‏3 11 بࣷوبَ فࣹڪّ نَ اࣵدِّنَ يࣵقُ،
تࣹ فࣹيسْ اَڪْ اَيْ،
يࣵلَّ تࣹگْ ݖِ بࣴتَمْ؞
‏4 12 يࣵلَّ گِسْ نࣹ اࣵدِّنَ يࣵقُ نَ،
ندَخْتࣹ جࣴفِ ݧࣺݒَّ بࣷڪّ ݧَاوْ فِ ڪَوْ سُوفْ؞
‏5 13 نࣷونُ يࣵلَّ نࣹ نُوحٍ:
«دࣷگَلْ نَا نࣹ دِنَا بࣷولࣹ فَاتْ مبࣷولࣹمْ بࣷرࣷومْ بَڪَّنْ،
ندَخْتࣹ اَيُ نِتْ دَجَلْ نَ ڪَوْ سُوفْ؞
دِنَا لࣹينْ بࣷولࣹ اَڪْ لِ ݖِ اࣵدِّنَ،
فَاگَاگَلْ؞
‏6 14 «يَوْ نَگْ وُتَلْ دࣺنڪُ سِݒَّرْ،
يࣹتَّلْ ݖِ سَ بࣷݒّ گَالْ گُ مَگْ؛
نگَ سࣵڪّ ݖِ اَيْ نࣺيگْ،
تࣹ دِوْ ڪࣷ ڪࣷلْتَارْ بِيرْ اَڪْ بِتِ؞
‏7 15 نِ نگَ ڪࣷيْ دࣹفَرࣹ نِي لَ؞
نَ گُدَّايُ گَالْ گِ دِ ݧࣹتِّ تࣺيمࣺيرِ خَسَبْ،
يَاتُوَايْ بِ دِ جُرٗومْ فُڪِّ خَسَبْ،
تَخَوَايْ بِ دِ فَنْوࣹيرِ خَسَبْ؞
‏8 16 يࣹتَّلْ مبَارْ،
مُ ڪࣷ تِيمْ،
تࣹ نگَ اࣵگّلࣹ گَالْ گِ،
بࣵيِّ لُ تࣷلّࣷوڪْ خَسَبْ دِگَّنْتࣹ ڪࣺمࣺجْ يِ اَڪْ مبَارْ مِ؞
سࣵڪَّلْ ڪࣷ بُنتّ ݖِ وࣹتْ گِ،
تࣹ سࣺدَّلࣹ ڪࣷ ݧࣹتِّ نࣺيگْ يُ تࣹگْلࣷو،
بࣹنّ ݖِ سُوفْ،
بࣹنّ ݖِ دِگّ بِ اَڪْ بࣹنّ ݖِ ڪَوْ؞
‏9 17 «مَا نگِ نِي دِ اِنْدِ مبࣴنُّ ندࣷخْ ݖِ ڪَوْ سُوفْ،
مُ فَرْ لࣺݒّ لُيْ دُندّ لُ اَسَمَانْ تِيمْ؛
تࣹ لࣺݒّ لُ نࣹڪّ ݖِ ڪَوْ سُوفْ دِنَ دࣹي؞
‏10 18 وَايࣹ دِنَا فَسْ ڪٗلّࣴرࣹ اَڪْ يَوْ،
نگَ دُگّ ݖِ گَالْ گِ،
يَاڪْ سَيْ دࣷومْ اَڪْ سَ سࣷخْنَ اَڪْ سَيْ سࣷخْنَيْ دࣷومْ؞
‏11 19 تࣹ خࣹيتِ رَبْ يࣺݒّ نَنگَ ݖِ دُگَلْ ݧَارْ ݖِ گَالْ گِ،
وُ ݖِ نࣹڪّ،
نگَ دࣹفْ ڪࣷ گٗورْ اَڪْ جِگࣺينْ؛
سُ ڪࣷ دࣹفࣹي نگࣹينْ اࣵندّ مُݖّ؞
‏12 20 خࣹيتُ ݒِݖّ وُ ݖِ نࣹڪّ اَڪْ خࣹيتُ بࣵيِّمَ وُ ݖِ نࣹڪّ اَڪْ وࣺݒّ خࣹيتُ رَبْ وُيْ رَامْ ݖِ سُوفْ،
دِنَ دࣹفْ ݧَارْ،
ݧࣴوْ ݖِ يَوْ،
بَ رَوْ؞
‏13 21 سࣵڪَّلْ ݖِ بࣺݒّ خࣹيتُ دُندّ،
دࣹنݖّ،
نگࣹينْ بࣷڪّ دُنْدࣹ؞»
‏14 22 نࣷونُ لَ نُوحٍ دࣹفْ؞
لࣺݒّ لُ ڪࣷ يࣵلَّ سَنتّ،
دࣹفْ نَ ڪࣷ؞
يࣵلَّ مُسَلْ نَ نُوحٍ ݖِ مبࣴنّ مِ
‏15 7 1 گَنَّاوْ بَ لࣷولُ اَمࣹي اَجِ سَخْ جِ دَفَ وَخْ نُوحٍ نࣹ ڪࣷ:
«دُگَّلْ ݖِ گَالْ گُ مَگْ گِ،
يَاڪْ سَ نجَبࣷوتْ گࣺݒّ،
ندَخْتࣹ نِتِ جَمࣷنࣷ جِي،
يَوْ رࣹڪّ لاَ ݖِ گِسْ،
نگَ جُبْ نِ مَ ڪࣷ بࣴگّࣹ؞
‏16 2 خࣹيتِ رَبْ يُ سࣹتْa يࣺݒّ يٗبَّالࣹ ݖِ جُرٗومْ ݧَارْ يُ گٗورْ اَڪْ جُرٗومْ ݧَارْ يُ جِگࣺينْ،
اَڪْ ݧَارْ ݖِ خࣹيتِ رَبْ يِ سࣹتُلْ،
تࣹ ݧُيْ گٗورْ اَڪْ جِگࣺينْ؞
‏17 3 تࣹ اِتْ خࣹيتُ ݒِݖّ مُ نࣹڪّ يٗبَّالࣹ ݖِ جُرٗومْ ݧَارْ يُ گٗورْ اَڪْ جُرٗومْ ݧَارْ يُ جِگࣺينْ،
نگِرْ سࣹينْ خࣹيتْ بَݧَ فࣹيْ ݖِ ڪَوْ سُوفْ؞
‏18 4 ندَخْ ڪَتْ لِ فࣹيڪْ جُرٗومْ ݧَارِ فَنْ دِنَا وࣵݖّࣹ تَوْ ݖِ ڪَوْ سُوفْ دِيرُبْ ݧࣹينتّ فُڪِّ گُدِّ اَڪْ ݧࣹينتّ فُڪِّ بࣴݖّࣴگْ،
بَ فَرْ ݖِ اࣵدِّنَ مبࣷولࣹمْ مبِنْدࣹيفْ مُ مَ سࣵڪّ؞»
‏19 5 نࣷونُ نُوحٍ دࣹفْ لَ ڪࣷ اَجِ سَخْ جِ سَنتّ لࣺݒّ؞
‏20 6 بِ مبࣴنّ مِيْ ݧࣴوْ ݖِ اࣵدِّنَ،
بࣷوبَ فࣹڪّ نَ نُوحٍ اَمْ جُرٗومْ بࣹنِّ تࣺيمࣺيرِ اَتْ؞
‏21 7 نُوحٍ دُگّ نَ ݖَ گَالْ گَ،
اࣵندّ اَڪْ اَيْ دࣷومَمْ اَڪْ سࣷخْنَامْ اَڪْ سࣷخْنَيْ دࣷومَمْ يَ،
نگِرْ مُݖّ ݖَ ندࣷخُ مبࣴنّ مَ؞
‏22 8 رَبْ يُ سࣹتْ يِ اَڪْ يِ سࣹتُلْ اَڪْ ݒِݖّ يِ اَڪْ يِيْ رَامْ
‏23 9 اࣵندّ نَݧُ اَڪْ نُوحٍ،
دُگّ ݖَ گَالْ گَ،
دࣹفْ ݧَارْ-ݧَارْ،
گٗورْ اَڪْ جِگࣺينْ،
نِ ڪࣷ يࣵلَّ سَنْتࣹ نُوحٍ؞
‏24 10 بَ جُرٗومْ ݧَارِ فَنْ وࣺيࣹي نَگْ،
ندࣷخُ مبࣴنّ مَ اَگْسِ ݖِ ڪَوْ سُوفْ؞
‏25 11 اَتْ مَ نُوحٍ اَمࣹي جُرٗومْ بࣹنِّ تࣺيمࣺيرِ اَتْ،
ݖَ فُڪِّ فَنْ يَ اَڪْ جُرٗومْ ݧَارْ ݖَ ݧَارࣹيلُ وࣹيرْ وَ،
ڪࣹرَ لَ بࣴتِ ندࣷخْ يَ نࣹ جࣵيّ،
جٗگࣹ خٗوتࣹيْ گࣺيجْ،
بُنْتِ ندࣷخِ اَسَمَانْ نࣹ ڪُلْبࣺتْ،
مُ نࣹ يُرࣹيتْ،
‏26 12 تَوْ بِيْ سࣷتِّڪُ ݖِ سُوفْ دِيرُبْ ݧࣹينتّ فُڪِّ گُدِّ اَڪْ ݧࣹينتّ فُڪِّ بࣴݖّࣴگْ؞
‏27 13 ڪࣹرࣷوگْ بࣺسْ بࣷوبُ لَ نُوحٍ اࣵندّ اَڪِ دࣷومَمْ يُ گٗورْ،
سࣹمْ اَڪْ خَمْ اَڪْ يَفࣹتْ،
اَڪْ سࣷخْنَامْ اَڪْ ݧࣹتِّ سࣷخْنَيْ دࣷومَمْ يࣷويُ،
ݧُ دُگّ ݖَ گَالْ گَ،
‏28 14 اࣵنْدَنْدࣷوڪْ رَبُ اࣵلّ يࣺݒّ،
لُ ݖِ نࣹڪّ اَڪْ وِرْگࣷومْ،
اَڪْ جُرْ گࣺݒّ،
لُ ݖِ نࣹڪّ اَڪْ وِرْگࣷومْ،
اَڪْ لࣺݒّ لُيْ رَامْ ݖِ ڪَوْ سُوفْ،
لُ ݖِ نࣹڪّ اَڪْ وِرْگࣷومْ،
اَڪْ لࣺݒّ لُيْ نَاوْ،
لُ ݖِ نࣹڪّ اَڪْ وِرْگࣷومْ،
مُيْ ݒِݖّ مبَا بࣷرࣷومِ لاَفْ؞
‏29 15 لࣺݒّ لُيْ بࣷرࣷومْ بَڪَّنْ دࣹفْ نَ ݧَارْ-ݧَارْ،
اࣵندّ اَڪْ نُوحٍ،
دُگّ ݖَ گَالْ گَ،
‏30 16 مبࣷولࣹمْ خࣹيتُ مبِنْدࣹيفْ رࣹڪّ،
دِ گٗورْ اَڪْ جِگࣺينْ،
نِ ڪࣷ ڪࣷ يࣵلَّ سَنْتࣹ؞
ݖِ ڪَوْ لࣷولُ اَجِ سَخْ جِ نࣹ رࣵݒّ بُنْتُ گَالْ گِ،
گَنَّاوْ بِ نُوحٍ دُگّࣹي؞
‏31 17 مبࣴنّ مَ نَگْ دِ وَلْ ݖِ اࣵدِّنَ دِيرُبْ ݧࣹينتّ فُڪِّ فَنْ؞
ندࣷخْ يَ دِ يࣷڪُّ،
گَالْ گَيْ جٗگْ،
بَ تࣴمبّ،
تِيمْ سُوفْ لُ سࣷرࣹ؞
‏32 18 مَ نگَيْ گࣴنْ دِ يࣷڪُّ،
تࣹيْ بَاوَانْ ݖِ ڪَوْ سُوفْ،
بَ گَالْ گَ تࣴمبّ،
دِ دࣹمْ ݖَ ڪَوْ ندࣷخْ مَ؞
‏33 19 ندࣷخْ مَ دِ وَلَنگَانْ،
بَ مࣴدّ تُندّ يُ ڪَوࣹ يَ فࣺݒّ فُ اَسَمَانْ تِيمْ؞
‏34 20 مُيْ يࣷڪُّ،
دِ يࣷڪُّ،
بَ گࣴنَ ڪَوࣹ تُندّ يَ لُ مَتْ فُڪِّ خَسَبْ اَڪْ جُرٗومْ؞
‏35 21 مبࣷولࣹمْ مبِنْدࣹيفْ مُ دَانْ دࣷخْ ݖِ ڪَوْ سُوفْ دࣹي نَ ݖَ؞
ݒِݖّ يِ اَڪْ جُرْ گِ اَڪْ رَبِ اࣵلّ يِ اَڪْ لࣺݒّ لُيْ رَامْ ݖِ ڪَوْ سُوفْ دࣹي نَ،
ݧࣷومْ اَڪْ نِتْ ݧࣺݒّ؞
‏36 22 مبࣷولࣹمْ مبِنْدࣹيفْ مُيْ بࣷرࣷومْ بَڪَّنْ ݖِ ڪَوْ سُوفَ بࣷڪّ دࣹي؞
‏37 23 لُيْ دُندّ ݖِ اࣵدِّنَ دَلْدِ سࣵنڪُ،
بَ نِتْ اَڪْ جُرْ اَڪْ لُيْ رَامْ اَڪْ لُيْ نَاوْ– لࣺݒّ لَ اَجِ سَخْ جِ رَافَلْ ݖِ اࣵدِّنَ؞
نُوحٍ رࣹڪَّ دࣹسْ،
مࣷومْ اَڪْ ݧَ اࣵنْدࣷونْ اَڪْ مࣷومْ ݖَ گَالْ گَ؞
‏38 24 نࣷونُ ندࣷخْ مِيْ وَلْ-وَلِ ݖِ ڪَوْ سُوفْ دِيرُبْ تࣺيمࣺيرِ فَنْ اَڪْ جُرٗومْ فُڪّ؞
نُوحٍ گࣺنّ نَ گَالْ گُ مَگْ گَ
‏39 8 1 ݖِ بِيرْ لࣷولُ يࣵلَّ خَلاَتْ نَ نُوحٍ اَڪْ لِ مُ نࣹڪَّلْ ݖِ گَالْ گِ،
دِ رَبِ اࣵلّ يِ اَڪْ جُرْ گِ؞
يࣵلَّ دَلْدِ يࣹبَلْ نگࣹلَوْ،
مُ وَلْ ڪَوْ سُوفْ،
بَ ندࣷخْ يِ وࣵݖّ؞
‏40 2 بࣴتِ ندࣷخْ يَ ݖَ خٗوتࣹ يَ دَفَ تَخَوْ،
بُنْتِ ندࣷخِ اَسَمَانْ تࣴجُ،
تَوْ بِ دَلْ؞
‏41 3 نࣷونُ ندࣷخْ مِ تَا ݖِ سُوفْ دِ وࣵݧِّڪُ؞
بِ تࣺيمࣺيرِ فَنْ يِ اَڪْ جُرٗومْ فُڪّ مَتࣹي نَگْ،
ندࣷخْ مَ گࣴنَ نࣺيوْ؞
‏42 4 بَ جُرٗومْ ݧَارࣹيلُ وࣹيرْ وَ اَمࣹي فُڪِّ فَنْ اَڪْ جُرٗومْ ݧَارْ،
گَالْ گَا نگَ تࣹيرࣹجِ تُنْدِ اَرَرَاتْ؞
‏43 5 ندࣷخْ مَيْ وࣵݧِّڪُ رࣹڪّ بَ ݖَ فُڪّࣹيلُ وࣹيرْ وَ؞
ڪࣹرࣷوگْ بَ فُڪّࣹيلُ وࣹيرْ وَ تࣹرࣷو،
ݖَ لَ نجࣷبَّخْتَلِ تُندّ يَ تࣵمْبَلࣹي فࣹيݧْ؞
‏44 6 تࣹگْ نَݧُ ݖَ ݧࣹينتّ فُڪِّ فَنْ،
نُوحٍ دࣷورَ اُبِّ ݒࣷرࣷخْندࣷلّ،
بَ مُ سࣵڪَّلࣷونْ گَالْ گَ،
‏45 7 دَلْدِ يࣹبَلْ اَبْ بَاخࣷݧْ،
مُيْ نَاوْ،
دِ دࣹمْ اَڪَ دِڪّ،
بَ ندࣷخْ مِ فࣹنْدِ ݖِ ڪَوْ سُوفْ؞
‏46 8 مُ تࣹگْ ݖَ يࣹبَلْ ݒِتَخْ،
بَ خَمْ ندَخْ ندࣷخْ مِ ݖِ سُوفْ فࣹرْ نَ؞
‏47 9 وَايࣹ ݒِتَخْ مَ گِسُلْ فࣹنّ فُ مُ تَگْ،
ندَخْ ندࣷخْ مِ مُورْ نَ سُوفْ سࣺݒّ؞
مُ دࣺلْسِ نَگْ فَ مࣷومْ ݖَ گَالْ گَ؞
نُوحٍ تࣵلَّلْ لࣷخࣷومْ،
جࣵݒّ ڪࣷ دُگَلْ ݖِ بِيرْ فِ وࣹتَمْ؞
‏48 10 نࣺگَنْدِڪُ نَ جُرٗومْ ݧَارِ فَنْ،
دࣷورَ يࣹبَلاَتْ ݒِتَخْ مَ؞
‏49 11 بَ مُ دࣺلْسࣹي ݖَ نگࣷونْ،
اِنْدَالࣹوُلْ لُ مࣷيْ خࣷبُ اࣷلِوْ وُ تࣷويْ وُ ڪࣹݒّࣹ ݖَ سࣵلّ وَ؞
نࣷونُ نُوحٍ خَمْ نࣹ ندࣷخْ مِ تَا وࣷونْ ݖِ سُوفْ،
فࣹرْ نَ؞
‏50 12 مُ نࣺگَتِ جُرٗومْ ݧَارِ فَنْ،
بࣵيِّوَاتْ ݒِتَخْ مَ،
وَايࣹ بِي يࣷونْ دࣺلْسِوَاتُلْ ݖَ مࣷومْ؞
‏51 13 اَتْ مَ نُوحٍ اَمࣹي جُرٗومْ بࣹنِّ تࣺيمࣺيرِ اَتْ اَڪْ بࣹنّ،
ڪࣹرࣷوگْ بࣹنّ فَنُ وࣹيرْ وَ جࣴڪّ،
ندࣷخْ مَ تَا وࣷونْ ݖَ سُوفْ دَلْدِ فࣹنْدِ؞
نُوحٍ اُبِّ ڪُبࣹيرْ گَ،
خࣷولْ،
بَ گِسْ نࣹ ڪَوْ سُوفْ سࣺݒّ وࣷوْ نَ؞
‏52 14 ڪࣹرࣷوگْ ݧَارْ فُڪِّ فَنْ اَڪْ جُرٗومْ ݧَارْ ݖَ ݧَارࣹيلُ وࣹيرْ وَ،
ݖَ لَ سُوفْ سِ دࣷورَ وࣷوْ ڪࣷݝّ؞
‏53 15 بَ لࣷولُ اَمࣹي يࣵلَّ نࣹ نُوحٍ:
‏54 16 «گࣺنَّلْ گَالْ گِ،
يَاڪْ سَ سࣷخْنَ اَڪْ سَيْ دࣷومْ اَڪْ سَيْ سࣷخْنَيْ دࣷومْ؞
‏55 17 گࣺنّࣹوَالࣹيلْ لࣺݒّ لُيْ خࣹيتُ مبِنْدࣹيفْ مُ نگࣹينْ اࣵنْدَلْ:
ݒِݖّ يِ اَڪْ بࣵيِّمَ يِ اَڪْ لُيْ رَامْ،
ݧُ فُلَنْدِوُ ݖِ ڪَوْ سُوفْ،
دَلْدِ گِيرْ تࣹ بَرࣹ؞»
‏56 18 نࣷونُ نُوحٍ گࣺنّ،
مࣷومْ اَڪِ دࣷومَمْ اَڪْ سࣷخْنَامْ اَڪْ سࣷخْنَيْ دࣷومَمْ يَ؞
‏57 19 رَبِ اࣵلّ يِ اَڪْ لُيْ رَامْ اَڪْ لُيْ نَاوْ اَڪْ لُيْ دࣷخْ ݖِ ڪَوْ سُوفْ– لࣺݒّ گࣺنّࣹ ݖَ گَالْ گَ،
لُ ݖِ نࣹڪّ اَڪْ وِرْگࣷومْ؞
‏58 20 نَڪَ لَ نُوحٍ گࣺنّ،
دَلْدِ سࣵڪَّلْ اَجِ سَخْ جِ سَرْخَلُڪَايْ،
مࣵگَّلࣹ ڪࣷ ڪࣷ؞
مُ سࣵڪّ نَگْ ݖِ مبࣷولࣹمْ رَبْ وُ سࣹتْ اَڪْ مبࣷولࣹمْ ݒِݖّ مُ سࣹتْ،
جࣺبَّلْ يࣵلَّ ݖَ سَرْخَلُڪَايْ بَ،
مُيْ سَرَخْ سُيْ لَڪّ بَ جࣹيخْ؞
‏59 21 نࣷونُ خࣹتُ جࣵمّ گِلّࣹي ݖَ سَرَخْ سَ،
يࣺيگْ فَ اَجِ سَخْ جَ؞
بࣷوبَ لَ وَخْ ݖِ خࣷلَمْ نࣹ:
«دࣷوتُمَ رࣴبّ سُوفْ مُڪّ ندَخْ نِتْ،
دࣷونْتࣹ بَ مُيْ ندَوْ لَيْ دَلࣹ خࣴنْتࣹ لُ ݧَاوْ ݖِ خࣷلَمْ؞
دࣷوتُمَ بࣷولࣹ فࣵدّ مُڪّ لࣺݒّ لُيْ دُندّ،
نِي مَ ڪࣷ دࣹفࣹ؞
‏60 22 لِ فࣹيڪْ سُوفْ دِ سُوفْ،
دࣷوتُ نجِوࣷويْ دَڪّ مبَا نگٗوبْ،
تࣹ دُ سࣹدّ اَڪُ تࣵنگࣷورْ،
دُ نَوࣹتْ اِتْ مبَا نࣷورْ،
مبَا گُدّࣹيڪْ بࣴݖّࣴگْ؞»
يࣵلَّ فَسْ نَ ڪٗلّࣴرࣹ اَڪْ نُوحٍ
‏61 9 1 نࣷونُ يࣵلَّ بَرْڪࣹيلْ نُوحٍ اَڪِ دࣷومَمْ نࣹ لࣹينْ:
«گِيرْلࣹينْ تࣹ بَرࣹ،
بَ فࣹيسْ اࣵدِّنَ؞
‏62 2 مبࣷولࣹمْ رَبِ اࣵلّ يِ اَڪْ ݒِݖِّ اَسَمَانْ اَڪْ بِنْدࣹيفْ يِيْ رَامْ ݖِ سُوفْ،
اَڪْ جࣴنِ گࣺيجْ يࣺݒّ،
دِنَݧُ لࣹينْ رَگَلْ لࣷولْ؞
تࣹگْ نَا لࣹينْ ݖِ سࣹينِ لࣷخࣷ؞
‏63 3 مَنْ نگࣹينَ دُنْدࣹ بࣷرࣷومْ بَڪَّنْ يࣷويُ يࣺݒّ؞
جَگْلࣹيلْ نَا لࣹينْ بِنْدࣹيفْ يࣷويُ يࣺݒّ،
نِ مَ لࣹينْ جَگْلࣹيلࣹ وࣷونْ گࣵنْݖَخْ؞
‏64 4 «وَايࣹ بُلࣹينْ لࣹڪّ يࣵݒّ وُ اࣵندّ اَڪْ دࣹرࣹتْ،
ندَخْتࣹ دُنْدَ نگِ ݖِ دࣹرࣹتْ؞
‏65 5 سࣹينْ دُنْدُ نِتْ اِتَمْ،
مُ نگِ ݖِ سࣹينْ دࣹرࣹتْ،
ڪࣷنْ ڪَتْ،
ڪࣺݒّ ڪُ تُورْ سࣹينْ دࣹرࣹتْ دِنَا ڪࣷ ڪࣷ تࣷݒّ؞
تࣹ اِتْ مَلَ مُ ݖِ رࣹيْ نِتْ دِنَا ڪࣷ ڪࣷ تࣷݒّ،
نِ مَيْ تࣷݒّࣹ نِتْ؞
دِنَا تࣷݒّ ڪُ نࣹڪّ دࣹرࣹتُ مࣷرࣷومَمْ؞
‏66 6 ڪُ تُورْ دࣹرࣹتُ نِتْ،
نَ نِتْ تُورْ دࣹرࣹتَمْ،
نگِرْ مࣹلࣷڪَانُ بࣷݒَّمْ
لَ يࣵلَّ سࣵڪّࣹ نِتْ؞
‏67 7 يࣺينْ نَگْ،
گِيرْلࣹينْ تࣹ بَرࣹ،
فُلَنْدِوُ ݖِ ڪَوْ سُوفْ تࣹ فࣹيسْ ڪࣷ؞»
‏68 8 نࣷونُ يࣵلَّ وَخْ نُوحٍ اَڪْ دࣷومَمْ،
يِ مُ اࣵنْدَلْ،
نࣹ لࣹينْ:
‏69 9 «بࣺسْ نِڪِ تࣹيْ فَسَلْ نَا سَمَ بࣷݒّ ڪٗلّࣴرࣹ سَمَ دِگَّنْتࣹ اَڪْ يࣺينْ اَڪْ سࣹينْ اَسْڪَنْ،
‏70 10 فَسْ ڪࣷ اِتْ سَمَ دِگَّنْتࣹ اَڪْ مبࣷولࣹمْ مبِنْدࣹيفْ مُ نگࣹينْ اࣵنْدَلْ،
مُيْ ݒِݖّ اَڪْ اُگْ جُرْ مبَا رَبِ اࣵلّ– لࣺݒّ دَالْ لُ گࣺنَّنْدࣷو اَڪْ يࣺينْ ݖِ گَالْ گِ،
دِ لࣺݒّ لُيْ دُندّ ݖِ ڪَوْ سُوفْ؞
‏71 11 نِ مَيْ فَسࣹيڪْ يࣺينْ ڪٗلّࣴرࣹ نِي لَ:
دࣹيسَتُلْ بࣷولࣹ بِنْدࣹيفْ يࣺݒّ،
فَاتْ ݖِ مبࣴنّ مُڪّ؞
دُ مࣹنّ مبࣴنّ مُ دࣹيتِ فَاگَاگَلْ اࣵدِّنَ؞»
‏72 12 يࣵلَّ نࣹيتِ:
«ڪٗلّࣴرࣹ گِ مَيْ فَسْ اَڪْ يࣺينْ اَڪْ مبࣷولࣹمْ مبِنْدࣹيفْ مُ اࣵندّ اَڪْ يࣺينْ،
دَيْ سَخْ بَ فࣵوّ تࣹ لِيْ مࣵنْدَرْگَامْ مࣷويْ لِي:
‏73 13 دࣹفْ نَا سَمَ خࣷنْ ݖِ نِيرْ يِ،
مُيْ مࣵنْدَرْگَالْ ڪٗلّࣴرࣹ،
گِ دࣷخْ سَمَ دِگَّنْتࣹ اَڪْ اࣵدِّنَ؞
‏74 14 «سَا سُ مَ دَجَلࣹي نِيرْ يِ ݖِ جَوّ جِ،
بَ خࣷنْ گِ فࣹيݧْ ݖِ نِيرْ يِ،
‏75 15 دَمَيْ دَلْدِ خَلاَتَاتْ ڪٗلّࣴرࣹ،
گِ مَ فَسْ سَمَ دِگَّنْتࣹيڪْ يࣺينْ اَڪْ مبࣷولࣹمْ خࣹيتِ بِنْدࣹيفْ،
سُ ڪࣷ دࣹفࣹي ندࣷخْ دࣷوتُلْ سَبَبْ مبࣴنّ مُيْ فَاگَاگَلْ لࣺݒّ لُيْ دُندّ؞
‏76 16 خࣷنْ گِ دَيْ نࣹڪّ ݖِ بِيرْ نِيرْ يِ،
مَ دِ ڪࣷ گِسْ،
دِ خَلاَتْ ڪٗلّࣴرࣹ،
گِ مَ فَسْ فࣵوّ،
سَمَ دِگَّنْتࣹ،
مَنْ يࣵلَّ،
اَڪْ مبࣷولࣹمْ خࣹيتُ مبِنْدࣹيفْ مُ نࣹڪّ ݖِ ڪَوْ سُوفْ؞»
‏77 17 ݖِ ڪَوْ لࣷولُ يࣵلَّ نࣹ نُوحٍ:
«لࣷولُ مࣷويْ مࣵنْدَرْگَالْ سَمَ ڪٗلّࣴرࣹ،
گِ مَ فَسْ اَڪْ مبࣷولࣹمْ مبِنْدࣹيفْ مُ نࣹڪّ ݖِ ڪَوْ سُوفْ؞»
نجࣵلْبࣺينْ گَ، سَارْ 1.11-9
يࣵلَّ سَفَانْ نَ لࣵڪُّ اࣵدِّنَ، مُ جَخَسࣷو
‏78 11 1 ݖَ ندࣷورْتࣹ لَ نَگْ وَا اࣵدِّنَ سࣺݒَّ بࣷڪّࣷونْ وࣹنّ لࣵڪّ،
سࣹينِ وَخْ دِ بࣹنّ؞
‏79 2 بِ نِتْ ݧِ تࣷخࣷو،
جࣴمْ ݒࣹنڪُ،
دَݧࣷو گِسْ اَگْ جࣷورْ ݖِ رࣺيوُ سِنࣹيَارْ،
ݧُ سَنݖّ فَ؞
‏80 3 ݧُ نࣹ:
«اَيْݖَ نُ دࣹفَرْ اَيْ مٗولْ يُ ݧُ لَڪّ ݖِبْ تَالْ،
بَ مُ ݧࣷرْ؞»
نࣷونُ ݧُ دࣹفْ مٗولْ يَ دࣷجِ تَبَخْ،
تَقَلࣹي ڪࣷ ڪࣷلْتَارْ،
‏81 4 دَلْدِ نࣹ:
«اَيْݖَ نُ تَبَخْ اَبْ دࣴڪّ بُ تَاخْ مَ بࣴتّ اَسَمَانْ؛
ڪࣷنْ دِنَنُ اَمْ وࣷيْ،
تࣹ دُنُ تَسَارࣷو ݖِ ڪَوْ سُوفْ؞»
‏82 5 وَايࣹ اَجِ سَخْ جِ وࣵݖّ،
خࣷولْسِ دࣴڪّ بِ اَڪْ تَاخْ،
مِ دࣷومْ آدَمَ يِ تَبَخْ؞
‏83 6 اَجِ سَخْ جِ دَ نࣹ:
«سُ ݧُ تࣵمْبَلࣹي نِي،
دِ مࣹنّ مبࣷولࣷو تࣹ بࣷڪّ وࣹنّ لࣵڪّ،
ڪࣷنْ دَرَ دࣷوتُ لࣹينْ تࣴ؞
‏84 7 اَيْݖَ نُ وࣵݖّ،
سَفَانْ سࣹينْ لࣵڪّ،
بَ دࣷوتُݧُ دࣺگّࣷو؞»
‏85 8 بَ لࣷولُ اَمࣹي اَجِ سَخْ جِ جࣴلࣹ لࣹينْ فࣷوفَ،
تَسَارࣹ لࣹينْ ݖِ ڪَوْ سُوفْ سࣺݒّ،
