ڪࣷرْنࣹيْ وࣷو نَ ݒِيࣹيرْ
10
‏1 ڪُ ݧُيْ وَخْ ڪࣷرْنࣹيْ، نجِيتُ تَڪّ-دࣹرْ لَ وࣷونْ ݖَ سࣹسَرࣹ؞ مَ نگَ بࣷڪّࣷونْ ݖَ مبࣷولࣷو مَ ݧُ دِݒّࣹي گࣵنگࣷورُ اِتَلِ؞ ‏2 ڪُ جُلِّتࣹ لَ وࣷونْ تࣹ رَگَلْ يࣵلَّ، مࣷوڪْ وَا ڪࣴرَمْ گࣺݒّ؛ بَرࣹ وࣷونْ نَ سَرَخْ يُ مُيْ يࣺوࣺنࣹ اَسْڪَنْ وِ، تࣹ ݧَانْ ݖِ يࣵلَّ لَ سَخࣷو وࣷونْ؞
‏3 اَمْ ݒࣹيݧُ نَگْ دِڪَّلْ ڪࣷ دِگّ نجࣷلّࣷورْ، مُ گِسْ ݖِ لُ لࣹيرْ مَلاَڪَمْ يࣵلَّ، مُ دُگّسِ، نࣹ ڪࣷ:‏ «ڪࣷرْنࣹيْ!‏» ‏4 ڪࣷرْنࣹيْ نࣹ ڪࣷ جࣵڪّ، تِيتْ، نࣹ ڪࣷ:‏ «سَنگّ بِ، لُ مُ دࣷونْ؟» مُ نࣹ ڪࣷ:‏ «سَيْ ݧَانْ اَڪْ سَيْ سَرَخْ يࣺيگْ نَ فَ يࣵلَّ، بَ مُ بࣵيِّ لَ خࣹلْ؞ ‏5 لࣺيگِ نَگْ يࣹبْلࣹيلْ ݖَ يࣷݒࣹ، نگَ وࣷولُجِ فَ ڪُ ݧُيْ وَخْ سِمࣷݝْ، ݧُ دِ ڪࣷ دࣵڪّࣹنْتَلࣹ ݒِيࣹيرْ؞ ‏6 مَ نگَ دَلْ اَڪْ مࣹنࣹينْ سِمࣷݝْ مَ، وُلِّڪَتْ بَ ڪࣴرَمْ فࣺيتࣹ وࣹتُ گࣺيجْ؞»
‏7 نَڪَ لَ مَلاَڪَ مَ وَخْ اَڪْ ڪࣷرْنࣹيْ بَ دࣹمْ، مُ وࣷو ݧَارْ ݖِيْ سُرْگَامْ، اَڪْ بࣹنّ تَڪّ-دࣹرْ بُ جُلِّتࣹ تࣹ بࣷڪّ ݖِيْ سُقَمْ؞ ‏8 مُ نࣹتَّلِ لࣹينْ لࣺݒّ نَگْ، دَلْدِ لࣹينْ يࣹبَلْ يࣷݒࣹ؞
‏9 ݖَ اࣴلّࣴگْ سَ، ݧُ دࣹمْ بَيْ جُبْ دࣴڪّ بَ، يࣹمࣷوڪْ ݒِيࣹيرْ يࣺيگْ ݖَ ڪَوْ تَاخْ مَ ݖِ دِگّ بࣴݖّࣴگْ، نگِرْ ݧَانْ؞ ‏10 مُ دࣹمْ بَ خِيفْ، بࣴگَّ لࣹڪّ، ݧُ دِ ڪࣷ تࣷگَّلْ نَگْ لُ مُ لࣹڪّ؞ ݖِ بِيرْ لࣷولُ مُ دَانُ لࣹيرْ؞ ‏11 مُ دَلْدِ گِسْ اَسَمَانْ اُبِّڪُ، لُ مࣹلْ نِ سࣺرْ بُ مَگْ بُ ݧُ تࣺيࣹي ݧࣹينْتِ لَفَمْ، وࣵݖّࣹ اَسَمَانْ، جࣴمْ سُوفْ؞ ‏12 سࣺرْ بَا نگَ دࣹفْ مبࣷولࣹمْ بࣷرࣷومْ ݧࣹينْتِ تࣵنڪّ، اَڪْ دُنْدࣷوتْ يِيْ رَامْ، اَڪْ مبࣷولࣹمْ نجَنَاوْ؞ ‏13 مُ اَمْ بَاتْ بُ جِبْ، نࣹ ڪࣷ:‏ «ݒِيࣹيرْ، جٗگَلْ، رࣹيْ تࣹ لࣹڪّ؞» ‏14 ݒِيࣹيرْ نࣹ ڪࣷ:‏ «مُڪّ، سَنگّ بِ، ندَخْتࣹ مَسُمَا لࣹڪّ لࣹنّ لُ دَگَنُلْ مبَا لُ سࣹتُلْ؞» ‏15 تࣹيوُلْ بَاتْ بِ وَخْ اَڪْ مࣷومْ ݧَارࣹيلْ بِ يࣷونْ، نࣹ ڪࣷ:‏ «لُ يࣵلَّ سࣹتَلْ، بُ ڪࣷ دَگَنَدِلْ؞» ‏16 مࣹنّ ݒࣹيݧُ مࣷومُ دِڪَّلْ نَ ڪࣷ ݧࣹتِّ يࣷونْ؞ لَ ݖَ تࣹگُ ݧُ يࣺيگࣹ ڪࣴفْ ڪَ اَسَمَانْ؞ ‏17 ݒِيࣹيرَ نگَ جَاخْلࣹ لࣷولْ، دِ سࣹيتْ ݖِمْ خࣹلَمْ لُ ݒࣹيݧُ مَ ڪࣷ دِڪَّلْ دِ تࣹڪِّ، ندࣹڪࣹ نِتْ ݧَ ڪࣷرْنࣹيْ يࣹبَلࣷونْ لاَجْ نَݧُ ڪࣴرْ سِمࣷݝْ، بَ دِڪّ تَخَوْ ݖَ بُنتّ بَ؞ ‏18 ݧُ وࣷوتࣹ، لاَجْتࣹ، نࣹ:‏ «سِمࣷݝْ، مِ ݧُيْ وَخْ ݒِيࣹيرْ، فِي لَ دࣴڪّ؟»
‏19 نَڪَ لَ ݒِيࣹيرْ دِ خَلاَتْ بَ تࣹيْ ݖَ ݒࣹيݧُ مَ، نࣷو گِ نࣹ ڪࣷ:‏ «ݧࣹتِّ نِتَ نگِ نِي، دِ لَ سࣹيتْ؞ ‏20 وࣵݖَّلْ بࣷوگْ، اࣵندّ اَڪْ ݧࣷومْ تࣹ بُلْ نࣵتَّبْلࣹ، ندَخْ مَنْ مَا لࣹينْ يࣹبَلْ؞» ‏21 ݒِيࣹيرْ وࣵݖّ نَگْ نࣹ نِتْ ݧِ:‏ «مَا نگِي؛ ڪِ نگࣹينْ دِ سࣹيتْ، مَنْ لَ؞ لُ دࣷونْ سࣹينِ تࣵنڪّ؟» ‏22 ݧُ نࣹ ڪࣷ:‏ «نجِيتُ تَڪّ-دࣹرْ بَ ڪࣷرْنࣹيْ، ڪُ جُبْ، رَگَلْ يࣵلَّ، تࣹ رَفࣹتْ سࣹيدࣹ بُ ڪࣷ خࣹيتُ يَوُتْ وࣺݒّ سࣹيدࣹيلْ، مࣷو جࣷتࣹ ݖِ مَلاَڪَ مُ سࣹلّ، ندِگَلَلْ وࣷولُ لَ ڪࣴرَمْ، نگِرْ دࣺگْلُ سَيْ ڪࣵدُّ؞» ‏23 ݖِ ڪَوْ لࣷولُ ݒِيࣹيرْ دُگَلْ لࣹينْ، دَلَلْ لࣹينْ؞
ݒِيࣹيرْ وُيْجِ نَ ڪࣷرْنࣹيْ
بَ بࣴتْ سࣹتࣹي، ݒِيࣹيرْ اࣵندّ اَڪْ ݧࣷومْ، ݧࣹنّ ݖَ بࣷڪِّ گࣴمْڪَتْ ݧَ دࣴڪّࣹ يࣷݒࣹ گُنگࣹ ڪࣷ؞ ‏24 ݖَ اࣴلّࣴگْ سَ ݒِيࣹيرْ اَگْسِ سࣹسَرࣹ؞ فࣹڪّ نَ ڪࣷرْنࣹيْ دِ ڪࣷ سࣺينْتُ، دَلْدِ وࣷو ݖَ ڪࣴرَمْ اَيْ بࣷڪَّمْ، اَڪْ خَرِتَمْ يِ ڪࣷ گࣴنَ جࣹگࣹ؞ ‏25 بَ ݒِيࣹيرْ دُگّسࣹي، ڪࣷرْنࣹيْ گَتَنْدُ ڪࣷ، نࣹ گُرُبْ ݖَ تࣵنڪِ ݒِيࣹيرْ، سُجّٗوتْ؞ ‏26 ݒِيࣹيرْ يࣺڪَّتِ ڪࣷ، نࣹ:‏ «دࣺيتْ جٗگَلْ، مَنْ اِتْ، نِتْ دࣷݝّ لاَ؞» ‏27 ݒِيࣹيرْ دِ وَخْتَانْ اَڪْ مࣷومْ، بَ دُگّسِ، يࣹمْ ݖَ نِتْ ݧُ بَرࣹ ݧَ فَ دَجࣹ؞ ‏28 مُ نࣹ لࣹينْ:‏ «يࣹينْ دࣹ خَمْ نگࣹينْ نࣹ اَبْ يَوُتْ، مَيࣹيسُ ڪࣷ مُ جَخَسࣷو اَڪْ خࣹيتُ جَامْبُرْ، مبَا مُ سࣹيتِ ڪࣷ؞ وَايࣹ مَنْ نَگْ يࣵلاَّ مَ وࣷنْ نࣹ دُ ڪࣹنّ نِتْ ڪُ مَ نَانْ دَگَنُلَ جَخَسࣷولْ، مبَا سࣹتُلْ؞ ‏29 لࣷولࣷو تَخْ، بِ نگࣹينْ مَ وࣷولࣷو، مَ ݧࣴوْ تࣹ تࣹنْدࣹيفَلُمَ بࣹنّ يࣷونْ؞ مَ دِ لࣹينْ لاَجْ نَگْ لِ وَرَلْ نگࣹينْ وࣷولُ مَ؞»
‏30 ݖِ ڪَوْ لࣷولُ ڪࣷرْنࣹيْ نࣹ:‏ «بࣴڪَّاتِ-دࣺمبّ ݖِ وَخْتُوْ دِگّ نجࣷلّࣷورْ وِي نُ تࣷلُّ تࣹيْ، ݖِ لاَ دࣷونْ ݧَانࣹ سَمَ بِيرْ نࣺيگْ؞ مُ اَمْ ڪُ جࣹڪِّ نࣹ جَاسْ ݖِ سَمَ ڪَنَمْ، سࣷلْ يࣺرࣹ يُ نࣹ رࣵݧّ؞ ‏31 مُ نࣹ مَ:‏ ‹ڪࣷرْنࣹيْ، نَنگُ نَݧُ سَيْ ݧَانْ، تࣹ نࣹمِّڪُ نَݧُ سَيْ سَرَخْ فَ ڪَنَمْ يࣵلَّ؞ ‏32 ڪࣷنْ نَگْ يࣹبْلࣹيلْ يࣷݒࣹ، نگَ وࣷولُ سِمࣷݝْ، مِ ݧُ دࣵڪّࣹنْتَلࣹ ݒِيࣹيرْ؛ مَ نگَ دَلْ ݖَ ڪࣴرْ مࣹنࣹينْ سِمࣷݝْ، وُلِّڪَتْ بَ ݖَ وࣹتُ گࣺيجْ گَ؞› ‏33 ݖَ سَا سَ نَگْ لاَ يࣹبْلࣹ ݖِ يَوْ؞ يَوْ اِتْ دࣹفْ نگَ لُ بَاخْ، بِ نگَ دِڪّࣹي؞ لࣺيگِ نَگْ نُنْ ݧࣺݒَّ نگِ فِ ݖِ ڪَنَمْ يࣵلَّ، نگِرْ دࣺگْلُ مبࣷولࣹمْ لُ لَ بࣷرࣷومْ بِ سَنتّ؞»
‏34 بَ لࣷولُ اَمࣹي ݒِيࣹيرْ اࣵدُّ، نࣹ:‏ «دࣴگّ-دࣴگّ گِسْ نَا نࣹ يࣵلَّ دَالْ دُ گࣴنَاتْلࣹ، ‏35 وَايࣹ خࣹيتࣷو خࣹيتْ، ڪُ ڪࣷ ݖِ رَگَلْ تࣹيْ دࣹفْ نجࣹڪّ، مࣷويْ ڪِ ڪࣷ نࣹيخْ؞ ‏36 مࣷو يٗنّࣹي ڪࣵدّࣷومْ بࣵنِّ إِسْرَاءِيلْ، اࣵگّࣹ لࣹينْ خِبَارُ جࣵمّ، بِيْ مَيࣹ جࣵمّ ݖِ عِيسَىٰ اَلْمَسِ، مِيْ سَنگُ ݧࣺݒّ؞ ‏37 يࣹينْ ݖِ سࣹينْ بࣷݒّ خَمْ نگࣹينْ لِ خࣹوࣷونْ ݖِ مبࣷولࣹمْ رࣺيوُمْ يَوُتْ يِ، تࣹ دَلࣹ ݖِ دِيوَانُ گَلِلࣹ، گَنَّاوْ بَ يَحْيَىٰ وࣷوتࣹي، دِ سٗوبࣹ ݖِمْ ندࣷخْ، ‏38 اَڪْ نِ يࣵلَّ اࣵنْدࣹ اَڪْ عِيسَىٰ مُ نَسَرࣹتْ، بَ جَگْلࣹيلْ ڪࣷ ݒَلْ گُ مُ دࣷگَلࣹ نࣷو گُ سࣹلّ گࣹيڪْ خَمْ-خَمْ بِ مُ ڪࣷ سࣷتِّ، مُيْ وࣴرْ، دِ دࣹفْ لُ بَاخْ، دِ فَجْ مبࣷولࣹمْ ݧِ سࣹيْتَانࣹ نࣷتࣷونْ؞
‏39 «نُنْ ݖِ سُنُ بࣷݒّ نࣷو سࣹيدࣹ لِ مُ دࣹفْ لࣺݒّ ݖِ رࣺيوُمْ يَوُتْ يِ اَڪْ يࣹرُسَلࣹمْ؞ مࣷومْ مِ ݧُ وࣺڪّ ݖِ بَنتّ، بَ رࣹيْ ڪࣷ، ‏40 مࣷومْ لَ يࣵلَّ دࣹڪَّلْ ݖِ ݧࣹتّࣹيلُ فَنَمْ، بَ مَيْ ڪࣷ مُ فࣹيݧْ؞ ‏41 دُ اَسْڪَنْ وࣺݒّ لَ ڪࣷ وࣷنْ، خَنَا نُنْ، سࣹيدࣹ يِ يࣵلَّ تࣵنّࣷونْ لُ جِيتُ، نُ بࣷڪّ اَڪْ مࣷومْ لࣹڪّ اَڪْ نَانْ، گَنَّاوْ ندࣹڪِّتࣹيمْ؞ ‏42 مࣷو نُ سَنتّ نَگْ نُ خَمَلْ اَسْڪَنْ وِ تࣹ سࣹيدࣹ نࣹ مࣷومْ لَ يࣵلَّ تَبّ اࣵتّࣹڪَتُبْ اَجِ دُندّ ݧِ، اَڪْ اَجِ دࣹي ݧِ؞ ‏43 مࣷومْ لَ يࣷنࣹنتّ يࣺݒّ سࣹيدࣹيلْ، نࣹ ڪࣺݒّ ڪُ ڪࣷ گࣴمْ، جࣷتْ نگَ سَ نجࣺگَّلُگْ بࣵڪَّارْ ݖِوْ تُرَمْ؞»
‏44 بَ ݒِيࣹيرْ دِ وَخْ ڪࣵدُّ گࣷوگُ، نࣷو گُ سࣹلّ گِ دَلْدِ وࣵݖّ ݖِ ڪَوْ مبࣷولࣹمْ ݧَيْ دࣺگْلُ ڪࣵدُّ گَ؞ ‏45 بَ لࣷولُ اَمࣹي مبࣷولࣹمْ گࣴمْڪَتِ يَوُتْ يَ اࣵندّ اَڪْ ݒِيࣹيرْ، وَارُ ݖَ نَ مَيُگْ نࣷو گُ سࣹلّ گِ تُورُوَالࣹي ݖَ خࣹيتِ جَامْبُرْ ݧَ، ‏46 ندَخْ دَݧُ لࣹينَ دࣺگّ ݧࣷومْ اِتَمْ، ݧُيْ لࣵڪّ يࣹنࣹينْ لࣵڪّ، اَڪَ مࣵگَّلْ يࣵلَّ؞
‏47 ݒِيࣹيرْ دࣹلُّ نࣹ:‏ «ݧِي جࣷتْ نࣷو گُ سࣹلّ گِ نِ نُنْ، ندَخْ مَنࣹيسْ نَ لࣹينَ بَݧَ سٗوبْ ݖِ ندࣷخْ؟» ‏48 مُ سَنْتَانࣹ نَگْ ݧُ سٗوبْ لࣹينْ ݖِ ندࣷخْ ݖِ تُرُ عِيسَىٰ اَلْمَسِ؞ گَنَّاوْ لࣷولُ ݧُ ݧَانْ ݒِيࣹيرْ، مُ تࣷوگْ ݖِ ݧࣷومْ اَيْ فَنْ؞
