نجࣵلْبࣺينْ گَ
دِ خَاجْ بِ جࣴڪّ ݖِ تَوْرࣹيتُ مُوسَىٰ
يࣵلَّ سࣵڪّ نَ اࣵدِّنَ، فَلْ ݖِ نِتْ
1
‏1 ݖَ نجࣵلْبࣺينْ گَ يࣵلَّ سࣵڪّ نَ اَسَمَانْ اَڪْ سُوفْ؞ ‏2 سُوفْ نَگْ دَا مَاسࣹ وࣷونْ تࣹ وࣺيتْ، لࣴنْدࣴمْ مُورْ ندࣷخْ مُ خٗوتْ مِ، لَلࣹ ݖَ ڪَوْ، نࣷووُگْ يࣵلَّ دِ وࣴرْ، تِيمْ ندࣷخْ مِ؞
‏3 بَ لࣷولُ اَمࣹي يࣵلَّ نࣹ:‏ «نَ لࣹيرْ نࣹڪّ،» لࣹيرْ دَلْدِ اَمْ؞ ‏4 يࣵلَّ گِسْ نࣹ لࣹيرْ گِ لُ بَاخْ لَ، مُ خࣵجَّلࣹ لࣹيرْ اَڪْ لࣴنْدࣴمْ؞ ‏5 يࣵلَّ تُدّࣹ لࣹيرْ گِ بࣴݖّࣴگْ، لࣴنْدࣴمْ گِ گُدِّ؞ نگࣷونْ جࣷتْ، سُبَ دُگّ، مُيْ بࣺسْ بَ جࣴڪّ؞
‏6 يࣵلَّ نࣹ:‏ «نَ اَوْ دࣺندّ دࣷخْ دِگَّنْتࣹ ندࣷخْ يِ، نگِرْ خࣵجَّلࣹ لࣹينْ، يِي فِي، يِي فَلࣹ؞» ‏7 يࣵلَّ نَگْ سࣵڪّ دࣺندّ وِ، خࣵجَّلࣹي ڪࣷ ندࣷخْ يِ فࣺيتࣹ دࣺندّ وِ سُوفْ اَڪْ يِ ڪࣷ فࣺيتࣹ ڪَوْ؞ مُ دَلْدِ نࣹڪّ نࣷونَ؞ ‏8 يࣵلَّ تُدّࣹ دࣺندّ وِ اَسَمَانْ؞ نگࣷونْ جࣷتْ، سُبَ دُگّ، مُيْ بࣺسُبْ ݧَارࣹيلْ بَ؞
‏9 يࣵلَّ نࣹ:‏ «نَ ندࣷخْ، مِ اَسَمَانْ تِيمْ، گࣴنْدࣷو بࣹنّ بࣺرَبْ، بَ جࣷورْ گُ وࣷوْ فࣹيݧْ؞» مُ دَلْدِ نࣹڪّ نࣷونَ؞ ‏10 يࣵلَّ تُدّࣹ جࣷورْ گُ وࣷوْ گِ سُوفْ، ندࣷخْ مِ گࣴنْدࣷو، مُ تُدّࣹ ڪࣷ گࣺيجْ؞ يࣵلَّ گِسْ نࣹ بَاخْ نَ؞
‏11 يࣵلَّ نࣹ:‏ «نَ سُوفْ سَخْ ݧَخْ، مُيْ گࣵنݖَخْ گُ اࣵندّ اَڪْ جِوࣷومْ، اَڪْ گَرَبْ يُيْ مࣹݧّ تࣹيْ جُرْ سࣹينِ جِوُ، لُ ݖِ نࣹڪّ اَڪْ وِرْگࣷومْ ݖِ ڪَوْ سُوفْ؞» مُ دَلْدِ نࣹڪّ نࣷونَ؞ ‏12 سُوفْ نَگْ سَخْ ݧَخْ، مُيْ گࣵنݖَخْ گُ اࣵندّ اَڪْ وِرْگࣷيْ جِوࣷومْ، اَڪْ گَرَبْ يُيْ مࣹݧّ تࣹيْ جُرْ سࣹينِ وِرْگࣷيْ جِوُ؞ يࣵلَّ گِسْ نࣹ بَاخْ نَ؞ ‏13 نگࣷونْ جࣷتْ، سُبَ دُگّ، مُيْ بࣺسُبْ ݧࣹتّࣹيلْ بَ؞
‏14 يࣵلَّ نࣹ:‏ «نَ اَمْ اَيْ لࣹيرْ ݖِ اَسَمَانْ، نگِرْ خࣵجَّلࣹ گُدِّ اَڪْ بࣴݖّࣴگْ، تࣹ دࣷونْ اَيْ تࣹگْتَلْ، يُيْ رࣵݧّلࣹ جَمࣷنࣷ يِ، تࣹيْ تَڪّ بࣺسْ يِ اَڪْ اَتْ يِ؞ ‏15 نَݧُ نࣹڪّ اَيْ لࣹيرْ ݖِ اَسَمَانْ سِ، دِ لࣹيرَلْ سُوفْ؞» مُ دَلْدِ نࣹڪّ نࣷونَ؞ ‏16 يࣵلَّ سࣵڪّ ݧَارِ لࣹيرْ يُ مَگْ، گِ گࣴنَ مَگْ يِلِفْ بࣴݖّࣴگْ، گِ گࣴنَ ندَوْ يِلِفْ گُدِّ؛ مُ سࣵڪّ اِتْ بِدِّيوْ يِ؞ ‏17 يࣵلَّ نَگْ تࣹگْ لࣹينْ ݖِ اَسَمَانْ، دِ ݖَ لࣹيرَلْ سُوفْ، ‏18 ݧُيْ يࣷرࣹ گُدِّ اَڪْ بࣴݖّࣴگْ، تࣹيْ خࣵجَّلࣹ لࣴنْدࣴمْ اَڪْ لࣹيرْ؞ يࣵلَّ گِسْ نࣹ بَاخْ نَ؞ ‏19 نگࣷونْ جࣷتْ، سُبَ دُگّ، مُيْ بࣺسُبْ ݧࣹينْتࣹيلْ بَ؞
‏20 بَ لࣷولُ اَمࣹي يࣵلَّ نࣹ:‏ «نَ ندࣷخْ فࣹيسْ اَڪْ اَيْ ندِيرَانِ خࣹيتِ بِنْدࣹيفْ، تࣹ اَيْ ݒِݖّ تِيمْ سُوفْ، دِ نَاوْ ݖِ اَسَمَانْ؞» ‏21 ݖِ بِيرْ لࣷولُ يࣵلَّ بِندّ رَبِ گࣺيجْ يُ مَگْ يِ اَڪْ مبࣷولࣹمْ مبِنْدࣹيفْ مُيْ فࣺييْ تࣹيْ رَمّ-رَمِّ ݖِ ندࣷخْ مِ، مُ ݖِ نࣹڪّ اَڪْ وِرْگࣷومْ، تࣹگْ ݖَ لࣺݒّ لُيْ بࣷرࣷومِ لاَفْ، لُ ݖِ نࣹڪّ اَڪْ وِرْگࣷومْ؞ يࣵلَّ گِسْ نࣹ بَاخْ نَ؞ ‏22 بَ مُ ڪࣷ دࣹفࣹي يࣵلَّ بَرْڪࣹيلْ لࣹينْ، دَلْدِ نࣹ:‏ «گِيرْلࣹينْ تࣹ بَرࣹ، بَ فࣹيسْ ندࣷخِ گࣺيجْ؞» مُ نࣹيتِ:‏ «نَ ݒِݖّ يِ بَرࣹ ݖِ ڪَوْ سُوفْ؞» ‏23 نگࣷونْ جࣷتْ، سُبَ دُگّ، مُيْ بࣺسُبْ جُرٗومࣹيلْ بَ؞
‏24 يࣵلَّ نࣹ:‏ «نَ سُوفْ گࣺنّࣹ اَيْ بِنْدࣹيفْ، مُ ݖِ نࣹڪّ اَڪْ وِرْگࣷومْ:‏ اَگْ جُرْ، لُيْ رَامْ اَڪِ رَبِ اࣵلّ، لُ ݖِ نࣹڪّ اَڪْ وِرْگࣷومْ؞» مُ دَلْدِ نࣹڪّ نࣷونَ؞ ‏25 يࣵلَّ دَفَ بِندّ رَبِ اࣵلّ يِ، لُ ݖِ نࣹڪّ اَڪْ وِرْگࣷومْ، جُرْ گُ ݖِ نࣹڪّ اَڪْ وِرْگࣷومْ، اَڪْ لࣺݒّ لُيْ رَامْ ݖِ ڪَوْ سُوفْ، لُ ݖِ نࣹڪّ اَڪْ وِرْگࣷومْ؞ يࣵلَّ نَگْ گِسْ نࣹ بَاخْ نَ؞
‏26 يࣵلَّ نࣹ:‏ «نَنُ بِندّ نِتْ ݖِ سُنُ مࣹلࣷڪَانْ، مُ نِرࣷوڪْ نُنْ، تࣹ نَ يِلِفْ جࣴنِ گࣺيجْ اَڪْ ݒِݖِّ اَسَمَانْ اَڪْ جُرْ گࣺݒّ اَڪْ سُوفْ سࣺݒّ اَڪْ لࣺݒّ لُ ݖِيْ رَامْ؞»
‏27 يࣵلَّ دَفَ بِندّ نِتْ ݖِ تَڪّندࣹيرَمْ؛
ݖِ تَڪّندࣹيرُ بࣷݒَّمْ لَ يࣵلَّ بِندّ نِتْ،
گٗورْ اَڪْ جِگࣺينْ لَ بِندّ؞
‏28 يࣵلَّ بَرْڪࣹيلْ لࣹينْ نࣹ لࣹينْ:‏ «گِيرْلࣹينْ تࣹ بَرࣹ، بَ فࣹيسْ اࣵدِّنَ، دِ ڪࣷ سَيْتُ؛ تࣹ اِتْ يِلِفْلࣹينْ جࣴنِ گࣺيجْ اَڪْ ݒِݖِّ اَسَمَانْ اَڪْ مبࣷولࣹمْ مبِنْدࣹيفْ مُيْ يࣴنگُ ݖِ ڪَوْ سُوفْ؞»
‏29 يࣵلَّ نࣹ:‏ «خࣷولْ-لࣹينْ، جࣷخْ نَا لࣹينْ گࣺݒّ گࣵنݖَخْ گُ اࣵندّ اَڪْ جِوࣷومْ ݖِ ڪَوْ سُوفْ سࣺݒّ اَڪْ گࣺݒّ گَرَبْ گُيْ مࣹݧّ تࣹيْ جُرْ جِوࣷومْ، مُيْ سࣹينْ دُندّ؞ ‏30 نَڪَ رَبِ اࣵلّ يࣺݒّ اِتْ، ݧࣷوڪْ ݒِݖِّ اَسَمَانْ يࣺݒّ اَڪْ لࣺݒّ لُيْ رَامْ ݖِ ڪَوْ سُوفْ، تࣹ دِ بࣷرࣷومْ بَڪَّنْ، جࣷخْ نَا لࣹينْ لࣺݒّ لُيْ ݧَخْ مُ تࣷويْ، ݧࣷومْ اِتَمْ، مُيْ سࣹينْ لࣹڪّ؞» مُ دَلْدِ نࣹڪّ نࣷونَ؞
‏31 يࣵلَّ نَگْ گِسْ لِ مُ دࣹفْ لࣺݒّ، فࣹڪّ لࣺݒّ بَاخْ لࣷولْ؞ نگࣷونْ جࣷتْ، سُبَ دُگّ، مُيْ بࣺسُبْ جُرٗومْ بࣹنّࣹيلْ بَ؞
