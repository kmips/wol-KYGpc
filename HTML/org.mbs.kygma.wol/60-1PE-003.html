جࣴڪّࣴرْ سَنگّ لَ، جَبَرْ دِ نَوْلࣹ
3
‏1 نَڪَ نࣷونُ يࣹينْ جِگࣺينْ ݧِ، نَنگُلْ-لࣹينْ سࣹينْ جࣴڪّࣴرْ، نگِرْ جࣴڪّࣴرْ جُ ݖِ نَنگُوُلَ گࣴمْ ڪࣵدُّ گِ اِتْ، مَنَ گِنْدِڪُ تࣹ وَخْ دُ ݖِ اَمْ يࣷونْ، نگِرْ نَ جَبَرَمْ دِ جࣴفࣹ، ‏2 ندَخْ دَيْ خࣷولْ ݖِ يࣹينْ اَڪْ سࣹينُگْ سࣹلَّلْ گُ اࣵندّ اَڪْ وࣹگࣹيلْ؞
‏3 سࣹينْ گࣵنجَرْ بُمُ دࣷونْ گࣵنجَرُ بِتِ بُ يࣹمْ ݖِ اَيْ لࣺتّ يُ رَفࣹتْ، اَڪْ تَڪَّايِ وُرُسْ، اَڪْ يࣺرࣹ يُ يࣵنجّ، ‏4 وَايࣹ نَيْ لُ لࣵقࣷو جࣴمُّ بِيرْ، نَيْ گࣵنجَرْ گُ دُلْ جࣹيخْ، تࣹ مࣷويْ لࣹوࣹتَايْ اَڪُگْ تࣹييْ، ندَخْ لࣷولࣷو رࣺيْ نجَرْتࣹ فَ يࣵلَّ؞ ‏5 بُ جࣴڪّࣷونْ لࣷولُ مࣷو دࣷونْ گࣵنجَرُ جِگࣺينْ ݧُ سࣹلّ ݧَ يَاڪَارࣷونْ يࣵلَّ تࣹ نَنگُلْ سࣹينْ جࣴڪّࣴرْ، ‏6 نࣷونُ سَارَتَ دࣺگَّلࣹ وࣷونْ إِبْرَاهِيمَ، نَانْ ڪࣷ «سَنگّ بِ»؞ يࣹينْ جِگࣺينْ ݧِ نَگْ، يࣹينْ اَيْ دࣷومِ سَارَتَ، لِ فࣹيڪْ يࣹينَ نگِ دࣹفْ لُ بَاخْ تࣹ بَݧْ ݖࣹي رَگَلْ ڪُ لࣹينْ خُݒّ؞
‏7 نَڪَ نࣷونُ يࣹينْ گٗورْ ݧِ، بࣵيِّلࣹينْ خࣹلْ نِ نگࣹينْ دࣴڪّࣹ اَڪْ سࣹينِ جَبَرْ، تࣹيْ سٗورَالࣹ نࣺيوْ دࣷولࣹيْ جِگࣺينْ؞ سࣺدّلࣹينْ لࣹينْ تࣹرَانگَيْ نَوْلࣹ، گَنَّاوْ يࣹينَ بࣷڪّ وࣹنّ يِوْ وِ لࣹينْ مَيْ نگࣹينْ دُندّ دࣴگّ؞ سُ ڪࣷ دࣹفࣹي دَرَ دُ گࣵلَّنڪࣷورْ سࣹينِ ݧَانْ؞
سࣷݒَّنْتࣹلࣹينْ تࣹ سَخࣷو بَاخْ
‏8 لِ مَيْ تࣴنڪࣹ نَگْ مࣷو دِ يࣹينْ ݧࣺݒّ، نگࣹينْ مࣵنڪࣷو، يࣺگَنْتࣹ، بࣴگَّنْتࣹ نِ مُ وَرࣹ اَيْ بࣷڪِّ گࣴمْڪَتْ، بَاخࣹينْتࣹلࣹينْ تࣹ وࣷيࣷفْ؞ ‏9 لُ بࣷنْ، بُلࣹينْ ڪࣷ فࣹيࣹ لُ بࣷنْ؛ اَيْ سَاگَ اِتْ، بُلࣹينْ ڪࣷ فࣹيࣹ اَيْ سَاگَ؞ خَنَا نگࣹينْ ݧَانَلْ لࣹينْ، فࣹيࣹ لࣹينْ ڪࣷ، ندَخْ ݖِ لࣷولُ لࣹيسْ لࣹينْ وࣷو، نگِرْ نگࣹينْ جَگࣷو بَرْڪࣹ؞ ‏10 ندَخْ ڪَتْ:‏
«ڪُ بࣴگَّ بࣺگࣹ بَڪَّنَمْ،
فࣹڪّࣹ بࣺسْ بُ رَفࣹتْ،
نَ لࣵمِّݧَمْ دَوْ لُ آيْ،
دَوِ فࣹنْ؞
‏11 نَ دَگّࣷوڪْ لُ بࣷنْ، تࣹيْ دࣹفْ لُ بَاخْ،
نَ سࣵڪُّ جࣵمّ، سَخࣷو ڪࣷ؞
‏12 ندَخْ اَيْ گࣴتْ لَ بࣷرࣷومْ بِ
نࣹ جࣵڪّ اَجِ جُبْ ݧِ،
تࣹ اَيْ نࣷݒَّمْ لَ دࣺڪّ سࣹينِ ݧَانْ،
وَايࣹ بࣷرࣷومْ بِ، ڪَنَمَمْ لَيْ تࣴجَلْ
ݧِيْ دࣹفْ لُ بࣷنْ¤؞»
نَ سࣹينُگْ نجُبْ وَرَلْ سࣹينْ ݖࣷونࣷ
‏13 تࣹ اِتْ سُ نگࣹينْ دࣹي ݧُ نجَخْلَفْ ݖِ دࣹفْ لُ بَاخْ، اَنَ ڪُ لࣹينْ دِ دࣹفْ لُ بࣷنْ؟ ‏14 تࣹ سَخْ سُ نگࣹينْ دࣷونْ جࣵنڪࣷونْتࣹيڪِ ݖࣷونࣷ ندَخْ سࣹينُگْ نجُبْ، ندࣷڪّلࣹي يࣹينْ؞ بُلࣹينْ اَمْ جࣹنّ تِيتَانگࣹ ݖِ سࣹينِ خُݒّࣹ، تࣹ بُلࣹينْ جَاخْلࣹ؞ ‏15 وَايࣹ سَنگّ اَلْمَسِ نگࣹينْ دِ سࣹلَّلْ ݖِ سࣹينْ خࣷلْ، تࣹ وَاجَلْ نُ نگࣹينْ دِ تࣷنْتࣷو ڪُ لࣹينْ لاَجْ لُ جࣴمْ ݖِ يَاڪَارْ جِ نگࣹينْ اَمْ؞ ‏16 وَايࣹ نَيْ تࣷنتّ لُ رَفࣹتْ تࣹ اࣵندّ اَڪُبْ يَرْ؞ سَخࣷولࣹينْ دِ جࣴفࣹ لِ دَلَلْ سࣹينُمْ خࣹلْ، بَ مبِرْ مَ ݧُ لࣹينْ دِ بَاتَلِ، ݧࣷوݧُيْ ݝࣵݧّ سࣹينْ دُنْدِنْ وُ رَفࣹتْ وِ لࣹينْ اَلْمَسِ مَنْلࣷو، گࣵݖّࣹ رࣹڪّ لَݧُيْ جࣴلࣹ ݖَ مبِرْ مَ ݧُ لࣹينْ دِ بَاتَلِ؞
‏17 ݖࣷونࣷ يُ لَ جࣴفْ جُ بَاخْ يٗبّࣹ ڪَتْ، ندࣹگَمْ دَ دِ ݖࣷوبَرࣹيْ يࣵلَّ، مࣷو گࣴنْ ݖࣷونࣷ يُ لَ جࣴفْ جُ بࣷنْ يٗبّࣹ؞ ‏18 اَلْمَسِ اِتَمْ بࣹنّ يࣷونْ لَ دَجْ ݖࣷونࣷ ندَخْ بࣵڪَّارِ نِتْ ݧِ، مࣷومْ مِيْ ڪِ جُبْ مࣷو لࣹينْ ڪࣷ دࣹفَلْ يࣹينْ ݧِ جُبَدِ، نگِرْ يٗبُّ لࣹينْ فَ يࣵلَّ؛ مࣷومْ لَݧُ رࣹيْ يَرَمُ سُوخَمْ، وَايࣹ مࣷو دُنْدَاتْ ݖِ دࣷولࣹيْ نࣷو گِ؞ ‏19 ݖِ دࣷولࣹيْ نࣷو گࣷوگُ لَ دࣹمࣹ بَ وَارِ بِنْدࣹيفْ يَ سࣹينِ فِتْ تࣴجࣹ¤، ‏20 ݧࣷوݧَ گࣴمَدِ وࣷونْ بُ يࣵگّ، جَنتّ يَ يࣵلَّيْ مُݧْ، دِ نࣺگَنْدِڪُ، تࣹ نُوحٍ دِ يࣹتّ گَالْ، گَ لِمْ بُ نࣺيوْ، جُرٗومْ ݧࣹتّ، مُݖّࣹ وࣷونْ ݖَ ندࣷخْ مَ؞ ‏21 ندࣷخْ مࣷومَ دِ مِسَالُبْ سٗوبُ گِ لࣹينْ مُسَلْ تࣹيْ؛ دُ سٗوبُ گُيْ دِنْدِ تࣴرُبْ يَرَمْ، وَايࣹ مࣷومْ نگࣹينْ دِ وُيࣷو وࣷوتࣹبْ يࣵلَّ ݖِ خࣹلْ مُ دَلْ، تࣹ لِ لࣹينْ دِ مُسَلْ مࣷويْ ندࣹڪِّتࣹلْ عِيسَىٰ اَلْمَسِ، ‏22 مِ يࣺيگْ اَسَمَانْ، تࣷوگْ فَ ندِجࣷورُ يࣵلَّ، ڪِ مَلاَڪَ يࣹيڪْ بࣷرࣷومْ سَݧْ-سَݧْ يࣹيڪْ بࣷرࣷومْ دࣷولࣹ يِ، بࣷڪّ نࣹڪّ ݖِگْ ݖُرْگَامْ؞
