2
‏1 بَ لࣷولُ اَمࣹي نُ وَلْبَتِڪُ، دࣹلُّ مࣵنْدِݝْ مَ، تࣷݒّ يࣷونُ گࣺيجُ بَرَخْ يَ، نَ مَ ڪࣷ اَجِ سَخْ جِ وَخࣹ وࣷونْ؞ وࣴرْ نَنُ دِيوَانُ تُنْدُ سࣹيِرْ اَيْ فَنِ فَنْ؞
بࣵنِّ إِسْرَاءِيلْ تࣹگِّ نَݧُ
‏2 اَجِ سَخْ جِ وَخْ مَ، نࣹ مَ:‏ ‏3 «سࣹينْ وࣴرُبْ تُندّ وِي دࣷيْ نَ؞ وَلْبَتِڪُلࣹينْ، جُبْلُ بࣴجْ-گࣵنَّارْ؞ ‏4 مبࣷولࣷو مِ نگَيْ سَنتّ نࣹ لࣹينْ:‏ ‹يࣹينَ نگِ وَاجَ جࣺگِّ سࣹينْ ڪࣹمُ رࣺيوُمْ بࣷڪِّ اࣹسَوࣹينْ، ݧِ دࣴڪّࣹ سࣹيِرْ؞ دِنَݧُ لࣹينْ رَگَلْ، وَايࣹ نَنگࣹينْ مࣷيْتُ بُ بَاخْ؞ ‏5 بُلࣹينْ خࣹيخْ اَڪْ ݧࣷومْ، ندَخْ دُمَ لࣹينْ جࣷخْ ݖِ سࣹينْ سُوفْ فُ تࣺسْتࣴنْ سَخْ تࣹگࣹ، نگِرْ اࣹسَوࣹينْ ݧِ لاَ سࣺدّ دِيوَانُ تُنْدُ سࣹيِرْ؞ ‏6 اَوْ ݧَمْ، وࣹݖِّڪࣷو لࣹينْ ڪࣷوڪْ ݧࣷومْ خَالِسْ، دࣷورْ ڪࣷو لࣹڪّ؛ اَمْ ندࣷخْ اِتْ، نَنگࣹينْ ڪࣷ جࣴنْدࣹيڪْ ݧࣷومْ، دࣷورْ ڪࣷو نَانْ؞ ‏7 ندَخْ يࣹينْ ڪَتْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ مࣷو بَرْڪࣹيلْ سࣹينْ مبࣷولࣹمُ ݧَقْ تࣹ مࣷو رࣵݧّࣹي سࣹينْ دࣷخُبْ مࣵنْدِݝْ مُ يَا مِي؞ ݧࣹينتّ فُڪِّ اَتَ نگِي سࣹينْ يࣵلَّ اَجِ سَخْ جَا نگࣹيڪْ يࣹينْ، ݧࣵڪُّلࣹينْ دَرَ؞›» ‏8 بَ مُ ڪࣷ دࣹفࣹي نُ رࣷمبّ سُنُ بࣷڪِّ اࣹسَوࣹينْ ݧِ دࣴڪّࣹ سࣹيِرْ؞ دَنࣷو وَݖّ يࣷونُ جࣷورْ گَ، اَڪْ اࣹلَتْ اَڪْ اࣹسِيࣷنْ گࣹبࣹرْ، دَلْدِ وَلْبَتِڪُ جَارࣹجِ مࣵنْدِݝُ مࣷوَبْ؞
‏9 بَ لࣷولُ اَمࣹي اَجِ سَخْ جِ نࣹ مَ:‏ «بُلࣹينْ دَلْ ݖِ ڪَوْ مࣷوَبࣹينْ ݧِ، دِ لࣹينْ سࣷنگّ اُبْ خَرࣹ، نگِرْ دُمَ لࣹينْ جࣷخْ بࣹنّ ݖࣺرْ ݖِ سࣹينُمْ رࣺيوْ، ندَخْ اَسْڪَنُ لٗوتْ لاَ سࣺدّ رࣺيوُمْ اَرْ؞»
‏10 اࣹمࣹينْ ݧَا فَ جࣴڪّࣷونَ دࣴڪّ، اَسْڪَنْ وُ اَمْ ڪࣵتَّنْ لَݧُ وࣷونْ، بَرࣹ، يࣹگّ ݖَ ڪَوْ نِ اَنَگࣹينْ ݧِ؞ ‏11 اَيْ رࣹفَيࣹينْ لَݧُ جࣵݒّࣹ وࣷونْ اࣹمࣹينْ ݧࣷوݧُ اَڪْ اَنَگࣹينْ ݧِ، تࣹيوُلْ مࣷوَبࣹينْ ݧِ دِ لࣹينْ وࣷويࣹ اَيْ اࣹمࣹينْ؞ ‏12 سࣹيِرْ نَگْ لَ اࣷرࣹينْ ݧَ جࣴڪّࣷونَ دࣴڪّ، وَايࣹ اࣹسَوࣹينْ ݧَا لࣹينْ فَ دࣵقࣹ، فَاگَاگَلْ لࣹينْ، وُوتُ لࣹينْ، نࣷونُ إِسْرَاءِيلْ دࣹفْ رࣹڪّ اَڪْ ݖࣺرُ سُوفَمْ بِ لࣹينْ اَجِ سَخْ جِ جࣷخْ؞
‏13 اَجِ سَخْ جِ نࣹيتِ:‏ «جࣵلّ-لࣹينْ وَلُمْ سࣹرࣹدّ؞» نُ جࣵلّ وَلُمْ سࣹرࣹدّ؞ ‏14 دِگَّنْتࣹ بَ نُ بࣵيِّڪࣷو ڪَدࣹسْ بَرْنࣹيَ، دِ دࣷخْ، بَ ڪࣹرࣷوگْ بࣺسْ بَ نُ جࣵلّࣹي وَلُمْ سࣹرࣹدّ، فَنْوࣹيرِ اَتْ اَڪْ جُرٗومْ ݧࣹتّ لَ؞ فࣹڪّ نَ مبࣷولࣹمْ مَاسُ گٗورْ ݧَ تࣷلُّ وࣷونْ ݖِ خَرࣹ، ݖَ ندࣷورْتࣹلْ يࣷونْ وَ، ݧࣷومْ ݧࣺݒَّ وُوتࣹ دَلْ بَ، دࣹي بَ جࣹيخْ، نَ لࣹينْ ڪࣷ اَجِ سَخْ جِ گِݧَلࣹ وࣷونْ؞ ‏15 اَجِ سَخْ جࣹي لࣹينْ تࣷݒّࣹ دِيسَايُ لࣷخࣷومْ، وࣷلّࣹي لࣹينْ ݖِ بِيرْ دَلْ بِ، بَ ݧُ جࣹيخْ تࣵڪّ؞
‏16 بَ مبࣷولࣹمْ ݧَ تࣷلُّ وࣷونْ ݖِ خَرࣹ دࣹييࣹي بَ جࣹيخْ، بَ ݧࣷومْ ݧࣺݒّ وُوتࣹ ݖَ مبࣷولࣷو مَ، ‏17 اَجِ سَخْ جِ وَخْ مَ، نࣹ مَ:‏ ‏18 «يَا نگِ وَاجَ جࣺگِّ ڪࣹمُ رࣺيوُمْ مࣷوَبْ، ݖَ اَرْ؞ ‏19 بُ نگࣹينْ اࣵگّࣹي ݖَ اَمࣷنࣹينْ ݧَ نَگْ، بُلࣹينْ دَلْ ݖِ سࣹينْ ڪَوْ، بُلࣹينْ خࣹيخْ اَڪْ ݧࣷومْ، ندَخْ دُمَ لࣹينْ جࣷخْ بࣹنّ ݖࣺرْ ݖِ رࣺيوُمْ اَمࣷنࣹينْ ݧِ، نگِرْ اَسْڪَنُ لٗوتْ وࣷووُ لاَ ڪࣷ سࣺدّ؞» ‏20 دِيوَانْ بࣷوبُ اِتْ رࣺيوُمْ رࣹفَيࣹينْ لࣹيسْ ڪࣷ جࣵݒّࣹ وࣷونْ، تࣹ رࣹفَيࣹينْ ݧَا فَ جࣴڪّࣷونَ دࣴڪّ؞ تࣹيوُلْ اَمࣷنࣹينْ ݧِ دِ لࣹينْ وࣷويࣹ اَيْ سَمْسُمࣹينْ، ‏21 اَسْڪَنْ وُ اَمْ ڪࣵتَّنْ، بَرࣹ، تࣹ يࣹگّ ݖَ ڪَوْ نِ اَنَگࣹينْ ݧِ؞ اَجِ سَخْ جࣹي لࣹينْ فࣵڪَّسَلْ اَمࣷنࣹينْ ݧِ، اَمࣷنࣹينْ ݧِ دࣵقْ لࣹينْ، سَنݖّ، وُوتُ لࣹينْ؞ ‏22 مࣷويْ نَ مُ دࣹفَلࣷونْ اࣹسَوࣹينْ ݧَ دࣴڪّࣹ سࣹيِرْ رࣹڪّ، بَ مُ لࣹينْ فࣵڪَّسَلࣹي اࣷرࣹينْ ݧَ، اࣹسَوࣹينْ ݧَ دࣵقْ لࣹينْ، سَنݖّ، وُوتُ لࣹينْ، بَ بࣺسُبْ تࣹيْ جِي؞ ‏23 اَوࣹينْ ݧَ دࣴڪّࣹ وࣷونْ دࣴڪّ يُ ندَوْ يَ دࣹمْ، بَ اࣵڪِّ گَسَ نَگْ، ڪَفْتࣷرࣹينْ ݧَ بࣵيِّڪࣷو ڪࣹرࣹتْ ݧࣷو لࣹينْ فَاگَاگَلْ، وُوتُ لࣹينْ؞
إِسْرَاءِيلْ نَنگُ نَ ݒࣹنڪُبْ دࣹخُ يُرْدَنْ
‏24 اَجِ سَخْ جِ نࣹيتِ:‏ «اَيْݖَ، جࣵلّ-لࣹينْ وَلُمْ اَرْنࣷنْ؞ مُ نگࣷوگْ، مَا تࣹگْ ݖِ سࣹينْ لࣷخࣷ، سِوࣷنْ اَمࣷرࣹينْ بَ، بُورُبْ اࣹسْبࣷنْ، مࣷوڪْ رࣺيوَمْ؞ دَلْ-لࣹينْ دِ نَنگُ رࣺيوْ مِ، سࣷنگّ ڪࣷ خَرࣹ؞ ‏25 بࣺسْ نِڪِ تࣹيْ مَا تࣵمْبَلࣹي وࣵݖّࣹ نجࣵقَرࣹيڪْ تِيتَانگࣹ ݖِ ڪَوْ مبࣷولࣹمْ اَسْڪَنْ يِ اَسَمَانْ تِيمْ، بَ بُ ݧُ دࣺگّࣹي سࣹينُمْ رِيرْ، دَݧُيْ جَاخْلࣹ، دِ لࣷخْ؞»
‏26 ݖِ ڪَوْ لࣷولُ مَ يࣹبَلْ اَيْ ندَوْ يُ بࣵيِّڪࣷو مࣵنْدِݝُ ڪࣹدࣹمࣷتْ، بَ ݖَ سِوࣷنْ بُورُبْ اࣹسْبࣷنْ؞ مَ يٗبَّنْتࣹ ڪࣷ ڪࣵدُّيْ جࣵمّ، نࣹ ڪࣷ:‏ ‏27 «مَيْ نُ، نُ جَارࣹ سَمْ رࣺيوْ؞ يࣷونْ وِ رࣹڪّ لَنُيْ تࣷݒّ؛ دُنُ جࣵدّ ندِجࣷورْ، دُنُ جࣵدّ ݖࣵمّࣷݧْ؞ ‏28 اَوْ ݧَمْ، خَالِسْ نگَ نُ ڪࣷيْ وࣹݖّࣹي، نُ لࣹڪّ؛ اَمْ ندࣷخْ سَخْ، خَالِسْ نگَ نُ ڪࣷيْ جࣷخࣹ، نُ نَانْ؞ مَيْ نُ رࣹڪّ نُ دࣷخْ، بَ جࣵلّ؞ ‏29 نࣷونُ نُ اࣹسَوࣹينْ ݧَ دࣴڪّࣹ سࣹيِرْ دࣹفَلࣷونْ، ݧࣷوڪْ مࣷوَبࣹينْ ݧَ دࣴڪّࣹ اَرْ رࣹڪّ؞ سُ ڪࣷ دࣹفࣹي نُ مَنَ جࣵلّ دࣹخُ يُرْدَنْ، بَ اࣵگّ ݖَ رࣺيوْ مَ نُ سُنُ يࣵلَّ اَجِ سَخْ جِ جࣷخْ؞»
‏30 سِوࣷنْ بُورُبْ اࣹسْبࣷنْ نَگْ نَنگُوُ نࣷو مَيْ نُ جَارࣹ فَ مࣷومْ، ندَخْ بࣷوبَ سࣹينْ يࣵلَّ اَجِ سَخْ جِ مࣷو دࣴگࣴرَلْ بࣷݒَّمْ، وࣷوَلْ خࣷلَمْ، نگِرْ تࣹگْ ڪࣷ ݖِ سࣹينْ لࣷخࣷ نِڪِ بࣺسُبْ تࣹيْ جِي؞ ‏31 اَجِ سَخْ جِ نࣹ مَ:‏ «گِسْ نگَ، مَا نگِ نِي تࣵمْبَلࣹي تࣹگْ سِوࣷنْ اَڪُمْ رࣺيوَمْ ݖِ سࣹينْ لࣷخࣷ؞ دَلْ-لࣹينْ دِ نَنگُمْ رࣺيوَمْ؞» ‏32 بَ مُ ڪࣷ دࣹفࣹي سِوࣷنْ دِڪّ دَجࣹيڪْ نُنْ، مࣷوڪْ گࣵنگࣷورَمْ گࣺݒّ، نگِرْ سࣷنگّ نُ خَرࣹ ݖَ يَخَݖّ؞ ‏33 سُنُ يࣵلَّ اَجِ سَخْ جِ دَانࣹ ڪࣷ فِ سُنُ ڪَنَمْ، نُ دُمَ ڪࣷ، مࣷوڪْ اَيْ دࣷومَمْ، اَڪْ گࣵنگࣷورَمْ گࣺݒّ؞ ‏34 جَنتّ يࣷويُ لَنُ نَنگُ دࣴڪَّمْ يُ مَگْ يࣺݒّ، فَاگَاگَلْ دࣴڪّ يُ مَگْ يࣺݒّ:‏ گٗورْ اَڪْ جِگࣺينْ، بَ ݖَ تُوتْ-تࣵنڪّ يَ، بࣵيِّوُنُ ڪُيْ دُندّ؞ ‏35 خَنَا جُرْ گَ لَنُ سࣴخࣴتࣷو، اَڪْ مبࣷولࣹمْ اَلَلِ دࣴڪّ يَ نُ نَنگُ؞ ‏36 لَ ڪࣷ دَلࣹ اَرࣷوࣹرْ گَ تِيمْ وَلُمْ اَرْنࣷنْ، بࣷولࣹ ݖَ دࣴڪّ بَ ݖَ خُرْ وَ، بَ ݖَ گَلࣵدّ، دُ بࣹنّ دࣴڪّ بُ نُ تࣴ؞ فࣷوفُ فࣺݒّ لَ سُنُ يࣵلَّ اَجِ سَخْ جِ دَانࣹ فِ سُنُ ڪَنَمْ؞ ‏37 رࣺيوُمْ اَمࣷنࣹينْ ݧَ دࣷݝّ نگࣹينْ جࣹگࣹوُلْ؛ مُيْ رࣺيوْ مَ لࣹرُ دࣹخُ يَبࣷگْ، اَڪْ دࣴڪِّ تُندّ يَ، مبࣷولࣹمْ دَالْ، فُ نُ سُنُ يࣵلَّ اَجِ سَخْ جِ آيࣹ وࣷونْ؞
