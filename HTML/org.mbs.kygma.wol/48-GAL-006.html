دِمْبَلَنْتࣹلࣹينْ
6
‏1 بࣷڪِّ گࣴمْڪَتْ ݧِ، سُ ݧُ بࣹتّࣹي نِتْ ݖِ اَگْ تࣷوݧْ، يࣹينْ ݧِ تࣷݒّ نࣷووُگْ يࣵلَّ، وَݧِّلࣹينْ ڪࣷ ندࣵنڪّ، تࣹ ڪࣹنّ ڪُ نࣹڪّ ݖِ يࣹينْ خࣷولاَتْ بࣷݒَّمْ بُ بَاخْ، نگِرْ سࣹينْ بَڪَّنْ بَݧْ لࣹينَ خِيرْتَلْ، بَ نگࣹينْ دَانُ ݖِ فِيرْ گُ نِ مࣹلْ؞ ‏2 يࣹنِّڪࣷونْتࣹلࣹينْ سࣹينِ يࣹنْ، سُ بࣷوبَا نگࣹينْ دِ تࣷݒّ يࣷونُ اَلْمَسِ بَ مُ مَتْ سࣴڪّ؞ ‏3 ڪُ فࣷوگْ نࣹ مࣷو اَمْ سࣷلࣷ، تࣹ دُ دَرَ نَگْ، بࣷݒَّمْ لَيْ نَخْ؞ ‏4 ڪُ ݖِ نࣹڪّ، جࣴفِ بࣷݒَّمْ، نَ ڪࣷ نِيرْ؞ سُ ݖِ اَمࣹي لࣹنّ لُ مُ مَنَ سَگࣷو، نَ ڪࣷ يࣹمَلࣹ ݖِ بࣷݒَّمْ، تࣹ بَݧَ سَگࣷو لࣹنِّ بࣷݒَّمْ لُ مُ خࣷولْ ݖِ ݧࣹنࣹينْ، ‏5 ندࣹ ڪُ نࣹڪّ اَڪْ سَ اَݧُ جࣴفِ بࣷݒّ؞
‏6 ڪِ ݧُيْ جࣵنگَلْ ڪࣵدُّگْ يࣵلَّ نَگْ وَرْ نَا سࣺدّ ڪِ ڪࣷيْ جࣵنگَلْ ݖِ مبࣷولࣹمْ اَلَلَمْ؞
‏7 بُ لࣹينْ لࣹنّ نَخْ؛ يࣵلَّ دࣹيسُ ڪࣷ فࣷونْتࣷو؞ نِتْ ڪِ، لِ مُ جِ، مࣷومْ لَيْ گٗوبْ؞ ‏8 ڪُ جِيَلْ بࣵنّࣹيخُ بَڪَّنَمْ، ݖِ بࣵنّࣹيخُ بَڪَّنَمْ لَيْ جࣴلࣹ نگٗوبُمْ يࣵقُتࣹ؛ ڪُ جِيَلْ نࣷووُگْ يࣵلَّ، ݖِ نࣷو گِ نگَيْ گٗوبࣹ تࣹخࣹ گُ سَخْ دࣵڪّ؞ ‏9 بُنُ تࣵيِّ ݖِ دࣹفْ لُ بَاخْ، ندࣹ بُ جࣷتࣹي رࣹڪّ نࣷويْ گٗوبْ، ندࣹگَمْ يࣷقِوُنُ؞ ‏10 ڪࣷنْ نَگْ بِ نُ ڪࣷيْ جࣹڪُّ، نَنُيْ دࣹفْ لُ بَاخْ، نگِرْ ݧࣺݒّ، رَوَتِنَ وَا ڪࣴرْ يࣷونُ نگࣴمْ وِ؞
نَ گࣴمْڪَتْ بِ دࣷونْ مبِنْدࣹيفْ مُ بࣹيسْ
‏11 سࣹيتْلُلࣹينْ مبِندّ مِ؛ ݖِ اَرَفْ يُ مَگْ يِي نگࣹينْ وَرَ خࣵمّࣹي، نࣹ سَمَ لࣷخࣷلْ بࣷݒّ لاَ ڪࣷ بِنْدࣹ!‏
‏12 ݧِ لࣹينْ دِ گࣺتࣴنْ ݖِ خَرَفْ مࣷويْ ݧِ نگِسْتَلْ تَخْ ݧُ بࣴگّ نِتْ ݧِ گࣴرࣴمࣹ لࣹينْ سࣹينْ مࣹلࣷوْ يَرَمْ، تࣹ جُبْلُوُݧُ ݖِ لُ مࣷيْ رَگَلْ ڪُ لࣹينْ بُندّخَتَلْ ندَخْ دࣹيوُگْ اَلْمَسِ گِ ݧُيْ خَمْلࣹ نࣹ مࣷويْ تَخَ مُݖّ؞ ‏13 ݧِ خَرَفْ ݧࣷومْ ݖِ سࣹينْ بࣷݒّ سَخْ تࣷݒُّݧُ يࣷونْ وِ؞ تࣹيوُلْ ݧُ بࣴگّ نگࣹينْ خَرَفْ، نگِرْ مَنَ دَمࣷو سࣹينْ مࣵنْدَرْگَمْ يَرَمْ، يࣹينْ ݧِ ݧُ خَرَفْلࣷو؞ ‏14 وَايࣹ مَنْ دࣹ يࣵلَّ بُمَ دَمࣷو لࣹنّ لُ مࣷيْ دࣹيوُگْ سُنُ بࣷرࣷومْ عِيسَىٰ اَلْمَسِ ݖَ بَنتّ بَ؞ بَنتّ بࣷوبُ مࣷو تَخْ اࣵدِّنَ دِ لُ ݧُ مَ دَاجَلْ ݖِ بَنتّ، بَ اࣵدِّنَ دࣹي؛ مَنْ اِتْ مَيْ لُ ݧُ دَاجَلْ اࣵدِّنَ ݖِ بَنتّ بَ مَ دࣹي؞ ‏15 خَرَفْ اَڪْ ݧࣵڪَّ خَرَفْ، دَرَ اَمُ ݖِ سࣷلࣷ؞ لِ اَمْ سࣷلࣷ مࣷويْ دࣷونْ مبِنْدࣹيفْ مُ بࣹيسْ؞ ‏16 مبࣷولࣹمْ ݧِيْ سࣵمّ ندِگَلْ لِي نَگْ، تࣹ دِ إِسْرَاءِيلُ يࣵلَّ، جࣵمّ اَڪْ يࣴرْمَنْدࣹ ݧࣹيلْ نَ لࣹينْ؞
‏17 لِ ݖِ دࣹسْ مࣷو دِ گَنَّاوْ-سِ-تࣹيْ، بُ مَ ڪࣹنّ لࣴجَلَتِ، ندَخْ لࣺگࣺتْ يِ ݖِ سَمَ يَرَمْ، سَمَگْ بࣷڪّ ݖِ عِيسَىٰ لَ مࣵنْدَرْگَالْ؞
‏18 بࣷڪّ يِ، يَلْ نَ سُنُ يِوُ بࣷرࣷومْ عِيسَىٰ اَلْمَسِ اࣵندّ اَڪْ سࣹينُمْ خࣹلْ؞ اَمِينْ؞
