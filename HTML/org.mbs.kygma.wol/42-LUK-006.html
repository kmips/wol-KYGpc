بࣺسُبْ نࣷفْلاَيْ، عِيسَىٰ مࣷويْ بࣷرࣷومْ
6
‏1 بࣺسُبْ نࣷفْلاَيْ لَ وࣷونْ، عِيسَىٰ دࣷونْ جَارࣹ اَيْ تࣷولْ يُ ݧࣷرْ، تَالِبࣹيمْ يَ فࣵقࣹ ݖَ اَيْ گُبْ، دِ بࣷخࣷمْ، دِ لࣹڪّ؞ ‏2 ݧࣹنّ ݖَ فَرِسࣹنْ يَ نࣹ لࣹينْ:‏ «لُ وَرَلْ نگࣹينْ دِ دࣹفْ لُ يࣷونْ مَيࣹوُلْ ݖِ بࣺسُبْ نࣷفْلاَيْ؟» ‏3 عِيسَىٰ نࣹ لࣹينْ:‏ «خَنَا جࣵنگُلࣹينْ لَ دَاوُدَ دࣹفࣷونْ، بَ مُ خِيفࣹي، مࣷومْ، اَڪْ ݧَ مُ اࣵنْدَلࣷونْ؟ ‏4 بَ مُ دُگّࣹي ݖَ ڪࣴرْ يࣵلَّ گَ، مبُرُمْ تࣹيوَلْ مَ لَ جࣴلْ، لࣹڪّ، جࣷخْ ݖَ ݧَ مُ اࣵنْدَلْ، تࣹ يࣷونْ مَيࣹوُلْ ڪࣹنّ ڪُ ݖَ لࣹڪّ ڪُ دُلْ سَرْخَلْڪَتْ يِ؞» ‏5 مُ نࣹيتِ لࣹينْ:‏ «بࣺسُبْ نࣷفْلاَيْ، دࣷومُ نِتْ ڪِ مࣷويْ بࣷرࣷومْ؞»
عِيسَىٰ فَجْ نَ نِتْ ڪُ لࣷخࣷومْ لࣵگِّ
‏6 بࣹنࣹينْ بࣺسُبْ نࣷفْلاَيْ، عِيسَىٰ دُگّ ݖَ جࣵنگُ بَ، دِ جࣵنگّلࣹ؞ جࣹنّ وَايْ جُ لࣷخࣷلْ ندِجࣷورَمْ لࣵگِّ مَ نگَ فَ وࣷونْ؞ ‏7 فِرِڪَتِ يࣷونِ مُوسَىٰ يَاڪْ فَرِسࣹنْ يَا نگَيْ خࣷولْ عِيسَىٰ، بَ خَمْ ندَخْ دِنَ ڪࣷ فَجْ ݖَ بࣺسُبْ نࣷفْلاَيْ بَ، نگِرْ ݧُ اَمْ لُ ݧُ ڪࣷ تُومَالࣹ؞ ‏8 عِيسَىٰ نَگْ خَمْ سࣹينْ خَلاَتْ، دَلْدِ نࣹ بࣷرࣷومْ لࣷخࣷ بَ:‏ «جٗگَلْ تَخَوْ فِي ݖِ دِگّ بِ؞» مُ جٗگْ تَخَوْ؞ ‏9 عِيسَىٰ نࣹ لࣹينْ:‏ «لاَجْ نَا لࣹينْ، اَنَ لُ يࣷونْ مَيࣹ ݖِ بࣺسُبْ نࣷفْلاَيْ؛ دࣹفْ لُ بَاخْ اَمْ لُ بࣷنْ؟ مُسَلْ بَڪَّنْ اَمْ لࣷرࣹ؟» ‏10 ݖِ ڪَوْ لࣷولُ مُ دَوَلْ بࣴتَمْ، خࣷولْ ݧَ ڪࣷ يࣺيوْ ݧࣺݒّ، بَ نࣷݒِّ نࣹ وَا جَ:‏ «تࣵلَّلَلْ سَ لࣷخࣷ؞» مُ تࣵلَّلْ، لࣷخࣷ بَ وࣺرْ؞ ‏11 ݧُ مࣹرْ نَگْ بَ فࣹيسْ، دِ دِيسࣷو ݖِ سࣹينْ بِيرْ نُ ݧُيْ دࣹفْ اَڪْ عِيسَىٰ؞
عِيسَىٰ تࣵنّ نَ فُڪِّ ندَوْ اَڪْ ݧَارْ
‏12 فَنْ يࣷويُ لَ عِيسَىٰ دࣹمْ ڪَوْ تُندّ وَ نگِرْ ݧَانِ فَ؛ مُ فَنَانࣹ فَا ݧَانْ ݖِ يࣵلَّ؞ ‏13 بَ بࣴتْ سࣹتࣹي، مُ وࣷو اَيْ تَالِبࣹيمْ، تࣵنّ ݖَ فُڪّ اَڪْ ݧَارْ ݧِي، تُدّࣹ لࣹينْ اَيْ ندَوَمْ:‏
‏14 سِمࣷݝْ مِ عِيسَىٰ دِ وࣷويࣹ اِتَمْ ݒِيࣹيرْ،
اَڪْ اࣵندّرࣹ مِ بࣷڪّ اَڪْ ݒِيࣹيرْ ندࣹيْ اَڪْ بَايْ،
اَڪْ يَعْقُوبَ اَڪْ يࣷوَانْ،
اَڪْ فِلِبْ اَڪْ بَرْتࣹلࣹمِ،
‏15 اَڪْ مَݖࣴ اَڪْ تࣷمَا،
اَڪْ يَعْقُوبَ دࣷومُ اَلْفࣹ،
اَڪْ سِمࣷݝْ مِ ݧُيْ وَخْ فَرْلُڪَتُ مࣷومْ-سَ-رࣺيوْ،
‏16 اَڪْ يُدَ دࣷومُ يَعْقُوبَ،
اَڪْ يُدَ اِسْڪَرْيࣷ مَ مُجّ وࣷرْ؞
نࣹڪِّنُ تࣹيْ، يࣷولُبْ اࣴلّࣴگْ
‏17 ݖِ ڪَوْ لࣷولُ عِيسَىٰ اࣵندّ اَڪْ ݧࣷومْ، وࣵݖّ، دَلْدِ تَخَوْ ݖَ جࣷورْ گَ؞ اَيْ تَالِبࣹيمْ يُ بَرࣹي نگَ فَ وࣷونْ اَڪْ مبࣷولࣷو مُ رࣺيَ رࣺيْ:‏ اَيْ نِتِ يُدࣹ گࣺݒّ، اَڪْ يࣹرُسَلࣹمْ اَڪْ تِرْ اَڪْ سِدࣷنْ، دࣴڪِّ وࣹتُ تࣹفࣹسْ گَ؞ ‏18 ݧُ دِڪّ، دِ ڪࣷ دࣺگْلُ تࣹيْ فَجُ بَ تࣵگّࣷوڪْ سࣹينِ جࣵنگࣷرࣷ؞ ݧِ رَبْ جࣵݒّ اِتْ، وࣺرْ؞ ‏19 مبࣷولࣷو مࣺݒَّ ڪࣷ دࣷونْ وُتَ لاَلْ ندَخْ لࣹيرْ گُيْ بࣵيِّڪࣷو ݖَ مࣷومْ، دِ فَجْ ݧࣺݒّ؞
‏20 عِيسَىٰ نَگْ خࣷولْ اَيْ تَالِبࣹيمْ، نࣹ لࣹينْ:‏
«ندࣷڪّلࣹي يࣹينْ ݧِ نࣺيوْلࣹ،
ندࣹ نگُورُگْ يࣵلَّ، يࣹينَيْ بࣷرࣷومْ؞
‏21 ندࣷڪّلࣹي يࣹينْ ݧِ خِيفْ تࣹيْ،
ندࣹ دِنگࣹينْ رࣹگّ؞
ندࣷڪّلࣹي يࣹينْ ݧِيْ جࣷويْ تࣹيْ،
ندࣹ دِنگࣹينْ رࣹي؞
‏22 ندࣷڪّلࣹي يࣹينْ، بُ لࣹينْ نِتْ ݧِ بَݧࣹي، دَگّࣷوڪْ يࣹينْ، سَاگَ لࣹينْ، سِڪَّلْ سࣹينْ دࣹرْ ندَخْ دࣷومُ نِتْ ڪِ؞ ‏23 بࣺسُبْ ڪࣹرࣷوگْ، بࣺگْلࣹينْ تࣹ بࣵنّࣹيخُ بَيْ فࣹݖّ ندَخْ سࣹينْ يࣷولُبْ اࣴلّࣴگْ بُ رࣺيْ، ندࣹ نࣷونُ لَ سࣹينْ مَامِ بَݧْ يࣷويُ دَانْ دࣹفْ اَڪْ يࣷنࣹنتّ يَ؞
‏24 وَايࣹ وٗويْ نگَلَّ يࣹينْ ݧِ بَرࣹلࣹ،
ندَخْ نࣹيخْلࣹ نگࣹينْ بَ نࣷݒِّ؞
‏25 وٗويْ نگَلَّ يࣹينْ ݧِ رࣹگّ تࣹيْ،
ندَخْ دِنگࣹينْ خِيفْ اࣴلّࣴگْ؞
وٗويْ يࣹينْ ݧِيْ رࣹي تࣹيْ،
ندَخْ دِنگࣹينْ تِيسْلُ، دِنگࣹينْ جࣷويْ؞
‏26 وٗويْ نگَلَّ يࣹينْ، بُ لࣹينْ ݧࣺݒّ دࣹي وَخَلْ لُ بَاخْ، ندَخْ نࣷونُ رࣹڪّ لَ مَامِ ݧࣷوݧُ دَانْ جࣴفࣹ اَڪْ يࣷنࣹنْتِ ݖَاخَانْ يَ؞
سࣷݒّلࣹينْ سࣹينِ نࣷونْ
‏27 «يࣹينْ ݧِ مَيْ دࣺگْلُ، دَمَ نࣹ، سࣹينِ نࣷونْ، سࣷݒّلࣹينْ لࣹينْ، سࣹينِ بَݧْ، دࣹفَلْ-لࣹينْ لࣹينْ لُ بَاخْ؞ ‏28 ݧِ لࣹينْ دِ مٗولُ، يࣺينࣹلࣹينْ لࣹينْ لُ بَاخْ، ݧِ لࣹينْ دِ سࣷخࣷرࣹ، ݧَانَلْ-لࣹينْ لࣹينْ؞ ‏29 ڪُ لࣹينْ تَلاَتَ ݖِبْ لࣹخْ، دࣺڪّلࣹينْ ڪࣷ بَ ݖَ دࣹسْ؞ ڪُ نَنگُ سࣹينْ مبُبّ مُ مَگْ، بُلࣹينْ ڪࣷ تࣹرࣹي جࣴلاَلࣹ سࣹينُبْ تُرْڪِ؞ ‏30 ڪࣺݒّ ڪُ لࣹينْ ݧَانْ، مَيْلࣹينْ ڪࣷ؛ ڪُ نَنگُ سࣹينْ يࣴفْ، بُلࣹينْ ڪࣷ ڪࣷ لاَجْ؞ ‏31 لُ نگࣹينْ بࣴگّ نِتْ ݧِ دࣹفَلْ لࣹينْ ڪࣷ نَگْ، نَنگࣹينْ لࣹينْ ڪࣷ دࣹفَلࣹ نࣷونُ؞
‏32 «سُ دࣹي ݧِ لࣹينْ سࣷݒّ دࣷݝّ نگࣹينْ سࣷݒّ، اَنَ لُ نگࣹينْ ݖِ وَرَ يࣷولࣷو؟ بࣵڪَّارْڪَتْ يِ سَخْ، ݧِ لࣹينْ سࣷݒّ، سࣷݒّ نَݧُ لࣹينْ؞ ‏33 تࣹ سُ دࣹي ݧِ لࣹينْ دࣹفَلْ لُ بَاخْ دࣷݝّ نگࣹينْ دࣹفَلْ لُ بَاخْ، اَنَ لُ نگࣹينْ ݖِ وَرَ يࣷولࣷو؟ بࣵڪَّارْڪَتْ يࣹيتْ نࣷونُ دࣷݝّ لَݧُيْ جࣴفࣹ؞ ‏34 تࣹ سُ دࣹي ݧِ نگࣹينْ يَاڪَارࣹ ندࣹلّࣷو دࣷݝّ نگࣹينْ لࣹبَلْ، اَنَ لُ نگࣹينْ ݖِ وَرَ يࣷولࣷو؟ بࣵڪَّارْڪَتْ يِ اِتْ، يَاڪَارْ ندࣹلّࣷو لُ مَتْ سࣴڪّ مࣷو تَخْ ݧُيْ لࣹبْلَنْتࣹ؞ ‏35 يࣹينْ ڪَيْ، سࣷݒّلࣹينْ سࣹينِ نࣷونْ، دࣹفْلࣹينْ لُ بَاخْ، لࣹبْلࣹلࣹينْ تࣹ بَݧَ يَاڪَارْ ندࣹلّࣷو؞ سُ بࣷوبَا سࣹينُبْ يࣷولْ رࣺيْ، نگࣹينْ دِ دࣷومِ اَجِ ڪَوࣹ جِ، ندَخْ مࣷومْ دࣹ ݧِ سَنْتَدِ تࣹ بࣷنْ، مࣷو لࣹينْ دِ بَاخࣹ؞ ‏36 نِ سࣹينْ بَايْ اَمࣹ يࣴرْمَنْدࣹ نَگْ، نَنگࣹينْ نِ اَمࣹ يࣴرْمَنْدࣹ؞
‏37 «بُلࣹينْ دِ سِڪَّلࣹ، سُ بࣷوبَا دࣹيسُلࣹينْ سِڪَّلْ؞ بُلࣹينْ يࣹيَلࣹ ڪࣹنّ اَبْ دَانْ، نگِرْ ݧُ بَݧْ لࣹينَ يࣹيَلࣹ اَبْ دَانْ؞ بَلࣹلࣹينْ، ݧُ بَالْ لࣹينْ؞ ‏38 مَيࣹلࣹينْ، ݧُ مَيْ لࣹينْ؛ نَتّ بُ دُونْ بُ ݧُ رࣷڪَّسْ، يࣴنگَلْ، مُ فࣹيسْ بَيْ تُورُ، مࣷومْ لࣹيسْ لࣹينْ دِ نࣵدِّلْ ݖِ سࣹينْ مبُبّ؞ ندَخْ نَتّ بِ نگࣹينْ دِ نَتَّلࣹ، مࣷومْ لࣹيسْ لࣹينْ دِ نَتَّلࣹ؞»
ندِمْبَلْ، ڪࣷو تَنࣹ
‏39 عِيسَىٰ لࣺيبَتِ لࣹينْ، نࣹ لࣹينْ:‏ «ندَخْ گُمْبَ دِنَ وࣷمَّتْ گُمْبَ؟ مبَا دُݧُ اࣵندّ تࣵبِّ ݖِ ڪࣵمبّ؟ ‏40 تَالِبࣹ سُتُلْ سࣴرِݧَمْ، وَايࣹ بࣺݒّ تَالِبࣹ بُ جࣵنگّ بَ وࣵݖّ، تࣷلّࣷوڪْ سࣴرِݧَمْ؞
‏41 «اَنَ لࣷويْ خࣷولࣹ خَتَخْ بِ ݖِ سَ بࣴتُ مبࣷڪّ تࣹ گࣵنجّ گِ ݖِ سَ بࣴتُ بࣷݒّ گِسُلࣷو ڪࣷ؟ ‏42 اَنَ نࣷو مَنࣹ نࣹ سَ مبࣷڪّ:‏ ‹مبࣷڪّ مࣹي، خَارَلْ، مَ دِنْدِلْ لَ خَتَخْ گِ ݖِ سَ بࣴتْ،› تࣹ يَوْ ݖِ سَ بࣷݒّ گࣵنجّ گِ ݖِ سَ بࣴتْ، گِسُلࣷو ڪࣷ؟ جِنِگَلْڪَتْ بِ!‏ جࣴڪَّلَ دِنْدِ گࣵنجّ گِ ݖِ سَ بࣴتْ؞ سُ بࣷوبَا دِنگَ گِسْ بُ بَاخْ خَتَخْ گِ ݖِ سَ بࣴتُ مبࣷڪّ، بَ دِنْدِلْ ڪࣷ ڪࣷ؞
دࣷومْ جَ، ندࣹيْ جَ
‏43 «گَرَبْ گُ نࣹيخْ دُ مࣹݧّ دࣷومْ يُ نَقَرِ، تࣹ گَرَبْ گُ نَقَرِ اِتْ دُ مࣹݧّ دࣷومْ يُ نࣹيخْ؞ ‏44 گَرَبْ گُ نࣹڪّ، مࣹݧّࣹيفَمْ لࣹيسْ ڪࣷيْ خࣵمّࣹي؞ اَگْ دࣺدّ، دࣹيسُ ݖِ وِتّࣹ دࣷومِ فِگّ؛ اَيْ دَگّ اِتْ، دࣹيسُ ݖِ جࣴلࣹ دࣷومِ رࣹسࣹݧْ؞ ‏45 نِتْ ڪُ بَاخْ، دࣹنݖُبْ نگࣴنࣹيلْ بَ ݖَ بِيرْ خࣷلَمْ، ݖَ لَيْ جࣴلࣹ لُ بَاخْ لَ مُ سࣵلّࣷو؛ نِتْ ڪُ بࣷنْ اِتْ، دࣹنݖُبْ سَفَانْ بَ ݖَ بِيرْ خࣷلَمْ، ݖَ لَيْ جࣴلࣹ سَفَانْ بَ مُ سࣵلّࣷو؞ گࣺمِّݧْ دَالْ، لَ فࣹيسْ خࣷلُ بࣷرࣷومْ لَيْ وَخْ؞
ݧَارِ تَبَخْڪَتَ نگِ
‏46 «اَنَ لُ نگࣹينْ مَ نَا ‹سَنگّ بِ، سَنگّ بِ،› تࣹ جࣴفࣹوُلࣹينْ سَمَيْ وَخْ؟ ‏47 ڪࣺݒّ ڪُ ݧࣴوْ ݖِ مَنْ، دࣺگّ سَمَيْ وَخْ تࣹ دِ ڪࣷ جࣴفࣹ، دِنَا لࣹينْ وࣷنْ نُ مُيْ مࣹلْ؞ ‏48 دَفَ مࣹلْ نِ نِتْ ڪِيْ تَبَخْ ڪࣴرْ، گَسَ گَسْ بَ مُ خٗوتْ، دَلْدِ سَمݒّ ڪࣹنُ گَ ݖَ ڪَوْ دࣷجْ وَ ݖَ سُوفْ؞ مبࣴنّ مَ دِڪّ، وَلْ مَ رࣴڪّ ڪࣴرْ گَاڪْ دࣷولࣹ تࣹ مَنُ ڪࣷو يࣴنگَلْ ندَخْ تَبَخْ بُ وࣺرْ بَ ݧُ ڪࣷ تَبَخࣹ؞ ‏49 وَايࣹ ڪُ دࣺگّ تࣹ جࣴفࣹوࣷو، دَنگَا مࣹلْ نِ نِتْ ڪُ تَبَخْ ڪࣴرْ ݖِ سُوفْ، تࣹ اَمُلْ ڪࣹنُ؞ وَلْ مَا ڪࣷيْ رࣴڪّ اَڪْ دࣷولࣹ، ڪࣴرْ گَ جࣹڪِّ مࣵبّ، يࣵقُ يَخࣹيتْ؞»
