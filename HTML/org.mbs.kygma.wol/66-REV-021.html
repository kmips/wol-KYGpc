اَسَمَانْ اَڪْ سُوفْ اَمْ نَݧُ ڪُوتَايْ
21
‏1 مَ دَلْدِ گِسْ اَسَمَانْ سُ بࣹيسْ اَڪْ سُوفْ سُ بࣹيسْ؞ فࣹڪّ نَ اَسَمَانْ سُ جࣴڪّ سَ اَڪْ سُوفْ سُ جࣴڪّ سَ وࣺيْ، تࣹ گࣺيجْ اَمَتُلْ؞ ‏2 مَ گِسْ نَگْ دࣴڪّ بُ سࣹلّ بِ، يࣹرُسَلࣹمْ گُ بࣹيسْ گِ، مُ وࣵݖّࣹ اَسَمَانْ، فَ يࣵلَّ، مࣹلْ نِبْ سࣺيتْ بُ دࣹفَرُ بَ جࣹڪّ نگِرْ جࣴڪّࣴرَمْ؞ ‏3 مَ دࣺگّ ڪࣵدُّ گُ خُمبّ جِبࣹ ݖَ نگࣵنگُنࣹ مَ، نࣹ:‏ «مࣵڪَّانُ يࣵلاَّ نگِي ݖِ بِيرْ نِتْ ݧِ؞ مࣷويْ دࣴڪّ اَڪْ ݧࣷومْ، ݧُ دِيْ ݧࣷݧَمْ؞ يࣵلَّ ݖِ بࣷݒَّمْ مࣷويْ دࣴڪّ اَڪْ ݧࣷومْ، دِ سࣹينْ يࣵلَّ؞ ‏4 مࣷويْ فࣷمݒّ سࣹينْ وࣺݒّ رࣷنگࣷوݧْ:‏ دࣹي دࣷوتُ اَمْ، دࣴجْ اَڪِ يُوخْ اَڪُ تِيسْ دࣷوتُ اَمْ، لُ جࣴڪّ لاَيْ وࣺيْ؞»
‏5 ݖِ ڪَوْ لࣷولُ ڪِ تࣷوگْ ݖِ نگࣵنگُنࣹ مِ نࣹ:‏ «مَا نگِي دِ يࣹيسَلْ لࣺݒّ!‏» مُ تࣹگْ ݖِ نࣹ مَ:‏ «بِنْدَلْ، يِي ڪࣵدُّ يُ وٗورْ لَ تࣹ دࣴگُّ؞» ‏6 مُ نࣹيتِ مَ:‏ «سࣷتِّ نَ!‏ مَايْ اَلْفَ، دِ اࣷمࣹگَ، دِ نجࣵلْبࣺينْ، دِ مُجْ؞ ڪُ مَرْ، مَا لَيْ مَيْ نگَ نَانْ ݖِ بࣴتُ ندࣷخُمْ دُندّ مِ، تࣹ دࣷو فࣹيْ دَرَ؞ ‏7 ڪࣺݒّ ڪُ اَمْ ندَمْ نَگْ، لࣷولࣷويْ دࣷونْ سَبْ ݖࣺرْ؞ مَايْ دࣷونْ سَ يࣵلَّ، نگَ دࣷونْ سَمَ دࣷومْ، ‏8 وَايࣹ ݧُ رَگَلْ ݧِ اَڪْ اَجِ گࣴمَدِ ݧِ اَڪْ دࣹفْڪَتِ ݧَاوْتࣺيفْ يِ اَڪْ بٗومْڪَتْ يِ اَڪْ مࣷيْڪَتْ يِ اَڪْ خࣴرࣴمْڪَتْ يِ اَڪْ بࣷڪَّالࣹڪَتْ يِ، مبࣷولࣹمْ نِتِ فࣹنْ ݧِ، سࣹينْ اَݧَ نگَ ݖَ دࣺيگُ سَوَرَ وَاڪْ تَمَرَخْ بَ؞ لࣷولُ مࣷو دِ ݧَارࣹيلُ دࣹي؞»
‏9 بَ لࣷولُ اَمࣹي مࣹنّ مَلاَڪَ دِڪّ؞ مُ نگِ بࣷڪّ ݖِ جُرٗومْ ݧَارِ مَلاَڪَ، يِ يࣷرْ جُرٗومْ ݧَارِ ندَبْ يِ فࣹيسْ اَڪْ جُرٗومْ ݧَارِ مُسِبَ يُ مُجّ يِ؞ مُ نࣹ مَ:‏ «دِڪَّلْ، مَ وࣷنْ لَ سࣺيتْ بِ، سࣷخْنَسْ گࣵتّ بِ؞»
‏10 ݖِ ڪَوْ لࣷولُ مُ يٗبّࣷو مَ نࣷووُگْ يࣵلَّ بَ ݖِ تُندّ وُ مَگْ تࣹ ڪَوࣹ، دَلْدِ مَيْ وࣷنْ دࣴڪّ بُ سࣹلّ بِ، يࣹرُسَلࣹمْ، مُيْ وࣵݖّࣹ اَسَمَانْ، فَ يࣵلَّ؞ ‏11 دࣴڪّ بَا نگَ اࣵندّ اَڪْ دَرَجَيْ يࣵلَّ، لࣹيرَايَمْ بَ مࣹلْ نِ ݒࣹرُ گࣵنجَرْ، نِرࣷوڪْ ݒࣹرُ جࣵسْݒࣴ، لࣹيرࣹ نِ ڪِرِسْتَالْ؞ ‏12 تَتَ جُ رࣺيْ تࣹ ڪَوࣹ لَ اَمْ اَڪْ فُڪِّ بُنتّ اَڪْ ݧَارْ يُ فُڪِّ مَلاَڪَ اَڪْ ݧَارْ تَخَوࣹ، اَيْ تُرْ بِنْدࣹ ݖَ ڪَوْ بُنتّ يَ، دِ تُرِ فُڪِّ گِيرِ إِسْرَاءِيلْ يَاڪْ ݧَارْ؞ ‏13 وࣹتُ ݒࣹنڪُ بَ، ݧࣹتِّ بُنتّ لَ؛ وࣹتُ بࣴجْ-گࣵنَّارْ دِ ݧࣹتِّ بُنتّ؛ وࣹتُ بࣴجْ-سَالُمْ، ݧࣹتِّ بُنتّ؛ وࣹتُ سࣷوُ بَ، ݧࣹتِّ بُنتّ؞ ‏14 تَتَيْ دࣴڪّ بَا نگَ سَمْݒࣹ ݖِ فُڪِّ دࣷجِ ڪࣹنࣷوڪْ ݧَارْ يُ ݧُ بِندّ فُڪِّ تُرْ اَڪْ ݧَارْ يُ ندَوِ گࣵتّ بِ؞
‏15 ڪِيْ وَخْ اَڪْ مَنْ نَگْ، مُ نگِ يࣷرْ يࣹتُ نَتُّڪَايُ وُرُسْ، نگِرْ نَتّ دࣴڪّ بِ اَڪِ بُنْتَمْ اَڪْ تَتَامْ؞ ‏16 دࣴڪّ بِ دَفَ ڪَارࣹ؛ گُدَّايْ بِ تࣷلّࣷوڪْ يَاتُوَايْ بِ؞ مُ نَتّࣹ دࣴڪّ بِ بَنتّ بِ مُ يࣷرْ، گُدَّايْ بِ اَڪْ يَاتُوَايْ بِ اَڪْ تَخَوَايْ بِ، لࣺݒّ تࣷلّࣷو، دِ فُڪِّ جُنّࣹيڪْ ݧَارِ دَيࣷ (12000)، يࣹمࣷوڪْ ݧَارِ جُنّࣹيڪْ ݧَارْ تࣺيمࣺيرِ ڪِلࣷمࣹيتَرْ (2200)؞ ‏17 مَلاَڪَ مِ نَتّ تَتَ جِ، مُيْ تࣺيمࣺيرِ خَسَبْ اَڪْ ݧࣹينتّ فُڪّ اَڪْ ݧࣹينتّ (144) يُ مَلاَڪَ مِ نَتّࣹ، نِ ڪࣷ نِتْ ݧِيْ نَتّࣹ نَڪَ جࣹڪّ، تࣹ يࣹمࣷوڪْ جُرٗومْ ݧَارْ فُڪّ اَڪْ ݧَارِ مࣹيتَرْ (72)؞ ‏18 تَتَ جِ ݒࣹرُ جࣵسْݒࣴ لَݧُ ڪࣷ لِگّࣺييࣹ، دࣴڪّ بِ دِ وُرُسُ نگَلَمْ، لࣹيرْ نࣵݧّ نِ ڪِرِسْتَالْ؞ ‏19 دࣷجُ ڪࣹنُگْ دࣴڪّ بِ، خࣹيتِ گࣵنجَرْ جُ نࣹڪّ لَݧُ ڪࣷ رَفࣹتَلࣹ؞
دࣷجُ ڪࣹنُ گُ جࣴڪّ گِ دِ جࣵسْݒࣴ؛
ݧَارࣹيلْ بِ دِ سَفِيرْ؛
ݧࣹتّࣹيلْ بِ دِ ڪَلْسࣹدُوَنْ؛
ݧࣹينْتࣹيلْ بِ دِ اࣹمࣴرٗدّ؛
‏20 جُرٗومࣹيلْ بِ دِ سَرْدُوَنْ؛
جُرٗومْ بࣹنّࣹيلْ بِ دِ ڪࣷرْنَلِنْ؛
جُرٗومْ ݧَارࣹيلْ بِ دِ ڪِرْسࣷلِتْ؛
جُرٗومْ ݧࣹتّࣹيلْ بِ دِ بࣹرِلْ؛
جُرٗومْ ݧࣹينْتࣹيلْ بِ دِ تࣷݒَاسْ؛
فُڪّࣹيلْ بِ دِ ڪِرْسࣷݒَرَسْ؛
فُڪّࣹيلْ بࣹيڪْ بࣹنّ دِ يَسࣹنتّ؛
فُڪّࣹيلْ بࣹيڪْ ݧَارْ دِ اَمࣹتِسْتࣴ؞
‏21 فُڪِّ بُنتّ يࣹيڪْ ݧَارْ، فُڪِّ ݒࣹرْ اَڪْ ݧَارْ لَݧُ؞ بُنتّ بُ نࣹڪّ سࣵڪّࣷو ݖِ بࣹنّ ݒࣹرْ ڪࣹݒّ، ݒࣺنݖُمْ دࣴڪّ بِ دِ وُرُسُ نگَلَمْ، لࣹيرْ نࣵݧّ نِ ڪِرِسْتَالْ؞
‏22 اَبْ جَامࣷوڪَايْ نَگْ، گِسُمَ ڪࣷ فࣹنّ ݖِ بِيرْ دࣴڪّ بِ، ندَخْ بࣷرࣷومْ بِ يࣵلَّ اَجِ مَنْ جِ اَڪْ گࣵتّ بِ ݧࣷويْ جَامࣷوڪَايْ بِ؞ ‏23 تࣹ دࣴڪّ بِ سࣷخْلَوُلْ جَنتّ بُ ڪࣷيْ لࣹيرَلْ مبَا وࣹيرْ، ندَخْ لࣹيرُ يࣵلاَّ ڪࣷيْ لࣹيرَلْ، گࣵتّ بِ دِ لࣵمْݒَمْ؞ ‏24 اَسْڪَنِ اࣵدِّنَ، لࣹيرُ دࣴڪّ بِ لَݧُيْ دࣷخࣹ، بُورِ اࣵدِّنَ دِ فِ اِنْدِ سࣹينْ تࣹرَانگَ؞ ‏25 بُنْتِ دࣴڪّ بِ دُ تࣴجࣹ مُڪّ بࣴݖّࣴگْ، تࣹ فࣷوفَ سَخْ گُدِّ اَمُ فَ؞ ‏26 دࣹيسْ نَ فَ اِنْدِ دَرَجَاڪْ تࣹرَانگَيْ اَسْڪَنْ يِ؞ ‏27 دُ لࣹنّ لُ سࣷبࣹوُ لُ فَيْ دُگّ نَگْ، مبَا دࣹفْڪَتُ جࣴفْ جُ رُسْلُ اَڪْ نجُبْلَݝْ؞ ݧِ ݧُ بِندّ ݖِ تࣺيرࣹبْ دُندّ بُ گࣵتّ بِ دࣷݝّ ݧࣷو فَيْ دُگّ؞
