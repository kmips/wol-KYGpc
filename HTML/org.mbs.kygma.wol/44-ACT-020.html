ݒٗولْ جَارࣹ نَ مَسࣹدُوَنْ اَڪْ گࣹرࣹسْ
20
‏1 بَ يࣴنگُ-يࣴنگُ بَ جࣹيخࣹي، ݒٗولْ دَا وࣷو تَالِبࣹ يَ، ݧَاخْ لࣹينْ، بَ نࣷݒِّ تࣵگّتࣷو اَڪْ ݧࣷومْ، جٗگࣹ فَ، دࣹلُّ دِيوَانُ مَسࣹدُوَنْ؞ ‏2 بَ لࣷولُ اَمࣹي مُ وࣴرْ گࣷخْ يࣷويُ، ݧَاخࣹ گࣴمْڪَتْ ݧَ ڪࣵدُّ يُ تَڪُّ، دࣷورَ دࣹمْ رࣺيوُمْ گࣹرࣹسْ؞ ‏3 مُ تࣷوگْ فَ ݧࣹتِّ وࣹيرْ؞ گَنَّاوْ گِ مُيْ وَاجَ دُگّ گَالْ جࣴمْ سِرِ، فَرْ ݒࣹخࣹ مُ ڪࣷ يَوُتْ يَ لَلَلْ فࣹيݧْ، مُ فࣷمّ، نَرَ دࣹلُّ جَارࣹجِ مَسࣹدُوَنْ؞ ‏4 ݧَ اࣵنْدࣷونْ اَڪْ مࣷومْ ݧࣷو دِ سࣷݒَتࣹرْ دࣷومُ ݒِرُسْ مَ ݖࣷسَانࣷو بࣹرࣹ، اَڪْ اَرِسْتࣵرڪّ اَڪْ سࣹگࣷندّ، وَا تࣹسَلࣷنِگْ يَ، اَڪْ گَيُسْ، مَ دࣴڪّࣹ دࣹرْبࣹ، اَڪْ تِمࣷتࣹ، اَڪْ وَا دِيوَانُ اَسِ يَ، تِسِگْ اَڪْ تࣷرࣷفِمْ؞ ‏5 ݧࣷوݧُ ݧࣷو جِيتُ، دِ نُ نࣺگْ ݖَ دࣴڪّ بَ ݧُيْ وَخْ تࣷرࣷوَسْ؞ ‏6 نُنْ نَگْ بَ مࣵگَّلُ يَوُتْ، گَ ݧُيْ وَخْ مبُرُ مُ اَمُلْ لَوِيرْ وࣹيسࣹي لَنُ دُگّࣹ گَالْ فَ دࣴڪّ بَ ݧُيْ وَخْ فِلِݒِ، اَمْ جُرٗومِ فَنْ ݖَ يࣷونْ وَ، دࣷورْ لࣹينَ فࣹڪّسِ ݖَ تࣷرࣷوَسْ، تࣷوگْ فَ جُرٗومْ ݧَارِ فَنْ؞
ݒٗولْ دࣹڪَّلْ نَ نِتْ ݖَ تࣷرࣷوَسْ
‏7 ݖَ ندࣷورْتࣹلْ اَيُ بࣺسْ بَ، نَڪَ لَنُ دَجࣹ نگِرْ دَگّ مبُرُمْ بࣷڪّࣷو مِ¤، ݒٗولْ، يࣺڪَّتِ ڪࣵدُّ، دِ وَخْ اَڪْ گࣴمْڪَتْ ݧِ، ندَخْ فࣹڪّ نَ مُ بࣴگّࣷونَ دࣹمْ ݖَ اࣴلّࣴگْ سَ؞ وَخْ جَ نَگْ لَوْ بَ گُدِّ گَ خَاجْ؞ ‏8 اَيْ لࣵمݒّ يُ بَرࣹي نگَ وࣷونْ ݖَ نࣺيگُ ڪَوْ تَاخْ مَ ݧُ دَجࣹ وࣷونْ؞ ‏9 ݖِ بِيرْ لࣷولُ اَبْ خَلࣹ بُ گٗورْ بُ ݧُيْ وَخْ اࣴتِڪُسْ مَ نگَ تࣷوگࣹ ڪࣺمࣺجُ ݒَلَنْتࣹيرْ بَ؞ نگࣴمّࣹنتّ دَبْ ڪࣷ، تࣹ ڪࣵدُّگْ ݒٗولْ گَ يࣵگّ لࣷولْ؞ اَيْ نࣹلَوْ نَگْ جࣵݒّ وَا جِ، مُ خࣵوِّڪࣷو ݖَ ݧࣹتّࣹيلُ تَاخْ مَ، دَانُ، ݧُ يࣺڪَّتِ ڪࣷ، فࣹڪّ مُ دࣹي؞
‏10 بَ لࣷولُ اَمࣹي ݒٗولْ وࣵݖّ، تِيمْ ڪࣷ، دَلْدِ ڪࣷيْ تࣺيࣹ ݖِ دِگَّنْتࣹ ݧَارِ لࣷخࣷومْ، نࣹ لࣹينْ:‏ «بُلࣹينْ جَاخْلࣹ، مُ نگِ دُندّ؞» ‏11 مُ يࣺيگَاتْ، دَگّ مبُرُ مَ، لࣹڪّ؞ وَخَاتْ نَ لُ يࣵگّ، بَ نجࣴلْ جࣷتْ، مُ سࣷوگَ دࣹمْ؞ ‏12 خَلࣹلُ گٗورْ بَ مࣷومْ، نِتْ ݧَا ڪࣷ يٗبُّ، مُيْ دُندّ، سࣹينْ خࣹلْ دࣷورَ دَلْ بُ بَاخْ؞
ݒٗولْ تࣵگّتࣷو نَاڪْ گࣴمْڪَتِ اࣹفࣹسْ
‏13 نُنْ نَگْ نُ جِيتُجِ، دُگّ گَالْ جࣴمْ اَسࣷسْ، فَ نُ نَرَ يࣹبࣹ ݒٗولْ، نَ مُ ڪࣷ مࣺبࣺتࣹ وࣷونْ، ندَخْ ݒٗولْ ݖِ بࣷݒَّمْ مࣷو خَلاَتࣷونَ جَارࣹ ݖِ يࣷونُ سُوفْ سِ بَ فࣹڪّسِ نُ فَ؞ ‏14 بَ مُ نُ فࣹڪّسࣹي ݖَ اَسࣷسْ، نُ يࣹبْ ڪࣷ، دَلْدِ دࣹمْ مِتِلࣹنْ؞ ‏15 نُ جٗگࣹ فَ، تࣴمبّ بَ اَگْسِ ݖَ اࣴلّࣴگْ سَ فَ جَنࣷوڪْ دُنُ ڪِيࣷسْ، بࣺسْ بَ ݖَ تࣷݒّ، نُ اࣵگّ سَمࣷسْ، بࣺسْ بَ ݖَ تࣷݒَّاتْ، نُ اࣵگّ مِلࣹ؞ ‏16 ندَخْ ݒٗولْ دَا فَسࣷونْ يࣺينࣹيْ تࣹگِّ اࣹفࣹسْ، نگِرْ بَݧَ يࣵگّ دِيوَانُ اَسِ؛ بࣷوبَ دَيْ گَاوْتُ، نگِرْ مࣵگَّلُ ݒࣵنْتَڪࣷتْ مَنْ ڪࣷو فࣹڪّ يࣹرُسَلࣹمْ؞
‏17 فَ مِلࣹ لَ ݒٗولْ يٗنّࣹي ݖَ اࣹفࣹسْ، وࣷولُ مَگِ گࣴمْڪَتْ ݧِ؞ ‏18 بَ ݧُ دِڪّࣹي، مُ نࣹ لࣹينْ:‏
«يࣹينْ ݖِ سࣹينْ بࣷݒّ خَمْ نگࣹينْ بُ بَاخْ، ݖَ بࣺسْ بَ مَ جࣴڪّࣹي تࣹگْ سَمَ تࣵنڪّ اَسِ، بَ تࣹيْ جِي، نِ مَ مَسَ دࣷخَلࣹ اَڪْ يࣹينْ، ‏19 دِ جَامࣷو بࣷرࣷومْ بِ اَگْ وࣷيࣷفْ گُ مَتْ سࣴڪّ، ݖِ بِيرِ رࣷنگࣷوݧْ اَڪْ نَتُّ يُ مَ ݒࣹخࣹيْ يَوُتْ يِ تࣹگْ؞ ‏20 تࣹ اَمُلْ لࣹنّ لُ لࣹينْ دِ جَرِݧْ، لُ مَ لࣹينْ نࣴبّ، بَݧْ لࣹينْ ڪࣷو خَمَلْ؛ جࣵنگَلْ نَا لࣹينْ ݖِ مبࣹدُّمْ بُورْ اَڪْ ݖِ بِيرْ ڪࣴرْ يِ؞ ‏21 دࣺنڪّ نَا يَوُتْ يِ اَڪْ گࣹرࣹگْ يِ نࣹ لࣹينْ، ݧُ تُوبْ ݖِ يࣵلَّ تࣹ گࣴمْ سُنُ سَنگّ عِيسَىٰ؞
‏22 «لࣺيگِ مَا نگِي ݖِ ݖُرْگَگْ نࣷو گُ سࣹلّ گِ، دِ دࣹمْ يࣹرُسَلࣹمْ، لُ مَ فَيْ دَلْ سَخْ، خَوْمَ ݖِ دَرَ، ‏23 لُ مࣷيْ لِ مَ نࣷو گُ سࣹلّ گِيْ سࣹيرࣹيلْ ݖِ دࣴڪّࣷو دࣴڪّ، نࣹ مَ اَيْ جࣺنگّ اَڪِ ݖࣷونࣷ مࣷو مَيْ نࣺگْ؞ ‏24 وَايࣹ سَمَ بَڪَّنْ سࣷخَلُ مَ، خَنَا مَ بࣹيْ سَمَوْ سَسْ، بَ سࣷتَّلْ لِگّࣺييْ، بِ مَ سَنگّ عِيسَىٰ دࣺنڪّ، تࣹ مࣷويْ سࣹيدࣹيلْ خِبَارُ جࣵمّ بُ يِوُ يࣵلَّ وِ؞
‏25 «لࣺيگِ نَگْ خَمْ نَا نࣹ مبࣷولࣹمْ يࣹينْ ݧِ مَ دُگّࣷونْ ݖِ سࣹينْ بِيرْ، دِ يࣺگْلࣹ نگُورُگْ يࣵلَّ، دࣷوتُلࣹينْ مَ گِسْ؞ ‏26 مࣷو مَ تَخْ دِ سࣹيرࣹ بࣺسُبْ تࣹيْ، نࣹ وࣵݖّࣷو نَاڪْ ݧࣺݒّ، دࣹيسُ مَ تࣷݒّ بࣹنّ بَڪَّنْ؞ ‏27 ندَخْ مبࣷولࣹمْ لِ يࣵلَّ نَرْ بَݧُمَ لࣹينْ ݖࣹي خَمَلْ لࣹنّ؞ ‏28 وَتُّلࣹينْ نَگْ سࣹينْ بࣷݒّ، وَتُّ مبࣷولࣹمْ گࣺتّ، گِ لࣹينْ نࣷو گُ سࣹلّ گِ سࣵمّلࣷو، تࣹ نگࣹينْ فࣷرَلْ مبࣷولࣷومْ گࣴمْڪَتِ يࣵلَّ، مِ يࣵلَّ جࣷتࣹ دࣹرࣹتُ بࣷݒَّمْ؞ ‏29 مَنْ نَگْ خَمْ نَا نࣹ گَنَّاوْ بُ مَ دࣹمࣹي، اَيْ بُڪِّ يُ ݧࣵنگّ دِنَݧُ تࣵبِّ ݖِ سࣹينْ بِيرْ تࣹ دُݧُ ݧࣺيبْلُ گࣺتّ گِ؞ ‏30 ݖِ سࣹينْ بِيرْ سَخْ لَ اَوْ نِتْ دِ جٗگࣹ تَخَوْ، دِ وَخْ ندࣴنگّتࣹ، نگِرْ خِيرْتَلْ تَالِبࣹ يِ، بَ ݧُ تࣷݒّ لࣹينْ؞ ‏31 تࣹيوْلُلࣹينْ نَگْ تࣹ فࣵتّلِڪُ نࣹ دِيرُ ݧࣹتِّ اَتْ، گُدِّ اَڪْ بࣴݖّࣴگْ، مَسُمَا نࣷݒّࣹي اࣵرْتُ ڪࣹنّ ڪُ نࣹڪّ ݖِ يࣹينْ، بَيْ جࣷويْ سَخْ؞
‏32 «لࣺيگِ نَگْ، دࣺنڪّ نَا لࣹينْ يࣵلَّ اَڪْ ڪࣵدُّگْ يِوَمْ، گِ لࣹينْ مَنَ دࣷولࣹيلْ، دࣹنݖَلْ لࣹينْ سࣹينْ مُورْ ݖِ بِيرْ مبࣷولࣹمْ نِتْ ݧِ ݧُ سࣹلَّلْ؞ ‏33 دُ لࣹنّ لُ مَ خࣹمّࣹمْ ݖِ ڪࣹنّ؛ دُ خَالِسْ، دُ وُرُسْ دُ ڪࣷدَّايْ؞ ‏34 يࣹينْ ݖِ سࣹينْ بࣷݒّ، خَمْ نگࣹينْ نࣹ، مُيْ سَمَ سࣷخْلَيْ بࣷݒّ، دِ يࣷيْ ݧِ اࣵندّ اَڪْ مَنْ، سَمَ لࣷخࣷ يِي لاَ لِگّࣺييࣹ لُ مَ ڪࣷ فَجࣹ؞ ‏35 نُ نࣹڪّ لاَ لࣹينْ وࣷنࣹ نࣹ سُنُ وَرْتࣺيفْ مࣷويْ نُ لِگّࣺييࣹ نࣷونُ بُ بَاخْ، نگِرْ دِمْبَلِ نࣺيوْ-دࣷولࣹ يِ، تࣹيْ فࣵتّلِڪُ ڪࣵدُّگْ سَنگّ عِيسَىٰ، گَ مُ نࣷونْ:‏ ‹نَنگُ، جࣷخࣹي ڪࣷ اࣴݒّ بَرْڪࣹ؞›»
‏36 نَڪَ لَ وَخْ لࣷولُ، دَلْدِ سُڪَّنْدࣷو اَڪْ ݧࣷومْ ݧࣺݒّ، ݧَانْ؞ ‏37 بَ لࣷولُ اَمࣹي ݧࣺݒّ جࣷويࣷو جࣷويْ يُ مࣹتِّ، لَخَسُ ڪࣷ، فٗونْ ڪࣷ؞ ‏38 لَ لࣹينْ يࣷڪّ اَوْ تِيسْ نَگْ مࣷويْ ڪࣵدُّ گَ مُ لࣹينْ وَخْ، نࣹ لࣹينْ:‏ «دࣷوتُلࣹينْ مَ گِسْ؞» بَ مُ ڪࣷ دࣹفࣹي، ݧُ گُنگࣹ ڪࣷ بَ ݖَ گَالْ گَ؞
