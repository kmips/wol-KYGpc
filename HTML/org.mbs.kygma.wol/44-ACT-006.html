تَبّ نَݧُ تَخَوُڪَتِ مبࣷولࣷومْ گࣴمْڪَتْ ݧِ
6
‏1 جَنتّ يࣷويُ نَگْ، نَڪَ تَالِبࣹ يِ دِ گࣴنَ بَرࣹ، يَوُتْ ݧِيْ لࣵڪّ گࣹرࣹگْ تࣵمْبَلࣹي ݧَخْتُ ݖِ ڪَوْ يَوُتْ ݧِيْ وَخْ سࣹينْ لࣵمِّݧُ بࣷݒّ ندَخْ نࣺيوَلْ گُ ݧُ گِسْ ݧُ نࣺيوَلْ ڪࣷ سࣹينِ جࣴتُنْ ݖَ نَ ݧُيْ سࣺدَّلࣹي اَگْ ندِمْبَلْ بࣺسْ بُ نࣹڪّ؞ ‏2 بَ مُ ڪࣷ دࣹفࣹي فُڪِّ ندَوْ يَ اَڪْ ݧَارْ وࣷو مبࣷولࣷومْ تَالِبࣹ يَ، نࣹ لࣹينْ:‏ «نُ سࣵگَّنࣹ ڪࣵدُّگْ يࣵلَّ، دِ تࣷݒّتࣷو مبِرُمْ سࣺدَّلࣹ دَالْ، جࣹڪُّلْ ݖِ نُنْ؞ ‏3 ڪࣷنْ نَگْ بࣷڪّ يِ، سࣹيتْلࣹينْ ݖِ سࣹينْ بِيرْ جُرٗومْ ݧَارْ ݧُ رَفࣹتْ سࣹيدࣹ، سࣷلࣷو نࣷو گُ سࣹلّ گِ تࣹ رَفࣹتْ خࣹلْ، نُ دࣺنڪّ لࣹينْ سَسْ وࣷووُ؞ ‏4 نُنْ نَگْ نَنُ وࣺيࣹ جُلِّ گِ، اَڪْ لِگّࣺييْ بِ ݖِ ڪࣵدُّ گِ؞»
‏5 وَخْ جࣷوجُ دَلْدِ نࣹيخْ مبࣷولࣷو مࣺݒّ؞ ݧُ سࣹݒِّ اࣹݖّࣹنْ، نِتْ ڪُفࣹيسࣹ نگࣴمْ اَڪْ نࣷو گُ سࣹلّ گِ، سࣹݒِّ فِلِبْ اَڪْ ݒࣷرࣷڪࣷرْ اَڪْ نِڪَنࣷرْ اَڪْ تِمࣷنْ اَڪْ ݒَرْمࣹنَسْ اَڪْ نِڪࣷلَسْ، مَ دࣴڪّࣹ اࣵنݖࣷسْ تࣹ دِبْ تُوبࣹينْ بُ دُگّࣷونْ لُ جِيتُ ݖِ يࣷونُ يَوُتْ؞ ‏6 ݧُ اِنْدِ لࣹينْ فَ ڪَنَمْ ندَوْ يَ، ݧُ ݧَانَلْ لࣹينْ، تࣹگْ لࣹينْ لࣷخࣷ؞
‏7 بَ لࣷولُ اَمࣹي ڪࣵدُّگْ يࣵلَّ دِ لَوْ، لِمُ تَالِبࣹ يِ يࣷڪُّ بَ بَرࣹ لࣷولْ ݖِ بِيرْ يࣹرُسَلࣹمْ، سَرْخَلْڪَتْ يُ بَرࣹ دَلْدِ تࣷݒّ يࣷونُ نگࣴمْ وِ؞
تࣹگْ نَݧُ اࣹݖّࣹنْ لࣷخࣷ
‏8 ݖِ ڪَوْ لࣷولُ اࣹݖّࣹنْ، مِ يِوُ يࣵلَّ دࣹفْ اَگْ لࣹيرْ فࣹيسْ ڪࣷ، دِ دࣹفْ اَيْ ڪࣺيمَانْ اَڪْ فِرْندࣹ يُ مَگْ ݖِ بِيرْ اَسْڪَنْ وِ؞ ‏9 بَ لࣷولُ اَمࣹي ݧࣹنّ ݖَ جࣵنگُ بَ ݧُيْ وَخْ جࣵنگُبْ وَا گࣷرࣹيلْ گَ، دَلْدِ جٗگْ، دِ تࣵمْبَلࣹي وࣹرَنْتࣹيڪْ اࣹݖّࣹنْ؞ اَيْ نِتِ وَا سِرࣹنَ نگَ ݖَ، اَڪْ وَا اَلࣹگْسࣵنْدِرِ، اَڪْ وَا دِيوَانِ سِلِسِ اَڪْ اَسِ؞ ‏10 تࣹيوُلْ اࣵتَّنُݧُ خࣹلَمْ مُ رَفࣹتْ، اَڪْ نࣷو گِ مُيْ وَخࣹ؞
‏11 ݧُ دَلْدِ گࣹرْ اَيْ نِتْ نگِرْ ݧُ دُورَلْ ڪࣷ، نࣹ:‏ «نࣷو ڪࣷ دࣺگّ، مُ يࣺڪَّتِ ڪࣵدُّيْ سَاگَ، جࣴمࣹ ݖِ مُوسَىٰ اَڪْ ݖِ يࣵلَّ؞» ‏12 بَ لࣷولُ اَمࣹي ݧُ خَبْتَلْ نِتْ ݧِ اَڪْ نجِيتْ يِ، اَڪْ فِرِڪَتِ يࣷونْ يِ، ݧُ نࣹ جَلَݧْ، جࣵݒّ اࣹݖّࣹنْ، يٗبُّ ݖَ ڪُرࣺلُ اࣵتّࣹڪَتْ يَ؞ ‏13 ݧُ اِنْدࣹيتِ فࣷوفَ اَيْ سࣹيدࣹيْ نَافࣺقْ يُ نࣹ:‏ «ڪِي دُ بࣵيّࣹي وَخْ مُڪّ لُ جُونࣷو اَڪْ بࣺرَبْ بُ سࣹلّ بِي اَڪْ يࣷونُ مُوسَىٰ؞ ‏14 نࣷو ڪࣷ دࣺگّ مُ نࣹ عِيسَىٰ نَسَرࣹينْ بࣷوبُ دِنَ مࣵبّ بࣺرَبْ بِي تࣹ دِنَ سࣷݒِّ آدَ يِ نُ مُوسَىٰ بࣵيّࣹي؞»
‏15 مبࣷولࣹمْ ݧَ ݖَ جَتَايُ اࣵتّࣹڪَتْ يَ نَگْ نࣹ جࣵڪّ اࣹݖّࣹنْ؛ ݧُ گِسْ خَرْ-ڪَنَمْ بَ نࣹ رࣵݧّ نِ خَرْ-ڪَنَمُ مَلاَڪَ؞
