لُ جࣴمْ ݖِ سࣺيُ جِگࣺينُ إِسْرَاءِيلْ، اَڪْ ݖࣺرُ سُوفَمْ
36
‏1 بَ لࣷولُ وࣺيࣹي ڪِلِفَيْ نجَبࣷوتْ يِ اَسْڪَنࣷو ݖِ يُوسُفَ، تࣹ بࣷڪّ ݖِ لࣵنگُ گَلࣵدّ دࣷومُ مَڪِرْ مِ سࣴتࣷو ݖِ مَنَسࣹ، ݧࣷو دِڪّ؞ ݧُ وَخْ فَ ڪَنَمْ مُوسَىٰ، اَڪْ ڪِلِفَيْ نجَبࣷوتِ بࣵنِّ إِسْرَاءِيلْ؞ ‏2 ݧُ نࣹ:‏ «سَنگّ بِ، خَمْ نگَ اَجِ سَخْ جِ مࣷو دِگْلࣹ وࣷونْ ݧُ تࣹگَلْ بَنتّ بࣵنِّ إِسْرَاءِيلْ، سࣺدَّلࣹ لࣹينْ رࣺيوْ مِ؞ ݖِ بِيرْ لࣷولُ، سَنگّ بِ، ندِگَلْ ݧࣹيلْ لَ، تُڪّࣹي ݖِ اَجِ سَخْ جِ، نگِرْ نگَ جࣴلْ سُنُ ݖࣺرُ دࣷومُ بَايْ سࣹلࣷفَتْ، جࣷخْ ڪࣷ اَيْ دࣷومَمْ يُ جِگࣺينْ؞ ‏3 وَايࣹ بُ ݧُ سࣺيࣹي ݖِ يࣹنࣹينْ گِيرِ بࣵنِّ إِسْرَاءِيلْ دࣹ، سࣹينْ ݖࣺرُ سُوفْ مࣷويْ وࣵݧِّڪࣷو ݖِ سُنُ ݖࣺرُ مَامْ، دࣷلِّڪُ ݖَ ݖࣺرُ گِيرْ گَ ݧُ سࣺيِ ݖَ سࣹينْ بِيرْ، تࣹ ڪࣷنْ ݖࣺرْ بَ نُ تࣹگࣷو وࣷونْ بَنتّ مࣷو ݖَيْ وࣵݧِّڪࣷو؞ ‏4 تࣹ اِتْ بُ اَتُمْ يِوِڪُ¤ مُ بࣵنِّ إِسْرَاءِيلْ اَگْسࣹي، سࣹينْ ݖࣺرْ دَيْ دࣷلِّڪُ ݖَ ݖࣺرُ گِيرْ گَ ݧُ سࣺيِ ݖَ سࣹينْ بِيرْ؞ تࣹ ڪࣷنْ اِتْ ݖِ سُنُ ݖࣺرُ گِيرُ مَامْ، لَ سࣹينْ ݖࣺرْ بࣷوبُ دِ وࣵݧِّڪࣷو؞»
‏5 بَ لࣷولُ وࣺيࣹي، مُوسَىٰ جࣷخْ بࣵنِّ إِسْرَاءِيلْ ندِگَلْ، ݖِ ندِگَلَلْ اَجِ سَخْ جِ، نࣹ:‏ «دࣴگّ لَ گِيرُگْ يُوسُفࣹينْ ݧِ وَخْ؞ ‏6 ڪࣵدُّ گِي نَگْ مࣷويْ لِ اَجِ سَخْ جِ سَنْتَانࣹ ݖِ دࣷومِ سࣹلࣷفَتْ يُ جِگࣺينْ يࣷويُ:‏ ڪُ لࣹينْ نࣹيخْ، نَݧُ سࣺيْ اَڪْ مࣷومْ، نَ فࣹڪّ رࣹڪّ، ڪَ ݧُيْ سࣺيَلْ اَسْڪَنࣷو ݖِ لࣵنگّ گُ بࣷڪّ ݖِ سࣹينْ گِيرُگْ بَايْ؞ ‏7 دُ بࣹنّ ݖࣺرُ سُوفْ ݖِ بِيرْ بࣵنِّ إِسْرَاءِيلْ، بُيْ جٗگࣹ ݖِ گࣹنّ گِيرْ، تࣷخُ ݖَ گࣹنࣹينْ، وَايࣹ نَ ڪُ نࣹڪّ ݖِ بࣵنِّ إِسْرَاءِيلْ يࣹمْ ڪࣹݒّ ݖَ سُوفَسْ گِيرَمْ، گَ مُ دࣷنّࣹ ݖَيْ مَامَمْ؞ ‏8 مبࣷولࣹمْ ڪُ جِگࣺينْ ڪُ دࣷنّ سُوفْ ݖِ گِيرِ بࣵنِّ إِسْرَاءِيلْ، ݖِ بِيرْ لࣵنگّ گُ بࣷڪّ ݖِ گِيرُگْ مَامَمْ رࣹڪّ لَيْ سࣺيْ، نگِرْ ڪُ نࣹڪّ ݖِ بࣵنِّ إِسْرَاءِيلْ مَنَ ݧࣷݝَلْ ݖࣺرُ مَامَمْ؞ ‏9 سُ ڪࣷ دࣹفࣹي دُ بࣹنّ ݖࣺرُ سُوفْ بُيْ جٗگࣹ ݖِ گࣹنّ گِيرْ، تࣷخُ ݖَ گࣹنࣹينْ، وَايࣹ گِيرُگْ بࣵنِّ إِسْرَاءِيلْ گُ ݖِ نࣹڪّ دَيْ يࣹمْ ڪࣹݒّ ݖَ ݖࣺرُ سُوفَمْ؞»
‏10 نَ ڪࣷ اَجِ سَخْ جِ سَنْتࣹ مُوسَىٰ نَگْ، نَ لَ ڪࣷ دࣷومِ سࣹلࣷفَتْ يُ جِگࣺينْ دࣹفࣹ؞ ‏11 مُيْ مَالَ، دِ تِرْݖَ، دِ اࣷگْلَ، دِ مِلْڪَ، دِ نࣷوَ، دࣷومِ سࣹلࣷفَتْ يࣷويُ، ݧࣷومْ ݧࣺݒّ سࣹينْ بࣷڪِّ بَايْ لَݧُ سࣺيَلْ؞ ‏12 ݖَ لࣵنگِ اَسْڪَنُ مَنَسࣹ دࣷومُ يُوسُفَ لَݧُ سࣺيْ، سࣹينْ ݖࣺرُ سُوفْ دَلْدِ دࣹسْ ݖَ گِيرْ گَ سࣹينْ لࣵنگُ بَايْ بࣷڪّ؞
‏13 لࣷولُ مࣷويْ سَنْتَانࣹ يࣹيڪْ اࣵتّࣹيْ يࣷونْ يِ اَجِ سَخْ جِ دࣺنڪࣷونْ بࣵنِّ إِسْرَاءِيلْ، مُوسَىٰ جࣷتّلِ لࣹينْ ڪࣷ ݖَ جࣷورِ مࣷوَبْ، فَ فࣹگّࣷوڪْ دࣹخُ يُرْدَنْ، تࣹ جَنࣷوڪْ يࣹرِڪࣷ؞
