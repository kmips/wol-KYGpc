ݧَارࣹيلُ بَتَاخَلْ بِ يࣵلَّ مَيْ ݒٗولْ، مُ بِندّ ڪࣷ وَا
ڪࣷرࣹنتّ
1
‏1 ݖِ مَنْ ݒٗولْ، ندَوَلْ اَلْمَسِ عِيسَىٰ ݖِ ݖࣷوبَرࣹيْ يࣵلَّ، ݖِ مَنْ لَ بَتَاخَلْ بِي بَوࣷو، بَوࣷو اِتَمْ ݖِ تِمࣷتࣹ مبࣷڪّ مِ، ݧࣹيلْ مبࣷولࣷومْ يࣵلَّ مِ ݖِ ڪࣷرࣹنتّ، اَڪْ مبࣷولࣹمْ ݧُ سࣹلّ ݧِ ݖِ دِيوَانُ اَڪَيِ گࣺݒّ؞ ‏2 اَوْ يِوْ اَڪْ جࣵمّ ݧࣹيلْ نَ لࣹينْ، بَوࣷو فَ يࣵلَّ سُنُ بَايْ، اَڪْ سَنگّ عِيسَىٰ اَلْمَسِ؞
يࣵلَّ مࣷويْ دࣴفَلْ خࣷلْ يِ
‏3 ݖَنتّ ݧࣹيلْ نَ يࣵلَّ سُنُ بَايُ سَنگّ عِيسَىٰ اَلْمَسِ، بَايُ يࣴرْمَنْدࣹ يِ، يࣵلَّ جِيْ دࣴفَلࣹ نُ نࣹڪّ، ‏4 مࣷومْ مِ نُيْ دࣴفَلْ ݖِ سُنُ جࣺݒّ نجࣵقَرࣹ، نگِرْ نࣷونُ نُ جࣷتࣹ نُنْ ݖِ سُنُ بࣷݒّ، ندࣴفَلاَيْ لُ نُ دࣴفَلࣷو فَ يࣵلَّ، نُ مَنْ ڪࣷو دࣴفَلࣹ ݧِ نࣹڪّࣹ مبࣷولࣹمْ خࣹيتِ نجࣵقَرࣹ؞ ‏5 ندَخْ ڪَتْ ݖࣷونࣷيْ اَلْمَسِ، نُ مُ گࣴنَ يࣷڪّࣷو ݖِ سُنُ ڪَوْ، ندَخْ لِ نُ ڪࣷ گࣴمْ، نࣷونُ اِتَمْ لَ ندࣴفَلاَيْ لِيْ بَوࣷو ݖِ اَلْمَسِ دِ يࣷڪّࣷو؞ ‏6 سُ نُ جَارࣹي ݖِ نجࣵقَرࣹ، سࣹينْ ندࣴفَلاَيْ اَڪْ سࣹينْ جࣵمّ لَيْ دࣷونْ؞ سُنُ اَمࣹي ندࣴفَلاَيْ اِتْ، سࣹينْ ندࣴفَلاَيْ لَيْ دࣷونْ، نگِرْ ݖࣷونࣷ يِ نُيْ دࣺڪُّ، نگࣹينْ مَنْ ڪࣷو دࣺڪُّ، يࣹينْ اِتَمْ؞ ‏7 تࣹ اِتَمْ اَمْ نَنُ يَاڪَارْ ݖِ يࣹينْ بُ بَاخْ؞ خَمْ نَنُ نࣹ نِ نگࣹينْ بࣷڪّࣹ اَڪْ نُنْ ݖࣷونࣷ، نِ نگࣹينْ دِ بࣷڪّࣹ اَڪْ نُنْ ندࣴفَلاَيْ؞
‏8 ندَخْ ڪَتْ، بࣷڪّ يِ، بُگُّنُ نگࣹينْ اُمݒّلࣹ نجࣵقَرࣹ، جِ نُ دَجْ ݖِ دِيوَانُ اَسِ؞ دَجْ نَنُ ݖࣷونࣷ بُ جࣺگِّ دَيࣷ، تࣹ وࣹيسْ سُنُ ڪࣵتَّنْ، بَ يَاڪَارَتُنُ وࣷونَ دُندّ سَخْ؞ ‏9 نُنْ ݖِ سُنُ وࣵلُّ بࣷݒّ، جࣵݒّࣷونْ نَنُ نࣹ سُنُ اࣵتّࣹبْ دࣹي مࣷو تَخَوْ؞ وَايࣹ لࣷولُ تَخْ نَ نُ بَݧَتࣹي يَاڪَارْ سُنُ بࣷݒّ، خَنَا دِ يَاڪَارْ ݖِ يࣵلَّ، مِيْ دࣹڪَّلْ ݧِ دࣹي؞ ‏10 مࣷو نُ وَلُّ ݖِ گࣷوگُ دࣹي گُ نَرࣷونَ مࣹتِّ، مࣷو نُيْ وَلُّوَاتْ، تࣹ مࣷومْ لَنُ يَاڪَارْ مُ وَلّࣷوتِ نُ؞ ‏11 يࣹينْ اِتَمْ، نِ نگࣹينْ نُ ݖِيْ جࣵݒّلࣹي مࣷويْ نگࣹينْ دِ نُ ݧَانَلْ؞ سُ ڪࣷ دࣹفࣹي يِوْ، وَ نُ ݧُ بَرࣹ ݧَانَلْ بَ نُ اَمْ ڪࣷ، دِنَ تَخْ ݧُ بَرࣹ بِرَلْ اَگْ ݖَنتّ ݖِ سُنُ مبِرْ؞
ݒٗولْ بِرَلْ نَ جࣴفِنَمْ اَڪْ مࣺبࣺتَمْ
‏12 لِ نُيْ دَمࣷو نَگْ مࣷو دِ سࣹيدࣹ بِ نُ سُنُ خࣹلْ مُ دَلْ سࣹيدࣹيلْ، نࣹ خࣷلْ بُ سࣹتْ اَڪْ جِڪّࣷ جُ دࣴگُّ جُ بَوࣷو فَ يࣵلَّ، مࣷويْ لِ نُ دࣷخَلࣹ ݖِ اࣵدِّنَ سِ، رَوَتِنَ سُنُ دِگّ اَڪْ يࣹينْ، تࣹ دُ لُ نُ سُنُ خَمْ-خَمُ نِتُ نࣹينْ مَيْ، وَايࣹ يِوُ يࣵلَّ مࣷو نُ ڪࣷ مَيْ؞ ‏13 بِنْدُنُ لࣹينْ لࣹنࣹينْ لُ مࣷيْ مبِندّ مࣷومُ نگࣹينْ جࣵنگّ تࣹ مَنْ ڪࣷو خَمْ؞ مَيْ يَاڪَارْ نَگْ نࣹ دِنگࣹينْ خَمْ بَ مࣷڪَّلْ، ‏14 نࣷونُ نگࣹينْ خَوَ خَمࣹ نࣹ نُنْ، مَنْ نگࣹينْ نࣷو دَمࣷو، تࣹ نُنْ اِتْ، يࣹينْ لَ نُيْ دَمࣷو، ڪࣹرࣷوگْ بࣺسُ سَنگّ عِيسَىٰ؞
‏15 ڪٗولُتࣹ گُ مَ اَمࣷونْ ݖِ نِ نگࣹينْ نُيْ دَمࣷو مࣷو تَخْ مَ نَرࣷونَ جࣴڪَّ دِڪّ بَ ݖِ يࣹينْ، نگِرْ سࣹينْ يِوْ مَنَ يَارْ، ‏16 بَ بُ مَ جٗگࣹي ݖِ يࣹينْ، دَلْدِ جَارࣹجِ دِيوَانُ مَسࣹدُوَنْ، بُ مَ جٗگࣹي مَسࣹدُوَنْ، دِڪَّاتْ ݖِ يࣹينْ، بَ نگࣹينْ يِوِ مَ، مَ جࣴمْ يُدࣹ؞ ‏17 ندَخْ ݧࣵڪّ ݒَسْتࣺيفْ لاَ مࣺبࣺتࣹ مࣷومُ مࣺبࣺتْ؟ اَمْ نِ نِتُ سُوخُ نࣹينْ دِ مࣺبࣺتࣹ رࣹڪّ لاَ تࣷݒّ بَيْ خࣹلْ-يَارْ، نَانْ «وَاوْ» تࣹ نَانْ «دࣺيتْ»؟
‏18 يࣵلَّ مِ مَتَ گࣴمْ دࣹ سࣹيرࣹ نَ نࣹ ڪࣵدُّ گِ نُ لࣹينْ دِ وَخْ دُ «وَاوْ، دࣺيتْ،» ‏19 ندَخْ دࣷومُ يࣵلَّ، عِيسَىٰ اَلْمَسِ، مِ نُ بِرَلْ ݖِ سࣹينْ بِيرْ، مَنْ اَڪْ سِلَسْ¤ اَڪْ تِمࣷتࣹ، دُ ڪࣵدُّگْ «وَاوْ، دࣺيتْ؛» «وَاوْ» دࣷݝَّ نࣹڪّ ݖِ مࣷومْ؞ ‏20 اَڪْ لِ دِگࣹيْ يࣵلَّ دِ بَرࣹ لࣺݒّ، ݖِ اَلْمَسِ لَ اَيْ دِگࣹيمْ دِ دࣷونࣹ «وَاوْ»؞ مࣷو تَخْ اِتَمْ، ݖِ مࣷومْ لَنُ نَانْ يࣵلَّ:‏ «اَمِينْ»، نگِرْ دࣹلّࣷو ڪࣷ نجُڪَّلْ؞ ‏21 تࣹ ڪِ نُيْ سَخَلْ ݖِ اَلْمَسِ، نࣷوڪْ يࣹينْ، مࣷو نُ تَبّࣹ دِوُ ݒَلْ گِ، تࣹ مࣷو دِ يࣵلَّ؞ ‏22 مࣷو نُ مࣵنْدَرْگَالࣹ خࣵمِّڪَايَمْ، تࣹ مࣷو يࣹبْ دَوَلُبْ نࣷو گُ سࣹلّ گِ ݖِ سُنُ بِيرْ خࣷلْ؞
‏23 مَنْ نَگْ سࣹيدࣹيْ يࣵلَّ لاَ ݖِيْ سࣵڪُّ، گِݧْ نَا ڪࣷ ݖِ سَمَ بَڪَّنْ، بَݧْ لࣹينَ يࣷڪُّ نَقَرْ مࣷو تَخْ دࣺلْسِوُمَ ڪࣷرࣹنتّ؞ ‏24 دُ ݖَاگࣹينُ يِلِفْ سࣹينْ نگࣴمْ نَگْ، ندَخْ ݖِ نگࣴمْ نگࣹينْ سَخْ، وَايࣹ نࣷوڪْ يࣹينْ اَيْ لِگّࣺييْ، نگِرْ سࣹينْ مبࣺگْتࣹ؞
