ݧࣹتِّ يࣷونْ لࣹيسْ دِ مࣵگَّلْسِ ݖِ بࣺرَبْ بُ سࣹلّ بِ
16
‏1 وࣷرْمَالْ-لࣹينْ وࣹيرُ اَبِيبْ، تࣹ نگࣹينْ اَمَلْ ݖِ مࣵگَّلُ بࣺسُبْ مُݖّ، ݧࣹيلْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ، ندَخْ ݖِ وࣹيرُ اَبِيبْ لَ لࣹينْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ گࣺنّࣹي مِسْرَ، اَگْ گُدِّ؞ ‏2 نَنگࣹينْ ݖِ سَرْخَلَلْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ، اَبْ گࣵتّ مبَا اَوْ نَگْ، ݖِ بࣺرَبْ بِ اَجِ سَخْ جِ تَامࣷو دࣴعࣴلْ تُرَمْ اࣴلّࣴگْ؞ ‏3 بُلࣹينْ ڪࣷ لࣹڪَّالࣹ لࣹنّ لُ اَمْ لَوِيرْ؞ دَلࣹلࣹينْ ݖَا لࣹڪّ، بَ دِيرُ جُرٗومْ ݧَارِ فَنْ، مبُرُ مُ اَمُلْ لَوِيرْ، مبُرُمْ نجࣵقَرࣹ، ندَخْ ݖِ بِيرْ گَاوْتُ نگࣹينْ گࣺنّࣹ وࣷونْ رࣺيوُمْ مِسْرَ؞ سُ ڪࣷ دࣹفࣹي نگࣹينْ بَاخَنْتَلْ سࣹينْ گِيرُ دُندّ، بࣺسْ بَ نگࣹينْ گࣺنّࣹي رࣺيوُمْ مِسْرَ؞ ‏4 بُ لࣹينْ ڪࣹنّ گِسَلْ لَوِيرْ ݖِ سࣹينْ تُندّ وࣺݒّ، دِيرُ جُرٗومْ ݧَارِ فَنْ، تࣹ بُ لࣹنّ فَنَانْ ݖِ يࣵݒُّ جُرْ گِ نگࣹينْ سَرْخَلْ، دِگَّنْتࣹ نگࣷونُ بࣺسْ بُ جࣴڪّ بَ، بَ ݖَ سُبَ سَ؞
‏5 جُرُ بࣺسُبْ مُݖّ بِ، دُنگࣹينْ ڪࣷ مَنَ رࣹنْدِ فࣹنّ ݖِ دࣴڪّ يِ لࣹينْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ جࣷخْ، ‏6 فُ مࣷيْ بࣺرَبْ بَ سࣹينْ يࣵلَّ اَجِ سَخْ جِ تَامࣷو دࣴعࣴلْ تُرَمْ؞ فࣷوفَ نگࣹينْ دِ رࣹنْدِ جُرُ بࣺسُبْ مُݖّ بِ ݖِ نگࣷونْ بُ جَنتّ بِيْ سࣷ، يࣹمࣷوڪْ وَخْتُ وَ نگࣹينْ بࣵيِّڪࣷو وࣷونْ مِسْرَ؞ ‏7 تࣷگّلࣹينْ ڪࣷ، تࣹ لࣹڪّ ڪࣷ ݖِ بࣺرَبْ بِ سࣹينْ يࣵلَّ اَجِ سَخْ جِ دِ تَامُجِ، ݖَ سُبَ سَ نگࣹينْ خࣴيْ، دࣹلُّ ݖَ سࣹينِ خَيْمَ؞ ‏8 جُرٗومْ بࣹنِّ فَنْ نگࣹينْ دِ لࣹڪّ مبُرُ مُ اَمُلْ لَوِيرْ، بࣺسُبْ جُرٗومْ ݧَارࣹيلْ دِ اَمْ ندَجࣹ، ݧࣹيلْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ؞ بُلࣹينْ ݖِ لِگّࣺييْ دَرَ؞
‏9 جُرٗومْ ݧَارِ اَيِ بࣺسْ نَگْ، نگࣹينْ دِ وَݧّ، تࣹ ݖِ ندࣷورْتࣹلْ نگٗوبْ نگࣹينْ دِ تࣵمْبَلࣹي وَݧّ جُرٗومْ ݧَارِ اَيِ بࣺسْ يِ، ‏10 نگࣹينْ دࣷورَ اَمَلْ مࣵگَّلُگْ اَيِ بࣺسْ، ݧࣹيلْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ، تࣹ سَرَخُ يࣺينࣹ بَ نگࣹينْ ݖَيْ جࣷخࣹ دِ ڪࣹمُ ڪࣷومْ گَ لࣹينْ سࣹينْ يࣵلَّ بَرْڪࣹيلࣹ؞ ‏11 نگࣹينْ دَلْدِ بࣵنّࣹيخُ فَ سࣹينْ ڪَنَمْ يࣵلَّ اَجِ سَخْ جِ، يࣹينْ اَڪْ سࣹينِ دࣷومْ، گٗورْ اَڪْ جِگࣺينْ اَڪْ سࣹينِ جَامْ، گٗورْ اَڪْ جِگࣺينْ، اَڪْ لࣹوࣹينْ ݧِ ݖِ سࣹينْ دࣴڪّ، اَڪْ دࣷخَنْدࣺيمْ يࣹيڪْ جِرِمْ يࣹيڪْ جࣴتُنْ ݧِ ݖِ سࣹينْ بِيرْ، ݖِ بࣺرَبْ بِ سࣹينْ يࣵلَّ اَجِ سَخْ جِ تَامࣷو دࣴعࣴلْ تُرَمْ اࣴلّࣴگْ؞ ‏12 فࣵتّلِڪُلࣹينْ نࣹ اَيْ جَامْ نگࣹينْ وࣷونْ ݖَ مِسْرَ تࣹ نگࣹينْ سࣵمّࣷو جࣴفࣹ دࣷگَلِ يࣷونْ يِي؞
‏13 مࣵگَّلُ مبَارْ يِ اِتْ نَنگࣹينْ ڪࣷ وࣷرْمَالْ دِيرُ جُرٗومْ ݧَارِ فَنْ، گَنَّاوْ بُ نگࣹينْ دَجَلࣹي سࣹينُمْ ݒࣹݒّ، اَڪْ سࣹينْ ندࣷخُمْ رࣹسࣹݧْ؞ ‏14 نگࣹينْ دَلْدِ بࣵنّࣹيخࣷو سࣹينُگْ مࣵگَّلْ، يࣹينْ اَڪْ سࣹينِ دࣷومْ، گٗورْ اَڪْ جِگࣺينْ، اَڪْ سࣹينِ جَامْ، گٗورْ اَڪْ جِگࣺينْ، اَڪْ لࣹوࣹينْ ݧࣹيڪْ دࣷخَنْدࣺيمْ يࣹيڪْ جِرِمْ يࣹيڪْ جࣴتُنْ، ݧِ ݖِ سࣹينْ بِيرْ دࣴڪّ؞ ‏15 جُرٗومْ ݧَارِ فَنْ نگࣹينْ ڪࣷيْ مࣵگَّلْ ݧࣹيلْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ ݖِ بࣺرَبْ بِ اَجِ سَخْ جِ دِ تَامُجِ، ندَخْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ مࣷو لࣹينْ دِ بَرْڪࣹيلَلْ سࣹينْ مࣹݧَّتْ مࣺݒّ، اَڪْ سࣹينْ مبࣷولࣹمْ ݧَقْ، سࣹينْ مبࣺگْتࣹ مَتْ سࣴڪّ؞
‏16 ڪࣷنْ ݧࣹتِّ يࣷونْ ݖِمْ اَتْ لَ مبࣷولࣹمْ سࣹينِ گٗورْ دِ تࣹيوْ فِ سࣹينْ ڪَنَمْ يࣵلَّ اَجِ سَخْ جِ، ݖِ بࣺرَبْ بِ مُيْ تَامُجِ:‏ فَنِ مࣵگَّلُ مبُرُ مُ اَمُلْ لَوِيرْ، اَڪْ فَنِ مࣵگَّلُگْ اَيِ بࣺسْ، اَڪْ فَنِ مࣵگَّلُ مبَارْ يِ؞ وَايࣹ دࣹيسُلْ تࣹيوࣹ لࣷخࣷيْ نࣹينْ فِ ڪَنَمْ اَجِ سَخْ جِ؞ ‏17 ڪُ نࣹڪّ، نَ سَرَخْ بَ مُيْ اِنْدِ، دِ ڪࣹمُ بَرْڪࣹ بَ ڪࣷ يࣵلَّ جَگْلࣹيلْ؞
يࣷونْ وِ سࣵرْتَلْ نَ مبِرُمْ اࣵتّࣹ
‏18 تࣹ اِتَمْ اَيْ اࣵتّࣹڪَتْ اَڪِ ڪِلِفَ نگࣹينْ دِ تَبّ نگِرْ سࣹينِ گِيرْ، ݖِ مبࣷولࣹمْ دࣴڪّ يِ لࣹينْ سࣹينْ يࣵلَّ اَجِ سَخْ جِيْ جࣷخْ، ݧُيْ اࣵتّࣹ مبࣷولࣷو مِ اࣵتّࣹبْ دࣴگّ؞ ‏19 بُلࣹينْ جَلْگَتِ يࣷونْ ݖِبْ اࣵتّࣹ، بُلࣹينْ جࣹنگّ تࣹ بُلࣹينْ نَنگُ اَلَلُ گࣹرْ، ندَخْ اَبْ گࣹرْ دَيْ مُورْ بࣴتِ بࣷرࣷومْ خࣹلْ، دِ خَݧْ اَجِ جُبْ دࣴگّ؞ ‏20 دࣴگّ دࣷݝّ نگࣹينْ وَرَ تࣷݒّ، ندَخْ نگࣹينْ دُندّ، جَگࣷو رࣺيوْ مِ لࣹينْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ دِ جࣷخْ؞
‏21 بُلࣹينْ سَمݒّ بࣹنّ خࣹرُ بَنتّ، بُ جٗگࣹ ݖِ گࣹنّ گَرَبْ، نگِرْ دِ ڪࣷ جَامࣷو اَسࣹرَ تُورْ مِ، ݖِ سࣹينْ وࣹتُ سَرْخَلُڪَايْ بِ نگࣹينْ سࣵڪّ نگِرْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ؞ ‏22 تࣹ اِتَمْ بُلࣹينْ يࣺڪَّتِ لࣹنّ لُ بࣷڪّ ݖِ دࣷجِ جَامࣷوڪَايْ يࣷويُ سࣹينْ يࣵلَّ اَجِ سَخْ جِ بَݧْ؞
