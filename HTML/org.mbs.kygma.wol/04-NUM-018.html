لُ جࣴمْ ݖِ سَرْخَلْڪَتْ يِ اَڪْ لࣹوࣹينْ ݧِ
18
‏1 اَجِ سَخْ جِ نࣹ هَـٰرُونَ:‏ «يَوْ، يَاڪْ سَيْ دࣷومْ اَڪْ سَ وَا ڪࣴرْ مَامْ، يࣹينَيْ گࣵدُّ ݒࣹيُگْ تࣷوݧْ ݖِ وࣵلُّ بࣺرَبْ بُ سࣹلّ بِ، تࣹ اِتَمْ يَوْ، يَاڪْ سَيْ دࣷومْ، يࣹينَيْ گࣵدُّ ݒࣹيُگْ تࣷوݧْ ݖِ سࣹينْ وࣵلُّ ݖَرْخَلْ؞ ‏2 تࣹ اِتَمْ سَ بࣷڪِّ لࣹوࣹينْ ݧِ، سَ گِيرُگْ بَايْ، جࣹگࣹݧَلْ لࣹينْ سَ بࣷݒّ، ݧُ جࣷڪُّ ݖِ يَوْ، دِ لَ جࣵݒّلࣹ، تࣹ يَوْ يَاڪْ سَيْ دࣷومْ، نگࣹينْ فࣺيتࣹ فِ ڪَنَمْ خَيْمَ بَ سࣹيدࣹسْ ڪٗلّࣴرࣹ گَ دࣹنݖࣹ؞ ‏3 ݧࣷويْ دࣺنڪࣷو سَ ندࣺنڪَانࣹ، اَڪْ دࣺنڪَانࣹيْ مبࣷولࣹمْ خَيْمَ بِ، وَايࣹ جُمْتُڪَايِ بࣺرَبْ بُ سࣹلّ بِ، اَڪْ سَرْخَلُڪَايْ بِ، بُݧُ ڪࣷ جࣹگࣹ، لُ ڪࣷ مࣷيْ دَݧُيْ دࣹي، ݧࣷومْ اَڪْ يࣹينْ اِتَمْ؞ ‏4 نَݧُ جࣷڪُّ ݖِ يَوْ، تࣹ ݧُ دࣺنڪࣷو ندࣺنڪَانࣹلْ خَيْمَبْ ندَجࣹ مِ، ݖِ مبࣷولࣹمْ لِگّࣺييُ خَيْمَ بِ؞ ڪࣹنّ ڪُ ݖِ بࣷڪُّلْ نَگْ دُ لࣹينْ جࣹگࣹ، يࣹينْ؞ ‏5 نَنگࣹينْ دࣺنڪࣷو ندࣺنڪَانࣹلْ بࣺرَبْ بُ سࣹلّ بِ، اَڪْ ندࣺنڪَانࣹلْ سَرْخَلُڪَايْ بِ، نگِرْ اَمْ سࣵنجّ بَݧَ دَلَتِ ݖِ ڪَوْ بࣵنِّ إِسْرَاءِيلْ؞
‏6 «مَنْ مِي نَگْ مَا سࣹݒّࣹي سࣹينْ بࣷڪِّ لࣹوࣹينْ ݧِ ݖِ بِيرْ بࣵنِّ إِسْرَاءِيلْ؞ يࣹينْ لَݧُ لࣹينْ جَگْلࣹيلَ جَگْلࣹيلْ، ݧࣹيلْ اَجِ سَخْ جِ، نگِرْ ݧُ لِگّࣺييْ لِگّࣺييُ خَيْمَبْ ندَجࣹ مِ؞ ‏7 يَوْ نَگْ يَاڪْ سَيْ دࣷومْ، دࣺنڪࣷولࣹينْ سࣹينُگْ ݖَرْخَلْ ݖِ مبࣷولࣹمْ مبِرُ سَرْخَلُڪَايْ بِ، اَڪْ ݖَ وࣵلاَّ رِدࣷبْ بِيرْ بِ، نگࣹينْ دِ ݖَ لِگّࣺييْ؞ سࣹينْ لِگّࣺييُ ݖَرْخَلْ بِ، اَبْ سَسُ تࣹرَانگَ لَ، مَا لࣹينْ ڪࣷ تࣹرَلࣹ؞ وَايࣹ ڪُ بࣷڪُّلْ ݖِ يࣹينْ، بُ ݖِ لاَلࣹي، دࣹيسْ ڪࣷيْ رࣹيْ؞»
لُ جࣴمْ ݖِ ݒࣹيُ سَرْخَلْڪَتْ يِ
‏8 اَجِ سَخْ جِ نࣹيتِ هَـٰرُونَ:‏ «مَنْ مِي مَا لَ سࣺدّ ندࣺنڪَانࣹبْ سَمَيْ جࣷوخࣹ؛ مبࣷولࣹمْ جࣷوخࣹ يُ سࣹلّ يِ بࣵنِّ إِسْرَاءِيلْ اِنْدِ، يَوْ لاَ ڪࣷ جَگْلࣹيلْ، نگَ جَگࣷو، يَاڪْ سَيْ دࣷومْ، ݖِ ڪَوْ دࣷگَلْ بُ سَخْ دࣵڪّ؞ ‏9 لِي نَگْ، يَا ڪࣷ مࣷومْ، ݖِ مبࣷولࣹمْ سَرَخْ يُ سࣹلّ يࣹي سࣹلّ، يِ مُݖّ سَوَرَ:‏ سࣹينْ بࣺݒّ سَرَخْ، سࣹينْ بࣺݒّ سَرَخُ ݒࣹݒّ اَڪْ سࣹينْ بࣺݒّ سَرَخُ ݒٗوتُمْ بࣵڪَّارْ، اَڪْ سࣹينْ بࣺݒّ سَرَخُ ݒࣹيُگْ تࣷوݧْ بُ ݧُ مَ اِنْدِلْ، لُ سࣹلَّ سࣹلّ لَ، ݧࣹيلْ لَ، يَاڪْ سَ دࣷومْ يُ گٗورْ؞ ‏10 فُ سࣹلَّ سࣹلّ نگࣹينْ ڪࣷيْ لࣹڪّࣹ؛ ڪࣺݒّ ڪُيْ گٗورْ ݖِ يࣹينْ، سَݧْ نَ ݖࣹي لࣹڪّ؞ لُ سࣹلّ لَيْ دࣷونْ ݖِ يَوْ؞
‏11 «لࣹنࣹينْ لُ لَ ݧࣹيلْ مࣷو دِ جࣷوخࣹ بُ جٗگࣹ ݖِ سَرَخِ نِتْ ݧِ، دِ بࣺݒّ سَرَخُ يࣺڪَّتِ-جࣺبَّلْ بُ بࣵنِّ إِسْرَاءِيلْ؞ يَوْ لاَ ڪࣷ سࣺدّ، يَاڪْ سَ دࣷومْ يُ گٗورْ، اَڪْ سَ دࣷومْ يُ جِگࣺينْ، ݖِ ڪَوْ دࣷگَلُ يࣷونْ بُ سَخْ دࣵڪّ؞ ڪࣺݒّ ڪُ سࣹتْ ݖِ سَ وَا ڪࣴرْ، سَݧْ نَ ݖࣹي لࣹڪّ؞
‏12 «لࣺݒّ لُيْ نگࣴنࣹيلُ دِوْ گُ بࣹيسْ، اَڪْ نگࣴنࣹيلُ بِيݧْ بُ بࣹيسْ، اَڪْ ݒࣹݒّ مُ جࣴڪَّ ݧࣷرْ، مُ ݧُ جࣷخْ اَجِ سَخْ جِ، يَوْ لاَ ڪࣷ جࣷخْ؞ ‏13 نجࣵلْبࣺينُ مࣹݧّࣹيفْ مِ ݧُيْ اِنْدِلْ اَجِ سَخْ جِ، ݖِ مبࣷولࣹمْ لُ مࣹݧّࣹي ݖِ سࣹينْ سُوفْ، يَا ڪࣷيْ مࣷومْ، تࣹ ڪࣺݒّ ڪُ سࣹتْ ݖِ سَ وَا ڪࣴرْ، سَݧْ نَ ݖࣹي لࣹڪّ؞ ‏14 مبࣷولࣹمْ لࣹيسْ آيَلَلْ اَجِ سَخْ جِ ݖِ إِسْرَاءِيلْ، يَا ڪࣷيْ مࣷومْ؞ ‏15 لࣺݒّ لُيْ اُبِّتࣹيْ نجُرْ بُ ݧُ جَگْلࣹيلْ اَجِ سَخْ جِ، ݖِ لࣺݒّ لُ بِنْدࣷو اَوْ سُوخْ، نِتْ اَڪْ مَلَ، يَا ڪࣷيْ مࣷومْ؞ فࣹخࣹيلْ رࣹڪّ بَ جࣷتْلُ تَاوُبْ نِتْ ڪِ، تࣹ تَاوُبْ مَلَ مُ سࣹتُلْ اِتْ، نَنگَ ڪࣷ جࣷتْلُ؞ ‏16 نجࣷتُ خَلࣹ بُ اَمْ لُ دَلࣹ ݖِ وࣹيرْ، اࣵݒّ ڪࣷ ݖِ جُرٗومِ سِيڪَلِ خَالِسْ يُ بࣺرَبْ بُ سࣹلّ بِ، بࣹنّ سِيڪَلْ يࣹمࣷوڪْ ݧَارْ فُڪِّ گࣹرَ مبَا فُڪِّ گَرَامْ؞ ‏17 وَايࣹ لُ نَگْ تَاوْلࣷو، اَڪْ لُ خَرْ تَاوْلࣷو، اَڪْ لُ بࣴيْ تَاوْلࣷو، دࣹيسُ ڪࣷ جࣷتْ؞ يࣷويِي يُ سࣹلّ لَ؞ دࣹرࣹتَمْ نگَيْ خࣴݒّ ݖِ سَرْخَلُڪَايْ بِ، نࣹبّࣷنْ بِ، نگَ دࣹفْ ڪࣷ سَخَرْ، مُيْ سَرَخُ سَوَرَ نگِرْ خࣹتُگْ جࣵمّ، ݧࣹيلْ اَجِ سَخْ جِ؞ ‏18 اَوْ يࣵݒَّمْ، يَا ڪࣷيْ مࣷومْ، نِ نگَ مࣷومࣹ دࣴنُّ سَرَخُ يࣺڪَّتِ-جࣺبَّلْ، اَڪْ تࣵنڪُبْ ڪَنَمْ بُ ندِجࣷورْ بَ؞ ‏19 مبࣷولࣹمْ جࣷوخࣹ يُ سࣹلّ يِ بࣵنِّ إِسْرَاءِيلْ دِ جࣷوخࣹيلْ اَجِ سَخْ جِ، سࣺدّ نَا لَ ڪࣷ، يَاڪْ سَ دࣷومْ يُ گٗورْ، اَڪْ سَ دࣷومْ يُ جِگࣺينْ، ݖِ ڪَوْ دࣷگَلُ يࣷونْ بُ سَخْ دࣵڪّ؞ ڪٗلّࣴرࣹيْ خࣷرࣷمَ¤ نگࣷوگُ، دُ فࣹݖِّڪُ، خَنَا سَخْ دࣵڪّ فِ ڪَنَمْ اَجِ سَخْ جِ، ݧࣹيلْ لَ يَوْ، يَاڪْ سَوْ اَسْڪَنْ؞» ‏20 اَجِ سَخْ جِ نࣹيتِ هَـٰرُونَ:‏ «سࣹينُمْ رࣺيوْ نَگْ دࣷو ݖِ اَمْ ݖࣺرْ، تࣹ وࣵلُّ سُوفْ، دࣷو ڪࣷ اَمْ ݖِ سࣹينْ بِيرْ؞ مَنْ مَايْ سَ وࣵلّ، دِ سَبْ ݖࣺرْ ݖِ بِيرْ بࣵنِّ إِسْرَاءِيلْ؞»
لُ جࣴمْ ݖِ ݒࣹيُگْ لࣹوࣹينْ ݧِ
‏21 لࣹوࣹينْ ݧِ ݧࣷومْ، مَا لࣹينْ جࣷخْ بࣺݒّ ݖࣺرُ فُڪّࣹيلْ بُ گࣺنّࣹ ݖِ إِسْرَاءِيلْ، مُيْ سࣹينْ ݖࣺرْ، ݧُ يࣷولࣷو ڪࣷ سࣹينْ لِگّࣺييْ بِ ݧُيْ لِگّࣺييْ ݖِ خَيْمَبْ ندَجࣹ مِ؞
‏22 بُ بࣵنِّ إِسْرَاءِيلْ جُوخَتِ ݖِ خَيْمَبْ ندَجࣹ مِ، دِ گࣵدُّ بࣵڪَّارْ بُ دࣹي دِ اࣵتّࣹيمْ؞ ‏23 لࣹوࣹينْ ݧِ ݧࣷويْ ݧِيْ لِگّࣺييْ لِگّࣺييِ خَيْمَبْ ندَجࣹ مِ، تࣹ لُ ݧُ ݖَ تࣷوݧْ، ݧࣷويْ گࣵدُّ ݒࣹيْ گَ، ݖِ ڪَوْ دࣷگَلْ بُ سَخْ دࣵڪّ، ݧࣹيلْ لࣹينْ، ݧࣷوڪْ سࣹينْ اَسْڪَنْ؞ وَايࣹ بِيرْ بࣵنِّ إِسْرَاءِيلْ، لࣹوࣹينْ ݧِ دُݧُ ݖِ سࣺدُّ ݖࣺرُ سُوفْ، ‏24 ندَخْ ݖࣺرِ فُڪّࣹيلْ يِ بࣵنِّ إِسْرَاءِيلْ يࣺڪَّتِ، جࣷوخࣹيلْ ڪࣷ اَجِ سَخْ جِ، جࣷخْ نَا ڪࣷ لࣹوࣹينْ ݧِ، مُيْ سࣹينْ ݖࣺرْ؞ مࣷو تَخْ مَ نࣹ لࣹينْ، بِيرْ بࣵنِّ إِسْرَاءِيلْ، دُݧُ ݖِ سࣺدُّ ݖࣺرُ سُوفْ؞
اَبْ جࣷوخࣹ ݧࣹيلْ نَ لࣹوࣹينْ ݧِ
‏25 اَجِ سَخْ وَخَتِ مُوسَىٰ نࣹ:‏ ‏26 «لࣹوࣹينْ ݧِ، وَخْ لࣹينْ، نࣹ لࣹينْ بُ ݧُ جࣴلࣹي ݖِ بࣵنِّ إِسْرَاءِيلْ سࣹينْ ݖࣺرِ فُڪّࣹيلْ يِ مَ لࣹينْ جَگْلࣹيلْ يࣹينْ، مُيْ سࣹينْ ݖࣺرْ، نَݧُ ݖِ جࣷوخࣹيلْ اَجِ سَخْ جِ، فُڪّࣹيلُبْ ݖࣺرُ فُڪّࣹيلْ يࣷويُ؞ ‏27 نَݧُ لࣹينْ جࣵݒَّلْ لࣷولُ، يࣹينْ لࣹوࣹينْ ݧِ، مُيْ سࣹينْ جࣷوخࣹبْ بࣷݒّ، نِ بُ دࣷونْ ݒࣹݒّ مُ جٗگࣹ ݖِ سࣹينْ دࣵگَّيْ بࣷݒّ، اَڪْ بِيݧْ بُ بࣹيسْ بُ جٗگࣹ ݖِ سࣹينْ نَلُڪَايْ؞ ‏28 سُ ڪࣷ دࣹفࣹي نگࣹينْ جࣷوخࣹ يࣹينْ اِتَمْ، جࣷوخࣹبْ اَجِ سَخْ جِ، ݖِ مبࣷولࣹمْ سࣹينْ ݖࣺرِ فُڪّࣹيلْ يِ نگࣹينْ جࣴلࣹ ݖِ بࣵنِّ إِسْرَاءِيلْ؞ ڪࣷنْ نَنگࣹينْ جࣴلࣹ ݖࣷوݖُ، جࣷوخࣹبْ اَجِ سَخْ جِ، جࣷخْ ڪࣷ هَـٰرُونَ سَرْخَلْڪَتْ بِ؞ ‏29 لࣺݒّ لُ ݧُ لࣹينْ سࣺدّ، نَنگࣹينْ ݖِ جࣷوخࣹ جࣷوخࣹبْ اَجِ سَخْ جِ؞ مبࣷولࣹمْ ݖࣺرِ نگࣴنࣹيلْ يࣷويُ، ݖِ نگࣹينْ دِ گࣺنّࣹ ݖࣺرْ بُ سࣹلّ بࣷوبُ؞
‏30 «وَخَلْ لࣹوࣹينْ ݧِ نࣹ لࣹينْ گَنَّاوْ بُ ݧُ ݖَ گࣺنّࣹي ݖࣺرُ نگࣴنࣹيلْ لِ، لِ ݖِ دࣹسْ لࣹوْ نَ لࣹينْ، نِ بُ دࣷونْ ݒࣹݒّ مُ جٗگࣹ ݖِ سࣹينْ دࣵگَّيْ بࣷݒّ، اَڪْ بِيݧْ بُ جٗگࣹ ݖِ سࣹينْ نَلُڪَايْ؞ ‏31 سَݧْ نَݧُ ڪࣷو لࣹڪّࣹ فࣺݒّ، ݧࣷومْ اَڪْ سࣹينْ وَا ڪࣴرْ، ندَخْ لࣷولُ سࣹينْ ݒࣹيْ لَ، ݧُ يࣷولࣷو ڪࣷ سࣹينْ لِگّࣺييُ خَيْمَبْ ندَجࣹ مِ؞ ‏32 دُݧُ ݖِ گࣵدُّ بࣹنّ بࣵڪَّارْ، گَنَّاوْ بُ ݧُ ݖِ جࣷوخࣹي ݖࣺرُ نگࣴنࣹيلْ لِ؞ وَايࣹ جࣷوخࣹ يُ سࣹلّ يُ بࣵنِّ إِسْرَاءِيلْ يࣷويُ، بُݧُ ݖِ تࣹدَّدِلْ لࣹنّ، بَ دِ ݖِ دࣹي؞»
