29
‏1 بَ مُ ڪࣷ دࣹفࣹي مُوسَىٰ وࣷولُ إِسْرَاءِيلْ گࣺݒّ؞ مُ نࣹ لࣹينْ:‏
يࣹينَ گِسْ مبࣷولࣹمْ لَ اَجِ سَخْ جِ دࣹفْ، نگࣹينْ تࣹگْ ݖِ سࣹينْ بࣴتْ، ݖَ رࣺيوُمْ مِسْرَ، ݖِ ڪَوْ فِرْعَوْنَ اَڪْ دَگَمْ ݧࣺݒّ، اَڪْ رࣺيوَمْ مࣺݒّ؞ ‏2 نَتُّ يُ مَگْ يࣷويُ، اَڪْ فِرْندࣹ يࣹيڪْ ڪࣺيمَانْ يُ رࣺيْ يࣷويُ، يࣹينَ ݖِ تࣹگْ سࣹينْ بࣴتْ، ‏3 تࣹ تَخُلْ اَجِ سَخْ جِ مَيْ لࣹينْ خࣹلْ مُ نگࣹينْ رࣵݧّࣹي، اَڪْ گࣴتْ يُ نگࣹينْ گِسࣹ، اَڪْ نࣷݒّ يُ نگࣹينْ دࣺگّࣹ، بࣷوبَ بَ سُنُ جࣷنِّ يࣵلَّيْ تࣹيْ!‏ ‏4 اَجِ سَخْ جِ نࣹي:‏ «مَا لࣹينْ جِيتࣹ ݧࣹينتّ فُڪِّ اَتْ ݖَ مࣵنْدِݝْ مَ، سࣹينِ يࣺرࣹ رࣵݒُّلْ، سࣹينُ دࣵلّ رࣵݒُّلْ؞ ‏5 مبُرُ، لࣹڪُّلࣹينْ ڪࣷ، بِيݧْ مبَا گࣹنࣹينْ نَانْ گُ دِيسْ، نَانُلࣹينْ ڪࣷ؞ مَا لࣹينْ لࣹيلْ، نگِرْ نگࣹينْ خَمْ نࣹ مَنْ مَايْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ؞» ‏6 بَ نگࣹينْ اَگْسࣹي ݖِ بࣺرَبْ بِي، سِوࣷنْ بُورُبْ اࣹسْبࣷنْ دِڪّ، اࣵندّ اَڪْ اࣷگْ بُورُ بَسَنْ، دِڪّ دَجࣹيڪْ نُنْ، نگِرْ سࣷنگّ نُ خَرࣹ، نُ دُمَ لࣹينْ؞ ‏7 نُ نَنگُ سࣹينُمْ رࣺيوْ، سࣺدّ ڪࣷ رُبࣹنࣹينْ ݧِ، اَڪْ گࣵدّࣹينْ ݧِ، اَڪْ گࣹنّ-وࣵلُّ گِيرُگْ مَنَسࣹينْ ݧِ؞ ‏8 سࣵمّلࣹينْ نَگْ سࣵرْتِ ڪٗلّࣴرࣹ گِي، دِ ڪࣷ جࣴفࣹ ندَخْ يࣵلَّ نگࣹينْ بَاخْلࣹ ݖِ مبࣷولࣹمْ لُ نگࣹينْ دࣹفْ؞
بُلࣹينْ ݖَاخَانࣹ ڪٗلّࣴرࣹگْ اَجِ سَخْ جِ
‏9 يࣹينَ نگِي تࣹيْ، تَخَوْ جࣷنّ، يࣹينْ ݧࣺݒّ فِ سࣹينْ ڪَنَمْ يࣵلَّ اَجِ سَخْ جِ، سࣹينْ نجِيتِ گِيرْ، اَڪْ سࣹينِ مَگْ اَڪْ سࣹينِ ڪࣵنگَمْ؛ مبࣷولࣹمْ گٗورِ إِسْرَاءِيلْ، ‏10 اَڪْ سࣹينِ تُوتْ-تࣵنڪّ، اَڪْ سࣹينِ جَبَرْ، اَڪُبْ دࣷخَنْدࣺيمْ ݖِ سࣹينُبْ دَلْ، سࣹينْ گࣷرْڪَتُبْ مَتّ، بَ ݖِ سࣹينُبْ رࣷوتْڪَتْ؞ ‏11 يࣹينَ نگِ تࣵبِّ ݖِ سࣹينْ بِيرْ ڪٗلّࣴرࣹگْ يࣵلَّ اَجِ سَخْ جِ، اَڪْ ݖِ نگِݧَمْ، ڪٗلّࣴرࣹ گِ سࣹينْ يࣵلَّ اَجِ سَخْ جِ فَسࣷو اَڪْ يࣹينْ بࣺسُبْ تࣹيْ، ‏12 نگِرْ دࣹفْ لࣹينْ بࣺسُبْ تࣹيْ، اَسْڪَنْ وِ مُ ݧࣷݧࣷو؛ مࣷومْ ݖِ بࣷݒَّمْ مُ دࣷونْ سࣹينْ يࣵلَّ، نَ مُ لࣹينْ ڪࣷ وَخࣹ وࣷونْ، نَ مُ ڪࣷ گِݧَلࣹ وࣷونْ سࣹينِ مَامْ، إِبْرَاهِيمَ اَڪْ إِسْحَاقَ اَڪْ يَعْقُوبَ؞ ‏13 اَجِ سَخْ جِ نࣹي:‏ «دُ يࣹينْ دࣷݝّ لاَيْ فَسࣷولْ ڪٗلّࣴرࣹ گِي، اَڪْ نگِݧْ لِي؞» ‏14 يࣹينْ ݧِ تَخَوْ ݖِ سُنُ بِيرْ تࣹيْ، فِ سُنُ ڪَنَمْ يࣵلَّ اَجِ سَخْ جِ، ݖِ نگࣹينْ بࣷڪّ، وَايࣹ ݧِ دِڪَّگُلْ بَيْ بࣷڪّ اَڪْ نُنْ تَخَوْ فِي اِتْ، ݖِ لَݧُ؞
‏15 يࣹينْ ݖِ سࣹينْ بࣷݒّ، خَمْ نگࣹينْ نَ نُ نࣹڪّࣹ وࣷونْ رࣺيوُمْ مِسْرَ اَڪْ نَ نُ جَارࣹ ݖَ بِيرْ خࣹيتْ يَ نگࣹينْ جَارࣹ سࣹينِ رࣺيوْ؞ ‏16 گِسْ نگࣹينْ سࣹينْ جࣴفْ جُ جࣷمبّلُ، سࣹينْ ڪَسَرَيْ تُورْ يَ:‏ يࣴفِ بَنتّ يࣹيڪْ يࣴفِ دࣷجْ يࣹيڪْ يࣴفِ خَالِسْ يࣹيڪْ يࣴفِ وُرُسْ يِ ݧُ اࣵنْدَلْ؞ ‏17 ڪࣷنْ نَگْ، مُيْ گٗورْ، دِ جِگࣺينْ، اَڪْ لࣵنگّ گُ مُ مَنَ بࣷڪّ، مبَا گِيرْ، فࣹخࣹلࣹينْ بَ ڪࣹنّ ݖِ يࣹينْ دُ وَلْبَتِڪُ، دࣴدُّ بࣺسُبْ تࣹيْ، سُنُ يࣵلَّ اَجِ سَخْ جِ، دِ جَامُجِ يࣵلَّيْ خࣹيتْ يࣷويُ؞ لُ ڪࣷ مࣷيْ اَبْ رࣹينْ دِنَ فࣴلّࣹ فِ سࣹينْ بِيرْ، مࣹݧَّلْ لࣹينْ تࣷوڪࣹ اَڪْ وࣹخْتَنْ؞ ‏18 جࣷمْبُلْ مُ اَمْ ڪُ دࣺگّ ڪࣵدُّيْ نگِݧْ لِي، تࣹيْ نَخْ بࣷݒَّمْ ݖِمْ خࣹلَمْ، نَانْ:‏ «جࣵمّ رࣹڪّ لاَيْ اَمْ، مَا تࣹيْ تࣴ،» بَ تَخْ ݧَخْ مُ وࣷوْ لَڪَّالࣹ مُ تࣷويْ؞ ‏19 اَجِ سَخْ جِ دُ نَنگࣷو بَالْ ڪࣷوڪُ، ندࣹ سُ بࣷوبَا، سࣵنجُمْ اَجِ سَخْ جࣹييْ اࣵندّ اَڪْ فِيرَانگࣹيمْ، تࣵڪّ ݖِ ڪَوَمْ، تࣹ مبࣷولࣹمْ اَلْڪࣵنْدࣹ گِ ݧُ بِندّ ݖِ تࣺيرࣹ بِي دَيْ ڪࣴݒُّ ݖِ ڪَوَمْ، تࣹ اَجِ سَخْ جࣹييْ فَرْ اَوْ تُرَمْ فُ اَسَمَانْ تِيمْ؞ ‏20 اَجِ سَخْ جࣹي ڪࣷيْ سࣹݒّࣹي ݖِ مبࣷولࣹمْ گِيرِ إِسْرَاءِيلْ، نگِرْ سࣺدّ ڪࣷ نجࣹڪَّرْ، نَ مُ دࣴݒّࣷوڪْ مبࣷولࣹمْ اَلْڪࣵنْدࣹيْ ڪٗلّࣴرࣹ گِ ݧُ بِندّ ݖِ تࣺيرࣹبْ يࣷونْ وِي؞
مبُگَلْ دُ جَاسْ
‏21 مَاسْ گِيْ ݧࣴوْ، سࣹينْ دࣷومْ يِ لࣹينْ دِ وُوتُجِ يࣹينْ، ݧࣷوڪْ دࣷخَنْدࣺيمْ يِ جٗگࣹ فُ سࣷرࣹ، بُ ݧُ گِسࣹي مبُگَلْ يِ ݖِ رࣺيوْ مِي، اَڪْ وࣷݒّ يِ فِ اَجِ سَخْ جِ وࣷݒَّلࣹ، ‏22 تَمَرَخْ اَڪْ خࣷرࣷمْ، اَڪْ لَڪّ گِ ݖِ سࣹينْ رࣺيوْ مࣺݒّ، بَ دُ لࣹنّ لࣹيسْ فَيْ جِ، مبَا لُ فَيْ جࣹبِّ، مبَا گࣹنّ گࣵنݖَخْ گُ فَيْ سَخࣹ، مبࣹتࣹ تَسْتࣹيْ سࣷدࣷمْ، اَڪْ گࣷمࣷرْ اَڪْ اَتْمَ اَڪْ ݖࣹبࣷيِمْ، دࣴڪّ يَ اَجِ سَخْ جِ تَسْ، ݖَ مࣹرَمْ مَ اَڪْ سࣵنجَمْ مَ، ‏23 سُ بࣷوبَا يࣺيفَرْ يࣺݒَّ نَانْ:‏ «اَنَ لُ وَرَلْ اَجِ سَخْ جِ دࣹفْ نِي رࣺيوْ مِ؟ لُ وَرَلْ مِي سࣵنجّ مُ تࣵنگّ تࣹ رࣺيْ؟» ‏24 ݧُ نࣹ لࣹينْ:‏ «لِ ڪࣷ وَرَلْ مࣷويْ، دَݧࣷو فࣹݖِّ ڪٗلّࣴرࣹيْ اَجِ سَخْ جِ، سࣹينْ يࣵلَّيْ مَامْ، تࣹ مُ فَسࣷونْ ڪࣷ اَڪْ ݧࣷومْ بَ مُ لࣹينْ جࣴلࣹي رࣺيوُمْ مِسْرَ؞ ‏25 ݧُ دࣹمْ نَگْ، دِ جَامُ يࣹنࣹينْ يࣵلَّ، دِ لࣹينْ سُجّٗوتَلْ، يࣵلَّ يُ ݧُ خَمُلࣷونْ، تࣹ دُ لُ لࣹينْ اَجِ سَخْ جِ فࣺيتَلࣹ؞ ‏26 مࣷو تَخْ سࣵنجُمْ اَجِ سَخْ جِ تࣵڪَّلْ رࣺيوْ مِي، نگِرْ وࣵݖّࣹيلْ لࣹينْ مبࣷولࣹمْ اَلْڪࣵنْدࣹ يِ ݧُ بِندّ ݖِ تࣺيرࣹ بِي؞ ‏27 اَجِ سَخْ جِ نَگْ مࣷو لࣹينْ دࣺجَّتࣹي سࣹينُمْ رࣺيوْ ݖِ بِيرْ مࣹرْ، اَڪْ خَدَرْ، اَڪْ سࣵنجّ مُ رࣺيْ، بَ سࣵنِّ لࣹينْ مࣹنࣹينْ رࣺيوْ، نِڪِ بࣺسُبْ تࣹيْ جِي؞»
‏28 مبِرْ يِ لࣵقُ دَالْ، ݧࣹيلْ نَ سُنُ يࣵلَّ اَجِ سَخْ جِ، مبِرْ يِ فࣹيݧْ نَگْ، ݧࣹيلْ نُ، نࣷوڪْ سُنُيْ دࣷومْ بَ فࣵوّ، تࣹ نُ وَرَ جࣴفࣹ مبࣷولࣹمْ ڪࣵدُّيْ يࣷونْ وِي؞
