ݒٗولْ جُبْلُ نَ يࣹرُسَلࣹمْ
21
‏1 گَنَّاوْ بَ نُ تࣵگّࣷو اَڪْ ݧࣷومْ، نُ دُگّ گَالْ، جُبَلْ دُنُ ڪࣷسْ، تࣹگْ ݖَ تࣹيرِ ݖَ دُنُ رࣷدّ، جٗگࣹ فَ، جࣵلّ بَ ݒَتَرَ؞ ‏2 نُ گِسْ فَ گَالْ گُ جࣴمْ فࣹنِسِ، دُگّ ݖَ، دَلْدِ تࣹدِّ؞ ‏3 نَڪَ لَنُ دࣹمْ بَ سࣺينْ دُنُ سِݒَّرْ، وَݖّ ڪࣷ نَگْ سُنُ وࣹتُ ݖࣵمّࣷݧْ، جُبْلُ دِيوَانُ سِرِ، بَ تࣹيرِ تِرْ، ندَخْ فَ لَ گَالْ گِ وَرࣷونَ يࣹبِّ اَبْ يࣹبَمْ؞ ‏4 نُ سࣹيتْ تَالِبࣹيْ فࣷوفَ، بَ گِسْ لࣹينْ، تࣷوگْ اَڪْ ݧࣷومْ جُرٗومْ ݧَارِ فَنْ؞ ݖِ بِيرْ لࣷولُ نࣷو گُ سࣹلّ گِ دِ لࣹينْ خِيرْ ݖِ وَخْ ݒٗولْ، نࣹ ڪࣷ بُمُ دࣹمْ يࣹرُسَلࣹمْ؞ ‏5 بَ نُ فَ اَمࣹي فَنْ يࣷويُ لَنُ فَ بࣵيِّڪࣷو، دࣹمْ، ݧُ گُنگࣹ نُ، ݧࣷومْ ݧࣺݒّ، ݧࣷوڪْ سࣹينِ سࣷخْنَ اَڪْ سࣹينِ دࣷومْ؞ نُ دࣹمْ بَ گࣺنّ دࣴڪّ بَ، دَلْدِ سُڪّ ݖَ تࣹفࣹسْ گَ، ݧَانْ؞ ‏6 بَ لࣷولُ اَمࣹي نُ تࣵگّتࣷو بَ نࣷݒِّ، دُگّ ݖَ گَالْ گَ، ݧࣷومْ ݧُ ݧِبِّ؞
‏7 نُنْ نَگْ بَ نُ جٗگࣹي تِرْ، دَنࣷو ݧࣷقْلِ بَ تࣹيرِ ݒَتࣷلࣹمَيِسْ؞ نُ نُيࣷو فَاڪْ بࣷڪِّ تَالِبࣹ يَ، تࣷوگْ فَاڪْ ݧࣷومْ بࣹنّ فَنْ؞ ‏8 ݖَ اࣴلّࣴگْ سَ نُ دࣹمْ بَ اࣵگّ سࣹسَرࣹ؞ نُ دࣹمْ ڪࣴرْ فِلِبْ، خَمْلࣹڪَتُ خِبَارُ جࣵمّ، بِ بࣷڪّ ݖِ جُرٗومْ ݧَارْ ݧِ¤؞ نُ دَلْ اَڪْ مࣷومْ؞ ‏9 فِلِبْ مࣷومُ اَمࣷونْ نَ ݧࣹينْتِ دࣷومِ جَنقّ يُيْ وَخࣹ اَيْ وَخْيُ؞
‏10 گَنَّاوْ بَ نُ فَ تࣷوگࣹي اَيْ فَنْ، اَبْ يࣷنࣹنتّ بُ ݧُيْ وَخْ اَگَبُسْ جٗگࣹ يُدࣹ، اَگْسِ؞ ‏11 مُ دِڪّ بَ ݖِ نُنْ، جࣴلْ نگَݧَايَلْ ݒٗولْ، يࣹيوࣹ ڪࣷ اَيْ تࣵنڪَمْ اَڪِ لࣷخࣷومْ، دَلْدِ نࣹ:‏ «نࣷو گُ سࣹلّ گِ دَفَ وَخْ نࣹ:‏ ‹ڪِ مࣷومْ نگَݧَايْ لِي، نِي لَ ڪࣷ يَوُتْ يِيْ يࣹيوࣹ ݖَ يࣹرُسَلࣹمْ، جࣺبَّلْ ڪࣷ جَامْبُرْ ݧِ دُلْ يَوُتْ؞›»
‏12 بَ نُ دࣺگّࣹي لࣷولُ، نُنْ اَڪْ وَا گࣷخْ بَا تِنُ ݒٗولْ، نࣹ ڪࣷ بُمُ دࣹمْ يࣹرُسَلࣹمْ؞ ‏13 ݖِ ڪَوْ لࣷولُ ݒٗولْ نࣹ:‏ «لُ نگࣹينْ دِ دࣹفْ نِي، دِ جࣹيخَلْ سَمَ خࣷلْ؟ وَخُمَ ݧُ يࣹيوْ مَ رࣹڪّ، وَايࣹ دࣹي سَخْ نَنگُ نَا ڪࣷو دࣹفْ ݖِ يࣹرُسَلࣹمْ نگِرْ سَنگّ عِيسَىٰ؞» ‏14 نَڪَ لَنُ وَخْ، وَخْ، مُ تࣴ، نُ بࣵيِّ، دَلْدِ نࣹ:‏ «ڪࣷنْ نَگْ يَلْ نَ بࣷرࣷومْ بِ سࣷتَّلْ ݖࣷوبَرࣹيمْ!‏»
‏15 گَنَّاوْ فَنْ يࣷويُ لَنُ فَبُ، نگِرْ دࣹمْ يࣹرُسَلࣹمْ؞ ‏16 ݖِ بِيرْ لࣷولُ اَيْ تَالِبࣹ يُ دࣴڪّࣹ سࣹسَرࣹ اࣵندّ اَڪْ نُنْ، يٗبُّ نُ فَ نُ وَرَ دَلِ، ݖَ ڪࣴرْ ڪُ ݧُيْ وَخْ مَنَسࣷنْ، مَ ݖࣷسَانࣷو سِݒَّرْ، تࣹ دِ تَالِبࣹ بُ يࣵگّ؞
ݒٗولْ اَگْسِ نَ يࣹرُسَلࣹمْ
‏17 بَ نُ اَگْسࣹي يࣹرُسَلࣹمْ نَگْ، بࣷڪِّ تَالِبࣹ يَ بࣺگࣹ نَݧُ نࣷو دَلَلْ؞ ‏18 ݖَ اࣴلّࣴگْ سَ ݒٗولْ اࣵندّ اَڪْ نُنْ، سࣹيتِ يَعْقُوبَ، فࣹڪّ مَگْ ݧࣺݒَّ نگَ فَ تࣹيوْ؞ ‏19 ݒٗولْ نُيࣷوڪْ ݧࣷومْ، بَ نࣷݒِّ بࣹنّ-بࣹنَّلَلْ لࣹينْ جࣴفْ، جِ يࣵلَّ سࣷتَّلࣹ لِگّࣺييَمْ ݖِ بِيرْ جَامْبُرْ ݧِ؞
‏20 ݧُ دࣺگّ ݖَ، دَلْدِ سࣵبَّالْ يࣵلَّ، بَ نࣷݒِّ نࣹ ڪࣷ:‏ «مبࣷڪّ مِ، خࣷولَلْ ݧَاتَ جُنِّيْ يَوُتَ دࣷونْ اَيْ گࣴمْڪَتْ، تࣹ ݧࣷومْ ݧࣺݒّ دِ ݧُ خࣺرْ ݖِ يࣷونُ مُوسَىٰ؞ ‏21 ݧࣷو دࣺگّ نَگْ سَبْ جࣴوْ، ݧُ نࣹ دࣴدُّ يࣷونُ مُوسَىٰ مࣷويْ لِ نگَيْ جࣵنگَلْ مبࣷولࣹمْ يَوُتْ يِ دࣴڪّ ݖِ بِيرْ جَامْبُرْ ݧِ، نَانْ لࣹينْ بُݧُ خَرْفَلْ سࣹينِ دࣷومْ، تࣹ بُݧُ تࣷݒّ سُنُيْ آدَ؞ ‏22 لُ ݖِيْ ݒࣹخࣹ نَگْ؟ ندَخْ ڪَتْ دُݧُ اُمݒّلࣹ نࣹ دِڪّ نگَ؞ ‏23 لࣺيگِ دَالْ، دَنگَيْ دࣹفْ لِ نُ لَيْ وَخْ نِي:‏ نُ نگِ فࣹيڪْ ݧࣹينْتِ نِتْ ݧُ اَمْ لُ ݧُ خَسَلْ يࣵلَّ؞ ‏24 ݧࣷومْ نگَيْ اࣵنْدَلْ، نگَ بࣷڪّ اَڪْ ݧࣷومْ سࣹتْلُ، تࣹ نگَ گࣵدُّ سࣹينْ ݒࣹيُ وَتࣹيفُ بࣷݒّ، وِ بࣷرࣷومْ خَسْ وُ نِ مࣹلْ آدَوࣷو؞ سُ ڪࣷ دࣹفࣹي ݧࣺݒّ اَيْ خَمْ نࣹ لِ ݧُ لَيْ جࣴوࣹ، دَرَ اَمُ ݖِ، تࣹ يࣷونُ مُوسَىٰ نگَيْ جࣴفࣹ، دِ ڪࣷ سࣵمّ؞ ‏25 نَڪَ گࣴمْڪَتْ ݧِ دُلْ يَوُتْ نَگْ، بِندّ نَنُ لࣹينْ لِ نُ دࣷگَلْ، نࣹ لࣹينْ ݧُ مࣷيْتُ ݧَمْ وُ ݧُ سَرْخَلْ اَيْ خࣴرࣴمْ، مࣷيْتࣷو لࣹڪّ دࣹرࣹتْ اَڪْ يࣵݒُّ مَلَ مُ ݧُ تُورُلْ دࣹرࣹتَمْ، تࣹ مࣷيْتُ ݒࣷوُمْ سࣺيْ؞»
‏26 ݖَ اࣴلّࣴگْ سَ ݒٗولْ اࣵندّ اَڪْ نِتْ ݧَ، بࣷڪّ اَڪْ ݧࣷومْ سࣹتْلُ، دُگّ ݖَ ڪࣴرْ يࣵلَّ گَ؞ مُ دَلْدِ خَمْلࣹ بࣺسْ بَ سࣹينْ اࣵݒُّ سࣹتْلُ دِ مَتْ، بَ جࣷوخࣹ بَ ݖِ وَرْ گࣺنّ، ݧࣹيلْ ڪࣹنّ ڪُ نࣹڪّ ݖِ ݧࣷومْ؞
جࣵݒّ نَݧُ ݒٗولْ
‏27 بَ اࣵݒُّ جُرٗومْ ݧَارِ فَنْ يَيْ وَاجَ جࣹيخْ، يَوُتْ يَ جٗگࣹ دِيوَانُ اَسِ گِسْ ݒٗولْ ݖَ ڪࣴرْ يࣵلَّ گَ؞ ݧُ يࣴنگَلْ مبࣷولࣷو مࣺݒّ، جࣵݒّ ڪࣷ، ‏28 دِ خَاݖُ نَانْ:‏ «وَلّࣹي، بࣷڪِّ إِسْرَاءِيلْ!‏ وَا جِي، مࣷويْ ڪِيْ وࣴرْ فࣺݒّ، دِ وَارࣹ ݧࣺݒّ ڪࣵدُّ گُ سَفَانࣷوڪْ نجَرِݧَلْ خࣹيتْ وِ، اَڪْ يࣷونُ مُوسَىٰ اَڪْ بࣺرَبْ بِي؞ رَخْ ݖِ دࣷلِّ مُ اِنْدِ ݖِ ڪࣴرْ يࣵلَّ گِ اَيْ گࣹرࣹگْ، بَ سࣷبࣹيلْ بࣺرَبْ بُ سࣹلّ بِي؞» ‏29 بࣷوبَ تࣷرࣷفِمْ مُ اࣹفࣹسْ لَ ݧُ گِسࣷونْ ݖَ دࣴڪّ بَ، مُ اࣵندّ اَڪْ ݒٗولْ، ݧُ يَاڪَارْ نࣹ ݒٗولْ دَ ڪࣷو يٗبُّ ݖَ ڪࣴرْ يࣵلَّ گَ؞ ‏30 بَ لࣷولُ اَمࣹي دࣴڪّ بَ بࣺݒّ مࣵبّ، نِتْ ݧَ نࣹ ڪࣴرْ-ڪࣴرْ، نࣹ ڪࣴݒّ، ݧُ جࣵݒّ ݒٗولْ، گࣺنّࣹ ڪࣷ ڪࣴرْ يࣵلَّ گَ رࣹڪّ، دَلْدِ تࣴجْ بُنتّ يَ؞
‏31 نَڪَ لَݧُ ڪࣷيْ وُتَ رࣹيْ، ݧُ اࣵگّࣹ نجِيتَلْ گࣵنگࣷورُ رࣷومْ بَ يࣷرࣹ ڪِلِفْتࣹيفْ گَ، نࣹ ڪࣷ يࣹرُسَلࣹمْ گࣺݒّ سَلْفَاݧࣷو نَ؞ ‏32 ݖَ سَا سَ مُ اࣵندّ اَڪْ اَيْ دَگْ، اَڪْ نجِيتِ تَڪّ-دࣹرْ، نࣹ جَاسْ ݖَ بِيرْ مبࣷولࣷو مَ؞ يَوُتْ يَ نَگْ گِسْ نجِيتَلْ گࣵنگࣷورْ گَ اَڪْ دَگْ يَ، دَلْدِ دَڪَّلْ يَرْ يَ ݧُ دࣷونْ دٗورْ ݒٗولْ؞
‏33 ݖِ ڪَوْ لࣷولُ نجِيتْ لَ اَگْسِ، جࣵݒّ ڪࣷ، دَلْدِ سَنْتَانࣹ ݧُ جࣺنگࣹ ڪࣷ ݧَارِ ݖࣵلَّلَ؞ بَ لࣷولُ وࣺيࣹي مُ لاَجْتࣹ، نࣹ:‏ «ڪِي ڪُ مُ دࣷونْ، اَڪْ لُ مُ دࣹفْ؟» ‏34 ݧࣹنّ ݖَ مبࣷولࣷو مَ خَاݖُ نࣹ لِي، ݧَ ݖَ دࣹسْ نࣹ لاَ؞ ݧࣷلْݧࣷلْ جَ نَگْ تَخْ مَنُ ݖَا خَمْ دَرَ لُ لࣹيرْ؞ مُ دَلْدِ سَنْتَانࣹ، ݧُ يٗبُّ ڪࣷ ݖَ بِيرْ تَتَ جَ؞ ‏35 بَ ݒٗولْ اࣵگّࣹ ݖَ دࣴگَّسْتَلِ تَتَ جَ، مبࣷولࣷو مَ دَا سࣵنجُ اَڪْ دࣷولࣹ، بَ تَڪّ-دࣹرْ يَ مُجّ فَبْ ݒٗولْ، گࣵدُّ ڪࣷ؞ ‏36 ندَخْ بࣷوبَ مبࣷولࣷو مُ رࣺيَ ڪࣷ تࣷݒّ، دِ خَاݖُ نَا:‏ «نَ دࣹي!‏»
ݒٗولْ لَيࣷو نَ اَڪْ يࣹرُسَلࣹمْ
‏37 نَڪَ لَݧُ ڪࣷيْ وَاجَ دُگَلْ ݖَ تَتَ جَ، ݒٗولْ نࣹ نجِيتُ گࣵنگࣷورْ گَ:‏ «ندَخْ مَنْ نَا لاَ وَخْ بَاتْ؟» مُ نࣹ ڪࣷ:‏ «ندࣹڪࣹ دࣺگّ نگَ گࣹرࣹگْ؟ ‏38 ڪࣷنْ دُ يَايْ وَايُ مِسْرَ جࣷوجُ يࣴنگَلࣷونْ دࣴڪّ بِ ݖِ فَنْ يِي، تࣹ يٗبُّ وࣷونْ ݧࣹينْتِ جُنِّيْ بٗومْڪَتْ يَ (4000) ݖَ مࣵنْدِݝْ مَ؟» ‏39 ݒٗولْ نࣹ ڪࣷ:‏ «مَنْ دࣹ اَبْ يَوُتْ لاَ، دِ اَسْ گࣷرُ تࣵرسّ، دࣴڪّ بُ ݧࣵڪُّلْ تُرْ بَ ݖَ دِيوَانُ سِلِسِ؞ دَمَ لَيْ ݧَانْ، نگَ مَيْ مَ، مَ وَخْ اَڪْ مبࣷولࣷو مِ؞» ‏40 مُ مَيْ ڪࣷ؞ ݒٗولْ تَخَوْ ݖَ دࣴگَّسْتَلْ يَ، تࣵلَّلْ مبࣷولࣷو مَ لࣷخࣷ، ݧُ نࣹ سࣹلَوْ؞ مُ وَخْ اَڪْ ݧࣷومْ ݖِ يَوُتْ، نࣹ:‏
