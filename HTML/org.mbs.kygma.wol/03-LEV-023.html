لُ جࣴمْ ݖِ بࣺسِ مࣵگَّلْ يِ
23
‏1 اَجِ سَخْ جِ دࣹلّࣷوتِ وَخْ مُوسَىٰ نࣹ ڪࣷ:‏ ‏2 «سَمَيْ مࣵگَّلْ، مَنْ اَجِ سَخْ جِ، دَنگࣹينْ ڪࣷو وَرَ سِيوَلْ، مُيْ ندَجࣹ يُ سࣹلّ؞ سَمَيْ مࣵگَّلْ يِي لَ؞
‏3 «جُرٗومْ بࣹنِّ فَنْ لࣹيسْ دِ لِگّࣺييْ، بࣺسُبْ جُرٗومْ ݧَارࣹيلْ بَ بࣺسُبْ نࣷفْلاَيْ دࣷݝّ لَيْ دࣷونْ، دِ بࣺسُبْ ندَجࣹ مُ سࣹلّ؞ دُ لࣹنّ لُ نگࣹينْ ݖِيْ لِگّࣺييْ؞ بࣺسُبْ نࣷفْلاَيْ لَ بُ ݧُيْ جَگْلࣹيلْ اَجِ سَخْ جِ، اَڪْ فࣺݒّ فُ نگࣹينْ دࣴڪّ؞
‏4 «اَجِ سَخْ جِ، دࣹيلࣹينْ يࣺگْلࣹ اَيْ مࣵگَّلَمْ، مُيْ اَيْ ندَجࣹ يُ سࣹلّ يُ نگࣹينْ دِ وࣷرْمَالْ سَا يُ جࣷتࣹي؞ مࣵگَّلْ يِي لَ؞
لُ جࣴمْ ݖِ بࣺسُبْ مُݖّ اَڪْ بࣺسِ مبُرُ مُ اَمُلْ لَوِيرْ
‏5 «فُڪِّ فَنْ اَڪْ ݧࣹينتّ ݖِ وࣹيرْ وِ جࣴڪّ، دِگَّنْتࣹ نگࣷونْ اَڪْ جَنتّ بُ سࣷ، مࣷويْ مࣵگَّلُ بࣺسُبْ مُݖّ بَ، ݧࣹيلْ اَجِ سَخْ جِ؞
‏6 «فُڪِّ فَنْ اَڪْ جُرٗومْ ݖِ وࣹيرْ وࣷووُ لَ اَيُ بࣺسُ مبُرُ مُ اَمُلْ لَوِيرْ دِ تࣵمْبَلِ، ݧࣹيلْ اَجِ سَخْ جِ؞ دِيرُ جُرٗومْ ݧَارِ فَنْ مبُرُ مُ اَمُلْ لَوِيرْ نگࣹينْ وَرَ لࣹڪّ؞ ‏7 بࣺسْ بُ جࣴڪّ بِ ندَجࣹ مُ سࣹلّ نگࣹينْ ݖِيْ اَمَلْ؞ مبࣷولࣹمْ لُ نگࣹينْ فَرَلْ دِ لِگّࣺييْ بُلࣹينْ ڪࣷ ݖِ لِگّࣺييْ؞ ‏8 دࣹفْلࣹينْ ݖِ سَرَخُ سَوَرَ، ݧࣹيلْ اَجِ سَخْ جِ دِيرُ جُرٗومْ ݧَارِ فَنْ؞ بࣺسُبْ جُرٗومْ ݧَارࣹيلْ بَ دِ بࣺسُبْ ندَجࣹ مُ سࣹلّ؞ مبࣷولࣹمْ لُ نگࣹينْ فَرَلْ دِ لِگّࣺييْ بُلࣹينْ ڪࣷ ݖِ لِگّࣺييْ؞»
لُ جࣴمْ ݖِ مࣵگَّلُ ندࣷورْتࣹلْ مࣹݧّࣹيفْ
‏9 سُ لࣷولُ وࣺيࣹي اَجِ سَخْ جِ وَخْ نَ مُوسَىٰ نࣹ ڪࣷ:‏ ‏10 «وَخَلْ بࣵنِّ إِسْرَاءِيلْ نࣹ لࣹينْ:‏ بُ نگࣹينْ دُگّࣹي ݖَ رࣺيوْ مَ مَ لࣹينْ جࣷخْ، بَ دَجَلࣹ لَ نگࣹينْ جࣴلࣹ ݖَ سُوفَمْ، يٗبُّلࣹينْ ݖِ سَرْخَلْڪَتْ بِ تَڪُّ ݒࣹݒُّمْ لࣷرسّ بِ نگࣹينْ جࣴڪَّ گٗوبْ؞ ‏11 سَرْخَلْڪَتْ بِ دَيْ يࣺڪَّتِ تَڪّ بِ، جࣺبَّلْ ڪࣷ اَجِ سَخْ جِ، نگِرْ مُ نَنگُلْ لࣹينْ؞ بࣺسْ بِ تࣷݒّ ݖِ بࣺسُبْ نࣷفْلاَيْ بِ لَيْ يࣺڪَّتِ تَڪّ بِ، جࣺبَّلْ ڪࣷ ڪࣷ؞ ‏12 بࣺسْ بِ ݧُيْ يࣺڪَّتِ تَڪّ بِ، جࣺبَّلࣹ، نَنگࣹينْ ݖِ سَرْخَلْ ڪُويْ مُ اَمُلْ سِڪّ، تࣷلُّ ݖِ مࣹنّ اَتْ، ݧُ دࣹفَلْ ڪࣷ اَجِ سَخْ جِ سَرَخُ رࣹنْدِ-دٗومَلْ؞ ‏13 سَرَخُ ݒࣹݒّ مِ مُيْ اࣵنْدَلْ جُرٗومْ بࣹنِّ ڪِلࣷيْ سُنگُفْ سُ مُݖّ اَيِبْ لَيْ دࣷونْ، ݧُ خِيوࣹ ڪࣷ دِوْ، مُيْ سَرَخُ سَوَرَ، ݧࣹيلْ اَجِ سَخْ جِ، نگِرْ خࣹتُگْ جࣵمّ؞ سَرَخْ سِ ݧُيْ تُورَلْ اَجِ سَخْ جِ، لِيتَرُ بِيݧْ لاَڪْ گࣹنّ-وࣵلّ؞ ‏14 دُ مبُرُ، دُ ݒࣹݒّ مُ ݧُ سࣺنْدَلْ، دُ ݒࣹݒّ مُ بࣹيسْ مُ نگࣹينْ وَرَ لࣹڪّ، بَ ڪࣹرࣷوگْ بࣺسْ بࣷوبُ نگࣹينْ دِ اِنْدِ سࣹينْ سَرَخُ يࣵلَّ؞ دࣷگَلَ نگࣷوگُ، دِ لُ لࣹينْ وَرْ فࣵوّ، يࣹينْ اَڪْ سࣹينْ اَسْڪَنْ، اَڪْ فࣺݒّ فُ نگࣹينْ دࣴڪّ؞
لُ جࣴمْ ݖِ مࣵگَّلُ بࣺسُ نگٗوبْ
‏15 «بࣺسُبْ نࣷفْلاَيْ بَ نگࣹينْ دِ اِنْدِ تَڪُّ بࣹلࣹ بِيْ سَرَخُ يࣺڪَّتِ-جࣺبَّلࣹ، بࣺسْ بَ ݖَ تࣷݒّ نگࣹينْ دِ دَلࣹ وَݧّ جُرٗومْ ݧَارِ اَيِ بࣺسْ يُ مَتْ سࣴڪّ؞ ‏16 وَݧّلࣹينْ بَ ݖَ بࣺسْ بَ تࣷݒّ ݖِ جُرٗومْ ݧَارࣹيلُ بࣺسُبْ نࣷفْلاَيْ بَ، مُيْ جُرٗومْ فُڪِّ فَنْ، نگࣹينْ اِنْدِلْ ݖَ اَجِ سَخْ جِ سَرَخُ ݒࣹݒّ مُ بࣹيسْ؞ ‏17 سࣹينِ ڪࣴرْ نگࣹينْ دِ جࣴلࣹ ݧَارِ مبُرُ يُيْ دࣷونْ سَرَخُ يࣺڪَّتِ-جࣺبَّلࣹ، مُ ݖِ نࣹڪّ دِ لُ ݧُ لَڪّࣹ جُرٗومْ بࣹنِّ ڪِلࣷيْ سُنگُفْ سُ مُݖّ اَيِبْ؛ دࣹيسْ ݖِيْ دࣹفْ لَوِيرْ، لَڪَّالࣹ ڪࣷ ڪࣷ، مُيْ سَرَخُ ندࣷورْتࣹلْ مࣹݧّࣹيفْ، ݧࣹيلْ اَجِ سَخْ جِ؞ ‏18 نَ مبࣷولࣷو مِ بࣷولࣹيڪْ مبُرُ مِ جُرٗومْ ݧَارِ ڪُويْ يُ اَمُلْ سِڪّ تࣹ تࣷلُّ ݖِ مࣹنّ اَتْ، اَڪْ يࣴڪّ اَڪْ ݧَارِ ڪُويْ، مُ دࣷونْ سَرَخُ رࣹنْدِ-دٗومَلْ، ݧࣹيلْ اَجِ سَخْ جِ؞ نَ اࣵندّ اَڪْ سَرَخُ ݒࣹݒّ اَڪْ سَرَخْ سِ ݧُيْ تُورْ؞ لࣷولُ سَرَخْ لَ نگِرْ خࣹتُگْ جࣵمّ، ݧࣹيلْ اَجِ سَخْ جِ؞ ‏19 تࣹ اِتَمْ نگࣹينْ سَرْخَلْ اَبْ سِڪّࣹتْ بُيْ دࣷونْ سَرَخُ ݒٗوتُمْ بࣵڪَّارْ اَڪْ ݧَارِ خَرِ مࣹنّ اَتْ، مُيْ سَرَخُ ݖَنتّ ݖِ بِيرْ جࣵمّ؞ ‏20 سَرْخَلْڪَتْ بِ دَيْ يࣺڪَّتِ ݧَارِ خَرْ يࣷويُ، بࣷولࣹ ڪࣷوڪْ مبُرُمْ ندࣷورْتࣹلْ مࣹݧّࣹيفْ مِ، دࣹفْ ڪࣷ سَرَخْ بُ ݧُيْ يࣺڪَّتِ، جࣺبَّلْ اَجِ سَخْ جِ؞ يࣷويُ سَرَخْ دِ لُ سࣹلّ ݧࣹيلْ اَجِ سَخْ جِ، سَرْخَلْڪَتْ بࣹي ڪࣷ مࣷومْ؞ ‏21 وࣷوتࣹلࣹينْ ݖِ بࣺسْ بࣷوبُ ندَجࣹ مُ سࣹلّ، تࣹ لُ نگࣹينْ فَرَلْ دِ لِگّࣺييْ بُلࣹينْ ڪࣷ ݖِ لِگّࣺييْ؞ دࣷگَلَ نگࣷوگُ، دِ لُ لࣹينْ وَرْ فࣵوّ، يࣹينْ اَڪْ سࣹينْ اَسْڪَنْ، اَڪْ فࣺݒّ فُ نگࣹينْ دࣴڪّ؞
‏22 «بُ نگࣹينْ دࣹي گٗوبْ سࣹينْ تࣷولْ، بُلࣹينْ گٗوبْ بَ فَ تࣷولْ بَ يࣹمْ تࣹ بُلࣹينْ ݖِ تࣷݒَّاتْ، دِ فࣷرَاتُ؞ اَجِ ݧࣵڪّ اَڪُبْ دࣷخَنْدࣺيمْ نگࣹينْ ڪࣷيْ بࣵيّࣹي؞ مَايْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ؞»
لِ ݖِ مࣵگَّلُ بࣺسُ لِيتْ يَ
‏23 گَنَّاوْ لࣷولُ اَجِ سَخْ جِ وَخْ نَ مُوسَىٰ نࣹ ڪࣷ:‏ ‏24 «وَخَلْ بࣵنِّ إِسْرَاءِيلْ نࣹ لࣹينْ:‏ بُ جُرٗومْ ݧَارࣹيلُ وࣹيرْ تَخَوࣹي، بࣺسْ بِ جࣴڪّ ݖِ وࣹيرْ وِ، نَ دࣷونْ سࣹينْ بࣺسُبْ نࣷفْلاَيْ بُ نگࣹينْ دِ اَمَلْ ندَجࣹ مُ سࣹلّ؞ دࣹيلࣹينْ فࣵتّلࣹي بࣺسْ بِ لِيتْ يُ خُمبّ؞ ‏25 مبࣷولࣹمْ لُ نگࣹينْ فَرَلْ دِ لِگّࣺييْ بُلࣹينْ ڪࣷ ݖِ لِگّࣺييْ، وَايࣹ نَنگࣹينْ ݖِ اِنْدِ سَرَخُ سَوَرَ، ݧࣹيلْ اَجِ سَخْ جِ؞»
لِ ݖِ بࣺسُبْ نجࣷتْلاَيْ بَ
‏26 اَجِ سَخْ جِ تࣹگْ ݖَ وَخْ مُوسَىٰ نࣹ ڪࣷ:‏ ‏27 «تࣹ اِتَمْ فُڪِّ فَنْ ݖِ جُرٗومْ ݧَارࣹيلُ وࣹيرْ وࣷووُ، بࣺسُبْ نجࣷتْلاَيْ بَ لَيْ دࣷونْ؞ ندَجࣹ مُ سࣹلّ نگࣹينْ ݖِيْ اَمَلْ؞ نَنگࣹينْ ݖِ تࣷرࣷخْلُ تࣹ اِنْدِ سَرَخُ سَوَرَ، ݧࣹيلْ اَجِ سَخْ جِ؞ ‏28 دُ لࣹنّ لُ نگࣹينْ دِ لِگّࣺييْ ݖِ بࣷوبُ بࣺسْ، ندَخْ بࣺسُبْ نجࣷتْلاَيْ لَ بُ نگࣹينْ دِ جࣷتࣷو فِ سࣹينْ ڪَنَمْ يࣵلَّ، اَجِ سَخْ جِ؞ ‏29 تࣹ ڪَتْ، ڪࣺݒّ ڪُ تࣷرࣷخْلُوُلْ بࣷوبُ بࣺسْ، دࣹيسْ ڪࣷيْ دَگّࣹ ݖِ بِيرْ بࣵنِّ إِسْرَاءِيلْ؞ ‏30 ڪࣺݒّ ڪُ لِگّࣺييْ لࣹنّ ݖِ بࣷوبُ بࣺسْ، سُ بࣷوبَا دِنَا فَرْ ڪࣷوڪُ ݖِ بِيرْ بࣵنِّ إِسْرَاءِيلْ، سࣵنڪّ ڪࣷ؞ ‏31 بُلࣹينْ ݖِ لِگّࣺييْ دَرَ؞ دࣷگَلَ نگࣷوگُ، دِ لُ لࣹينْ وَرْ فࣵوّ، يࣹينْ اَڪْ سࣹينْ اَسْڪَنْ اَڪْ فࣺݒّ فُ نگࣹينْ دࣴڪّ؞ ‏32 سࣹينْ بࣺسُبْ نࣷفْلاَيْ لَ بُ نگࣹينْ دِ نࣷݒّلُ دࣷݝّ تࣹ وَرْ ݖࣹي تࣷرࣷخْلُ؞ لِ ڪࣷ دَلࣹ ݖِ نگࣷونُگْ جُرٗومْ ݧࣹينْتِ فَنْ ݖِ وࣹيرْ وِ بَ ݖَ اࣴلّࣴگْ سَ، بَ جَنتّ سࣷ، ݖِ نگࣹينْ دِ وࣷرْمَالْ سࣹينْ بࣺسُبْ نࣷفْلاَيْ؞»
لِ ݖِ مࣵگَّلُ بࣺسِ مبَارْ يِ
‏33 سُ لࣷولُ وࣺيࣹي اَجِ سَخْ جِ وَخْ نَ مُوسَىٰ نࣹ ڪࣷ:‏ ‏34 «وَخَلْ بࣵنِّ إِسْرَاءِيلْ نࣹ لࣹينْ:‏ ݧُ وَݧّ دَلࣹ ڪࣷ ݖِ فُڪِّ فَنْ اَڪْ جُرٗومْ ݖِ وࣹيرُ جُرٗومْ ݧَارࣹيلْ وࣷووُ بَ مُ اَمْ جُرٗومْ ݧَارِ فَنْ؞ دِيرْ بࣷوبُ مࣷويْ مࣵگَّلُ بࣺسِ مبَارْ يِ، ݧࣹيلْ اَجِ سَخْ جِ؞ ‏35 بࣺسْ بُ جࣴڪّ بِ بࣺسُبْ ندَجࣹ مُ سࣹلّ لَ؞ مبࣷولࣹمْ لُ نگࣹينْ فَرَلْ دِ لِگّࣺييْ بُلࣹينْ ڪࣷ ݖِ لِگّࣺييْ؞ ‏36 دِيرُ جُرٗومْ ݧَارِ فَنْ نگࣹينْ دِ اِنْدِ سَرَخُ سَوَرَ، ݧࣹيلْ اَجِ سَخْ جِ؞ بࣺسُبْ جُرٗومْ ݧࣹتّࣹيلْ، بَ وَرَ دࣷونْ سࣹينْ ندَجࣹ مُ سࣹلّ، نَنگࣹينْ ݖِ اِنْدِ سَرَخُ سَوَرَ، ݧࣹيلْ اَجِ سَخْ جِ؞ ندَجࣹمْ مُجَّنْتَلْ بَ لَ؞ مبࣷولࣹمْ لُ نگࣹينْ فَرَلْ دِ لِگّࣺييْ بُلࣹينْ ڪࣷ ݖِ لِگّࣺييْ؞
‏37 «يࣷويࣷو دِ مࣵگَّلِ اَجِ سَخْ جِ، يِ ݧُ وَرَ يࣺگْلࣹ، دِ ندَجࣹ يُ سࣹلّ يُ ݧُيْ اِنْدِ اَيْ سَرَخِ سَوَرَ، ݧࣹيلْ اَجِ سَخْ جِ، مُيْ سَرَخُ رࣹنْدِ-دٗومَلْ، اَڪْ سَرَخُ ݒࣹݒّ، اَڪْ يࣹنࣹينْ سَرَخِ جُرْ گُ ݧُيْ رࣹنْدِ، اَڪْ سَرَخْ يِ ݧُيْ تُورَلْ اَجِ سَخْ جِ، بࣺسْ بُ نࣹڪّ اَڪْ لَ ݖَ وَرْ؞ ‏38 يࣷويُ سَرَخْ جࣷتࣹوُلْ دَرَ اَڪْ سَرَخِ بࣺسِ نࣷفْلاَيْ يُ اَجِ سَخْ جِ؞ جࣷتࣹوُلْ دَرَ اِتْ اَڪْ يِ نگࣹينْ دِ جࣷخࣹيلْ اَجِ سَخْ جِ مبَا مبࣷولࣹمْ سَرَخْ يِ نگࣹينْ دِ وࣵݖّࣷو نگِݧْ، اَڪْ مبࣷولࣹمْ سَرَخِ يࣺينࣹ يِ نگࣹينْ دِ دࣹفَلْ اَجِ سَخْ جِ؞
‏39 «ݖِ بِيرْ لࣷولُ، فُڪِّ فَنْ اَڪْ جُرٗومْ ݖِ جُرٗومْ ݧَارࣹيلُ وࣹيرْ وِ، گَنَّاوْ بُ نگࣹينْ دَجَلࣹي مࣹݧّࣹيفُمْ رࣺيوْ مِ، اَمَلْ-لࣹينْ مࣵگَّلُگْ اَجِ سَخْ جِ دِيرُ جُرٗومْ ݧَارِ فَنْ؞ نَ بࣺسْ بُ جࣴڪّ بَ دِ بࣺسُبْ نࣷفْلاَيْ، بࣺسُبْ جُرٗومْ ݧࣹتّࣹيلْ بَ دِ بࣺسُبْ نࣷفْلاَيْ؞ ‏40 بࣺسْ بُ جࣴڪّ بَ وِتّلࣹينْ دࣷومِ گَرَبْ يُ رَفࣹتْ اَڪْ ݖَرِ تِيرْ اَڪْ ݖَرِ گَرَبْ يُ سࣴقْ اَڪْ ݖَرِ گَرَبْ يُ ݧُيْ وَخْ سٗولْ، تࣹيْ سَخࣹ تࣵڪَّلْ دࣹخْ؞ نَنگࣹينْ بࣵنّࣹيخُ فِ سࣹينْ ڪَنَمْ يࣵلَّ اَجِ سَخْ جِ دِيرُ جُرٗومْ ݧَارِ فَنْ؞ ‏41 وࣷرْمَالْ-لࣹينْ مࣵگَّلُگْ اَجِ سَخْ جࣷوجُ، دِ جُرٗومْ ݧَارِ فَنْ ݖِمْ اَتْ، تࣹ وࣷرْمَالْ-لࣹينْ ڪࣷ ݖِ جُرٗومْ ݧَارࣹيلُ وࣹيرْ وِ؞ دࣷگَلَ نگࣷوگُ، دِ لُ لࣹينْ وَرْ فࣵوّ، يࣹينْ اَڪْ سࣹينْ اَسْڪَنْ؞ ‏42 ݖِ اَيْ مبَارْ نگࣹينْ دِ دࣴڪّ دِيرُ جُرٗومْ ݧَارِ فَنْ يࣷويُ؞ مبࣷولࣹمْ نجُدُّ-جِ-رࣺيوْ ݖِ إِسْرَاءِيلْ نَ ݖِ دࣴڪّࣹ مبَارْ، ‏43 سُ ڪࣷ دࣹفࣹي سࣹينِ سࣴتْ خَمْ نࣹ ݖِيْ مبَارْ لاَ دࣴڪَّلࣷونْ بࣵنِّ إِسْرَاءِيلْ، بَ مَ لࣹينْ گࣺنّࣹي رࣺيوُمْ مِسْرَ؞ مَايْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ؞»
‏44 نࣷونُ لَ مُوسَىٰ خَمَلࣹ بࣵنِّ إِسْرَاءِيلْ بࣺسِ مࣵگَّلُگْ اَجِ سَخْ جِ؞
