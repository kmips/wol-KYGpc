لُ جࣴمْ ݖِ سَݧْ-سَݧِ سَرْخَلْڪَتْ يِ
21
‏1 گَنَّاوْ لࣷولُ اَجِ سَخْ جِ وَخْ نَ مُوسَىٰ نࣹ ڪࣷ:‏ «وَخَلْ سَرْخَلْڪَتْ يِيْ دࣷومِ هَـٰرُونَ نࣹ لࣹينْ:‏ بُ ݖِ ڪࣹنّ سࣷبࣹيلْ بࣷݒَّمْ، دِ لاَلْ مبࣷڪَّمْ مُ دࣹي، ‏2 سُ دُلْ ݧِ ڪࣷ گࣴنَ جࣹگࣹ ݖِ وࣵلُّ مبࣷڪّ، مُيْ ندࣹيَمْ مبَا بَايَمْ مبَا دࣷومَمْ جُ گٗورْ اَڪْ دࣷومَمْ جُ جِگࣺينْ، مبَا ڪُ گٗورْ ڪُ مُ بࣷڪَّلْ وَاجُرْ، ‏3 مبَا اَبْ جِگࣺينَمْ بُ سࣺيَگُلْ تࣹ بࣷوتُ ݖِ مࣷومْ ندَخْ اَمُلْ جࣴڪّࣴرْ— ڪُ مُ بࣷڪَّلْ نࣷونُ، سَݧْ نَا سࣷبࣹوُ ندَخْ مࣷومْ؞ ‏4 وَرُلَ سࣷبࣹيلْ بࣷݒَّمْ، دِ تࣹدَّدِلْ بࣷݒَّمْ، ندَخْ ڪِلِفَ لَ ݖِوْ خࣹيتَمْ؞
‏5 «بُݧُ وَتْ فࣹنّ ݖِ سࣹينْ بࣷݒّ تࣹ بُݧُ وَتْ سࣹينْ ݒࣹگِّ سِڪِّمْ، بُݧُ دَگّ سࣹينْ يَرَمْ اِتْ، دِ ڪࣷ مِتِتْلࣷو ڪُ دࣹي؞ ‏6 نَݧُ نࣹڪَّلْ سࣹينْ يࣵلَّ اَيْ نِتْ ݧُ سࣹلّ، تࣹ بَݧَ تࣹدَّدِلْ سࣹينْ تُرُ يࣵلَّ، ندَخْ سَرَخِ سَوَرْ يِ ݧࣹيلْ اَجِ سَخْ جِ، تࣹ دِ سࣹينْ ݧَمُ يࣵلَّ، ݧࣷو ڪࣷيْ جࣷخࣹ؞ ڪࣷنْ نَگْ دَݧࣷو وَرَ سࣹلّ؞
‏7 «اَبْ گࣵنݖّ مبَا ڪُ ݧُ تࣷرْخَلْ، تࣴدّࣹ ڪࣷ، ݧࣷومْ دُݧُ ڪࣷ جࣴلْ سࣷخْنَ؞ ڪُ جࣴڪّࣴرَمْ بَالْ اِتْ دُݧُ ڪࣷ جࣴلْ سࣷخْنَ، ندَخْ دَݧࣷو سࣹلَّلَلْ سࣹينْ بࣷݒّ سࣹينْ يࣵلَّ؞ ‏8 تࣹ اِتَمْ نَنگࣹينْ رࣵݧّࣹي نࣹ ݧُ سࣹلّ لَݧُ، ندَخْ سࣹينْ ݧَمُ يࣵلَّ، ݧࣷو ڪࣷيْ جࣷخࣹ؞ نَنگࣹينْ لࣹينْ جࣷخْ ݖࣺرُ نِتْ ݧُ سࣹلّ، ندَخْ ڪُ سࣹلّ لاَ، مَنْ اَجِ سَخْ جِ لࣹينْ سࣹلَّلْ، يࣹينْ؞
‏9 «سُ فࣹڪّࣹي نࣹ دࣷومُ سَرْخَلْڪَتْ دَ دِ جِگࣺينْ جُيْ تࣹدَّدِلْ بࣷݒَّمْ، دِ گࣵنݖّتُ، ڪࣷوڪُ بَايَمْ لَيْ تࣷرْخَلْ؞ دࣹيسْ ڪࣷيْ تَالْ، لَڪّ ڪࣷ؞
‏10 «سَرْخَلْڪَتْ، بِ گࣴنَ مࣵگّ ݖِ بࣷڪَّمْ يِيْ سَرْخَلْڪَتْ، تࣹ ݧُ سࣷتِّ دِوْ گِ ݧُيْ فَلࣹ ݖِ بࣷݒَّمْ، سࣹلَّلْ ڪࣷ، مُيْ سࣷلْ يࣺرࣹ يُ سࣹلّ يِ، ڪࣷوڪُ دُ نجَݧُ تࣹ دُ خࣷتِّ اَيْ يࣺرࣹيمْ، دِ ڪࣷ ݧَاوْلࣷو ڪُ دࣹي؞ ‏11 فُ اَبْ نࣺيوْ نࣹڪّ دُ فَ دࣹمْ، بُ دࣷونْ ندࣹيَمْ مبَا بَايَمْ سَخْ، دُ ڪࣷ تَخَ سࣷبࣹيلْ بࣷݒَّمْ؞ ‏12 دُ گࣺنّ اࣴتُّ بࣺرَبْ بُ سࣹلّ بِ، مبَا مُيْ تࣹدَّدِلْ بࣺرَبُ يࣵلاَّمْ بُ سࣹلّ بِ، ندَخْ دِوُ ݒَلْ گُ سࣹلّ گِ ڪࣷ يࣵلاَّمْ فَلࣹ؞ مَايْ اَجِ سَخْ جِ؞ ‏13 سَرْخَلْڪَتْ بُ مَگْ بِ، ڪُ خَمُلْ گٗورْ لَيْ جࣴلْ سࣷخْنَ؞ ‏14 ڪُ جࣴڪّࣴرَمْ فَاتُ مبَا ڪُ ݧُ بَالْ مبَا ڪُ ݧُ تࣷرْخَلْ، تࣴدّࣹ ڪࣷ، مبَا اَبْ گࣵنݖّ، دُ ڪࣷ مَنَ جࣴلْ سࣷخْنَ؞ اَبْ جَنقّ بُ مُ بࣷڪَّلْ دࣷݝّ لَ مَنَ جࣴلْ سࣷخْنَ؞ ‏15 سُ ڪࣷ دࣹفࣹي دُ تࣹدَّدِلْ اَوْ اَسْڪَنَمْ ݖِ بِيرْ بࣵنِّ إِسْرَاءِيلْ، ندَخْ مَنْ اَجِ سَخْ جِ مَا ڪࣷ سࣹلَّلْ؞»
لِيْ نگࣵنتّ ݖِ سَرْخَلْڪَتْ
‏16 بُ لࣷولُ وࣺيࣹي اَجِ سَخْ جِ وَخْ نَ مُوسَىٰ نࣹ ڪࣷ:‏ ‏17 «وَخَلْ هَـٰرُونَ نࣹ ڪࣷ:‏ مُيْ تࣹيْ مُيْ اࣴلّࣴگْ بَ فࣵوّ، گٗورْ گُ اَسْڪَنࣷو ݖِ يَوْ اَڪْ مَاسْ گُ مُ مَنَ بࣷڪّ، سُ اَمࣹي سِڪّ دُ مَنَ اَگْسِ بَ ݖِ سَرْخَلُڪَايْ بِ، دِ جࣷخࣹ ݧَمُ يࣵلاَّمْ؞ ‏18 دُ ڪࣹنّ سَخْ ڪُ اَمْ سِڪّ ڪُيْ اَگْسِ بَ فِي، دُ گُمْبَ، دُ ڪُيْ سࣷوخْ، دُ ڪُ خَرْ-ڪَنَمَمْ اَمْ سِڪّ مبَا ڪُ ݖࣺرْ يِ سُتَاتࣹ؞ ‏19 دُ اِتْ ڪُ تࣵنڪّ بِ لࣵگِّ مبَا لࣷخࣷ بِ، ‏20 مبَا اَبْ خُوگࣹ مبَا اَبْ تُنگُنࣹ؛ دُ اِتْ ڪُ بࣴتَمْ اَمْ سِڪّ مبَا ڪُ رࣵمّ مبَا ڪُ اَمْ گٗومْ يُ سࣷلْ دࣴتّ، مبَا ڪُ ݧُ تࣵݒّ؞ ‏21 اَسْڪَنُ هَـٰرُونَ سَرْخَلْڪَتْ بِ، دُ ڪࣹنّ ڪُ ݖِ اَمْ سِڪّ ڪُيْ اَگْسِ بَ فِي، دِ جࣷخࣹ سَرَخِ سَوَرَ يِ ݧࣹيلْ اَجِ سَخْ جِ؞ سِڪّ سِ ݖِ مࣷومَ تَخْ دُ مَنَ اَگْسِ بَ فِي، دِ جࣷخࣹ ݧَمُ يࣵلاَّمْ؞ ‏22 دُ تࣹرࣹ مُ سَݧَ لࣹڪّ ݖِ ݧَمُ يࣵلاَّمْ، مُيْ ݧَمْ وُ سࣹلّ اَڪْ وُ سࣹلَّ سࣹلّ؞ ‏23 وَايࣹ وَرُلَ جࣹگࣹ رِدࣷ بِ، مبَا مُيْ دࣹمْ بَ ݖَ سَرْخَلُڪَايْ بَ، ندَخْ سِڪّ سِ ݖِ مࣷومْ؞ بُمُ تࣹدَّدِلْ سَمَ بࣺرَبْ يُ سࣹلّ، نگِرْ مَنْ اَجِ سَخْ جِ مَا سࣹلَّلْ بࣺرَبْ يࣷويُ؞»
‏24 مُوسَىٰ جࣷتّلِ نَ هَـٰرُونَ اَڪِ دࣷومَمْ ڪࣵدُّ يࣷويُ، ݧࣷومْ اَڪْ مبࣷولࣹمْ بࣵنِّ إِسْرَاءِيلْ؞
