لِ وَرْ ݖِ جࣵنگࣷرࣷيْ دࣹرْ
13
‏1 اَجِ سَخْ جِ وَخْ نَ مُوسَىٰ اَڪْ هَـٰرُونَ نࣹ لࣹينْ:‏ ‏2 «سُ نِتْ اَمࣹي اَبْ تࣴرْگࣴنْ ݖِ يَرَمَمْ مبَا اَبْ اࣴرْ، مبَا اَبْ گࣵڪّ بُ وࣹيخْ بُ مَنَ دࣷونْ جࣵنگࣷرࣷيْ دࣹرْ جُيْ لَوْ ݖِ يَرَمَمْ، نَݧُ ڪࣷ يٗبُّ ݖِ هَـٰرُونَ سَرْخَلْڪَتْ بِ مبَا ڪࣹنّ ݖِيْ دࣷومَمْ يُ گٗورْ يِيْ سَرْخَلْڪَتْ يِ؞ ‏3 نَ سَرْخَلْڪَتْ بِ سࣹيتْ گࣵڪّ بِ ݖِ يَرَمُ نِتْ ڪِ؛ سُ فࣹڪّࣹي نࣹ ڪَوَرْ گِ ݖِ گࣵڪّ بِ دَفَ سࣷݒِّڪُ وࣹيخْ، تࣹ مُ مࣹلْ نِ لُ دࣹفْ ݒَخْ ݖِ دࣹرْ بِ، جࣵنگࣷرࣷيْ دࣹرْ جُيْ لَوْ لَ؞ بُ سَرْخَلْڪَتْ بِ گِسࣹي لࣷولُ، نَ بِرَلْ نࣹ نِتْ ڪِ سࣷبࣹوُ نَ؞ ‏4 وَايࣹ سُ دࣹي اَبْ گࣵڪّ بُ وࣹيخْ ݖِ يَرَمْ وِ تࣹ مࣹلُلْ نِ لُ دࣹفْ ݒَخْ ݖِ دࣹرْ بِ، تَخُلْ اِتَمْ ڪَوَرْ گِ ݖِ نࣹڪّ سࣷݒِّڪُ وࣹيخْ، نَ سَرْخَلْڪَتْ بِ بࣹرْ نِتْ ڪِ اَمࣹ گࣵڪّ بِ دِيرُ جُرٗومْ ݧَارِ فَنْ؞ ‏5 نَ ڪࣷ سَرْخَلْڪَتْ بِ خࣷولاَتْ بࣺسُبْ جُرٗومْ ݧَارࣹيلْ بَ؞ بُ سَرْخَلْڪَتْ بِ سࣹيتْلࣷو نࣹ گࣵڪّ بِ سࣷݒِّڪُوُلْ مبَا مُ لَوْ ݖِ يَرَمَمْ، نَ ڪࣷ بࣹرَاتْ جُرٗومْ ݧَارِ فَنْ؞ ‏6 نَ ڪࣷ سَرْخَلْڪَتْ بِ خࣷولاَتْ ݖَ بࣺسُبْ جُرٗومْ ݧَارࣹيلْ بَ؞ سُ فࣹڪّࣹي گࣵڪّ بِ دِ رَافْ تࣹ لَوُلْ ݖِ ڪَوْ دࣹرُ يَرَمْ وِ، نَ سَرْخَلْڪَتْ بِ بِرَلْ نࣹ نِتْ ڪِ سࣹتْ نَ؛ اَبْ اࣴرْ دࣷݝّ لَ؞ دَيْ فٗوتْ اَيْ يࣺرࣹيمْ رࣹڪّ دَلْدِ سࣹتْ؞ ‏7 سُ فࣹڪّࣹي نࣹ وٗورْ نَ نࣹ اࣴرْ بِ لَوْ نَ ݖِ دࣹرُ يَرَمُ نِتْ ڪِ، گَنَّاوْ بَ مُ ڪࣷ وࣷنࣹي سَرْخَلْڪَتْ بِ، بَ مُ بِرَلْ نࣹ سࣹتْ نَ، دَيْ دࣹلُّ ݖَ سَرْخَلْڪَتْ بَ؞ ‏8 سُ بࣷوبَا نَ ڪࣷ سَرْخَلْڪَتْ بِ خࣷولاَتْ بَ گِسْ اࣴرْ بِ لَوْ ݖِ دࣹرُ يَرَمْ وِ، مُ دࣷورَ بِرَلْ نࣹ نِتْ ڪِ سࣷبࣹوُ نَ تࣹ جࣵنگࣷرࣷيْ دࣹرْ جُيْ لَوْ لَ؞
‏9 «ڪࣺݒّ ڪُ اَمࣹ جࣵنگࣷرࣷيْ دࣹرْ جُيْ لَوْ، نَݧُ ڪࣷ يٗبُّ ݖَ سَرْخَلْڪَتْ بَ؞ ‏10 نَ ڪࣷ سَرْخَلْڪَتْ بِ سࣹيتْ؞ سُ دࣹي اَبْ تࣴرْگࣴنْ بُ وࣹيخْ تࣹ ڪَوَرْ گِ ݖِ نࣹڪّ سࣷݒِّڪُ وࣹيخْ، بࣷولࣹ ݖِ دࣹفْ گٗومْ ݖَ بِيرْ، ‏11 ڪࣷنْ جࣵنگࣷرࣷيْ دࣹرْ جُيْ لَوْ لَ تࣹ خَسْ نَا لَوْ ݖِ يَرَمْ وِ؞ نَ سَرْخَلْڪَتْ بِ بِرَلْ نࣹ نِتْ ڪِ سࣷبࣹوُ نَ؞ دُ جَرْ مُ دِ ڪࣷ بࣹرْ ندَخْ تَقْ نَ سࣷبࣹ بَ نࣷݒِّ؞ ‏12 سُ فࣹڪّࣹي نࣹ نَگْ جࣵنگࣷرࣷ جِ گࣺنّ نَا گࣺنّ ݖِ يَرَمُ نِتْ ڪِ، سَرْخَلْڪَتْ بِ گِسْ ڪࣷ فࣺݒّ، لِ ڪࣷ دَلࣹ ݖِ بࣷݒّ بِ بَ ݖِ تࣵنڪّ يِ، ‏13 نَ ڪࣷ سَرْخَلْڪَتْ بِ سࣹيتْ بُ بَاخْ؞ بُ گِسࣹي نࣹ لِ گࣺنّ ݖِ يَرَمَمْ دَجْ نَ يَرَمْ وِ يࣺݒّ، نَ بِرَلْ نࣹ نِتْ ڪِ مُ دَلْ سࣹتْ نَ؛ گَنَّاوْ لࣺݒَّمَ سࣷݒِّڪُ وࣹيخْ، سࣹتْ نَ؞ ‏14 بَ تࣹيْ بࣺسْ بُ اَيْ گٗومْ گࣺنَّاتࣹي ݖِ يَرَمْ وِ، نِتْ ڪِ سࣷبࣹوُ نَ؞ ‏15 سُ سَرْخَلْڪَتْ بِ گِسࣹي يَرَمُ نِتْ ڪِ دࣹفْ گٗومْ رࣹڪّ، نَ بِرَلْ نࣹ سࣷبࣹوُ نَ؞ گٗومْ بࣷوبُ گࣺنّ ݖِ يَرَمْ وِ سࣷبࣹ لَ؞ جࣵنگࣷرࣷيْ دࣹرْ جُيْ لَوْ لَ؞ ‏16 تࣹ اِتْ بُ گٗومْ بِ سࣷݒِّڪࣷو وࣹيخْ، نَ دࣹمْ ݖَ سَرْخَلْڪَتْ بَ؞ ‏17 نَ ڪࣷ سَرْخَلْڪَتْ بِ سࣹيتْ، بَ گِسْ نࣹ وࣹيخَاتْ نَ، مُ بِرَلْ نࣹ ڪِ جࣵنگࣷرࣷ جِ دَلْ سࣹتْ نَ تࣹ ڪࣷنْ اِتْ سࣹتْ نَ؞
‏18 «سُ نِتْ تَابࣹي بَ وࣺرْ، ‏19 اَبْ تࣴرْگࣴنْ بُ وࣹيخْ مبَا گࣵڪّ بُ وࣹيخْ بُ رَخْ خࣷنقّ فࣹيݧْ فَ تَابْ بَ نࣹڪّࣷونْ، نَ دࣹمْ ݖَ سَرْخَلْڪَتْ بَ؞ ‏20 سُ ڪࣷ سَرْخَلْڪَتْ بِ سࣹيتࣹي بَ گِسْ فَ تَابْ بَ نࣹڪّࣷونْ مࣹلْ نِ لُ دࣹفْ ݒَخْ، تࣹ ڪَوَرْ گِ ݖِ نࣹڪّ سࣷݒِّڪُ وࣹيخْ، نَ سَرْخَلْڪَتْ بِ بِرَلْ نࣹ نِتْ ڪِ سࣷبࣹوُ نَ؞ جࣵنگࣷرࣷيْ دࣹرْ جُيْ لَوْ لَ، جُ جُدّࣷو ݖَ تَابْ بَ؞ ‏21 وَايࣹ سُ ڪࣷ سَرْخَلْڪَتْ بِ سࣹيتࣹي تࣹ گِسُ ݖَ ڪَوَرْ گُ وࣹيخْ، دࣹفُلْ ݒَخْ اِتْ تࣹ رَافْ نَ، نَ بࣹرْ نِتْ ڪِ جُرٗومْ ݧَارِ فَنْ؞ ‏22 سُ وࣺرࣹي نࣹ جࣵنگࣷرࣷ جِ لَوْ نَ ݖِ يَرَمْ وِ، نَ سَرْخَلْڪَتْ بِ بِرَلْ نࣹ نِتْ ڪِ سࣷبࣹوُ نَ؞ جࣵنگࣷرࣷيْ دࣹرْ جُيْ لَوْ لَ؞ ‏23 سُ گࣵڪّ بِ يࣹمࣹي فࣷوفُ، بَݧَ لَوْ، لࣺگࣺتُبْ تَابْ بَ لَ، تࣹ نَ سَرْخَلْڪَتْ بِ بِرَلْ نࣹ نِتْ ڪِ سࣹتْ نَ؞
‏24 «سُ نِتْ لَڪّࣹي، اَبْ گࣵڪّ بُ وࣹيخْ مبَا بُ وࣹيخْ تࣹ رَخْ خࣷنقّ فࣹيݧْ ݖَ ڪَوْ سُوخْ وُ خࣷنقّ وَ لَڪّ، ‏25 نَ ڪࣷ سَرْخَلْڪَتْ بِ سࣹيتْ؞ سُ ڪَوَرْ گِ ݖِ نࣹڪّ سࣷݒِّڪࣷو وࣹيخْ تࣹ گࣵڪّ بِ مࣹلْ نِ لُ دࣹفْ ݒَخْ، ڪࣷنْ جࣵنگࣷرࣷيْ دࣹرْ جُيْ لَوْ لَ، جُ جُدّࣷو ݖَ لَڪّ-لَڪّ بَ؞ نَ سَرْخَلْڪَتْ بِ بِرَلْ نࣹ نِتْ ڪِ سࣷبࣹوُ نَ، ندَخْ جࣵنگࣷرࣷيْ دࣹرْ جُيْ لَوْ لَ؞ ‏26 وَايࣹ سُ ڪࣷ سَرْخَلْڪَتْ بِ سࣹيتࣹي، گِسُلْ ڪَوَرْ گُ وࣹيخْ ݖِ گࣵڪّ بِ، دࣹفُلْ ݒَخْ اِتْ تࣹ رَافْ نَ، نَ بࣹرْ نِتْ ڪِ جُرٗومْ ݧَارِ فَنْ؞ ‏27 سَرْخَلْڪَتْ بِ دَ ڪࣷيْ سࣹيتْ ݖَ بࣺسُبْ جُرٗومْ ݧَارࣹيلْ بَ؞ سُ وࣺرࣹي نࣹ جࣵنگࣷرࣷ جِ لَوْ نَ ݖِ يَرَمْ وِ، نَ بِرَلْ نࣹ نِتْ ڪِ سࣷبࣹوُ نَ؞ جࣵنگࣷرࣷيْ دࣹرْ جُيْ لَوْ لَ؞ ‏28 سُ گࣵڪّ بِ يࣹمࣹي فࣷوفُ تࣹ لَوُلْ ݖِ يَرَمْ وِ، اَبْ تُتْ لَ بُ جُدّࣷو ݖِ لَڪّ-لَڪّ بِ؞ نَ سَرْخَلْڪَتْ بِ بِرَلْ نࣹ نِتْ ڪِ سࣹتْ نَ، ندَخْ لࣺگࣺتُبْ لَڪّ-لَڪّ بَ لَ؞
‏29 «سُ فࣹڪّࣹي نࣹ گٗورْ مبَا جِگࣺينْ دَفَ اَمࣹ جࣵنگࣷرࣷ ݖِ بࣷݒّ مبَا ݖِ سِڪِّمْ، ‏30 نَ سَرْخَلْڪَتْ بِ سࣹيتْ فِ جࣵنگࣷرࣷ جِ نࣹڪّ؞ سُ مࣹلࣹي نِ لُ دࣹفْ ݒَخْ ݖِ دࣹرْ بِ، ڪَوَرْ گُ خَلّ تࣹ سࣹوْ نࣹڪّ ݖَ، نَ سَرْخَلْڪَتْ بِ بِرَلْ نࣹ نِتْ ڪِ سࣷبࣹوُ نَ؞ لࣷولُ وَتْ-وَتَانْ لَ، دِ اࣴرْ يُيْ گࣺنّ ݖِ بࣷݒّ مبَا سِڪِّمْ؞ ‏31 ندࣹگَمْ سَرْخَلْڪَتْ بِ سࣹيتْ نَ اࣴرْ وِ تࣹ نِرُوُلْ لُ دࣹفْ ݒَخْ، اَمُلْ اِتْ ڪَوَرْ گُ ݧُولْ ݖَ بِيرْ، دَيْ بࣹرْ نِتْ ڪِ اَمࣹ اࣴرْ وِ دِيرُ جُرٗومْ ݧَارِ فَنْ؞ ‏32 سَرْخَلْڪَتْ بِ دَيْ سࣹيتَاتْ اࣴرْ وِ ݖَ بࣺسُبْ جُرٗومْ ݧَارࣹيلْ بَ؞ سُ وَتْ-وَتَانْ جِ لَوُلْ، ڪَوَرْ گُ خَلّ نࣹڪُّ ݖَ، تࣹ مࣹلُلْ نِ لُ دࣹفْ ݒَخْ، ‏33 نَ نِتْ ڪِ وَتُ وَايࣹ دُ وَتَالࣹ وَتْ-وَتَانْ جِ؞ سَرْخَلْڪَتْ بِ دَ ڪࣷيْ بࣹرَاتْ جُرٗومْ ݧَارِ فَنْ؞ ‏34 نَ سَرْخَلْڪَتْ بِ سࣹيتَاتْ وَتْ-وَتَانْ جِ ݖَ بࣺسُبْ جُرٗومْ ݧَارࣹيلْ بَ؞ سُ وَتْ-وَتَانْ جِ لَوُلْ ݖِ يَرَمْ وِ تࣹ نِرُوُلْ لُ دࣹفْ ݒَخْ، سَرْخَلْڪَتْ بِ دَيْ بِرَلْ نࣹ نِتْ ڪِ سࣷبࣹوُوُلْ؞ نِتْ ڪِ دَيْ فٗوتْ اَيْ يࣺرࣹيمْ دَلْدِ سࣹتْ؞ ‏35 وَايࣹ سُ وࣺرࣹي نࣹ وَتْ-وَتَانْ جِ لَوْ نَ ݖِ يَرَمْ وِ گَنَّاوْ بَ ݧُ بِرَلࣹي نࣹ سࣹتْ نَ، ‏36 نَ ڪࣷ سَرْخَلْڪَتْ بِ دࣹلُّ سࣹيتْ بَ گِسْ نࣹ وَتْ-وَتَانْ جِ لَوْ نَ ݖِ يَرَمْ وِ؞ سُ بࣷوبَا سࣷخْلاَتُلْ مُ دِ ݖَ سࣹيتْ ڪَوَرْ گُ خَلّ، ندَخْ نِتْ ڪِ سࣷبࣹوُ نَ؞ ‏37 وَايࣹ سُ سࣹيتْلࣷو نࣹ وَتْ-وَتَانْ جِ سࣷݒِّڪُوُلْ تࣹ ڪَوَرْ گُ ݧُولْ سَخْ ݖَ، وَتْ-وَتَانْ جَ وࣺرْ نَ، نِتْ ڪِ اِتْ سࣹتْ نَ؞ نَ سَرْخَلْڪَتْ بِ بِرَلْ نࣹ نِتْ ڪِ سࣹتْ نَ؞
‏38 «سُ گٗورْ مبَا جِگࣺينْ اَمࣹي اَيْ گࣵڪّ-گࣵڪّ يُ وࣹيخْ ݖِ يَرَمْ وِ، ‏39 سَرْخَلْڪَتْ بِ سࣹيتْ ڪࣷ، گࣵڪّ يࣷويُ ݖِ يَرَمْ وِ دِ لُ وࣹيخْ، خَوَ خَلّ، خَمْ رࣹڪّ لَ، جُ فࣹيݧْ ݖِ يَرَمْ وِ وَايࣹ نِتْ ڪِ سࣹتْ نَ؞
‏40 «گٗورْ گُ بࣷݒّ بَ اَمَتُلْ ڪَوَرْ دَفَ لࣹيلْ وَايࣹ سࣹتْ نَ؞ ‏41 بُ بࣷݒّ بِ ݧࣵڪّࣹي ڪَوَرْ فِ فࣺيتࣹيڪْ ڪَنَمْ گِ، جࣴ بَا لࣹيلْ وَايࣹ سࣹتْ نَ؞ ‏42 وَايࣹ سُ اَمࣹي گࣵڪّ بُ وࣹيخْ بُ فࣹيݧْ تࣹ رَخْ خࣷنقّ ݖِ لࣹيلُبْ ندَالْ بࣷݒّ مبَا جࣴ، جࣵنگࣷرࣷيْ دࣹرْ لَ جُيْ مࣵگّ تࣹيْ گࣺنّ ݖِ لࣹيلُبْ ندَالْ بࣷݒّ مبَا جࣴ؞ ‏43 سَرْخَلْڪَتْ بِ دَ ڪࣷيْ سࣹيتْ؞ ندࣹگَمْ اَيْ تࣴرْگࣴنْ يُ وࣹيخْ تࣹ رَخْ خࣷنقَ نگِ ݖِ لࣹيلُبْ ندَالْ بࣷݒّ بِ مبَا جࣴ بِ، مُ مࣹلْ نِ جࣵنگࣷرࣷيْ دࣹرْ جُيْ لَوْ ݖِ يَرَمْ وِ، ‏44 نِتْ ڪِ اࣵندّ نَاڪْ جࣵنگࣷرࣷ تࣹ تَقْ نَ سࣷبࣹ؞ نَ سَرْخَلْڪَتْ بِ بِرَلْ ݖِ لُ لࣹيرْ نࣹ نِتْ ڪِ سࣷبࣹوُ نَ ندَخْ جࣵنگࣷرࣷ جِ ݖِ بࣷݒّ بِ؞
‏45 «ڪُ اࣵندّ اَڪْ جࣵنگࣷرࣷيْ دࣹرْ جُيْ لَوْ نَگْ يࣺرࣹ يُ خࣷتِّڪُ لَيْ سࣷلْ تࣹ دَيْ نجَݧُ¤؞ نَ مُورْ لِ ڪࣷ دَلࣹ ݖِ تُݧُمْ ڪَوْ مِ بَ ݖِ سِڪِّمْ بِ، تࣹ نَيْ جࣵݒّࣷو اࣵدُّ ݖَ ڪَوْ نَانْ:‏ ‹سࣹتُمَ دࣹ، سࣹتُمَ!‏› ‏46 لِ فࣹيڪْ جࣵنگࣷرࣷ جَا نگِ ݖِ مࣷومْ، دَيْ سࣷبࣹوُ تࣹ گَنَّاوْ دَا سࣷبࣹوُ، نَ دࣴڪّ مࣷومْ دࣷݝّ تࣹ نَ دࣴڪُّوَايَمْ گࣺنّ دَلْ بِ؞
لِ وَرْ ݖِ ڪࣴفْ ڪُ تࣹگْ لِيرْ
‏47 «سُ ندِمࣷ خُورࣹي، تࣹگْ اَوْ لِيرْ، مُيْ ندِمࣷلْ ڪَوَرُ گࣵتّ گُ ݧُ اࣴݖّ مبَا يُ وࣴݧْ گُ اࣴݖُّ¤، ‏48 مُ تࣹگُ ݖِ بࣵگّ مبَا ݖِ ݒࣷقࣹ، سُ دࣹي ڪَوَرْ گُ اࣴݖُّ، سُ دࣹي وࣴݧْ گُ اࣴݖُّ وَلَّ دࣹرْ بُ ݧُ وُلِّ وَلَّ مبࣷولࣹمْ لُ ݧُ دࣹفَرࣹ دࣹرْ، ‏49 سُ لِيرْ وِ نࣴتࣴخࣹي مبَا مُ خࣷنقّ، تࣹگࣹ ݖِ يࣺرࣹ مبَا دࣹرْ مبَا بࣵگّ مبَا ݒࣷقࣹ مبَا لُ ݧُ دࣹفَرࣹ دࣹرْ، لِيرْ وُيْ لَوْ لَ؞ دࣹيسْ ڪࣷيْ وࣷنْ سَرْخَلْڪَتْ بِ؞ ‏50 نَ سَرْخَلْڪَتْ بِ نࣹمِّڪُ لِيرْ وِ، بَ نࣷݒِّ بࣹرْ ڪࣴفْ ڪِ تࣹگْ لِيرْ وِ دِيرُ جُرٗومْ ݧَارِ فَنْ؞ ‏51 بࣺسُبْ جُرٗومْ ݧَارࣹيلْ بَ نَ سࣹيتْ لِيرْ وَ؞ سُ لِيرْ وَ لَوࣹي ݖَ ڪَوْ ڪࣴفْ ڪَ، مُيْ ندِمࣷ مبَا بࣵگّ مبَا ݒࣷقࣹ مبَا دࣹرْ مبَا جُمْتُڪَايُ دࣹرْ، ڪࣷنْ لِيرْ وُ آيْتَلْ لَ تࣹ ڪࣴفْ ڪَ سࣷبࣹوُ نَ؞ ‏52 دَ ڪࣷيْ لَڪّ، ندَخْ لِيرْ وُ آيْتَلْ لَ، مُيْ ندِمࣷ مبَا بࣵگّ مبَا ݒࣷقࣹ مبَا ڪَوَرُ گࣵتّ گُ ݧُ اࣴݖّ مبَا وࣴݧْ گُ اࣴݖُّ مبَا لُ ݧُ دࣹفَرࣹ دࣹرْ تࣹ لِيرْ وَ تࣹگُ ݖَ؞ دࣹيسْ ڪࣷيْ لَڪّ، بَ مُ دِبْ دٗومْ؞ ‏53 وَايࣹ ڪَتْ بُ سَرْخَلْڪَتْ بِ سࣹيتْلࣷو نࣹ لِيرْ وِ لَوُلْ ݖِ ڪࣴفْ ڪِ، مُيْ ندِمࣷ مبَا بࣵگّ مبَا ݒࣷقࣹ مبَا جُمْتُڪَايُ دࣹرْ، ‏54 نَ سَنْتَانࣹ ݧُ فٗوتْ ڪࣷ، مُ دَلْدِ ڪࣷيْ بࣹرَاتْ جُرٗومْ ݧَارِ فَنْ؞ ‏55 ندࣹگَمْ نَگْ گَنَّاوْ بَ ݧُ فٗوتࣹي ڪࣴفْ ڪَ تࣹگْ لِيرْ، سَرْخَلْڪَتْ بِ سࣹيتْ نَ ڪࣷ، تࣹ لِيرْ وِ سࣷݒِّوُلْ مࣹلࣷ، بُ لَوُلْ اِتْ سࣹتُلْ؞ دࣹيسْ ڪࣷيْ لَڪّ، بَ مُ دِبْ دٗومْ، سُ يࣵقࣷو بِيرْ اَڪْ سُ يࣵقࣷو بِتِ يࣺݒّ؞ ‏56 سُ سَرْخَلْڪَتْ بِ گِسࣹي گَنَّاوْ بَ ݧُ فٗوتࣹي ڪࣴفْ ڪَ نࣹ، لِيرْ وِ رَافْ نَ، نَ دَگّ سࣵنِّ فَ تࣹگࣷونْ لِيرْ ݖَ ڪࣴفْ ڪَ، مُيْ ندِمࣷ مبَا دࣹرْ مبَا بࣵگّ مبَا ݒࣷقࣹ؞ ‏57 سُ لِيرْ وَ گࣺنَّاتࣹي ݖَ ڪࣴفْ ڪَ، مُيْ ندِمࣷ مبَا بࣵگّ مبَا ݒࣷقࣹ مبَا لُ ݧُ دࣹفَرࣹ دࣹرْ، ڪࣷنْ لِيرْ وُيْ مࣵگّ لَ؞ ڪࣴفْ ڪَ تࣹگْ لِيرْ دࣹيسْ ڪࣷيْ لَڪّ، بَ مُ دِبْ دٗومْ؞ ‏58 ندࣹگَمْ لِيرْ وَ دࣹݧْ نَ، گَنَّاوْ بَ ݧُ فٗوتࣹي ڪࣴفْ ڪَ، مُيْ ندِمࣷ مبَا بࣵگّ مبَا ݒࣷقࣹ مبَا لُ ݧُ دࣹفَرࣹ دࣹرْ، دࣹيسْ ڪࣷيْ فٗوتَاتْ، مُ دࣷورَ سࣹتْ؞»
‏59 لࣷولُ لَ يࣷونْ دِگْلࣹ ݖِ لِيرْ وُيْ لَوْ وُ تࣹگُ ݖِ ندِمࣷلْ ڪَوَرْ گُ اࣴݖُّ مبَا لُ وࣴݧْ گُ اࣴݖُّ مبَا بࣵگّ مبَا ݒࣷقࣹ مبَا لُ ݧُ دࣹفَرࣹ دࣹرْ؞ ݖِ دِگْلࣹ يࣷويُ لَݧُيْ تࣹگُ، بَ مَنَ بِرَلْ نࣹ ڪࣴفْ ڪَ سࣹتْ نَ اَمْ سࣹتُلْ؞
