ندِگَلْ يِ جࣴمْ ݖِ سࣹلَّلْ
(سَارْ 17—26)
لِ وَرْ ݖِ جَامُ اَڪْ سَرْخَلْ
17
‏1 بُ لࣷولُ وࣺيࣹي اَجِ سَخْ جِ وَخْ نَ مُوسَىٰ نࣹ ڪࣷ:‏ ‏2 «وَخَلْ هَـٰرُونَ اَڪِ دࣷومَمْ يُ گٗورْ اَڪْ بࣵنِّ إِسْرَاءِيلْ گࣺݒّ نࣹ لࣹينْ لِ اَجِ سَخْ جِ سَنْتَانࣹ مࣷويْ لِي:‏ ‏3 ڪࣺݒّ ڪُيْ گٗورْ ݖِ وَا ڪࣴرْ إِسْرَاءِيلْ، بُ رࣹنْدࣹي اَوْ نَگْ مبَا خَرْ مبَا بࣴيْ ݖِ بِيرْ دَلْ بِ، اَڪْ فُ گࣺنّ دَلْ بِ، ‏4 تࣹ اِنْدِوُ ڪࣷ ݖِ بُنتّ خَيْمَبْ ندَجࣹ مِ، تࣹيوَلَلْ ڪࣷ اَجِ سَخْ جِ، مُيْ سَرَخْ ݖَ بُنتّ جَامࣷوڪَايُ اَجِ سَخْ جِ، ڪࣷوڪُ دࣹيسْ نَ ڪࣷ تࣷݒّ اࣵقُ دࣹرࣹتْ؞ دَفَ تُورْ دࣹرࣹتْ؞ ڪࣷنْ دࣹيسْ نَ ڪࣷ دَگّࣹ ݖِ بِيرْ بࣵنِّ إِسْرَاءِيلْ؞ ‏5 سُ ڪࣷ دࣹفࣹي بࣵنِّ إِسْرَاءِيلْ دِنَݧُ اِنْدِلْ اَجِ سَخْ جِ جُرِ سَرَخْ يِ ݧُيْ رࣹنْدِ ݖِ اࣵلّ بِ؞ دَݧُ ڪࣷو وَرَ يٗبُّ ݖَ سَرْخَلْڪَتْ بَ، فِ اَجِ سَخْ جِ، ݖَ بُنتّ خَيْمَبْ ندَجࣹ مَ، دࣹفَلْ ڪࣷ اَجِ سَخْ جِ سَرَخِ ݖَنتّ ݖِ بِيرْ جࣵمّ؞ ‏6 سَرْخَلْڪَتْ بِ دَيْ خࣴݒّ دࣹرࣹتْ جِ ݖِ سَرْخَلُڪَايُ اَجِ سَخْ جِ، ݖِ بُنتّ خَيْمَبْ ندَجࣹ مِ، بَ نࣷݒِّ، لَڪَّلْ اَجِ سَخْ جِ نࣹبّࣷنْ بِ، مُيْ خࣹيݧْ خࣹتُگْ جࣵمّ؞ ‏7 بُݧُ دِ سَرْخَلَتِلْ تُورْ، يِ ݧُيْ جࣴمَّلࣹ سِڪّࣹتْ، دِ ڪࣷ بࣷڪَّالࣹي؞ نَ لࣷولُ دِ دࣷگَلْ بُ لࣹينْ وَرْ فࣵوّ، ݧࣷومْ اَڪْ سࣹينْ اَسْڪَنْ؞ ‏8 وَخْ لࣹينْ بࣷوگْ نࣹ لࣹينْ:‏ ڪࣺݒّ ڪُيْ گٗورْ، ڪُ بࣷڪّ ݖِ وَا ڪࣴرْ إِسْرَاءِيلْ، مبَا مُ بࣷڪّ ݖِ دࣷخَنْدࣺيمْ يِ ݖِ سࣹينْ بِيرْ، بُيْ گࣺنّࣹ اَبْ سَرَخُ رࣹنْدِ-دٗومَلْ مبَا بࣹنࣹينْ سَرَخْ، ‏9 تࣹ اِنْدِوُ ڪࣷ ݖِ بُنتّ خَيْمَبْ ندَجࣹ مِ، سَرْخَلَلْ ڪࣷ اَجِ سَخْ جِ، ڪࣷوڪُ دࣹيسْ ڪࣷو وَرَ دَگّࣹ ݖِ بِيرْ بࣵنِّ إِسْرَاءِيلْ؞
‏10 «تࣹ اِتْ ڪࣺݒّ ڪُ بࣷڪّ ݖِ وَا ڪࣴرْ إِسْرَاءِيلْ مبَا دࣷخَنْدࣺيمْ يِ ݖِ سࣹينْ بِيرْ، بُ لࣹڪّࣹي لࣹنّ لُيْ دࣹرࣹتْ، ڪࣷوڪُ لࣹڪّ دࣹرࣹتْ، مَنْ اَجِ سَخْ جِ، دِنَا ڪࣷ نࣷونࣷو تࣹ دِنَا ڪࣷ دَگّࣹ ݖِ بِيرْ بࣵنِّ إِسْرَاءِيلْ؞ ‏11 ندَخْتࣹ دُنْدُ مبِنْدࣹيفْ ݖِ دࣹرࣹتْ لَ نࣹڪّ، تࣹ مَنْ مَا لࣹينْ سࣺدّࣹ دࣹرࣹتْ، نگࣹينْ دِ ڪࣷ جࣷتࣷو سࣹينْ بَڪَّنْ ݖِ ڪَوْ سَرْخَلُڪَايْ بِ؞ ندَخْ ڪَتْ بَڪَّنُ مبِنْدࣹيفْ دࣹرࣹتَمْ لَ تَخَوࣹ؞ مࣷو تَخْ دࣹرࣹتْ مَنَ دࣹفَلْ نِتْ نجࣷتْلاَيَمْ؞ ‏12 لࣷولࣷو وَرَلْ مَ نࣹ بࣵنِّ إِسْرَاءِيلْ، بُ ڪࣹنّ ݖِ ݧࣷومْ لࣹڪّ دࣹرࣹتْ، تࣹ دࣷخَنْدࣺيمْ بِ ݖِ سࣹينْ بِيرْ اِتْ بُمُ لࣹڪّ دࣹرࣹتْ؞
‏13 «ڪࣺݒّ ڪُ بࣷڪّ ݖِ بࣵنِّ إِسْرَاءِيلْ مبَا ݖِ دࣷخَنْدࣺيمْ يِ ݖِ سࣹينْ بِيرْ، بُ رࣴبّࣹي بَ جࣵݒّ مَلَ مبَا نجَنَاوْ، مُ دِ لُ دَگَنَ لࣹڪّ، نَ تُورْ دࣹرࣹتْ جِ، سُولْ، ‏14 ندَخْ دُنْدُ مبࣷولࣹمْ مبِنْدࣹيفْ مࣷويْ دࣹرࣹتَمْ؞ مࣷو وَرَلْ، مَنْ اَجِ سَخْ جِ، مَ نࣹ بࣵنِّ إِسْرَاءِيلْ بُݧُ لࣹڪّ دࣹرࣹتُ مࣹنّ مبِنْدࣹيفْ نگِرْ دُنْدُ مبࣷولࣹمْ مبِنْدࣹيفْ مࣷويْ دࣹرࣹتَمْ؞ ڪࣺݒّ ڪُ لࣹڪّ دࣹرࣹتْ، دࣹيسْ ڪࣷيْ دَگّࣹ ݖِ اَسْڪَنْ وِ؞
‏15 «تࣹ اِتَمْ ڪࣺݒّ ڪُ لࣹڪّ مࣺدّ مبَا لُ رَبْ فࣵدّ، مُيْ نجُدُّ-جِ-رࣺيوْ مبَا دࣷخَنْدࣺيمْ، نَ فٗوتْ اَيْ يࣺرࣹيمْ، سَنگُ، تࣹ دُ تࣹرࣹ مُ يࣹنْدࣷو سࣷبࣹ بَ جَنتّ سࣷ، مُ دࣷورَ سࣹتْ؞ ‏16 وَايࣹ سُ فٗوتُلْ، سَنگُوُلْ، مࣷويْ گࣵدُّ بࣵڪَّارَمْ؞»
