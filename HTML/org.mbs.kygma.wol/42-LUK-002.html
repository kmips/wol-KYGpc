عِيسَىٰ جُدُّ نَ ݖَ بࣹتْلࣹيࣹمْ
2
‏1 جَنتّ يࣷويُ ندِگَلْ مࣷو تُڪّࣹي وࣷونْ ݖِ بُورْ بُ مَگْ بُ ݧُيْ وَخْ سࣹسَارْ اࣷگُستّ؞ مُ نࣹ نَ ݧࣺݒّ بِنْدُجِ ݖَ لِمࣹيفْ يَ؞ ‏2 مبِنْدُ مࣷومَ مࣷويْ مُ جࣴڪّ مَ، يࣹمࣷوڪْ بَ ڪِرِݧُسْ دࣹي بࣷرࣷومْ رࣺيوُمْ سِرِ؞ ‏3 ݖِ ڪَوْ لࣷولُ ݧࣺݒّ بِنْدُجِ، ڪُ نࣹڪّ ݖَ سَ دࣴڪُّ ݖࣷسَانْ؞ ‏4 ڪُ ݧُيْ وَخْ يُوسُفَ اِتَمْ جٗگࣹ نَسَرࣹتْ ݖَ دِيوَانُ گَلِلࣹ، جࣴمْ ݖَ دِيوَانُ يُدࣹ ݖَ بࣹتْلࣹيࣹمْ، دࣴڪّ بَ دَاوُدَ ݖࣷسَانࣷو، ندَخْ ݖِ وَا ڪࣴرْ دَاوُدَ لَ اَسْڪَنَمْ بࣷڪّ؞ ‏5 مَرْيَامَ مِ دِگࣷوڪْ مࣷومْ سࣺيْ لَ اࣵنْدَلْ بِنْدُجِ، تࣹ فࣹڪّ ڪࣷ مُ اࣴمبّ؞ ‏6 نَڪَ لَݧُ دَلْ بࣹتْلࣹيࣹمْ، وَخْتُوْ مُݖَّمْ اَگْسِ؞ ‏7 مُ اَمْ دࣷومْ جُ گٗورْ، مُ دِبْ تَاوَمْ؞ مُ لَخَسْ ڪࣷ، تࣴرَلْ ڪࣷ ݖَ گࣵمْبُگْ جُرْ گَ¤، ندَخْ ݧࣵڪّ فُ ݧُ خَجْ ݖَ نࣺيگُ گَنْ بَ؞
مَلاَڪَ فࣹيݧُ نَ اَيْ سࣵمّ
‏8 بࣷوبَ اَيْ سࣵمَّ نگَ ݖَ گࣷخْ بࣷوبࣹي، دِ فَنَانࣹي وَتُّ سࣹينُگْ جُرْ؞ ‏9 اَمْ مَلاَڪَمْ بࣷرࣷومْ بِ جࣹڪِّ فࣹيݧُ لࣹينْ، لࣹيرُ بࣷرࣷومْ بِ مࣹلَخْ، وࣴرْ لࣹينْ، ݧُ تِيتْ تِيتَانگࣹ لُ رࣺيْ؞ ‏10 مَلاَڪَ مِ نࣹ لࣹينْ:‏ «بُلࣹينْ تِيتْ، ندَخْ ڪَتْ مَا لࣹينْ اِنْدِلْ خِبَارُ جࣵمّ بُ دࣷونْ مبࣺگْ مُ رࣺيْ، ݧࣹيلْ اَسْڪَنْ وࣺݒّ؞ ‏11 بࣺسُبْ تࣹيْ جِي ݖَ دࣴڪُّ دَاوُدَ، جُدُّلࣹ نگࣹينْ اَبْ مُسَلْڪَتْ، تࣹ مࣷويْ اَلْمَسِ، دِ سَنگّ بِ؞ ‏12 لِ نگࣹينْ ݖِيْ دࣹفْ اَبْ تَڪّ مࣷو دِ:‏ دِنگࣹينْ گِسْ لِيرْ بُ ݧُ لَخَسْ، تࣴرَلْ ڪࣷ ݖِ گࣵمْبُگْ جُرْ؞»
‏13 ݖَ سَا سَ گࣵنگࣷورُ اَسَمَانْ گُ مَگْ اࣵندّ اَڪْ مَلاَڪَ مِ، دِ سࣵبَّالْ يࣵلَّ، نࣹ:‏
‏14 «دَرَجَ ݧࣹيلْ نَ يࣵلَّ فَ اَسَمَانْ سَ گࣴنَ ڪَوࣹ،
جࣵمّ ݖِ ڪَوْ سُوفْ ݧࣹيلْ ݧِ يࣵلَّ بَاخࣹ؞»
‏15 بَ مَلاَڪَ يَ بࣵيِّڪࣷو ݖَ سࣵمّ سَ، دࣹلُّ اَسَمَانْ، ݧُيْ وَخْ ݖَ سࣹينْ بِيرْ، نَانْ:‏ «نَنُ دࣹمْ بࣷوگْ بࣹتْلࣹيࣹمْ، سࣹيتِ لَ فَ خࣹوْ، تࣹ بࣷرࣷومْ بِ خَمَلْ نُ ڪࣷ؞»
‏16 ݧُ گَاوْ دࣹمْ بَ فࣷوفَ، دَلْدِ گِسْ مَرْيَامَ اَڪْ يُوسُفَ، اَڪْ لِيرْ بَ تࣴدّ ݖَ گࣵمْبُگْ جُرْ گَ؞ ‏17 نَڪَ لَݧُ لࣹينْ گِسْ، دَلْدِ لࣹينْ خَمَلْ لَ ݧُ لࣹينْ وَخْ ݖَ مبِرُمْ خَلࣹ بَ؞ ‏18 مبࣷولࣹمْ ݧَ دࣺگّ لَ لࣹينْ سࣵمّ سَ وَخْ نَگْ وَارُ لࣷولْ؞ ‏19 تࣹيوُلْ مَرْيَامَ مࣷومْ، بࣵيِّ وࣷونْ نَ خࣹلْ ݖِ مبِرْ مࣷومُ مࣺݒّ، دِ ڪࣷ خَلاَتْ ݖِ خࣷلَمْ؞ ‏20 بَ لࣷولُ اَمࣹي سࣵمّ سَ دࣹلُّ، دِ سࣵبَّالْ اَڪَ سَنتّ يࣵلَّ ݖِ مبࣷولࣹمْ لَ ݧُ دࣺگّ، گِسْ ڪࣷ، تࣹ لࣺݒّ اَمࣹ نࣷونࣹي ݧُ لࣹينْ ڪࣷ وَخࣹ وࣷونْ؞
يٗبُّ نَݧُ عِيسَىٰ ݖَ ڪࣴرْ يࣵلَّ گَ ݖَ يࣹرُسَلࣹمْ
‏21 بَ جُرٗومْ ݧࣹتِّ فَنِ خَلࣹ بِ مَتࣹي، ݧُ خَرْفَلْ ڪࣷ، تُدّࣹ ڪࣷ عِيسَىٰ، نَ ڪࣷ مَلاَڪَ مَ تُدّࣹ وࣷونْ، بَلَ مࣷو سࣷسُ؞
‏22 گَنَّاوْ بَ سࣹينْ فَنِ سࣹتْلُ¤ مَتࣹي، نِ ڪࣷ يࣷونُ مُوسَىٰ لاَجࣹ، ݧُ يٗبُّ خَلࣹ بِ يࣹرُسَلࣹمْ، نگِرْ تࣹيوَلْ ڪࣷ فِ ڪَنَمْ بࣷرࣷومْ بِ، ‏23 نࣷونࣹي ݧُ ڪࣷ بِنْدࣹ ݖِ يࣷونُ بࣷرࣷومْ بِ نࣹ:‏ «بࣺݒّ تَاوْ بُ گٗورْ، دࣹيسْ ڪࣷيْ تُدّࣹ سࣺدّࣷو بُ سࣹلّ، ݧࣹيلْ بࣷرࣷومْ بِ؞» ‏24 ݧُ يٗبَّالࣹ سَرَخْ بِ ݖِ وَرْ، نِ ڪࣷ يࣷونُ بࣷرࣷومْ بِ سَنْتَانࣹي:‏ «ݧَارِ ݒِتَخُ اࣵلّ مبَا ݧَارِ خَتِ يُ ندَوْ؞»
سِمࣹيࣷنْ سࣵبَّالْ نَ يࣵلَّ
‏25 اَمࣷونْ نَ نَگْ ݖَ يࣹرُسَلࣹمْ ڪُ ݧُيْ وَخْ سِمࣹيࣷنْ؛ ڪُ جُبࣷونْ، رَگَلْ يࣵلَّ، دِ سࣺينْتُ جَمࣷنࣷ جِ نࣹڪِّنُ بࣵنِّ إِسْرَاءِيلْ دِ اَمࣹ اَگْ تَنْ، تࣹ نࣷو گُ سࣹلّ گِ اࣵنْدࣷونْ اَڪْ مࣷومْ؞ ‏26 ندࣺييْ نَگْ تُڪّࣹي ݖِ نࣷو گُ سࣹلّ گِ، خَمَلْ ڪࣷ نࣹ ڪࣷ دُ دࣹي مُڪّ تࣹ گِسُلْ اَلْمَسِ بُ بࣷرࣷومْ بِ؞ ‏27 ݖِ ڪَوْ لࣷولُ مُ دࣹمْ بَ ݖَ بِيرْ ڪࣴرْ يࣵلَّ گَ ݖِ دࣷولࣹيْ نࣷو گُ سࣹلّ گِ؞ گَنَّاوْ لࣷولُ وَاجُرِ عِيسَىٰ اِنْدِ عِيسَىٰ نگِرْ دࣹفَلْ ڪࣷ لِ آدَيْ يࣷونْ وِ لاَجْ؞ ‏28 سِمࣹيࣷنْ اِتَمْ جࣴلْ خَلࣹ بِ، لࣹيوُ ڪࣷ، دَلْدِ سࣵبَّالْ يࣵلَّ، نࣹ:‏
‏29 «لࣺيگِ نَگْ بُورْ يࣵلَّ، يِوِلْ سَبْ جَامْ،
مُ دࣹمْ اَڪْ جࣵمّ، نࣷونࣹي نگَ ڪࣷ وَخࣹ وࣷونْ؞
‏30 ندࣹ سَايْ گࣴتْ تࣹگُ نَ ݖِ مُݖّ گِ نگَ دࣷگَلْ،
‏31 وَاجَلْ ڪࣷ فِ ڪَنَمْ اَسْڪَنْ يࣺݒّ،
‏32 مُ دِگْ لࣹيرْ، ݧࣹيلْ جَامْبُرْ ݧِ،
سَگَلْ إِسْرَاءِيلْ سَ ݧࣷݧْ؞»
‏33 بَايْ بِ اَڪْ ندࣹيْ جِ نَگْ يࣺيمُ ݖِ ڪࣵدُّ يࣷويُ ݧُ وَخْ ݖِ خَلࣹ بِ؞ ‏34 بَ لࣷولُ اَمࣹي سِمࣹيࣷنْ ݧَانَلْ لࣹينْ، بَ نࣷݒِّ نࣹ ندࣹيَمْ مَرْيَامَ:‏ «دَمَ نࣹ، ڪِي لَݧُ يࣹبَلْ نگِرْ سࣵنڪُتࣹيْ ݧُ بَرࣹ مبَا سࣹينُگْ لࣹقْلِڪُ ݖِ دِگّ إِسْرَاءِيلْ، نگِرْ اِتَمْ مُ دࣷونْ فِرْندࣹ جُ ݧُيْ گࣵنْتَلْ؞ ‏35 يَوْ نَگْ، سَبْ خࣷلْ لَ اَوْ نَقَرْ وُ مࣹلْ نِ سَامَرْ دِ خَرْ، بَ ݧُ بَرࣹ لَ سࣹينْ خَلاَتِ بِيرْ خࣷلْ دِ فࣹيݧْ؞»
آنَ سَنتّ نَ يࣵلَّ
‏36 جِگࣺينْ جُ ݧُيْ وَخْ آنَ، اَبْ يࣷنࣹنتّ لَ وࣷونْ ݖَ يࣹرُسَلࣹمْ، دِ دࣷومُ فَنُوࣹلْ مِ اَسْڪَنࣷو ݖِ گِيرُگْ اَسࣹرْ؞ مَگْ لَ وࣷونْ لࣷولْ؞ بَ مُيْ جَنقّ لَ سࣺيْ، نࣹڪّ اَڪْ جࣴڪّࣴرَمْ جُرٗومْ ݧَارِ اَتْ؞ ‏37 جࣴڪّࣴرْ جَ فَاتُ، مُ تࣷوگْ، سࣺيَاتُلْ، بَ بࣷوبَ مُ اَمࣹي جُرٗومْ ݧࣹتّ فُڪِّ اَتْ اَڪْ ݧࣹينتّ (84)؞ دَاوُلْ جٗگࣹ ݖَ ڪࣴرْ يࣵلَّ گَ، ڪࣷورْ اَڪْ جُلِّ لَ دَانْ جَامࣷو گُدِّ اَڪْ بࣴݖّࣴگْ؞ ‏38 آنَ نَگْ دِڪّ مࣷومْ اِتْ ݖَ وَخْتُ وࣷووَ، دَلْدِ دࣹلّࣷو يࣵلَّ نجُڪَّلْ؞ ݖِ ڪَوْ لࣷولُ مُيْ وَخْتَانࣹ مبِرُمْ خَلࣹ بِ اَڪْ مبࣷولࣹمْ ݧَ دࣷونْ سࣺينْتُ بࣺسُ گࣷرࣹيلُگْ يࣹرُسَلࣹمْ؞
‏39 ݖِ ڪَوْ لࣷولُ ݧُ مَتَلْ مبࣷولࣹمْ لُ يࣷونُ بࣷرࣷومْ بِ لاَجْ، دࣷورَ دࣹلُّ نَسَرࣹتْ، سࣹينْ دࣴڪّ بَ ݖَ دِيوَانُ گَلِلࣹ؞ ‏40 خَلࣹ بِ نَگْ دِ مࣵگّ، دِ دࣴگࣴرْ، فࣹيسࣹ اَمْ خࣹلْ، تࣹ يِوُ يࣵلَّ اࣵندّ اَڪْ مࣷومْ؞
عِيسَىٰ فࣹيݧْ نَ ݖَ ڪࣴرْ يࣵلَّ گَ
‏41 اَتْ مُ جࣷتَانْ نَگْ وَاجُرِ عِيسَىٰ دࣹمْ يࣹرُسَلࣹمْ، نگِرْ مࣵگَّلِ بࣺسُبْ مُݖّ بَ؞ ‏42 بَ عِيسَىٰ اَمࣹي فُڪِّ اَتْ اَڪْ ݧَارْ، ݧُ دࣹمْ مࣵگَّلِ نَ ݧُ ڪࣷ بَاخࣷو وࣷونْ؞ ‏43 بَ مࣵگَّلْ گَ تَسࣹي، ݧُ ݧِبِّ، نگٗورْ سَ عِيسَىٰ مࣷومْ دࣹسْ يࣹرُسَلࣹمْ، تࣹ اَيْ وَاجُرَمْ يࣺگُݧُ ڪࣷ؞ ‏44 ݧُ خَلاَتْ نࣹ، عِيسَىٰ دَا اࣵندّ اَڪْ ݧُ ݧُ بࣷڪَّلࣷونْ يࣷونْ؞ دࣷخْ نَݧُ بࣺسُبْ لࣴمّ، دࣷورْ ڪࣷو وࣴرِ ݖِ سࣹينِ بࣷڪّ اَڪْ سࣹينِ خَمࣹ، ‏45 تࣹ تَخُلْ ݧُ گِسْ ڪࣷ؞ ݧُ دَلْدِ دࣹلُّ يࣹرُسَلࣹمْ، سࣹيتِ ڪࣷ؞
‏46 ݖَ گَنَّاوْ اࣴلّࣴگْ سَ لَݧُ ڪࣷ فࣹڪّ ݖَ ڪࣴرْ يࣵلَّ گَ، مُ تࣷوگْ ݖَ دِگّ جࣵنگّلࣹڪَتْ يَ، دِ لࣹينْ دࣺگْلࣷوڪَ لاَجْ؞ ‏47 مبࣷولࣹمْ ݧَ ڪࣷ دࣺگّ يࣺيمُ ݖِ دࣺگِّنَمْ اَڪِ تࣷنْتَمْ؞ ‏48 اَيْ وَاجُرَمْ گِسْ ڪࣷ، وَارُ؞ يَايَمْ نࣹ ڪࣷ:‏ «دࣷومْ، لِ نگَ نُ دࣹفْ نَگْ؟ مَاڪْ سَ بَايَ نگِ جَاخْلࣹ، دِ لَ سࣹيتْ!‏» ‏49 مُ نࣹ لࣹينْ:‏ «لُ نگࣹينْ مَيْ سࣹيتࣹ؟ خَمُلࣹينْ نࣹ سَمَ ڪࣴرْ بَايْ لاَ وَرَ نࣹڪّ؟»
‏50 اَيْ وَاجُرَمْ ݧࣷومْ دࣺگُّݧُ ڪࣵدُّ گࣷوگُ مُ لࣹينْ وَخْ؞ ‏51 گَنَّاوْ لࣷولُ عِيسَىٰ اࣵندّ اَڪْ ݧࣷومْ، دࣹلُّ نَسَرࣹتْ، تࣹيْ سَخࣷو دࣺگَّلْ اَيْ وَاجُرَمْ؞ يَايَمْ نَگْ تَڪّ مبِرْ يࣷويُ يࣺݒّ ݖِ خࣷلَمْ؞ ‏52 عِيسَىٰ مࣷومْ، مَ نگَيْ مࣵگّ ݖِ دَيࣷ اَڪْ خࣹلْ مُ رَفࣹتْ، اَڪْ ݖࣺرُبْ يِوْ فِ يࣵلَّ اَڪْ نِتْ ݧِ؞
