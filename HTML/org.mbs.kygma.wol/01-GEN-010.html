اَسْڪَنُ دࣷومِ نُوحٍ
(سَارْ 10‏.1—11‏.9)
خࣹيتْ يِ سࣷسࣷو ݖِ دࣷومِ نُوحٍ
10
‏1 اَسْڪَنُ دࣷومِ نُوحٍ مࣷويْ لِي؞ گَنَّاوْ بَ مبࣴنّ مَ وࣺيࣹي، سࣹمْ اَڪْ خَمْ اَڪْ يَفࣹتْ اَمْ نَݧُ اَيْ دࣷومْ؞
لِ جࣴمْ ݖِ اَسْڪَنُ يَفࣹتْ
‏2 دࣷومِ يَفࣹتْ يُ گٗورْ ݧِي لَ:‏ گࣷمࣹيرْ اَڪْ مَگࣷگْ اَڪْ مَدࣺيْ اَڪْ يَوَانْ اَڪْ تُبَلْ اَڪْ مࣹسࣹگْ اَڪْ تِرَاسْ؞ ‏3 دࣷومِ گࣷمࣹيرْ يُ گٗورْ دِ اَسْڪࣹنَسْ، رِفَاتْ اَڪْ تࣷگَرْمَ؞ ‏4 دࣷومِ يَوَانْ يُ گٗورْ دِ اࣹلِسَ اَڪْ تَرْسِسْ اَڪْ ڪِتِمْ اَڪْ دࣷدَنِمْ؞ ‏5 ݧِ سࣷسࣷو ݖِ ݧࣷومْ ݧࣷو لَوْ ݖَ وࣵلاَّ گࣺيجْ، مࣹݧّ نَگْ اَيْ خࣹيتْ ݖِ سࣹينِ سُوفْ، خࣹيتْ وُ ݖِ نࣹڪّ اَڪْ لࣵڪَّمْ، دࣹفْ اِتْ اَيْ لࣵنگّ ݖِ سࣹينْ بِيرِ خࣹيتْ؞
لِ جࣴمْ ݖِ اَسْڪَنُ خَمْ
‏6 دࣷومِ خَمْ يُ گٗورْ ݧࣷو دِ ݧِي:‏ ڪُوسْ اَڪْ مِسْرَيِمْ اَڪْ ݒُوتْ اَڪْ ڪَنَانْ؞ ‏7 دࣷومِ ڪُوسْ يُ گٗورْ دِ سࣹبَ اَڪْ اَوِلَ اَڪْ سَبْتَ اَڪْ رَامَ اَڪْ سَبْتࣹڪَ؞ دࣷومِ رَامَ يُ گٗورْ دِ سَبَ اَڪْ دࣹدَنْ؞
‏8 ڪُوسْ مࣷومْ اِتَمْ جُرْ نَمْرٗوتْ، ڪِ جࣴڪّ دِ جࣵمْبَارْ جُ سِيوْ ݖِ ڪَوْ سُوفْ، ‏9 مُ دࣷونࣷونْ رࣴبّڪَتْ بُ مَگَ مَگْ؞ لࣷولࣷو وَرَلْ ݧُ نَانْ:‏ «مبࣹتࣹ نَمْرٗوتْ، رࣴبّڪَتْ بُ مَگَ مَگْ بَ؞» ‏10 دࣴڪّ يِيْ ݒࣺييِ نگُورَمْ ݧࣷو دِ:‏ بَبࣹلْ اَڪْ اࣹرࣹگْ اَڪْ اَڪࣵدّ اَڪْ ڪَلْنࣹ، گِ ݖِ رࣺيوُ سِنࣹيَارْ؞ ‏11‏-12 ݖَ رࣺيوْ مࣷومَ لَ نَمْرٗوتْ جٗگࣹ دࣹمْ اَسُرْ، سَنݖّ دࣴڪّ بُ مَگْ بࣷوبࣹي دِ نِنِوْ، اَڪْ دࣴڪّ يࣷويࣹي دِ رࣹيࣷبࣷتْ اِرْ اَڪْ ڪَلَ، بَ رࣹسࣹنْ، گِ ݖِ دِگَّنْتࣹ نِنِوْ اَڪْ ڪَلَ؞
‏13 مِسْرَيِمْ نَگْ مࣷويْ مَامِ لُدّࣹينْ ݧِ اَڪْ اَنَمࣹينْ ݧِ اَڪْ لࣹيَبࣹينْ ݧِ اَڪْ نَفْتُيࣹينْ ݧِ ‏14 اَڪْ ݒَتَرُسࣹينْ ݧِ اَڪْ ڪَسْلُيࣹينْ، ݧࣷوݧࣹي وَا فِلِسْتِ بࣵيِّڪࣷو ݖَ سࣹينْ رࣺيوْ؛ اَڪْ اِتْ ڪَفْتࣷرࣹينْ ݧِ؞
‏15 ڪَنَانْ مࣷو تَاوْلࣷو سِدࣷنْ، تࣹگْ ݖَ اࣹتّ، ‏16 تࣹ مࣷويْ مَامِ يࣹبُسࣹينْ ݧِ اَڪْ اَمࣷرࣹينْ ݧِ اَڪْ گِرْگَسࣹينْ ݧِ ‏17 اَڪْ اࣹوࣹينْ ݧِ اَڪْ اَرْڪࣹينْ ݧِ اَڪْ سِنࣹينْ ݧِ، ‏18 اَڪْ اَرْوࣵدّࣹينْ ݧِ اَڪْ سࣹمَرࣹينْ ݧِ اَڪْ اَمَتࣹينْ ݧِ؞ لَ ݖَ تࣹگُ لࣵنگِ ڪَنَانْ تَسَارࣷو؞ ‏19 ڪࣹمُ سُوفَسْ ڪَنَانࣹينْ ݧَ، مُ نگِ تࣵمْبَلࣹي سِدࣷنْ، جَارْ وࣹتِ گࣹرَرْ، دࣹمْ بَ گَسَ، وُتِ سࣷدࣷمْ اَڪْ گࣷمࣷرْ اَڪْ اَتْمَ اَڪْ ݖࣹبࣷيِمْ، دَلْدِ تࣹيرِ لࣹسَ؞
‏20 ݧࣷوݧُ ݧࣷويْ اَسْڪَنِ خَمْ، اَسْڪَنْ وُ ݖِ نࣹڪّ اَڪْ لࣵنگَمْ اَڪْ لࣵڪَّمْ، دِ اَيْ خࣹيتْ ݖِ سࣹينِ رࣺيوْ؞
لِ جࣴمْ ݖِ اَسْڪَنُ سࣹمْ
‏21 سࣹمْ، مِيْ مَگُ يَفࣹتْ، جُرْ نَ مࣷومْ اِتَمْ اَيْ دࣷومْ يُ گٗورْ؞ مࣷويْ مَامِ اࣹبࣹرْ اَڪْ اَسْڪَنَمْ؞ ‏22 دࣷومِ سࣹمْ يُ گٗورْ ݧِي لَ:‏ اࣹلَمْ اَڪْ اَسُرْ اَڪْ اَرْݒَگْسࣵدّ اَڪْ لُدّ اَڪْ اَرَمْ؞ ‏23 گٗورْ ݧِ اَرَمْ جُرْ ݧِي لَ:‏ اُوسْ اَڪْ خُولْ اَڪْ گࣹتࣹرْ اَڪْ مَاسْ؞ ‏24 اَرْݒَگْسࣵدّ جُرْ سࣹلَ، سࣹلَ جُرْ اࣹبࣹرْ؛ ‏25 اࣹبࣹرْ مࣷومْ، جُرْ ݧَارِ دࣷومْ يُ گٗورْ، ڪࣹنّ ڪِ تُدّ ݒࣹلࣹگْ (مُ فِرِ سࣺدّلِڪࣷو)، ندَخْتࣹ ݖِ جَمࣷنࣷومْ اࣵدِّنَ دَفَ سࣺدّلِڪࣷو وࣷونْ؞ رَڪُّ ݒࣹلࣹگْ مࣷويْ يࣷگْتَانْ؞
‏26 يࣷگْتَانَ جُرْ اَلَمࣷدࣵدّ اَڪْ سࣹلࣹفْ اَڪْ اَسَرْمَوࣹتْ اَڪْ يࣹرَ ‏27 اَڪْ اَدࣷرَمْ اَڪْ اُسَلْ اَڪْ دِگْلَ ‏28 اَڪْ اࣷبَلْ اَڪْ اَبِمَيࣹلْ اَڪْ سَبَ ‏29 اَڪْ اࣷفِرْ اَڪْ اَوِلَ اَڪْ يࣷبَبْ؞ ݧࣷوݧُ ݧࣺݒّ دِ دࣷومِ يࣷگْتَانْ يُ گٗورْ؞ ‏30 ݧَ نگَ دࣴڪّࣷونْ ݖَ وࣵلُّ مࣹسَ، گَ وُتِ سࣹفَرْ، دِ دِيوَانُ تُندّ يَ ݖَ ݒࣹنڪُ؞
‏31 ݧࣷوݧࣷو اَسْڪَنࣷو ݖِ سࣹمْ، اَسْڪَنْ وُ ݖِ نࣹڪّ اَڪْ لࣵنگَمْ اَڪْ لࣵڪَّمْ، دِ اَيْ خࣹيتْ ݖِ سࣹينِ رࣺيوْ؞
‏32 لࣵنگِ دࣷومِ نُوحٍ يُ گٗورَ نگࣷوگُ، لࣵنگّ گُ نࣹڪّ اَڪْ اَسْڪَنَمْ، ݧُ دِيْ خࣹيتْ؞ ݖِ ݧࣷومْ لَ خࣹيتْ يࣺݒّ جٗگࣹ تَسَارࣷو ݖِ ڪَوْ سُوفْ، گَنَّاوْ بَ مبࣴنّ مَ وࣺيࣹي؞
