خِبَارُ عِيسَىٰ يࣴنگَلْ نَ تࣹسَلࣷنِگْ
17
‏1 بَ مُ ڪࣷ دࣹفࣹي ݧُ جَارࣹ دࣴڪّ يَ ݧُيْ وَخْ اَمْفِݒࣷلِسْ اَڪْ اَݒࣷلࣷنِ، دࣷورَ اࣵگّ تࣹسَلࣷنِگْ، گَ يَوُتْ يَ اَمࣷونْ اَبْ جࣵنگُ؞ ‏2 ݒٗولْ نَگْ دِ لࣹينْ فَ فࣹڪّ، نَ مُ ڪࣷ بَاخࣷو؞ دِيرُ ݧࣹتِّ بࣺسِ نࣷفْلاَيْ، مُ دِ لࣹينْ وَخَلْ ݖِ مبِندّ مِ؞ ‏3 مُ دِ لࣹينْ تࣹڪِّلْ مبِندّ مِ، دِ لࣹينْ ݖِ فِرْندࣹيلْ نࣹ ݖࣷونࣷبْ اَلْمَسِ اَڪْ ندࣹڪِّتࣹيمْ مَنُلْ وࣷونَ ݧࣵڪّ؞ مُ نࣹ لࣹينْ:‏ «عِيسَىٰ مِي مَ دِ لࣹينْ خَمَلْ، مࣷويْ اَلْمَسِ؞» ‏4 بَ لࣷولُ اَمࣹي ݧࣹنّ ݖَ يَوُتْ يَ گࣴمْ، فࣹڪّسِ ݒٗولْ اَڪْ سِلَسْ، ݧࣷومْ اَڪْ گࣹرࣹگْ يُ بَرࣹ يُ رَگَلْ يࣵلَّ، اَڪْ جِگࣺينْ ݧُ رࣵݧِّڪُ ݧُ بَرࣹ؞
‏5 يَوُتْ يَ نَگْ اَمْ ݖَ ڪَݧَانْ، دَلْدِ فࣷرَاتࣷو ݖَ مبࣹدّ مَ اَيْ تَخَوَالُڪَتْ، دَجَلࣹ اَيْ گࣵنگࣷورْ، بَ يࣴنگَلْ دࣴڪّ بَ؞ ݧُ سࣷنگّ ڪࣴرْ يَسࣷنْ، دِ سࣹيتْ ݒٗولْ اَڪْ سِلَسْ، نگِرْ يٗبُّ لࣹينْ ݖَ وَا دࣴڪّ بَ؞ ‏6 ݖِ ڪَوْ لࣷولُ گِسُݧُ لࣹينْ، ݧُ دِرِ يَسࣷنْ اَڪْ يࣹنࣹينْ بࣷڪِّ گࣴمْڪَتْ بَ ݖَ ڪَنَمْ اࣵتّࣹڪَتْ يَ؞ ݧَ نگَيْ بِرَلْ، نَانْ:‏ «نِتْ ݧِ تَسْ اࣵدِّنَا نگِي، اَگْسِ نَݧُ فِ، ‏7 تࣹ يَسࣷنَ لࣹينْ دَلَلْ؞ ݧࣷومْ ݧࣺݒَّ خࣴتّ ندِگَلِ بُورْ سࣹسَارْ، تࣹ نَانْ اَمْ نَ بࣹنࣹينْ بُورْ بُ ݧُيْ وَخْ عِيسَىٰ؞» ‏8 بَ مبࣷولࣷو مَ اَڪْ اࣵتّࣹڪَتِ دࣴڪّ بَ دࣺگّࣹي لࣷولُ، دَݧࣷو جَاخْلࣹ؞ ‏9 بَ لࣷولُ اَمࣹي ݧُ فࣹيْلࣷو يَسࣷنْ اَڪْ ݧَ ݖَ دࣹسْ دَانْ بَ ݖَ وَرْ، دࣷورْ لࣹينَ يِوِ؞
ݒٗولْ اَڪْ سِلَسْ دࣹمْ نَݧُ بࣹرࣹ
‏10 بَ مُ ڪࣷ دࣹفࣹي بࣷڪِّ گࣴمْڪَتْ ݧَ يࣹبَلْ ݖَ سَا سَ ݒٗولْ اَڪْ سِلَسْ ݖَ گُدِّ گَ، ݧُ جࣴمْ بࣹرࣹ؞ ݧُ اَگْسِ، دࣹمْ ݖَ جࣵنگُبْ يَوُتْ يَ؞ ‏11 وَا بࣹرࣹ نَگْ گࣴنَ اَمْ دࣺگِّنْ وَا تࣹسَلࣷنِگْ؞ ݧُ گࣴمْ ڪࣵدُّ گِ، خࣺرْ ݖِ لࣷولْ، بࣺسْ بُ نࣹڪّ ݧُيْ نِيرْ مبِندّ مِ، نگِرْ خَمْ ندَخْ لَ ݧُ لࣹينْ دِ وَخْ نࣷونُ لَ؞ ‏12 ݖِ ڪَوْ لࣷولُ يَوُتْ يُ بَرࣹ گࣴمْ، ݧࣷومْ اَڪْ جِگࣺينِ گࣹرࣹگْ ݧُ رࣵݧِّڪُ، اَڪْ اَيْ گٗورِ گࣹرࣹگْ ݧُ بَرࣹ؞
‏13 بَ لࣷولُ اَمࣹي يَوُتِ تࣹسَلࣷنِگْ دࣺگّ نࣹ ݒٗولَ نگَيْ خَمْلࣹ ڪࣵدُّگْ يࣵلَّ ݖَ بࣹرࣹ اِتَمْ؞ ݧُ دِڪَّتِ، يࣴنگَلْ نِتْ ݧَ، ݧُ دࣹفْ اَيْ گࣵنگࣷورْ؞ ‏14 ݖَ لَ بࣷڪِّ گࣴمْڪَتْ ݧِ يࣹبَلْ ݒٗولْ ݖَ سَا سَ، مُ جࣴمْ تࣹفࣹسْ گَ، سِلَسْ اَڪْ تِمࣷتࣹ ݧࣷومْ، دࣹسْ ݖَ بࣹرࣹ؞ ‏15 ݧَ دَرْ ݒٗولْ نَگْ گُنگࣹ ڪࣷ بَ ݖَ دࣴڪّ بَ ݧُيْ وَخْ اَتࣹنْ، ݧُ دࣷورَ دࣴݒّ، جࣷتّلِجِ سِلَسْ اَڪْ تِمࣷتࣹ ڪࣵدُّگْ ݒٗولْ، نگِرْ ݧُ فࣹڪّسِ ڪࣷ نُ مُ گࣴنَ گَاوࣹ؞
ݒٗولْ وَارࣹ نَ ݖَ اَتࣹنْ
‏16 دِگَّنْتࣹ بَ ݒٗولْ دِ نࣺگْ سِلَسْ اَڪْ تِمࣷتࣹ ݖَ اَتࣹنْ، خࣷلَمْ جࣹيخْ نَ ندَخْ دَفَ گِسْ نَ دࣴڪّ بَ يࣹبࣷو ݖِ جَامُ اَيْ تُورْ؞ ‏17 ݖِ بِيرْ لࣷولُ مُيْ وَخْ اَڪْ يَوُتْ يَاڪْ جَامْبُرِ رَگَلْڪَتِ يࣵلَّ يَ ݖَ جࣵنگُ بَ، دِ وَخْ اِتْ اَڪْ وَا ݒࣺنݖّ مَ بࣺسْ بُ نࣹڪّ؞
‏18 ݖِ ڪَوْ لࣷولُ اَيْ بࣷرࣷومْ خَمْ-خَمْ يُ بࣷڪّ ݖِ نگࣺرُمْ اࣹݒِڪُرْ¤ اَڪْ نگࣺرُمْ اࣹسْتࣷوِسࣹنْ¤، دِ وࣹرَنْتࣹ اَڪْ مࣷومْ؞ ݧࣹنّ ݧَ نࣹ:‏ «ڪࣹبَتُڪَتْ بِي، لُ مُيْ وَخْ نِي؟» ݧَ ݖَ دࣹسْ دࣺگّ ݒٗولْ دِ وَارࣹ خِبَارُ جࣵمّ بِ ݖِ عِيسَىٰ اَڪْ ندࣹڪِّتࣹيمْ، ݧُ نࣹ:‏ «دَا مࣹلْ نِ اَيْ تُورْ-تُورَانِ دࣷخَنْدࣺيمْ لَيْ وَارࣹيلْ؞» ‏19 ݧُ جࣵݒّ ڪࣷ نَگْ، يٗبُّ ڪࣷ ݖَ اࣴتُّبْ اࣵتّࣹڪَايْ بَ ݧُيْ وَخْ اَرࣹيࣷݒَسْ؞ ݧُ نࣹ ڪࣷ:‏ «ندَخْ مَنْ نَنࣷو خَمْ نجࣵنگّلࣹ مُ بࣹيسْ مِي نگَيْ سِيوَلْ، لُ مُ دࣷونْ؟ ‏20 ندَخْ لِ نگَيْ وَخْ ڪَتْ سُنُ نگَنْ-گِ-نࣷݒّ لَ؞ نُ بࣴگَّ خَمْ نَگْ لࣷولُ لُ مُيْ تࣹڪِّ؞» ‏21 بࣷوبَ وَا اَتࣹنْ ݧࣺݒّ، دࣷخَنْدࣺيمْ اَڪْ نجُدُّ-جِ-رࣺيوْ، خِنْتࣹوُݧُ لࣹنّ لُ مࣷيْ دِ نࣹتَّلِ اَڪَ دࣺگْلُ خࣹوْ-خࣹوْ وُ بࣹيسْ؞
‏22 ݖِ ڪَوْ لࣷولُ ݒٗولْ تَخَوْ ݖَ دِگّ اࣴتُّبْ اَرࣹيࣷݒَسْ، نࣹ لࣹينْ:‏ «يࣹينْ وَا اَتࣹنْ، گِسْ نَا نࣹ ݖِ وࣹتْ گُ نࣹڪّ ݧُ فَرْلُ نگࣹينْ ݖِ دِينࣹ؞ ‏23 ندَخْ مَا دࣷونْ دࣷخَنْتُ، دِ سࣹيتْ سࣹينِ جَامࣷوڪَايْ، بَ يࣹمْ ݖِ بُ ݧُ بِندّ مِي مبِندّ:‏ ‹ݧࣹيلْ يࣵلَّ جِ نُ خَمُلْ؞› لࣷولُ نگࣹينْ دِ مࣵگَّلْ تࣹ خَمُلࣹينْ ڪࣷ نَگْ، مࣷومْ لاَ لࣹينْ دِ خَمَلْ؞
‏24 «يࣵلَّ مِ سࣵڪّ اࣵدِّنَ اَڪْ مبࣷولࣹمْ لِ ݖِ بِيرَمْ، دِ بࣷرࣷومْ اَسَمَانْ اَڪْ سُوفْ، دࣴڪّࣹوُلْ مࣵڪَّانْ مُ اَيْ لࣷخࣷ سࣵڪّ؞ ‏25 تࣹ اِتْ جَرِݧࣷووُلْ لࣷخࣷلْ نِتْ، مبࣹتࣹ ڪُ سࣷخْلَ لࣹنّ، ندَخْ مࣷويْ مَيْ ݧࣺݒّ بَڪَّنْ اَڪُگْ نࣷو، اَڪْ لُ مُ مَنَ دࣷونْ؞ ‏26 مࣷو سࣵڪّࣹ خࣹيتِ اࣵدِّنَ يࣺݒّ ݖِ ڪࣹنّ نِتْ، ݧُ دࣴڪّࣹ دࣺنْدُ سُوفْ سࣺݒّ، تࣹ مࣷو لࣹينْ اࣵݒَّلْ سࣹينِ جَمࣷنࣷ، رࣴدَّلْ لࣹينْ سࣹينْ ڪࣹمِ دࣴڪُّوَايْ؞ ‏27 يࣵلاَّ دࣹفْ لࣷولُ، نگِرْ خࣹيتِ اࣵدِّنَ وُتْ ڪࣷ، لࣵمْبَتُ بَ دَجْ ڪࣷ، ندࣹگَمْ مَنْ نَا اَمْ، دࣷونْتࣹ سࣷرࣹوُلْ ڪࣹنّ ڪُ نࣹڪّ ݖِ نُنْ؞ ‏28 ‹ندَخْ سُنُ بَڪَّنْ اَڪْ سُنُ يࣴنگُ اَڪْ سُنُگْ نࣹڪّ، لࣺݒّ ݖِ مࣷومْ لَ؞› مࣷويْ بَ تࣹيْ لَ ݧࣹنّ ݖِ سࣹينِ وࣷيْڪَتْ نࣹ:‏ ‹نُنْ اِتَمْ اَوْ خࣹيتَمْ لَنُ؞› ‏29 ندࣹگَمْ نَگْ خࣹيتُ يࣵلَّ لَنُ، وَرُنࣷو نِرُلࣹ يࣵلَّ اَڪْ لࣹنّ لُيْ وُرُسْ، مبَا خَالِسْ مبَا اَوْ دࣷجْ تࣹ سࣷسࣷو ݖِ خَرࣹݧْتࣹيْ نِتْ اَڪُمْ خࣹلَمْ؞ ‏30 يࣵلَّ نَگْ جࣺلَّلࣹ نَ جَنْتِ ݧࣵڪَّ خَمْ گࣷوگَ وࣷونْ، وَايࣹ تࣹيْ مُ نگِ وࣷو ݧࣺݒّ، اَڪْ فࣺݒّ فُ ݧُ فࣺيتࣹ، نگِرْ ݧُ تُوبْ؞ ‏31 ندَخْتࣹ يࣵلَّ جࣵݒّ نَ بࣺسْ بُ مُيْ اࣵتّࣹي اࣵدِّنَ ݖِ دࣴگّ، اࣵتّࣹ بُ نِتْ ڪِ مُ تَامُ دِ سࣷتَّلِ؛ تࣹ لࣷولُ يࣵلَّ جࣷخْ نَ ݖِ ݧࣺݒّ فِرْندࣹ لُ وٗورْ، بَ مُ دࣹڪَّلࣹي ڪࣷوڪُ مُ تَامُ، ندࣹڪِّتࣹ لُ دࣹي تࣷݒُّلْ؞»
‏32 نَڪَ لَݧُ دࣺگّ وَخِ ندࣹي-ندࣹڪِّ، ݧࣹنّ ݧَ دِ ڪࣷ فࣷونْتࣷو، ݧَ ݖَ دࣹسْ نࣹ ڪࣷ:‏ «دِنَنُ لَ دࣺگْلُ، نگَ وَخَلاَتْ نُ ݖِ مࣷومُ مبِرْ؞» ‏33 بَ لࣷولُ اَمࣹي ݒٗولْ بࣵيِّڪࣷو ݖَ سࣹينْ بِيرْ؞ ‏34 تࣹيوُلْ مُ اَمْ ݧُ ڪࣷ فࣹڪّسِ، بَ مُجّ گࣴمْ؛ دࣹنِسْ مِ بࣷڪّ ݖِ اࣴتُّبْ اَرࣹيࣷݒَسْ ݖِ لَ، اَڪْ جِگࣺينْ جُ ݧُيْ وَخْ دَمَرِسْ، اَڪْ ݧࣹنࣹينْ؞
