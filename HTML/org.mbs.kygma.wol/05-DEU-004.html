مُوسَىٰ ݧَاخْ نَ بࣵنِّ إِسْرَاءِيلْ
4
‏1 لࣺيگِ نَگْ يࣹينْ وَا إِسْرَاءِيلْ، دࣺگْلُلࣹينْ دࣷگَلِ يࣷونْ يࣹيڪْ اࣵتّࣹيْ يࣷونْ يِي مَ لࣹينْ دِ خَمَلْ، نگِرْ نگࣹينْ دِ ڪࣷ جࣴفࣹ، ندَخْ نگࣹينْ دُندّ بَ دُگّ، نَنگُجِ رࣺيوْ مَ لࣹينْ اَجِ سَخْ جِ سࣹينْ يࣵلَّيْ مَامْ، دِ جࣷخْ؞ ‏2 بُلࣹينْ يࣷڪّ دَرَ ݖِ ڪࣵدُّ گِ مَ لࣹينْ دِ دࣺنڪّ، تࣹ بُلࣹينْ ݖِ وࣵݧِّ دَرَ، ندَخْ نگࣹينْ سࣵمّ سࣹينْ سَنْتَانࣹيْ يࣵلَّ اَجِ سَخْ، جِ مَ لࣹينْ دِ دࣺنڪّ؞ ‏3 سࣹينْ بࣴتْ مࣷو لࣹينْ مَيْ لَ اَجِ سَخْ جِ دࣹفْ ݖَ بَالْ ݒࣹوࣷرْ؞ مبࣷولࣹمْ ڪُ تࣷݒّࣷونْ تُورْ مَ ݧُيْ وَخْ بَالْ ݖَ ݒࣹوࣷرْ، سࣹينْ يࣵلَّ اَجِ سَخْ جِ مࣷو ڪࣷ سࣵنڪࣹ ݖِ سࣹينْ بِيرْ؞ ‏4 وَايࣹ يࣹينْ ݧِ ݝࣷيࣷونْ ݖِ سࣹينْ يࣵلَّ اَجِ سَخْ جِ، يࣹينْ ݧࣺݒَّ نگِ دُندّ تࣹيْ جِي؞
‏5 مُ نگࣷوگْ، جࣵنگَلْ نَا لࣹينْ اَيْ دࣷگَلِ يࣷونْ، اَڪِ اࣵتّࣹيْ يࣷونْ، نَ مَ ڪࣷ سَمَ يࣵلَّ اَجِ سَخْ جِ سَنْتࣹ، نگِرْ نگࣹينْ وَرْ ڪࣷو جࣴفࣹ ݖَ بِيرْ رࣺيوْ مَ نگࣹينْ دِ نَنگُجِ؞ ‏6 سࣵمّࣷولࣹينْ ڪࣷو جࣴفࣹ، ندَخْ مࣷويْ دࣷونْ سࣹينُگْ رَفࣹتْ خࣹلْ، اَڪْ سࣹينُگْ رࣵݧّࣹي فِ ڪَنَمْ خࣹيتْ يِ، بَ بُ ݧُ دࣺگّࣹي مبࣷولࣹمْ دࣷگَلِ يࣷونْ يِي، دِنَݧُ نࣹ:‏ «وِي اَسْڪَنْ وُ مَگْ رࣹڪَّ رَفࣹتُمْ خࣹلْ تࣹ رࣵݧّࣹي!‏» ‏7 اَنَ وَنْ خࣹيتْ وُ مَگْ وُ اَمْ يࣵلَّ جُ لࣹينْ جࣹگࣹ نِ نُ سُنُ يࣵلَّ اَجِ سَخْ جِيْ جࣹگࣹي سَا يُ نُ ڪࣷ وࣷويࣹي وَلّ؟ ‏8 اَنَ خࣹيتْ وُ ڪࣷ مࣵگّࣹ، بَ اَمْ دࣷگَلِ يࣷونْ اَڪْ اࣵتّࣹيْ يࣷونْ يُ جࣹڪّࣹ نِ يࣷونْ وِي مَ جࣷخࣹ فِ سࣹينْ ڪَنَمْ بࣺسُبْ تࣹيْ؟
‏9 وَايࣹ وَتُّلࣹينْ، وَتُّلࣹينْ سࣹينْ بࣷݒّ بُ بَاخْ، بَلَ نگࣹينَ فࣵتّࣹ مبِرْ يِ نگࣹينْ تࣹگْ سࣹينْ بࣴتْ، بَ دُ گࣺنّ سࣹينُمْ خࣹلْ سࣹينْ گِيرُ دُندّ گࣺݒّ؞ تࣹ نَنگࣹينْ ڪࣷ خَمَلْ سࣹينِ دࣷومْ اَڪْ سࣹينِ سࣴتْ؞ ‏10 بࣺسْ بَ نگࣹينْ تَخَوࣹي فَ سࣹينْ ڪَنَمْ يࣵلَّ اَجِ سَخْ جِ، فَ تُنْدُ اࣷرࣹبْ، اَجِ سَخْ جࣹي مَ نࣹ:‏ «دَجَلࣹيلْ مَ مبࣷولࣷو مِ، مَ دࣺگْتَلْ لࣹينْ سَمَيْ ڪࣵدُّ، نگِرْ ݧُ تࣵگَّتُ ݖِ رَگَلْ مَ لِ فࣹيڪْ ݧُيْ دُندّ ݖِ ڪَوْ سُوفْ، تࣹ سࣹينِ دࣷومْ اِتْ، ݧُ تࣵگَّتْ لࣹينْ ݖِ؞» ‏11 يࣹينَ دِڪّ تَخَوْ ݖَ سُوفُ تُندّ وَ، تُندّ وَ نࣹ جِݒّࣺتْ، سَوَرَ وَ جࣷلِّ بَ ݖَ خࣷلُ اَسَمَانْ، ݖِ بِيرْ خࣵمْبَارِ گُدِّ گُ نࣹ ڪࣴرُوسْ؞ ‏12 اَجِ سَخْ جِ اࣵدّࣷو ݖَ دِگّ سَوَرَ وَ، وَخْ اَڪْ يࣹينْ، نگࣹينْ دِ دࣺگّ بَاتْ، تࣹ جࣴمّ، گِسُلࣹينْ ڪࣷ، خَنَا بَاتْ دࣷݝّ؞ ‏13 ݖِ ڪَوْ لࣷولُ مُ بِرَلَلْ لࣹينْ ڪٗلّࣴرࣹيمْ گِ مُ لࣹينْ سَنتّ نگࣹينْ سࣵمّ ڪࣷ، مُيْ فُڪِّ سَنْتَانࣹ يِ؞ مُ دَلْدِ لࣹينْ بِنْدَلْ سَنْتَانࣹ يِ ݖِ ݧَارِ اࣵلُّوَيْ دࣷجْ؞ ‏14 مَنْ نَگْ لَ اَجِ سَخْ جِ سَنتّ ݖَ يࣷويَ جَنتّ، نࣹ مَ خَمَلْ لࣹينْ دࣷگَلِ يࣷونْ يࣹيڪْ اࣵتّࣹيْ يࣷونْ، يِ نگࣹينْ وَرَ جࣴفࣹ ݖَ رࣺيوْ مَ نگࣹينْ جࣴمْ، نگِرْ نَنگُجِ ڪࣷ؞
يࣷونْ وِ آيَلْ نَ بࣷڪَّالࣹ
‏15 نَنگࣹينْ وَتُّ سࣹينْ بࣷݒّ بُ بَاخْ نَگْ ندَخْ ڪࣹرࣷوگْ بَ اَجِ سَخْ جِيْ اࣵدّࣷو ݖَ دِگّ سَوَرَ وَ، وَخْ اَڪْ يࣹينْ ݖَ تُنْدُ اࣷرࣹبْ، گِسُلࣹينْ جࣹنّ جࣴمّ؞ ‏16 ڪࣷنْ بُلࣹينْ دࣹفُگْ يࣵقُتࣹ، دِ سࣵڪَّلْ سࣹينْ بࣷݒّ مࣹنّ تُورْ مُيْ جࣴمَّلْ لࣹنّ لُ مࣹلْ نِ گٗورْ مبَا جِگࣺينْ، ‏17 مبَا لُ مࣹلْ نِ مࣹنّ مَلَمْ سُوفْ، مبَا مَلَمْ ڪَوْ مُ مࣹلْ نِ نجَنَاوْ، ‏18 مبَا لُ مࣹلْ نِ ندࣴگْمࣹينتّ، دِ رَامْ، مبَا لُ مࣹلْ نِوْ جࣴنْ، نࣹڪّ ݖِ ندࣷخْ مُ سُوفْ تِيمْ؞ ‏19 بُلࣹينْ سࣺينْتُ اَسَمَانْ، گِسْ جَنتّ بِ، مبَا وࣹيرْ وِ، مبَا بِدِّيوْ يِ، اَڪْ مبࣷولࣹمْ لࣹيرِ اَسَمَانْ، بَ مُ خِيرْتَلْ لࣹينْ ݖِ دِ لࣹينْ سُجّٗوتَلْ اَڪَ جَامُ، تࣹ سࣹينْ يࣵلَّ اَجِ سَخْ جِ فࣺيتَلࣹ يࣷويِي، مبࣷولࣹمْ خࣹيتْ يِ اَسَمَانْ تِيمْ؞ ‏20 يࣹينْ نَگْ اَجِ سَخْ جࣹي لࣹينْ جࣴلْ، بࣹلّࣹي لࣹينْ ݖَ مِسْرَ، تَالُ خࣹلِّڪَايُ وࣹݧْ گُ ݧُولْ بَ، نگِرْ نگࣹينْ دࣷونْ ݧࣷݧَمْ ݧِ مُ سࣺدّࣷو نِڪِ بࣺسُبْ تࣹيْ؞
‏21 وَايࣹ يࣹينَ تَخْ اَجِ سَخْ جِ مࣹرࣹ مَ بَ گِݧْ نࣹ دُمَ جࣵلّ دࣹخُ يُرْدَنْ، تࣹ دُمَ دُگّ ݖَ رࣺيوْ مُ بَاخْ، مَ لࣹينْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ مࣷومَلࣹ؞ ‏22 مَنْ نَگْ ݖِ رࣺيوْ مِي لاَيْ دࣹي؞ دُ مَايْ جࣵلّ دࣹخُ يُرْدَنْ؞ وَايࣹ يࣹينْ يࣹينَيْ جࣵلّ، بَ نَنگُجِ رࣺيوْ مُ بَاخْ مࣷومُ؞ ‏23 وَتُّلࣹينْ سࣹينْ بࣷݒّ بَ بَݧَ فࣵتّࣹ ڪٗلّࣴرࣹ گِ سࣹينْ يࣵلَّ اَجِ سَخْ جِ فَسْ اَڪْ يࣹينْ، دِ سࣵڪَّلْ سࣹينْ بࣷݒّ تُورْ مُيْ جࣴمَّلْ لࣹنّ، تࣹ سࣹينْ يࣵلَّ اَجِ سَخْ جِ آيࣹ لࣹينْ ڪࣷ؞ ‏24 سࣹينْ يࣵلَّ اَجِ سَخْ جِ ڪَتْ، مࣷويْ سَوَرَ وُيْ خࣷيࣷمࣹ، يࣵلَّ جُ فِيرْ لَ؞
‏25 بُ نگࣹينْ دࣹمࣹي بَ اَمْ اَيْ دࣷومْ اَڪِ سࣴتْ، فࣹڪّ لࣹينْ يࣵگّ ݖَ رࣺيوْ مَ، سُ بࣷوبَا سُ نگࣹينْ دࣹفࣹي اَگْ يࣵقُتࣹ، دِ سࣵڪّ اَمْ تُورْ، اَڪْ جࣴمُّ لُ مُ مَنَ دࣷونْ، مبَا نگࣹينْ دࣹفْ لࣹنّ لُ سࣹينْ يࣵلَّ اَجِ سَخْ جِ ݧَاوْلُ، لُ ڪࣷيْ مࣹرْلࣷو، ‏26 مَا نگِ سࣵڪُّ اَسَمَانْ اَڪْ سُوفْ سࣹيدࣹيلْ مَاڪْ يࣹينْ بࣺسُبْ تࣹيْ جِي، نࣹ سُ بࣷوبَا يࣹينَيْ سࣵنڪࣷو سࣵنڪُ ݖِ لُ گَاوْ، بَ وُوتࣹ ݖَ رࣺيوْ، مَ نگࣹينْ جࣵلّ دࣹخُ يُرْدَنْ، جࣴمْ فَ، نگِرْ نَنگُجِ ڪࣷ؞ سُ بࣷوبَا دُنگࣹينْ فَ يࣵگّࣹ، ندَخْ دࣹيسْ لࣹينْ دِ رَافَلْ، بَ نگࣹينْ جࣹيخْ تࣵڪّ؞ ‏27 اَجِ سَخْ جِ مࣷو لࣹينْ دِ تَسَارࣹ ݖِ بِيرْ خࣹيتْ يِ، تࣹ لُ نࣺيوْ لَ سࣹينُ ندࣹسْ دِ دࣷونْ ݖِ بِيرْ خࣹيتِ جَامْبُرْ يِ لࣹينْ اَجِ سَخْ جِيْ دࣵقْ جࣴمࣹ؞ ‏28 فࣷوفَ نگࣹينْ دِ جَامࣷو اَيْ يࣵلَّ يُ لࣷخࣷيْ نِتْ سࣵڪّࣹ بَنتّ، مبَا دࣷجْ، يࣵلَّ يُ دُلْ گِسْ، دُ دࣺگّ، دُ لࣹڪّ، دُ خࣹيݧْتُ؞ ‏29 وَايࣹ سُ نگࣹينْ فَ سࣵڪّࣷو سࣹينْ يࣵلَّ اَجِ سَخْ جِ، دِنگࣹينْ ڪࣷ دَجْ، ندࣹگَمْ سࣹينْ لࣺݒِّ خࣷلْ، اَڪْ سࣹينْ لࣺݒِّ بَڪَّنْ نگࣹينْ ڪࣷ سࣵڪّࣷو؞ ‏30 بُ لࣹينْ نجࣵقَرࣹ دِڪَّلࣹي، فࣹڪّ مبࣷولࣹمْ مبِرْ يࣷويُ دَبْ لࣹينْ، سُ بࣷوبَا دِنگࣹينْ مُجّ دࣺلْسِ ݖِ سࣹينْ يࣵلَّ اَجِ سَخْ جِ تࣹ دِنگࣹينْ ڪࣷ دࣺگَّلْ؞ ‏31 ندَخْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ يࣵلَّيْ يࣴرْمَنْدࣹ لَ؞ دُ لࣹينْ وَݖّ، دُ لࣹينْ يࣵقْتࣹ تࣹ دُ فࣵتّࣹ ڪٗلّࣴرࣹ، گَ مُ فَسࣷونْ اَڪْ سࣹينِ مَامْ ݖِ ڪَوْ نگِݧْ؞
اَبْ جَگْلࣹ ݧࣹيلْ نَ إِسْرَاءِيلْ
‏32 سࣹيتْلࣹينْ رࣹڪّ ݖِ جَنْتِ دࣺمبّ يِ لࣹينْ جِيتُ، لَ دَلࣹ بࣺسْ بَ يࣵلَّ سࣵڪّࣹ نِتْ فِ ڪَوْ سُوفْ، مبَا نگࣹينْ سࣹيتْ اَسَمَانْ، ݖَتْ بَ ݖَتْ؞ ندَخْ لُ مࣵگّࣹ نِي مَسْ نَا اَمْ؟ اَمْ مَسࣹيسْ نَا دࣺگّ لُ نِ مࣹلْ؟ ‏33 اَنَ اَسْڪَنْ وُ مَسَ دࣺگّ بَاتُ يࣵلَّ بُ جِبࣹ ݖِ سَوَرَ، نِ نگࣹينْ ڪࣷ دࣺگّࣹ تࣹ دࣹسࣹ بَڪَّنْ؟ ‏34 جَنْ يࣵلَّ مࣷو مَسَ فࣹخࣹي دِڪّ فࣷقَتِلْ بࣷݒَّمْ اَوْ خࣹيتْ، بࣹلّࣹي لࣹينْ ݖِ وࣹنࣹينْ خࣹيتْ، ݖِ بِيرِ نَتُّ، اَڪِ فِرْندࣹ اَڪِ ڪࣺيمَانْ اَڪْ خَرࣹ يُ مُ سࣷتَّلࣹ دࣷولࣹيْ لࣷخࣷومْ، اَڪْ ڪࣵتَّنُ ݒࣴرࣴگَمْ، اَڪْ جࣴفْ جُ رَگْلُ، نࣷونࣹي لࣹينْ ڪࣷ سࣹينْ يࣵلَّ اَجِ سَخْ جِ دࣹفَلࣹ ݖَ مِسْرَ، نگࣹينْ تࣹگْ ݖِ بࣴتْ؟ ‏35 يࣹينْ دࣺيْ لَݧُ وࣷنْ لࣷولُ لࣺݒّ نگِرْ نگࣹينْ خَمْ نࣹ اَجِ سَخْ جِ دࣷݝّ مࣷويْ يࣵلَّ، تࣹ جࣹنࣹينْ يࣵلَّ اَمُلْ، جُ دُلْ مࣷومْ؞ ‏36 فَ اَسَمَانْ لَ اࣵدّࣷو، دࣺگْتَلْ لࣹينْ بَاتَمْ، نگِرْ يَرْ لࣹينْ، ݖِ ڪَوْ سُوفْ اِتْ لَ لࣹينْ وࣷنࣹ سَوَرَامْ سُ مَگْ سَ، تࣹ اَيْ ڪࣵدّࣷومْ نگࣹينْ دࣺگّ، مُ جِبࣹ ݖَ دِگّ سَوَرَ سَ؞ ‏37 سࣷݒّ سࣹينِ مَامْ نَگْ مࣷو تَخْ مُ تَامُ لࣹينْ، يࣹينْ اَسْڪَنْ وِ لࣹينْ وُوتُ، گࣺنّࣹيلْ لࣹينْ بࣷݒَّمْ مِسْرَ، ݖِ دࣷولࣹيمْ جُ بَرࣹ، ‏38 نگِرْ دࣵقَلْ لࣹينْ خࣹيتْ يُ لࣹينْ سُتْ، اࣴݒّ لࣹينْ دࣷولࣹ، بَ دُگَلْ لࣹينْ سࣹينُمْ رࣺيوْ، سࣺدّ لࣹينْ ڪࣷ بࣺسْ نِڪِ تࣹيْ؞
‏39 ڪࣷنْ خَمْلࣹينْ تࣹيْ تࣹ دࣺلّࣷوسِ ݖِ سࣹينْ خࣹلْ، نࣹ اَجِ سَخْ جِ مࣷويْ يࣵلَّ فَ ڪَوْ اَسَمَانْ اَڪْ فِ ڪَوْ سُوفْ؞ دُ ڪࣹنࣹينْ؞ ‏40 سࣵمّلࣹينْ دࣷگَلِ يࣷونَمْ، اَڪْ سَنْتَانࣹيمْ يِ مَ لࣹينْ دࣺنڪّ بࣺسْ نِڪِ تࣹيْ، ندَخْ نگࣹينْ بَاخْلࣹ يࣹينْ اَڪْ سࣹينْ دࣷومْ يِ لࣹينْ دِ وُوتُ، تࣹ نگࣹينْ يࣵگّ ݖَ رࣺيوْ مَ لࣹينْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ جࣷخْ بَ فࣵوّ؞
اَمَلْ نَݧُ اَيْ رَوْتُ ݖَ ݒࣹنڪُبْ يُرْدَنْ
‏41 بَ لࣷولُ وࣺيࣹي مُوسَىٰ بࣹرْ نَ ݧࣹتِّ دࣴڪّ ݖَ ݒࣹنڪُبْ دࣹخُ يُرْدَنْ، ‏42 مُ دِيْ رَوْتُ ݖِ ڪُ رࣹيْ مࣷرࣷومُ نِتَمْ، تࣹ رࣹيُ ڪࣷ ݖِگْ تࣹيࣹيفْ، وࣷنُ ڪࣷ وࣷونْ اِتْ مبَݧࣹيلْ، لُ جِيتُ لࣷولَ؞ ڪࣷوڪُ سَݧْ نَا لࣵقُجِ ݖِ دࣴڪّ يࣷويُ بࣹنّ، نگِرْ رَوَلࣹ بَڪَّنَمْ؞ ‏43 دࣴڪّ يࣷويُ ݧࣷو دࣷونْ بࣹݖّࣹرْ، گَ ݖَ مࣵنْدِݝْ مَ، ݖَ ڪَوْ دࣺندّ وُ تࣹلّ وَ ݖَ ڪَوْ تُندّ وَ، سُ دࣹي اَبْ رُبࣹنࣹينْ؛ اَڪْ رَمࣷتْ گَ ݖَ گَلࣵدّ، سُ دࣹي اَبْ گࣵدّࣹينْ؛ اَڪْ گࣷلَنْ گَ ݖَ دِيوَانُ بَسَنْ، سُ دࣹي اَبْ مَنَسࣹينْ؞
مُوسَىٰ بَامْتُ نَ يࣷونْ وِ
(سَارْ 4‏.44—11‏.32)
‏44 لِي مࣷو دِ ندِگَلُ يࣷونْ وِ مُوسَىٰ بِرَلࣷونْ فِ ڪَنَمْ بࣵنِّ إِسْرَاءِيلْ؞ ‏45 لِي مࣷو دِ سࣹيدࣹيْ يࣷونْ يِ، اَڪْ دࣷگَلِ يࣷونْ يِ، اَڪْ اࣵتّࣹيْ يࣷونْ يِ مُوسَىٰ دࣺنڪࣷونْ بࣵنِّ إِسْرَاءِيلْ، گَنَّاوْ بَ ݧُ بࣵيِّڪࣷو مِسْرَ، ‏46 ݖَ وࣵلاَّ دࣹخُ يُرْدَنْ، ݖَ خُرْ وَ جَنࣷوڪْ بࣹتْ ݒࣹوࣷرْ، ݖَ رࣺيوُمْ سِوࣷنْ بُورُبْ اَمࣷرࣹينْ بَ تࣷوگࣷونْ اࣹسْبࣷنْ، تࣹ مُوسَىٰ اَڪْ بࣵنِّ إِسْرَاءِيلْ دُمَ وࣷونْ ڪࣷ، بَ ݧُ گࣺنّࣹي مِسْرَ؞ ‏47 ݧࣷو نَنگُ اَمْ رࣺيوَمْ، اَڪْ رࣺيوُمْ اࣷگْ بُورُ بَسَنْ، ݧَارِ بُورِ اَمࣷرࣹينْ ݧَ مࣷومࣷونْ وࣵلاَّ دࣹخُ يُرْدَنْ، ݖَ ݒࣹنڪُ، ‏48 فَ دَلࣹ ݖَ اَرࣷوࣹرْ، گَ ݖَ ݖَتَلْ وَلُمْ اَرْنࣷنْ، بَ ݖَ تُنْدُ سِيࣷݝْ¤، گَ ݧُيْ وَخْ اِتَمْ اࣹرْمࣷنْ، ‏49 بࣷولࣹ ݖَ مبࣷولࣹمْ جࣷورُگْ وࣵلاَّ دࣹخُ يُرْدَنْ، ݖَ ݒࣹنڪُ، بَ اࣵگّ ݖَ گࣺيجُ خࣷرࣷمْ گَ، ݖَ سُوفُ تُنْدُ ݒِسْگَ؞
