لُ جࣴمْ ݖِ ندࣷخُمْ سࣹتْلُ اَڪْ سࣹتْلُ گِ وَرْ
19
‏1 اَجِ سَخْ جِ وَخَتِ مُوسَىٰ اَڪْ هَـٰرُونَ، نࣹ:‏ ‏2 «لِي مࣷويْ دࣷگَلُ يࣷونْ بِ اَجِ سَخْ جِ سَنْتَانࣹ:‏ نࣹلࣹينْ بࣵنِّ إِسْرَاءِيلْ ݧُ اِنْدِلْ لࣹينْ نَگْ وُ جِگࣺينْ وُ خࣷنقّ، وُ مَتْ، مُݖّ سِڪّ، تࣹ ڪࣹنّ مَسُ ڪࣷو تَڪّ، ‏3 نگࣹينْ جࣷخْ ڪࣷ اࣹلَسَرْ سَرْخَلْڪَتْ بِ، ݧُ گࣺنّࣹ ڪࣷ دَلْ بِ، تࣹ ݧُ رࣹنْدِ ڪࣷ فِ ڪَنَمَمْ؞ ‏4 نَ اࣹلَسَرْ سَرْخَلْڪَتْ بِ ݖَݒّ بَارَامَمْ ݖِ دࣹرࣹتْ جِ، وِسْ-وِسَلْ ڪࣷ جُرٗومْ ݧَارِ يࣷونْ فِ فࣺيتࣹيڪْ بُنتّ خَيْمَبْ ندَجࣹ مِ؞ ‏5 نَݧُ لَڪّ نَگْ وِ‎ ݖِ ڪَنَمَمْ؞ دࣹرْ بِ اَڪْ يࣵݒّ وِ اَڪْ دࣹرࣹتْ جِ اَڪْ سࣺبّرِيتْ مِ، لࣺݒّ لَݧُيْ لَڪّ؞ ‏6 سُ ڪࣷ دࣹفࣹي سَرْخَلْڪَتْ بِ سࣵڪّ بَنْتُ سࣹيدَرْ اَڪْ ݖَرُ اِسࣷبْ اَڪْ ندِمࣷ لُ خࣷنقّ ݖُرّ، بࣷولࣹ ڪࣷ سࣵنِّ ݖَ سَوَرَ وَ ݧُيْ لَڪّ نَگْ وَ؞ ‏7 گَنَّاوْ لࣷولُ نَ سَرْخَلْڪَتْ بِ فٗوتْ اَيْ يࣺرࣹيمْ، سَنگُ، دࣷورَ مَنَ دُگَّاتْ ݖِ دَلْ بِ، تࣹ دُ ڪࣷ تࣹي يࣹنْدࣷو سࣷبࣹ بَ ݖَ نگࣷونْ؞ ‏8 ڪِيْ لَڪّ نَگْ وِ اِتَمْ، نَ فٗوتْ اَيْ يࣺرࣹيمْ، سَنگُ، تࣹ دُ ڪࣷ تࣹي يࣹنْدࣷو سࣷبࣹ بَ ݖَ نگࣷونْ؞ ‏9 نَ ڪُ سࣹتْ تࣷنِّ دٗومُ نَگْ وَ، تࣹ سࣷتِّ ڪࣷ بࣺرَبْ بُ سࣹتْ، فࣷوفَ ݖَ بِتِ؞ دٗومْ بَ دࣹنݖُبْ مبࣷولࣷومْ بࣵنِّ إِسْرَاءِيلْ لَيْ دࣷونْ، ݧُ دِ ڪࣷ وَاجَلࣹ ندࣷخُمْ سࣹتْلُ، نگِرْ سَرَخُ ݒٗوتُمْ بࣵڪَّارْ؞ ‏10 ڪُ گٗورْ ڪِيْ تࣷنِّ دٗومِ نَگْ وِ اِتَمْ دَيْ فٗوتْ اَيْ يࣺرࣹيمْ، تࣹ دُ تࣹي مُ يࣹنْدࣷو سࣷبࣹ بَ ݖَ نگࣷونْ؞ نَ لࣷولُ دِ دࣷگَلُ يࣷونْ بُ سَخْ فࣵوّ، ݧࣹيلْ بࣵنِّ إِسْرَاءِيلْ، اَڪْ دࣷخَنْدࣺيمْ بُ دࣴڪّ ݖِ سࣹينْ بِيرْ؞
‏11 «بُ لࣷولُ وࣹيسࣹي ڪُ لاَلْ نِتْ ڪُ دࣹي، اَڪْ نࣺيوْ بُ مُ مَنَ دࣷونْ، دِنَ سࣷبࣹوُ دِيرُ جُرٗومْ ݧَارِ فَنْ؞ ‏12 نَ سࣹتْلُ ݖَ ندࣷخْ مࣷومࣹي ݖَ بࣺسُبْ ݧࣹتّࣹيلْ بَ، گَنَّاوْ بَ مُ ڪࣷ لاَلࣹي، تࣹ نَ سࣹتْلُوَاتْ ݖَ بࣺسُبْ جُرٗومْ ݧَارࣹيلْ بَ؞ سُ ڪࣷ دࣹفࣹي مُ سࣹتْ؞ وَايࣹ سُ سࣹتْلُوُلْ ݖَ بࣺسُبْ ݧࣹتّࣹيلْ بَ، اَڪْ بࣺسُبْ جُرٗومْ ݧَارࣹيلْ بَ، دُ سࣹتْ؞ ‏13 ڪࣺݒّ ڪُ لاَلْ نِتْ ڪُ دࣹي، اَڪْ نࣺيوْ بُ مُ مَنَ دࣷونْ، بُ سࣹتْلُوُلْ، مࣵڪَّانُ اَجِ سَخْ جِ لَ سࣷبࣹيلْ؞ ڪࣷوڪُ دࣹيسْ ڪࣷيْ دَگّࣹ ݖِ بࣵنِّ إِسْرَاءِيلْ؞ گَنَّاوْ ندࣷخُمْ سࣹتْلُ مَ، وِسࣹيسُ ڪࣷ ڪࣷ، سࣷبࣹوُ لَيْ وࣺيࣹ؞
‏14 «لِي لَ يࣷونْ وِ تࣴرَلْ:‏ بُ نِتْ دࣹييࣹي ݖِ بِيرْ خَيْمَ، ڪࣺݒّ ڪُ دُگّ ݖِ بِيرْ خَيْمَ بِ، اَڪْ ڪࣺݒّ ڪُ مُ فࣹڪّ ݖِ خَيْمَ بِ، دَيْ سࣷبࣹوُ دِيرُ جُرٗومْ ݧَارِ فَنْ، ‏15 تࣹ مبࣷولࣹمْ ندَبْ لُ فَ اُبِّڪُ وࣷونْ، ڪُبࣹيرْ اِتْ اُبُ ڪࣷ وࣷونْ رࣵݒّ، لࣷولَ ندَبْ سࣷبࣹوُ نَ؞ ‏16 ڪࣺݒّ ڪُ لاَلْ ݖَ اࣵلّ بَ، نِتْ ڪُ ݧُ بٗومࣹ سَامَرْ، مبَا ڪُ يࣵلَّ مࣷومَلْ بࣷݒَّمْ، مبَا مُ لاَلْ يَخِ نِتْ ڪُ دࣹي، مبَا مُ لاَلْ بࣵمّࣹيلْ، ڪࣷوڪَ دَيْ سࣷبࣹوُ دِيرُ جُرٗومْ ݧَارِ فَنْ؞ ‏17 نَݧُ سࣵڪَّلْ ڪِ سࣷبࣹوُ ݖَ دٗومُ سَرَخُ ݒٗوتُمْ بࣵڪَّارْ بَ ݧُ لَڪّࣷونْ، تࣹ ݧُ سࣷتِّ ندࣷخْ مُيْ دُندّ ݖِ ندَبْ، ݖَ ڪَوْ دٗومْ بَ؞ ‏18 بُ لࣷولُ وࣺيࣹي، نَ ڪُ سࣹتْ جࣴلْ ݖَرُ اِسࣷبْ، ݖَݒّ ڪࣷ ݖَ ندࣷخْ مَ، وِسْ-وِسَلْ ڪࣷ ݖَ ڪَوْ خَيْمَ بَ نِتْ دࣹييࣹ، وِسْ ڪࣷ ݖَ ڪَوْ ندَبْ يࣺݒّ، اَڪْ ݖَ ڪَوْ نِتْ ݧَ فَ نࣹڪّࣷونْ؞ نَ ڪࣷ وِسْ اِتَمْ ݖِ ڪَوْ نِتْ ڪُ لاَلْ يَخِ ڪُ دࣹي، مبَا مُ لاَلْ ڪُ ݧُ بٗومࣹ سَامَرْ، مبَا مُ لاَلْ ڪُ يࣵلَّ مࣷومَلْ بࣷݒَّمْ، مبَا مُ لاَلْ بࣵمّࣹيلْ؞ ‏19 بࣺسُبْ ݧࣹتّࣹيلْ بَ نَگْ، اَڪْ بࣺسُبْ جُرٗومْ ݧَارࣹيلْ بَ، ݖِ لَ نِتْ ڪُ سࣹتْ دِ وِسْ-وِسَلْ ندࣷخْ مِ ݖِ ڪَوْ ڪِ سࣷبࣹوُ؞ بࣺسُبْ جُرٗومْ ݧَارࣹيلْ بَ ݖَ لَ ڪࣷيْ سࣹتّلِ؞ سُ ڪࣷ دࣹفࣹي ڪَ سࣷبࣹوُ فٗوتْ اَيْ يࣺرࣹيمْ، سَنگُ، بُ جَنتّ سࣷوࣹي مُ دَلْدِ سࣹتْ؞ ‏20 ڪࣺݒّ ڪُ سࣷبࣹوُ تࣹ سࣹتْلُوُلْ، ڪࣷوڪُ دࣹيسْ ڪࣷيْ دَگّࣹ ݖِ مبࣷولࣷو مِ، ندَخْ بࣺرَبُ سࣹلّنگَيْ اَجِ سَخْ جِ لَ سࣷبࣹيلْ؞ گَنَّاوْ وِسࣹيسُ ڪࣷ ندࣷخُمْ سࣹتْلُ مِ، سࣷبࣹوُ نَ؞
‏21 «نَ لࣹينْ لࣷولُ ݧࣹيلْ ݖِ ڪَوْ دࣷگَلُ يࣷونْ بُ سَخْ دࣵڪّ؞ ڪَ دࣷونْ وِسْ-وِسَلْ ندࣷخُمْ سࣹتْلُ مِ نَگْ، نَ فٗوتْ اَيْ يࣺرࣹيمْ، تࣹ ڪُ لاَلْ ݖِ ندࣷخُمْ سࣹتْلُ مِ دَيْ يࣹنْدࣷو سࣷبࣹ بَ ݖَ نگࣷونْ؞ ‏22 نِتْ ڪِ سࣷبࣹوُ، مبࣷولࣹمْ لُ مُ لاَلْ دَيْ سࣷبࣹوُ، تࣹ ڪُ لاَلْ ݖَ لࣷولَ اِتْ دَيْ يࣹنْدࣷو سࣷبࣹ بَ ݖَ نگࣷونْ؞»
