بِنْدࣹيسْ نَ وَا سࣵردّ
3
‏1 «مَلاَڪَمْ مبࣷولࣷومْ گࣴمْڪَتْ ݧَ ݖَ سࣵردّ نَگْ، بِندّ ڪࣷ نࣹ ڪࣷ:‏
«لِي مࣷو دِ ڪࣵدُّيْ بࣷرࣷومْ جُرٗومْ ݧَارِ نࣷو يُ يࣵلَّ يِ اَڪْ جُرٗومْ ݧَارِ بِدِّيوْ يِ؞ مُ نࣹ:‏ گِسْ نَا سَيْ جࣴفْ؞ دَݧُ لاَ جࣵݒّࣹ ڪُيْ دُندّ تࣹ فࣹڪّ لَ دࣹي؞ ‏2 تࣹيوْلُلْ تࣹ نگَ لࣹقْلِ ندࣹسْ وِ نگَيْ وَاجَ دࣹيلࣹ؞ ندَخْ ڪَتْ، گِسُمَ سَ جࣴفْ جُ مَتْ فَ سَمَ ڪَنَمْ يࣵلَّ؞ ‏3 ڪࣷنْ نَگْ فࣵتّلِڪُلْ لَ نگَ جࣷتࣷونْ تࣹ دࣺگّ ڪࣷ؛ نگَ سࣵمّ ڪࣷ، تࣹ تُوبْ؞ سࣷو تࣹيوْلُوُلْ نَگْ، دِنَا ݧࣴوْ نِبْ سࣵݖّ، تࣹ دࣷو خَمْ مُڪّ وَنْ وَخْتُ لاَ نَا جِمࣹيتْ فِ سَ ڪَوْ؞
‏4 «مࣷونَ اَمْ نگَ فࣷوفُ ݖِ سࣵردّ نِتْ ݧُ نࣺيوْ ݧُ تِلِمَلُلْ سࣹينْ مبُبّ؞ ݧࣷوݧࣷويْ اࣵندّ اَڪْ مَنْ، سࣷلْ لُ وࣹيخْ، ندَخْ ݧࣷو ڪࣷ يࣹيࣷو؞ ‏5 ڪُ اَمْ ندَمْ مࣷويْ سࣷلࣷو نࣷونُ مبُبّ مُ وࣹيخْ، تࣹ تُرَمْ، دُمَ ڪࣷ فَرْ ݖِ تࣺيرࣹبْ دُندّ بِ، وَايࣹ مَا ڪࣷيْ وُيُلْ ݖِ سَمَ ڪَنَمْ بَايْ اَڪْ اَيْ مَلاَڪَامْ؞ ‏6 ڪُ اَمِ نࣷݒّ ڪَتْ، نَ دࣺگّ لِ نࣷووُگْ يࣵلَّ وَخْ مبࣷولࣷويْ گࣴمْڪَتْ ݧِ؞
بِنْدࣹيسْ نَ وَا فِلَدࣹلْفِ
‏7 «مَلاَڪَمْ مبࣷولࣷومْ گࣴمْڪَتْ ݧَ ݖَ فِلَدࣹلْفِ نَگْ، بِندّ ڪࣷ، نࣹ ڪࣷ:‏
«لِي مࣷو دِ ڪࣵدُّيْ ڪُ سࣹلّ ڪِ، ڪِيْ دࣴگّ، تࣹ يࣷرْ ݖَابِيْ دَاوُدَ، دِ تِجِّ تࣹ ڪࣹنّ دُ تࣴجْ، دِ تࣴجْ تࣹ ڪࣹنّ دُ تِجِّ؞ مُ نࣹ:‏ ‏8 گِسْ نَا سَيْ جࣴفْ؞ مَا دࣹفْ فِ سَ ڪَنَمْ، بُنتّ بُ تِجِّڪُ بُ ڪࣹنّ مَنُلَ تࣴجْ؞ ندَخْ نࣺيوْ نگَ دࣷولࣹ، وَايࣹ سࣵمّ نگَ سَمَ ڪࣵدُّ تࣹ وࣹدِّوࣷو مَ؞ ‏9 ݧࣹنّ ݖِ جࣵنگُبْ سࣹيْتَانࣹ بِ، ݧࣷوݧُيْ فࣹنْ، دِ تُدّࣷو اَيْ يَوُتْ تࣹ دُݧُيْ يَوُتْ، مَا نگِي دِ لࣹينْ خِيرْتَلْ، دِڪّلࣷو لࣹينْ، سُجّٗوتْلࣷو لࣹينْ فِ سَ ڪَنَمْ، بَ ݧُ خَمْ نࣹ مَنْ مَا لَ سࣷݒّ؞ ‏10 گَنَّاوْ يَا سࣵمّ سَمَ ڪࣵدُّ، بَ مُݧْ نِ مَ لَ ڪࣷ سَنْتࣹ، مَنْ اِتَمْ دِنَا لَ سࣵمّ ݖِ وَخْتُوْ نَتُّ بِيْ وࣵݖّسِ ݖِ اࣵدِّنَ سࣺݒّ، نگِرْ نَتُّ وَا ڪَوْ سُوفْ؞
‏11 «مَا نگِ ݧࣴوْ لࣺيگِ؞ ݝࣷيَلْ ݖِ لِ نگَ يࣷرْ، نگِرْ ڪࣹنّ بَݧَ فࣴڪّ سَ ڪَالَگْ ندَمْ؞ ‏12 ڪُ اَمْ ندَمْ نَگْ، مَا لَيْ دࣹفْ نگَ دࣷونْ اَگْ ڪࣹنُ فِ سَمَ بِيرْ نࣺيگْ يࣵلَّ، تࣹ دࣷو فَ گࣺنَّتِ مُڪّ؞ مَايْ بِندّ فِ سَ ڪَوْ سَمَ تُرُ يࣵلَّ اَڪْ سَمَ تُرُ دࣴڪُّ يࣵلَّ، تࣹ مࣷويْ يࣹرُسَلࣹمْ گُ بࣹيسْ، گِيْ وࣵݖّࣹ فَ ڪَوْ، بࣵيِّڪࣷو فَ سَمَ يࣵلَّ، مَايْ بِندّ اِتْ فِ سَ ڪَوْ سَمَ تُرْ وُ يࣹيسْ؞ ‏13 ڪُ اَمِ نࣷݒّ ڪَتْ، نَ دࣺگّ لِ نࣷووُگْ يࣵلَّ وَخْ مبࣷولࣷويْ گࣴمْڪَتْ ݧِ؞
بِنْدࣹيسْ نَ وَا لَوْدِسࣹ
‏14 «مَلاَڪَمْ مبࣷولࣷومْ گࣴمْڪَتْ ݧَ ݖَ لَوْدِسࣹ نَگْ، بِندّ ڪࣷ، نࣹ ڪࣷ:‏
«لِي مࣷو دِ ڪࣵدُّيْ وَاوُ يࣵلَّ جِ، سࣹيدࣹ بِ وٗورْ تࣹ دࣴگُّ، ندࣹيُ ݖࣵڪّتࣺيفُ يࣵلَّ جِ؞ مُ نࣹ:‏ ‏15 خَمْ نَا سَيْ جࣴفْ؛ خَمْ نَا نࣹ سࣹدّࣷو، تࣵنگࣷو؞ سُ مَ سَݧࣷونْ نگَ سࣹدّ مبَا نگَ تࣵنگّ؞ ‏16 گَنَّاوْ نِي نگَ نُگّࣹ نَگْ؛ تࣵنگࣷو، سࣹدّࣷو، دِنَا لَ يࣵبِّ؞ ‏17 گَنَّاوْ يَا نگِ نَانْ:‏ ‹اَمْ نَا اَلَلْ، بَرࣹلࣹ نَا، سࣷخْلاَتُمَ دَرَ،› تࣹ يَوْ خَمُلࣷو نِ نگَ تࣷرࣷخࣹ، جَرَ يࣴرࣴمْ، ݧࣵڪّ، گُمْبَ، نࣹ دُݝّ، ‏18 دَمَ لَيْ دِگَلْ، نگَ جࣴندّ ݖِ مَنْ وُرُسْ وُ ݧُ خࣹلّࣹي ݖِ سَوَرَ بَ مُ سࣹتْ وࣹݖّ، نگِرْ نگَ اَمْ اَلَلْ؞ جࣴنْدَلْ اِتَمْ يࣺرࣹ يُ وࣹيخْ يࣷو سࣷلࣷو، بَ لࣵقْ سَ دُݝَّارَلْ گُ گࣵݖّࣹلُ گِ؛ اَڪْ تُوفَايْ لࣷويْ تُوفࣷو، نگِرْ نگَ گِسْ؞
‏19 «مَنْ، ڪࣺݒّ ڪُ مَ سࣷݒّ، مَا ڪࣷيْ جُبَّنْتِ، يَرْ ڪࣷ؞ ڪࣷنْ نَگْ فَرْلُلْ تࣹ تُوبْ!‏ ‏20 مَا نگِ نِي تَخَوْ ݖِ بُنتّ بِ، دِ فࣴگّ؞ ڪُ دࣺگّ سَمَ بَاتْ، بَ اُبِّلْ مَ، دِنَا اَگْسِ مࣷوسْ سَ نࣺيگْ، نُ بࣷڪّ رࣹيرْ؞
‏21 «ڪُ اَمْ ندَمْ نَگْ، مَا لَيْ مَيْ نگَ تࣷوگْ اَڪْ مَنْ فِ سَمَ نگࣵنگُنࣹ، نَ مَ اَمࣹ وࣷونْ ندَمْ، بَ تࣷوگْ اَڪْ سَمَ بَايْ فِ نگࣵنگُنࣹيمْ؞ ‏22 ڪُ اَمِ نࣷݒّ ڪَتْ، نَ دࣺگّ لِ نࣷووُگْ يࣵلَّ وَخْ مبࣷولࣷويْ گࣴمْڪَتْ ݧِ؞»
