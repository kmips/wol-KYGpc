سࣵرْتَلْ نَݧُ لُ سࣹتْ اَڪْ لُ سࣹتُلْ
(سَارْ 11—15)
لُ جࣴمْ ݖِ مَلَ مُ سࣹتْ اَڪْ مُ سࣹتُلْ
11
‏1 اَجِ سَخْ جِ وَخْ نَ مُوسَىٰ اَڪْ هَـٰرُونَ نࣹ لࣹينْ:‏ ‏2 «وَخْلࣹينْ بࣵنِّ إِسْرَاءِيلْ نࣹ لࣹينْ:‏ بࣷرࣷومْ بَڪَّنْ يِ نگࣹينْ دِ لࣹڪّ سࣹينْ يࣵݒّ، ݖِ مبࣷولࣹمْ مَلَ يِيْ دࣷخْ ݖِ سُوفْ مࣷو دِ يِي:‏ ‏3 مبࣷولࣹمْ مَلَ مُ وࣹ وَ خَرْ، سࣺدّلِڪࣷو ݧَارْ، تࣹ مُيْ دُݧّ، مࣷومْ نگࣹينْ دِ لࣹڪّ اَوْ يࣵݒَّمْ؞ ‏4 يِي ݧُيْ لِمْ، بُلࣹينْ ڪࣷ لࣹڪّ:‏ مَلَ مُيْ دُݧّ تࣹ وࣹ وِ خَرُلْ، مبَا وࣹ وِ خَرْ تࣹ دُ دُݧّ؞ لُ مࣹلْ نِ گِلࣺيمْ، بُلࣹينْ ڪࣷ لࣹڪّ ندَخْ دَيْ دُݧّ، وَايࣹ وࣹ وِ خَرُلْ؞ دَگَنُلْ ݖِ يࣹينْ؞ ‏5 دَمَنْ¤ دَيْ دُݧّ وَايࣹ وࣹ وِ خَرُلْ، ڪࣷنْ دَگَنُلْ ݖِ يࣹينْ؞ ‏6 لࣴگْ دَيْ دُݧّ وَايࣹ وࣹ وِ خَرُلْ، ڪࣷنْ دَگَنُلْ ݖِ يࣹينْ؞ ‏7 مبَامْ-خُوخْ، وࣹ وِ خَرْ نَ وَايࣹ دُ دُݧّ، ڪࣷنْ دَگَنُلْ ݖِ يࣹينْ؞ ‏8 بُلࣹينْ ݖِ لࣹڪّ لࣹنّ تࣹ بُلࣹينْ لاَلْ مࣺدَّمْ ندَخْ لࣹنّ دَگَنُ ݖِ، ݖِ يࣹينْ؞
‏9 «لِ نگࣹينْ مَنَ لࣹڪّ ݖِ مبࣷولࣹمْ لُيْ نࣹڪّ ݖِ ندࣷخْ مࣷويْ لِي:‏ مبࣷولࣹمْ لُ اَمْ اَيْ ݖِيرِيرْ اَڪِ وَاسِنْتٗورْ تࣹيْ نࣹڪّ ݖِ ندࣷخُمْ گࣺيجْ مبَا دࣹخْ، مَنْ نگࣹينْ ڪࣷو لࣹڪّ؞ ‏10 وَايࣹ مبࣷولࣹمْ لُيْ نࣹڪّ ݖِ گࣺيجْ مبَا دࣹخْ، دِ لࣺݒّ لُيْ رَمّ-رَمِّ ݖِمْ ندࣷخْ مبَا مبࣷولࣹمْ يࣹنࣹينْ بِنْدࣹيفْ يُيْ نࣹڪّ ݖِ ندࣷخْ، سُ اَمُلْ ݖِيرِيرْ اَڪِ وَاسِنْتٗورْ، سِبْلࣹينْ ڪࣷ؞ ‏11 لُ نگࣹينْ سِبْ لَيْ دࣷونْ، ڪࣷنْ بُلࣹينْ ݖِ لࣹڪّ يࣵݒُّ لࣹنّ تࣹ سِبْلࣹينْ مࣺدَّمْ؞ ‏12 مبࣷولࣹمْ لُيْ نࣹڪّ ݖِ ندࣷخْ تࣹ اَمُلْ ݖِيرِيرْ اَڪْ وَاسِنْتٗورْ، سِبْلࣹينْ ڪࣷ؞
‏13‏-18 «لِ نگࣹينْ وَرَ سِبْ ݖِ نجَنَاوْ يِ تࣹ دࣹيسُ ڪࣷ لࣹڪّ مࣷو دِ:‏ جَخَايْ اَڪْ مبࣷولࣹمْ خࣹيتِ تَنْ؛ اَڪْ ݖࣺيلِ، بُ خࣷنقّ بَاڪْ بُ ݧُولْ بَ، بُ مُ مَنَ دࣷونْ، اَڪْ مبࣷولࣹمْ خࣹيتِ بَاخࣷݧْ، اَڪْ مبࣷولࣹمْ خࣹيتِ لࣷويْ، مُيْ لࣷويْ مُ ندَوْ اَڪْ لࣷويْ مُيْ نَݒّ اَڪْ لࣷويْ مُ گُدّ نࣷݒّ اَڪْ لࣷويُ مࣵنْدِݝْ، بَ ݖِ خَرْ-جࣺجّ اَڪْ تِيوِ؞ نَنگࣹينْ سِبْ مَفْ اِتَمْ، اَڪْ ڪِرِگࣺيجْ اَڪْ مبࣷولࣹمْ خࣹيتِ لِݖِّنْ؛ ‏19 اَڪْ يࣺبّ اَڪْ لُ خࣹيتࣷوڪْ خࣷدّ، اَڪْ مُسࣹجُبْرُ اَڪْ نجُگُبْ؞
‏20 «مبࣷولࣹمْ گُنٗورْ گُيْ نَاوْ تࣹيْ دࣷخࣹ ݧࣹينْتِ تࣵنڪّ سِبْلࣹينْ ڪࣷ؞ ‏21 لِ نگࣹينْ مَنَ لࣹڪّ دَالْ ݖِ مبࣷولࣹمْ گُنٗورْ گُيْ نَاوْ تࣹيْ دࣷخࣹ ݧࣹينْتِ تࣵنڪّ مࣷو دِ:‏ لࣺݒّ لُ تࣵنڪّ يِ اࣵندّ اَڪِ يࣹيلْ، مُ مَنْ ݖَا تࣴبْ-تࣴبࣹ ݖِ سُوفْ؞ ‏22 لِ نگࣹينْ مَنَ لࣹڪّ ݖِ يࣷويُ مࣷو دِ:‏ مبࣷولࣹمْ خࣹيتُ نجࣺيرࣺيرْ اَڪْ مبࣷولࣹمْ خࣹيتُ سَلّࣹيرْ اَڪْ مبࣷولࣹمْ خࣹيتُ سࣷݖّࣹتْ اَڪْ يࣹنࣹينْ يُ ݧُ خࣹيتࣷولْ يࣺݒّ؞ ‏23 وَايࣹ مبࣷولࣹمْ يࣹنࣹينْ گُنٗورْ يُيْ نَاوْ تࣹيْ دࣷخࣹ ݧࣹينْتِ تࣵنڪّ سِبْلࣹينْ ڪࣷ؞
‏24 «يِي نَگْ دَ لࣹينْ دِ سࣷبࣹيلْ تࣹ ڪࣺݒّ ڪُ لاَلْ لُ ݖِ دࣹي دِنَ يࣹنْدࣷو سࣷبࣹ بَ جَنتّ سࣷ؞ ‏25 تࣹ اِتَمْ ڪࣺݒّ ڪُ فࣷرْ لࣹنّ لُ دࣹي ݖِ يِي، نَ فٗوتْ اَيْ يࣺرࣹيمْ، وَايࣹ دِنَ يࣹنْدࣷو سࣷبࣹ بَ جَنتّ سࣷ؞ ‏26 مبࣷولࣹمْ مَلَ مُ وࣹ وَ خَرْ تࣹ سࣺدّلِڪࣷووُلْ ݧَارْ مبَا دُ دُݧّ دَگَنُلْ ݖِ يࣹينْ، ڪُ ݖِ لاَلْ لࣹنّ اِتْ سࣷبࣹوُ نَ؞ ‏27 بَ تࣹيْ لࣺݒّ لُيْ دࣷخࣹ ندࣴگُّيْ تࣵنڪَمْ ݖِ مبࣷولࣹمْ بࣷرࣷومْ ݧࣹينْتِ تࣵنڪّ، دَگَنُلْ ݖِ يࣹينْ تࣹ ڪُ لاَلْ مࣺدَّمْ دِنَ يࣹنْدࣷو سࣷبࣹ بَ جَنتّ سࣷ؞ ‏28 ڪُ فࣷرْ مࣺدَّمْ، نَ فٗوتْ اَيْ يࣺرࣹيمْ تࣹ دِنَ يࣹنْدࣷو سࣷبࣹ بَ جَنتّ سࣷ؞ يࣷويُ دَگَنُلْ ݖِ يࣹينْ؞
‏29 «بُ لࣷولُ وࣹيسࣹي، لࣹنࣹينْ لُ دَگَنُلْ ݖِ يࣹينْ ݖِ رَبْ يُ سࣹوْ-سࣹوَانْ يِيْ دࣷخْ نࣹ ݧࣵݒّ ݖِ سُوفْ، مࣷويْ ڪَݧَ اَڪْ جَنَخْ اَڪْ خࣹيتِ مبࣴتّ يࣺݒّ؛ ‏30 اَڪْ اُنڪّ اَڪْ بَرْ اَڪْ سِنْدَخْ بُيْ يࣺيگْ، اَڪْ سِنْدَخْ بُيْ دࣷخْ ݖِ سُوفْ، اَڪْ ڪࣵڪَّتَرْ؞ ‏31 يࣷويࣷو دَگَنُلْ ݖِ يࣹينْ ݖِ مبࣷولࣹمْ رَبْ يُ سࣹوْ-سࣹوَانْ يِيْ دࣷخْ نࣹ ݧࣵݒّ ݖِ سُوفْ؞ ڪࣺݒّ ڪُ لاَلْ لࣹنّ لُ ݖِ دࣹي دِنَ يࣹنْدࣷو سࣷبࣹ بَ جَنتّ سࣷ؞ ‏32 تࣹ اِتْ لُ ݖِ دࣹي بَ وَدّ ݖِ ڪَوْ لࣹنّ، لࣷولَ دَيْ سࣷبࣹوُ، نَيْ ندَبَلْ بَنتّ، مبَا ندِمࣷ، دࣹرْ مبَا اَبْ سَاڪُ، مبَا جُمْتُڪَايْ بُ مُ مَنَ دࣷونْ؞ نَݧُ سٗوبْ لࣷولَ ݖِ ندࣷخْ تࣹ دُ تࣹي مُ يࣹنْدࣷو سࣷبࣹ بَ جَنتّ سࣷ، مُ دࣷورَ سࣹتْ؞ ‏33 سُ فࣹڪّࣹي نࣹ لࣹنَّ دࣹي ݖِ يࣷويُ، وَدّ ݖِ بِيرْ ندَبَلْ خَنْدࣹيرْ، لُ وࣷونْ ݖَ بِيرْ سࣷبࣹوُ نَ، تࣹ نَݧُ تࣷجْ ندَبْ لَ؞ ‏34 ندࣷخْ مُ جٗگࣹ ݖِ ندَبْ لࣷولُ، بُ دَلࣹي ݖِ ݧَمْ وُ دَگَنْ، ݧَمْ وࣷووُ دَلْ نَيْ سࣷبࣹوُ مࣷومْ اِتَمْ؛ بُ دࣷونْ اَگْ نَانْ گُ دَگَنْ مࣷو ݖَ نࣹڪّ اِتْ، سࣷبࣹوُ نَ، اَڪْ ندَبْ لُ نَانْ گَ مَنَ نࣹڪّ؞ ‏35 لࣺݒّ لُ لࣹنّ ݖِ يࣷويُ دࣹي، بَ دَلْ ݖَ، لَ مُ دَلْ سࣷبࣹوُ نَ؞ سُ دࣹي اَبْ ݒُورُ بَنْ مبَا ݖِنْ، دࣹيسْ ڪࣷيْ تࣷجْ ندَخْ سࣷبࣹوُ نَ تࣹ لُ سࣹتُلْ لَ ݖِ يࣹينْ؞ ‏36 سُ فࣹڪّࣹي نࣹ لࣷولَ دࣹي دَا وَدّ ݖِ بِيرْ بࣴتُ ندࣷخْ مبَا ڪࣵمبّ گُ دࣹفْ ندࣷخْ، دُ تࣹي ندࣷخْ مَ لاَبْ، وَايࣹ ڪُ لاَلْ لَ دࣹي ݖَ بِيرْ سࣷبࣹوُ نگَ؞ ‏37 سُ لَ دࣹي وَدّࣹي ݖِ ݒࣹݒُّمْ جِوُ، دُ تࣹي ݒࣹݒُّمْ جِوُ مَ سࣹتْ؛ ‏38 وَايࣹ سُ فࣹڪّࣹي ݧُ ڪَاسْ ݒࣹݒّ مَ تࣹ لࣹنّ لُ دࣹي لُ بࣷڪّ ݖِ يࣷويُ دَلْ ݖَ، ݒࣹݒّ مَ دَگَنُلْ ݖِ يࣹينْ؞
‏39 «سُ مَلَ مُ دَگَنْ مࣺدّࣹي، ڪُ لاَلْ مࣺدّ مَ دِنَ يࣹنْدࣷو سࣷبࣹ بَ جَنتّ سࣷ؞ ‏40 ڪࣺݒّ ڪُ لࣹڪّ ݖَ مࣺدّ مَ، نَ فٗوتْ اَيْ يࣺرࣹيمْ، تࣹ دُ تࣹي مُ يࣹنْدࣷو سࣷبࣹ بَ جَنتّ سࣷ؞ ڪࣺݒّ ڪُ فࣷرْ مࣺدّ مَ، نَ فٗوتْ اَيْ يࣺرࣹيمْ تࣹ دُ تࣹي مُ يࣹنْدࣷو سࣷبࣹ بَ جَنتّ سࣷ؞
‏41 «مبࣷولࣹمْ بِنْدࣹيفْ يُ سࣹوْ-سࣹوَانْ يِيْ دࣷخْ نࣹ ݧࣵݒّ ݖِ سُوفْ، سِبْلࣹينْ ڪࣷ؞ دࣹيسُ ڪࣷ لࣹڪّ؞ ‏42 لࣺݒّ لُيْ وَتَتُ، مبَا لُ سࣹوْ دِ بࣷرࣷومْ ݧࣹينْتِ تࣵنڪّ، مبَا مُ بَرࣹ اَيْ تࣵنڪّ ݖِ مبࣷولࣹمْ بِنْدࣹيفْ يُ سࣹوْ-سࣹوَانْ يِيْ دࣷخْ نࣹ ݧࣵݒّ ݖِ سُوفْ، وَرُلࣹينْ ڪࣷو لࣹڪّ، ندَخْ لُ ݧُ وَرَ سِبْ لَ؞ ‏43 بُلࣹينْ اَرَامَلْ سࣹينْ بࣷݒّ ݖِ لࣹنّ ݖِ يُ سࣹوْ-سࣹوَانْ يࣷويُ؛ بُلࣹينْ ݖِ تَقْ، دِ ݖِ سࣷبࣹيلْ سࣹينْ بࣷݒّ، ‏44 ندَخْتࣹ مَنْ اَجِ سَخْ جِ مَايْ سࣹينْ يࣵلَّ؞ ڪࣷنْ نَگْ سࣹلّ-لُلࣹينْ تࣹ سࣹلّ ندَخْ مَنْ ڪُ سࣹلّ لاَ؞ ڪࣷنْ بُلࣹينْ سࣷبࣹيلْ مُڪّ سࣹينْ بࣷݒّ ݖِ مبࣷولࣹمْ بِنْدࣹيفْ يُ سࣹوْ-سࣹوَانْ يِيْ دࣷخْ ݖِ سُوفْ؞ ‏45 ندَخْتࣹ ڪَتْ مَنْ اَجِ سَخْ جِ مَايْ ڪِ لࣹينْ گࣺنّࣹ وࣷونْ رࣺيوُمْ مِسْرَ، دِ سࣹينْ يࣵلَّ؞ سࣹلّ-لࣹينْ بࣷوگْ، ندَخْ مَنْ ڪُ سࣹلّ لاَ؞»
‏46 لࣷولُ لَ يࣷونْ دِگْلࣹ ݖِ وࣵلُّ مَلَ اَڪْ نجَنَاوْ اَڪْ بِنْدࣹيفْ يِ نࣹڪّ ݖِ ندࣷخْ، اَڪْ مبࣷولࣹمْ بِنْدࣹيفْ يُ سࣹوْ-سࣹوَانْ يِيْ دࣷخْ نࣹ ݧࣵݒّ ݖِ سُوفْ؞ ‏47 دِنَ تَخَ رࣵݧَّتْلࣹ لُ سࣹتْ اَڪْ لُ سࣹتُلْ؞ دِنَ تَخَ رࣵݧَّتْلࣹ بࣷرࣷومْ بَڪَّنْ يِ دَگَنَ لࣹڪّ ݧࣷوڪْ بࣷرࣷومْ بَڪَّنْ يِ دَگَنُلَ لࣹڪّ؞
