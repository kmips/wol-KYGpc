لُ جࣴمْ ݖِ دَلِنُ گِيرْ يِ
2
‏1 بَ لࣷولُ وࣺيࣹي اَجِ سَخْ جِ وَخْ مُوسَىٰ اَڪْ هَـٰرُونَ، نࣹ لࣹينْ:‏ ‏2 «نَ بࣵنِّ إِسْرَاءِيلْ دِ دَلْ، ڪُ نࣹڪّ اَڪْ رَايَامْ اَڪْ خࣵمِّڪَايُ ڪࣴرْ مَامَمْ؞ فِ وࣴرْ خَيْمَبْ ندَجࣹ مِ تࣹ دࣵندّ ڪࣷ اَسْ لࣴفْ، فِ لَݧُيْ دَلࣹ:‏
 
‏3 «ݧَيْ دَلࣹ ݖَ ڪَنَمْ، فَ فࣺيتࣹ ݒࣹنڪُ، ݧࣷويْ وَا رَايَبْ يُدَ اَڪْ سࣹينِ گࣵنگࣷورْ، ڪِلِفَگْ يُدࣹينْ ݧِ دِ نَسࣷنْ دࣷومُ اَمِنَدَبْ؞ ‏4 ݧَ ݧُ لِمْ ݖِ گࣵنگࣷورَمْ دِ جُرٗومْ ݧَارْ فُڪِّ جُنّࣹيڪْ ݧࣹينتّ اَڪْ جُرٗومْ بࣹنِّ تࣺيمࣺيرْ (74600)؞
‏5 «ݧَيْ دَلࣹ فَ فࣹگّࣷوڪْ ݧࣷومْ ݧࣷويْ گِيرُگْ اِسَڪَرࣹينْ ݧِ، ڪِلِفَگْ اَسْڪَنُ اِسَاڪَرْ دِ نࣹتَنࣹلْ دࣷومُ سُوَرْ؞ ‏6 ݧَ ݧُ لِمْ ݖِ گࣵنگࣷورَمْ دِ جُرٗومْ فُڪِّ جُنّࣹيڪْ ݧࣹينتّ اَڪْ ݧࣹينْتِ تࣺيمࣺيرْ (54400)؞
‏7 «گِيرُگْ سَبُلࣷنࣹينْ ݧࣹييْ فࣹگّࣷوڪْ ݧࣷومْ، ڪِلِفَگْ اَسْڪَنُ سَبُلࣷنْ دِ اࣹلْيَبْ دࣷومُ اࣹلࣷنْ؞ ‏8 ݧَ ݧُ لِمَلْ گࣵنگࣷورَمْ دِ جُرٗومْ فُڪِّ جُنّࣹيڪْ جُرٗومْ ݧَارْ اَڪْ ݧࣹينْتِ تࣺيمࣺيرْ (57400)؞
‏9 «مبࣷولࣹمْ ݧَ ݧُ لِمَلْ دَلُبْ يُدَ اَڪِ گࣵنگࣷورَمْ دَلْدِ دࣷونْ تࣺيمࣺيرِ جُنّࣹيڪْ جُرٗومْ ݧࣹتّ فُڪّ اَڪْ جُرٗومْ بࣹنّ اَڪْ ݧࣹينْتِ تࣺيمࣺيرْ (186400)؞ بُ ݧُيْ دࣹمْ، ݧࣷوݧࣷويْ دࣷخْ، جِيتُ؞
 
‏10 «وَا رَايَبْ رُبࣹنْ ݧࣷويْ دَلࣹ بࣴجْ-سَالُمْ، ݧࣷوڪْ سࣹينِ گࣵنگࣷورْ، ڪِلِفَگْ اَسْڪَنُ رُبࣹنْ دِ اࣹلِسُرْ دࣷومُ سࣹدࣹيُرْ؞ ‏11 ݧَ ݧُ لِمَلْ گࣵنگࣷورَمْ دِ ݧࣹينتّ فُڪِّ جُنّࣹيڪْ جُرٗومْ بࣹنّ اَڪْ جُرٗومِ تࣺيمࣺيرْ (46500)؞
‏12 «ݧَيْ دَلࣹ فَ فࣹگّࣷوڪْ ݧࣷومْ ݧࣷويْ گِيرُگْ ݖِمْيࣷنࣹينْ ݧِ، ڪِلِفَگْ اَسْڪَنُ سِمࣹيࣷنْ دِ سࣹلُمْيࣹلْ دࣷومُ سُرِسَدَايْ؞ ‏13 ݧَ ݧُ لِمَلْ گࣵنگࣷورَمْ دِ جُرٗومْ فُڪِّ جُنّࣹيڪْ جُرٗومْ ݧࣹينتّ اَڪْ ݧࣹتِّ تࣺيمࣺيرْ (59300)؞
‏14 «گِيرُ گࣵدّࣹينْ ݧࣹييْ فࣹگّࣷوڪْ ݧࣷومْ، ڪِلِفَگْ اَسْڪَنُ گࣵدّ دِ اࣹلْيَسَفْ دࣷومُ رࣹوࣹلْ؞ ‏15 ݧَ ݧُ لِمَلْ گࣵنگࣷورَمْ دِ ݧࣹينتّ فُڪِّ جُنّࣹيڪْ جُرٗومْ اَڪْ جُرٗومْ بࣹنِّ تࣺيمࣺيرْ اَڪْ جُرٗومْ فُڪّ (45650)؞
‏16 «مبࣷولࣹمْ ݧَ ݧُ لِمَلْ دَلُبْ رُبࣹنْ اَڪِ گࣵنگࣷورَمْ دࣷونْ تࣺيمࣺيرِ جُنّࣹيڪْ جُرٗومْ فُڪّ اَڪْ بࣹنّ، اَڪْ ݧࣹينْتِ تࣺيمࣺيرْ اَڪْ جُرٗومْ فُڪّ (151450)؞ بُ ݧُيْ دࣹمْ، ݧࣷوݧࣷويْ دࣷونْ ݧَارࣹيلُ ڪُرࣺلْ، دࣷخْ، تࣷݒّ ݖَ؞
 
‏17 «سُ ڪࣷ دࣹفࣹي خَيْمَبْ ندَجࣹ مِ جٗگْ تࣷݒّ ݖَ، اࣵندّ اَڪْ لࣹوࣹينْ ݧِ ڪࣷيْ گࣵدُّ، تࣹ سࣹينْ دَلْ نࣹڪّ ݖِ دِگّ دَلْ يِ؞ نࣷونࣹي ڪُرࣺلْ يِيْ دَلࣹ دَالْ، نِ لَݧُيْ جٗگࣹ دࣹمْ، ڪُ نࣹڪّ تࣷݒّ اَيَمْ اَڪْ رَايَامْ؞
 
‏18 «وَا رَايَبْ اࣹفْرَيِمْ ݧࣷويْ دَلࣹ وࣹتُ سࣷوُ، ݧࣷوڪْ سࣹينِ گࣵنگࣷورْ، ڪِلِفَگْ اَسْڪَنُ اࣹفْرَيِمْ دِ اࣹلِسَمَ دࣷومُ اَمِيُتْ؞ ‏19 ݧَ ݧُ لِمَلْ گࣵنگࣷورَمْ دِ ݧࣹينتّ فُڪِّ جُنِّ اَڪْ جُرٗومِ تࣺيمࣺيرْ (40500)؞
‏20 «ݧَيْ دَلࣹ فَ فࣹگّࣷوڪْ ݧࣷومْ ݧࣷويْ گِيرُگْ مَنَسࣹينْ ݧِ، ڪِلِفَگْ اَسْڪَنُ مَنَسࣹ دِ گَمَلْيࣹلْ دࣷومُ ݒࣹدَسُرْ؞ ‏21 ݧَ ݧُ لِمَلْ گࣵنگࣷورَمْ دِ فَنْوࣹيرِ جُنّࣹيڪْ ݧَارْ اَڪْ ݧَارْ تࣺيمࣺيرْ (32200)؞
‏22 «گِيرُگْ بࣹݧَمِنࣹينْ ݧࣹييْ فࣹگّࣷوڪْ ݧࣷومْ، ڪِلِفَگْ اَسْڪَنُ بࣹݧَمِنْ دِ اَبِدَنْ دࣷومُ گِدࣷنِ؞ ‏23 ݧَ ݧُ لِمَلْ گࣵنگࣷورَمْ دِ فَنْوࣹيرِ جُنّࣹيڪْ جُرٗومْ اَڪْ ݧࣹينْتِ تࣺيمࣺيرْ (35400)؞
‏24 «مبࣷولࣹمْ ݧَ ݧُ لِمَلْ دَلُبْ اࣹفْرَيِمْ اَڪِ گࣵنگࣷورَمْ دَلْدِ دࣷونْ تࣺيمࣺيرِ جُنّࣹيڪْ جُرٗومْ ݧࣹتّ اَڪْ تࣺيمࣺيرْ (108100)؞ بُ ݧُيْ دࣹمْ، ݧࣷوݧࣷو دِ ݧࣹتّࣹيلُ ڪُرࣺلْ، دࣷخْ تࣷݒّ ݖَ؞
 
‏25 «وَا رَايَبْ دَنْ ݧࣷويْ دَلࣹ بࣴجْ-گࣵنَّارْ، ݧࣷوڪْ سࣹينِ گࣵنگࣷورْ، ڪِلِفَگْ اَسْڪَنُ دَنْ دِ اَيࣹسࣹرْ دࣷومُ اَمِسَدَايْ؞ ‏26 ݧَ ݧُ لِمَلْ گࣵنگࣷورَمْ دِ جُرٗومْ بࣹنّ فُڪِّ جُنّࣹيڪْ ݧَارْ اَڪْ جُرٗومْ ݧَارْ تࣺيمࣺيرْ (62700)؞
‏27 «ݧَيْ دَلࣹ فَ فࣹگّࣷوڪْ ݧࣷومْ ݧࣷويْ گِيرُگْ اَسࣹرࣹينْ ݧِ، ڪِلِفَگْ اَسْڪَنُ اَسࣹرْ دِ ݒَگْيࣹلْ دࣷومُ اࣷڪّرَنْ؞ ‏28 ݧَ ݧُ لِمَلْ گࣵنگࣷورَمْ دِ ݧࣹينتّ فُڪِّ جُنّࣹيڪْ بࣹنّ اَڪْ جُرٗومِ تࣺيمࣺيرْ (41500)؞
‏29 «گِيرُگْ نࣹفْتَلࣹينْ ݧࣹييْ فࣹگّࣷوڪْ ݧࣷومْ، ڪِلِفَگْ اَسْڪَنُ نࣹفْتَلِ دِ اَيْرَ دࣷومُ اࣹنَنْ؞ ‏30 ݧَ ݧُ لِمَلْ گࣵنگࣷورَمْ دِ جُرٗومْ فُڪِّ جُنّࣹيڪْ ݧࣹتّ اَڪْ ݧࣹينْتِ تࣺيمࣺيرْ (53400)؞
‏31 «مبࣷولࣹمْ ݧَ ݧُ لِمَلْ دَلُبْ دَنْ دَلْدِ دࣷونْ تࣺيمࣺيرْ اَڪْ جُرٗومْ فُڪِّ جُنّࣹيڪْ جُرٗومْ ݧَارْ اَڪْ جُرٗومْ بࣹنِّ تࣺيمࣺيرْ (157600)؞ ݧࣷوݧࣷويْ مُجَّ جٗگْ، تࣷݒّ سࣹينِ رَايَ؞»
 
‏32 ݧࣷوݧُ لَݧُ لِمَلْ بࣵنِّ إِسْرَاءِيلْ نَ ݧُ بࣷڪّࣹ سࣹينْ ڪࣴرِ مَامْ؞ مبࣷولࣹمْ ݧَ ݧُ لِمَلْ دَلْ يَ اَڪْ سࣹينِ گࣵنگࣷورْ دࣷونْ جُرٗومْ بࣹنِّ تࣺيمࣺيرِ جُنّࣹيڪْ ݧࣹتّ اَڪْ جُرٗومِ تࣺيمࣺيرْ اَڪْ جُرٗومْ فُڪّ (603550)؞ ‏33 لࣹوࣹينْ ݧِ نَگْ لِمَالࣹيسُ لࣹينْ ݖِ بࣵنِّ إِسْرَاءِيلْ، تࣹ نࣷونُ لَ ڪࣷ اَجِ سَخْ جِ سَنْتࣹ وࣷونْ مُوسَىٰ؞
‏34 بࣵنِّ إِسْرَاءِيلْ دَلْدِ دࣹفْ نࣷونُ؞ مبࣷولࣹمْ نࣷونࣹي ڪࣷ اَجِ سَخْ جِ سَنْتࣹ وࣷونْ مُوسَىٰ، نَ لَݧُ دَانْ دَلࣹ، تࣷݒّ سࣹينِ رَايَ، نَ لَݧُ دَانْ جٗگࣹ، ڪُ نࣹڪّ اَڪْ لࣵنگَمْ اَڪْ ڪࣴرْ مَامَمْ؞
