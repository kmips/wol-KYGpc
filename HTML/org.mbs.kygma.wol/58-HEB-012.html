بُلࣹينْ خࣹيبْ يَرُ بࣷرࣷومْ بِ
12
‏1 ڪࣷنْ نَگْ گَنَّاوْ گِي ندِيرَانُ سࣹيدࣹ وُ رࣺيْ مࣷو نُ يࣺيوْ، نَنُ سِݒِّڪُ مبࣷولࣹمْ سࣴفْ بُ نُيْ يࣹيخَلْ، اَڪْ بࣵڪَّارْ بُ نُ يࣷمْبَ تࣺقْتَلْ؞ ݧُ دَوْ نَگْ خࣺلْ وِ ݧُ نُ وَاجَلَلْ، تࣹ بَݧَ يࣷقِ؞ ‏2 نَنُ نࣹ جࣵڪّ عِيسَىٰ، ڪِيْ سࣷسْ، دِ سࣷتَّلْ نگࣴمْ، مࣷومْ مِ مبࣺگْ مِ ݧُ ڪࣷ وَاجَلَلْ تَخْ مُ دࣺڪُّ بَنتّ بَ ݧُ ڪࣷ دَاجْ، تࣹگُلْ مُيْ دَرَ تࣷرࣷخْتَانگࣹ جَ ڪࣷ ݖَ فࣹڪّ، تࣹ دࣷخࣹ ݖَ تࣷوگْ ݖَ ندِجࣷورُ جَلُبْ يࣵلَّ؞ ‏3 خࣷولْ-لࣹينْ ݖِ مࣷومْ مِ بࣵڪَّارْڪَتْ يِ دࣷونْ خࣹيخْ، بَلَ نگࣹينَ يࣷقِ بَيْ خࣵدِّ؞ ‏4 يࣹينْ ڪَتْ خَرࣹيگُلࣹينْ اَڪْ بࣵڪَّارْ، بَ سࣹينْ دࣹرࣹتْ تُورُ ݖِ؞ ‏5 تࣹ اِتْ فࣵتّࣹ نگࣹينْ ڪࣵدُّ يِي ݧُ لࣹينْ ݧَاخࣹ، نِ نِتْ اَڪِ دࣷومَمْ:‏
«دࣷومْ، بُلْ خࣹيبْ يَرُ بࣷرࣷومْ بِ،
بُلْ خࣵدِّ، بُ لَ وَݧّࣹي،
‏6 ندَخْ بࣷرࣷومْ بِ، ڪُ مُ سࣷݒّ، يَرْ لَ،
مࣷويْ بَنْتَلْ جࣺݒّ دࣷومْ جُ ڪࣷ نࣹيخْ¤؞»
‏7 مُݧْلࣹينْ ݖࣷونࣷ بُ لࣹينْ دِ يَرْ؞ نِ اَيْ دࣷومْ لَ يࣵلَّ دِ جࣴفْلَنْتࣹيڪْ يࣹينْ؞ اَنَ دࣷومْ جُ بَايْ بَ دُلْ يَرْ؟ ‏8 بُ ݧُ لࣹينْ خَݧࣹي يَرْ بِ جࣺݒّ دࣷومْ اَمْ وࣵلّ، دَفَ فࣹڪّ نگࣹينْ دِ دࣷومِ ݒࣷ، وَايࣹ دُ نگࣹينْ دࣷومِ يࣷونْ؞ ‏9 تࣹ اِتَمْ اَمْ نَنُ اَيْ بَايْ ݖِ وࣵلُّ دࣹرࣹتْ، ݧࣷو نُ دَانْ يَرْ، تࣹ دَنُ لࣹينَ وࣹگࣷونْ؞ ڪࣷنْ نَگْ خَنَا دُ بَايْ بِ يࣷرُگْ نࣷو لَنُ وَرَ گࣴنَتࣹي نَنگُلْ، نگِرْ نُ دُندّ؟ ‏10 سُنُيْ بَايَ نُ يَرْ فَنْ يُ نࣺيوْ، نَ مُ جَادࣷو وࣷونْ ݖِ سࣹينُمْ خࣹلْ؞ وَايࣹ سُنُ بَايْ بِي، سُنُ نجَرِݧَلْ بࣷݒّ لَنُيْ يَرࣹ، نگِرْ نُ سࣺدُّ ݖِ سࣹلاَّيَمْ؞ ‏11 دࣴگّ لَ، بࣹنّ يَرُبْ تࣹيْ نࣹيخُلْ، خَنَا نَقَرِ؞ وَايࣹ لَ يَرْ بَيْ مُجّ جُرَلْ ݧَ مُ تࣵگَّتْ، جࣵمّ لَ اَڪْ نجُبْ؞
‏12 ڪࣷنْ نَگْ «دࣴگࣴرَلْ-لࣹينْ لࣷخࣷ يِ تࣷقِ، اَڪْ اٗومْ يِ بَݧْ¤،» ‏13 تࣹ «نگࣹينْ خࣵلَّلْ سࣹينْ بࣷݒّ خࣵلّ يُ جُبْ يُ نگࣹينْ دِ جَارࣹ سࣹينِ تࣵنڪّ¤،» نگِرْ ڪُيْ سࣷوخْ دࣷخࣹ ݖَ وࣺرْ تࣹ بَݧَ گࣴنَ تࣴلࣹ؞
بُلࣹينْ تَنقَمْلُ يࣵلَّ
‏14 سࣵڪُّلࣹينْ جࣵمّ سࣹينْ دِگَّنْتࣹ اَڪْ ݧࣺݒّ، تࣹ اِتْ سࣵڪُّلࣹينْ سࣹلّنگَ، ندࣹ ڪُ سࣹلُّلْ دُ گِسْ بࣷرࣷومْ بِ؞ ‏15 مࣷيْتُلࣹينْ بَلاَ ڪࣹنّ دِ ݧࣵڪّ يِوُ يࣵلَّ؞ مࣷيْتُلࣹينْ بَلَ اَبْ رࣹينُ وࣹخْتَنْ دِ فࣴلّ، دِ لࣴجْلࣹ، بَ رَخْ مبࣷولࣷومْ لࣴمّ؞ ‏16 مࣷيْتُلࣹينْ بَلَ ڪࣹنَّ يࣹبُ ݖِ ݒࣷوُمْ سࣺيْ، مبَا مُ تࣹدَّدِلْ لُ سࣹلّ بَ مࣹلْ نِ اࣹسَوُ مِ وࣹݖّࣹي وࣷونْ اَگْ لࣹڪّ، ݖࣺرُ تَاوْ بَ مُ يࣹلّࣷو؞ ‏17 خَمْ نگࣹينْ نࣹ گَنَّاوْ گِ، بَ مُ بࣴگّࣹي جࣷتْ ݖَ بَرْڪࣹ بَ، دࣹيسْ ڪࣷو گࣵنْتَلْ؞ اَمُلْ وࣷونْ فࣹنّ فُ مُ جَارࣹ اِتْ بَ سࣷݒَّرْݧِ دࣹمِنَمْ، تࣹ جࣺيمْ نَ ڪࣷ بَيْ جࣷويْ؞
‏18 يࣹينْ ڪَتْ دُ يࣹينَ اَگْسِ وࣷونْ بَ جࣹگࣹ تُندّ وُ نگࣹينْ مَنَ لاَلࣹ سࣹينْ لࣷخࣷ، مُيْ تࣵڪّ، اࣵندّ اَڪِ خࣵمْبَارْ، اَڪْ لࣴنْدࣴمْ گُ نࣹ ڪࣴرُوسْ، اَڪْ نگࣹلَوْ لُ رࣺيْ؞ ‏19 دُ يࣹينَ دࣺگّ لِيتْ گُ جِبْ، اَڪْ ݖࣷووَلْ بَاتْ بُ جِبْ تࣹ ݧَيْ دࣺگّ ڪࣵدُّ يَ دِ ݧَانْ ݧُ بَݧْ لࣹينَ دࣺگْتَلَتِ بࣹنّ بَاتْ بُ نِ مࣹلْ؞ ‏20 مَنُݧُ وࣷونَ خَجࣷو ندِگَلْ لَ نࣹ:‏ «مَلَ سَخْ، سُ لاَلࣹي تُندّ وِ، دٗورْلࣹينْ ڪࣷيْ دࣷجْ بَ مُ دࣹي¤؞» ‏21 ݒࣹيݧُ مࣷومَ مࣷو رَگْلُ وࣷونْ بَ مُوسَىٰ نࣹ:‏ «دَمَا تِيتْ بَيْ لࣷخْ¤؞»
‏22 وَايࣹ يࣹينْ فِ نگࣹينْ اَگْسِ مࣷو دِ تُنْدُ سِيࣷݝْ، دࣴڪُّ يࣵلَّ جِيْ دُندّ، تࣹ مࣷويْ يࣹرُسَلࣹمْ گَ ݖَ اَسَمَانْ، يࣹينَ اࣵگّ فَ جُنِّيْ جُنِّيْ مَلاَڪَ، دَجࣹي دِ مࣵگَّلْ؛ ‏23 يࣹينَ اࣵگّ فَ ندَجࣹمْ تَاوْ، يَ سࣹينِ تُرْ بِنْدࣹ فَ اَسَمَانْ؞ يࣹينَ اࣵگّ فَ يࣵلَّ، اࣵتّࣹڪَتُبْ ݧࣺݒّ؛ يࣹينَ اࣵگّ فَ نࣷووِ اَجِ جُبْ ݧَ ݧُ مَتَلْ سࣴڪّ؞ ‏24 يࣹينَ اࣵگّ فَ عِيسَىٰ، رَمُّڪَتْ بِ دࣷخْ ڪٗلّࣴرࣹ گُ يࣹيسْ گِ، يࣹينَ اࣵگّ فَ دࣹرࣹتْ جِيْ سࣹتَلْ تُورࣷو، تࣹ ڪࣵدُّگْ دࣹرࣹتْ جَ گࣴنَ نࣹيخْ ڪࣵدُّگْ دࣹرࣹتُ اَبࣹلْ گَ جِبࣷونْ؞
‏25 وَتُّلࣹينْ بَلَ نگࣹينَ تَنقَمْلُ ڪِيْ وَخْ اَڪْ يࣹينْ؞ ندࣹگَمْ ݧَ تَنقَمْلُ وࣷونْ ڪَ يࣺڪَّتِ وࣷونْ ڪࣵدّࣷومْ، وَخْ اَڪْ ݧࣷومْ ݖِ ڪَوْ سُوفْ ݧࣷو رࣴݖُّلْ، اَسْتࣹ نُنْ ݧِي، سُ نُ دࣴدّࣷو ڪِيْ وَخࣹ فَ اَسَمَانْ؞ ‏26 بࣷوبَ بَاتَمْ لَ يࣴنگَلࣹ وࣷونْ سُوفْ، وَايࣹ تࣹيْ مࣷو دِگࣹ نࣹ:‏ «بࣹنࣹينْ يࣷونْ، مَايْ يࣴنگَلَتِ سُوفْ، تࣹ دُ سُوفْ دࣷݝّ، وَايࣹ اَسَمَانْ اِتَمْ¤؞»
‏27 لِ مُ نࣹ «بࣹنࣹينْ يࣷونْ»، دِ وُندّ نࣹ لࣹيسْ مَنَ يࣴنگَلْ، لَيْ دِنْدِ، تࣹ مࣷويْ لِ سࣵڪّࣹ، نگِرْ لࣹيسْ مَنُلَ يࣴنگَلْ سَخْ بَ فࣵوّ؞
‏28 ڪࣷنْ نَگْ گَنَّاوْ نگُورْ گُ دࣹيسُلْ يࣴنگَلْ لَنُ جَگࣷو، نَنُ جࣵݒّࣷو ݖَنتّ، دِ جَامُ يࣵلَّ نِ مُ ڪࣷ بࣴگّࣹ ݖِگْ نجُلِّتࣹ اَڪُگْ رَگَلْ، ‏29 ندَخْ ڪَتْ «سُنُ يࣵلَّ، سَوَرَ وُيْ خࣷيࣷمࣹ لَ¤؞»
