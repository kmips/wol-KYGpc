ݒٗولْ دُگّ نَ گَالْ، جࣴمْ رࣷومْ
27
‏1 بَ لࣷولُ وࣺيࣹي ݧُ دࣷگُ نࣷو دُگَلْ گَالْ، جࣴمࣹ نُ رࣺيوُمْ اِتَلِ؞ ݧُ تࣹگْ ݒٗولْ اَڪْ ݧࣹنࣹينْ ݧُ ݧُ تࣴجࣷونْ، ݖِ لࣷخࣷولْ نجِيتُ تَڪّ-دࣹرْ بُ ݧُيْ وَخْ يُلْيُسْ، بࣷڪّ ݖِ مبࣷولࣷومْ خَرࣹمْ بُورْ بُ مَگْ بِ؞ ‏2 نُ دُگّ گَالُ دࣴڪّ بَ ݧُيْ وَخْ اࣵدّرَمِتْ تࣹيْ جَارࣹجِ تࣹيرُيْ اَسِ، دَلْدِ تࣵبِّ بِيرْ گࣺيجْ، اَرِسْتࣵرڪّ جࣹنّ وَايْ جُ دࣴڪّࣹ تࣹسَلࣷنِگْ ݖَ دِيوَانُ مَسࣹدُوَنْ، اࣵندّ اَڪْ نُنْ؞
‏3 ݖَ اࣴلّࣴگْ سَ نُ تࣹيرِ سِدࣷنْ؛ يُلْيُسْ نَگْ بَاخࣹ ݒٗولْ، مَيْ ڪࣷ مُ دࣹمْ ݖِ اَيْ خَرِتَمْ، نگِرْ ݧُ تࣷݒّتࣷو ڪࣷ؞ ‏4 نُ جٗگࣹ فَ، دࣹلُّ گࣺيجْ، نگࣹلَوْ لِ سࣷفْلُ نُ، نُ لࣹرُ تࣹفࣹسُ سِݒَّرْ، جَارࣹ وࣹتْ گُ نُ فࣹگࣷو نگࣹلَوْ لِ؞ ‏5 ݖِ ڪَوْ لࣷولُ نُ جࣵلّ گࣺيجْ، گِ جَنࣷوڪْ دِيوَانِ سِلِسِ اَڪْ ݒَمْفِلِ، دَلْدِ تࣹيرِ مِرَ ݖَ لِسِ؞ ‏6 فࣷوفَ نَگْ نجِيتُ تَڪّ-دࣹرْ بَ گِسْ گَالْ گُ جٗگࣹ اَلࣹگْسࣵنْدِرِ، جࣴمْ اِتَلِ، مُ دُگَلْ نُ ݖَ؞ ‏7 بَ لࣷولُ اَمࣹي نُيْ دࣹمࣹ ندࣵنڪّ اَيْ فَنِ فَنْ؛ سࣷنّ نَنُ لࣷولْ دࣷورَ جُبْ ڪَنِدّ، وَايࣹ نگࣹلَوْ لِ مَيُ نُ نُ تࣹيرْ؞ نُ لࣹرُ تࣹفࣹسُ ڪࣹرࣹتْ، جَارࣹ وࣹتْ گُ نُ فࣹگࣷو نگࣹلَوْ لِ، جَنࣷوڪْ سَلْمࣷنْ؞ ‏8 سࣷنّ نَنُ لࣷولْ بَلَ نࣷو رࣷمبّ فࣷوفُ، دࣷورَ اَگْسِ فَ ݧُيْ وَخْ نࣹيخْ تࣹيرُ، تࣹ دࣹندّ اَڪْ دࣴڪّ بَ ݧُيْ وَخْ لَسࣹ؞
‏9 فࣹڪّ نَ نَگْ نُ يࣵگّ ݖَ يࣷونْ وَ، بَ تࣷلُّ ݖِ جَمࣷنࣷ جُ تُڪِّبْ گࣺيجْ آيْتَلْ، ندَخْ فࣹڪّ نَ ڪࣷورُگْ يَوُتْ¤ وࣹيسْ؞ لࣷولُ تَخْ ݒٗولْ دِگَلْ لࣹينْ نࣹ:‏ ‏10 «بࣷڪّ يِ، گِسْ نَا نࣹ يࣷونْ وِي لࣷرَانگࣹ اَڪْ يࣵقُلࣹ گُ رࣺيْ لَ؞ وَخُمَ گَالْ گِ اَڪْ لِ مُ يࣹبْ، وَايࣹ سُنُيْ بَڪَّنْ سَخْ خَجْ نَ ݖِ؞» ‏11 تࣹيوُلْ نجِيتُ تَڪّ-دࣹرْ بَ سࣷبࣹنْتَلْ وَخِ ݒٗولْ، دࣺگَّلْ دَوَلْڪَتْ بَ اَڪْ بࣷرࣷومْ گَالْ گَ؞ ‏12 تࣹ اِتَمْ تࣹيرُ بࣷوبُ نࣹيخُلْ وࣷونَ لࣷلِّڪࣷو؞ لࣷولُ وَرَلْ ݧَ اࣴݒّ ݖَ وَا گَالْ گَ مࣵنڪࣷو ݖِ جٗگࣹ فَ، دࣹلُّ گࣺيجْ، تࣹ فࣹخࣹ نُ ݧُ اࣵگّࣹ فࣹنِگْسࣴ، نگِرْ لࣷلِّڪࣷو فَ؞ مࣷويْ تࣹيرُبْ دُنُ ڪࣹرࣹتْ بِ جُبْلُ سࣷوُبْ سُوفْ اَڪْ سࣷوُبْ ڪَوْ؞
نگࣴلࣺنْ سࣵقِ نَ
‏13 ݖِ ڪَوْ لࣷولُ نگࣹلَوَلْ بࣴجْ-سَالُمْ وَلْ ندࣵنڪّ، ݧُ نجࣷرتّ نࣹ مَنْ نَݧࣷو سࣷتَّلْ سࣹينْ ݒࣹخࣹ؞ ݧُ وࣴگّ دِيگَلْ، دَلْدِ تَفࣷو تࣹفࣹسُ ڪࣹرࣹتْ؞ ‏14 تࣹيوُلْ نࣹسْ تُوتْ نگࣹلَوْ لُ رࣺيْ لُ ݧُيْ وَخْ ݒࣹنڪُبْ ڪَوْ، رِدّࣹي ݖَ دُنْ بَ؞ ‏15 گَالْ گِ بُوبُ، مَنَتُلَ دࣺڪُّ نگࣹلَوْ لِ، بَ نُ مُجّ بَيْلِڪُ، دَلْدِ يَلْ؞ ‏16 نُ لࣹرُ نَگْ دُنْ بُ تُوتِ بُ ݧُيْ وَخْ ڪࣷدَ، جَارࣹ وࣹتْ گُ نُ فࣹگࣷو نگࣹلَوْ لِ، تࣹگْ ݖِ سࣷنّ لࣷولْ دࣷورَ تࣺيࣹ لࣷوݖࣷبْ رَوْتُ بُ گَالْ گِ؞ ‏17 ݧُ يࣺيگࣹ لࣷوݖࣷ بَ، دَلْدِ جࣴلْ اَيْ بُومْ يُ ݧُ لَخَسْ ݖَ يَرَمُ گَالْ گَ، دࣵبّلࣹي ڪࣷ ڪࣷ؞ ݖِ ڪَوْ لࣷولُ ݧُ رَگَلَ نگِرࣷجِ ݖَ بࣹيݧْ بُ ݧُيْ وَخْ سِرتّ، بَ تَخْ ݧُ سࣵنِّ دِيگَلْ، نگِرْ يࣹيخَلْ گَالْ گَ، دَلْدِ يَلَلْ؞ ‏18 ݖَ اࣴلّࣴگْ سَ نگࣹلَوْ لَ وࣺيࣹ نࣷو ݒَمْتِ-ݒَمْتࣹي، بَ ݧُ مُجّ وࣵݧِّ ݖَ يࣹبُ گَالْ گَ، سࣵنِّ بِيرْ گࣺيجْ؞ ‏19 ݖَ گَنَّاوْ اࣴلّࣴگْ سَ، سࣹينْ لࣷخࣷيْ بࣷݒّ لَݧُ جࣴلࣹ يࣹنّ جُمْتُڪَايِ گَالْ گَ، سࣵنِّ؞ ‏20 اَيْ فَنِ فَنْ جَنتّ بࣹيڪْ بِدِّيوْ يِ نࣹ مࣹݝّ، تࣹ نگࣹلَوْ لِيْ رِدِّ رࣹڪّ، بَ نُ نࣷݒّࣹي يَاڪَارْ نࣹ دِنَنُ مُݖّ؞
‏21 لࣷولُ نَگْ فࣹڪّ نَ ݧُ گࣴجࣷونَ لࣹڪّ؞ ݒٗولْ دَلْدِ تَخَوْ ݖَ سࣹينْ بِيرْ نࣹ:‏ «بࣷڪّ يِ، سُ نگࣹينْ يࣺگࣷونْ دࣺگَّلْ مَ، بَݧَ جٗگࣹ ڪࣹرࣹتْ، بَ مࣷيُ لࣷرَانگࣹ گِي اَڪْ يࣵقُلࣹ گِي؞ ‏22 لࣺيگِ نَگْ دَمَ لࣹينْ دِ ݧَاخْ، نگِرْ نگࣹينْ دَلَلْ سࣹينْ خࣹلْ، ندَخْ ڪࣹنّ ݖِ يࣹينْ دُ ݧࣵڪّ بَڪَّنَمْ، گَالْ گِ رࣹڪّ اَيْ يࣵقُ؞ ‏23 ندَخْ مَلاَڪَمْ يࣵلَّ جِ مَ مࣷومْ، تࣹ مَ دِ ڪࣷ جَامُ، مࣷو مَ دِڪَّلْ بِيگْ؞ ‏24 مُ نࣹ مَ:‏ ‹ݒٗولْ، بُلْ تِيتْ؛ فࣵوّ نگَ تَخَوِ فَ ڪَنَمْ سࣹسَارْ، تࣹ اِتْ يࣵلَّ مࣷو لَ بَاخࣹ مُݖُّگْ مبࣷولࣹمْ ݧِ نگَ اࣵنْدَلْ ݖِ گَالْ گِ؞› ‏25 ڪࣷنْ نَگْ، بࣷڪّ يِ، دَلَلْ-لࣹينْ سࣹينْ خࣹلْ، ندَخْ گࣴمْ نَا يࣵلَّ، خَمْ نَا نࣹ نِ مُ ڪࣷ وَخࣹ، نِ لَيْ اَمࣹ؞ ‏26 دُ ݧࣵڪّ اَوْ دُنْ وُ نُ لُفࣹجِ؞»
گَالْ گَ تَسْ نَ
‏27 ݖَ فُڪّࣹيلُ بࣺسْ بَاڪْ ݧࣹينتّ، نُ نگَيْ يَلْ-يَلࣹي فُ نࣹ ݖَ گࣺيجُ اَدِرَتِگْ؛ بَ گُدِّ گَ خَاجࣹي مࣷولْ يَ نجࣷرتّ نࣹ نُ نگَيْ جࣹگࣹسِ اَبْ تࣹفࣹسْ؞ ‏28 ݧُ سࣵنِّ نَتُّڪَايْ بَ بِيرْ گࣺيجْ، گِسْ ندࣷخْ مَ خٗوتࣹ ݧَارْ فُڪِّ دَيࣷ، خَوَ يࣹمࣷوڪْ ݧࣹينتّ فُڪِّ مࣹيتَرْ؞ ݧُ دࣹمَتِ تُوتِ، سࣵنِّوَاتْ ڪࣷ، دَلْدِ گِسْ ندࣷخْ مَ خٗوتࣹ فَنْوࣹيرِ مࣹيتَرْ، خَوَ يࣹمࣷوڪْ فُڪِّ دَيࣷ اَڪْ جُرٗومْ؞ ‏29 ݖِ ڪَوْ لࣷولُ ݧُ رَگَلَ لُفࣹجِ بࣺرَبْ بُ اَمِ دࣷجْ، بَ تَخْ ݧُ سࣵنّࣹي ݖَ گࣹينُ گَالْ گَ، ݧࣹينْتِ دِيگَلْ، تࣹيْ ݧَانْ جَنتّ فࣹنڪࣹ لࣹينْ؞ ‏30 تࣹيوُلْ مࣷولْ يَ ݧࣷومْ دِ فࣹخࣹي رَوْ، دَلْدِ دࣹمْ بَ يࣷورِ لࣷوݖࣷبْ رَوْتُ بَ بِيرْ گࣺيجْ، تࣹيْ تَافَنْتࣷو نࣹ اَيْ دِيگَلْ لَݧُيْ سࣵنّࣹي ݖَ بࣷݒُّ گَالْ گَ؞ ‏31 ݒٗولْ نَگْ نࣹ نجِيتُ تَڪّ-دࣹرْ بَ اَڪْ خَرࣹڪَتْ يَ:‏ «ݧِي بُ ݧُ دࣹسُلْ ݖِ گَالْ گِ، يࣹينْ دُنگࣹينْ رَوْ؞» ‏32 ݖَ لَ تَڪّ-دࣹرْ يَ دࣷگْ بُومْ يَ تࣺيࣹ لࣷوݖࣷ بَ، بࣵيِّ، مُ وࣺيْ؞
‏33 نَڪَ لَ جَنتّ دِ وَاجَ فࣹنڪّ، ݒٗولْ ݧَاخْ ݧࣺݒّ، نگِرْ ݧُ لࣹڪّ؛ مُ نࣹ لࣹينْ:‏ «فُڪِّ فَنْ اَڪْ ݧࣹينْتَ نگِي، يࣹينَ نگِ تࣷوگْ، لࣹڪُّلࣹينْ، سࣹخُلࣹينْ دَرَ؞ ‏34 دَمَ لࣹينْ دِ ݧَاخْ نَگْ نگِرْ نگࣹينْ لࣹڪّ، ندَخْ نࣷونُ دࣷݝّ نگࣹينْ مَنَ رَوࣹ؞ دُ گࣹنّ ڪَوَرْ سَخْ گُيْ رࣷتࣹ ݖِ بࣷݒُّ ڪࣹنّ ݖِ يࣹينْ؞» ‏35 لࣷولُ لَ وَخْ، دَلْدِ جࣴلْ مبُرُ، سَنتّ يࣵلَّ ݖِ ڪَنَمْ ݧࣺݒّ، بَ نࣷݒِّ دَگّ، لࣹڪّ؞ ‏36 بَ لࣷولُ اَمࣹي، ݧࣺݒّ دࣹلُّ سَوَرْ، دَلْدِ لࣹڪّ ݧࣷومْ اِتَمْ؞ ‏37 ݧَ نࣹڪّࣷونْ ݖَ گَالْ گَ ݧࣺݒّ ݧَارْ تࣺيمࣺيرْ اَڪْ جُرٗومْ ݧَارْ فُڪّ اَڪْ جُرٗومْ بࣹنّ (276) لَݧُ؞ ‏38 ݧُ لࣹڪّ بَ سُورْ، دَلْدِ سࣵنِّ ݒࣹݒّ مَ بِيرْ گࣺيجْ، نگِرْ وࣷيࣷفَلْ گَالْ گَ؞
‏39 بَ بࣴتْ سࣹتࣹي، خࣵمِّوُݧُ رࣺيوْ مَ، وَايࣹ اَبْ ݖَتُ تࣹفࣹسْ لَݧُ سࣺينْ؞ ݧُ فَسْ يࣺينࣹ ݖَا تࣹيرَلِ، ندࣹگَمْ مَنُ نَ؞ ‏40 بَ مُ ڪࣷ دࣹفࣹي ݧُ دَگّ دِيگَلْ يَ، وَݖّ بِيرْ گࣺيجْ، يِوِوَالࣹ گࣷجْ يَ تࣺيࣹ جࣷووُ يَيْ دَوَلْ گَالْ گَ؞ ݖِ ڪَوْ لࣷولُ ݧُ فِرِ وِيرُ ڪَنَمْ گَ، وَݖّࣹ اَڪْ نگࣹلَوْ لَ، نُ وُتِ تࣹفࣹسْ گَ؞ ‏41 تࣹيوُلْ ݧُ دَلْ فُ ݧَارِ ندَوْ دَجࣹ، ݖِ اَبْ تُنْدُ بࣹيݧْ، گَالْ گَ لُفْ؞ بࣷݒُّ گَالْ گَ نُورْ ݖَ سُوفْ سَ، يࣴنگࣷوتُلْ، گࣹينُ گَالْ گَ تَسْ ندَخْ دࣷولࣹيْ گَنَّخْ يَ؞
‏42 بَ لࣷولُ اَمࣹي تَڪّ-دࣹرْ يَ نَرَ رࣹيْ ݧَ ݧُ تࣴجْ، بَلَ ݧࣷو فࣺييْ، بَ رَوْ؞ ‏43 نجِيتُ تَڪّ-دࣹرْ بَ نَگْ نَمَّ رَوَلࣹ بَڪَّنُ ݒٗولْ، بَ تَخْ مُ آيࣹ لࣷولُ؞ مُ سَنتّ ݧِ مَنَ فࣺييْ، نࣹ لࣹينْ ݧُ جࣴڪَّ سٗوبُ، وُتِ تࣹفࣹسْ گَ؞ ‏44 ݧَ ݖَ دࣹسْ نَگْ، تࣷݒّ ݖَ، ݧِي لَنگَامࣷو اَيْ دࣺنڪّ، ݧࣹي، تࣷجِتِ گَالْ گَ؞ نࣷونُ لَ ݧࣺݒّ مُجّࣹ جࣷتْ تࣹفࣹسْ گَ ݖِ جࣵمّ؞
