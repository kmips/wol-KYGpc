عِيسَىٰ دࣺڪُّ نَ ݒࣹخࣹيْ سࣹيْتَانࣹ
4
‏1 عِيسَىٰ مࣷو فࣹيسࣹ وࣷونْ نࣷو گُ سࣹلّ گِ، بࣵيِّڪࣷو دࣹخُ يُرْدَنْ، ݖِ نجِيتَلْ نࣷو گِ، ݖَ بِيرْ مࣵنْدِݝْ مَ؞ ‏2 ݧࣹينتّ فُڪِّ فَنْ لَ ڪࣷ فَ سࣹيْتَانࣹ فِيرْ، لࣹڪُّلْ دَرَ دِيرْ بࣷوبُ بࣺݒّ؞ گَنَّاوْ بَ فَنْ يࣷويُ وࣺيࣹي، مُ خِيفْ؞
‏3 سࣹيْتَانࣹ نَگْ نࣹ ڪࣷ:‏ «گَنَّاوْ يَا دِ دࣷومُ يࣵلَّ، نࣹيلْ دࣷجْ وِي مُ سࣷݒّلِڪُ مبُرُ؞» ‏4 عِيسَىٰ نࣹ ڪࣷ:‏ «بِنْدࣹيسْ نَ نࣹ دُ اَوْ ݧَمْ دࣷݝّ لَ نِتْ دِ دُنْدࣹ؞»
‏5 سࣹيْتَانࣹ يࣺيگࣹ نَ ڪࣷ اِتَمْ، وࣷنْ ڪࣷ مبࣷولࣹمْ نگُورِ اࣵدِّنَ، ݖِ خࣹفْ اَڪْ خِݒِّ؞ ‏6 سࣹيْتَانࣹ نࣹ عِيسَىٰ:‏ «يَوْ لاَيْ مَيْ مبࣷولࣹمْ ڪِلِفْتࣹيفْ گِي اَڪْ دَرَجَ جِ مُ اࣵنْدَلْ، ندَخْ مَنْ لࣹيسْ ڪࣷ جࣷخْ، تࣹ ڪُ مَ سࣷوبْ لاَ ڪࣷيْ مَيْ؞ ‏7 ڪࣷنْ نَگْ سࣷو مَ سُجّٗوتَلࣹي، يَا مࣷومْ لࣺݒّ؞» ‏8 عِيسَىٰ نࣹ ڪࣷ:‏ «بِنْدࣹيسْ نَ نࣹ:‏ ‹بࣷرࣷومْ بِ سَ يࣵلَّ نگَيْ سُجّٗوتَلْ، تࣹ مࣷومْ دࣷݝّ نگَيْ جَامُ؞›»
‏9 سࣹيْتَانࣹ يٗبُّ نَ ڪࣷ يࣹرُسَلࣹمْ، اَجْ ڪࣷ ݖَ نجࣷبَّخْتَلُ ڪࣴرْ يࣵلَّ گَ، نࣹ ڪࣷ:‏ «گَنَّاوْ يَا دِ دࣷومُ يࣵلَّ، تࣴبࣹيلْ فِي، وࣵݖّ، ‏10 ندَخْ بِنْدࣹيسْ نَ نࣹ:‏
‹اَيْ مَلاَڪَامْ لَيْ جࣷخْ ندِگَلْ ݖِ سَ مبِرْ،
ݧُ سࣵمّ لَ، ‏11 تࣹ سࣹينِ لࣷخࣷ لَݧُ لَيْ لࣹيوࣷو،
بَ دࣷو فَڪّتَلࣷو اَوْ دࣷجْ؞›»
‏12 عِيسَىٰ نࣹ ڪࣷ:‏ «بِنْدࣹيسْ نَ نࣹ:‏ ‹بُلْ سࣹيتْلُ بࣷرࣷومْ بِ سَ يࣵلَّ؞›»
‏13 سࣹيْتَانࣹ لَلَلْ نَ عِيسَىٰ گࣺݒّ فِيرْ بَ سࣷتَّلْ، دࣷورَ دࣹمْ، بࣵيِّ ڪࣷ اَبْ دِيرْ؞
لُ جࣴمْ ݖِ عِيسَىٰ فَ نَسَرࣹتْ
‏14 بَ لࣷولُ اَمࣹي عِيسَىٰ دࣹلُّ گَلِلࣹ ݖِ مَنࣷورࣹيْ نࣷو گِ، اَبْ جࣴوَمْ نَگْ لَوْ ݖَ دِيوَانْ بَ بࣺݒّ بَ مُ دَجْ؞ ‏15 مُيْ جࣵنگّلࣹ ݖَ سࣹينِ جࣵنگُ، ݧࣺݒّ دِ ڪࣷ ڪَݧْ؞
‏16 عِيسَىٰ مࣷو دِڪّࣷونْ نَسَرࣹتْ گَ مُ فࣹڪّ بَاخْ، دَلْدِ دُگّ ݖَ جࣵنگُ بَ، نَ مُ ڪࣷ بَاخࣷو وࣷونْ ݖِ بࣺسُبْ نࣷفْلاَيْ؞ مُ تَخَوْ نگِرْ بِرَلْ اَبْ دࣷگْ؞ ‏17 ݧُ جࣷخْ ڪࣷ تࣺيرࣹبْ يࣷنࣹنتّ يࣵلَّ اࣹسَيِ؞ مُ اُبِّ تࣺيرࣹ بِ، گِسْ دࣷگْ بِ ݧُ بِندّ نࣹ:‏
‏18 «نࣷووُگْ بࣷرࣷومْ بَا نگِ فِ سَمَ ڪَوْ،
مࣷو مَ فَلْ،
نگِرْ مَ يࣺگَلْ نࣺيوْ-دࣷولࣹ يِ خِبَارُ جࣵمّ بِ؞
دَ مَا يࣹبَلْ نگِرْ مَ يࣺگَلْ جَامْ يِ نࣹ دࣹيسْ نَ لࣹينْ گࣷرࣹيلْ،
يࣺگَلْ گُمْبَ يِ نࣹ دِنَݧُ گِسْ،
يࣺگَلْ ݧِ ݧُ نࣷتْ، نࣹ دࣹيسْ نَ لࣹينْ جࣷتْ،
‏19 تࣹيْ يࣺگْلࣹ اَتُمْ يِوْ وُ بࣷرࣷومْ بِ؞»
‏20 ݖِ ڪَوْ لࣷولُ مُ اُبْ تࣺيرࣹ بِ، دࣹلّࣷو ڪࣷ جَوْرِݧْ بَ، دَلْدِ تࣷوگْ؞ ݧَ ݖَ جࣵنگُ بَ ݧࣺݒّ نَگْ نࣹ ڪࣷ جࣵڪّ؞ ‏21 مُ اࣵدُّ، نࣹ لࣹينْ:‏ «تࣹيْ جِي لَ مبِندّ مِي مَتْ، نگࣹينْ دࣺگّ؞» ‏22 ݧࣺݒَّ دࣷونْ سࣹيدࣹيلْ عِيسَىٰ لُ رَفࣹتْ، تࣹيْ يࣺيمُ ݖِ ڪࣵدُّيْ يِوْ يِ تُڪّࣹي ݖِ گࣺمِّݧَمْ؞ ݧُ نَانْ:‏ «خَنَا ڪِي دُ دࣷومُ يُوسُفَ جِ؟» ‏23 مُ نࣹ لࣹينْ:‏ «خَمْ نَا نࣹ دِنگࣹينْ مَ وَخَلْ وَخِنْ وِ ݧُ نَانْ:‏ ‹فَجْڪَتْ بِ، فَجَلْ سَ بࣷݒّ؞› مبࣷولࣹمْ لِ نُ دࣺگّ نࣹ مࣷو خࣹوْ ڪَݒࣹرْنَوُمْ، دࣹفْ ڪࣷ فِي ݖِ سَ رࣺيوُمْ بࣷݒّ اِتَمْ؞» ‏24 مُ نࣹيتِ:‏ «مَا لࣹينْ ڪࣷ وَخْ دࣺيْ، اَمُلْ بࣹنّ يࣷنࣹنتّ بُ ݧُ تࣹرَلْ ݖِ رࣺيوُمْ بࣷݒَّمْ؞ ‏25 مَا لࣹينْ وَخْ نࣹ لِيْ دࣴگّ مࣷو دِ، ݖَ جَمࣷنࣷيْ اِلْيَاسْ، بَ اَسَمَانْ تࣴجࣷو دِيرُ ݧࣹتِّ اَتْ اَڪْ جُرٗومْ بࣹنِّ وࣹيرْ، بَ خِيفْ بُ مࣹتِّ اَمْ ݖَ رࣺيوْ مࣺݒّ، اَيْ جࣴتُنْ ݧُ بَرࣹي نگَ وࣷونْ ݖَ إِسْرَاءِيلْ؞ ‏26 مࣷونَ دُ ڪࣹنّ ݖِ ݧࣷومْ ڪُ ݧُ يࣹبَلْ اِلْيَاسْ ݖَ مࣷومْ ڪُ مࣷيْ جࣴتُنْ بَ دࣴڪّࣹ سَرࣹبْتَ، ݖَ سِدࣷنْ؞ ‏27 جَمࣷنࣷيْ يࣷنࣹنتّ يࣵلَّ اَلْيَاسَ اِتَمْ، اَيْ گَانَ يُ بَرࣹي نگَ وࣷونْ ݖَ إِسْرَاءِيلْ؞ وَايࣹ فَجࣹيسُ ݖِ ڪࣹنّ ڪُ مࣷيْ نَامَنْ، وَا سِرِ بَ؞»
‏28 بَ مبࣷولࣹمْ وَا جࣵنگُ بَ دࣺگّࣹي لࣷولُ دَݧࣷو مࣹرْ بَ فࣹيسْ؞ ‏29 ݧُ جٗگْ، بࣴمࣴخْ ڪࣷ بَ گࣺنّࣹ ڪࣷ دࣴڪّ بَ، يٗبُّ ڪࣷ ݖَ ڪࣺمࣺجُ تُندّ، وَ سࣹينْ دࣴڪّ بَ سَنݖࣹ، نگِرْ خَلَبْ ڪࣷ ݖِ سُوفْ؞ ‏30 تࣹيوُلْ مࣷومْ مُ جَارࣹ ݖِ سࣹينْ بِيرْ، دࣹمْ يࣷونَمْ؞
عِيسَىٰ فَجْ نَ ڪُ رَبْ جࣵݒّ
‏31 بَ مُ ڪࣷ دࣹفࣹي عِيسَىٰ دࣹمْ ڪَݒࣹرْنَوُمْ، اَبْ دࣴڪّ ݖَ دِيوَانُ گَلِلࣹ، دِ لࣹينْ جࣵنگَلْ ݖَ بࣺسِ نࣷفْلاَيْ يَ؞ ‏32 نِتْ ݧَ نَگْ يࣺيمُ ݖَ نجࣵنگّلࣹيمْ، ندَخْ سَݧْ-سَݧْ بَ مُيْ وَخࣹ؞ ‏33 ندࣹڪࣹ فࣷوفَ ݖَ جࣵنگُ بَ، جࣹنّ وَايْ جُ رَبْ جࣵݒَّ نگَ فَ وࣷونْ؞ وَا جِ جࣹڪِّ خَاݖُ، نࣹ:‏ ‏34 «مࣷو يَوْ عِيسَىٰ مُ نَسَرࣹتْ، لُ نُ جࣷتࣹيڪْ يَوْ؟ اَلَگْسِ نُ مࣷويْ سَيْ تࣵنڪّ؟ خَمْ نَا يَا دِ ڪَنْ، يَا دِ اَجِ سࣹلّ جُ يࣵلَّ جِ!‏» ‏35 عِيسَىٰ نَگْ گࣴدّ ڪࣷ، نࣹ ڪࣷ:‏ «نࣷݒِّلْ تࣹ گࣺنّ نِتْ ڪِ!‏» رَبْ وَ دَانࣹيلْ نِتْ ڪِ ݖَ دِگّ مبࣷولࣷو مَ، دࣷورَ گࣺنّ تࣹ گَاݧُ ڪࣷ فࣹنّ؞
‏36 بَ لࣷولُ اَمࣹي ݧࣺݒَّ وَارُ، دِ وَخْ ݖِ سࣹينْ بِيرْ، نَانْ:‏ «گِي ڪࣵدُّ لُ مُ دࣷونْ؟ مࣷومْ دࣹ سَݧْ-سَݧْ اَڪْ خَمْ-خَمْ لَيْ جࣷخࣹ رَبْ يِ ندِگَلْ، ݧُ گࣺنّ؞» ‏37 ݖࣷووَلْ عِيسَىٰ نَگْ لَوْ بَ فࣺݒّ فُ بࣷڪّ ݖَ دِيوَانْ بَ؞
عِيسَىٰ فَجْ نَ جَرَگْ يُ بَرࣹ
‏38 عِيسَىٰ جٗگࣹ جࣵنگُ بَ، دࣹمْ ڪࣴرْ سِمࣷݝْ، دَلْدِ فࣹڪّ يَايُ جَبَرُ سِمࣷݝْ اࣵندّ اَڪْ تࣵنگࣷورُ يَرَمْ وُ رࣺيْ؞ ݧُ ݧَانْ ݖِ عِيسَىٰ ندِمْبَلْ؞ ‏39 مُ دِڪّ، تِيمْ سࣷخْنَ سَ، دَلْدِ گࣴدّ تࣵنگࣷورُ يَرَمْ وَ، تࣵنگࣷورْ وَ تࣹقْلِڪࣷوڪْ مࣷومْ؞ سࣷخْنَ سَ جٗگْ ݖَ سَا سَ، دِ لࣹينْ تࣷݒّتࣷو؞
‏40 بَ جَنتّ سࣷوࣹي، مبࣷولࣹمْ ݧَ اَمࣷونْ اَيْ جَرَگْ يُ اَمࣹ اَيْ جࣵنگࣷرࣷيْ جࣵنگࣷرࣷ، دَلْدِ لࣹينْ اِنْدِلْ عِيسَىٰ؞ مࣷومْ اِتْ، مُ تࣹگْ ڪࣹنّ ڪُ ݖِ نࣹڪّ اَيْ لࣷخࣷومْ، وࣺرَلْ لَ؞ ‏41 اَيْ رَبْ اِتَمْ دِ گࣺنّ ݧُ بَرࣹ، دِ يُوخُ، تࣹ نَانْ:‏ «يَايْ دࣷومُ يࣵلَّ!‏» تࣹيوُلْ مُ گࣴدّ لࣹينْ، آيࣹ لࣹينْ ݧُ وَخْ، ندَخْ خَمࣷونْ نَݧُ نࣹ مࣷويْ اَلْمَسِ؞
‏42 نجࣴلْ گُ جࣴڪّ ݖَ اࣴلّࣴگْ سَ لَ عِيسَىٰ گࣺنّ دࣴڪّ بَ، دࣹمْ بَ بࣺرَبْ بُ وࣺيتْ؞ مبࣷولࣷو مَ نَگْ دِ ڪࣷ سࣹيتْ، بَ دِڪّ، دِ ڪࣷ تࣺيࣹ، بَݧْ مُ دࣹمْ، بَ لࣹينْ؞ ‏43 مُ نࣹ لࣹينْ:‏ «يࣹنࣹينْ دࣴڪّ يࣹيتْ، وَرْ نَ مَ فَا يࣺگْلࣹجِ خِبَارُ جࣵمّ بُ نگُورُگْ يࣵلَّ، ندَخْ لࣷولُ لَݧُ مَ يࣹبَلࣹ؞» ‏44 ݖِ ڪَوْ لࣷولُ مُيْ وَارࣹ ݖَ جࣵنگُيْ دِيوَانُ يُدࣹ؞
