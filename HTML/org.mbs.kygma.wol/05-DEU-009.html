إِسْرَاءِيلْ گࣴنُلْ ݖِ خࣹيتْ يِ
9
‏1 إِسْرَاءِيلْ دࣺگْلُلࣹينْ!‏ يࣹينَ نگِي تࣹيْ، دِ جࣵلّ دࣹخُ يُرْدَنْ نگِرْ دࣵقِ خࣹيتْ يُ لࣹينْ اࣴݒّ، اࣴݒّ لࣹينْ دࣷولࣹ، نگِرْ نَنگُ سࣹينْ دࣴڪّ يُ رࣺيْ، دࣵبّلِڪࣷو تَتَ يُ اࣵڪِّ اَسَمَانْ؞ ‏2 اَسْڪَنْ وُ اَمْ ڪࣵتَّنْ، يࣹگّ ݖَ ڪَوْ، سࣷقِڪࣷو ݖِ اَنَگࣹينْ، ݧِ نگࣹينْ خَمْ، تࣹ دࣺگّ ݧُيْ وَخْ ݖِ ݧࣷومْ، نَانْ:‏ «ڪُ مَنَ تࣴ اَنَگࣹينْ ݧِ؟» ‏3 وَايࣹ خَمْلࣹينْ بࣺسْ نِڪِ تࣹيْ نࣹ سࣹينْ يࣵلَّ اَجِ سَخْ جِ مࣷويْ ڪِيْ جࣵلّ جِيتࣹ لࣹينْ، دِ سَوَرَ وُيْ خࣷيࣷمࣹ؞ مࣷو لࣹينْ دِ فࣵڪَّسْ، مࣷويْ ڪِ لࣹينْ لࣹينْ دِ نࣷتَلْ، نگࣹينْ دࣵقْ لࣹينْ، سࣵنڪّ لࣹينْ ݖِ لُ گَاوْ، نِ لࣹينْ ڪࣷ اَجِ سَخْ جِ وَخࣹ وࣷونْ؞
‏4 گَنَّاوْ بُ لࣹينْ لࣹينْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ دࣵقَلࣹي نَگْ، بُلࣹينْ وَخْ ݖِ سࣹينْ خࣹلْ نࣹ:‏ «سُنُ نجࣹڪَّ نُ مَيْ اَجِ سَخْ جِ اِنْدِ نُ، نگِرْ نُ مࣷومْ رࣺيوْ مِي؞» مبࣷنُگْ خࣹيتْ يِي ڪَيْ مࣷو وَرَلْ اَجِ سَخْ جِ دࣵقَلْ لࣹينْ لࣹينْ؞ ‏5 دُ سࣹينْ گࣹنّ نجࣹڪّ، اَڪُگْ نجُبْ گُ لࣹينْ مَيْ نگࣹينْ نَنگُجِ سࣹينُمْ رࣺيوْ؞ وَايࣹ مبࣷنُگْ خࣹيتْ يِي مࣷو وَرَلْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ دࣵقَلْ لࣹينْ لࣹينْ، نگِرْ سࣷتَّلْ ڪࣵدُّ گَ مُ گِݧَلࣷونْ سࣹينِ مَامْ إِبْرَاهِيمَ اَڪْ إِسْحَاقَ اَڪْ يَعْقُوبَ؞ ‏6 ڪࣷنْ نَ لࣹينْ بِرْ نࣹ دُ سࣹينُگْ نجࣹڪّ مࣷوسْ لَ لࣹينْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ جࣷخࣹ مِي رࣺيوْ مُ بَاخْ، نگِرْ نگࣹينْ مࣷومْ ڪࣷ ندَخْ، اَسْڪَنْ وُ دَݝِّيرَلْ نگࣹينْ؞
إِسْرَاءِيلْ جَامُ نَ سࣴلُّوْ وُرُسْ
‏7 فࣵتّلِڪُلࣹينْ!‏ بُلࣹينْ فࣵتّࣹ نَ نگࣹينْ مࣹرْلࣷو وࣷونْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ ݖَ مࣵنْدِݝْ مَ؞ لَ دَلࣹ بࣺسْ بَ نگࣹينْ بࣵيِّڪࣷو رࣺيوُمْ مِسْرَ، بَ بَ نگࣹينْ اَگْسࣹي بࣺرَبْ بِي، فِݒُّ ݖِ ڪَوْ اَجِ سَخْ جِ نگࣹينْ مَسَ نࣹڪّࣹ؞ ‏8 ݖَ تُنْدُ اࣷرࣹبْ نگࣹينْ مࣹرْلࣷو اَجِ سَخْ جِ، اَجِ سَخْ جِ مࣹرࣹ لࣹينْ، بَ نَرْ لࣹينَ فَاگَاگَلْ؞ ‏9 بَ مَ يࣺيگࣹي ݖَ تُندّ وَ، دِ جࣴلِ اࣵلُّوَيْ دࣷجْ يَ، اࣵلُّوَيْ ڪٗلّࣴرࣹ، گِ اَجِ سَخْ جِ فَسْ اَڪْ يࣹينْ؛ تࣷوگْ نَا ݖَ ڪَوْ تُندّ وَ ݧࣹينتّ فُڪِّ بࣴݖّࣴگْ اَڪْ ݧࣹينتّ فُڪِّ گُدِّ؞ دُ ݧَمْ وُ مَ لࣹڪّ، دُ ندࣷخْ مُ مَ نَانْ؞ ‏10 اَجِ سَخْ جِ جࣷخْ مَ ݧَارِ اࣵلُّوَيْ دࣷجْ يُ لࣷخࣷلْ يࣵلَّ بِندّ مبࣷولࣹمْ ڪࣵدُّ يَ لࣹينْ اَجِ سَخْ جِ وَخࣷونْ ݖَ ڪَوْ تُندّ وَ، بَ مُ اࣵدّࣷو فَ دِگّ سَوَرَ وَ، ڪࣹرࣷوگْ بࣺسُبْ ندَجࣹ مَ؞
‏11 بَ ݧࣹينتّ فُڪِّ گُدِّ اَڪْ ݧࣹينتّ فُڪِّ بࣴݖّࣴگْ مَتࣹي، اَجِ سَخْ جِ جࣷخْ مَ ݧَارِ اࣵلُّوَيْ دࣷجْ يَ، اࣵلُّوَيْ ڪٗلّࣴرࣹ گَ؞ ‏12 اَجِ سَخْ جِ نࣹ مَ:‏ «جٗگَلْ وࣵݖّࣹ فِي گَاوْ، ندَخْ سَ بࣷڪّ يَ نگَ گࣺنّࣹي مِسْرَ دࣹفْ نَݧُ يࣵقُتࣹ؞ تࣹگِّ نَݧُ خَاتْ يࣷونْ وَ مَ لࣹينْ سَنْتࣷونْ؞ مٗولَلْ نَݧُ سࣹينْ بࣷݒّ جࣴمُّ تُورْ؞»
‏13 اَجِ سَخْ جِ وَخَتِ مَ، نࣹ مَ:‏ «گِسْ نَا نࣹ مبࣷولࣷو مِي ڪَتْ، ݧُ دࣴگࣴرْ بࣷݒّ لَݧُ؞ ‏14 بࣵيِّ مَ رࣹڪّ مَ فَاگَاگَلْ لࣹينْ، بَ فَرْ سࣹينْ تُرْ فُ اَسَمَانْ تِيمْ، تࣹ دࣹفْ لَ نگَيْ خࣹيتْ وُ اࣴݒّ وِي، اࣴݒّ ڪࣷ دࣷولࣹ؞»
‏15 بَ لࣷولُ اَمࣹي مَ وَلْبَتِڪُ، وࣵݖّࣹ ݖَ تُندّ وَ، مَاڪْ يَارِ اࣵلُّوَيْ ڪٗلّࣴرࣹ گَ ݖَ سَمَ يَارِ لࣷخࣷ، تُندّ وࣺݒّ دِ تࣵڪّ؞ ‏16 مَ گِسْ نࣹ ندࣹڪࣹ مࣷيْ نگࣹينْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ، دَلْدِ سࣵڪَّلْ سࣹينْ بࣷݒّ وࣴلُّ وُ ݧُ مٗولْ، دࣴدُّ خَاتْ يࣷونْ وَ لࣹينْ اَجِ سَخْ جِ سَنْتࣷونْ؞ ‏17 ݖِ ڪَوْ لࣷولُ مَ نࣹ ݖَسْ يَارِ اࣵلُّوَ يَ ݖَ سَمَ يَارِ لࣷخࣷ، سࣵنِّ، رَجَخࣹ ڪࣷ، نگࣹينْ دِ گِسْ؞
مُوسَىٰ تِنُلْ نَ بࣵنِّ إِسْرَاءِيلْ
‏18 بَ لࣷولُ اَمࣹي مَ دَانُ تࣴدّ فَ ڪَنَمْ اَجِ سَخْ جِ، نَ وࣷونْ، دِيرُ ݧࣹينتّ فُڪِّ بࣴݖّࣴگْ اَڪْ ݧࣹينتّ فُڪِّ گُدِّ؞ دُ ݧَمْ وُ مَ لࣹڪّ، دُ ندࣷخْ مُ مَ نَانْ ندَخْ سࣹينْ گࣺݒّ مࣷيْ گُ نگࣹينْ مࣷيْ، دِ دࣹفْ لُ اَجِ سَخْ جِ ݧَاوْلُ، بَ مࣹرْلࣷو ڪࣷ؞ ‏19 بࣷوبَ اَمْ نَا تِيتَانگࣹ ݖِ مࣹرُمْ اَجِ سَخْ جِ، اَڪْ سࣵنجَمْ مِ مُ نَرَ سࣵنجُ ݖِ سࣹينْ ڪَوْ، نگِرْ سࣵنڪّ لࣹينْ؞ تࣹيوُلْ اَجِ سَخْ جِ نَنگُلْ مَ بࣷوبُ يࣷونْ اِتَمْ؞ ‏20 هَـٰرُونَ نَگْ، اَجِ سَخْ جِ مࣹرࣹ ڪࣷ لࣷولْ، نَرْ ڪࣷو سࣵنڪّ؞ مَ دَلْدِ ݧَانَلاَلࣹ هَـٰرُونَ؞ ‏21 سࣹينْ بࣵڪَّارُڪَايْ بࣷوبُ نگࣹينْ سࣵڪّࣷونْ نَگْ، جࣴمُّ وࣴلُّ وَ، مَ جࣴلْ ڪࣷ، لَڪّ ڪࣷ، تࣷجَتْ، وࣷلْ بَ مُ مࣷڪّ، بَ دࣷونْ ݒࣴندّ؞ مَ دَلْدِ سࣷتِّ ݒࣴندّ بَ ݖَ بِيرْ يࣷلْ گَ وَلࣹ ڪَوْ تُندّ وَ؞
‏22 ݖَ تَبࣹرَ اِتَمْ، اَڪْ ݖَ مَاسَ، اَڪْ ݖَ ڪِبّرࣷتْ تَاوَ، نگࣹينْ مࣹرْلࣷوتِ فَ اَجِ سَخْ جِ؞ ‏23 اَڪْ بَ لࣹينْ اَجِ سَخْ جِ يࣹبَلࣹي فَ ڪَدࣹسْ بَرْنࣹيَ، نࣹ لࣹينْ:‏ «دࣹمْلࣹينْ نَنگُجِ رࣺيوْ مَ مَ لࣹينْ جࣷخْ؞» يࣹينَ گࣵنْتَلْ سࣹينْ ڪࣵدُّگْ يࣵلَّ اَجِ سَخْ جِ، گࣴمُلࣹينْ ڪࣷ، دࣺگَّلُلࣹينْ ڪࣵدّࣷومْ؞ ‏24 گࣵنْتَلْ اَجِ سَخْ جِ نگࣹينْ نࣹڪّࣹ نَڪَ جࣹڪّ، دَلࣹ ڪࣷ ݖَ بࣺسْ بَ مَ لࣹينْ خَمࣹي؞
‏25 مَا دَانُ تࣴدّ فَ ڪَنَمْ اَجِ سَخْ جِ دِيرُ ݧࣹينتّ فُڪِّ بࣴݖّࣴگْ يࣷويُ، اَڪْ ݧࣹينتّ فُڪِّ گُدِّ يَ مَ تࣴدّ فَ ڪَنَمَمْ ندَخْ لَ اَجِ سَخْ جِ نࣷونْ دَ لࣹينْ دِ سࣵنڪّ؞ ‏26 مَ ݧَانْ اَجِ سَخْ جِ، نࣹ ڪࣷ:‏ «اࣺيْ بࣷرࣷومْ بِ اَجِ سَخْ جِ، بُلْ سࣵنڪّ سَ ݧࣷݧْ ݧِ نگَ سࣺدّࣷو، ݧِ نگَ جࣷتࣹ سَ مࣵگَّايْ، گࣺنّࣹي لࣹينْ مِسْرَ ݖِ سَ دࣷولࣹيْ لࣷخࣷ؞ ‏27 نگَلَّ بࣵيِّلْ خࣹلْ سَ جَامْ ݧَ، إِبْرَاهِيمَ اَڪْ إِسْحَاقَ اَڪْ يَعْقُوبَ؞ بُلْ خࣷولْ ݖِ تࣴوُگْ مبࣷولࣷو مِي اَڪْ سࣹينْ مبࣷنْ، اَڪْ سࣹينُگْ مࣷيْ؞ ‏28 لُ ڪࣷ مࣷيْ ڪَتْ وَا رࣺيوْ مَ نگَ نُ جࣴلࣹ دِنَݧُ نࣹ:‏ ‹تࣴلࣹ لࣹينَ يٗبُّ رࣺيوْ مَ مُ لࣹينْ دِگࣷونْ، اَڪْ جࣺݒِّ لࣹينْ مࣷو تَخْ اَجِ سَخْ جِ گࣺنّࣹ لࣹينْ، نگِرْ رࣹيْ لࣹينْ ݖَ مࣵنْدِݝْ مَ؞› ‏29 ݧِي دࣹ ݧࣷويْ سَ ݧࣷݧْ ݧِ نگَ سࣺدّࣷو، ݧِ نگَ گࣺنّࣹي سَ دࣷولࣹ جُ بَرࣹ، ݖِ سَ ڪࣵتَّنُ لࣷخࣷ؞»
