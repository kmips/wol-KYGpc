فِرْعَوْنَ گࣺنتّ نَ
41
‏1 ݧَارِ اَتِ لࣴمّ وࣹيسْ نَ گَنَّاوْ لࣷولُ، گࣺنتّ دِڪَّلْ فِرْعَوْنَ، مُ جࣹنࣹيرْ تَخَوْ ݖَ وࣹتُ دࣹخُ نِيلْ؞ ‏2 مُ گِسْ جُرٗومْ ݧَارِ نَگْ يُ رَفࣹتْ تࣹ سُورْ گࣺنّࣹ ݖَ دࣹخْ گَ، دِ فࣷرْ بَرَخِ تࣵڪّ گَ؞ ‏3 مُ دࣹلُّ گِسْ جُرٗومْ ݧَارِ نَگْ يُ ݧَاوْ تࣹ يࣷويْ تࣷݒّ ݖَ ݧࣷومْ، گࣺنّࣹ ݖَ دࣹخْ گَ، لࣵنگّ اَڪْ ݧࣷومْ ݖَ تࣵڪّ گَ؞ ‏4 نَگْ يُ ݧَاوْ يَ تࣹ يࣷويْ دَلْدِ لࣹڪّ جُرٗومْ ݧَارِ نَگْ يُ رَفࣹتْ يَ تࣹ سُورْ؞ فِرْعَوْنَ يࣹوُّ، ‏5 بَ نࣷݒِّ نࣹلَوَاتْ، دࣹلُّ گࣺنتّ، گِسْ جُرٗومْ ݧَارِ گُبْ¤ يُ نَنگُ تࣹ بَاخْ سَخْ ݖِ بࣹنّ گَتَّخْ؞ ‏6 گَنَّاوْ گَ جُرٗومْ ݧَارِ گُبْ يُ گࣷويْ يُ نگࣹلَوْ لِ لَڪّ سَخْ، تࣹگُ ݖَ يَ جِيتُ؞ ‏7 ݖَ لَ گُبْ يُ گࣷويْ يَ وࣷنّ جُرٗومْ ݧَارِ گُبْ يُ نَنگُ يَ تࣹ بَاخْ؞ گَنَّاوْ لࣷولُ فِرْعَوْنَ يࣹوُّ، خَمْ نࣹ گࣺنتّ لَ وࣷونْ؞
‏8 بَ بࣴتْ سࣹتࣹي، فِرْعَوْنَ جَاخْلࣹ لࣷولْ، بَ وࣷولُ جِبَرْ يَ اَڪْ بࣷرࣷومْ خَمْ-خَمِ مِسْرَ يࣺݒّ، نࣹتَّلِ لࣹينْ لَ مُ گࣺنتّ، تࣹ ڪࣹنّ مَنُ ڪࣷو فِرِلْ فِرْعَوْنَ؞
‏9 بَ لࣷولُ اَمࣹي ڪِلِفَ گَيْ تࣷݒّتࣷو نَانِ فِرْعَوْنَ نࣹ ڪࣷ:‏ «مَنْ دࣹ تࣷوݧْ نَا تࣹ دِنَا لَ ڪࣷ وَخْ تࣹيْ؞ ‏10 سَنگّ بِ، بَ نگَ نُ مࣹرࣹي، يَوْ فِرْعَوْنَ، بَ تࣴجْ مَ ڪَسࣷ ݖَ ڪࣴرْ نجِيتُ دَگْ يَ، مَنْ اَڪْ ڪِلِفَگْ لَڪّ-ڪَتِ مبُرُ يَ، ‏11 گࣹنّ گُدِّ دَنࣷو گࣺنْتَنْدࣷو، مَنْ اَڪْ مࣷومْ، گࣺنتّ گُ ݖِ نࣹڪّ اَڪْ لَ مُيْ فِرِ؞ ‏12 فࣹڪّ اَمࣷونْ نَ وَخَمْبَانࣹوْ اࣹبْرࣴ وُ فَ نࣹڪّࣷونْ اَڪْ نُنْ، تࣹ دࣷونࣷونْ جَامُبْ نجِيتَلْ دَگْ يَ؞ نُ نࣹتَّلِ ڪࣷ سُنُيْ گࣺنتّ، مُ فِرِلْ نُ ڪࣷ، وَخْ نُ ڪُ نࣹڪّ لَ سَ گࣺنتّ دِ تࣹڪِّ؞ ‏13 تࣹ نَ مُ نُ ڪࣷ فِرِلࣹ وࣷونْ، نَ لَ اَمࣹ:‏ دࣹلّࣷو نَݧُ مَ سَمَ لِگّࣺييْ، ڪَ ݖَ دࣹسْ ݧُ وࣺڪّ ڪࣷ ݖِ بَنتّ؞»
يُوسُفَ فِرِ نَ گࣺنْتِ فِرْعَوْنَ
‏14 بَ فِرْعَوْنَ دࣺگّࣹي لࣷولُ، دَفَ وࣷولُ يُوسُفَ؞ ݧُ گَاوْ تِجِّ ڪࣷ، مُ وَتُ، سࣷلُ، دࣷورَ وُيْجِ فِرْعَوْنَ؞ ‏15 فِرْعَوْنَ نࣹ يُوسُفَ:‏ «دَمَا گࣺنتّ، تࣹ ڪࣹنّ مَنُ ڪࣷو فِرِ، تࣹ مَ دࣺگّ ݧُيْ وَخْ ݖِ يَوْ نَانْ، گࣺنتّ گࣷو دࣺگّ رࣹڪّ، خَمْ ڪࣷ، بَ مَنْ ڪࣷو فِرِ؞» ‏16 يُوسُفَ نࣹ ڪࣷ:‏ «دُ ݖِ مَنْ دࣹ، وَايࣹ يࣵلَّ مࣷويْ بَاخࣹ فِرْعَوْنَ ݒِرِمْ جࣵمّ؞»
‏17 فِرْعَوْنَ نࣹ يُوسُفَ:‏ «دَمَا گࣺنتّ، تَخَوْ ݖِ تࣵڪَّلْ دࣹخُ نِيلْ، ‏18 مَ گِسْ جُرٗومْ ݧَارِ نَگْ يُ سُورْ تࣹ رَفࣹتْ گࣺنّࣹ ݖَ دࣹخْ گَ، دِ فࣷرْ بَرَخُ تࣵڪّ گَ؞ ‏19 مَ دࣹلُّ گِسْ يࣹنࣹينْ جُرٗومْ ݧَارِ نَگْ يُ رࣵگّ، ݧَاوْ تࣹ يࣷويْ گࣺنّࣹ ݖَ دࣹخْ گَ، تࣷݒّ ݖِ ݧࣷومْ، دِ يُ ݧَاوْ، بَ مَسُمَا گِسْ لُ نِ مࣹلْ ݖِ مِسْرَ؞ ‏20 نࣷونُ يُ يࣷويْ يَ تࣹ ݧَاوْ دَلْدِ لࣹڪّ جُرٗومْ ݧَارِ نَگْ، يَ جࣴڪّࣷونَ فࣹيݧْ تࣹ سُورْ؞ ‏21 وَايࣹ بِ ݧُ لࣹينْ لࣹڪّࣹي بَ نࣷݒِّ، اِتَمْ دَفَ مࣹلْ نِ لࣹڪُّݧُ لࣹينْ، ندَخْ تࣹرࣹوُلࣹينَ مࣹلْ نَ وࣷونْ؞ بࣷوبَ لاَ يࣹوُّ، ‏22 بَ نࣷݒِّ دࣹلُّ گࣺنتّ، گِسْ جُرٗومْ ݧَارِ گُبْ يُ فࣹيسْ تࣹ بَاخْ سَخْ ݖِ بࣹنّ گَتَّخْ؞ ‏23 گَنَّاوْ گَ مَ گِسْ جُرٗومْ ݧَارِ گُبْ يُ گࣷويْ يُ نگࣹلَوْ لِ لَڪّ، سَخْ تࣹگُ ݖَ يَ جِيتُ؞ ‏24 گُبْ يُ گࣷويْ يَ نَگْ وࣷنّ جُرٗومْ ݧَارِ گُبْ يُ بَاخْ يَ؞ نࣹتَّلِ نَا ڪࣷ سَمَيْ جِبَرْ، وَايࣹ ڪࣹنّ ݖِ ݧࣷومْ مَنُ مَ ڪࣷو فِرِلْ؞»
‏25 يُوسُفَ نࣹ فِرْعَوْنَ:‏ «سَ ݧَارِ گࣺنتّ يِ دࣹ، سَنگّ بِ، ݧࣷو بࣷڪّ ݒِرِ؞ سَنگّ بِ، يࣵلَّ دَ لاَ خَمَلْ لِ مُيْ دࣹفْ اࣴلّࣴگْ؞ ‏26 جُرٗومْ ݧَارِ نَگْ يُ بَاخْ يِ، جُرٗومْ ݧَارِ اَتْ لَ؞ تࣹ اِتْ جُرٗومْ ݧَارِ گُبْ يُ بَاخْ يِ، جُرٗومْ ݧَارِ اَتْ لَ؞ مُيْ گࣹنّ گࣺنتّ گِ؞ ‏27 تࣹ اِتَمْ جُرٗومْ ݧَارِ نَگْ يُ يࣷويْ، يِ ݖِ تࣷݒّ تࣹ ݧَاوْ، جُرٗومْ ݧَارِ اَتْ لَ؞ نَڪَ نࣷونُ جُرٗومْ ݧَارِ گُبْ يُ گࣷويْ، يِ نگࣹلَوْ لِ لَڪّ، جُرٗومْ ݧَارِ اَتِ خِيفْ لَيْ فِرِ؞ ‏28 مࣷويْ لِ مَ لَ وَخْ رࣹڪّ، سَنگّ بُ تࣹدّ بِ:‏ لِ يࣵلَّ دِ دࣹفِ اࣴلّࣴگْ لَ لَ وࣷنْ؞ ‏29 نُ نگِ دࣴگْمَلْ جُرٗومْ ݧَارِ اَتِ نَاتَانگࣹ يُ رࣺيْ ݖِ مبࣷولࣹمْ رࣺيوُمْ مِسْرَ؞ ‏30 تࣹ جُرٗومْ ݧَارِ اَتِ خِيفَ ݖِيْ تࣷݒّ، بَ ݧُ فࣵتّࣹ نَاتَانگࣹ گُ رࣺيْ گَ اَمࣷونْ رࣺيوُمْ مِسْرَ؞ خِيفْ بَ ݖَيْ تࣷݒّ دِنَ يࣵقْ رࣺيوْ مِ؞ ‏31 خِيفْ بِ دِنَ تَخْ لࣹنّ لُيْ نِرࣷوڪْ نَاتَانگࣹ دࣷوتُ دࣹسْ ݖِ رࣺيوْ مِ، ندَخْ دِنَ مࣹتِّ لࣷولْ؞ ‏32 تࣹ نَگْ، سَنگّ بِ، گَنَّاوْ گࣺنتّ نگَ ڪࣷ ݧَارِ يࣷونْ، ڪࣷنْ يࣵلَّ دࣷگَلْ نَ مبِرْ مِ، تࣹ دِنَ ڪࣷ اَمَلْ بَلاَ يࣵگّ؞»
‏33 يُوسُفَ تࣹگْ ݖَ نࣹ فِرْعَوْنَ:‏ «ڪࣷنْ لࣺيگِ سَنگّ بِ، وُتَلْ نِتْ ڪُ مُوسْ تࣹ رَفࣹتْ خࣹلْ، جِيتَلْ ڪࣷ ݖِ رࣺيوُمْ مِسْرَ؞ ‏34 تࣹ اِتَمْ سَنگّ بِ، دَنگَيْ فَلْ اَيْ ندَوْ ݖِ رࣺيوْ مِ، ݧُيْ سࣵڪُّ جُرٗومِ نَتّ يُ نࣹ بࣹنّ ݖِ مبࣷولࣹمْ لُ ݧُ گٗوبࣹ رࣺيوُمْ مِسْرَ دِيرُ جُرٗومْ ݧَارِ اَتِ نَاتَانگࣹ يِيْ ݧࣴوْ؞ ‏35 ݧࣷومْ نَݧُ دَجَلࣹ مبࣷولࣹمْ دُنْدِ اَتْ يُ بَاخْ يِ، تࣹ بࣷولࣹ ݒࣹݒّ مِ ݖِ سَ ندِگَلُ بࣷݒّ، يَوْ فِرْعَوْنَ، مُيْ دُندّ بُ ݧُيْ دࣹنݖّ ݖِ دࣴڪّ يِ؞ ‏36 دُندّ بࣷوبࣷويْ دࣷونْ سࣵقُ رࣺيوْ مِ دِيرُ جُرٗومْ ݧَارِ اَتِ خِيفْ يِيْ دَلْ ݖِ رࣺيوُمْ مِسْرَ؛ سُ ڪࣷ دࣹفࣹي وَا رࣺيوْ مِ دُݧُ لࣷرُ ݖِ خِيفْ بِ؞»
فِرْعَوْنَ فَلْ نَ يُوسُفَ، مُ جِيتࣹ مِسْرَ گࣺݒّ
‏37 ݒࣹخࣹ مࣷومُ نَگْ دࣷونْ لُ فِرْعَوْنَ اَڪِ جَوْرِݧَمْ يࣺݒّ نجࣷرتّ نࣹ بَاخْ نَ؞ ‏38 فِرْعَوْنَ دَلْدِ نࣹ جَوْرِݧْ يَ:‏ «ندَخْ مَسࣹيسْ نَا گِسْ ڪُ مࣹلْ نِ ڪِيَ وَايْ، نِتْ ڪُ يࣵلَّ سࣷلْ نࣷووُگْ بࣷݒَّمْ؟» ‏39 فِرْعَوْنَ وَخْ اَڪْ يُوسُفَ نࣹ ڪࣷ:‏ «ندࣹگَمْ يࣵلَّ خَمَلْ نَ لَ لِي لࣺݒّ ڪَيْ، لࣹيرْ نَ نࣹ اَمُلْ ڪࣹنّ ڪُ مُوسْ تࣹ رَفࣹتْ خࣹلْ نِ يَوْ؞ ‏40 ڪࣷنْ دِنَا لَ مَيْ، نگَ جِيتࣹ ڪࣴرْ بُورْ، تࣹ وَا رࣺيوْ مࣺݒّ ݖِ سَ وَاوْ لَݧُيْ وࣺيْ؛ جَلْ بِ مَ تࣷوگْ دࣷݝّ لاَ لَيْ سُتࣹ؞»
‏41 گَنَّاوْ لࣷولُ فِرْعَوْنَ نࣹيتِ يُوسُفَ:‏ «بࣺسْ نِڪِ تࣹيْ، دࣺنڪّ نَا لَ رࣺيوُمْ مِسْرَ گࣺݒّ» ‏42 بَ لࣷولُ اَمࣹي فِرْعَوْنَ تࣹڪّࣹي جَارࣷبْ مࣵنْدَرْگَالْ بَ ݖَ لࣷخࣷومْ، تَڪَّلْ ڪࣷ يُوسُفَ، سࣷلَلْ ڪࣷ يࣺرࣹيْ تࣵنّࣺيفْ، تَڪَّلْ ڪࣷ ݖَقُ وُرُسْ، ‏43 بَ نࣷݒِّ وَرَلْ ڪࣷ وَتِيرُ ڪُ سࣴسْ رࣴڪّ ݖِ مࣷومْ فِرْعَوْنَ، تࣹ فُ مُ وَرࣹ ݧُ جِيتُ ڪࣷ، دِ يࣺينࣹ نَانْ:‏ «سُجّٗوتْلࣹينْ¤!‏» نࣷونُ لَ ڪࣷ فِرْعَوْنَ فَلࣹ، مُ يِلِفْ مِسْرَ گࣺݒّ؞
‏44 ݖِ ڪَوْ لࣷولُ فِرْعَوْنَ نࣹ يُوسُفَ:‏ «مَنْ مَايْ فِرْعَوْنَ، وَايࣹ گَنَّاوْ-سِ-تࣹيْ مبࣷولࣹمْ وَا رࣺيوُمْ مِسْرَ، ڪࣹنّ دُ ݖِ سَݧَتࣹي يࣺڪَّتِ لࣷخࣷ مبَا مُيْ سࣺقِ جࣺيگࣷ، تࣹ اࣵنْدࣷو ݖِ؞» ‏45 فِرْعَوْنَ نَگْ تُدّࣹ يُوسُفَ سَفْنَتْ ݒَنࣹيَ، مَيْ ڪࣷ جَبَرْ اَسْنَتْ دࣷومُ ݒࣷتِفࣹرَ مَ دࣷونْ سَرْخَلْڪَتْ ݖَ دࣴڪّ بَ ݧُيْ وَخْ اࣷنْ¤؞ يُوسُفَ دࣷخࣹ ݖَ تࣹگࣷو ڪِلِفْتࣹيفُ مِسْرَ گࣺݒّ؞ ‏46 بَ يُوسُفَيْ تࣵمْبَلࣹي لِگّࣺييَلْ فِرْعَوْنَ بُورُ مِسْرَ، مُ نگِ تࣷلُّ ݖِ فَنْوࣹيرِ اَتْ؞ يُوسُفَ دَلْدِ بࣵيِّڪࣷو ݖَ فِرْعَوْنَ، وࣴرْ رࣺيوْ مَ مࣺݒّ؞
‏47 دِيرُ جُرٗومْ ݧَارِ اَتِ نَاتَانگࣹ يَ سُوفْ سَ نَنگُ، بَ مࣹݧّࣹيفْ مَ تُورُ؞ ‏48 مُ دَجَلࣹ مبࣷولࣹمْ دُنْدُبْ رࣺيوُمْ مِسْرَ دِيرُ جُرٗومْ ݧَارِ اَتْ يࣷويࣹي، دࣴڪّ بُ نࣹڪّ مُ دࣹنݖّ فَ دُندّ، بَ جٗگࣹ ݖَ تࣷولْ يَ ڪࣷ وࣴرْ؞ ‏49 فَ لَ يُوسُفَ جَارࣹ دَجَلࣹ ݒࣹݒّ، مُ نࣹ گࣵݧّ، نِ سُوفَسْ گࣺيجْ، بَ تࣵيّࣹيتِ ڪࣷو وَݧّ، ندَخْ مُجّ نَ وࣹيسْ اَبْ لِمْ؞
‏50 لاَتَ اَتِ خِيفْ يَيْ اَگْسِ، يُوسُفَ اَمْ نَ ݧَارِ دࣷومْ يُ گٗورْ اَڪْ اَسْنَتْ دࣷومُ ݒࣷتِفࣹرَ، ڪِلِفَگْ دِينࣹ، گَ ݖَ اࣷنْ؞ ‏51 يُوسُفَ تُدّࣹ تَاوْ بَ مَنَسࣹ (مُ فِرِ ڪِ مَ تَخَ فࣵتّࣹ)، ندَخْتࣹ دَ نࣹ:‏ «لِ مَ سࣷنّ اَڪْ لِ مَ سࣷرࣹ سَمَ ڪࣴرْ بَايْ لࣺݒّ، يࣵلَّ فࣵتّࣹلࣷو نَ مَ ڪࣷ؞» ‏52 دࣷومَمْ جَ ݖَ تࣷݒّ مُ تُدّࣹ ڪࣷ اࣹفْرَيِمْ (مُ فِرِ گِيرْ)، ندَخْتࣹ دَ نࣹ:‏ «يࣵلاَّ مَ گِيرَلْ ݖَ رࣺيوْ، مَ مَ سࣷنّࣹ؞»
‏53 ݖِ ڪَوْ لࣷولُ جُرٗومْ ݧَارِ اَتِ نَاتَانگࣹ يَ جࣹيخْ ݖَ رࣺيوُمْ مِسْرَ، ‏54 جُرٗومْ ݧَارِ اَتِ خِيفْ يَ تࣵمْبَلِ، نَ ڪࣷ يُوسُفَ وَخࣹ وࣷونْ؞ رࣺيوْ يࣺݒّ نَگْ نࣹڪّ ݖِ خِيفْ، وَايࣹ اَمُلْ فُ ݧࣵڪّ دُندّ ݖَ مِسْرَ؞
‏55 بَ وَا مِسْرَ خِيفࣹي، دَݧࣷو دࣹمْ جࣷويْتُجِ دُندّ ݖَ فِرْعَوْنَ؞ مُ وَخْ اَڪْ ݧࣺݒّ نࣹ لࣹينْ:‏ «دࣹمْلࣹينْ ݖَ يُوسُفَ؛ لُ مُ لࣹينْ وَخْ، دࣹفْلࣹينْ ڪࣷ؞» ‏56 بَ ݧُ دࣹمࣹي، بَ مࣹتِتُ خِيفْ بِ دَجَلْ رࣺيوْ مࣺݒّ، يُوسُفَ دَفَ اُبِّ دࣹنݖّ يࣺݒّ، دِ جَايْ ݒࣹݒّ مِ وَا مِسْرَ، ‏57 بَ نِتْ ݧِيْ جٗگࣹ فُ نࣹڪّ ݖِ اࣵدِّنَ، دِ وُتْسِ نجࣴلْ ݖِ يُوسُفَ ندَخْ خِيفْ بُ مࣹتّࣹي فࣺݒّ ݖَ اࣵدِّنَ؞
