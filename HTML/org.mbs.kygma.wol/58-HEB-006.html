6
‏1 ڪࣷنْ نَگْ نَنُ جٗگࣹ ݖِ ندࣷورْتࣹلْ اࣵلُّوَيْ اَلْمَسِ، دَلْدِ جࣵلّ جࣴمْ ݖِ نجࣵنگُمْ مَگْ، تࣹ ݧُ بَݧَ لَلاَتْ ڪࣹنُ گِ؞ ڪࣹنُ گِ مࣷويْ گࣴمْ يࣵلَّ گَنَّاوْ تُوبࣹيلْ گِ وَرْ ݖِ جࣴفْ جُ بࣷنْ جِ نِتْ ڪِيْ دࣹفْ تࣹ مُيْ جࣴفِ ندࣹي؞ ‏2 ڪࣹنُ گِ مࣷويْ اِتَمْ نجࣵنگّلࣹمْ خࣹيتِ نجࣵݒّ، اَڪْ ݧَانْ گُ اࣵندّ اَڪْ تࣹگࣹبْ لࣷخࣷ، اَڪْ مبِرُمْ ندࣹي-ندࣹڪِّ، اَڪْ اࣵتّࣹبْ بࣺسْ-ݒࣺنݖّ بِيْ سَخْ دࣵڪّ؞ ‏3 نَنُ جࣴمْ ڪَنَمْ، تࣹ مࣷومْ لَنُيْ دࣹفْ، ندࣹگَمْ سࣷوبْ نَ يࣵلَّ؞
‏4 ݧِ ݧُ خَسَ لࣹيرَلْ نَگْ، تࣹ ݧُ جࣷتَ مࣷسْ مَيْ گُ بَوࣷو اَسَمَانْ، سࣺدُّ ݖِ نࣷو گُ سࣹلّ گِ، دࣹيسُ لࣹينْ مَنَ دࣹفَلْ دَرَ؞ ‏5 ݧࣷو مࣷسْ ڪࣵدُّگْ يࣵلَّ گُ بَاخْ گِ، اَڪْ مَنࣷورࣹ يُ جَمࣷنࣷ جِيْ ݧࣴوْ، ‏6 بَ نࣷݒِّ خࣵݖِّ؞ ڪࣷنْ دࣹيسُ لࣹينْ وَݧِّ، نگِرْ ݧُيْ تُوبَتِ، ندࣹ ݧࣷو دَاجَاتَلْ سࣹينْ بࣷݒّ دࣷومُ يࣵلَّ جِ ݖِ بَنتّ، تࣹگْ ڪࣷ تࣷرࣷخْتࣹ جُ نࣹ فࣵݝّ؞
‏7 ندَخْ سُوفْ ڪَتْ، سُ نَانࣹي ندࣷخْ مُ مُ تَوْلُ اَيْ يࣷونْ، بَ مࣹݧَّلْ ݧِ ݧُ ڪࣷيْ بࣹيَلْ، گࣵنݖَخْ گُ لࣹينْ جَرِݧْ، سࣷوسَ سُوفْ دِنَ سࣺدُّ ݖِ بَرْڪࣹ بُ بَوࣷو فَ يࣵلَّ؞ ‏8 وَايࣹ سُ مࣹݧّࣹي اَيْ دࣺگْ اَڪِ خَاخَامْ، اَمُلْ نجَرِݧْ، اَلْڪُتࣹيمْ لࣹيسْ دِ تُدّ لࣺيگِ، تࣹ لَڪّ لَيْ مُجّࣹ؞
‏9 وَايࣹ نَگْ سࣷݒّࣹ يِ، نِي نُيْ وَخࣹ اَڪْ يࣹينْ تࣹيوُلْ نُ دِ لࣹينْ يَاڪَارࣹ ݖِ لُ وࣺرْ، لِ لࣹينْ گࣴنَ اَمَلْ نجَرِݧْ، تࣹ بࣷڪّ ݖِ سࣹينُگْ مُݖّ؞ ‏10 يࣵلَّ ݧࣵڪُّلْ وࣷرْمَ بَيْ فࣵتّࣹ سࣹينْ لِگّࣺييْ، اَڪْ ݖࣷفࣹيلْ گِ نگࣹينْ سࣷݒّ تُرَمْ، تࣹ فِرْندࣹيلࣹ ڪࣷ نِ نگࣹينْ تَخَوࣷو ݧُ سࣹلّ ݧِ، دࣺمبّ اَڪْ تࣹيْ؞ ‏11 وَايࣹ دَنࣷو بࣴگّ ڪࣹنّ ڪُ نࣹڪّ ݖِ يࣹينْ وࣺيࣹ گࣹنّ فَرْلُتࣹ گࣷوگُ نگࣹينْ نࣹڪّࣹ، بَ ڪࣹرَ مُجْ گَ، نگِرْ سࣹينْ يَاڪَارْ سࣷتِّ بَ مَتْ سࣴڪّ؞ ‏12 سُ ڪࣷ دࣹفࣹي دُنگࣹينْ يࣵݖَّارَلْ، وَايࣹ دَنگࣹينْ دِ رࣷيْ ݖِ ݧِ گࣴمْ تࣹ مُݧْ، بَ جَگࣷو دِگࣹيْ يࣵلَّ؞
دِگࣹبْ يࣵلَّ دُ جَاسْ
‏13 مُ مࣹلْ نِ إِبْرَاهِيمَ، بَ يࣵلَّ دِ دِگࣷو اَڪْ مࣷومْ، يࣵلَّ مَنُلْ وࣷونَ گِݧࣹ ڪُ سُتْ يࣵلَّ؞ مࣷو تَخْ مُ گِݧَلْ ڪࣷ ݖِ بࣷݒَّمْ، ‏14 نࣹ ڪࣷ:‏ «مَا لَيْ بَرْڪࣹيلَ بَرْڪࣹيلْ دࣺيْ، فُلْ لاَ فُلْ¤؞» ‏15 نࣷونُ لَ إِبْرَاهِيمَ مُݧࣹ، بَ ڪࣹرَ مُ جࣷتࣹ ݖَ دِگࣹ بَ؞
‏16 نِتْ نَگْ، ڪُ ڪࣷ سُتْ لَيْ گِݧࣹ، نگِرْ دࣷگَلْ ڪࣵدُّ، بَ تࣹيسَلْ وࣺݒّ وࣹرَنْتࣹ؞ ‏17 يࣵلاَّ بࣴگّࣷونَ فِرْندࣹيلْ نُ مُ گࣴنَ لࣹيرࣹ ݧࣹيلْ ݧِ مࣷومْ دِگࣹ بِ، نࣹ اَبْ دࣷگَلَمْ دُ تࣹبِّڪُ مُڪّ؞ مࣷو تَخْ مُ دࣷگَلࣹ نگِݧْ اَبْ دِگࣹيمْ؞ ‏18 ݧَارْ يࣷويُ دُ تࣹبِّڪُ ندَخْ يࣵلَّ دُ فࣹنْ، تࣹ نُنْ ݧِ دَوْ لࣵقُسِ ݖِ مࣷومْ، ݖِ لَنُ اَمࣹ ندࣴگࣴرْلاَيْ لُ رࣺيْ لُ نُ ݝࣷيࣹ ݖِ يَاڪَارْ جِ ݖِ سُنُ ڪَنَمْ؞ ‏19 يَاڪَارْ جࣷوجُ لَنُ اَمْ، مُ مࣹلْ نِ دِيگَلُبْ گَالْ، دِيگَلْ بُ دࣴگࣴرْ تࣹ وٗورْ، ݧࣹيلْ سُنُ خࣷلْ، تࣹ يَاڪَارْ جࣷوجُ مࣷويْ بࣴتّ رِدࣷ بَ، بَ جࣵلّ ݖَ نࣺيگُ بِيرْ بُ يࣵلَّ بَ؞ ‏20 فَ لَ ندَوْ لِ نُ جِيتُ دُگّ نگِرْ نُنْ، تࣹ ڪࣷوڪࣷويْ عِيسَىٰ مِ ݧُ فَلْ سَرْخَلْڪَتْ بُ مَگْ بَ فࣵوّ، مُ تࣷݒّ دَيࣷبْ مࣹلْڪِسࣹدࣹگْ؞
