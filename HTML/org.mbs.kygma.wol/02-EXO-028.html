لُ جࣴمْ ݖِ يࣺرࣹيْ سَرْخَلْڪَتْ يِ
28
‏1 «لࣺيگِ نَگْ يَوْ، نَنگَ سࣹݒّࣹي ݖِ بࣵنِّ إِسْرَاءِيلْ سَ مَگْ هَـٰرُونَ، اَڪْ دࣷومَمْ يُ گٗورْ، نگَ وࣷو لࣹينْ فِ يَوْ، ݧُ دِ مَ سَرْخَلَلْ:‏ هَـٰرُونَ اَڪْ دࣷومَمْ يُ گٗورْ، نَدَبْ اَڪْ اَبِيُ اَڪْ اࣹلَسَرْ اَڪْ اِتَمَرْ؞ ‏2 سࣵڪَّلَلْ سَ مَگْ هَـٰرُونَ يࣺرࣹيْ سࣹلّنگَ نگِرْ دَرَجَ، اَڪْ تࣹرَانگَ بُ سَرْخَلْڪَتْ وَرَ يࣹلّࣷو؞ ‏3 دَنگَيْ سَنتّ مبࣷولࣹمْ لِگّࣺييْڪَتْ يُ خَرࣹݧْ، يِ مَ فࣹيسَلْ خࣹلُمْ خَرࣹݧْتࣹ، ݧُ دࣹفَرْ يࣺرࣹيْ هَـٰرُونَ، نگِرْ فَلْ ڪࣷ، مُ دِ مَ سَرْخَلَلْ؞ ‏4 يِي يࣺرࣹ نَگْ لَݧُ ڪࣷيْ دࣹفَرَلْ:‏ ڪِيرَايَلْ دࣴنّ اَڪْ اَبْ خَرْ-سࣵنِّ اَڪْ اَبْ فࣴخْيَ، اَڪْ مبُبّ مُ گُدّ مُ ݧُ رࣵبّ، اَڪْ ڪَالَ اَڪْ لَخَسَايْ؞ يࣺرࣹيْ سࣹلّنگَ لَݧُيْ دࣹفَلْ سَ مَگْ هَـٰرُونَ اَڪْ دࣷومَمْ يُ گٗورْ، نگِرْ ݧُ دِ مَ سَرْخَلَلْ؞ ‏5 لِگّࣺييْڪَتْ يࣷويُ ݧࣷويْ جࣴلْ وࣴݧِ وُرُسْ، اَڪْ وࣴݧْ گُ بَخَ تࣹ لاَلْ يࣷلࣹتْ، اَڪْ وࣴݧْ گُ خࣹوَرْ، اَڪْ گُ خࣷنقّ ݖُرّ، اَڪْ وࣴݧُ لࣹي نگِرْ لِگّࣺييࣹ ڪࣷ؞
لُ جࣴمْ ݖِ خَرْ-سࣵنِّ مِ
‏6 «نَݧُ رࣵبّࣹ خَرْ-سࣵنِّ مِ وࣴݧُ وُرُسْ، اَڪْ وࣴݧْ گُ بَخَ تࣹ لاَلْ يࣷلࣹتْ، اَڪْ وࣴݧْ گُ خࣹوَرْ اَڪْ گُ خࣷنقّ ݖُرّ اَڪْ وࣴݧُ لࣹي، ݧُ ݧَوَالࣹ ݖِ اَيْ نَتَالْ؞ ‏7 ݧَارِ نگࣵلِّ مبَگّ يُ ݧُيْ فَسْتࣹ لَ خَرْ-سࣵنِّ مِيْ اَمْ، ݧُ دِ لࣹينْ فَسْتࣹي ݖِ ݧَارِ ݖَتْ يِ؞ ‏8 نگَݧَايْ لِيْ تࣹگࣹ ݖِ ڪَوْ خَرْ-سࣵنِّ مِ، نِ خَرْ-سࣵنِّ مِ لَݧُ ڪࣷيْ لِگّࣺييࣹ، مُ اࣵندّ اَڪْ مࣷومْ، دِ بࣹنّ، دِ وࣴݧُ وُرُسْ، اَڪْ وࣴݧْ گُ بَخَ تࣹ لاَلْ يࣷلࣹتْ، اَڪْ وࣴݧْ گُ خࣹوَرْ، اَڪْ گُ خࣷنقّ ݖُرّ، اَڪْ وࣴݧُ لࣹي؞ ‏9 نَنگَ جࣴلْ ݧَارِ ݒࣹرِ اࣷنِگْسࣴ، نگَ ݧَاسْ ݖِ تُرِ دࣷومِ إِسْرَاءِيلْ يُ گٗورْ:‏ ‏10 يِي جُرٗومْ بࣹنِّ تُرْ ݖِ بࣹنّ ݒࣹرْ، جُرٗومْ بࣹنِّ تُرْ يِ ݖِ دࣹسْ ݖِ بࣹنࣹينْ ݒࣹرْ بِ، تُرْ يِ تࣷݒَّنْتࣹ، نِ بࣷرࣷومْ تُرْ يِ تࣷݒَّنْتࣹي ݖِ جُدُّ؞ ‏11 نِ ڪࣷ اَبْ لِگّࣺييْڪَتْ بُ فَرࣹوࣷو مبِنْدُمْ دࣷجْ دِ دࣹفࣹ، نِ لَݧُيْ ݧَاسࣹ تُرِ دࣷومِ إِسْرَاءِيلْ يُ گٗورْ ݖِ ڪَوْ ݧَارِ ݒࣹرْ يِ، مُ مࣹلْ نِ نِ ݧُيْ ݧَاسࣹ خࣵمِّڪَايْ يِ ݧُيْ تࣷرْلࣷو، تࣹ نَݧُ وࣹ ݒࣹرْ يِ ݖِ اَيْ ݝَنڪُڪَايِ وُرُسْ؞ ‏12 تَݒَّلْ ݧَارِ ݒࣹرْ يِ ݖِ ڪَوْ نگࣵلِّ وَگِّ خَرْ-سࣵنِّ مِ، ݒࣹرْ يِ وُوتُ بࣵنِّ إِسْرَاءِيلْ، هَـٰرُونَ گࣵدُّ سࣹينِ تُرْ ݖِ ڪَوْ ݧَارِ نگࣵلِّ وَگَّمْ، فِ ڪَنَمْ اَجِ سَخْ جِ، نگِرْ سَخَلْ اَگْ ݒࣵتّلِڪُ؞
‏13 «سࣵڪَّلْ اَيْ ݝَنڪُڪَايِ وُرُسْ، ‏14 اَڪْ ݧَارِ ݖَقِ وُرُسُ نگَلَمْ، بُ ݖِ نࣹڪّ نگَ رَوَاسࣹ ڪࣷ نِگْ بُومْ، بَ نࣷݒِّ نگَ وࣹ ݖَقْ يُ رَوَاسࣷو يࣷويُ ݖِ ݝَنڪُڪَايْ يِ؞
لُ جࣴمْ ݖِ ڪِيرَايَلْ دࣴنّ بِ
‏15 «نَنگَ سࣵڪّ ڪِيرَايَلْ دࣴنّ بِ ݧُيْ جࣴفࣷو ݖِبْ اࣵتّࣹ، ݧُ لِگّࣺييَالࣹ ڪࣷ اَيْ نَتَالْ نِ خَرْ-سࣵنِّ مِ، تࣹ نَݧُ ڪࣷ رࣵبّࣹ وࣴݧُ وُرُسْ، اَڪْ وࣴݧْ گُ بَخَ تࣹ لاَلْ يࣷلࣹتْ، اَڪْ وࣴݧْ گُ خࣹوَرْ، اَڪْ گُ خࣷنقّ ݖُرّ، اَڪْ وࣴݧُ لࣹي؞ ‏16 اَبْ ڪَارࣹ بُ ݧُ لࣹمْ لَيْ دࣷونْ، گُدَّايْ بِ دِ ݝَرَلࣹبْ بَارَامُ دࣺيْ اَڪْ دِگّ، يَاتُوَايْ بِ دِ ݝَرَلࣹبْ بَارَامُ دࣺيْ اَڪْ دِگّ؞ ‏17 نَنگَ وࣹ اَيْ ݒࣹرْ ݖِ ڪَوْ ڪِيرَايَلْ دࣴنّ بِ، نگَ دࣹفْ ڪࣷ ݧࣹينْتِ لࣵنگِ ݒࣹرْ؞ لࣵنگّ گِ جِيتُ، ݒࣹرُ سَرْدُوَنْ اَڪْ ݒࣹرُ تࣷݒَاسْ اَڪْ ݒࣹرُ اࣹسْڪَرْبُڪَلْ مࣷو ݖَيْ نࣹڪّ؛ ‏18 ݧَارࣹيلُ لࣵنگّ گِ دِ ݒࣹرُ اࣹمࣴرٗدّ اَڪْ ݒࣹرُ سَفِيرْ اَڪْ ݒࣹرُ جَمَا؛ ‏19 ݧࣹتّࣹيلُ لࣵنگّ گِ دِ ݒࣹرُ يَسࣹنتّ اَڪْ ݒࣹرُ اَگَاتْ اَڪْ ݒࣹرُ اَمࣹتِسْتࣴ؛ ‏20 ݧࣹينْتࣹيلُ لࣵنگّ گِ دِ ݒࣹرُ ڪِرْسࣷلِتْ اَڪْ ݒࣹرُ اࣷنِگْسࣴ اَڪْ ݒࣹرُ جࣵسْݒࣴ، تࣹ ݒࣹرْ بُ ݖِ نࣹڪّ، ݝَنڪُڪَايُ وُرُسْ لࣹيسْ ڪࣷيْ وࣹوࣹ؞ ‏21 ݒࣹرْ يࣹيڪْ تُرِ دࣷومِ إِسْرَاءِيلْ ݧࣷويْ تࣷلّࣷو اَبْ لِمْ:‏ مُيْ فُڪّ اَڪْ ݧَارْ، نِ تُرْ يِ تࣷلُّ؞ دࣹيسْ ݖِيْ ݧَاسْ تُرْ يِ نِ خࣵمِّڪَايِ تࣷرْلُ، ڪُ نࣹڪّ اَڪْ تُرَمْ، بَ فُڪِّ گِيرْ يࣹيڪْ ݧَارْ دَجْ؞ ‏22 «نَنگَ سࣵڪَّلْ ڪِيرَايَلْ دࣴنّ بِ ݧَارِ ݖَقِ وُرُسُ نگَلَمْ يُ ݖِيْ تࣹگࣹ، تࣹ رَوَاسࣷو نِ اَيْ بُومْ، ‏23 اَڪْ ݧَارِ جَارࣷيْ وُرُسْ يُ ݧُيْ تَڪّ ݖِ ݧَارِ ݖَتِ ڪِيرَايَلْ دࣴنّ بِ ݖِ ڪَوْ، ‏24 تࣹ نگَ نَسْ ݧَارِ بُومِ وُرُسْ يِ ݖِ ݧَارِ جَارࣷ، يِ ݖِ ݖَتِ ڪِيرَايْ لِ؞ ‏25 يࣹنࣹينْ ݧَارِ ݖَتِ بُومِ وُرُسْ يِ، نگَ وࣹ لࣹينْ ݖِ ݧَارِ ݝَنڪُڪَايْ، يِ ݖِ ڪَوْ نگࣵلِّ وَگِّ خَرْ-سࣵنِّ مِ، ݖِ ڪَنَمْ؞ ‏26 سࣵڪَّلْ ڪࣷ اِتْ ݧَارِ جَارࣷيْ وُرُسْ يُ ݧُيْ وࣹ ݖِ ݧَارِ ݖَتِ سُوفِ ڪِيرَايْ لِ، فِ فࣺيتࣹ بِيرْ، وࣹتْ گِ تَفُ ݖِ خَرْ-سࣵنِّ مِ؞ ‏27 نگَ سࣵڪَّلاَتْ ڪࣷ يࣹنࣹينْ ݧَارِ جَارࣷيْ وُرُسْ يࣷويْ وࣹ ݖِ سُوفُ نگࣵلِّ خَرْ-سࣵنِّ مِ، فِ فࣺيتࣹ ڪَنَمْ، دࣹنْدࣹيڪْ سࣷفُ نگࣵلّ يِ، تࣹ تِيمْ نگَݧَايُ خَرْ-سࣵنِّ مِ؞ ‏28 ڪِيرَايَلْ دࣴنّ بِ، بُومْ گُ سࣹوْ گُ بَخَ تࣹ لاَلْ يࣷلࣹتْ، مࣷومْ لࣹيسْ ڪࣷيْ تࣺيࣹي، لࣴڪّلࣹي ڪࣷ جَارࣷيْ ڪِيرَايْ لࣹيڪْ جَارࣷيْ خَرْ-سࣵنِّ مِ، سُ ڪࣷ دࣹفࣹي ڪِيرَايْ لِ دࣴڪّࣹ تِيمْ نگَݧَايْ لِ، بَ دُ تࣹقْلِڪࣷوڪْ خَرْ-سࣵنِّ مِ؞ ‏29 هَـٰرُونَ نَگْ نَيْ تࣹگࣷو تُرِ بࣵنِّ إِسْرَاءِيلْ ݖِ ڪَوْ ڪِيرَايَلْ دࣴنّ بِ ݧُيْ اࣵتّࣹي، تُرْ يِ تَفُ ݖِ خࣷلَمْ، سَا يُيْ دُگّ ݖِ بࣺرَبْ بُ سࣹلّ بِ، نگِرْ سَخَلْ ݒࣵتّلِڪُ بَ فࣵوّ فِ ڪَنَمْ اَجِ سَخْ جِ؞ ‏30 نَنگَ يࣹبْ ݖِ بِيرْ ڪِيرَايَلْ دࣴنّ بِ ݧُيْ اࣵتّࣹي، جُمْتُڪَايِ تࣹگْتَلْ يࣷويُ دِ اُرِمْ اَڪْ تُمِمْ¤، نگِرْ ݧُ تَفُ ݖِ خࣷلُ هَـٰرُونَ، سَا يُيْ تَخَوِ فِ ڪَنَمْ اَجِ سَخْ جِ؞ نࣷونُ لَ هَـٰرُونَ دِ دࣴڪّࣹ يٗبَّالࣹ ݖِ ڪَوْ خࣷلَمْ جُمْتُڪَايِ اࣵتّࣹ يُ بࣵنِّ إِسْرَاءِيلْ، سَا سُيْ تَخَوِ فِ ڪَنَمْ اَجِ سَخْ جِ، بَ فࣵوّ؞
لُ جࣴمْ ݖِ يࣺرࣹ يُ سࣹلّ، يِ ݖِ دࣹسْ
‏31 «فࣴخْيَ بِ خَرْ-سࣵنِّ مِيْ تࣹگࣹ، لࣺݒّ، وࣴݧْ گُ بَخَ تࣹ لاَلْ يࣷلࣹتْ نگَ ڪࣷيْ لِگّࣺييࣹ بَ مُ دَجْ؞ ‏32 نَ بَاتُ فࣴخْيَ بِ نࣹڪّ ݖِ دِگّ بِ، مࣹلْ نِ بَاتُ مبُبَّمْ خَرࣹ، تࣹ نَݧُ ڪࣷ اࣷمْبࣹ اࣷمبّ بُ دࣴگࣴرْ بَ مُ دَجْ، نگِرْ دُ خࣷتِّڪُ؞ ‏33 نَنگَ تَڪّ ݖِ اࣷمْبُ فࣴخْيَ بِ بَ مُ دَجْ، اَيْ دࣷومِ گࣴرࣴنَاتْ يُ ݧُ دࣹفَرࣹ وࣴݧْ گُ بَخَ تࣹ لاَلْ يࣷلࣹتْ، اَڪْ وࣴݧْ گُ خࣹوَرْ اَڪْ گُ خࣷنقّ ݖُرّ، تࣹ دِگَّنْتࣹ گࣴرࣴنَاتْ يِ، نگَ دࣷخَلࣹ ݖِ اَيْ جٗولٗولِ يُ وُرُسْ بَ اࣷمبّ بِ دَجْ؞ ‏34 اَبْ جٗولٗولِبْ وُرُسْ، اَبْ دࣷومُ گࣴرࣴنَاتْ، اَبْ جٗولٗولِبْ وُرُسْ، اَبْ دࣷومُ گࣴرࣴنَاتْ، بَ اࣷمْبُ فࣴخْيَ بِ دَجْ؞ ‏35 نَ ڪࣷ هَـٰرُونَ سࣷلْ بُ نࣹڪّࣹي ݖِ لِگّࣺييُ سَرْخَلࣹيمْ؞ سُ ڪࣷ دࣹفࣹي ݖࣷووَلْ جٗولٗولِ يِ دِ جِبْ، بَ بُيْ جࣵلّ ݖِ بِيرْ بࣺرَبْ بُ سࣹلّ بِ، جࣴمْ فِ ڪَنَمْ اَجِ سَخْ جِ، مبَا مُ دِ ݖِ گࣺنّࣹ، دُ تَخْ مُ دࣹي؞
‏36 «سࣵڪَّلْ دࣷگُ وُرُسُ نگَلَمْ وُ تࣹلّ، نگَ ݧَاسْ ݖِ مبِندّ مِي نِ ݧُيْ ݧَاسࣹ مبِنْدُمْ تࣷرْلُ:‏ ‹لُ سࣹلَّ نگِي، ݧࣹيلْ اَجِ سَخْ جِ؞› ‏37 تَڪّࣹ ڪࣷ بُومْ گُ بَخَ تࣹ لاَلْ يࣷلࣹتْ ݖِ ڪَوْ ڪَالَ گِ، مُ فࣺيتࣹ ڪَنَمْ؞ ‏38 نَ تَفُ ݖِ جࣴوُ هَـٰرُونَ، مُ گࣵدّࣷو ڪࣷ اَيْ وُ تُڪّࣹي ݖِ لُ بࣵنِّ إِسْرَاءِيلْ مࣷيْ ݖِ وࣵلُّ يࣴفْ يُ سࣹلّ يِ ݧُ وَرَ سࣹلَّلْ ݖِ سࣹينْ مبࣷولࣹمْ سَرَخْ يُ سࣹلّ؞ نَ نࣹڪّ فࣵوّ ݖِ جࣴعࣴمْ، ندَخْ اَجِ سَخْ جِ نَنگُلْ لࣹينْ؞
‏39 «مبُبّ مُ گُدّ مِ نَگْ، نَنگَ ڪࣷ رࣵبّࣹ وࣴݧُ لࣹي، ڪَالَ گِ اِتَمْ دِ وࣴݧُ لࣹي؞ تࣹ لَخَسَايْ گِ، لِگّࣺييُ رࣵبّ بُ رَفࣹتْ لَيْ دࣷونْ؞ ‏40 دࣷومِ هَـٰرُونَ يُ گٗورْ يِ اِتَمْ، سࣵڪَّلْ لࣹينْ اَيْ مبُبّ؞ سࣵڪَّلْ لࣹينْ اَيْ لَخَسَايْ، تࣹ اَيْ ڪَالَ اِتْ، سࣵڪَّلْ لࣹينْ ڪࣷ نگِرْ دَرَجَ اَڪْ تࣹرَانگَ بُ سَرْخَلْڪَتْ وَرَ يࣹلّࣷو؞ ‏41 نَنگَ سࣷلَلْ يࣷويِي سَ مَگْ هَـٰرُونَ، مࣷوڪْ اَيْ دࣷومَمْ يُ گٗورْ، تࣹ نگَ سࣷتِّ لࣹينْ دِوُ ݒَلْ¤ گِ، دࣺنڪّ لࣹينْ سࣹينْ سَسْ؞ نࣷونُ نگَ لࣹينْ دِ سࣺدّࣹ ݖࣺرُ سࣹلّنگَ، ݧُ دِ مَ سَرْخَلَلْ؞ ‏42 سࣵڪَّلْ لࣹينْ اَيْ تُبࣺيِ نجِيتْلاَيِ لࣹي، ݧُ سࣵنگࣹ ڪࣷ سࣹينِ ݖࣺرْ، دَلࣹ ڪࣷ ݖِ ندِگّ لِ بَ ݖِ لُݒّ يِ؞ ‏43 نَ هَـٰرُونَ اَڪِ دࣷومَمْ جِيتْلࣷو لࣷولُ، بُ ݧُيْ دُگّ ݖِ خَيْمَبْ ندَجࣹ مِ، مبَا ݧُيْ جࣹگࣹجِ سَرْخَلُڪَايْ بِ، نگِرْ لِگّࣺييْ ݖِ بࣺرَبْ بُ سࣹلّ بِ، ندَخْ ݧُ بَݧَ يٗبّࣹ سࣹينْ بࣷݒّ مبُگَلْ، مُيْ سࣹينْ دࣹي؞ نَ هَـٰرُونَ اَڪْ اَسْڪَنَمْ سࣵرْتࣷو لࣷولُ فࣵوّ؞
