دࣹفَرْ نَݧُ يࣺرࣹيْ سَرْخَلْڪَتْ يِ
39
‏1 وࣴݧْ گُ بَخَ گِ تࣹ لاَلْ يࣷلࣹتْ نَگْ اَڪْ وࣴݧْ گُ خࣹوَرْ گِ اَڪْ گُ خࣷنقّ گِ ݖُرّ، ݖِ لَݧُ سࣵڪّ يࣺرࣹ يُ يࣵنجّ يِ ݧُيْ لِگّࣺييࣹ ݖِ بࣺرَبْ بُ سࣹلّ بِ؞ ݧُ دَلْدِ دࣹفَرْ يࣺرࣹ يُ سࣹلّ، يِ ݧࣹيلْ هَـٰرُونَ نࣷونࣹي ڪࣷ اَجِ سَخْ جِ سَنْتࣹ وࣷونْ مُوسَىٰ؞
‏2 خَرْ-سࣵنِّ مِ، وࣴݧُ وُرُسْ لࣹيسْ ڪࣷ رࣵبّࣹ، اَڪْ وࣴݧْ گُ بَخَ تࣹ لاَلْ يࣷلࣹتْ، اَڪْ وࣴݧْ گُ خࣹوَرْ، اَڪْ گُ خࣷنقّ ݖُرّ اَڪْ وࣴݧُ لࣹي؞ ‏3 اَيْ خࣷبِ وُرُسْ لَݧُ تࣴگّ بَ مُ جࣹڪّ، ݧُ دَگّ ڪࣷ اَيْ وࣴݧْ يُ ݧُ رࣵبَّالࣹيڪْ وࣴݧْ گُ بَخَ گِ تࣹ لاَلْ يࣷلࣹتْ، اَڪْ وࣴݧْ گُ خࣹوَرْ گِ، اَڪْ گُ خࣷنقّ گِ ݖُرّ، اَڪْ وࣴݧُ لࣹي گِ؞ ݧُ دَلْدِ ݖِيْ ݧَوَالࣹ اَيْ نَتَالْ؞ ‏4 ݧَارِ نگࣵلِّ مبَگّ لَݧُ ڪࣷ دࣹفَلْ، دَلْدِ فَسْتࣹ وُ ݖِ نࣹڪّ ݖِ ݧَارِ ݖَتِ خَرْ-سࣵنِّ مِ؞ ‏5 نگَݧَايُ تَارَلْ لِ تࣹگࣹ ݖِ ڪَوْ، نِ خَرْ-سࣵنِّ مِ لࣹيسْ ڪࣷ دࣹفَرࣹ، لِگّࣺييَالࣹ ڪࣷوڪْ مࣷومْ، ݧُ اࣵندّ دِ بࣹنّ؞ ݧُ رࣵبّࣹ ڪࣷ وࣴݧُ وُرُسْ، اَڪْ وࣴݧْ گُ بَخَ تࣹ لاَلْ يࣷلࣹتْ، اَڪْ وࣴݧْ گُ خࣹوَرْ، اَڪْ گُ خࣷنقّ ݖُرّ، اَڪْ وࣴݧُ لࣹي، نࣷونࣹي ڪࣷ اَجِ سَخْ جِ سَنْتࣹ وࣷونْ مُوسَىٰ؞
‏6 ݖِ ڪَوْ لࣷولُ ݧُ وࣹ ݒࣹرِ اࣷنِگْسࣴ يِ، فࣹگࣹ ݒࣹرْ يِ ݝَنڪُڪَايِ وُرُسْ، دَلْدِ ݖِيْ ݧَاسْ تُرِ دࣷومِ إِسْرَاءِيلْ يُ گٗورْ، نࣷونُ ݧُيْ ݧَاسࣹ خࣵمِّڪَايْ يِ ݧُيْ تࣷرْلࣷو؞ ‏7 ݧُ تَݒّ ݒࣹرْ يِ ݖِ ڪَوْ نگࣵلِّ وَگِّ خَرْ-سࣵنِّ مِ، ݒࣹرْ يِ نَگْ تَخَوࣹ تَخَوَايُ بࣵنِّ إِسْرَاءِيلْ، نࣷونُ ڪࣷ اَجِ سَخْ جِ سَنْتࣹ وࣷونْ مُوسَىٰ؞
دࣹفَرْ نَݧُ ڪِيرَايَلْ دࣴنّ بِ
‏8 ݧُ سࣵڪّ ڪِيرَايَلْ دࣴنّ بِ، لِگّࣺييَالࣹ ڪࣷ اَيْ نَتَالْ نِ خَرْ-سࣵنِّ مِ، رࣵبّࣹ ڪࣷ وࣴݧُ وُرُسْ، اَڪْ وࣴݧْ گُ بَخَ تࣹ لاَلْ يࣷلࣹتْ، اَڪْ وࣴݧْ گُ خࣹوَرْ، اَڪْ گُ خࣷنقّ ݖُرّ، اَڪْ وࣴݧُ لࣹي؞ ‏9 اَبْ ڪَارࣹ بُ ݧُ لࣹمْ لَݧُ ڪࣷ دࣹفْ، گُدَّايْ بِ، گَنَّاوْ بُ ݧُ ڪࣷ لࣹمࣹي، دِ ݝَرَلࣹبْ بَارَامُ دࣺيْ اَڪْ دِگّ، يَاتُوَايْ بِ دِ ݝَرَلࣹبْ بَارَامُ دࣺيْ اَڪْ دِگّ؞ ‏10 ݧُ وࣹ نَگْ اَيْ ݒࣹرْ ݖِ ڪَوْ ڪِيرَايَلْ دࣴنّ بِ، دࣹفْ ڪࣷ ݧࣹينْتِ لࣵنگِ ݒࣹرْ؞ لࣵنگّ گِ جِيتُ، ݒࣹرُ سَرْدُوَنْ اَڪْ ݒࣹرُ تࣷݒَاسْ اَڪْ ݒࣹرُ اࣹسْڪَرْبُڪَلْ مࣷو ݖَ نࣹڪّ؛ ‏11 ݧَارࣹيلُ لࣵنگّ گِ دِ ݒࣹرُ اࣹمࣴرٗدّ اَڪْ ݒࣹرُ سَفِيرْ اَڪْ ݒࣹرُ جَمَا، ‏12 ݧࣹتّࣹيلُ لࣵنگّ گِ دِ ݒࣹرُ يَسࣹنتّ اَڪْ ݒࣹرُ اَگَاتْ اَڪْ ݒࣹرُ اَمࣹتِسْتࣴ، ‏13 ݧࣹينْتࣹيلُ لࣵنگّ گِ دِ ݒࣹرُ ڪِرْسࣷلِتْ اَڪْ ݒࣹرُ اࣷنِگْسࣴ اَڪْ ݒࣹرُ جࣵسْݒࣴ، تࣹ ݒࣹرْ بُ ݖِ نࣹڪّ ݝَنڪُڪَايُ وُرُسْ لࣹيسْ ڪࣷ وࣹوࣹ؞ ‏14 ݒࣹرْ يࣹيڪْ تُرِ دࣷومِ إِسْرَاءِيلْ يُ گٗورْ ݧࣷو تࣷلّࣷو اَبْ لِمْ:‏ مُيْ فُڪّ اَڪْ ݧَارْ، نِ تُرْ يِ تࣷلُّ؞ ݧُ ݧَاسْ ݖِ تُرْ يِ نِ خࣵمِّڪَايِ تࣷرْلُ، ڪُ نࣹڪّ اَڪْ تُرَمْ، بَ فُڪِّ گِيرْ يࣹيڪْ ݧَارْ دَجْ؞
‏15 ݖِ بِيرْ لࣷولُ ݧُ سࣵڪَّلْ ڪِيرَايَلْ دࣴنّ بِ ݧَارِ ݖَقِ وُرُسُ نگَلَمْ يُ ݖِ تࣹگࣹ، تࣹ رَوَاسࣷو نِ اَيْ بُومْ؞ ‏16 ݧُ سࣵڪّ اِتَمْ ݧَارِ ݝَنڪُڪَايِ وُرُسْ اَڪْ ݧَارِ جَارࣷيْ وُرُسْ يُ ݧُ تَڪّ ݖِ ݧَارِ ݖَتِ ڪِيرَايَلْ دࣴنّ بِ ݖِ ڪَوْ ‏17 ݧُ نَسْ نَگْ ݧَارِ بُومِ وُرُسْ يِ ݖِ ݧَارِ جَارࣷ يِ ݖِ ݖَتِ ڪِيرَايْ لِ؞ ‏18 يࣹنࣹينْ ݧَارِ ݖَتِ بُومِ وُرُسْ يِ، ݧُ وࣹ لࣹينْ ݖِ ݧَارِ ݝَنڪُڪَايْ، يِ ݖِ ڪَوْ نگࣵلِّ وَگِّ خَرْ-سࣵنِّ مِ، ݖِ ڪَنَمْ؞ ‏19 ݧُ سࣵڪَّلْ ڪࣷ اِتَمْ ݧَارِ جَارࣷيْ وُرُسْ، يُ ݧُ وࣹ ݖِ ݧَارِ ݖَتِ سُوفِ ڪِيرَايْ لِ، فِ فࣺيتࣹ بِيرْ، وࣹتْ گِ تَفُ ݖِ خَرْ-سࣵنِّ مِ؞ ‏20 ݧُ سࣵڪَّلاَتْ ڪِيرَايْ لِ يࣹنࣹينْ ݧَارِ جَارࣷيْ وُرُسْ، يُ ݧُ وࣹ ݖِ سُوفُ نگࣵلِّ خَرْ-سࣵنِّ مِ، فِ فࣺيتࣹ ڪَنَمْ، دࣹنْدࣹيڪْ سࣷفُ نگࣵلّ يِ، تࣹ تِيمْ نگَݧَايُ خَرْ-سࣵنِّ مِ؞ ‏21 ڪِيرَايَلْ دࣴنّ بِ، بُومْ گُ سࣹوْ گُ بَخَ تࣹ لاَلْ يࣷلࣹتْ، مࣷومْ لࣹيسْ ڪࣷ تࣺيࣹي، لࣴڪّلࣹي ڪࣷ جَارࣷيْ ڪِيرَايْ لࣹيڪْ جَارࣷيْ خَرْ-سࣵنِّ مِ، ڪِيرَايْ لِ دࣴڪّࣹ تِيمْ نگَݧَايْ لِ، بَ دُ تࣹقْلِڪࣷوڪْ خَرْ-سࣵنِّ مِ، نࣷونࣹي ڪࣷ اَجِ سَخْ جِ سَنْتࣹ وࣷونْ مُوسَىٰ؞
دࣹفَرْ نَݧُ يࣺرࣹيْ سَرْخَلْ، يِ ݖِ دࣹسْ
‏22 فࣴخْيَ بِ خَرْ-سࣵنِّ مِيْ تࣹگࣹ، لࣺݒّ، وࣴݧْ گُ بَخَ تࣹ لاَلْ يࣷلࣹتْ لࣹيسْ ڪࣷ رࣵبّࣹ بَ مُ دَجْ؞ ‏23 بَاتُ فࣴخْيَ بَا نگِ ݖِ دِگّ بِ، مࣹلْ نِ بَاتُ مبُبَّمْ خَرࣹ، ݧُ اࣷمْبࣹ ڪࣷ اࣷمبّ بُ دࣴگࣴرْ، بَ مُ دَجْ، نگِرْ دُ خࣷتِّڪُ؞ ‏24 ݖِ بِيرْ لࣷولُ ݧُ تَڪّ ݖِ اࣷمْبُ فࣴخْيَ بِ، اَيْ دࣷومِ گࣴرࣴنَاتْ يُ ݧُ دࣹفَرࣹ وࣴݧْ گُ بَخَ تࣹ لاَلْ يࣷلࣹتْ، اَڪْ وࣴݧْ گُ خࣹوَرْ، اَڪْ گُ خࣷنقّ ݖُرّ، اَڪْ وࣴݧُ لࣹي؞ ‏25 ݧُ دࣹفْ اِتَمْ اَيْ جٗولٗولِيْ وُرُسُ نگَلَمْ، بَ تَڪّ جٗولٗولِ يِ ݖِ دِگَّنْتࣹ دࣷومِ گࣴرࣴنَاتْ يِ، وࣴرَلࣹ ڪࣷ اࣷمْبُ فࣴخْيَ بِ، بَ مُ دَجْ، جٗولٗولِ يِ دࣷخْ دِگَّنْتࣹ دࣷومِ گࣴرࣴنَاتْ يِ؞ ‏26 اَبْ جٗولٗولِ، اَبْ گࣴرࣴنَاتْ، اَبْ جٗولٗولِ، اَبْ گࣴرࣴنَاتْ بَ اࣷمْبُ فࣴخْيَ، بِ ݧُيْ سࣷلْ دِ سَرْخَلْ دَجْ، نࣷونࣹي ڪࣷ اَجِ سَخْ جِ سَنْتࣹ وࣷونْ مُوسَىٰ؞
‏27 ݧُ سࣵڪَّلْ نَگْ هَـٰرُونَ اَڪِ دࣷومَمْ مبُبّ يُ گُدّ يُ ݧُ رࣵبّࣹ وࣴݧُ لࣹي، ‏28 اَڪْ ڪَالَگْ لࣹي، اَڪْ مبَخَنَيْ لࣹي، اَڪْ تُبࣺيِ نجِيتْلاَيْ يُ لࣹي، ‏29 اَڪْ لَخَسَايْ گِيْ لِگّࣺييُ رࣵبّ بُ رَفࣹتْ، بُ ݧُ رࣵبّࣹ وࣴݧُ لࣹي بُ دࣴگࣴرْ، اَڪْ وࣴݧْ گُ بَخَ تࣹ لاَلْ يࣷلࣹتْ، اَڪْ وࣴݧْ گُ خࣹوَرْ، اَڪْ گُ خࣷنقّ ݖُرّ، نࣷونࣹي ڪࣷ اَجِ سَخْ جِ سَنْتࣹ وࣷونْ مُوسَىٰ؞
‏30 ݧُ سࣵڪّ نَگْ مࣹيتࣹيلْ گُ سࣹلّ گُ وُرُسُ نگَلَمْ، مُيْ دࣷگُ وُرُسْ وُ تࣹلّ، ݧُ ݧَاسْ ݖِ ڪَوَمْ مِي مبِندّ، نِ ݧُيْ ݧَاسࣹ مبِنْدُمْ تࣷرْلُ:‏ «سࣹلّ نَ، ݧࣹيلْ اَجِ سَخْ جِ»؞ ‏31 ݧُ دَلْدِ ڪࣷيْ تَڪّࣹ بُومْ گُ بَخَ تࣹ لاَلْ يࣷلࣹتْ، تࣺيࣹي ڪࣷ ڪࣷ ݖِ ڪَوْ ڪَالَ گِ، نَ ڪࣷ اَجِ سَخْ جِ سَنْتࣹ وࣷونْ مُوسَىٰ؞
مُوسَىٰ مࣹمِّڪُ نَ لِگّࣺييْ بِ
‏32 نࣷونُ لَ مبࣷولࣹمْ لِگّࣺييُ مࣵڪَّانُ خَيْمَبْ ندَجࣹ مِ مَتࣹ؞ نࣷونࣹي ڪࣷ اَجِ سَخْ جِ سَنْتࣹ وࣷونْ مُوسَىٰ رࣹڪّ، نِ لَ ڪࣷ بࣵنِّ إِسْرَاءِيلْ دࣹفࣹ؞
‏33 بَ لࣷولُ اَمࣹي ݧُ اِنْدِلْ مُوسَىٰ جَامࣷوڪَايْ بِ:‏ خَيْمَ بِ اَڪْ يࣴفَمْ يࣺݒّ؛ لࣷنڪُ يِ اَڪْ لࣵݖّ يِ اَڪْ گَلَنْ يِ اَڪْ جࣴنْ يࣹيڪْ تࣹگُ يِ، ‏34 اَڪْ ڪِيرَايَلْ دࣹرِ ڪُويْ يِ ݧُ سٗوبْ خࣷنقّ، اَڪْ ڪِيرَايَلْ دࣹرِ ݒِيݒِ يِ، اَڪْ ڪِيرَايَلْ خࣵجَّتْلࣹ بِ، ‏35 اَڪْ گَالْ گِ اࣵلُّوَيْ سࣹيدࣹ سِ دࣹنݖࣹ، اَڪِ نجࣵݒّࣷومْ، اَڪْ ڪُبࣹيرُ سَرَخُ نجࣷتْلاَيْ گِ، ‏36 اَڪْ تَابَلْ جࣹيڪْ ندَبَمْ يࣺݒّ، اَڪْ مبُرُمْ تࣹيوَلْ مِ ݖِيْ تࣹگࣹ، ‏37 اَڪْ تࣹگُڪَايُ لࣵمݒّ بُ وُرُسُ نگَلَمْ بِ اَڪِ لࣵمْݒَمْ، لࣵمݒّ يِ سࣴسَلࣷو بَ جࣹڪّ اَڪْ مبࣷولࣹمِ جُمْتُڪَايَمْ، اَڪْ دِوْ گِ ݧُيْ تَالࣷو، ‏38 اَڪْ سَرْخَلُڪَايْ بُ وُرُسْ بِ، اَڪْ دِوُ ݒَلْ گِ، اَڪْ ݖُورَايْ لُ خࣹيݧْ لِ، اَڪْ ڪِيرَايَلْ بُنْتُ خَيْمَ بِ، ‏39 اَڪْ سَرْخَلُڪَايُ خࣵنجَرْ بࣹيڪْ ݖَاخُبْ خࣵنجَرَمْ، اَڪِ نجࣵݒّࣷومْ اَڪْ ندَبَمْ يࣺݒّ، اَڪْ مبَلْڪَمْ نجࣵݒّ مِ اَڪُبْ تࣹگࣷومْ، ‏40 اَڪْ سࣷرِ وُرْمبَلُ اࣴتّ بِ اَڪِ جࣴنَمْ اَڪِ تࣹگࣷومْ، اَڪْ ڪِيرَايَلْ لَفُ بُنْتُ اࣴتّ بِ، اَڪِ بُومَمْ اَڪِ خࣹرَمْ؛ مبࣷولࣹمْ جُمْتُڪَايْ، يِ ݧُيْ لِگّࣺييࣹ ݖِ مࣵڪَّانُ خَيْمَبْ ندَجࣹ مِ رࣹڪّ، ‏41 اَڪْ يࣺرࣹ يُ يࣵنجّ يِ ݧُيْ لِگّࣺييࣹ ݖِ بࣺرَبْ بُ سࣹلّ بِ؛ يࣺرࣹ يُ سࣹلّ يُ هَـٰرُونَ سَرْخَلْڪَتْ بِ، اَڪْ يࣺرࣹيْ دࣷومَمْ يِ ݧُيْ سࣷلْ، دِ سَرْخَلࣹ؞ ‏42 نࣷونࣹي ڪࣷ اَجِ سَخْ جِ سَنْتࣹ وࣷونْ مُوسَىٰ رࣹڪّ، نِ لَ بࣵنِّ إِسْرَاءِيلْ سࣷتَّلࣹ لِگّࣺييْ بِ بࣺݒّ؞ ‏43 مُوسَىٰ نَگْ نࣹمِّڪُ مبࣷولࣹمْ لِگّࣺييْ بِ، بَ گِسْ نࣹ سࣷتَّلْ نَݧُ ڪࣷ؛ نࣷونࣹي ڪࣷ اَجِ سَخْ جِ سَنْتَانࣹي وࣷونْ، نِ لَݧُ ڪࣷ دࣹفࣹ؞ مُوسَىٰ دَلْدِ لࣹينْ ݧَانَلْ؞
