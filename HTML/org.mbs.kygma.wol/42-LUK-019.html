عِيسَىٰ دَلْ نَ اَڪْ بࣵڪَّارْڪَتْ بَ
19
‏1 گَنَّاوْ لࣷولُ عِيسَىٰ دُگّ بِيرْ يࣹرِڪࣷ، جَارࣹ بِيرْ دࣴڪّ بَ، دِ دࣹمْ؞ ‏2 ندࣹڪࣹ جࣹنّ وَايَ نگَ فَ، دِ ڪِلِفَگْ جُوتِڪَتْ بُ بَرࣹ اَلَلْ، ݧُ دِ ڪࣷ وَخْ سَسࣹ؞ ‏3 مُ بࣴگَّ گِسْ نَگْ ڪُيْ عِيسَىٰ، تࣹ مبࣷولࣷو مَ تَخْ مَنُ ڪࣷو گِسْ، ندَخْ ندَامَ لَ وࣷونْ؞ ‏4 ݖِ ڪَوْ لࣷولُ سَسࣹ دَوْ، جِيتُجِ، دَلْدِ يࣺيگْ ݖِ گࣹنّ گَرَبْ ݖَ يࣷونْ وࣷووُ عِيسَىٰ وَرَ جَارࣹ، نگِرْ مُ مَنْ ڪࣷو گِسْ؞ ‏5 عِيسَىٰ دِڪّ بَ تࣷلُّ فࣷوفَ، تࣺينْ نࣹ ڪࣷ:‏ «سَسࣹ، وࣵݖَّلْ گَاوْ، ندَخْ سَ ڪࣴرْ لاَ وَرَ دَلِ تࣹيْ جِي؞»
‏6 سَسࣹ دَلْدِ وࣵݖّ گَاوْ، بࣺگࣹ ڪࣷو دَلَلْ لࣷولْ؞ ‏7 ݧࣺݒّ گِسْ لࣷولُ، دِ ݧُرُمْتُ نَانْ بࣵڪَّارْڪَتْ بِ لَ عِيسَىٰ دࣹمْ ڪࣴرَمْ، دَلْ اَڪْ مࣷومْ؞ ‏8 سَسࣹ نَگْ تَخَوْ نࣹ سَنگّ بِ:‏ «سَنگّ بِ، مَ نࣹ، سَمَ گࣹنّ-وࣵلُّ اَلَلْ، جࣷخْ نَا ڪࣷ نࣺيوْ-دࣷولࣹ يِ، تࣹ جࣺݒّ اَلَلْ جُ لࣹوُلْ جُ مَ مَسَ جࣴلَلْ ڪࣹنّ، دِنَا فࣹيْ ݧࣹينْتَمْ؞» ‏9 عِيسَىٰ نࣹ ڪࣷ:‏ «بࣺسْ نِڪِ تࣹيْ، مُݖّ دِڪَّلْ نَ ڪࣴرْ گِي، ندَخْ وَا جِي ڪَتْ ݖِ اَسْڪَنُ إِبْرَاهِيمَ لَ بࣷڪّ، مࣷومْ اِتَمْ؞ ‏10 ندَخْ دࣷومُ نِتْ ڪِ، ݧِ رࣺيرْ لَ سࣹيتْسِ، نگِرْ مُسَلْ لࣹينْ؞»
ندࣺنڪَانࣹ اَوْ سَسْ لَ
‏11 نَڪَ لَ نِتْ ݧَ دࣺگّ لࣷولُ، عِيسَىٰ لࣺيبَتِ لࣹينْ، ندَخْ بࣷوبَ جࣹگࣹ نَ يࣹرُسَلࣹمْ، تࣹ فࣹڪّ نَ ݧُ فࣷوگْ نࣹ نگُورُگْ يࣵلَّ دَيْ دَلْدِ تࣹيوْ ݖَ سَا سَ؞ ‏12 مࣷو تَخْ مُ نࣹ:‏ «بࣷرࣷومْ جُدُّ بُ رࣺيْ لَ وࣷونْ؞ مُيْ وَاجَ دࣹمْ رࣺيوْ مُ سࣷرࣹ، نگِرْ ݧُ فَلْ ڪࣷ فَ بُورْ، مُ دࣷورَ ݧِبّسِ؞ ‏13 ݖِ بِيرْ لࣷولُ مُ وࣷو فُڪِّ سُرْگَامْ، جࣷخْ لࣹينْ ڪُ نࣹڪّ بࣹنّ ݒࣷسࣹتُبْ وُرُسْ بُ گࣵنجَرࣹ، نࣹ لࣹينْ:‏ ‹لِگّࣺييْلࣹينْ ݖِ، بَ مَ ݧِبّسِ؞› ‏14 وَايࣹ وَا رࣺيوَمْ ݧَ دَݧُ ڪࣷو بَݧࣷونْ؞ ݖِ گَنَّاوَمْ لَݧُ يٗنِّ ندَوْ، نࣹ:‏ ‹ڪِي، بُگُّݧُ مُيْ سُنُ بُورْ؞›
‏15 «گَنَّاوْ بَ ݧُ ڪࣷ فَلࣹي بُورْ، بَ مُ دࣺلْسِ، دَا وࣷولُ سُرْگَ، يَ مُ جࣷخࣷونْ وُرُسْ وَ، نگِرْ خَمْ ݧَاتَ لَ لࣹينْ ݖَ سࣹينْ لِگّࣺييْ جُرَلْ؞ ‏16 ڪࣹنّ ڪَ دِڪّ، نࣹ ڪࣷ:‏ ‹سَنگّ بِ، سَ ݒࣷسࣹتْ بَ جُرْ نَ فُڪِّ ݒࣷسࣹتْ؞› ‏17 مُ نࣹ ڪࣷ:‏ ‹جَارَامَا، سُرْگَ بُ بَاخْ بِ؞ گَنَّاوْ ݖِ لُ تُوتْ نگَ وٗورࣹ نَگْ، فَلُلْ ݖِ فُڪِّ دࣴڪّ؞› ‏18 ڪࣹنࣹينْ دِڪّ، نࣹ:‏ ‹سَنگّ بِ، سَ ݒࣷسࣹتْ بَ اِنْدِ نَ جُرٗومِ ݒࣷسࣹتْ؞› ‏19 مُ نࣹيتِ ڪࣷوڪَ:‏ ‹يَوْ، فَلُلْ ݖِ جُرٗومِ دࣴڪّ؞› ‏20 ڪࣹنࣹينْ دِڪّ نࣹ:‏ ‹سَنگّ بِ، سَ ݒࣷسࣹتْ بَا نگِي مَ فَسࣷونْ ݖِوْ سࣹڪِّتْ؞ ‏21 دَمَ لاَ رَگَلْ، ندَخْتࣹ ڪُ نࣺيگْ نگَ؞ دَنگَيْ فࣷقَتِ لࣷو دࣺنڪَانࣹوُلْ وࣷونْ، دِ دَجَلࣹ لࣷو جِوُلࣷونْ؞› ‏22 مُ نࣹ ڪࣷ:‏ ‹سَيْ وَخْ لاَ لَيْ اࣵتّࣹي، سُرْگَ بُ بࣷنْ بِ!‏ خَمࣷونْ نگَ نࣹ نِتْ ڪُ نࣺيگْ لاَ، دِ فࣷقَتِ لُ مَ دࣺنڪَانࣹوُلْ وࣷونْ، دِ دَجَلࣹ لُ مَ جِوُلࣷونْ؞ ‏23 لُ لَ تࣹي وࣷونَ دࣺنڪَانࣹجِ سَمَ اَلَلْ ݖَ دࣹنݖُڪَايْ بَ، نگِرْ بُ مَ دِڪّࣹي جࣷتْ ڪࣷ، بࣷولࣹ ݖَ تࣷنࣷومْ؟›
‏24 «ݖِ ڪَوْ لࣷولُ مُ نࣹ ݧَ فَ تَخَوࣷونْ:‏ ‹نَنگُلࣹينْ ݒࣷسࣹتْ بِ ݖِ مࣷومْ، جࣷخْ ڪࣷ بࣷرࣷومْ فُڪِّ ݒࣷسࣹتْ يِ؞› ‏25 ݧُ نࣹ ڪࣷ:‏ ‹سَنگّ بِ، ڪࣹي نگࣹيڪْ فُڪِّ ݒࣷسࣹتْ بَ نࣷݒِّ!‏› ‏26 مُ نࣹ:‏ ‹مَ نࣹ لࣹينْ، ڪࣺݒّ ڪُ اَمْ، دࣹيسْ نَ ڪࣷ دࣷلِّ، وَايࣹ ڪُ اَمُلْ، لَ مُ اَمْ اَسْ نࣺيوْ سَخْ، دࣹيسْ نَ ڪࣷ نَنگُ؞ ‏27 سَمَ نࣷونْ يࣷويُ بُگُّلْ وࣷونْ مَ دࣷونْ سࣹينْ بُورْ نَگْ، اِنْدِلࣹينْ لࣹينْ فِي، تࣹ رࣹنْدِ لࣹينْ ݖِ سَمَ ڪَنَمْ؞›»
عِيسَىٰ دُگّ نَ يࣹرُسَلࣹمْ
‏28 بَ عِيسَىٰ وَخࣹي لࣷولُ بَ نࣷݒِّ، ݖَ لَ جِيتُ، جࣴمْ يࣹرُسَلࣹمْ؞ ‏29 بَ عِيسَىٰ دِڪّࣹي بَ جُبْ دࣴڪّ يَ ݧُيْ وَخْ بࣹتْفَسࣹ اَڪْ بࣹتَنِ، ݖَ وࣹتُ تُندّ وَ ݧُيْ وَخْ تُنْدُ اࣷلِوْ يَ، دَفَ يࣹبَلْ ݧَارِ تَالِبࣹيمْ، ‏30 نࣹ لࣹينْ:‏ «دࣹمْلࣹينْ ݖِ دࣴڪّ بِ ݖِ ڪَنَمْ؞ بُ نگࣹينْ اَگْسࣹي، دِنگࣹينْ گِسْ ݖُمْبُرْ بُ يࣹيوࣹ، بُ ڪࣹنّ مَسُلَ وَرْ؞ يِوِلࣹينْ ڪࣷ، اِنْدِ؞ ‏31 بُ ݧُ لࣹينْ نࣹي:‏ ‹لُ تَخْ نگࣹينْ دِ ڪࣷ يِوِ؟› نࣹ لࣹينْ:‏ ‹سَنگّ بࣹي ڪࣷ سࣷخْلَ؞›»
‏32 بَ ندَوْ يَ دࣹمࣹي، نَ لࣹينْ ڪࣷ عِيسَىٰ وَخࣹ لَݧُ ڪࣷ گِسࣹ؞ ‏33 نَڪَ لَݧُيْ يِوِ ݖُمْبُرْ بَ، اَيْ بࣷرࣷومَمْ نࣹ لࣹينْ:‏ «لُ تَخْ نگࣹينْ دِ يِوِ ݖُمْبُرْ بِ؟» ‏34 ݧُ نࣹ:‏ «بࣷرࣷومَمْ مࣷو ڪࣷ سࣷخْلَ؞» ‏35 ݧُ اِنْدِلْ عِيسَىٰ ݖُمْبُرْ بِ، سࣵنِّ سࣹينِ يࣺرࣹ ݖَ ڪَوَمْ، دَلْدِ وَرَلْ عِيسَىٰ؞
‏36 نَڪَ لَ عِيسَىٰ دِ دࣹمْ، نِتْ ݧِ دِ ڪࣷ لَلَلْ سࣹينِ يࣺرࣹ ݖَ يࣷونْ وَ؞ ‏37 بَ ݧُ جُبْسࣹي يࣹرُسَلࣹمْ، بَيْ وࣵݖّࣹ ݖَ مبَرْتَلُمْ تُنْدُ اࣷلِوْ يَ، مبࣷولࣷومْ تَالِبࣹ مࣺݒّ لَ مبࣺگْتࣹ جࣵݒّ، ݧُ تࣵمْبَلࣹي سࣵبَّالْ يࣵلَّ ݖَ ڪَوْ ندَخْ ڪࣺيمَانْ يَ ݧُ گِسْ؞ ‏38 ݧَ نگَ نَانْ:‏
«نَ بَرْڪࣹ وࣵݖَّلْ بُورْ بِ ݧࣴوْ ݖِ تُرُ بࣷرࣷومْ بِ!‏
جࣵمّ فَ اَسَمَانْ؛ دَرَجَ فَ بࣺرَبْ يَ گࣴنَ ڪَوࣹ!‏»
‏39 يࣹنّ فَرِسࣹنْ يَ وࣷونْ ݖَ مبࣷولࣷو مَ نَگْ نࣹ عِيسَىٰ:‏ «سࣴرِݧْ بِ، يࣹدَّلْ سَ تَالِبࣹ يِ!‏» ‏40 مُ نࣹ:‏ «مَا لࣹينْ ڪࣷ وَخْ، سُ ݧِي نࣷݒِّ وࣷونْ، دࣷجْ يِ ݧࣷويْ خَاݖُ؞»
‏41 بَ عِيسَىٰ جُبْسࣹي بَ سࣺينْ دࣴڪّ بَ، جࣷويْ نَ ندَخْ مࣷومْ؞ ‏42 مُ نࣹ:‏ «ݖࣺيْ يَوْ اِتَمْ، بࣷو خَمࣷونْ بࣺسُبْ تࣹيْ جِي لِ لَيْ مَيْ جࣵمّ!‏ وَايࣹ فِي مُ نࣹ، لࣵقُ نَ سَيْ گࣴتْ؞ ‏43 اَيْ بࣺسَ نگِ دِڪّ ݖِ سَ ڪَوْ، ڪࣹرࣷوگْ سَيْ نࣷونَ لَيْ گَوࣹ اَبْ سࣴڪّ، يࣺيوْ لَ، تَنݖࣹ لَ وࣹتْ گُ نࣹڪّ؞ ‏44 ݧࣷو لَيْ نَݒَّاجࣹ ݖِ سُوفْ، يَاڪْ سَيْ دࣷومْ يِ ݖِ سَ بِيرْ، تࣹ دُݧُ بࣵيِّ دࣷجْ ݖِ ڪَوْ دࣷجْ ݖِ سَ بِيرْ، ندَخْ بَ سَ بࣺسُبْ وَلّ دِڪّࣹي، يَا ڪࣷ رࣵݧّࣹيوُلࣷونْ؞»
عِيسَىٰ جࣵنگّلࣹ نَ ݖَ ڪࣴرْ يࣵلَّ گَ
‏45 گَنَّاوْ لࣷولُ عِيسَىٰ دُگّ ݖَ ڪࣴرْ يࣵلَّ گَ، دَلْدِ دࣵقْ ݧَ فَيْ جَايࣹ؞ ‏46 مُ نࣹ لࣹينْ:‏ «بِنْدࣹيسْ نَ نࣹ:‏ ‹سَمَ ڪࣴرْ گِ ڪࣴرُگْ جُلِّ لَ، وَايࣹ يࣹينْ، مࣷومْ نگࣹينْ دࣹفْ رَوْتُبْ سࣵݖّ!‏›»
‏47 عِيسَىٰ نَگْ دِ جࣵنگّلࣹ بࣺسْ بُ نࣹڪّ ݖَ ڪࣴرْ يࣵلَّ گَ؞ بَ مُ ڪࣷ دࣹفࣹي سَرْخَلْڪَتْ يُ مَگْ يَ، اَڪْ فِرِڪَتِ يࣷونْ يَ، ݧࣷوڪْ نجِيتِ خࣹيتْ وَ، دِ ڪࣷ وُتَ رࣹيْ؞ ‏48 وَايࣹ خَمُݧُ نُ ݧُيْ دࣹفْ، ندَخْ ݧࣺݒَّ ڪࣷ دࣷونْ دࣺگْلُ، تࣹيوْلُ ڪࣷ لࣷولْ؞
