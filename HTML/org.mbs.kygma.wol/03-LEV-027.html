لُ جࣴمْ ݖِ جَگْلࣹيلْ اَجِ سَخْ جِ نِتْ اَڪْ اَلَلْ
27
‏1 بُ لࣷولُ وࣹيسࣹي اَجِ سَخْ جِ وَخْ نَ مُوسَىٰ نࣹ ڪࣷ:‏ ‏2 «وَخَلْ بࣵنِّ إِسْرَاءِيلْ نࣹ لࣹينْ:‏ ڪࣺݒّ ڪُ دࣷگࣷو جَگْلࣹيلْ اَجِ سَخْ جِ نِتْ¤، مَنْ نَا وࣵݖّࣷوڪْ ڪࣵدّࣷومْ ݖِ ڪَوْ نِي ݧُ ڪࣷ خَيْمَا؞ ‏3 سُ دࣹي گٗورْ گُ اَمْ ݧَارْ فُڪِّ اَتْ بَ جُرٗومْ بࣹنّ فُڪِّ اَتْ، نَنگَ ڪࣷ خَيْمَ ݖِ جُرٗومْ فُڪِّ دࣷگِ خَالِسْ، ݖِ خَالِسُ بࣺرَبْ بُ سࣹلّ بِ؞ ‏4 سُ دࣹي جِگࣺينْ، نَنگَ ڪࣷ خَيْمَ ݖِ فَنْوࣹيرِ دࣷگِ خَالِسْ؞ ‏5 سُ اَمࣹي جُرٗومِ اَتْ بَ ݧَارْ فُڪِّ اَتْ، دَنگَ ڪࣷيْ خَيْمَ ݖِ ݧَارْ فُڪِّ دࣷگِ خَالِسْ، سُ دࣹي گٗورْ، مبَا فُڪِّ دࣷگِ خَالِسْ، سُ دࣹي جِگࣺينْ؞ ‏6 سُ دࣹي ڪُ اَمْ وࣹيرْ جࣵݒّ جُرٗومِ اَتْ، نَنگَ ڪࣷ خَيْمَ ݖِ جُرٗومِ دࣷگِ خَالِسْ سُ دࣹي گٗورْ؛ مبَا ݧࣹتِّ دࣷگِ خَالِسْ سُ دࣹي جِگࣺينْ؞ ‏7 سُ دࣹي ڪُ اَمْ جُرٗومْ بࣹنّ فُڪِّ اَتْ اَڪْ لُ ڪࣷ اࣴݒّ، نَنگَ ڪࣷ خَيْمَ ݖِ فُڪِّ دࣷگِ خَالِسْ اَڪْ جُرٗومْ سُ دࣹي گٗورْ؛ مبَا فُڪِّ دࣷگِ خَالِسْ سُ دࣹي جِگࣺينْ؞ ‏8 سُ فࣹڪّࣹي نࣹ ڪِ دࣷگُ وࣷونَ جَگْلࣹيلْ نِتْ ڪِ اَجِ سَخْ جِ دَ دِ ڪُ اَمْ-اَمَمْ يࣺيسْ سَ خَيْمَ، نَ ڪࣷوڪُ يٗبُّ نِتْ ڪِ بَ ݖَ سَرْخَلْڪَتْ بَ، سَرْخَلْڪَتْ بِ خَيْمَالْ ڪࣷ نِتْ ڪِ ݖِ لُ دࣴݒّࣷوڪْ اَمْ-اَمَمْ، مࣷومْ مِ دࣷگُ وࣷونْ؞
‏9 «سُ فࣹڪّࣹي نࣹ لَ ݧُ دࣷگُ وࣷونَ جࣷخࣹ اَگْ جُرْ لَ گُ دَگَنَ سَرْخَلَلْ اَجِ سَخْ جِ، دَيْ دَلْدِ سࣹلّ؞ ‏10 دࣹيسُ ڪࣷ وࣹݖّࣹي؞ دُ جُرْ گُ بَاخْ گُ ݧُ ڪࣷ مَنَ وُوتࣷو، دࣷونْتࣹ جُرْ گُ بࣷنْ لَ، تࣹ دُ جُرْ گُ بࣷنْ گُ ݧُ ڪࣷ مَنَ وُوتࣷو، تࣹ فࣹڪّ مُيْ گُ بَاخْ؞ ݖِ بِيرْ جَگْلࣹيلْ گࣷوگُ ڪُ خَسَ وُوتࣷو اَگْ جُرْ مࣷرࣷومْ مَ، جُرْ گُ جࣴڪّ گَاڪْ ڪُوتَايْ لَ دَݧُيْ بࣷڪّ دࣷونْ لُ سࣹلّ؞ ‏11 سُ فࣹڪّࣹي نࣹ لَ ݧُ دࣷگُ وࣷونَ جࣷخࣹ مَلَ مُ دَگَنُلْ لَ تࣹ دَگَنُلَ سَرْخَلَلْ اَجِ سَخْ جِ، نَ نِتْ ڪِ يٗبُّ جُرْ گَ بَ ݖَ سَرْخَلْڪَتْ بَ، ‏12 سَرْخَلْڪَتْ بَ خَيْمَ ڪࣷ؞ سُ بَاخࣹي سُ بࣷنࣹي، لُ مُ ݖِ دࣷگَلْ مࣷويْ وࣺيْ؞ ‏13 سُ ڪࣷ بࣷرࣷومْ بࣴگّࣹي جࣷتَاتْ، دَيْ فࣹيْ نجࣺگْ گَ، يࣷڪّ ݖَ خَاجُبْ جُرٗومࣹيلُ نجࣺگْ گَ؞
‏14 «سُ نِتْ جَگْلࣹيلࣹي اَجِ سَخْ جِ دࣴڪُّوَايَمْ، سࣹلَّلَلْ ڪࣷ ڪࣷ، نَ سَرْخَلْڪَتْ بِ خَيْمَ دࣴڪُّوَايْ بِ؞ سُ بَاخࣹي سُ بࣷنࣹي، لُ مُ ݖِ دࣷگَلْ مࣷويْ وࣺيْ؞ ‏15 سُ فࣹڪّࣹي نࣹ ڪِ دࣷگُ وࣷونَ جࣷخࣹ دࣴڪُّوَايَمْ دَ ڪࣷو بࣴگَّ جࣷتَاتْ، دَيْ فࣹيْ نجࣺگْ گَ ݧُ ڪࣷ خَيْمَا، يࣷڪّ ݖَ خَاجُبْ جُرٗومࣹيلَمْ، مُ دࣷونَاتْ اَلَلَمْ؞
‏16 «سُ نِتْ جَگْلࣹيلࣹي اَجِ سَخْ جِ سُوفْ سُ مُ مࣷومْ ݖِ سُوفَسْ ندࣷنࣷومْ، نَ ݧُيْ خَيْمَا سُوفْ سَ، نَ دࣴݒّࣷوڪْ دَيࣷبْ ݒࣹݒّ بَيْ مَتْ جِوُمْ سُوفْ سَ؞ ݧࣹتِّ بَرِگࣷيْ ݒࣹݒُّمْ لࣷرسّ يُ نࣹ جُرٗومْ فُڪِّ دࣷگِ خَالِسَ¤ ڪࣷيْ وࣹݖِّ؞ ‏17 سُ فࣹڪّࣹي نࣹ ݖِ اَتُمْ يِوِڪُ لَ جࣷخࣹ سُوفْ سِ، خَيْمَبْ سُوفْ سِ دَيْ مَتْ سࣴڪّ اࣵندّ اَڪْ مࣷومْ؞ ‏18 وَايࣹ سُ فࣹڪّࣹي نࣹ گَنَّاوْ اَتُمْ يِوِڪُ لَ جࣷخࣹ سُوفْ سِ، سَرْخَلْڪَتْ بِ دَ ڪࣷيْ نَتَّلْ نجࣺگْ گِ ݖِ اَتْ يِ دࣷخْ سࣹينْ دِگَّنْتࣹيڪْ اَتُمْ يِوِڪُ مَ ݖَ تࣷݒّ، گࣺنّࣹ ݖَ اَتْ يَ وࣹيسْ؞ ‏19 سُ فࣹڪّࣹي نࣹ ڪِ دࣷگُ وࣷونَ جࣷخࣹ تࣷولْ بِ دَ ڪࣷو بࣴگَّ جࣷتَاتْ، دَيْ فࣹيْ نجࣺگْ گَ ݧُ ڪࣷ خَيْمَا، يࣷڪّ ݖَ خَاجُبْ جُرٗومࣹيلَمْ، مُ دࣷونَاتْ اَلَلَمْ؞ ‏20 سُ جࣷتَاتُلْ تࣷولْ بِ، بَ ݧُ جَايْ تࣷولْ بِ ڪࣹنࣹينْ، دࣹيسُ ڪࣷ مَنَ جࣷتَاتْ؞ ‏21 بُ تࣷولْ بِ يِوِڪࣷو ݖِ اَتُمْ يِوِڪُ نَگْ، دَيْ دَلْدِ دࣷونْ لُ ݧُ سࣹلَّلَلْ اَجِ سَخْ جِ، مࣹلْ نِ تࣷولْ بُ ݧُ آيࣹ، مُ دࣷونْ اَلَلُ سَرْخَلْڪَتْ؞
‏22 «سُ فࣹڪّࣹي نࣹ نِتْ ڪِ دَفَ جَگْلࣹيلْ اَجِ سَخْ جِ تࣷولْ بُ مُ جࣴندّ، تࣹ دࣷنُّ ڪࣷ، ‏23 سَرْخَلْڪَتْ بِ دَ ڪࣷيْ خَيْمَالْ نجࣺگْ گِ، مࣺنگَلࣹ ڪࣷوڪْ اَتْ يِ دࣷخْ سࣹينْ دِگَّنْتࣹيڪْ اَتُمْ يِوِڪُ مَيْ ݧࣴوْ؞ نَ نِتْ ڪِ فࣹيْ لَ ݧُ خَيْمَ ݖَ بࣺسْ بَ، خَالِسْ بَ دِ لُ ݧُيْ سࣹلَّلَلْ اَجِ سَخْ جِ؞ ‏24 بُ اَتُمْ يِوِڪُ دِڪّࣹي، تࣷولْ بِ دَيْ دࣹلُّ ݖَ ڪَ ݧُ ڪࣷ جࣴنْدࣹ وࣷونْ تࣹ مُ دِ اَلَلُ ندࣷنࣷومْ؞ ‏25 مبࣷولࣹمْ لُ نگࣹينْ خَيْمَ، نَ دࣴݒّࣷوڪْ نَتّ بَ ݧُ يࣷونَلْ ݖِ بࣺرَبْ بُ سࣹلّ بِ، مَانَامْ لِ ݧُيْ وَخْ سِيڪَلْ تࣹ مُ تࣷلُّ ݖِ فُڪِّ گَرَامْ؞
‏26 «وَايࣹ ڪَتْ لُيْ تَاوْ ݖِگْ جُرْ، اَجِ سَخْ جࣹي ڪࣷ مࣷومْ جࣹيگْ؞ دُ لُ ڪࣹنّ دِ دࣷگࣷو جَگْلࣹيلْ اَجِ سَخْ جِ؞ مُيْ بَرْمࣷلْ، مُيْ مبࣷتࣹ، مُيْ تࣹفْ، اَجِ سَخْ جࣹييْ بࣷرࣷومْ؞ ‏27 سُ دࣹي لُ مَلَ مُ دَگَنُلْ تَاوْلࣷو، ڪَ ڪࣷ مࣷومࣷونْ مَنْ نَ ڪࣷو جࣷتَاتْ ندࣹگَمْ فࣹيْ نَ نجࣺگَمْ، بَ دࣷلِّ ݖَ خَاجُبْ جُرٗومࣹيلُ نجࣺگْ گَ؞ بُ ݧُ ڪࣷ جࣷتَاتُلْ، دࣹيسْ ڪࣷيْ جَايْ ݖِ نجࣺگْ گِ ݧُ ڪࣷ خَيْمَا؞ ‏28 مبࣷولࣹمْ لُ ݧُ آيَلَلْ اَجِ سَخْ جِ، لُ سࣹلَّ سࣹلّ لَ ݧࣹيلْ اَجِ سَخْ جِ؞ لࣺݒّ لُ نِتْ مࣷومْ، سُ دࣹي نِتْ مبَا جُرْ مبَا سُوفْ سُ مُ دࣷنّ، اَمُ ݖِ لࣹنّ لُ مُ آيَلَلْ اَجِ سَخْ جِ تࣹ ݧُ سَݧْ ڪࣷو جَايْ مبَا مُ دِ ڪࣷ جࣷتَاتْ؞ ‏29 بَ تࣹيْ دُ ڪࣹنّ نِتْ ڪُ ݧُ آيَلَلْ بَڪَّنَمْ اَجِ سَخْ جِ، مُ وَرَ سࣵنڪُ، بَ نࣷݒِّ ݧُ دِ ڪࣷ جࣷتَاتْ؞ دࣹيسْ ڪࣷيْ رࣹيْ رࣹڪّ؞
‏30 «بُ لࣷولُ وࣹيسࣹي مبࣷولࣹمْ لُ جٗگࣹ ݖِ سُوفْ، مُيْ ݒࣹݒّ مبَا دࣷومِ گَرَبْ، ݖࣺرُبْ فُڪّࣹيلْ بَ اَجِ سَخْ جࣹييْ بࣷرࣷومْ؞ دࣹيسْ ڪࣷيْ سࣹلَّلَلْ اَجِ سَخْ جِ؞ ‏31 سُ فࣹڪّࣹي نࣹ نِتْ ڪِ دَفَ نَمَّ جࣷتَاتْ لࣹنّ ݖَ ݖࣺرُ فُڪّࣹيلُ مࣹݧّࣹيفَمْ مَ مُيْ گࣺنّࣹ، دَيْ يࣷڪّ ݖَ نجࣺگْ گَ خَاجُبْ جُرٗومࣹيلَمْ؞ ‏32 مبࣷولࣹمْ ݖࣺرُ فُڪّࣹيلْ بُ ݧُ گࣺنّࣹي ݖِگْ جُرْ، گُ گُدّ گَاڪْ گُ گࣵتّ گَ، مُيْ لࣺݒّ لُ سࣵمّ وَݧّࣹ بَنْتَمْ، ݖࣺرُبْ فُڪّࣹيلْ لَ بُ ݧُيْ سࣹلَّلَلْ اَجِ سَخْ جِ؞ ‏33 دࣹيسُ ݖِ سࣹيتْ جُرْ گُ بَاخْ اَڪْ گُ بࣷنْ تࣹ دࣹيسُ ݖِ وُوتَلْ لࣹنّ؞ بُ ݖِ نِتْ خَسࣹي وُوتَلْ لࣹنّ، لَ جࣴڪّ اَڪْ لَ ڪࣷ وُوتُ لࣺݒّ دَيْ دَلْدِ دࣷونْ لُ سࣹلّ؞ دࣹيسُ ڪࣷ مَنَ جࣷتَاتْ؞»
‏34 يࣷويُ سَنْتَانࣹ لَ اَجِ سَخْ جِ دࣺنڪࣷونْ مُوسَىٰ ݖَ تُنْدُ سِنَيِ، نگِرْ مُ جࣷتّلِ ڪࣷ بࣵنِّ إِسْرَاءِيلْ؞
