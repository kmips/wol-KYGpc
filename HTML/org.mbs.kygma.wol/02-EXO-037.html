دࣹفَرْ نَݧُ گَالُ ڪٗلّࣴرࣹ گِ
37
‏1 بࣹسَلࣹلْ نَگْ دࣹفَرࣹ گَالُ ڪٗلّࣴرࣹ گِ دࣺنڪُ سࣺنگّ، گُدَّايْ بِ دِ ݧَارِ خَسَبْ اَڪْ گࣹنّ-وࣵلّ، يَاتُوَايْ بِ دِ خَسَبْ اَڪْ گࣹنّ-وࣵلّ، تَخَوَايْ بِ دِ خَسَبْ اَڪْ گࣹنّ-وࣵلّ؞ ‏2 مُ خࣷوبْ ڪࣷ وُرُسُ نگَلَمْ بِيرْ اَڪْ بِتِ، سࣵڪَّلْ ڪࣷ ڪࣺمࣺجُ وُرُسْ گُ ڪࣷ وࣴرْ؞ ‏3 مُ دَلْدِ ڪࣷيْ مٗولَلْ ݧࣹينْتِ لَمِ وُرُسْ ݖِ ݧࣹينْتِ تࣵنڪَمْ، ݧَارِ لَمْ ݖِ گࣹنّ وࣹتْ گِ، ݧَارْ، وࣹتْ گِ ݖِ دࣹسْ؞ ‏4 مُ سࣵڪّ نَگْ اَيْ نجࣵݒُّيْ بَنْتُ سࣺنگّ، خࣷوبْ بُ ݖِ نࣹڪّ وُرُسْ، ‏5 دَلْدِ رࣷوفْ نجࣵݒُّ يِ ݖِ لَمِ وࣹتِ گَالْ گِ، نگِرْ ݧُ دِ ڪࣷ يٗبّࣷو گَالْ گِ؞
‏6 ݖِ ڪَوْ لࣷولُ مُ سࣵڪّ ڪُبࣹيرُ سَرَخُ نجࣷتْلاَيْ گُ وُرُسُ نگَلَمْ، گُدَّايْ بِ دِ ݧَارِ خَسَبْ اَڪْ گࣹنّ-وࣵلّ، يَاتُوَايْ بِ دِ خَسَبْ اَڪْ گࣹنّ-وࣵلّ؞ ‏7 مُ سࣵڪّ اِتَمْ ݧَارِ مَلاَڪَيْ سࣹرُبْ يُ وُرُسْ؞ لِگّࣺييُ تࣴگّ بُ جࣹڪّ لَ دࣹفْ سࣹرُبْ يِ ݖِ ݧَارِ ݖَتِ ڪُبࣹيرْ گِ، ‏8 ݖَتْ لِي اَبْ سࣹرُبْ، ݖَتْ لࣹي اَبْ سࣹرُبْ، مُ مٗولاَلࣹ سࣹرُبْ يِ اَڪْ ݧَارِ ݖَتْ يِ، سࣹرُبْ يِ اࣵندّ اَڪْ ڪُبࣹيرْ گِ، دِ بࣹنّ؞ ‏9 سࣹرُبْ يَا نگِ تࣵلَّلْ سࣹينِ لاَفْ، جࣴمࣹ ڪَوْ، يِيرࣹ ڪُبࣹيرْ گِ سࣹينِ لاَفْ تࣹ جَنࣷو، سࣹينِ ڪَنَمْ جُبْلࣷوڪْ ڪُبࣹيرْ گِ؞
دࣹفَرْ نَݧُ تَابَلُ مبُرُمْ تࣹيوَلْ مِ
‏10 ݖِ بِيرْ لࣷولُ بࣹسَلࣹلْ سࣵڪّ تَابَلُ بَنْتُ سࣺنگّ، گُدَّايْ بِ دِ ݧَارِ خَسَبْ، يَاتُوَايْ بِ دِ خَسَبْ، تَخَوَايْ بِ دِ خَسَبْ اَڪْ گࣹنّ-وࣵلّ؞ ‏11 مُ خࣷوبْ ڪࣷ وُرُسُ نگَلَمْ، دࣹفَلْ ڪࣷ اࣷمْبُ وُرُسْ بُ ڪࣷ وࣴرْ؞ ‏12 مُ دࣹفَلْ ڪࣷ اِتْ ݒࣹگّ بُ ڪࣷ وࣴرْ بُ تࣷلُّ ݖِ يَاتُوَايُ ݧࣹينْتِ بَارَامْ، بَ نࣷݒِّ سࣵڪَّلْ ݒࣹگّ بِ اࣷمْبُ وُرُسْ، وࣴرَلࣹي ڪࣷ ڪࣷ؞ ‏13 مُ مٗولَلْ تَابَلْ جِ ݧࣹينْتِ لَمِ وُرُسْ، دَلْدِ وࣹ لَمْ يِ ݖِ ݧࣹينْتِ وࣹتْ يِ، فِ فَرࣹيڪْ ݧࣹينْتِ تࣵنڪّ يِ، ‏14 لَمْ يِ دࣹنْدࣹيڪْ ݒࣹگّ بِ، دِ تࣺيࣹ نجࣵݒُّ، يِ ݧُيْ يٗبّࣷو تَابَلْ جِ؞ ‏15 مُ سࣵڪّ اَيْ نجࣵݒُّيْ بَنْتُ سࣺنگّ، خࣷوبْ ڪࣷ وُرُسْ، ݧُ دِ ڪࣷ يٗبّࣷو تَابَلْ جِ؞ ‏16 ݖِ ڪَوْ لࣷولُ مُ سࣵڪّ جُمْتُڪَايْ يِيْ تࣹگࣹ ݖِ تَابَلْ جِ؛ ندَبَمْ يُ تࣹلّ يࣹيڪْ مبࣵتُّ يࣹيڪْ نجَقْ يࣹيڪْ ڪࣴلّ، يِ ݧُيْ تُورَلْ يࣵلَّ، لࣺݒّ دِ وُرُسُ نگَلَمْ؞
دࣹفَرْ نَݧُ تࣹگُڪَايُ لࣵمݒّ بِ
‏17 بࣹسَلࣹلْ سࣵڪّ اِتَمْ تࣹگُڪَايُ لࣵمݒّ بُ وُرُسُ نگَلَمْ؞ لِگّࣺييُ تࣴگّ بُ جࣹڪّ لَ دࣹفْ تࣹگُڪَايُ لࣵمݒّ بِ، اَڪْ تَاتُ تࣹگُڪَايْ بِ، اَڪْ ݒࣹرُ تࣹگُڪَايْ بِ، اَيْ ڪَاسَمْ اَڪِ ڪࣵمْبٗوتَمْ اَڪِ مبَرَمْ، لࣺݒّ اࣵندّ اَڪْ تࣹگُڪَايْ بِ، دِ بࣹنّ؞ ‏18 جُرٗومْ بࣹنِّ ݖَرَ سࣷقِڪࣷو ݖِ وࣹتِ تࣹگُڪَايُ لࣵمݒّ بِ، ݧࣹتِّ ݖَرْ ݖِ گِي وࣹتْ، ݧࣹتّ ݖِ گِ ݖِ دࣹسْ؞ ‏19 ݧࣹتِّ ڪَاسْ يُ بِنْدࣷو نِ تٗورْ-تٗورِ گَرَبُ اَمࣵندّ ݧࣷو نࣹڪّ ݖِ بࣹنّ ݖَرْ بِ، ڪَاسْ بُ ݖِ نࣹڪّ اࣵندّ اَڪْ اَيْ ڪࣵمْبٗوتَمْ اَڪِ مبَرَمْ، ݧࣹتِّ ڪَاسْ يُ بِنْدࣷو نِ تٗورْ-تٗورِ گَرَبُ اَمࣵندّ نࣹڪّ ݖِ ݖَرْ بِ ݖِ تࣷݒّ، ڪَاسْ بُ ݖِ نࣹڪّ اࣵندّ اَڪْ اَيْ ڪࣵمْبٗوتَمْ اَڪِ مبَرَمْ، بَ جُرٗومْ بࣹنِّ ݖَرْ، يِ سࣷقِڪࣷو ݖِ ݒࣹرْ بِ، دَجْ؞ ‏20 ݒࣹرُ تࣹگُڪَايُ لࣵمݒّ بِ اِتَمْ اَمْ نَ ݧࣹينْتِ ڪَاسْ يُ بِنْدࣷو نِ تٗورْ-تٗورِ گَرَبُ اَمࣵندّ، اࣵندّ اَڪْ اَيْ ڪࣵمْبٗوتَمْ اَڪِ مبَرَمْ؞ ‏21 اَبْ ڪࣵمْبٗوتَ رࣷنُ ݧَارِ ݖَرْ، جࣵݒّ ݖِ، بࣹنࣹينْ ڪࣵمْبٗوتْ رࣷنُ ݧَارِ ݖَرْ يِ ݖِ تࣷݒّ، جࣵݒّ ݖِ، اَبْ ڪࣵمْبٗوتْ رࣷنُ ݧَارِ ݖَرْ يِ مُجّ، جࣵݒّ ݖِ، جُرٗومْ بࣹنِّ ݖَرْ، يِ سࣷقِڪࣷو ݖِ وࣹتِ ݒࣹرْ بِ، دَجْ؞ ‏22 ڪࣵمْبٗوتْ يِ اَڪْ ݖَرْ يِ نَگْ اࣵندّ اَڪْ تࣹگُڪَايْ بِ، دِ وࣹنّ دࣷنجُ وُرُسُ نگَلَمْ وُ ݧُ تࣴگّ بَ مُ جࣹڪّ؞ ‏23 مُ سࣵڪّ جُرٗومْ ݧَارِ لࣵمْݒَمْ اَڪْ اَيْ ݧِيمِ فࣹيُڪَايَمْ اَڪْ اَيْ اَنْدَمْ، لࣺݒّ دِ وُرُسُ نگَلَمْ؞ ‏24 بࣹنّ تَلَݝُ وُرُسُ نگَلَمْ، يࣹمࣷوڪْ فَنْوࣹيرِ ڪِلࣷ، ݖِ لَ سࣵڪّ تࣹگُڪَايُ لࣵمݒّ بِ اَڪْ مبࣷولࣹمْ اَيْ جُمْتُڪَايَمْ؞
دࣹفَرْ نَݧُ سَرْخَلُڪَايُ ݖُورَايْ بِ
‏25 ݖِ ڪَوْ لࣷولُ بࣹسَلࣹلْ سࣵڪّ ݖِ بَنْتُ سࣺنگّ سَرْخَلُڪَايُ ݖُورَايْ بُ ڪَارࣹ، گُدَّايْ بِ دِ خَسَبْ، يَاتُوَايْ بِ دِ خَسَبْ، تَخَوَايْ بِ دِ ݧَارِ خَسَبْ، اَيْ بࣺجّࣺنَمْ اࣵندّ اَڪْ مࣷومْ دِ بࣹنّ بَنتّ بِ؞ ‏26 مُ خࣷوبْ ڪࣷ وُرُسُ نگَلَمْ ݖِ ڪَوْ اَڪْ ݖِ وࣹتْ يِ، بَ مُ دَجْ، اَڪْ بࣺجّࣺنْ يِ، دَلْدِ ڪࣷيْ سࣵڪَّلْ وࣷمْبُ وُرُسْ بُ ڪࣷ وࣴرْ؞ ‏27 ݧَارِ نࣷݒِّ وُرُسْ لَ ڪࣷ سࣵڪَّلْ ݖِ رࣷنْ ڪࣺمࣺجْ گِ، ݖِ ݧَارِ وࣹتَمْ، ݧُ جࣵڪَّارْلࣷو، دِ تࣺيࣹ بَنْتِ نجࣵݒُّ، يِ ݧُ ڪࣷيْ يٗبّࣷو؞ ‏28 ݖِ بِيرْ لࣷولُ مُ سࣵڪّ اَيْ نجࣵݒُّيْ بَنْتِ سࣺنگّ، خࣷوبْ لُ ݖِ نࣹڪّ وُرُسْ؞ ‏29 مُ سࣵڪّ نَگْ دِوُ ݒَلْ گُ سࣹلّ گِ، سࣵڪّ ݖُورَايْ لُ خࣹيݧْ لِ رَخُلْ، دࣹفْ ڪࣷ نجَفَانْ لُ نجَفَانْ-ڪَتْ وَاجَلْ بَ مُ بَاخْ؞
