4
‏1 ڪࣷنْ نَگْ دَمَ لَيْ دࣺنڪّ فِ ڪَنَمْ يࣵلَّ اَڪْ اَلْمَسِ عِيسَىٰ، مِيْ اࣵتّࣹجِ ݧِيْ دُندّ اَڪْ ݧِ دࣹي، دِ لَ دࣺنڪّ ݖِ دَرَجَيْ بࣺسَمْ بَ مُيْ فࣹيݧْ، اَڪْ ݖِ دَرَجَيْ نگُورَمْ، ‏2 وَارࣹيلْ ڪࣵدُّگْ يࣵلَّ تࣹ تَخَوْ ݖِ تࣹمّ، سُ نࣹيخࣹي اَڪْ سُ نَقَرࣹي؛ دِ يࣹدّࣹ، دِ فࣹمّࣷو، دِ ݧَاخࣹ، تࣹيْ سَخࣷو مُݧْ ݖِ سَمْ نجࣵنگّلࣹ؞ ‏3 ندَخْ ڪَتْ اَيْ جَمࣷنࣷو نگِ ݧࣴوْ، نِتْ ݧِ دࣷوتُݧُ خَجࣷو اࣵلُّوَ جُ وࣺرْ، وَايࣹ سࣹينِ بࣴگّ-بࣴگِّ بࣷݒّ لَݧُيْ تࣷݒّ، بَيْ فࣷرْتَاتࣷو فُ نࣹڪّ، اَيْ سࣴرِݧْ يُ لࣹينْ دِ وَخْ لُ لࣹينْ نࣹيخْ؞ ‏4 دࣴگّ لَݧُيْ تَنقَمْلُ، لَجّ، وُتِ لࣺيبِ نࣹينْ؞ ‏5 وَايࣹ يَوْ، اࣵنْدَلْ اَڪْ سَ سَگࣷ، اَڪْ دِگَّنْتࣹ بࣷو مَنَ تࣷلُّ، تࣹ ݧࣹمࣹ ݖࣷونࣷ؞ جࣵݒَّلْ ݖِ لِگّࣺييُبْ جࣷتّلِ خِبَارُ جࣵمّ بِ، بࣹيَلْ سَ وَارْ بَ مُ مَتْ سࣴڪّ؞
‏6 مَنْ نَگْ، دَانَڪَ سَرَخُ تُورُ بُ سࣷتِّڪُ لاَ بَ نࣷݒِّ؛ سَمَ وَخْتُوْ دࣹمْ دࣴگْمَلْ نَ؞ ‏7 خَرࣹ بُ بَاخْ بِ، خَرࣹ نَا ڪࣷ، رَوَنْتࣹ بِ، سࣷتَّلْ نَا ڪࣷ، نگࣴمْ گِ مَ اَمْ، سࣵمّ نَا ڪࣷ؞ ‏8 لࣺيگِ نَگْ ڪَالَگْ نجࣹڪَّ مَيْ خَارْ، تࣹ بࣷرࣷومْ بِ، اࣵتّࣹڪَتْ بُ جُبْ بِ مࣷو مَ ڪࣷيْ يࣷولْ، ڪࣹرࣷوگْ ݖَ بࣺسْ بَ، تࣹ دُ مَنْ رࣹڪّ لَ ڪࣷيْ يࣷولْ، وَايࣹ اِتَمْ ڪࣺݒّ ڪُ ڪࣷيْ سࣺينْتُ تࣹ يࣵڪَّمْتِ بࣺسْ بَ مُيْ فࣹيݧْ؞
فِي لَ ݒٗولْ تࣴجࣹ اَيْ دࣺنڪَانࣹيمْ
‏9 دࣹفَلْ سَ ڪࣹمْ-ڪࣵتَّنْ، بَ فࣹڪّسِ مَ نُ مُ گࣴنَ گَاوࣹ؞ ‏10 ندَخْ ڪَتْ دࣹمَسْ، اࣵدِّنَيْ تࣹيْ لَ سࣷݒّ بَ تَخْ مُ وَݖّ مَ، دࣹمْ تࣹسَلࣷنِگْ، تࣹ ڪࣹرࣹسࣹݧْسࣴ دࣹمْ نَ دِيوَانُ گَلَسِ، تِتْ دࣹمْ دِيوَانُ دَلْمَسِ؞ ‏11 لُوگْ رࣹڪَّ دࣹسْ فِ مَنْ؞ اِنْدَالࣹيلْ مࣵرڪّ، ندَخْ اَمَلْ نَ مَ نجَرِݧْ ݖِ لِگّࣺييْ بِ؞ ‏12 تِسِگْ نَگْ لاَ يࣹبَلْ اࣹفࣹسْ؞ ‏13 مبُبّ مَ مَ وَݖّࣷونْ اَڪْ ڪَرْݒُسْ ݖَ تࣷرࣷوَسْ، بࣷويْ ݧࣴوْ، اِنْدَالࣹيلْ مَ ڪࣷ، اَڪْ اِتَمْ تࣺيرࣹ يِ، رَوَتِنَ تࣺيرࣹيْ دࣹرْ يِ؞
‏14 اَلࣹگْسࣵنْدَرْ تࣴگّ بِ دࣹفْ نَ مَ لُ بࣷنْ لُ رࣺيْ، وَايࣹ بࣷرࣷومْ بࣹي ڪࣷيْ فࣹيْ جࣴفَمْ؞ ‏15 يَوْ اِتَمْ، مࣷيْتُ ڪࣷ بُ بَاخْ، ندَخْ مࣷو گࣵنْتَلْ لࣷولْ سُنُ اࣵلُّوَ؞
‏16 ڪࣹرࣷوگْ بَ مَ جࣴڪّࣹي دِ لَيࣷولْ سَمَ بࣷݒّ، ڪࣹنّ تَخَوُوْمَ، ݧࣺݒَّ مَ دَوْ، يࣵلَّ بُ لࣹينْ سࣹينْ جࣴفْ تࣷݒّ؞ ‏17 وَايࣹ بࣷرࣷومْ بࣹي مَ دِمْبَلِ، دࣷولࣹيلْ مَ، نگِرْ ڪࣵدُّ گِ سِيوࣹ ݖِ مَنْ، بَ خࣹيتِ جَامْبُرْ يࣺݒّ دࣺگّ ڪࣷ، تࣹ بࣷرࣷومْ بࣹي مَ مُسَلْ ݖِ ݒࣵدُّمْ گَيْندࣹ؞ ‏18 بࣷرࣷومْ بࣹي مَيْ وَلُّ ݖِ جࣺݒّ جࣴفْ جُ بࣷنْ، تࣹ مࣷو مَيْ آرْ بَ تࣵبَّلْ مَ فَ نگُورَمْ گَ فَ اَسَمَانْ؞ دَرَجَ ݧࣹيلْ نَ ڪࣷ، تࣹيْ اَڪْ اࣴلّࣴگْ بَ فࣵوّ؞ اَمِينْ؞
ݒٗولْ تࣵگُّ نَ ݧࣷومْ تِمࣷتࣹ
‏19 نُيُلْ مَ ݒِرِسْڪَ¤ اَڪْ اَڪِلَسْ اَڪْ وَا ڪࣴرْ اࣷنࣹسِفࣷرْ؞ ‏20 اࣹرࣵستّ مࣷومْ، ڪࣷرࣹنتّ لَ دࣹسْ، وَايࣹ تࣷرࣷفِمْ، مِلࣹ لاَ ڪࣷ بࣵيِّ، ندَخْ دَا وࣷݒّࣷونْ؞ ‏21 گَاوَلَ ݧࣴوْ، بَلاَ سࣹدُّبْ لࣷلِّ دِ جَمْ؞ اَبْ نُيࣷو ݧࣹيلْ نَ لَ، تُڪّࣹي ݖِ اࣴبُلُسْ اَڪْ ݒُدࣹݧْسࣴ اَڪْ لِنُسْ اَڪْ ڪࣴلٗجَا اَڪْ بࣷڪّ يࣺݒّ؞ ‏22 يَلْ نَ بࣷرࣷومْ بِ وࣺتَّلِ سَمْ خࣹلْ؞ اَوْ يِوْ ݧࣹيلْ نَ لࣹينْ؞
