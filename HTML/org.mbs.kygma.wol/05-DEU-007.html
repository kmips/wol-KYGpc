دࣷگࣷوڪْ يࣹنّ خࣹيتْ يِ وَرْ نَ إِسْرَاءِيلْ
7
‏1 بُ لࣹينْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ يٗبّࣷو رࣺيوْ مَ نگࣹينْ جࣴمْ، نگِرْ نَنگُجِ ڪࣷ، مࣷو فَيْ دࣵقࣹ خࣹيتْ يُ بَرࣹ فِ سࣹينْ ڪَنَمْ:‏ اࣹتّࣹينْ ݧِ اَڪْ گِرْگَسࣹينْ ݧِ اَڪْ اَمࣷرࣹينْ ݧِ اَڪْ ڪَنَانࣹينْ ݧِ اَڪْ ݒࣹرِسࣹينْ ݧِ اَڪْ اࣹوࣹينْ ݧࣹيڪْ يࣹبُسࣹينْ ݧِ؛ جُرٗومْ ݧَارِ خࣹيتْ يُ لࣹينْ اࣴݒّ، اࣴݒّ لࣹينْ دࣷولࣹ؞ ‏2 بُ لࣹينْ لࣹينْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ جࣺبَّلࣹي، بَ نگࣹينْ دُمَ لࣹينْ، نَنگࣹينْ لࣹينْ فَاگَاگَلْ بَ ݧُ جࣹيخْ تࣵڪّ؞ بُلࣹينْ فَسࣷوڪْ ݧࣷومْ ڪٗلّࣴرࣹ، تࣹ بُلࣹينْ لࣹينْ يࣴرࣴمْ؞ ‏3 بُلࣹينْ سࣺيَنْتࣹيڪْ ݧࣷومْ:‏ سࣹينْ دࣷومْ جُ جِگࣺينْ، بُلࣹينْ ڪࣷ مَيْ دࣷومُ خࣹيتْ يࣷويُ، جِگࣺينُ خࣹيتْ يࣷويُ اِتْ، بُلࣹينْ ڪࣷ جࣴلَلْ سࣹينْ دࣷومْ جُ گٗورْ؞ ‏4 ندَخْ دَݧُ مَيْ دࣴدَّلࣹ اَڪْ سࣹينْ دࣷومْ يُ گٗورْ، ݧُ مُجّ دِ جَامُ يࣹنࣹينْ يࣵلَّ، بَ سࣵنجُمْ اَجِ سَخْ جِ تࣵڪّ ݖِ سࣹينْ ڪَوْ يࣹينْ، فࣵڪَّسْ لࣹينْ ݖِ لُ گَاوْ؞ ‏5 نِ نگࣹينْ دِ دࣹفْ اَڪْ ݧࣷومْ ڪَيْ مࣷويْ نِي:‏ سࣹينِ سَرْخَلُڪَايْ، نگࣹينْ مࣵبّ؛ سࣹينْ تُورِ دࣷجْ، نگࣹينْ رَجَخࣹ؛ سࣹينْ خࣹرْ يِ ݧُيْ جَامࣷو اَسࣹرَ تُورْ مِ، نگࣹينْ گࣷرْ؛ سࣹينْ جࣴمِّ تُورْ، نگࣹينْ لَڪّ؞ ‏6 ندَخْ ڪَتْ اَسْڪَنْ وُ سࣹلّ نگࣹينْ، ݧࣹيلْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ؞ يࣹينْ لَ سࣹينْ يࣵلَّ اَجِ سَخْ جِ تَامُ، نگِرْ نگࣹينْ دࣷونْ اَسْڪَنْ وِ مُ سࣺدّࣷولْ بࣷݒَّمْ، ݖِ مبࣷولࣹمْ اَسْڪَنِ ڪَوْ سُوفْ؞
‏7 دُ سࣹينْ بَرࣹوَايْ بُ نگࣹينْ اࣴݒّلࣹي خࣹيتْ يࣺݒّ مࣷو تَخْ اَجِ سَخْ جِ نَوْ لࣹينْ، بَ تَامُ لࣹينْ، ندࣹ يࣹينَ گࣴنَ نࣺيوْ ݖِ خࣹيتْ يࣺݒّ؞ ‏8 وَايࣹ ݖࣷفࣹيلْ گُ لࣹينْ اَجِ سَخْ جِ سࣷݒّ، اَڪْ سࣵمّ گُ مُ سࣵمّ نگِݧَمْ لَ مُ گِݧَلࣷونْ سࣹينِ مَامْ، مࣷو تَخْ مُ گࣺنّࣹي لࣹينْ دࣷولࣹيْ لࣷخࣷومْ، جࣷتࣹ لࣹينْ ݖَ ڪࣴرُگْ نجَامْ گَ، اَڪْ ݖَ لࣷخࣷلْ فِرْعَوْنَ بُورُ مِسْرَ؞ ‏9 خَمْلࣹينْ بࣷوگْ نࣹ سࣹينْ يࣵلَّ اَجِ سَخْ جِ مࣷومْ مࣷويْ يࣵلَّ؛ مࣷويْ يࣵلَّ جُ دࣴگُّ جِيْ سࣵمّ ڪٗلّࣴرࣹ اَڪْ نگࣷرْ ݧࣹيلْ سࣷݒّࣹيمْ ݧِيْ سࣵمّ اَيْ سَنْتَانࣹيمْ، سࣴتْ بَ سࣴتَاتْ، بَ ݖِ جُنِّيْ مَاسْ؞ ‏10 وَايࣹ مࣷويْ فࣹيْ ݖِ سَا سِ اَيْ بَݧَمْ، نگِرْ سࣵنڪّ لࣹينْ؞ دُ يࣹيخࣹ بَݧَمْ، ݖِ سَا سِ لَ ڪࣷيْ فࣹيْ؞
‏11 ڪࣷنْ نَگْ سࣵمّلࣹينْ سَنْتَانࣹ بِ، مُيْ دࣷگَلِ يࣷونْ يِ، اَڪْ اࣵتّࣹيْ يࣷونْ يِ مَ لࣹينْ دࣺنڪّ تࣹيْ، نگِرْ دِ ڪࣷ جࣴفࣹ؞ ‏12 بُ نگࣹينْ دࣺگَّلࣹي اࣵتّࣹيْ يࣷونْ يِي، تࣹ سࣵمّࣷو ڪࣷو جࣴفࣹ، سُ بࣷوبَا سࣹينْ يࣵلَّ اَجِ سَخْ جِ دِنَ سࣵمّ دِگَّمْ اَڪْ يࣹينْ، ڪٗلّࣴرࣹ اَڪْ نگࣷرْ، گَ مُ گِݧَلࣷونْ سࣹينِ مَامْ؞ ‏13 مࣷو لࣹينْ دِ سࣷݒّ، بَرْڪࣹيلْ لࣹينْ، يࣷڪّ لࣹينْ، بَرْڪࣹيلْ سࣹينْ مࣹݧّࣹيفُمْ نجُرُڪَايْ، اَڪْ سࣹينْ مࣹݧّࣹيفُمْ سُوفْ؛ سࣹينْ ݒࣹݒّ اَڪْ سࣹينْ بِيݧْ، اَڪْ سࣹينُگْ دِوْ، اَڪْ سࣹينْ دࣷومِ جُرْ گُ گُدّ، اَڪْ گُ گࣵتّ، ݖَ رࣺيوْ مَ مُ گِݧَلࣷونْ سࣹينِ مَامْ، نࣹ يࣹينْ لَ ڪࣷيْ جࣷخْ؞ ‏14 سُ بࣷوبَا يࣹينَيْ گࣴنَ بَرْڪࣹيلْ مبࣷولࣹمْ خࣹيتْ يِ، تࣹ دُ لࣹنّ، گٗورْ اَڪْ جِگࣺينْ، لُيْ جَاسِرْ ݖِ سࣹينْ بِيرْ، اَڪْ سࣹينْ بِيرُگْ جُرْ؞ ‏15 سُ بࣷوبَا اَجِ سَخْ جࣹي لࣹينْ دِ خࣵجِّلْ مبࣷولࣹمْ وࣺرَدِ؞ مبࣷولࣹمْ جࣵنگࣷرࣷيْ مِسْرَ يُ بࣷنْ يُ نگࣹينْ خَمࣷونْ، دُ لࣹينْ ݖِ تࣹگْ لࣹنّ، تࣹ مبࣷولࣹمْ سࣹينِ بَݧْ لَ ڪࣷيْ تࣹگْ؞ ‏16 فࣵڪَّسْلࣹينْ مبࣷولࣹمْ خࣹيتْ يِ لࣹينْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ جࣺبَّلْ نَگْ؞ بُلࣹينْ لࣹينْ خࣷولْ بࣴتُ يࣴرْمَنْدࣹ، بُلࣹينْ جَامُ سࣹينِ تُورْ:‏ لُ ڪࣷ مࣷيْ مُيْ سࣹينُگْ فِيرْ؞
‏17 جࣷمْبُلْ نگࣹينْ دِ وَخْ ݖِ سࣹينْ خࣹلْ نَانْ:‏ «خࣹيتْ يِي گࣴنْ نࣷو بَرࣹ، اَنَ نُ نُ لࣹينْ مَنَ دࣵقࣹ؟» ‏18 بُلࣹينْ لࣹينْ رَگَلْ؞ فࣵتّلِڪُلࣹينْ بُ بَاخْ نِ سࣹينْ يࣵلَّ اَجِ سَخْ جِ دࣹفْ فِرْعَوْنَ اَڪْ مِسْرَ گࣺݒّ، ‏19 اَڪْ مِتِتْ يُ رࣺيْ يَ نگࣹينْ گِسَلْ سࣹينْ بࣷݒّ، اَڪْ فِرْندࣹ يَاڪْ ڪࣺيمَانْ يَ، اَڪْ سࣹينْ دࣷولࣹيْ لࣷخࣷلْ يࣵلَّ اَجِ سَخْ جࣹيڪْ ڪࣵتَّنَمْ گُ رࣺيْ، گَ مُ لࣹينْ گࣺنّࣹي؞ نࣷونُ لَ سࣹينْ يࣵلَّ اَجِ سَخْ جِ دِ دࣹفْ مبࣷولࣹمْ خࣹيتْ يِ نگࣹينْ رَگَلْ؞ ‏20 تࣹ اِتَمْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ، اَمْ مبَسْ لَيْ يࣹبَلْ ݖِ سࣹينْ ڪَوْ، بَ سࣹينُ ندࣹسْ وُ لࣹينْ لࣵقُ وࣷونْ سࣵنڪُ؞ ‏21 بُلࣹينْ اَمْ تِيتَانگࣹ ݖِ ݧࣷومْ، ندَخْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ ݖِ سࣹينْ بِيرْ، يࣵلَّ جُ مࣵگّ لَ تࣹ رَگْلُ؞ ‏22 سࣹينْ يࣵلَّ اَجِ سَخْ جࣹي لࣹينْ دِ دࣵقࣹ فِ سࣹينْ ڪَنَمْ، ندࣵنڪّ-ندࣵنڪّ؛ دُنگࣹينْ لࣹينْ مَنَ گَاوَ جࣹيخَلْ، لُ ڪࣷ مࣷيْ رَبِ اࣵلّ يِ مُجّ نࣹ گࣵݧّ ݖِ سࣹينْ ڪَوْ؞ ‏23 سࣹينْ يࣵلَّ اَجِ سَخْ جِ مࣷو لࣹينْ دِ جࣺبَّلْ خࣹيتْ يࣷويُ، مࣷو لࣹينْ دِ سَلْفَاݧࣹ سَلْفَاݧࣷو گُ رࣺيْ، بَ ݧُ سࣵنڪُ؞ ‏24 مࣷويْ تࣹگْ سࣹينِ بُورْ ݖِ سࣹينِ لࣷخࣷ، نگࣹينْ فَرْ سࣹينُ تُرْ فِ اَسَمَانْ تِيمْ؞ ڪࣹنّ ݖِ ݧࣷومْ دُ لࣹينْ تࣴ، بَ ڪࣹرَ نگࣹينْ لࣹينْ سࣵنڪࣹي؞ ‏25 سࣹينْ جࣴمِّ تُورْ يࣷويُ نَگْ، لَڪّلࣹينْ لࣹينْ؞ بُلࣹينْ خࣹمّࣹمْ لࣹنّ لُيْ خَالِسْ مبَا وُرُسْ ݖِ ڪَوْ تُورْ يِ؞ بُلࣹينْ ڪࣷ آڪِمࣷو، لُ ڪࣷ مࣷيْ مُيْ سࣹينُگْ فِيرْ، ندَخْ لُ سࣹيخْلُوࣷو نگࣷوگُ ݖِ سࣹينْ يࣵلَّ اَجِ سَخْ جِ؞ ‏26 بُلࣹينْ دُگَلْ يࣴفْ يُ سࣹيخْلُوُ سࣹينْ بِيرْ نࣺيگْ، لُ ڪࣷ مࣷيْ اࣵتّࣹبْ فَاگَاگَلْ بِ وَرْ ݖِ يࣴفْ يࣷويُ، مࣷو لࣹينْ دِ دَبْ، يࣹينْ اِتَمْ؞ خࣷفْلࣹينْ ڪࣷو خࣷفْ، سࣹيخْلُ ڪࣷو سࣹيخْلُ، ندَخْ يࣴفْ يࣷويُ، فَاگَاگَلْ مࣷو ݖِ وَرْ؞
