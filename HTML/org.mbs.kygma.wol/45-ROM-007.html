يࣷونُ مُوسَىٰ وࣵݖّ نَ گࣴمْڪَتُبْ اَلْمَسِ
7
‏1 اَمْ، بࣷڪّ يِ، يࣹينْ ݧِ خَمْ يࣷونْ وِ لاَيْ وَخَلْ، دَنگࣹينَ اُمݒّلࣹ نࣹ يࣷونْ يِلِفُلْ نِتْ ڪِ لُ مࣷيْ ݖِ گِيرُ دُنْدَمْ؟ ‏2 دَفَ مࣹلْ نِ ندَوْ سُيْ سࣺيْ؛ لِ فࣹيڪْ جࣴڪّࣴرَمْ دِ دُندّ، يࣷونَ يࣹيوْ بُومُ جࣴڪّࣴرَمْ ݖِ بَاتَمْ؞ وَايࣹ سُ جࣴڪّࣴرْ جِ دࣹييࣹي، ندَوْ سِ تࣹقْلِڪࣷو نَ اَڪْ يࣷونْ وِ ڪࣷ يࣹيوࣷونْ بُومُ جࣴڪّࣴرَمْ؞ ‏3 ڪࣷنْ نَگْ، لِ فࣹيڪْ جࣴڪّࣴرُ ندَوْ سِ دِ دُندّ، سُ سࣺيࣹي اَڪْ گࣹنࣹينْ گٗورْ، اَبْ جَالࣷوڪَتْ لࣹيسْ ڪࣷيْ وَخْ؞ سُ جࣴڪّࣴرْ جِ دࣹييࣹي نَگْ، يࣷونْ يِوِ نَ ندَوْ سِ، تࣹ سُ سࣺيࣹي اَڪْ ڪࣹنࣹينْ، دُ دࣷونْ نجَالࣷو؞
‏4 نࣷونُ لَ دࣹمࣹ اَڪْ يࣹينْ اِتَمْ، بࣷڪّ يِ؞ يَرَمُ اَلْمَسِ وِ نگࣹينْ بࣹنّࣷولْ، ݖِ نگࣹينْ دࣷونࣹ ݧُ دࣹييَلْ يࣷونُ مُوسَىٰ؞ نࣷونُ نگࣹينْ دࣷونࣹ مࣷومࣹيلُ ڪࣹنࣹينْ، تࣹ ڪࣷوڪُ مࣷويْ ڪِ دࣹڪِّ، نگِرْ نُ مࣹݧّ نجَرِݧْ لُ ݧࣹيلْ يࣵلَّ؞ ‏5 بَ نُ تࣷݒّࣹي سُنُ بَڪَّنْ، سُنُ بࣴگّ-بࣴگّ يُ بࣷنْ يِ ندِگَلِ يࣷونُ مُوسَىٰ يࣹي، ݧࣷو دَانْ لِگّࣺييْ ݖِ سُنُيْ ݖࣺرْ، نگِرْ نُ مࣹݧّ لُ نُ جࣴمࣹ ݖِ ندࣹي؞ ‏6 وَايࣹ تࣹيْ ݖِ سُنُ دِگّ اَڪْ يࣷونْ وِ نُ تࣴنڪُ وࣷونْ، لَ نُ دِ ݧُ دࣹي، بَ تࣹقْلِڪࣷو اَڪْ مࣷومْ؞ مࣷو تَخْ نُ دࣷونْ اَيْ جَامْ يُ تࣹگࣷو ڪِلِفْتࣹيفْ گُ يࣹيسْ، گِ نࣷووُگْ يࣵلَّ اِنْدِ، گَنَّاوْ بَ نُ وࣵݖّࣷو اَڪْ ڪِلِفْتࣹيفُ مبِنْدُ يࣷونُ مُوسَىٰ مَ وࣷونْ؞
يࣷونُ مُوسَىٰ مࣷو وࣷنࣹ نࣹ بࣵڪَّارْ مࣷو يࣺيسْ
‏7 اَنَ لُ لࣷولُ دِ تࣹڪِّ؟ خَنَا ڪࣷنْ يࣷونُ مُوسَىٰ بࣵڪَّارْ لَ؟ مُڪّ ڪَيْ!‏ وَايࣹ يࣷونُ مُوسَىٰ مࣷو مَ خَمَلْ لُيْ بࣵڪَّارْ؞ ندَخْ ڪَتْ سُ دُلْ وࣷونْ اَڪْ يࣷونْ وِ نࣹ:‏ «بُلْ خࣹمّࣹمْ يࣴفِ جَامْبُرْ¤،» ڪࣷنْ خࣹمّࣹمْتࣺيفْ سَخْ دُمَ خَمْ لُ مُ دࣷونْ؞ ‏8 وَايࣹ بࣵڪَّارْ مࣷو جَارࣹ بُنْتُبْ ندِگَلُ يࣷونْ، بَ يࣹي ݖِ مَنْ مبࣷولࣹمِ خࣹمّࣹمْتࣺيفْ يُ بࣷنْ؞ ندَخْ ڪَتْ، سُ يࣷونُ مُوسَىٰ اَمُلْ وࣷونْ، بࣵڪَّارْ دِ لُ دࣹي؞ ‏9 مَنْ دࣹ، بُ جࣴڪّࣷونْ، بَ مَ خَمَگُلْ يࣷونُ مُوسَىٰ، تࣹيوُلْ وࣷونْ مَيْ دُندّ؞ وَايࣹ بَ يࣷونُ مُوسَىٰ دِڪّࣹي، ݖَ لَ بࣵڪَّارْ نࣹ خِفࣺيتْ، دِ دُندّ، ‏10 مَنْ نَگْ، مَ دࣹي؞ ڪࣷنْ ندِگَلُ يࣷونْ وِ وَرَ جࣴمࣹ ݖِ اَگْ دُندّ، مࣷومْ ݖِ بࣷݒَّمْ مࣷو مَ مُجّ جࣴمࣹ ݖِ ندࣹي؞ ‏11 بࣵڪَّارْ مࣷو مَ جࣹڪّࣷو ندِگَلُ يࣷونْ، بَ نَخْ مَ، تࣹ ندِگَلُ يࣷونْ لَ مَ رࣹيࣹ اِتْ؞
‏12 ڪࣷنْ يࣷونُ مُوسَىٰ لُ سࣹلّ لَ، ندِگَلَمْ دِ لُ سࣹلّ، تࣹ جُبْ تࣹ بَاخْ؞ ‏13 لُ بَاخْ لࣷولُ دِ يࣷونُ مُوسَىٰ، خَنَا ڪࣷنْ مࣷو مَ تَخَ دࣹي؟ دࣺيتْ ڪَيْ!‏ بࣵڪَّارْ مࣷو فࣹيݧَلࣹ نࣷونُ مࣹلࣷومْ؛ مࣷو جࣴفَنْدِڪࣷو لُ بَاخْ، بَ رࣹيْ مَ، نگِرْ ندِگَلُ يࣷونْ دِ جُمْتُڪَايْ بُ بࣵڪَّارْ فࣺسَلࣹ گࣺݒّ مبࣷنَمْ گُ جࣺگِّ دَيࣷ؞
اَبْ گࣴمْڪَتْ نَمّ نَ لُ مُ تࣴلࣹي جࣴفࣹ
‏14 خَمْ نَنُ نࣹ يࣷونُ مُوسَىٰ ݖِ يࣵلَّ لَ بࣷڪّ، تࣹ مَنْ مَيْ نِتُ سُوخُ نࣹينْ، دِبْ جَامْ بُ ݧُ جَايْ بࣵڪَّارْ؞ ‏15 خَوْمَ سَخْ لُ وَرَلْ مَيْ جࣴفࣹ نِ مَيْ جࣴفࣹ؛ لِ مَ نَمّ، دُ لࣷولُ لاَيْ جࣴفࣹ، وَايࣹ لِ مَ بَݧْ، مࣷومْ لاَيْ جࣴفࣹ؞ ‏16 گَنَّاوْ لِ مَ بُگُّلْ، مࣷومْ لاَيْ دࣹفْ نَگْ، جُبࣷو نَا اَڪْ يࣷونُ مُوسَىٰ ݖِ دࣴگَّلْ نࣹ يࣷونُ مُوسَىٰ بَاخْ نَ؞ ‏17 ڪࣷنْ نَگْ دࣷوتُ مَنْ مَايْ جࣴفࣹ نࣷونُ، وَايࣹ دࣷولࣹيْ بࣵڪَّارْ بِ ݖِ مَنْ لَ؞ ‏18 خَمْ نَا نࣹ مَنْ، سَمَ بِنْدُ سُوخُ نࣹينْ وِي، لࣹنّ لُ بَاخْ دࣴڪُّ ݖِ؞ يࣺينࣹ جَا نگࣹيڪْ مَنْ، ڪࣵتَّنُ جࣴفࣹ لُ بَاخْ مࣷو فِ نࣹڪُّلْ؞ ‏19 ندَخْ ڪَتْ دُ لُ بَاخْ لِ مَ نَمَّ جࣴفࣹ لاَيْ جࣴفࣹ، وَايࣹ لُ بࣷنْ لِ مَ نَمُّلْ، مࣷومْ لاَيْ جࣴفࣹ؞ ‏20 تࣹ سُ دࣹي لِ مَ نَمُّلْ، مࣷومْ لاَيْ دࣹفْ، ڪࣷنْ ڪَيْ دࣷوتُ مَنْ مَا ڪࣷيْ دࣹفْ، وَايࣹ دࣷولࣹيْ بࣵڪَّارْ بِ دࣴڪّ ݖِ مَنْ مࣷو ڪࣷيْ دࣹفْ؞
‏21 تَڪّ بِ مَ ݖِ گِسْ دَالْ مࣷويْ لِي:‏ بُ مَ نَمّࣹي دࣹفْ لُ بَاخْ، لُ بࣷنَ مَيْ تࣹيوَلْسِ؞ ‏22 ندَخَمْ ݖِ سَمَ بِيرْ خࣷلْ، يࣷونُ يࣵلاَّ مَ نࣹيخْ؞ ‏23 وَايࣹ گِسْ نَا ݖِ سَمَيْ ݖࣺرْ، وࣹنࣹينْ يࣷونْ وُيْ خࣹيخْ اَڪْ يࣷونْ وِ سَمَمْ خࣹلْ جُبࣷولْ، تࣹ يࣷونْ وࣷووࣷو مَ جࣵݒّ نجَامْ، جࣺبَّلْ مَ دࣷولࣹيْ بࣵڪَّارْ بِ دࣴڪّࣹ سَمَيْ ݖࣺرْ؞ ‏24 ݖࣺيْ مَاڪَ رࣺيْ اَوْ تِيسْ!‏ اَنَ ڪُ مَيْ وَلّࣷوڪْ سَمَ يَرَمُ ندࣹي وِي؟ ‏25 وَايࣹ ݖَنتّ ݧࣹيلْ نَ يࣵلَّ مِ نُيْ وَلّࣷو جࣴمُّ عِيسَىٰ اَلْمَسِ، سُنُ سَنگّ؞
ڪࣷنْ دَالْ مَنْ، ݖِ سَمَ بࣷݒّ، ݖِ سَمَ وࣵلُّ خࣹلْ، يࣷونُ يࣵلَّ لاَيْ سُرْگَوُ، وَايࣹ ݖِ سَمَ وࣵلُّ يَرَمُ سُوخْ وِي، يࣷونُ بࣵڪَّارْ لاَيْ سُرْگَوُ؞
