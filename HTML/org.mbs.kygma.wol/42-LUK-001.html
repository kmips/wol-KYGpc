خِبَارُ جࣵمّ بُ اَلْمَسِ عِيسَىٰ، ݖِ جࣷتّلِبْ
لُوگْ
1
‏1 بَرࣹ نَ ݧُ فࣹخࣹي بِندّ اَبْ نࣹتَّلِبْ مبِرْ، يِ سࣷتِّ وࣷونْ ݖِ سُنُ بِيرْ، ‏2 نࣷونࣹي نُ ڪࣷ جࣷتࣹ ݖَ ݧَ ڪࣷ فࣹڪّࣹ وࣷونْ ݖَ ندࣷورْتࣹ لَ، تࣹ سࣹيدࣹ يࣷويُ مُجّ دࣷونْ جَوْرِݧِ ڪࣵدُّگْ يࣵلَّ؞ ‏3 مࣷو تَخْ مَنْ اِتْ، مَ فَسْ لاَ يࣺينࣹي بِنْدَلْ لࣺݒّ بَ مُ جࣹڪّ، يَوْ تࣹوࣷفِلْ مُ تࣹدّ مِ، گَنَّاوْ بَ مَ گࣴسْتࣷو مبِرْ مࣺݒّ، دَلࣹ ڪࣷ ݖَ ندࣷورْتࣹ لَ، بَ مُ لࣹيرْ نࣵݧّ؞ ‏4 سُ ڪࣷ دࣹفࣹي نگَ خَمْ خࣺلّ نࣹ لِ ݧُ لَ جࣵنگَلْ، لُ وࣺرْ ݒࣺݝّ لَ؞
مَلاَڪَ يࣺگْلࣹ نَ جُدُّبْ يَحْيَىٰ
‏5 جَنتّ يَ اࣹرࣷدّ دࣹي بُورُ رࣺيوُمْ يَوُتْ يِ، اَبْ سَرْخَلْڪَتْ لَ وࣷونْ بُ ݧُيْ وَخْ سࣵڪَّرْيَا، بࣷڪّ ݖِ لࣵنگُ سَرْخَلْڪَتْ، يِ اَسْڪَنࣷو ݖِ اَبِيَ، تࣹ سࣷخْنَامْ اࣹلِسَبࣹتْ اَسْڪَنࣷو ݖِ هَـٰرُونَ؞ ‏6 ݧࣷومْ ݧَارْ ݧࣺݒَّ جُبࣷونْ، يࣵلَّ سࣹيدࣹ، تࣹ مُݖّ اَيِبْ نِ ݧُيْ جࣴفࣹ سَنْتَانࣹ يࣹيڪْ ندِگَلِ يࣷونِ بࣷرࣷومْ بِ؞ ‏7 اَمُݧُ وࣷونْ دࣷومْ نَگْ، ندَخْ اࣹلِسَبࣹتْ مَنُلْ وࣷونَ اَمْ دࣷومْ، تࣹ اِتْ بࣷوبَ ݧࣷومْ ݧَارْ ݧࣺݒَّ دࣹمْ بَ مࣵگَّتْ؞
‏8 سࣵڪَّرْيَا لَ لࣵنگَمْ اَيࣹ وࣷونْ، مُيْ دࣹفْ لِگّࣺييُ ݖَرْخَلَمْ فَ ڪَنَمْ يࣵلَّ؞ ‏9 بَ ݧُ تࣹگࣷو بَنتّ، نَ ڪࣷ سَرْخَلْڪَتْ يِ بَاخࣷو وࣷونْ، مࣷومْ لَ بَنتّ بَ دَلْ، مُ وَرَ دُگّ بِيرْ نࣺيگُ بࣷرࣷومْ بِ، نگِرْ تَالْ فَ سَرَخُ ݖُورَايْ؞
‏10 مبࣷولࣷو مࣺݒّ نَگَ نگَ وࣷونْ ݖَ بِتِ دِ جُلِّ، ݖَ وَخْتُوْ سَرَخُ ݖُورَايْ بࣷوبُ؞ ‏11 ݖِ بِيرْ لࣷولُ مࣹنّ مَلاَڪَمْ بࣷرࣷومْ بِ جࣹڪِّ فࣹيݧُ سࣵڪَّرْيَا، ݖَ وࣹتُ ندِجࣷورُ سَرْخَلُڪَايُ ݖُورَايْ بَ؞ ‏12 بَ ڪࣷ سࣵڪَّرْيَا گِسࣹي دَا جَاخْلࣹ، تِيتَانگࣹ جࣵݒّ ڪࣷ؞ ‏13 مَلاَڪَ مَ نࣹ ڪࣷ:‏ «سࣵڪَّرْيَا، بُلْ تِيتْ، ندَخْ سَ ݧَانْ نَنگُ نَ؞ اࣹلِسَبࣹتْ سَ سࣷخْنَ دِنَ اَمْ دࣷومْ جُ گٗورْ، نَنگَ ڪࣷ تُدّࣹ يَحْيَىٰ؞ ‏14 دِنگَ اَمْ مبࣺگْتࣹ اَڪْ بࣵنّࣹيخْ، تࣹ ݧُ بَرࣹييْ بࣺگْ ݖِ جُدّࣷومْ، ‏15 ندَخْ ڪُ مࣵگّ لَيْ دࣷونْ، بࣷرࣷومْ بِ سࣹيدࣹ؞ بِيݧْ اَڪْ لࣹنّ لُيْ مࣵنْدࣹي نَگْ، دُ ڪࣷ نَانْ، نࣷو گُ سࣹلّ گِ مࣷويْ دَلࣹ ݖَ بِيرُ ندࣹيَمْ، سࣷلُ ڪࣷ بَ فࣹيسْ؞ ‏16 نِتِ بࣵنِّ إِسْرَاءِيلْ ݧُ بَرࣹ، مࣷو لࣹينْ دِ وَݧِّ ݖِ بࣷرࣷومْ بِ سࣹينْ يࣵلَّ؞ ‏17 مࣷويْ ندَوْ لِ ڪࣷيْ جِيتُ، اࣵندّ اَڪْ لࣹيرْ گَ اَڪْ خَمْ-خَمْ، بَ وࣷونْ ݖَ يࣷنࣹنتّ يࣵلَّ اِلْيَاسْ، بَ وَݧِّ خࣷلُ بَايْ ݖِ دࣷومَمْ، وَݧِّ ڪُ دࣺگَّدِ ݖِ خࣹلُمْ اَجِ جُبْ، لࣷولُ يࣺݒّ نگِرْ وَاجَلْ اَوْ خࣹيتْ بَ ݧُ جࣹڪّ نگِرْ بࣷرࣷومْ بِ؞»
‏18 سࣵڪَّرْيَا نَگْ نࣹ مَلاَڪَ مِ:‏ «لُ مَيْ دࣹفْ اَبْ تَڪّ ݖِ لࣷولُ بَ مُ بِرْ مَ؟ ندَخْ مَنْ مࣵگَّتْ نَا، تࣹ سَمَ سࣷخْنَ اِتْ لࣵقْ نَ اَيْ فَنْ؞» ‏19 مَلاَڪَ مَ نࣹ ڪࣷ:‏ «مَنْ مَايْ جِبْرِلْ مِيْ تَخَوْ فِ ڪَنَمْ يࣵلَّ؞ دَݧُ مَا يٗنِّ نگِرْ مَ وَخْ اَڪْ يَوْ، يࣺگَلْ لَ بِي خِبَارُ جࣵمّ؞ ‏20 لࣺيگِ نَگْ، دَمَ نࣹ، گَنَّاوْ گࣴمُلࣷو سَمَ ڪࣵدُّ، گِيْ جࣷتْ، سࣷتِّ، دِنگَ لُو تࣹ دࣷوتُلࣷو مَنَ وَخْ، بَ ڪࣹرَ لࣷولُ دِ اَمْ؞»
‏21 ݖِ بِيرْ لࣷولُ مبࣷولࣷو مَا نگَيْ خَارْ سࣵڪَّرْيَا، جَاخْلࣹ لࣷولْ ندَخْ يࣵگَّايَمْ ݖَ بِيرْ نࣺيگْ يࣵلَّ بَ؞ ‏22 بَ مُ گࣺنّࣹي نَگْ، مَنُلَ وَخْ اَڪْ ݧࣷومْ؞ ݧُ دَلْدِ خَمْ نࣹ اَمْ نَ لُ ڪࣷ فࣹيݧُ ݖَ نࣺيگْ يࣵلَّ بَ؞ مَ نگَ لࣹينْ دِ لِيَارْ، مَنَتُلَ وَخْ؞
‏23 نَڪَ لَ سࣵڪَّرْيَا فࣺگَلْ اَيُ ݖَرْخَلَمْ، دَلْدِ ݧِبِّ؞ ‏24 گَنَّاوْ گِ، سࣷخْنَامْ اࣹلِسَبࣹتْ اࣴمبّ؞ لࣴلُ نَ جُرٗومِ وࣹيرْ؞ مَ نگَ نَانْ:‏ ‏25 «لِي بࣷرࣷومْ بࣹي مَ ڪࣷ دࣹفَلْ ݖِ جَنتّ يِ مُ مَ گࣹيسࣷو، بَ تࣹگِّلْ مَ سَمَ گࣵݖّࣹ گِ ݖِ بِيرْ نِتْ ݧِ!‏»
جِبْرِلْ يࣺگْلࣹ نَ جُدُّبْ عِيسَىٰ
‏26 بَ اࣹلِسَبࣹتْ تࣷلّࣷو ݖِ جُرٗومْ بࣹنِّ وࣹيرَمْ، ݖَ لَ يࣵلَّ يࣹبَلْ جِبْرِلْ مَلاَڪَ مِ، دࣴڪّ بِ ݧُيْ وَخْ نَسَرࣹتْ ݖَ دِيوَانُ گَلِلࣹ، ‏27 ݖِ جَنقّ بُ دِگࣷو سࣺيْ اَڪْ وَا جُ ݧُيْ وَخْ يُوسُفَ تࣹ بࣷڪّ ݖِ وَا ڪࣴرْ دَاوُدَ؞ جَنقّ بَ مَرْيَامَ لَݧُ ڪࣷيْ وَخْ؞ ‏28 مَلاَڪَ مَ اَگْسِ بَ ݖَ مࣷومْ، نࣹ ڪࣷ:‏ «جࣵمّ نگَ اَمْ، يَوْ مِ ݧُ بَاخࣹ، بࣷرࣷومْ بِ يَوْ لَ اࣵنْدَلْ؞»
‏29 وَخِ مَلاَڪَ مَ نَگْ جَاخَلْ مَرْيَامَ، مُيْ خَلاَتْ لُ نُيࣷو بࣷوبُ مَنَ تࣹڪِّ؞ ‏30 مَلاَڪَ مَ نࣹ ڪࣷ:‏ «مَرْيَامَ، بُلْ تِيتْ، ندَخْ دَجࣹ نگَ اَڪْ يِوُ يࣵلَّ؞ ‏31 دِنگَ اࣴمبّ، بَ جُرْ دࣷومْ جُ گٗورْ؛ نَنگَ ڪࣷ تُدّࣹ عِيسَىٰ؞ ‏32 ڪُ مࣵگّ لَيْ دࣷونْ، تࣹ دࣹيسْ نَ ڪࣷ وࣷويࣹ دࣷومُ اَجِ ڪَوࣹ جِ؞ بࣷرࣷومْ بِ يࣵلَّ مࣷو ڪࣷيْ جَگْلࣹيلْ نگُورُگْ دَاوُدَ مَامَمْ؞ ‏33 مࣷويْ نگُورُ فِ ڪَوْ وَا ڪࣴرْ يَعْقُوبَ بَ فࣵوّ، تࣹ نگُورَمْ دُ فࣷقْ مُڪّ؞»
‏34 ݖِ ڪَوْ لࣷولُ مَرْيَامَ لاَجْ مَلاَڪَ مِ نࣹ ڪࣷ:‏ «نُ لࣷولُ دِ اَمࣹ، تࣹ مَنْ خَوْمَ گٗورْ؟» ‏35 مَلاَڪَ مِ نࣹ ڪࣷ:‏ «نࣷو گُ سࣹلّ گِ مࣷويْ وࣵݖّسِ ݖِ يَوْ، دࣷولࣹيْ اَجِ ڪَوࣹ جِ يِيرْ لَ؞ مࣷو تَخْ اِتَمْ ڪُ سࣹلّ ڪِيْ جُدّࣷو ݖِ يَوْ، دࣹيسْ نَ ڪࣷ وࣷويࣹ دࣷومُ يࣵلَّ؞ ‏36 تࣹ ڪَتْ اࣹلِسَبࣹتْ سَ مبࣷڪّ مِ اِتَمْ اࣴمبّ نَ دࣷومْ جُ گٗورْ تࣹ فࣹڪّ ڪࣷ مࣵگَّتْ؞ ڪِ ݧُ دࣷونْ وࣷويࣹ جِگࣺينْ جُ جَاسِرْ، مُ نگِ نِي ݖِ جُرٗومْ بࣹنِّ وࣹيرَمْ، ‏37 ندَخْ يࣵلَّ، دَرَ تࣴوُ ڪࣷ؞» ‏38 مَرْيَامَ دَلْدِ نࣹ:‏ «مَا نگِ، دِ جَامُبْ بࣷرࣷومْ بِ؞ نِي نگَ وَخࣹ، نَ اَمࣹ نࣷونُ ݖِ مَنْ؞»
بَ لࣷولُ اَمࣹي مَلاَڪَ مَ بࣵيِّڪࣷو ݖَ مࣷومْ، دࣹمْ؞
مَرْيَامَ سࣹيتِ نَ اࣹلِسَبࣹتْ
‏39 ݖِ فَنْ يࣷويُ لَ مَرْيَامَ فَبُ گَاوْ، جࣴمْ دِيوَانُ تُندّ يَ، ݖَ دࣴڪُّبْ يُدࣹ، ‏40 بَ سࣵڪَّرْيَا دࣴڪّࣹ؞ مُ اَگْسِ ݖَ ڪࣴرْ گَ، نُيُ اࣹلِسَبࣹتْ؞ ‏41 نَڪَ لَ اࣹلِسَبࣹتْ دࣺگّ نُيࣷوبْ مَرْيَامَ، دࣷومْ جَ مُ اࣴمبّ يࣴنگُ ݖَ بِيرَمْ؞ اࣹلِسَبࣹتْ دَلْدِ فࣹيسࣹ نࣷو گُ سࣹلّ گِ؞ ‏42 مُ اࣵدُّ ݖَ ڪَوْ نࣹ:‏ «يَا گࣴنَ بَرْڪࣹيلْ ݖِ جِگࣺينْ ݧِ، سَ دࣷومْ جِ نگَ اࣴمبّ اِتْ بَرْڪࣹيلْ نَ!‏ ‏43 اَنَ ڪُ مَ دࣷونْ بَ سَمَ ندࣹيُ سَنگّ دِ مَ سࣹيتْسِ؟ ‏44 نَڪَ لَ سَ ڪࣵدُّگْ نُيࣷو تࣵبِّ سَمَ نࣷݒّ رࣹڪّ، سَمَ دࣷومْ جِ يࣴنگُ ݖِ سَمَ بِيرْ بِ ندَخْ بࣺگْ؞ ‏45 ندࣷڪّلࣹي يَوْ مِ گࣴمْ نࣹ مبِرْ مِ ݧُ لَ يࣺگَلْ، تࣹ مُ تُڪّࣹي ݖِ بࣷرࣷومْ بِ دِنَ سࣷتِّ!‏»
مَرْيَامَ وࣷيْ نَ مبَاخُ يࣵلَّ
‏46 مَرْيَامَ نَگْ نࣹ:‏
‏47 «سَمَ خࣷلْ لاَيْ مࣵگَّلࣹ بࣷرࣷومْ بِ،
سَمَگْ نࣷو لاَيْ بࣺگࣹ يࣵلَّ مِ مَيْ مُسَلْ،
‏48 مࣷو گࣹيسُ بِي جَامَمْ بُ تࣹڪّࣹيدِ،
بَ مَاسࣷو مَاسْ دࣷخࣹ فِ وࣷويࣹ مَ بࣷرࣷومْ مبࣺگْ مِ،
‏49 ندࣹ ڪِيْ بࣷرࣷومْ دࣷولࣹ مࣷو مَ دࣹفَلْ جَلࣷورࣹ،
سࣹلّنگَ ݧࣹيلْ نَوْ تُرَمْ؞
‏50 «يࣴرْمَنْدࣹيمَ ݧࣹيلْ اَيْ رَگَلْڪَتَمْ،
مَاسْ ݖِ ڪَوْ مَاسْ؞
‏51 دࣷولࣹيْ لࣷخࣷومْ لَ دࣹفࣹ اَيْ جَلࣷورࣹ،
تَسَارࣹ ݧِ جِڪّࣷوࣷو رࣺيْ-رࣺيْلُ،
‏52 وࣵݖّࣹي بُورْ يِ ݖِ سࣹينِ گࣵنگُنࣹ،
يࣺڪَّتِ بَادࣷولࣷ يِ؞
‏53 ݧِ خِيفْ، مُ رࣹگَّلْ لࣹينْ لُ نࣹيخْ،
ݧِ دُونْلࣹ، مُ دࣵقࣹ لࣷخࣷيْ نࣹينْ؞
‏54 مࣷو وَلُّ بࣵنِّ إِسْرَاءِيلْ، جَامَمْ،
بࣵيّࣹي ڪࣷ خࣹلْ يࣴرْمَنْدࣹيمْ،
‏55 نࣷونࣹي مُ ڪࣷ دِگࣹ وࣷونْ سُنُيْ مَامْ،
ݧࣹيلْ إِبْرَاهِيمَ اَڪْ اَسْڪَنَمْ بَ فࣵوّ؞»
‏56 مَرْيَامَ تࣷوگْ نَ فَ اࣹلِسَبࣹتْ لُ وَرَ تࣷلّࣷوڪْ ݧࣹتِّ وࣹيرْ، سࣷوگَ ݧِبِّ؞
لُ جࣴمْ ݖِ جُدُّبْ يَحْيَىٰ
‏57 گَنَّاوْ گِ اࣹلِسَبࣹتْ دࣹمْ بَ اࣵݒَّمْ مَتْ، مُ مُݖّ، اَمْ دࣷومْ جُ گٗورْ؞ ‏58 دࣴڪَّنْدࣷومْ يَاڪْ اَيْ بࣷڪَّمْ يࣺگْ نࣷونُ ڪࣷ بࣷرࣷومْ بِ خࣺيوَلࣹي يࣴرْمَنْدࣹيمْ، ݧُ بࣷڪّ اَڪْ مࣷومْ مبࣺگْتࣹ مَ؞
‏59 بَ بࣺسْ بَ دࣺلْسࣹي، ݧُ دِڪّ خَرْفَلْسِ خَلࣹ بَ، بࣴگّ ڪࣷو تُدّࣹ سࣵڪَّرْيَا، دِݒّࣹي ڪࣷ بَايَمْ؞ ‏60 يَايَمْ نَگْ نࣹ لࣹينْ:‏ «دࣺيتْ، يَحْيَىٰ لَيْ تُدّ؞» ‏61 ݧُ نࣹ ڪࣷ:‏ «وَايࣹ ڪࣹنّ ݖِ سَيْ بࣷڪّ تُدُّلْ نࣷونُ!‏»
‏62 ݖِ ڪَوْ لࣷولُ ݧُ لِيَارْ بَايَمْ، نگِرْ خَمْ نُ مُ بࣴگّ ݧُ تُدّࣹ خَلࣹ بِ؞ ‏63 سࣵڪَّرْيَا لاَجْ اࣵلُّوَ، ݧُ جࣷخْ ڪࣷ، مُ بِندّ ݖِ:‏ «يَحْيَىٰ مࣷويْ تُرَمْ؞» ݧࣺݒّ وَارُ؞ ‏64 ݖَ سَا سَ گࣺمِّݧْ گَ اُبِّڪُ، لࣵمِّݧْ وَ نَنگُ، مُيْ وَخْ، دَلْدِ سَنتّ يࣵلَّ؞ ‏65 بَ لࣷولُ اَمࣹي تِيتَانگࣹ دِڪَّلْ مبࣷولࣹمْ ݧَ لࣹينْ سࣺقْ، مبِرْ مࣷومُ مࣺݒّ نَگْ دࣷونْ وَخْتَانْ ݖَ دِيوَانُ تُنْدِ يُدࣹ گࣺݒّ؞ ‏66 ݧَ ݖَ دࣺگّ ݧࣺݒّ دَلْدِ بࣵيِّ خࣹلْ ݖَ مبِرْ مَ، تࣹ نَانْ:‏ «لُ خَلࣹ بِي نَرَ دࣷونْ اࣴلّࣴگْ؟» نگِرْ مَنࣷورࣹيْ بࣷرࣷومْ بࣹي اࣵنْدࣷونْ اَڪْ مࣷومْ؞
سࣵڪَّرْيَا سَنتّ نَ يࣵلَّ
‏67 ݖِ ڪَوْ لࣷولُ سࣵڪَّرْيَا بَايُ يَحْيَىٰ فࣹيسࣹ نࣷو گُ سࣹلّ گِ، دَلْدِ تࣵمْبَلࣹي بِرَلْ ڪࣵدُّگْ وَخْيُ، نࣹ:‏
 
‏68 «ݖَنتّ ݧࣹيلْ نَ بࣷرࣷومْ بِ يࣵلَّيْ إِسْرَاءِيلْ،
مࣷو سࣹيتْسِ ݧࣷݧَمْ، بَ جࣷتْ لࣹينْ!‏
‏69 مࣷو نُ فࣹيݧَلَلْ مُسَلْڪَتْ بُ مَنࣷورࣹوُ،
فِ بِيرْ ڪࣴرْ دَاوُدَ جَامَمْ بَ،
‏70 نࣷونࣹي مُ ڪࣷ وَخࣹ وࣷونْ ݖَ جَنتّ يُ جࣴڪّ يَ، يࣷنࣹنْتَمْ يُ سࣹلّ يَ وࣷونْ جࣷتّلِ،
‏71 مࣷو نُ مُسَلْ ݖِ سُنُيْ نࣷونْ،
جࣴلࣹ نُ ݖِ سُنُ لࣷخࣷيْ مبࣷولࣹمْ بَݧَالࣹ؞
‏72 مࣷو جࣴفࣹ يࣴرْمَنْدࣹ ݧࣹيلْ سُنُيْ مَامْ،
تࣹيْ بࣵيِّ خࣹلْ ڪٗلّࣴرࣹيمْ گُ سࣹلّ،
‏73 اَڪْ نگِݧْ لَ مُ گِݧَلࣷونْ سُنُ مَامْ إِبْرَاهِيمَ،
‏74 نࣹ مࣷو نُيْ مَيْ نُ مُݖّࣹ ݖِ لࣷخࣷيْ نࣷونْ،
بَ مَنْ ڪࣷو جَامࣷو خࣹلْ مُ دَلْ
‏75 ݖِ بِيرْ سࣹلّنگَ اَڪْ نجُبْ ݖِ ڪَنَمَمْ، سُنُ گِيرُ دُندّ؞
 
‏76 «نگٗورْ سِ نَگْ يَوْ، يࣷنࣹنْتُ اَجِ ڪَوࣹ جِ لࣹيسْ لَيْ وࣷويࣹ،
ندࣹ يَايْ دࣷخْ، جِيتُ سَنگّ بِ، خࣵلَّلْ ڪࣷوْ يࣷونَمْ،
‏77 نگِرْ يࣺگَلْ ݧࣷݧَمْ اَگْ نجࣷتْ
گُيْ تࣷݒّ سࣹينْ نجࣺگَّلُگْ بࣵڪَّارْ؞
‏78 سُنُ يࣵلاَّ يَاتُ يࣴرْمَنْدࣹ،
مࣷو نُ فࣹنڪَلْ اَبْ جَنتّ،
‏79 دِ نِيتَلْ ݧِ تࣷوگࣹ بِيرْ لࣴنْدࣴمْ اَڪْ تَڪّندࣹيرُ ندࣹي،
دِ نُ گِنْدِ، تࣹگْ نُ ݖِ يࣷونُ جࣵمّ؞»
‏80 نگٗورْ سَ نَگْ دِ مࣵگّ، خࣹلْ مَ دِ يࣷڪُّ؞ مُ دࣴڪّ ݖَ مࣵنْدِݝْ مَ، بَ بࣺسْ بَ مُ فࣹيݧࣷو بࣵنِّ إِسْرَاءِيلْ؞
