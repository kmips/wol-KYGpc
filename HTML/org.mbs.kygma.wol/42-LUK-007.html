نجِيتُ تَڪّ-دࣹرْ بَ يࣷولࣷو نَ نگࣴمَمْ
7
‏1 عِيسَىٰ وَخْ نَ ڪࣵدّࣷومْ گࣷوگُ گࣺݒّ، اَسْڪَنْ وَيْ دࣺگّ، مُ دࣷورَ جࣵلّ بِيرْ ڪَݒࣹرْنَوُمْ؞ ‏2 فࣷوفَ نَگْ اَبْ نجِيتُ تَڪّ-دࣹرْ بُ وَا رࣷومَ نگَ فَ وࣷونْ، اَبْ جَامَمْ وࣷݒّࣷونْ بَيْ بࣴگَّ دࣹي، تࣹ نجِيتْ لَ سࣷݒّࣷونْ ڪࣷ لࣷولْ؞ ‏3 بَ نجِيتْ لَ دࣺگّࣹي مبِرِ عِيسَىٰ، دَا يࣹبَلْ ݧࣹنّ ݖَ مَگِ يَوُتِ دࣴڪّ بَ ݖَ مࣷومْ، ݧَانْ ڪࣷ نگِرْ مُ دِڪّ خࣹتّلِ جَامَمْ؞ ‏4 مَگِ دࣴڪّ بَ اَگْسِ ݖَ عِيسَىٰ، تِنُ ڪࣷو تِنُ، نࣹ ڪࣷ:‏ «ڪِ نُ يࣹبَلْ ݖِ يَوْ دࣹ، يࣹلّࣷو نَ سَ ندِمْبَلْ، ‏5 ندَخْ ڪُ سࣷݒّ سُنُوْ خࣹيتْ لَ، تࣹ سُنُ جࣵنگُ بِ سَخْ، مࣷو نُ ڪࣷ تَبَخَلْ؞»
‏6 عِيسَىٰ دَلْدِ اࣵندّ اَڪْ ݧࣷومْ؞ نَڪَ لَ جُبْسِ ڪࣴرْ گَ، نجِيتْ لَ يٗبَّنْتࣹ اَيْ خَرِتَمْ ڪࣵدّࣷومْ، ݧُ گَتَنْدُ عِيسَىٰ، نࣹ ڪࣷ:‏ «سَنگّ بِ، جَرُلْ نگَيْ سࣷنّ، ندَخْ سَبْ تࣵنڪّ سَمَ بِيرْ ڪࣴرْ، مَنْ يࣹيࣷووُمَ ڪࣷ؞ ‏7 مࣷو مَ لَ تࣹي گَتَنْدُلْ سَمَ بࣷݒّ، ندَخْ جࣵݒّ نَا نࣹ لࣷولُ جࣷمبّ نَ مَ؞ وَخَلْ گࣹنّ ڪࣵدُّ رࣹڪّ، سَمَ سُرْگَ وࣺرْ؞ ‏8 ندَخْ مَنْ ݖِ سَمَ بࣷݒّ ݖِ ڪِلِفْتࣹيفْ لاَ تࣴنڪُ، تࣹ مَنْ اِتَمْ اَيْ تَڪّ-دࣹرَ نگِ ݖِ سَمَ ندِگَلْ؞ سُ مَ نࣹي ڪِي:‏ ‹دࣹمَلْ،› مُ دࣹمْ؞ مَ نࣹ ڪࣹي:‏ ‹ݧࣴوَلْ،› مُ ݧࣴوْ؞ تࣹ سَمَ جَامْ بُ مَ نࣹ:‏ ‹دࣹفَلْ لِي،› مُ دࣹفْ ڪࣷ؞»
‏9 بَ عِيسَىٰ دࣺگّࣹي لࣷولُ دَا يࣺيمُ ݖَ نجِيتْ لَ؞ مُ نࣹ گࣹيسْ مبࣷولࣷو مَ تࣷݒّ ݖَ مࣷومْ، نࣹ لࣹينْ:‏ «دَمَ نࣹ، گِي نگࣴمْ گُ رࣺيْ، بࣵنِّ إِسْرَاءِيلْ سَخْ، گِسُمَ ڪࣷ ݖِ؞» ‏10 ندَوْ يَ دَلْدِ دࣹلُّ ݖَ ڪࣴرْ گَ، فࣹڪّ جَامْ بَ وࣺرْ؞
عِيسَىٰ دࣹڪَّلْ نَ دࣷومُ جࣴتُنْ بَ
‏11 لَ ݖَ تࣹگُ عِيسَىٰ دࣹمَتِ دࣴڪّ بُ ݧُيْ وَخْ نَيِنْ، اَيْ تَالِبࣹيمْ اَڪْ مبࣷولࣷو مُ بَرࣹ اࣵندّ اَڪْ مࣷومْ؞ ‏12 بَ مُ جُبْسࣹي بُنْتُ دࣴڪّ بَ، يࣹمُلْ ݖِ لُ مࣷيْ اَبْ رࣷبْ، ڪَ دࣹي دِ دࣷومُ جࣴتُنْ، تࣹ مࣷومْ رࣹڪّ لَ اَمࣷونْ؞ نِتِ دࣴڪّ بَ دࣹفْ مبࣷولࣷو مُ بَرࣹ، اࣵندّ اَڪْ سࣷخْنَ سَ؞ ‏13 سَنگّ بِ نَگْ گِسْ ڪࣷ، ݧࣹيوَنْتࣹ ڪࣷ، نࣹ ڪࣷ:‏ «بُلْ جࣷويْ؞» ‏14 مُ دِڪّ بَ تࣹگْ لࣷخࣷومْ ݖَ جَاتْ گَ، ݧَ ڪࣷ جࣵݒّࣷو تَخَوْ؞ مُ نࣹ:‏ «نگٗورْ سِ، يَوْ لاَيْ وَخْ، جٗگَلْ؞» ‏15 نࣺيوْ بَ جٗگْ تࣷوگْ، دِ وَخْ، مُ دࣹلّࣷو ڪࣷ يَايْ جَ؞ ‏16 تِيتَانگࣹ نَگْ جࣵݒّ ݧࣺݒّ، ݧُ سࣵبَّالْ يࣵلَّ، نَانْ:‏ «يࣷنࣹنتّ بُ مَگْ فࣹيݧْ نَ ݖِ سُنُ بِيرْ!‏» تࣹ نَانْ «يࣵلاَّ وَلُّسِ ݧࣷݧَمْ!‏»
‏17 ݖࣷووْ لࣷولُ نَگْ لَوْ ݖَ مبࣷولࣹمْ رࣺيوُمْ يَوُتْ اَڪْ مبࣷولࣹمْ فَ ڪࣷ وࣴرْ؞
عِيسَىٰ جࣵنگّلࣹ نَ ݖِ مبِرُمْ يَحْيَىٰ
‏18 ݖِ ڪَوْ لࣷولُ تَالِبࣹيْ يَحْيَىٰ اࣵگّࣹ يَحْيَىٰ لࣷولُ لࣺݒّ؞ مُ وࣷو ݧَارْ ݖَ تَالِبࣹ يَ، ‏19 يࣹبَلْ لࣹينْ ݖِ سَنگّ بِ، لاَجْ ڪࣷ نࣹ ڪࣷ:‏ «ندَخْ يَوْ يَا دِ ڪِ وَرَ ݧࣴوْ، اَمْ ڪࣹنࣹينْ لَنُيْ سࣺينْتُ؟» ‏20 تَالِبࣹ يَ دِڪّ نࣹ ڪࣷ:‏ «يَحْيَىٰ مࣷو نُ يࣹبَلْ، دِ لَ لاَجْ ندَخْ يَا دِ ڪِ وَرَ ݧࣴوْ، اَمْ ڪࣹنࣹينْ لَنُيْ سࣺينْتُ؟»
‏21 وَخْتُ وࣷووَ تࣹمْبࣹ نَگْ عِيسَىٰ فَجْ نَ ݧُ بَرࣹ، تࣵگَّلࣹ لࣹينْ اَڪِ جࣵنگࣷرࣷ اَڪِ جَگَدِ اَڪِ رَبْ، سِلْمَخَ يُ بَرࣹ، مُ بَاخࣹ لࣹينْ، ݧُيْ گِسْ؞ ‏22 مُ نࣹ ندَوْ يَ:‏ «لِ نگࣹينْ گِسْ تࣹ دࣺگّ ڪࣷ، دࣹمْلࣹينْ اࣵگّࣹ ڪࣷ يَحْيَىٰ؞ سِلْمَخَ يَا نگِ گِسْ، لَفَݧْ يِ دࣷخْ، گَانَ يِ وࣺرْ، تࣴخْ يِيْ دࣺگّ، نࣺيوْ يِ دࣹڪِّ، نࣺيوْ-دࣷولࣹ ݧِ دࣺگّ خِبَارُ جࣵمّ بِ¤؞ ‏23 ڪُ مَ تَخُلْ نگَ خࣵݖّ سَ ݒࣺيتࣹ اَڪْ مَنْ نَگْ، ندࣷڪّلࣹي يَوْ؞»
‏24 بَ ندَوِ يَحْيَىٰ دࣹمࣹي، عِيسَىٰ دَلْدِ وَخْ اَڪْ مبࣷولࣷو مَ ݖِ مبِرُمْ يَحْيَىٰ، نࣹ لࣹينْ:‏ «لُ نگࣹينْ سࣹيتَانِ وࣷونْ ݖَ مࣵنْدِݝْ مَ؟ خَنَا بَرَخْ بُ نگࣹلَوْ لِيْ يࣴنگَلْ؟ ‏25 ڪࣷنْ لُ نگࣹينْ سࣹيتَانِ وࣷونْ؟ بࣷرࣷومْ يࣺرࣹ يُ رَفࣹتْ؟ وَايࣹ بࣷرࣷومْ يࣺرࣹ يُ يࣵنجّ يࣷويُ تࣹ سࣹينْ دُندّ بَڪَّنࣹ، ݖِ ڪࣴرِ بُورْ لَݧُ دࣴڪّ؞ ‏26 ڪࣷنْ نَگْ لُ نگࣹينْ سࣹيتَانِ وࣷونْ؟ اَبْ يࣷنࣹنتّ؟ وَاوَاوْ، مَا لࣹينْ ڪࣷ وَخْ، تࣹ وࣹيسْ نَ اَبْ يࣷنࣹنتّ سَخْ؞ ‏27 ݖِ مࣷومْ مࣷومُ لَݧُ بِنْدࣷونْ نࣹ:‏
‹مَا نگِي دِ يࣹبَلْ سَمَ ندَوْ، مُ جِيتُ لَ،
خࣵلَّلْ لَ سَوْ يࣷونْ؞›
‏28 مَا لࣹينْ ڪࣷ وَخْ، لِ جِگࣺينْ جُرْ، ڪࣹنّ سُتُ ݖِ يَحْيَىٰ؞ وَايࣹ ڪِ گࣴنَ تُوتْ ݖِ نگُورُگْ يࣵلَّ سَخْ مࣷو ڪࣷ سُتْ؞»
‏29 مبࣷولࣹمْ ݧَ دࣺگْلُ عِيسَىٰ، بَ ݖَ جُوتِڪَتْ يَ سَخْ، دࣴگَّلْ نَݧُ يࣵلَّ، ندَخْ نَنگُ نَݧࣷو سٗوبُ سٗوبࣹبْ يَحْيَىٰ؞ ‏30 وَايࣹ فَرِسࣹنْ يَ اَڪْ خَمْڪَتِ يࣷونْ يَ ݧࣷو گࣵنْتَلْ لَ لࣹينْ يࣵلَّ نَمَّلࣷونْ، بَ يَحْيَىٰ سٗوبُ لࣹينْ ݖِ ندࣷخْ؞
‏31 عِيسَىٰ نࣹيتِ:‏ «اَنَ ڪࣷنْ لُ مَيْ نِرُلࣹيلْ نِتِ تࣹيْ جِي؟ اَنَ لُ ݧُيْ نِرࣷولْ؟ ‏32 ݧُ نگِ نِرࣷوڪْ خَلࣹ يُ تࣷوگْ ݖَ ݒࣺنݖّ مَ دِ اَوࣷونْتࣹ، نَانْ:‏
‹نࣷو لࣹينْ تࣷخࣷرࣷولْ، فࣹݖُّلࣹينْ،
نُ دࣴجَلْ لࣹينْ، جࣷويُلࣹينْ؞›
‏33 ندَخْتࣹ يَحْيَىٰ دَا فࣹيݧْ، لࣹڪُّلْ دُگُبْ، نَانُلْ بِيݧْ، نگࣹينْ نࣹي:‏ ‹اَوْ رَبْ لَ اࣵنْدَلْ؞› ‏34 دࣷومُ نِتْ ڪِ فࣹيݧْ، دِ لࣹڪّ اَڪَ نَانْ، نگࣹينْ نَايْ:‏ ‹ڪࣹي فُقْلࣹ، نَانْ-ڪَتُبْ بِيݧْ بِييْ خَرِتࣷوڪْ جُوتِڪَتْ يࣹيڪْ بࣵڪَّارْڪَتْ يِ؞› ‏35 وَايࣹ خࣹلْ مُ رَفࣹتْ، مبࣷولࣹمْ لَ مُ جُرْ مࣷويْ لَ ڪࣷيْ دࣴگَّلْ؞»
عِيسَىٰ گَنࣹ نَ سِمࣷݝْ فَرِسࣹنْ بَ
‏36 ڪࣹنّ ݖَ فَرِسࣹنْ يَ مࣷو وࣷو وࣷونْ عِيسَىٰ، نگِرْ ݧُ بࣷڪّ لࣹڪّ؞ عِيسَىٰ اَگْسِ ڪࣴرْ فَرِسࣹنْ بَ، دَلْدِ تࣷوگْ ݖَ ندَبْ لَ؞ ‏37 ندࣹڪࣹ سࣹنّ اَسْ ندَوَ نگَ ݖَ دࣴڪّ بَ، اَبْ مࣷيْڪَتْ لَ وࣷونْ؞ مُ يࣺگْ نَگْ نࣹ عِيسَىٰ نگَ ݖَ ڪࣴرْ فَرِسࣹنْ بَ، دِ لࣹڪّ؞ مُ دِڪّ، اِنْدَالࣹ نجَقَلْ دࣷجُ اَلْبَتَرْ لُ دࣹفْ دِوُ خࣹيݧْلُ؞ ‏38 مُ دِڪّ بَ فَ عِيسَىٰ سٗونُ، ݖَ گَنَّاوَمْ، فَ تࣵنڪِ عِيسَىٰ فࣺيتࣹ، دَلْدِ تَخَوْ، دِ جࣷويْ، رࣷنگࣷوݧْ يَ تࣷويَلْ خࣹݒّ تࣵنڪّ يَ؞ ندَوْ سَ نَگْ دِ فࣷمْݒࣹ تࣵنڪِ عِيسَىٰ ڪَوَرُ بࣷݒَّمْ گَيْ لَنگّ، دِ فٗونْ تࣵنڪّ يَاڪْ ݖࣷفࣹيلْ، دِ ݖَ سࣷتِّ دِوُ خࣹيݧْلُ گَ؞
‏39 فَرِسࣹنْ بَ وࣷو عِيسَىٰ نَگْ گِسْ لࣷولُ، نَانْ ݖِ خࣹلَمْ:‏ «ڪِي سُ دࣷونْ اَبْ يࣷنࣹنتّ، دِنَ خَمْ ندَوْ سِي دِ ڪࣷ نَبّ-نَبّࣹي ڪُ مُ دࣷونْ، اَڪْ لُ مُ دࣷونْ:‏ جِگࣺينُ مࣷيْڪَتْ!‏» ‏40 عِيسَىٰ نَگْ اࣵدُّ، نࣹ ڪࣷ:‏ «سِمࣷݝْ، دِنَا لَ وَخْ لࣹنّ؞» مُ نࣹ ڪࣷ:‏ «وَخَلْ، سࣴرِݧْ بِ؞»
‏41 عِيسَىٰ نࣹ ڪࣷ:‏ «ݧَارِ بࣷرࣷومْ بࣷرْ لَ وࣷونْ اَڪْ سࣹينُبْ لࣹبْلࣹڪَتْ؞ بࣷرُ ڪࣹنّ ڪَ دِ تࣺيمࣺيرِ جُنِّ (500000) ݖِ خَالِسْ، ڪَ ݖَ دࣹسْ فُڪِّ جُنِّ (50000)¤؞ ‏42 وَايࣹ ݧࣷومْ ݧَارْ ݧࣺݒّ، ڪࣹنّ مَنُ ݖࣹي فࣹيْ؞ مُ فَرْ جࣺگَّلْ لࣹينْ ݧࣷومْ ݧࣺݒّ؞ اَنَ ڪَنْ ݖِ ݧَارْ ݧࣷوݧُ نَگْ مࣷو ݖِيْ گࣴنَ فࣷنڪّ لࣹبْلࣹڪَتْ بِ؟» ‏43 سِمࣷݝْ نࣹ ڪࣷ:‏ «خَنَا ڪِ اࣴݒّلࣹ لُ ݧُ ڪࣷ جࣺگَّلْ؞»
عِيسَىٰ نࣹ ڪࣷ:‏ «سَبْ اࣵتّࣹ جُبْ نَ؞» ‏44 مُ وَلْبَتِڪُ ݖَ ندَوْ سَ، دَلْدِ نࣹ سِمࣷݝْ:‏ «گِسْ نگَ ندَوْ سِي دࣹ؟ دُگّ نَا سَ ڪࣴرْ، مَيࣹوࣷو ندࣷخْ مُ مَ جࣵنگࣷو؞ وَايࣹ مࣷومْ، اَيْ رࣷنگࣷوݧَمْ لَ تࣷويَلࣹ سَمَيْ تࣵنڪّ، فࣷمْݒࣹ ڪࣷ ڪَوَرُ بࣷݒَّمْ؞ ‏45 فٗونُلࣷو مَ، دَلَلࣹ مَ، وَايࣹ مࣷومْ، بَ مَ دُگّسࣹي بَ لࣺيگِ نࣷݒِّوُلَ فٗونْ سَمَيْ تࣵنڪّ؞ ‏46 اَگْ دِوْ، دِوࣹوࣷو ڪࣷ سَمَ بࣷݒّ، وَايࣹ مࣷومْ، دِوُ خࣹيݧْلُ لَ دِوْ سَمَيْ تࣵنڪّ؞ ‏47 مࣷو تَخْ مَ دِ لَ وَخْ، نࣹ بࣵڪَّارَمْ يُ بَرࣹ يِ ݧُ ڪࣷ جࣺگَّلْ مࣷو وَرَلْ فࣷنڪࣹيلَمْ گُ بَرࣹ؞ وَايࣹ نجࣺگَّلْ لُ تُوتْ، فࣷنڪࣹيلْ گُ تُوتْ رࣹڪّ؞»
‏48 عِيسَىٰ نَگْ نࣹ ندَوْ سَ:‏ «جࣺگَّلࣹيسْ نَ لَ سَيْ بࣵڪَّارْ؞» ‏49 ݖِ ڪَوْ لࣷولُ ݧَ مُ تࣷوگَنْدࣷولْ دِ لࣹڪّ، نَانْ ݖِ سࣹينْ خࣹلْ:‏ «ڪِي مࣷويْ ڪَنْ بَيْ جࣺگَّلࣹيْ بࣵڪَّارْ سَخْ؟» ‏50 عِيسَىٰ نَگْ نࣹ ندَوْ سَ:‏ «سَ نگࣴمْ مُسَلْ نَ لَ، دࣹمَلْ اَڪْ جࣵمّ؞»
