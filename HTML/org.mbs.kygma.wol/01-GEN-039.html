جَايْ نَݧُ يُوسُفَ جَوْرِݧُ فِرْعَوْنَ
39
‏1 بَ ݧُ يٗبّࣷو يُوسُفَ مِسْرَ، ݒࣷتِفَارْ، جَوْرِݧُ فِرْعَوْنَ¤، مَ جِيتࣹ اَيْ دَگَمْ، دَ ڪࣷو جࣴندّ ݖَ اِسْمَيْلࣹينْ، ݧَ ڪࣷ فَ اِنْدِ؞
‏2 ݖِ ڪَوْ لࣷولُ اَجِ سَخْ جِ اࣵندّ اَڪْ يُوسُفَ، مَيْ ڪࣷ ندَمْ، مُ دࣴڪّ ݖَ ڪࣴرْ سَنگُبْ وَا مِسْرَامْ بَ؞ ‏3 سَنگّ بَ نَگْ گِسْ نࣹ اَجِ سَخْ جَا نگِ اࣵندّ اَڪْ يُوسُفَ، دِ ڪࣷ مَيْ ندَمْ ݖِ لࣺݒّ لُ مُ سَبَبْلُ؞ ‏4 سَنگّ بَ اَمْ ݖࣷفࣹيلْ ݖِ يُوسُفَ، دࣹفْ ڪࣷ فَرَ بِيرْ ڪࣴرَمْ، دࣺنڪّ ڪࣷ سَيْتُبْ ڪࣴرْ گَ، بَ تࣹگْ ݖِيْ لࣷخࣷومْ لࣺݒّ لُ مُ مࣷومْ؞ ‏5 بࣷوبࣹي وَايِ مِسْرَ جَ دࣺنڪࣹي يُوسُفَ ڪࣴرَمْ اَڪْ لࣺݒّ لُ مُ مࣷومْ، مُ دِ ڪࣷ سَيْتُ، ݖَ لَ اَجِ سَخْ جِ دَلࣹ بَرْڪࣹيلْ ڪࣴرَمْ، بَرْڪࣹيلاَلࣹ مبࣷولࣹمْ لُ مُ مࣷومْ، ڪࣴرْ اَڪْ تࣷولْ؞ ‏6 مُ دَلْدِ بࣵيِّ لࣺݒّ ݖِ لࣷخࣷلْ يُوسُفَ، تࣹ لِ مُ فَ نࣹڪّ تَخْ بَ تࣷݒّتࣷوتُلْ دَرَ ݖِيْ مبِرَمْ، لُ مࣷيْ لِ مُيْ لࣹڪّ رࣹڪّ؞ بࣷوبَ يُوسُفَ ڪُ جࣹڪّࣷونْ بِندّ لَ تࣹ گٗورَيِوْ؞
سࣷخْنَسْ ݒࣷتِفَارْ فࣹخࣹيلْ نَ يُوسُفَ
‏7 مُ اَمْ بࣺسْ نَگْ سࣷخْنَسْ سَنگُبْ يُوسُفَ بَ خࣹمّࣹمْ يُوسُفَ، دَلْدِ نࣹ ڪࣷ:‏ «ڪَايْ تࣴدّ اَڪْ مَنْ؞» ‏8 تࣹيوُلْ يُوسُفَ بَݧْ، نࣹ سࣷخْنَسْ سَنگّ بَ:‏ «دࣺگْلُلْ، سَمَ سَنگّ دَ مَا جࣺبَّلْ لࣺݒّ، بَ تࣷݒّتࣷوتُلْ دَرَ ݖِ ڪࣴرْ گِ؞ ‏9 اࣴݒّلࣹوُ مَ سَݧْ-سَݧْ ݖِ بِيرْ ڪࣴرْ گِ، تࣹ آيࣹوُ مَ لࣹنّ خَنَا يَوْ، ندَخْ سࣷخْنَامْ نگَ؞ ڪࣷنْ لࣺيگِ نُ مَيْ مَنَ دࣹفࣹ ݧَاوْتࣺيفْ وُ رࣺيْ وُ نِ تࣷلُّ، دِ مࣷيْ يࣵلَّ؟» ‏10 بَ لࣷولُ اَمࣹي بࣺسْ بُ يࣵلَّ سࣵڪّ مُ دِ ڪࣷ وَخْ يُوسُفَ، تࣹ دࣺگْلُوُ ڪࣷ، وࣺيتُلْ اَڪْ مࣷومْ سَخْ، بَيْ تࣴدّ اَڪْ مࣷومْ؞
‏11 مُ اَمْ بࣺسْ يُوسُفَ دُگّ ݖَ ڪࣴرْ گَ، دِ دࣹفْ لِگّࣺييَمْ، تࣹ ڪࣹنّ ݖِ سُرْگَ يُ گٗورْ يَ نࣹڪُّ فَ؞ ‏12 ندَوْ سِ جࣵݒّ ݖِ مبُبّ مِ، نࣹ ڪࣷ:‏ «ڪَايْ تࣴدّ اَڪْ مَنْ؞» يُوسُفَ نَگْ وَݖّ مبُبّ مَ ݖَيْ لࣷخࣷومْ، دَوْ گࣺنّ ڪࣴرْ گَ؞ ‏13 ندَوْ سِ گِسْ مبُبّ، مِ مُ ڪࣷ وَݖّࣹ، بَ دَوْ گࣺنّ، ‏14 مُ وࣷو سُرْگَيْ ڪࣴرْ گَ، نࣹ لࣹينْ:‏ «خࣷولْ-لࣹينْ، اࣹبْرࣴ، سِي سَمَ جࣴڪّࣴرْ اِنْدِ ݖِ ڪࣴرْ گِ، مُ دِ نُ يَبْ؞ دَفَ ݧࣴوْ، بࣴگّ مَا تࣴدّࣹ، مَ يُوخُ ݖَ ڪَوْ؞ ‏15 نَڪَ لاَ يُوخُ ݖَ ڪَوْ، مُ وَݖّࣹ مَ مبُبَّمْ، دَوْ گࣺنّ؞» ‏16 بَ مُ ڪࣷ دࣹفࣹي ندَوْ سِ دࣹنݖّ مبُبّ مَ، بَ سَنگُبْ يُوسُفَ بَ ݧِبّسِ، ‏17 مُ نࣹتَّلࣹي ڪࣷ ڪࣷ نࣷونُ نࣹ ڪࣷ:‏ «جَامُ اࣹبْرࣴ، بِ نگَ اِنْدِ ݖِ ڪࣴرْ گِ دࣹ، دَ مَا فࣹڪّسِ، دِ مَ سࣵڪُّ؞ ‏18 نَڪَ لاَ يُوخُ ݖَ ڪَوْ، مُ دَوْ گࣺنّ، وَݖّࣹ مَ مبُبَّمْ؞»
تࣴجْ نَݧُ يُوسُفَ ڪَسࣷ
‏19 سَنگّ بِ دࣺگّ لِ ڪࣷ سࣷخْنَامْ وَخْ نࣹ مࣷومْ لَ ڪࣷ جَامَمْ دࣹفْ، مُ مࣹرْ لࣷولْ، ‏20 بَ جࣵݒّ يُوسُفَ، تࣴجْ ڪࣷ ڪَسࣷ، بَ ݧُيْ دࣹنݖّ ݧِ بُورْ جࣵݒّ؞ يُوسُفَ نࣹڪّ فࣷوفَ ݖَ ڪَسࣷ بَ؞
‏21 تࣹيوُلْ اَجِ سَخْ جِ اࣵندّ اَڪْ يُوسُفَ، لاَيࣹ ڪࣷ بِيرْ، دَلْدِ خِيرْ وَتُّڪَتُ ڪَسࣷ بَ ݖَ مࣷومْ، مُ يࣺوࣺنࣹ ڪࣷ؞ ‏22 ݖِ ڪَوْ لࣷولُ مُ مَيْ يُوسُفَ، مُ يِلِفْ ݧِ ݧُ تࣴجْ ݧࣺݒّ، تࣹ مبࣷولࣹمْ لُ ݧُ فَيْ دࣹفْ، مُ دِ ڪࣷ جِيتࣹ؞ ‏23 وَتُّڪَتْ بَ سࣹيتَتُلࣷونْ دَرَ لُ يُوسُفَ جِيتࣹ، ندَخْ اَجِ سَخْ جࣹي اࣵندّ اَڪْ مࣷومْ، تࣹ لُ مُ سَبَبْلُ، اَجِ سَخْ جِ مَيْ ڪࣷ ݖَ ندَمْ؞
