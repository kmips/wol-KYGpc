اَيْ گِيرْ دࣹسْ نَ ݒࣹنڪُبْ يُرْدَنْ
32
‏1 جُرْ گُ نࣹ گࣵݧّ، بَرࣹ لࣷولْ، لَ رُبࣹنࣹينْ ݧِ اَڪْ گࣵدّࣹينْ ݧِ اَمࣷونْ؞ ݧُ خࣷولاَتْ دِيوَانُ يَسࣹرْ اَڪْ دِيوَانُ گَلࣵدّ، گِسُݧُ لُ مࣷيْ اࣵلّ بُ نࣹيخَ يَرࣹگْ جُرْ؞
‏2 بَ مُ ڪࣷ دࣹفࣹي گࣵدّࣹينْ ݧَاڪْ رُبࣹنࣹينْ ݧَ دِڪّ، وَخْ اَڪْ مُوسَىٰ اَڪْ اࣹلَسَرْ سَرْخَلْڪَتْ بَ، اَڪْ ڪِلِفَيْ مبࣷولࣷو مَ؞ ݧُ نࣹ لࣹينْ:‏ ‏3 «گࣷخْ يِي دِ اَتَرࣷتْ، اَڪْ دِبࣷنْ، اَڪْ يَسࣹرْ، اَڪْ نِمْرَ، اَڪْ اࣹسْبࣷنْ، اَڪْ اࣹلَلࣹ، اَڪْ سࣹبَمْ، اَڪْ نࣹبࣷ، اَڪْ بࣹوࣷنْ، ‏4 مبࣷولࣹمْ رࣺيوْ مِي اَجِ سَخْ جِ دَانْ فِ ڪَنَمْ مبࣷولࣷومْ إِسْرَاءِيلْ دࣹ، رࣺيوْ مُ نࣹيخَ يَرࣹگْ جُرْ لَ، تࣹ نُنْ، سَنگّ بِ، بࣷرࣷومِ جُرْ لَنُ؞» ‏5 ݧُ تࣹگْ ݖَ نࣹ:‏ «ڪࣷنْ نَگْ نگَلَّ سَنگّ بِ، بَاخࣹ نُ رࣺيوْ مِي، مُيْ سُنُبْ ݖࣺرْ، تࣹ نگَ بَݧْ نࣷو جࣵلّࣹ دࣹخُ يُرْدَنْ؞»
‏6 مُوسَىٰ نࣹ گࣵدّࣹينْ ݧَاڪْ رُبࣹنࣹينْ ݧَ:‏ «خَنَا سࣹينِ بࣷڪّ دُݧُ خَرࣹجِ، يࣹينْ نگࣹينْ تࣷوگْ فِي؟ ‏7 دَنگࣹينَ نَرَ يࣷقِلࣷو بࣵنِّ إِسْرَاءِيلْ، بَ تࣹرࣹ لࣹينَ جࣵلِّ ݖَ رࣺيوْ، مَ لࣹينْ اَجِ سَخْ جِ جࣷخْ؟ ‏8 لُ مࣹلْ نِي دࣹ، لَ سࣹينِ بَايْ دࣹفࣷونْ بَ مَ لࣹينْ يࣹبَلࣹي، ݧُ جٗگࣹ ڪَدࣹسْ بَرْنࣹيَ، نگِرْ يࣴرِ رࣺيوْ مَ؞ ‏9 دَݧࣷو دࣹمْ بَ اࣵگّ خُرُ اࣹسْڪࣷلْ، بَ گِسْ رࣺيوْ مَ، دࣷورَ دࣺلْسِ، دِ يࣷقِلࣷو بࣵنِّ إِسْرَاءِيلْ، نگِرْ ݧُ بَݧَ دُگِّ ݖَ رࣺيوْ، مَ لࣹينْ اَجِ سَخْ جِ جࣷخࣷونْ؞ ‏10 سࣵنجُمْ اَجِ سَخْ جِ تࣵنگّ نَ بࣺسُبْ ڪࣹرࣷوگْ، بَ مُ گِݧْ، نࣹ:‏ ‏11 ‹ڪࣹنّ دُ گِسْ مُڪّ رࣺيوْ مَ مَ گِݧَلࣷونْ إِبْرَاهِيمَ اَڪْ إِسْحَاقَ اَڪْ يَعْقُوبَ، ݖِ گٗورْ ݧِ جٗگࣹ مِسْرَ تࣹ اَمْ ݧَارْ فُڪِّ اَتْ، جࣴمْ ڪَوْ، گَنَّاوْ مَتَلُݧُ ڪٗولُتࣹ ݖِ مَنْ؞ ‏12 خَنَا ڪَلࣹبْ دࣷومُ يࣹفُنࣹ، ڪࣹنࣹسࣹينْ بِ، اَڪْ يࣷسُوࣹ دࣷومُ نُونْ، ندَخْ ݧࣷو مَتَلْ ڪٗولُتࣹ ݖِ اَجِ سَخْ جِ؞› ‏13 بَ لࣷولُ اَمࣹي سࣵنجُمْ اَجِ سَخْ جِ تࣵڪَّلْ إِسْرَاءِيلْ، مُ وࣴرْلࣷو لࣹينْ ݖَ مࣵنْدِݝْ مَ دِيرُ ݧࣹينتّ فُڪِّ اَتْ، بَ مبࣷولࣹمْ مَاسْ گࣷوگَ دࣹفࣷونْ لُ اَجِ سَخْ جِ ݧَاوْلُ، ݧࣺݒّ جࣹيخْ تࣵڪّ؞ ‏14 يࣹينْ نَگْ دࣹفُلࣹينْ لُ مࣷيْ جٗگْ وُوتُ سࣹينِ بَايْ، نجُرُمْ بࣵڪَّارْڪَتْ يِ نگࣹينْ دࣷونْ، دِ خَمبّ سࣵنجُمْ اَجِ سَخْ، جِ تࣵڪَّلْ بࣵنِّ إِسْرَاءِيلْ؟ ‏15 سُ نگࣹينْ ڪࣷ دࣴدّࣷو اِتْ، دَيْ دࣹلُّ وَݖّ بࣵنِّ إِسْرَاءِيلْ ݖِ مࣵنْدِݝْ مِ رࣹڪّ، نگࣹينْ دَلْدِ سࣵنڪّلُ مِي مبࣷولࣷو مࣺݒّ؞»
‏16 بَ لࣷولُ وࣹيسࣹي، ݧُ دِڪَّاتْ، نࣹ ڪࣷ:‏ «اَيْ گࣺتّ لَنُ بࣴگَّ سࣵڪّ فِي نگِرْ سُنُيْ گࣵتّ، اَڪِ دࣴڪّ نگِرْ سُنُيْ نجَبࣷوتْ؞ ‏17 وَايࣹ نُنْ ڪَيْ دَنُيْ دَلْدِ گࣵنَّايُ، جِيتُ بࣵنِّ إِسْرَاءِيلْ، بَ ڪࣹرࣷوگْ ݧُ لࣹينْ يٗبّࣷو ݖَ سࣹينْ سُوفْ؞ سُنُ نجَبࣷوتَ نُيْ خَارࣹ فِي، بِيرْ دࣴڪّ يُ تَتَ وࣴرْ، نگِرْ ݧُ مُݖّ ݖِ وَا رࣺيوْ مِ؞ ‏18 دُنُ دࣹلُّ مُڪّ سُنُيْ ڪࣴرْ، بَ ڪࣹرࣷوگْ بࣵنِّ إِسْرَاءِيلْ دِ سࣺدُّ، ڪُ ݖِ نࣹڪّ جࣷتْ ݖࣺرُ سُوفَمْ؞ ‏19 وَايࣹ نُنْ دُنُ بࣷڪّ اَڪْ ݧࣷومْ اَبْ ݖࣺرُ سُوفْ ݖَ وࣵلاَّ دࣹخُ يُرْدَنْ، اَڪْ لَ ݖَ تࣷݒّ، ندَخْ دَيْ فࣹڪّ نُ جࣷتْ سُنُ ݖࣺرُ سُوفْ، ݖِ وࣹتْ گِي فࣺيتࣹ دࣹخُ يُرْدَنْ ݒࣹنڪُ؞»
‏20 مُوسَىٰ نࣹ لࣹينْ:‏ «سُ نگࣹينْ ڪࣷ دࣹفࣹي نࣷونُ، گࣵنَّايُ فِ ڪَنَمْ اَجِ سَخْ جِ نگِرْ خَرࣹجِ، ‏21 سُ سࣹينْ مبࣷولࣹمْ بࣷرࣷومِ گࣵنَّايْ جࣵلّࣹي دࣹخُ يُرْدَنْ فِ ڪَنَمْ اَجِ سَخْ جِ، بَ ڪࣹرࣷوگْ مُ دࣵقࣹي اَيْ نࣷونَمْ، ‏22 بَ رࣺيوْ مَ نَنگُلْ اَجِ سَخْ جِ، سُ بࣷوبَا دࣺيْ، سَݧْ نگࣹينَ دࣺلْسِ، تࣹ وࣵݖّࣷو نگࣹينْ اَڪْ اَجِ سَخْ جِ، وࣵݖّࣷو اَڪْ بࣵنِّ إِسْرَاءِيلْ، تࣹ سُ بࣷوبَا اِتْ رࣺيوْ مِي ݧࣹيلْ لࣹينْ، دِ سࣹينْ مࣷومࣹيلْ فِ ڪَنَمْ اَجِ سَخْ جِ؞ ‏23 وَايࣹ سُ نگࣹينْ جࣴفࣹوُلْ نࣷونُ ڪَتْ، يࣹينَ مࣷيْ اَجِ سَخْ جِ، تࣹ خَمْلࣹينْ نࣹ سࣹينُگْ مࣷيْ دِنَ لࣹينْ دَبْ؞ ‏24 ڪࣷنْ نَگْ سࣵڪّلࣹينْ اَيْ دࣴڪّ نگِرْ سࣹينِ نجَبࣷوتْ، اَڪِ گࣺتّ نگِرْ سࣹينِ گࣵتّ، وَايࣹ لِ نگࣹينْ دِگࣹ دَالْ، دࣹفْلࣹينْ ڪࣷ؞»
‏25 بَ لࣷولُ اَمࣹي گࣵدّࣹينْ ݧَاڪْ رُبࣹنࣹينْ ݧَ نࣹ مُوسَىٰ:‏ «سَنگّ بِ، نࣷونُ نگَ ڪࣷ سَنْتَانࣹي، نِ لَنُ ڪࣷيْ دࣹفࣹ؞ ‏26 سُنُيْ گࣷنࣹيڪْ سُنُيْ جَبَرْ، اَڪْ سُنُيْ گࣵتّ، اَڪْ سُنُ جُرْ گࣺݒّ فِي ݖِ دࣴڪِّ گَلࣵدّ لَيْ دࣹسْ؞ ‏27 وَايࣹ نُنْ، مبࣷولࣹمْ ݧِ يࣷرْ نگࣵنَّايْ دَنُيْ جࣵلّ دࣹخْ گِ فِ ڪَنَمْ اَجِ سَخْ جِ، نگِرْ خَرࣹجِ نِ نگَ ڪࣷ وَخࣹ، سَنگّ بِ؞»
‏28 مُوسَىٰ نَگْ سَنْتَانࣹ ݖِ سࣹينْ مبِرْ ݖَ اࣹلَسَرْ سَرْخَلْڪَتْ بَ، اَڪْ يࣷسُوࣹ دࣷومُ نُونْ، اَڪْ ڪِلِفَيْ بࣵنِّ إِسْرَاءِيلْ، يَ جِيتࣹ سࣹينِ ڪࣴرْ مَامْ؞ ‏29 مُوسَىٰ نࣹ لࣹينْ:‏ «سُ رُبࣹنࣹينْ ݧࣹيڪْ گࣵدّࣹينْ ݧِ اࣵنْدࣹيڪْ يࣹينْ، ݧࣷومْ ݧࣺݒّ گࣵنَّايُ نگِرْ خَرࣹجِ فِ ڪَنَمْ اَجِ سَخْ جِ، بَ رࣺيوْ مَ نَنگُلْ لࣹينْ، نَنگࣹينْ لࣹينْ جࣷخْ رࣺيوُمْ گَلࣵدّ، مُيْ سࣹينْ مࣷومࣹيلْ؞ ‏30 وَايࣹ سُ ݧُ اࣵنْدُلْ اَڪْ يࣹينْ، گࣵنَّايُ جࣵلّ، ݖِ سࣹينْ بِيرْ لَݧُيْ سࣺدُّ سُوفْ، ݖَ بِيرْ رࣺيوُمْ ڪَنَانْ؞»
‏31 بَ لࣷولُ اَمࣹي گࣵدّࣹينْ ݧَ اَڪْ رُبࣹنࣹينْ ݧَ دࣹلُّ نࣹ:‏ «سَنگّ بِ، لِ نُ اَجِ سَخْ جِ وَخْ رࣹڪّ لَنُيْ دࣹفْ؞ ‏32 نࣷويْ گࣵنَّايُ جࣵلِّ رࣺيوُمْ ڪَنَانْ فِ ڪَنَمْ اَجِ سَخْ جِ، وَايࣹ نُنْ، ݖࣺرْ بِ نُيْ مࣷومْ، وࣵلُّ دࣹخُ يُرْدَنْ وِي لَيْ دࣷونْ؞»
‏33 مُوسَىٰ نَگْ تُدّࣹ ݖࣺرْ يِي، اَسْڪَنُ گࣵدّ، اَڪْ اَسْڪَنُ رُبࣹنْ، اَڪْ گࣹنّ-وࣵلُّ گِيرُگْ مَنَسࣹ دࣷومُ يُوسُفَ:‏ نگُورُگْ سِوࣷنْ بُورُبْ اَمࣷرࣹينْ ݧِ لَ لࣹينْ جࣷخْ، اَڪْ نگُورُگْ اࣷگْ بُورُ بَسَنْ، رࣺيوْ مَ مࣺݒّ، بࣷولࣹ ݖَ دࣴڪّ يُ مَگْ يَ، اَڪْ اࣵلّ يَ ڪࣷ وࣴرْ؞ ‏34 بَ مُ ڪࣷ دࣹفࣹي گࣵدّࣹينْ ݧَ تَبَخْ دِبࣷنْ اَڪْ اَتَرࣷتْ اَڪْ اَرࣷوࣹرْ، ‏35 اَڪْ اَتࣹرࣷتْ سࣷفَنْ، اَڪْ يَسࣹرْ اَڪْ يࣷگْبࣷوَ، ‏36 اَڪْ بࣹتْ نِمْرَ، اَڪْ بࣹتْ اَرَنْ، لࣺݒّ دِ دࣴڪّ يُ تَتَ وࣴرْ؞ ݧُ سࣵڪَّالࣹ اَيْ وُرْمبَلْ نگِرْ سࣹينِ گࣵتّ؞ ‏37 رُبࣹنࣹينْ ݧَ اِتْ تَبَخْ اࣹسْبࣷنْ اَڪْ اࣹلَلࣹ اَڪْ ڪِرْيَاتَيِمْ، ‏38 اَڪْ نࣹبࣷ، اَڪْ بَالْ مࣹوࣷنْ، لࣹنّ ݖَ تُرِ دࣴڪّ يَ سࣷݒِّڪُ، اَڪْ سِبْمَ؞ ݧُ تُدّࣹ دࣴڪّ يࣷويُ ݧُ تَبَخَاتْ، سࣹينْ تُرِ بࣷݒّ؞
‏39 ݖِ ڪَوْ لࣷولُ ݧِ اَسْڪَنࣷو ݖِ مَڪِرْ دࣷومُ مَنَسࣹ، سࣷنگِ بࣴجْ-گࣵنَّارُ گَلࣵدّ، نَنگُ ڪࣷ، دࣵقࣹ فَ اَمࣷرࣹينْ ݧَ؞ ‏40 بَ مُ ڪࣷ دࣹفࣹي مُوسَىٰ جࣷخْ گَلࣵدّ، اَسْڪَنُ مَڪِرْ، ݧُ دࣴڪّࣹ؞ ‏41 ݖِ بِيرْ لࣷولُ اَسْڪَنُ يَيِرْ، مِ اَسْڪَنࣷو ݖِ مَنَسࣹ، دࣹمْ نَنگُجِ يࣹنࣹينْ دࣴڪِّ گَلࣵدّ يُ ندَوْ؞ ݧُ تُدّࣹ لࣹينْ اَوࣷتْ يَيِرْ؞ ‏42 نࣷبَ اِتْ نَنگُجِ ڪࣹنَتْ اَڪْ دࣴڪّ يَ ڪࣷ وࣴرْ، دَلْدِ تُدّࣹ گࣷخْ بَ بࣷݒَّمْ، مُيْ نࣷبَ؞
