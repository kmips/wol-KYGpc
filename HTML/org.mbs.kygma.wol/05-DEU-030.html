دࣹلُّسِ ݖِ اَجِ سَخْ جِ، بَرْڪࣹيلْ
30
‏1 اࣴلّࣴگْ نَگْ، سُ مبࣷولࣹمْ مبِرْ يࣷويُ دِڪّࣹي فِ سࣹينْ ڪَوْ، مُيْ بَرْڪࣹ يِ، دِ اَلْڪࣵنْدࣹ يِ مَ تࣹگْ فِ سࣹينْ ڪَنَمْ، سُ بࣷوبَا بُ نگࣹينْ دࣺلّࣷوسࣹي سࣹينْ خࣹلْ فࣷوفَ مُ لࣹينْ فࣹڪّ، ݖَ بِيرْ مبࣷولࣹمْ خࣹيتْ يَ لࣹينْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ دࣵقْ جࣴمࣹ، ‏2 بَ نگࣹينْ دࣹلُّسِ ݖِ سࣹينْ يࣵلَّ اَجِ سَخْ جِ، دِ ڪࣷ دࣺگَّلْ ݖِ مبࣷولࣹمْ نِ مَ لࣹينْ ڪࣷ سَنْتࣹ بࣺسُبْ تࣹيْ، يࣹينْ اَڪْ سࣹينِ دࣷومْ، تࣹ نگࣹينْ دࣹفࣹ ڪࣷ سࣹينْ لࣺݒِّ خࣷلْ اَڪْ سࣹينْ لࣺݒِّ بَڪَّنْ، ‏3 سُ بࣷوبَا اَجِ سَخْ جࣹييْ تِجِّ سࣹينْ وࣴرْسࣴگْ، يࣴرࣴمْ لࣹينْ؞ سࣹينْ يࣵلَّ اَجِ سَخْ جِ مࣷو لࣹينْ دِ دࣹلُّ دَجَلࣹي فࣷوفَ، ݖِ بِيرْ مبࣷولࣹمْ خࣹيتْ يَ مُ لࣹينْ تَسَارࣹ وࣷونْ ݖَ سࣹينْ بِيرْ؞ ‏4 سُ ݧُ لࣹينْ تࣷخَلࣷونْ بَ ݖَ ݖَتِ اَسَمَانْ سَخْ، فࣷوفَ لَ لࣹينْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ دِ دَجَلࣹي، تࣹ فَ لَ لࣹينْ دِ جࣴلࣹ؞ ‏5 سࣹينْ يࣵلَّ اَجِ سَخْ جࣹي لࣹينْ دِ دࣹلّࣷو ݖَ رࣺيوْ مَ سࣹينْ مَامْ مࣷومࣷونْ، نگࣹينْ مࣷومْ ڪࣷ، تࣹ مࣷو لࣹينْ دِ بَاخࣹ، يࣷڪّ لࣹينْ بَ نگࣹينْ گࣴنَ بَرࣹ سࣹينِ مَامْ؞ ‏6 سُ بࣷوبَا سࣹينْ يࣵلَّ اَجِ سَخْ جࣹي لࣹينْ دِ خَرْفَلْ خَرَفُ خࣷلْ، تࣹ مࣷويْ خَرْفَلْ خَرَفُ خࣷلْ، سࣹينْ اَسْڪَنْ، نگِرْ نگࣹينْ سࣷݒّࣹ سࣹينْ يࣵلَّ اَجِ سَخْ جِ سࣹينْ لࣺݒِّ خࣷلْ، اَڪْ سࣹينْ لࣺݒِّ بَڪَّنْ، ندَخْ نگࣹينْ دُندّ؞ ‏7 سُ لࣷولُ اَمࣹي سࣹينْ يࣵلَّ اَجِ سَخْ جࣹييْ تࣹگْ مبࣷولࣹمْ اَلْڪࣵنْدࣹ يِي مَ لࣹينْ لِمَلْ، ݖِ ڪَوْ سࣹينِ نࣷونْ اَڪْ سࣹينْ بَݧْ يِ لࣹينْ تࣷݝَلْ؞ ‏8 يࣹينْ نَگْ دِنگࣹينْ وَلْبَتِڪُ، دِ دࣺگَّلْ اَجِ سَخْ جِ، تࣹيْ جࣴفࣹ مبࣷولࣹمْ سَنْتَانࣹ يِ مَ لࣹينْ دِ دࣺنڪّ بࣺسُبْ تࣹيْ جِي؞ ‏9 ݖِ ڪَوْ لࣷولُ سࣹينْ يࣵلَّ اَجِ سَخْ جِ مࣷويْ بَاخَلْ مبࣷولࣹمْ سࣹينُ ݧَقْ، اَڪْ سࣹينْ مࣹݧّࣹيفُمْ نجُرُڪَايْ، اَڪْ سࣹينْ مࣹݧّࣹيفُمْ جُرْ، اَڪْ سࣹينْ مࣹݧّࣹيفُمْ سُوفْ، نگِرْ لُ بَاخْ؞ اَجِ سَخْ جِ مࣷويْ بࣺگࣹوَاتْ دِ لࣹينْ بَاخࣹ نَ مُ دَانْ بࣺگࣹي بَاخࣹ سࣹينِ مَامْ؞ ‏10 ندࣹگَمْ دࣺگَّلْ نگࣹينْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ، دِ تࣷݒّ اَيْ سَنْتَانࣹيمْ، اَڪْ دࣷگَلِ يࣷونَمْ، يِ ݧُ بِندّ ݖِ تࣺيرࣹبْ يࣷونْ وِي؛ ندࣹگَمْ يࣹينَ وَلْبَتِڪࣷو سࣹينْ لࣺݒِّ خࣷلْ اَڪْ سࣹينْ لࣺݒِّ بَڪَّنْ، بَ دࣹلُّ ݖِ سࣹينْ يࣵلَّ اَجِ سَخْ جِ؞
مُوسَىٰ نࣹي دࣺگَّلْ يࣵلَّ يࣷمبّ نَ
‏11 سَنْتَانࣹ بِي مَ لࣹينْ دِ دࣺنڪّ تࣹيْ نَگْ، جَفࣹوُلْ تࣹ سࣷرࣹوُلْ بَ تࣴ لࣹينْ، ‏12 اَجࣹوُلْ ڪَوْ اَسَمَانْ بَ نگࣹينْ نَانْ:‏ «اَنَ ڪُ نُيْ يࣺيگَلِ ڪَوْ اَسَمَانْ، اࣵجِّلْ نُ ڪࣷ، دࣺگْتَلْ نُ ڪࣷ، بَ نُ جࣴفࣹ ڪࣷ؟» ‏13 نࣹڪُّلْ اِتْ وࣵلاَّ گࣺيجْ بَ نگࣹينْ نَانْ:‏ «اَنَ ڪُ نُيْ جࣵلِّلْ وࣵلاَّ گࣺيجْ، جࣴلِلْ نُ ڪࣷ، دࣺگْتَلْ نُ ڪࣷ، بَ نُ جࣴفࣹ ڪࣷ؟» ‏14 ڪࣵدُّ گِ ڪَيْ جࣹگࣹ نَ لࣹينْ لࣷولْ؛ ݖِ سࣹينْ گࣺمِّݧْ اَڪْ ݖِ سࣹينْ بِيرْ خࣷلْ لَ نࣹڪّ، نگِرْ نگࣹينْ دِ ڪࣷ جࣴفࣹ؞
‏15 ڪࣷنْ نَگْ گِسْ نگࣹينْ ڪࣷ:‏ تࣹگْ نَا فِ سࣹينْ ڪَنَمْ، بࣺسْ نِڪِ تࣹيْ، دُندّ اَڪْ مبَاخْ، تࣹگْ نَا فِ سࣹينْ ڪَنَمْ، ندࣹي اَڪْ لࣷرَانگࣹ؞ ‏16 لِ مَ لࣹينْ سَنتّ، مَنْ، بࣺسُبْ تࣹيْ، مࣷويْ نگࣹينْ سࣷݒّ سࣹينْ يࣵلَّ اَجِ سَخْ جِ، دِ تࣷݒّ اَيْ يࣷونَمْ، دِ سࣵمّ اَيْ سَنْتَانࣹيمْ، اَڪْ دࣷگَلِ يࣷونَمْ، اَڪِ اࣵتّࣹيمْ، نگِرْ نگࣹينْ دُندّ تࣹ يࣷڪُّ، سࣹينْ يࣵلَّ اَجِ سَخْ جِ بَرْڪࣹيلْ لࣹينْ ݖَ رࣺيوْ مَ نگࣹينْ جࣴمْ، نگِرْ نَنگُجِ ڪࣷ؞ ‏17 وَايࣹ سُ نگࣹينْ وَلْبَتࣹي سࣹينْ خࣷلْ، دࣺگْلُوُلࣹينْ، خَنَا نگࣹينْ لَجّ، دِ سُجّٗوتَلْ يࣹنࣹينْ يࣵلَّ، دِ لࣹينْ جَامُ، ‏18 مَ وَخْ لࣹينْ ڪࣷ بࣺسْ نِڪِ تࣹيْ، سُ بࣷوبَا يࣹينَيْ سࣵنڪࣷو سࣵنڪُ، تࣹ دُنگࣹينْ گُدّ فَنْ ݖَ رࣺيوْ مَ نگࣹينْ دِ جࣵلّ دࣹخُ يُرْدَنْ، جࣴمْ فَ، نگِرْ نَنگُجِ ڪࣷ؞
‏19 وࣷو نَا سࣹيدࣹ اَسَمَانْ اَڪْ سُوفْ ݖِ يࣹينْ، بࣺسُبْ تࣹيْ:‏ دُندّ اَڪْ ندࣹي لاَ تࣹگْ فِ سࣹينْ ڪَنَمْ، اَڪْ بَرْڪࣹ اَڪْ اَلْڪࣵنْدࣹ؞ تَامُلࣹينْ دُندّ ندَخْ نگࣹينْ دُندّ، يࣹينْ اَڪْ سࣹينْ اَسْڪَنْ؞ ‏20 سࣷݒّلࣹينْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ، دࣺگَّلْ ڪࣷ تࣹ تَقْ ݖِ مࣷومْ، ندَخْ مࣷومْ مࣷويْ سࣹينُگْ دُندّ، دِ سࣹينْ گُدّ-گِ-فَنْ ݖَ رࣺيوْ مَ مُ گِݧَلࣷونْ سࣹينِ مَامْ إِبْرَاهِيمَ اَڪْ إِسْحَاقَ اَڪْ يَعْقُوبَ، نࣹ ݧࣷومْ لَ ڪࣷيْ جࣷخْ؞
