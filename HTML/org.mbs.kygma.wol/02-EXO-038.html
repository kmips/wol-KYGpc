دࣹفَرْ نَݧُ سَرْخَلُڪَايْ بِ اَڪْ مبَلْڪَمْ نجࣵݒّ مِ
38
‏1 بَ تࣹيْ بࣹسَلࣹلْ سࣵڪّ ݖِ بَنْتُ سࣺنگّ سَرْخَلُڪَايُ رࣹنْدِ-دٗومَلْ بُ ڪَارࣹ، گُدَّايْ بِ دِ جُرٗومِ خَسَبْ، يَاتُوَايْ بِ دِ جُرٗومِ خَسَبْ، تَخَوَايْ بِ دِ ݧࣹتِّ خَسَبْ؞ ‏2 مُ سࣵڪَّلْ ڪࣷ اَيْ بࣺجّࣺنَمْ ݖِ ݧࣹينْتِ ݖࣷلَّمْ، مُ اࣵندّ اَڪِ بࣺجّࣺنَمْ، دِ بࣹنّ؞ مُ خࣷوبْ نَگْ لࣺݒّ خࣵنجَرْ؞ ‏3 مُ سࣵڪّ جُمْتُڪَايِ سَرْخَلُڪَايْ بِ يࣺݒّ:‏ ندَبِ تِبُّڪَايُ دٗومْ يِ اَڪْ ݧِيتُڪَايْ يِ اَڪْ ڪࣴلّ يِ، اَڪْ لࣷنڪُڪَايِ يࣵݒّ يِ، اَڪْ اَندّ يِ، جُمْتُڪَايْ يࣺݒّ دِ خࣵنجَرْ؞ ‏4 مُ سࣵڪَّلْ سَرْخَلُڪَايْ بِ ݖَاخْ بُ ݧُ تࣴگّࣹ خࣵنجَرْ، مُ رࣷنُ ݒࣹگُّ سَرْخَلُڪَايْ بِ، دَلࣹ ݖِ سُوفْ بَ ݖِ دِگّ سَرْخَلُڪَايْ بِ؞ ‏5 مُ دَلْدِ ڪࣷيْ مٗولَلْ ݧࣹينْتِ نࣷݒّ ݖِ ݧࣹينْتِ ݖࣷلِّ ݖَاخُ خࣵنجَرْ بِ، ݧُيْ تࣺيࣹ نجࣵݒُّ يِ؞ ‏6 مُ سࣵڪّ نجࣵݒُّيْ بَنْتُ سࣺنگّ، خࣷوبْ لُ ݖِ نࣹڪّ خࣵنجَرْ، ‏7 دَلْدِ رࣷوفْ نجࣵݒُّ يِ ݖِ نࣷݒّ يِ ݖِ وࣹتِ سَرْخَلُڪَايْ بِ، نگِرْ بُ ݧُ ڪࣷيْ يٗبُّ؞ اَمْ ݒَخْ، مُ خࣵنقّ يُ تࣵݒَّنْدَارْ اُبْ، مࣷومْ لَ دࣹفْ سَرْخَلُڪَايْ بِ؞ ‏8 مُ سࣵڪّ اِتَمْ مبَلْڪَمْ نجࣵݒّ مُ اࣵندّ اَڪْ تࣹگࣷومْ، دِ خࣵنجَرْ جُ جٗگࣹ ݖِ سࣹيتُيْ جِگࣺينْ¤، ݧِ دࣷونْ لِگّࣺييْ ݖِ بُنتّ خَيْمَبْ ندَجࣹ مِ؞
دࣹفَرْ نَݧُ اࣴتُّ جَامࣷوڪَايْ بِ
‏9 بࣹسَلࣹلْ نَگْ سࣵڪّ وُرْمبَلُ جَامࣷوڪَايْ بِ؞ وࣹتَمْ، گِ فَرࣹ بࣴجْ-سَالُمْ دِ تࣺيمࣺيرِ خَسَبِ سࣷرِ رَبَلْ يُ ݧُ رࣵبّࣹ وࣴݧُ لࣹي، ‏10 سࣹينْ ݧَارْ فُڪِّ جࣴنْ اࣵندّ اَڪْ سࣹينْ ݧَارْ فُڪِّ تࣹگُ، دِ خࣵنجَرْ، لࣷنڪُ يࣹيڪْ گَلَنِ جࣴنْ يِ دِ خَالِسْ؞
‏11 وࣹتُ بࣴجْ-گࣵنَّارْ اِتْ دِ تࣺيمࣺيرِ خَسَبِ سࣷرِ رَبَلْ، سࣹينْ ݧَارْ فُڪِّ جࣴنْ اࣵندّ اَڪْ سࣹينْ ݧَارْ فُڪِّ تࣹگُ، دِ خࣵنجَرْ، لࣷنڪُ يࣹيڪْ گَلَنِ جࣴنْ يِ دِ خَالِسْ؞
‏12 وࣹتُ سࣷوُ دِ جُرٗومْ فُڪِّ خَسَبِ سࣷرِ رَبَلْ، اَڪْ سࣹينْ فُڪِّ جࣴنْ يُ اࣵندّ اَڪْ سࣹينْ فُڪِّ تࣹگُ، لࣷنڪُ يࣹيڪْ گَلَنِ جࣴنْ يِ دِ خَالِسْ؞
‏13 وࣹتُ ݒࣹنڪُ بِ ݖِ ڪَنَمْ دِ جُرٗومْ فُڪِّ خَسَبْ؛ ‏14‏-15 فُڪِّ خَسَبِ سࣷرِ رَبَلْ اَڪْ جُرٗومْ فِ لࣵنگّ اَڪْ بُنْتُ اࣴتّ بِ ݖِ گࣹنّ وࣹتَمْ، اَڪْ سࣹينْ ݧࣹتِّ جࣴنْ يُ اࣵندّ اَڪْ سࣹينْ ݧࣹتِّ تࣹگُ، اَڪْ فُڪِّ خَسَبِ سࣷرِ رَبَلْ اَڪْ جُرٗومْ، وࣹتْ گِ ݖِ دࣹسْ، اَڪْ سࣹينْ ݧࣹتِّ جࣴنْ يُ اࣵندّ اَڪْ سࣹينْ ݧࣹتِّ تࣹگُ؞
‏16 مبࣷولࣹمْ سࣷرِ وُرْمبَلُ اࣴتّ بِ بَ مُ دَجْ، وࣴݧُ لࣹي لَ؞
‏17 تࣹگُيْ جࣴنْ يِ خࣵنجَرْ لَ، وࣺڪُّڪَايِ جࣴنْ يࣹيڪْ اَيْ گَلَنَمْ دِ خَالِسْ، ݧُ خࣷوبْ بࣷݒِّ جࣴنْ يِ خَالِسْ؛ جࣴنِ اࣴتّ بِ يࣺݒّ نَگْ، گَلَنِ خَالِسْ لَݧُ لࣹينْ لࣴڪّلࣹي؞ ‏18 لَفُ بُنْتُ اࣴتّ بِ، لِگّࣺييُ رࣵبّ بُ رَفࣹتْ لَ، بُ ݧُ رࣵبّࣹ وࣴݧْ گُ بَخَ تࣹ لاَلْ يࣷلࣹتْ، اَڪْ وࣴݧْ گُ خࣹوَرْ، اَڪْ گُ خࣷنقّ ݖُرّ اَڪْ وࣴݧُ لࣹي؞ گُدَّايُ لَفْ بِ ݧَارْ فُڪِّ خَسَبْ لَ، تَخَوَايْ بِ يࣹمࣷوڪْ يَاتُوَايْ بِ، دِ جُرٗومِ خَسَبْ، تࣷلّࣷوڪْ سࣷرِ وُرْمبَلُ اࣴتّ بِ؞ ‏19 جࣴنِ لَفُ بُنتّ بِ ݧࣹينتّ لَ، سࣹينْ ݧࣹينْتِ تࣹگُ دِ خࣵنجَرْ، سࣹينِ وࣺڪُّڪَايْ دِ خَالِسْ، ݧُ خࣷوبْ ڪَوْ جࣴنْ يِ خَالِسْ، گَلَنْ يِ دِ خَالِسْ؞ ‏20 مبࣷولࣹمْ خࣹرِ جَامࣷوڪَايْ بِ اَڪْ خࣹرِ وُرْمبَلُ اࣴتّ بِ بَ مُ دَجْ، لࣺݒّ خࣵنجَرْ لَ؞
لِمْ نَݧُ جُمْتُڪَايْ يِ
‏21 لِي نَگْ مࣷو دِ تࣴنڪُبْ دَيࣷيْ گࣵنجَرْ يِ ݧُ جࣴفَنْدِڪࣷو ݖِ لِگّࣺييُ جَامࣷوڪَايْ بِ، جَامࣷوڪَايْ بࣷوبُ اࣵلُّوَيْ سࣹيدࣹ سِ دࣹنݖࣹ؞ تࣴنڪࣹيسْ نَ ڪࣷ ݖِ ندِگَلُ مُوسَىٰ، لࣹوࣹينْ ݧِ سَسࣷو تࣴنڪّ بِ ݖِ نجِيتَلْ اِتَمَرْ دࣷومُ هَـٰرُونَ سَرْخَلْڪَتْ بِ؞ ‏22 فࣹڪّ نَ يُدࣹينْ بِ بࣹسَلࣹلْ دࣷومُ اُرِ، مِيْ دࣷومُ اُرْ، سࣷتَّلْ نَ مبࣷولࣹمْ لِ اَجِ سَخْ جِ سَنْتࣷونْ مُوسَىٰ؞ ‏23 مُ اࣵنْدࣷونْ ݖِ اَڪْ دَنࣹينْ بِ، دࣷومُ اَيِسَمَاگْ، اࣷلِيَبْ؞ مُ دِبْ يࣹتّڪَتْ، دِبْ تَارَلْڪَتْ تࣹ دِ لِگّࣺييْڪَتُ وࣴݧْ گُ بَخَ تࣹ لاَلْ يࣷلࣹتْ، اَڪْ وࣴݧْ گُ خࣹوَرْ اَڪْ گُ خࣷنقّ ݖُرّ اَڪْ وࣴݧُ لࣹي؞
‏24 وُرُسْ وِ دࣹمْ ݖِ لِگّࣺييْ بِ بࣺݒّ، مبࣷولࣹمْ لِگّࣺييُ بࣺرَبْ بُ سࣹلّ بِ، تࣹ مُيْ وُرُسْ وُ ݧُ دࣹفْ سَرَخُ يࣺڪَّتِ-جࣺبَّلْ يࣵلَّ، جُرٗومْ ݧࣹتِّ تࣺيمࣺيرِ ڪِلࣷ اَڪْ جُرٗومْ ݧَارْ فُڪّ اَڪْ جُرٗومْ ݧَارْ لَ (877)، اَڪْ ݧࣹتِّ تࣺيمࣺيرِ گَرَامْ (300)، يࣹمࣷوڪْ ݧَارْ فُڪِّ تَلَݝْ اَڪْ جُرٗومْ ݧࣹينتّ، اَڪْ جُرٗومْ ݧَارْ تࣺيمࣺيرْ اَڪْ فَنْوࣹيرِ (730) سِيڪَلْ ݖِ سِيڪَلْ بِ ݧُ يࣷونَلْ ݖِ بࣺرَبْ بُ سࣹلّ بِ؞ ‏25 خَالِسْ بِ تُڪّࣹي ݖِ لِمࣹيفُ اَسْڪَنْ وِ، ݧࣹتِّ جُنِّيْ ڪِلࣷ اَڪْ فُڪّ اَڪْ جُرٗومْ ݧَارْ (3017)، اَڪْ جُرٗومْ ݧَارْ تࣺيمࣺيرِ گَرَامْ اَڪْ جُرٗومْ فُڪّ (750)، يࣹمࣷوڪْ تࣺيمࣺيرِ (100) تَلَݝْ اَڪْ جُنّࣹيڪْ جُرٗومْ ݧَارْ تࣺيمࣺيرْ اَڪْ جُرٗومْ ݧَارْ فُڪّ اَڪْ جُرٗومِ (1775) سِيڪَلْ، ݖِ سِيڪَلْ بِ ݧُ يࣷونَلْ ݖِ بࣺرَبْ بُ سࣹلّ بِ، ‏26 مُيْ جُرٗومِ گَرَامِ خَالِسْ بࣷݒّ بُ نࣹڪّ، يࣹمࣷوڪْ گࣹنّ-وࣵلُّ سِيڪَلْ، ݖِ سِيڪَلْ بِ ݧُ يࣷونَلْ ݖِ بࣺرَبْ بُ سࣹلّ بِ، ݧࣹيلْ ڪࣺݒّ ڪُ بِنْدُ ݖِ لِمࣹيفْ بِ، دَلࣹ ڪࣷ ݖِ ݧِ اَمْ ݧَارْ فُڪِّ اَتْ جࣴمْ ڪَوْ، ݧࣺݒّ دِ جُرٗومْ بࣹنِّ تࣺيمࣺيرِ جُنِّيْ گٗورْ اَڪْ ݧࣹتِّ جُنّࣹيڪْ جُرٗومِ تࣺيمࣺيرْ اَڪْ جُرٗومْ فُڪّ (603550)؞ ‏27 ݧࣹتِّ جُنِّيْ (3000) ڪِلࣷيْ خَالِسْ، دِ تࣺيمࣺيرِ تَلَݝْ، ݧࣹيلْ مٗولُبْ تࣹگُيْ بࣺرَبْ بُ سࣹلّ بِ، اَڪْ تࣹگُيْ رِدࣷ بِ؞ مُيْ تࣺيمࣺيرِ تࣹگُ ݖِ ݧࣹتِّ جُنِّيْ ڪِلࣷ، دِ فَنْوࣹيرِ ڪِلࣷ ݖِ بࣹنّ تࣹگُ، مُيْ تࣺيمࣺيرِ تࣹگُ ݖِ تࣺيمࣺيرِ تَلَݝْ، بࣹنّ تَلَݝْ ݖِ بࣹنّ تࣹگُ؞ ‏28 فُڪِّ ڪِلࣷوڪْ جُرٗومْ ݧَارْ يࣹيڪْ جُرٗومْ ݧَارْ تࣺيمࣺيرِ گَرَامْ يࣹيڪْ جُرٗومْ فُڪّ (17 ڪِلࣷ 750 گَرَامْ) يِ ݖِ تࣷݒّ، يࣹمࣷوڪْ جُنّࣹيڪْ جُرٗومْ ݧَارْ تࣺيمࣺيرْ اَڪْ جُرٗومْ ݧَارْ فُڪّ اَڪْ جُرٗومِ سِيڪَلِ خَالِسْ، اَيْ وࣺڪُّڪَايْ لَ ݖِ سࣵڪّ نگِرْ جࣴنْ يِ، خࣷوبْ ݖِ بࣷݒِّ جࣴنْ يِ، سࣵڪّ ݖِ گَلَنْ يِ؞
‏29 خࣵنجَرْ بِ ݧُ دࣹفْ سَرَخُ يࣺڪَّتِ-جࣺبَّلْ يࣵلَّ، ݧَارِ جُنِّيْ ڪِلࣷ اَڪْ تࣺيمࣺيرْ اَڪْ ݧَارْ فُڪّ اَڪْ ݧࣹينتّ لَ (2124)، يࣹمࣷوڪْ جُرٗومْ ݧَارْ فُڪِّ (70) تَلَݝْ اَڪْ ݧَارِ جُنّࣹيڪْ ݧࣹينْتِ تࣺيمࣺيرِ (2400) سِيڪَلْ؞ ‏30 مُ سࣵڪّ ݖِ تࣹگُيْ بُنْتُ خَيْمَبْ ندَجࣹ مِ، اَڪْ سَرْخَلُڪَايُ خࣵنجَرْ بِ، اَڪْ ݖَاخُبْ خࣵنجَرَمْ، اَڪْ مبࣷولࣹمْ جُمْتُڪَايِ سَرْخَلُڪَايْ بِ، ‏31 اَڪْ تࣹگُيْ وُرْمبَلُ اࣴتّ بِ، اَڪْ تࣹگُيْ بُنْتُ اࣴتّ بِ، اَڪْ خࣹرِ جَامࣷوڪَايْ بِ يࣺݒّ، اَڪْ خࣹرِ وُرْمبَلُ اࣴتّ بِ؞
