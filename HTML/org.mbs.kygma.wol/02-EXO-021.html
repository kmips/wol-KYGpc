21
‏1 اَجِ سَخْ جِ تࣹگْ ݖَ نࣹ:‏ «اࣵتّࣹيْ يࣷونْ يِي، نَنگَ لࣹينْ ڪࣷ جࣷخْ:‏
لُ جࣴمْ ݖِ اࣹبْرࣴ يِيْ اَيْ جَامْ
‏2 «سُ نگࣹينْ جࣴنْدࣹي جَامْ بُ گٗورْ بُ دِبْ اࣹبْرࣴ، نَ لِگّࣺييْ جُرٗومْ بࣹنِّ اَتْ؞ جُرٗومْ ݧَارࣹيلْ بَ لَيْ مࣷومْ بࣷݒَّمْ تࣹ دُ اَمࣹيلْ ڪࣹنّ دَرَ؞ ‏3 سُ فࣹڪّࣹي نࣹ بَ مُيْ دُگّ نجَامْ، مࣷومْ دࣷݝّ لَ وࣷونْ، مࣷومْ دࣷݝّ اَيْ مࣷومْ بࣷݒَّمْ؛ سُ ڪࣷ فࣹڪّࣹي مُ اَمْ جَبَرْ، نَ جَبَرَمْ اࣵندّ اَڪْ مࣷومْ، ݧُ مࣷومْ سࣹينْ بࣷݒّ؞ ‏4 سُ فࣹڪّࣹي نࣹ سَنگَمَ ڪࣷ مَيْ جَبَرْ، ندَوْ سِ اَمْ اَڪْ مࣷومْ اَيْ دࣷومْ، جَبَرْ جࣹيڪْ دࣷومَمْ يُ گٗورْ مبَا يُ جِگࣺينْ، سَنگّ بࣹي لࣹينْ دِ مࣷومْ؞ مࣷومْ ݖِ بࣷݒَّمْ دࣷݝّ اَيْ مࣷومْ بࣷݒَّمْ؞ ‏5 وَايࣹ سُ جَامْ بِ بِرَلࣹي نࣹ:‏ ‹مَا بࣴگّ سَمَ سَنگّ، مَاڪْ سَمَ جَبَرْ اَڪْ سَمَيْ دࣷومْ، تࣹ بُگُّمَا مࣷومْ سَمَ بࣷݒّ›، ‏6 سُ بࣷوبَا نَ ݖِ سَنگّ بِ سࣵڪُّ سࣹيدࣹ اࣵتّࣹڪَتْ يِ؞ سُ ڪࣷ دࣹفࣹي مُ سࣴسَلْ وَا جِ ݖِ لَفُ بُنتّ بِ مبَا جࣴنُ بُنتّ بِ؞ فࣷوفَ لَيْ بࣴنّࣹ نࣷݒُّ وَا جِ اَبْ ݒُسࣷ، مُ دَلْدِ دࣷونْ اَبْ جَامَمْ گِيرُ دُنْدَمْ؞
‏7 «سُ فࣹڪّࣹي نࣹ گٗورْ دَفَ جَايْ دࣷومَمْ جُ جِگࣺينْ نجَامْ، دࣹيسُ ڪࣷ مَنَ گࣷرࣹيلْ نِ بُ دࣷونْ گٗورْ؞ ‏8 سُ ڪࣷ سَنگَمْ جࣴنْدࣹي، دࣹفْ ڪࣷ جَبَرْ، بَ نࣷݒِّ بࣴگَّتُ ڪࣷ، نَ فࣹخࣹ ݧُ فࣹيَلْ ندَوْ سِ اَگْ نجࣷتَمْ، وَايࣹ اَمُلْ سَݧْ-سَݧُ جَايْ ڪࣷ اَيْ دࣷخَنْدࣺيمْ، ندَخْ لࣷولُ وࣷرْ لَ؞ ‏9 سُ فࣹڪّࣹي نࣹ دَ ڪࣷو جࣴنْدَلْ دࣷومَمْ جُ گٗورْ، نَ جࣴفࣹيڪْ مࣷومْ، نَ مُ دࣴݒّࣷوڪْ وَرࣹيفُ دࣷومْ جُ جِگࣺينْ؞ ‏10 ڪِ جࣴندّ اَسْ ندَوْ، دࣹفْ ڪࣷ جَبَرْ نَگْ، سُ تَڪّࣹي ڪࣹنࣹينْ اِتْ، وَرُلَ وࣵݧِّ لَ مُ وَرْلࣷوڪْ جَامْ بَ مُ دࣹفْ جَبَرْ ݖِ وࣵلُّ دُندّ اَڪْ ڪࣷدَّايْ اَڪْ دࣴڪّࣷو؞ ‏11 سُ ڪࣷ سَنگّ بَ مَتَلُلْ ݧࣹتّ يࣷويِي نَگْ، دَيْ مࣷومْ بࣷݒَّمْ، دࣹمْ تࣹ دُ فࣹيْ دَرَ؞
لُ جࣴمْ ݖِ لࣷرْ سَ مࣷرࣷومْ
‏12 «ڪُ دٗورْ نِتْ، رࣹيْ ڪࣷ، دࣹي رࣹڪّ مࣷويْ اࣵتّࣹيمْ؞ ‏13 سُ دُلْ وࣷونْ ݒࣹخࣹ مُ مُ لَلْ، خَنَا مُ فࣹڪّ يࣵلَّ يࣹمَلࣹ نِتْ ڪِ ݖِيْ لࣷخࣷومْ، دِنَا لࣹينْ وࣷنْ بࣺرَبْ بُ مُيْ دَوْ، لࣵقُجِ فَ؞
‏14 «وَايࣹ نَگْ، سُ نِتْ فࣹخࣹيلࣹي مࣷرࣷومَمْ، نَخْ ڪࣷ، بَ رࣹيْ ڪࣷ، سُ لࣵقࣷو وࣷونْ سَمَ سَرْخَلُڪَايْ سَخْ، جࣵݒّلࣹينْ ڪࣷ، رࣹيْ؞
‏15 «ڪُ دٗورْ بَايَمْ مبَا ندࣹيَمْ، دࣹي رࣹڪّ مࣷويْ اࣵتّࣹيمْ؞
‏16 «ڪُ سࣵݖّ نِتْ، دࣹي رࣹڪّ مࣷويْ اࣵتّࣹيمْ، مࣷو خَمْ دَ ڪࣷو جَايْ، مبَا ݧُ فࣹڪّ ڪࣷ ݖِ اَيْ لࣷخࣷومْ؞
‏17 «ڪُ سَاگَ بَايَمْ مبَا ندࣹيَمْ، دࣹي رࣹڪّ مࣷويْ اࣵتّࣹيمْ؞
‏18 «سُ اَيْ نِتْ جࣵݒَّنْتࣹي، ڪࣹنّ ڪِ دٗورْ مࣷرࣷومْ مَ لࣷخࣷومْ مبَا اَوْ دࣷجْ، سُ ݖَ دࣹيوُلْ، تࣹ اَبْ تࣴرَايَمْ سَبَبࣷو ݖَ، ‏19 سُ ݖِ جٗگࣹي مَنَ گࣺنّ ݖِ بِتِ، جࣵݒّ ݖِوْ يࣹتْ سَخْ، دِ دࣷخْ، دࣹيسُلْ دَانْ ڪِ ڪࣷ دٗورْ، وَايࣹ دِنَ دࣵمݒّ ڪِ مُ دٗورْ نگِرْ لِ مُ تࣴدّ لِگّࣺييُلْ، اَڪْ لِ ݒَجَمْ لاَجْ؞
‏20 «ڪُ دٗورْ يࣹتْ جَامَمْ بُ گٗورْ مبَا بُ جِگࣺينْ، بَ مُ دࣹي ݖِيْ لࣷخࣷومْ، دࣹيسْ نَ ڪࣷ فࣹيُلْ مࣷوسْ؞ ‏21 وَايࣹ سُ جَامْ بَ دُنْدࣹي بَ ݖَ اࣴلّࣴگْ سَ، مبَا گَنَّاوْ-اࣴلّࣴگْ سَ، دࣹيسُ ڪࣷ فࣹيُلْ ندَخْتࣹ مࣷومࣹيلُ سَنگَمْ لَ؞
‏22 «سُ اَيْ گٗورْ دࣹي جࣵݒَّنْتࣹ، بَ ڪࣹنّ ݖِ ݧࣷومْ دَلْ ݖِ جِگࣺينُ وࣺرُلْ، بَ لِيرْ بِ گࣺنّ، بُ ݖِ گࣹنّ لࣷرَانگࣹ تࣷݒُّلْ، دࣹيسْ نَ فࣹيْلࣷو مࣷوسْ ڪِ دَلْ ݖِ ڪَوَمْ، ندࣵمْݒَايْ لِ جࣴڪّࣴرْ جِ لاَجْ، تࣹ ݧَ لࣹينْ دِ اࣵتّࣹ جُبࣷو ݖَ لِمُ ݒࣹيْ گَ؞
‏23 «وَايࣹ سُ ݖِ لࣷرَانگࣹ سَبَبࣷو نَگْ، نَ بَڪَّنْ فࣹيْ بَڪَّنْ، ‏24 بࣴتْ فࣹيْ بࣴتْ، بࣴݧْ فࣹيْ بࣴݧْ، لࣷخࣷ فࣹيْ لࣷخࣷ؛ تࣵنڪّ فࣹيْ تࣵنڪّ، ‏25 لَڪّ-لَڪّ فࣹيْ لَڪّ-لَڪّ، تࣴݖُّ-تࣴݖُّ فࣹيْ تࣴݖُّ-تࣴݖُّ، گٗومْ فࣹيْ گٗومْ؞
‏26 «ڪُ دٗورْ اَبْ جَامَمْ بُ گٗورْ مبَا بُ جِگࣺينْ، بَ تُورْ بࣴتَمْ، نَ ڪࣷ گࣷرࣹيلْ ندَخْ بࣴتْ بَ؞ ‏27 سُ فࣷقࣹي بࣴݧُ جَامَمْ بُ گٗورْ مبَا بُ جِگࣺينْ، نَ ڪࣷ گࣷرࣹيلْ ندَخْ بࣴݧْ بَ؞
‏28 «سُ اَوْ نَگْ دَݧࣹي گٗورْ مبَا جِگࣺينْ، بَ رࣹيْ ڪࣷ، دٗورْلࣹينْ نَگْ وِ اَيْ خࣹيرْ بَ مُ دࣹي، تࣹ بُلࣹينْ لࣹڪّ يࣵݒّ وِ، وَايࣹ بࣷرࣷومْ نَگْ وِ، يࣷونْ تࣷݒُّ ڪࣷ؞ ‏29 سُ فࣹڪّࣹي نࣹ نَگْ وِ دَ دَانْ دَݧࣹ، ݧُ يࣺگَلࣷونْ ڪࣷ بࣷرࣷومْ، وَتُّوُ ڪࣷ، بَ نَگْ وِ رࣹيْ گٗورْ مبَا جِگࣺينْ، دٗورْلࣹينْ نَگْ وِ اَيْ خࣹيرْ، بَ مُ دࣹي، تࣹ تࣹگْ ݖِ رࣹيْ بࣷرࣷومْ نَگْ وِ؞ ‏30 سُ فࣹڪّࣹي نࣹ دࣹيسْ ڪࣷو سَسْ ندࣵمْݒَايْ، نَ فࣹيْ لِمْ بُ ݧُ ڪࣷ مَنَ سَسْ، مُيْ نجࣷتْ-گِ-بَڪَّنَمْ؞ ‏31 بَ تࣹيْ اِتَمْ، سُ نَگْ وِ دَݧࣹي خَلࣹ، بَ مُ دࣹي، مُيْ گٗورْ مبَا جِگࣺينْ، نَ اࣵتّࣹ بِيْ بࣹنّ رࣹڪّ ݖِ بࣷرࣷومْ نَگْ وِ؞ ‏32 سُ نَگْ وِ دَݧࣹي اَبْ جَامْ بَ مُ دࣹي، مُيْ گٗورْ مبَا جِگࣺينْ، نَ بࣷرࣷومْ نَگْ وِ فࣹيْ بࣷرࣷومْ جَامْ بِ فَنْوࣹيرِ دࣷگِ خَالِسْ، تࣹ نَݧُ دٗورْ نَگْ وِ اَيْ خࣹيرْ، بَ مُ دࣹي؞
‏33 «سُ فࣹڪّࣹي نࣹ نِتْ دَفَ اُبِّ ڪࣵمبّ گُ اُبࣹ وࣷونْ، مبَا مُ گَسْ ڪࣵمبّ، وَݖّ ڪࣷ مُ نࣹ ݝࣵݒّ، بَ نَگْ مبَا مبَامْ دَانُ ݖِ، ‏34 نَ بࣷرࣷومْ ڪࣵمبّ گِ فࣹيْ؛ دَيْ دࣹلّࣷو بࣷرࣷومْ مَلَ مِ نجࣺگُ مَلَ مِ، دَلْدِ فࣺيتࣹوࣷو مَلَ مِ ݖِ دࣹي؞
‏35 «سُ نَگُ نِتْ جَمࣹي نَگُ مࣷرࣷومَمْ، بَ مُ دࣹي، نَݧُ بࣷڪّ جَايْ نَگْ وِيْ دُندّ، سࣺدّࣷو نجࣺگْ گِ، تࣹ سࣺدّࣷو نَگْ وِ دࣹي؞
‏36 «وَايࣹ سُ فࣹڪّࣹي مُيْ نَگْ وُ ݧُ خَمْ نࣹ دَ دَانْ دَݧࣹ، تࣹ بࣷرࣷومَمْ ݧࣵڪّ ڪࣷو وَتُّ، بࣷرࣷومَمْ دَيْ فࣹيࣹ نَگْ وِ دࣹي اَوْ نَگْ وُيْ دُندّ، دَلْدِ فࣺيتࣹوࣷو وِ دࣹي؞
‏37 «ڪُ سࣵݖّ اَوْ نَگْ مبَا اَبْ گࣵتّ، رࣹيْ ڪࣷ مبَا مُ جَايْ ڪࣷ، نَ فࣹيࣹ وࣹنّ نَگْ، جُرٗومِ نَگْ؛ بࣹنّ گࣵتّ، مُ فࣹيࣹ ڪࣷ ݧࣹينْتِ گࣵتّ؞
