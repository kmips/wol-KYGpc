لُ جࣴمْ ݖِ بࣺسُبْ مُݖّ بَ
12
‏1 اَجِ سَخْ جِ وَخْ نَ اَڪْ مُوسَىٰ اَڪْ هَـٰرُونَ ݖَ رࣺيوُمْ مِسْرَ، نࣹ لࣹينْ:‏ ‏2 «وࣹيرْ وِي مࣷويْ دࣷونْ ݖِ يࣹينْ وِ جِيتُ ݖِ وࣹيرْ يِ، تࣹ نَ دࣷونْ ݖِ يࣹينْ وࣹيرْ، وِ جࣴڪّ ݖِ اَتْ مِ؞ ‏3 وَخْلࣹينْ اَڪْ مبࣷولࣷومْ إِسْرَاءِيلْ مࣺݒّ نࣹ لࣹينْ، بࣺسُبْ فُڪّࣹيلْ بَ ݖِ وࣹيرْ وِي، نَ ڪُ نࣹڪّ جࣴلَلْ نجَبࣷوتَمْ اَبْ گࣵتّ، ڪࣴرْ گُ نࣹڪّ اَڪْ گࣵتَّمْ؞ ‏4 وَايࣹ سُ دࣹي گࣵتّ بُ اࣴݒّ وَا ڪࣴرْ گَ، دَݧُ ڪࣷيْ بࣷڪّ اَڪْ ڪࣴرْ، گِ لࣹينْ گࣴنَ جࣹگࣹ، تࣹ نَݧُ سࣹيتْ نُ نِتْ ݧَ تࣷلُّ، دࣷورَ تࣵنّ گࣵتّ بُ دࣴݒّࣷوڪْ لَ ݧُ مَنَ لࣹڪّ؞ ‏5 گࣵتّ بَ بُمُ اَمْ سِڪّ تࣹ نَ دِ ڪُويُ مࣹنّ اَتْ؞ مَنْ نَا دࣷونْ خَرْ مبَا سِڪّࣹتْ؞ ‏6 دَنگࣹينْ ڪࣷيْ دࣹنݖّ بَ فُڪّࣹيلُ بࣺسْ بَ اَڪْ ݧࣹينتّ ݖِ وࣹيرْ وِي؞ بُ بࣺسْ بَا ݖِ دِگَّنْتࣹ نگࣷونْ اَڪْ جَنتّ بُ سࣷ، نَ مبࣷولࣷومْ إِسْرَاءِيلْ مࣺݒّ بࣷڪّ رࣹنْدِ سࣹينِ گࣵتّ؞ ‏7 سُ ڪࣷ دࣹفࣹي ݧُ سࣵڪّ ݖَ دࣹرࣹتْ جَ، تَقَلْ ݖَ بُنْتُ ڪࣴرْ، گَ ݧُ ڪࣷ نَرَ لࣹڪّࣹ، تَقَلْ ݖَ ݧَارِ جࣴنِ بُنتّ بَ اَڪْ نجࣴعࣴلْ بُنتّ بَ ݖَ ڪَوْ؞ ‏8 دَݧُيْ لَڪّ يࣵݒّ وَ ݖِبْ تَالْ، لࣹڪّ ڪࣷ گُدِّگْ ڪࣹرࣷوگْ، بࣷولࣹ ڪࣷوڪْ مبُرُ مُ اَمُلْ لَوِيرْ اَڪْ خࣷبْ يُ وࣹخْ؞ ‏9 نَݧُ ڪࣷ وࣵجّ، مُ نࣹ ݧُمّ، بࣷݒّ بِ اَڪْ تࣵنڪّ يِ اَڪْ يࣺرࣹيْ بِيرْ يِ؞ بُلࣹينْ ݖِ لࣹڪّ دَرَ لُ ݧࣷرُلْ، مبَا لُ ݧُ بَخَلْ؞ ‏10 بُلࣹينْ ݖِ وَݖّ دَرَ مُ فَنَانْ، تࣹ لࣺݒّ لُ ݖِ دࣹسْ بُ بࣴتْ سࣹتࣹي، لَڪّلࣹينْ بَ مُ جࣹيخْ تࣵڪّ؞ ‏11 نِ نگࣹينْ ڪࣷيْ لࣹڪّࣹ نَگْ، نِي لَ:‏ بُ نگࣹينْ ڪࣷيْ لࣹڪّ، دَنگࣹينْ دِ تَڪُّ، سࣷلْ سࣹينِ دࣵلّ، ݝࣵبّ سࣹينْ يࣹتْ، تࣹ نگࣹينْ لࣹڪّ ڪࣷ ݖِ لُ گَاوْ؞ لࣷولُ گࣵتُّ سَرَخُ بࣺسُبْ مُݖّ بَ لَيْ دࣷونْ، ݧࣹيلْ اَجِ سَخْ جِ؞
‏12 «گُدِّگْ ڪࣹرࣷوگْ مَايْ وࣴرْ رࣺيوُمْ مِسْرَ تࣹ مَايْ فࣵدّ لࣺݒّ لُيْ تَاوْ بُ گٗورْ ݖِ رࣺيوُمْ مِسْرَ، نِتْ اَڪْ مَلَ، تࣹ اِتَمْ مَايْ مبُگَلْ مبࣷولࣹمْ تُورِ مِسْرَ؞ مَنْ مَايْ اَجِ سَخْ جِ؞ ‏13 دࣹرࣹتْ جِ ݖِ ڪࣴرْ، يِ نگࣹينْ دࣴڪّࣹ نَگْ، مࣷويْ دࣷونْ سࣹينْ خࣵمِّڪَايْ، بَ تَخْ بُ مَ گِسࣹي دࣹرࣹتْ جِ، مُسَلْ لࣹينْ؞ سُ ڪࣷ دࣹفࣹي بُ مَيْ دَلْ ݖِ ڪَوْ رࣺيوُمْ مِسْرَ، مࣹنّ مُسِبَ دُ لࣹينْ لࣷرْ؞
‏14 «بࣺسُبْ ڪࣹرࣷوگْ بࣺسُ ݒࣵتّلِڪُ لَيْ دࣷونْ ݖِ يࣹينْ، نگࣹينْ وَرْ ڪࣷو مࣵگَّلْ نگِرْ اَجِ سَخْ جِ، مُيْ دࣷگَلْ بُ لࣹينْ وَرْ فࣵوّ، يࣹينْ اَڪْ سࣹينْ اَسْڪَنْ؞
لُ جࣴمْ ݖِ اَيُ بࣺسُ مبُرُ مُ اَمُلْ لَوِيرْ
‏15 «دِيرُ جُرٗومْ ݧَارِ فَنْ نگࣹينْ دِ لࣹڪّ مبُرُ مُ اَمُلْ لَوِيرْ؛ تࣹ بࣺسْ بُ جࣴڪّ بِ ݖِ بࣷݒَّمْ نَنگࣹينْ جࣴلࣹ سࣹينِ ڪࣴرْ لُيْ لَوِيرْ، ندَخْتࣹ ڪࣺݒّ ڪُ ݖِ لࣹڪّ مبُرُ مُ اَمْ لَوِيرْ، لِ ڪࣷ دَلࣹ بࣺسْ بُ جࣴڪّ بِ، بَ ڪࣹرَ بࣺسُبْ جُرٗومْ ݧَارࣹيلْ بَ، دࣹيسْ نَ ڪࣷ دَگّࣹ ݖِ اَسْڪَنُ إِسْرَاءِيلْ؞ ‏16 بࣺسْ بُ جࣴڪّ بِ نَنگࣹينْ ݖِ اَمَلْ ندَجࣹ مُ سࣹلّ؞ جُرٗومْ ݧَارࣹيلُ بࣺسْ بَ اِتَمْ نَنگࣹينْ ݖَ اَمَلْ ندَجࣹ مُ سࣹلّ؛ دࣹيسُلْ لِگّࣺييْ لࣹنّ ݖِ يࣷويُ بࣺسْ، لُ مࣷيْ نگࣹينْ تࣷگّ ݧَمْ وُ ݧࣺݒّ لࣹڪّ؞ لࣷولُ دࣷݝّ لࣹيسْ ݖِيْ مَنَ دࣹفْ؞
‏17 «نَنگࣹينْ وࣷرْمَالْ اَيُ بࣺسُ مبُرُ مُ اَمُلْ لَوِيرْ، ندَخْتࣹ ݖِ بࣺسْ بࣷوبُ تࣹمْبࣹ لاَ لࣹينْ گࣺنّࣹي رࣺيوُمْ مِسْرَ، ڪُرࣺلْ ݖِ ڪَوْ ڪُرࣺلْ؞ ڪࣷنْ نَگْ نَنگࣹينْ وࣷرْمَالْ بࣺسْ بِي، مُيْ دࣷگَلْ بُ لࣹينْ وَرْ فࣵوّ، يࣹينْ اَڪْ سࣹينْ اَسْڪَنْ؞ ‏18 لِ ڪࣷ دَلࣹ نگࣷونُ فُڪِّ فَنْ اَڪْ ݧࣹينتّ ݖِ وࣹيرْ وُ جࣴڪّ وِ، بَ ݖَ نگࣷونُ ݧَارْ فُڪِّ فَنْ اَڪْ بࣹنّ ݖِ وࣹيرْ وࣷووُ، نَنگࣹينْ ݖِ لࣹڪّ مبُرُ مُ اَمُلْ لَوِيرْ؞ ‏19 دِيرُ جُرٗومْ ݧَارِ فَنْ، بُ دَرَ ݖِ لَوِيرْ اَمْ فࣹنّ ݖِ سࣹينِ ڪࣴرْ، ندَخْ ڪࣺݒّ ڪُ لࣹڪّ لُ اَمْ لَوِيرْ، دࣹيسْ نَ ڪࣷ دَگّࣹ ݖِ اَسْڪَنُ إِسْرَاءِيلْ، مُيْ دࣷخَنْدࣺيمْ مبَا دࣷومُ رࣺيوْ مِ؞ ‏20 بُلࣹينْ ݖِ لࣹڪّ لࣹنّ لُ اَمْ لَوِيرْ، تࣹ فࣺݒّ فُ نگࣹينْ دَلْ، نَ مبُرُ مُ نگࣹينْ مَنَ لࣹڪّ دِ مُ اَمُلْ لَوِيرْ؞»
وَاجَلْ نَݧُ بࣺسُبْ مُݖّ بَ
‏21 بَ لࣷولُ اَمࣹي مُوسَىٰ وࣷو مَگِ إِسْرَاءِيلْ ݧࣺݒّ، نࣹ لࣹينْ:‏ «دࣹمْلࣹينْ تࣵنَّلِ سࣹينِ نجَبࣷوتْ اَيْ گࣵتّ، تࣹ نگࣹينْ رࣹنْدِ گࣵتّ يࣷويُ، دࣹفْ ڪࣷ سَرَخُ بࣺسُبْ مُݖّ بِ؞ ‏22 دَنگࣹينْ دِ جࣴلْ اَبْ تَڪُّ ݖَرِ اِسࣷبْ¤، ݖَݒّ ݖِ دࣹرࣹتْ جِ ݖِ ندَبْ لِ، دِوْ ݖِ سࣹينْ نجࣴعࣴلْ بُنتّ اَڪْ سࣹينْ ݧَارِ جࣴنِ بُنتّ، تࣹ بُ ڪࣹنّ ݖِ يࣹينْ وࣹيسْ بُنْتُ ڪࣴرَمْ بَ بࣴتْ سࣹتْ؞ ‏23 سُ ڪࣷ دࣹفࣹي، بُ اَجِ سَخْ جِيْ وࣴرْ ݖِ بِيرْ رࣺيوْ مِ، دِ فࣵدّ وَا مِسْرَ، بَ گِسْ دࣹرࣹتْ جِ تَقْ ݖِ سࣹينْ نجࣴعࣴلْ بُنتّ اَڪْ سࣹينْ ݧَارِ جࣴنِ بُنتّ، دَيْ تࣹگِّ بُنتّ بَ، تࣹ دُ بࣵيِّ مَلاَڪَمْ رࣹيْڪَتْ بِ، مُ دُگّ سࣹينِ ڪࣴرْ، دِ فࣵدّࣹ؞
‏24 «نَنگࣹينْ سࣵمّࣷنْتࣹيڪْ لࣷولُ، مُيْ دࣷگَلْ بُ لࣹينْ وَرْ فࣵوّ، يࣹينْ اَڪْ سࣹينْ اَسْڪَنْ؞ ‏25 ڪࣹرࣷوگْ بُ نگࣹينْ دُگّࣹي نَگْ ݖَ رࣺيوْ مَ لࣹينْ اَجِ سَخْ جِ جࣷخْ نَ مُ لࣹينْ ڪࣷ دِگࣹ وࣷونْ، نَنگࣹينْ سࣵمّࣷنْتࣹيڪْ جࣷوجُ آدَ؞ ‏26 تࣹ بُ لࣹينْ سࣹينِ دࣷومْ لاَجࣹي نࣹ لࣹينْ:‏ ‹لُ آدَ جِي دِ وُندّ؟› ‏27 وَخْلࣹينْ لࣹينْ نࣹ لࣹينْ:‏ ‹لࣷولُ سَرَخْ لَ ݧࣹيلْ اَجِ سَخْ جِ نگِرْ بࣺسُبْ مُݖّ بَ، ندَخْ لَ مُ تࣹگِّ وࣷونْ ڪࣴرِ بࣵنِّ إِسْرَاءِيلْ ݖَ مِسْرَ، ڪࣹرَ بَ مُيْ فࣵدّ وَا مِسْرَ تࣹ مُسَلْ ݖَ سُنُيْ ڪࣴرْ؞›»
بَ مُ وَخࣹي لࣷولُ بَ نࣷݒِّ، مبࣷولࣷو مَ سُجّٗوتْ، مࣵگَّلْ يࣵلَّ؞ ‏28 لَ اَجِ سَخْ جِ سَنْتࣷونْ مُوسَىٰ اَڪْ هَـٰرُونَ، نࣷونُ رࣹڪّ لَ ڪࣷ بࣵنِّ إِسْرَاءِيلْ دࣹفࣹ؞
يَرُبْ فُڪّࣹيلْ بَ، دِ دࣹيوُ تَاوْ يَ
‏29 ݖَ خَاجُ گُدِّ لَ اَجِ سَخْ جِ فࣵدّ مبࣷولࣹمْ تَاوِ رࣺيوُمْ مِسْرَ يُ گٗورْ، دَلࣹ ڪࣷ ݖَ تَاوْ بُ گٗورْ بُ فِرْعَوْنَ، مَ تࣷوگْ ݖَ جَلْ بَ، بَ ݖَ تَاوْ يُ گٗورْ يُ نِتْ ݧَ ݧُ تࣴجْ ڪَسࣷ، بࣷولࣹ ݖَ لࣺݒّ لُيْ تَاوْ بُ گٗورْ ݖَ جُرْ گَ؞ ‏30 فِرْعَوْنَ نَگْ جٗگْ ݖَ گُدِّ گَ، مࣷومْ اَڪْ دَگَمْ يࣺݒّ اَڪْ وَا مِسْرَ گࣺݒّ، يُوخْ گُ رࣺيْ جِبࣹ ݖَ مِسْرَ، ندَخْ ڪࣴرࣷو ڪࣴرْ، دُ گࣹنّ گُ نِتْ دࣹيوُلْ؞
‏31 بَ لࣷولُ اَمࣹي فِرْعَوْنَ وࣷولُ مُوسَىٰ اَڪْ هَـٰرُونَ ݖَ گُدِّ گࣷوگُ، نࣹ لࣹينْ:‏ «اَيْݖَ!‏ مࣷتّ!‏ سࣷرࣹلࣹينْ سَمَ وَا رࣺيوْ، يࣹينْ اَڪْ سࣹينِ بࣷڪّ!‏ لَگّلࣹينْ، جَامُجِ اَجِ سَخْ جِ، لِ نگࣹينْ بࣴگّ؞ ‏32 تࣹ اِتَمْ يٗبَّالࣹلࣹينْ سࣹينْ جُرࣷو جُرْ، گُ گُدّ اَڪْ گُ گࣵتّ، لِ نگࣹينْ بࣴگّ رࣹڪّ، تࣹ نࣹ مࣷتّ!‏ وَايࣹ ݧَانَالࣹلࣹينْ مَ سࣹينْ يࣵلَّ نَگْ، مُ بَرْڪࣹيلْ مَ؞»
بࣵنِّ إِسْرَاءِيلْ گࣵدَّايْ نَ
‏33 بَ مُ ڪࣷ دࣹفࣹي وَا مِسْرَ سࣷݧّ بࣵنِّ إِسْرَاءِيلْ، نگِرْ ݧُ گَاوْ گࣺنّ رࣺيوْ مَ، ندَخْ بࣷوبَ ݧَ نگَ نَانْ:‏ «نُنْ ݧࣺݒّ اَيْ دࣹي!‏» ‏34 لࣷولُ تَخْ مبࣷولࣷومْ إِسْرَاءِيلْ رࣷݧْ سࣹينْ خِيوِ سُنگُفْ يُ اَمَگُلْ لَوِيرْ؞ ݧُ اࣴمبّ سࣹينْ ندَبِ نࣷتُڪَايْ ݖَ سࣹينِ مَلاَنْ، گࣵدُّ؞ ‏35 فࣹڪّ نَ بࣵنِّ إِسْرَاءِيلْ دࣹفْ لَ لࣹينْ مُوسَىٰ سَنْتࣷونْ ݖَ وَا مِسْرَ؞ سࣵڪُّ نَݧُ ݖَ ݧࣷومْ اَيْ گࣵنجَرِ خَالِسْ اَڪْ وُرُسْ اَڪِ يࣺرࣹ؞ ‏36 اَجِ سَخْ جِ نَگْ دࣹفْ وَا مِسْرَ نَنگُلْ بࣵنِّ إِسْرَاءِيلْ، بَ جࣷخْ لࣹينْ لَ ݧُ سࣵڪُّ؛ نࣷونُ لَݧُ فُتّࣹي وَا مِسْرَ؞
‏37 ݖِ ڪَوْ لࣷولُ بࣵنِّ إِسْرَاءِيلْ جٗگࣹ دࣴڪّ بَ ݧُيْ وَخْ رَمْسࣹسْ، جࣴمْ سُڪّٗوتْ؞ ݧَ نگَ تࣷلُّ ݖِ جُرٗومْ بࣹنِّ تࣺيمࣺيرِ جُنِّيْ گٗورْ (600000) تࣹ لِمَالࣹيسُلْ جِگࣺينْ ݧَاڪْ خَلࣹ يَ؞ ‏38 اَمࣷونْ نَ اِتْ مبࣷولࣷومْ نِتْ ݧُ اࣵندّ اَڪْ ݧࣷومْ، تࣹ دِ اَسْڪَنْ يُ رَخࣹ، اَڪْ جُرْ گُ بَرࣹ، گُ گُدّ اَڪْ گُ گࣵتّ؞ ‏39 ݧُ دَلْدِ لَڪّ مبُرُ مُ تࣵݒَّنْدَارْ ݖَ خِيوْ وُ اَمُلْ لَوِيرْ، وَ ݧُ جࣴلࣹ مِسْرَ؞ لَ تَخࣷونْ خِيوْ وَ ݧࣵڪّ لَوِيرْ مࣷويْ دَݧُ لࣹينَ دࣵقࣹ مِسْرَ، بَ تَلُݧُ وࣷونَ فࣵگُّ لُ ݧُ لࣹڪّ؞
‏40 دِيرْ بَ بࣵنِّ إِسْرَاءِيلْ نࣹڪّ مِسْرَ تࣷلُّ نَ ݖِ ݧࣹينْتِ تࣺيمࣺيرِ اَتْ اَڪْ فَنْوࣹيرْ (430)؞ ‏41 ڪࣹرࣷوگْ ݖَ بࣺسْ، بَ ݧࣹينْتِ تࣺيمࣺيرِ اَتْ يَ اَڪْ فَنْوࣹيرْ مَتࣹي سࣴڪّ، ݖَ لَ ݧࣷݧِ اَجِ سَخْ جِ گࣺنّ رࣺيوُمْ مِسْرَ، ڪُرࣺلْ ݖِ ڪَوْ ڪُرࣺلْ؞ ‏42 گُدِّگْ ڪࣹرࣷوگْ گُدِّگْ تࣹيوْلُ لَ، ݧࣹيلْ اَجِ سَخْ جِ، نگِرْ مࣷو لࣹينْ ݖَ گࣺنّࣹ رࣺيوُمْ مِسْرَ، مࣷو تَخْ گُدِّ گࣷوگُ دِ گُ ݧُ جَگْلࣹيلْ اَجِ سَخْ جِ، بࣵنِّ إِسْرَاءِيلْ گࣺݒّ وَرْ ݖࣹي فَنَانࣹي تࣹيوْلُ، ݧࣹيلْ اَجِ سَخْ جِ، بَ فࣵوّ؞
اَيْ دࣷگَلِ اَجِ سَخْ جَا نگِي، ݧࣹيلْ بࣺسُبْ مُݖّ بَ
‏43 گَنَّاوْ لࣷولُ اَجِ سَخْ جِ دَفَ وَخْ اَڪْ مُوسَىٰ اَڪْ هَـٰرُونَ، نࣹ لࣹينْ:‏ «لِ مَ دࣷگَلْ ݖِ سَرَخُ بࣺسُبْ مُݖّ بِ مࣷويْ لِي:‏ بࣹنّ دࣷخَنْدࣺيمْ دُ ݖِ لࣹڪّ؞ ‏44 وَايࣹ بࣺݒّ جَامْ بُ ݧُ جࣴندّ تࣹ مُ خَسَ خَرَفْ، مَنْ نَ ݖࣹي لࣹڪّ؞ ‏45 بࣹنّ مࣵنگَانْ مبَا لِگّࣺييْڪَتْ بُ ݧُ بِندّ اِتْ دُ ݖِ لࣹڪّ؞ ‏46 گࣹنّ ڪࣴرْ لࣹيسْ ڪࣷيْ لࣹڪّࣹ:‏ بُ لࣹنّ ݖِ يࣵݒّ وِ گࣺنّ ڪࣴرْ گِ، تࣹ دࣹيسُ ݖِ دَمّ بࣹنّ يَخْ سَخْ؞ ‏47 مبࣷولࣹمْ ڪُ بࣷڪّ ݖِ بࣵنِّ إِسْرَاءِيلْ، نَ دࣹفْ سَرَخْ بِ؞ ‏48 سُ فࣹڪّࣹي نࣹ اَبْ دࣷخَنْدࣺيمْ بُ نگࣹينْ دࣴڪَّلْ دَفَ بࣴگَّ وࣷرْمَالْ مࣵگَّلُ بࣺسُبْ مُݖّ بِ نگِرْ اَجِ سَخْ جِ، خَرَفْ وَرْ نَ ڪࣷ، تࣹ وَرْ نَ گࣺݒّ گٗورْ گُ بࣷوتُ ݖِ مࣷومْ؛ سُ لࣷولُ اَمࣹي، دࣹيسْ نَ ڪࣷ مَيْ، مُ دࣹفْ ڪࣷ، ندَخْ ڪࣷنْ دَيْ مࣹلْ نِ دࣷومُ رࣺيوْ مِ؞ وَايࣹ ڪࣹنّ ڪُ خَرَفُلْ دُ ݖِ لࣹڪّ؞ ‏49 دࣷگَلْ بِ بࣹنّ لَيْ دࣷونْ، دࣷومُ رࣺيوْ مِ اَڪْ دࣷخَنْدࣺيمْ بِ دࣴڪّ ݖِ سࣹينْ بِيرْ، بࣷڪّ ݖِ يࣹمْ ڪࣹݒّ؞»
‏50 لَ اَجِ سَخْ جِ سَنْتࣷونْ مُوسَىٰ اَڪْ هَـٰرُونَ، نࣷونُ رࣹڪّ لَ ڪࣷ بࣵنِّ إِسْرَاءِيلْ دࣹفࣹ؞
‏51 بࣺسُبْ ڪࣹرࣷوگْ بࣷوبࣹي لَ اَجِ سَخْ جِ يِوِ بࣵنِّ إِسْرَاءِيلْ، ݧُ دࣹفْ ڪُرࣺلْ ݖِ ڪَوْ ڪُرࣺلْ، گࣺنّ رࣺيوُمْ مِسْرَ؞
