يࣷونْ وِ سࣵرْتَلْ نَ دُنْدِنْ وُ جࣹڪّ
22
‏1 بُلࣹينْ گِسْ سࣹينْ نَگُ مبࣷڪّ مبَا گࣵتَّمْ بُ رࣺيرْ، بَ جࣵݒّ ݖِ سࣹينْ نجَامْبُرْ؞ فࣹخࣹلࣹينْ ڪࣷو وَݧِّ، بَ دࣹلّࣷو ڪࣷ سࣹينْ مبࣷڪّ مَ؞ ‏2 سُ مبࣷڪّ مِ جࣹگࣹوࣷووُلْ اَڪْ يࣹينْ، مبَا خَمُلࣹينْ ڪࣷ، يٗبُّلࣹينْ جُرْ گِ سࣹينْ ڪࣴرْ، مُ دࣹسْ فَ، بَ مبࣷڪّ مِ دِڪّ، دِ ڪࣷ سࣹيتْ، نگࣹينْ دࣹلّࣷو ڪࣷ ڪࣷ؞ ‏3 نࣷونُ نگࣹينْ دِ دࣹفْ، سُ دࣹي مبَامْ، نࣷونُ نگࣹينْ دِ دࣹفْ سُ دࣹي مبُبّ، نࣷونُ اِتْ نگࣹينْ دِ دࣹفْ، اَڪْ مبࣷولࣹمْ لُ سࣹينْ مبࣷڪّ رࣺيرْلࣹ، بَ نگࣹينْ گِسْ ڪࣷ؞ دُنگࣹينْ ݖِ مَنَ جࣵݒّ سࣹينْ نجَامْبُرْ؞
‏4 بُلࣹينْ گِسْ سࣹينْ مبَامُ مبࣷڪّ، مبَا نَگَمْ وُ دَانُ ݖِوْ يࣷونْ، بَ نگࣹينْ جࣵݒّ ݖِ سࣹينْ نجَامْبُرْ؞ فࣹخࣹلࣹينْ بَ يࣺڪَّتِلࣹ ڪࣷ ڪࣷ؞
‏5 بُ لࣹنّ ݖِ مبࣷولࣹمْ يࣴفِ گٗورْ، تࣹگُ ݖِ ڪَوْ ڪُ جِگࣺينْ، تࣹ بُ گٗورْ سࣷلْ يࣺرࣹيْ جِگࣺينْ، ندَخْ اَجِ سَخْ جِ سࣹينْ يࣵلَّ سࣹيخْلُ نَ ڪࣺݒّ ڪُيْ دࣹفْ يࣷويُ؞
‏6 بُ نگࣹينْ دࣷخࣹي بَ يࣹمْ ݖِ تࣵگُّمْ ݒِݖّ ݖِ سࣹينُ يࣷونْ، مُ نࣹڪّ ݖِ گَرَبْ گُ مُ مَنَ دࣷونْ، مبَا ݖِ سُوفْ، ݖُوجْ يَ مبَا نࣹنْ يَ نࣹڪّ ݖَ، تࣹ يَايْ جَ بٗوفْ ݖَ ڪَوْ ݖُوجْ يَ، مبَا نࣹنْ يَ، بُلࣹينْ بࣷولࣹ جࣴلْ يَايْ جَ اَڪْ ݖُوجْ يَ؞ ‏7 فࣹخࣹلࣹينْ بَ بࣵيِّ يَايْ جَ، تࣹ جࣴلْ ݖُوجْ يَ، ندَخْ نگࣹينْ بَاخْلࣹ تࣹ گُدّ فَنْ؞
‏8 سُ نگࣹينْ دࣹي تَبَخْ ڪࣴرْ گُ بࣹيسْ، سࣵڪَّلْ-لࣹينْ سࣹينْ ڪَوُمْ تَاخْ، اَبْ فࣹگُوَايْ، بَلَ ݖَا نِتَ دَانࣷو، اࣵقُ دࣹرࣹتْ تࣷݒّ سࣹينْ ڪࣴرْ؞
‏9 بُلࣹينْ جِ ݖِ سࣹينْ تٗوڪࣴرُ رࣹسࣹݧْ، وࣹنࣹينْ وِرْگࣷ، لُ ڪࣷ مࣷيْ لࣺݒّ اَيْ دࣷونْ ݖࣺرُ سࣹلّنگَ بُ نگࣹينْ دُلْ سَݧَ جَرِݧࣷو:‏ لِ جٗگࣹ ݖِ جِوُ مِ رَخْ رࣹسࣹݧْ جِ، اَڪْ اِتَمْ مࣹݧّࣹيفُمْ رࣹسࣹݧْ جِ؞ ‏10 بُلࣹينْ رࣵنگَنْدࣷو نَگْ اَڪْ مبَامْ، دِ لࣹينْ رِجِّلࣷو؞
‏11 بُلࣹينْ سࣷلْ ندِمࣷلْ نجَخَسْ لُ ݧُ رَخࣹ ڪَوَرُ گࣵتّ اَڪْ وࣴݧُ لࣹي؞
‏12 اَيْ ݖِمْبِرْ-ݖِمْبِرْ نَگْ نگࣹينْ دِ دࣹفْ ݖِ ݧࣹينْتِ وࣹتِ مَلاَنُمْ سࣵنگُڪَايْ مُ نگࣹينْ دِ سࣵنگࣷو؞
سࣵمّࣷنْتࣹلࣹينْ اَڪْ اࣵقُ جِگࣺينْ
‏13 ڪُ جࣴلْ جَبَرْ، دࣴڪّࣷو ڪࣷ، بَ نࣷݒِّ جࣺݒِّ ڪࣷ، ‏14 دِ ڪࣷ تُومَالْ، دِ تࣹگْ لُ ݧَاوْ ݖِ دࣹرَمْ، نَانْ:‏ «ندَوْ سِي مَ جࣴلْ، بَ مَ اࣵگّࣹي ݖِ مࣷومْ، فࣹڪُّمَ ڪࣷ جَنقّ،» ‏15 سُ بࣷوبَا نَ بَايُ ندَوْ سِ اَڪْ ندࣹيَمْ، جࣴلْ فِرْندࣹيْ ندَوْ لَ، يٗبُّ ڪࣷ بَ ݖَ ڪَنَمْ مَگِ دࣴڪّ بَ، ݖَ بُنتّ دࣴڪّ بَ؞ ‏16 سُ ڪࣷ دࣹفࣹي بَايُ ندَوْ سِ نࣹ مَگْ ݧِ:‏ «سَمَ دࣷومْ لاَ مَيࣷونْ وَا جِي جَبَرْ؛ مُ وَلْبَتِڪُ جࣺݒِّ ڪࣷ؞ ‏17 مُ نگِي دِ ڪࣷ تُومَالْ، نَانْ:‏ ‹فࣹڪُّمَ سَ دࣷومْ جِ جَنقّ؞› تࣹ سَمَ ندَوَلْ دࣷومَ نگِي!‏» نَݧُ تࣵلَّلْ مَلاَنْ مَ ݖَ ڪَنَمْ مَگِ دࣴڪّ بَ؞ ‏18 سُ لࣷولُ اَمࣹي، نَ مَگِ دࣴڪّ بࣷوبَ جࣵݒّ وَا جِ، يَرْ ڪࣷ؞ ‏19 نَݧُ ڪࣷ دَانْ تࣺيمࣺيرِ دࣷنجِ خَالِسْ، جࣷخْ ڪࣷ بَايُ ندَوْ سِ، ندَخْ مࣷو يࣵقْ دࣹرُ جَنقُبْ إِسْرَاءِيلْ؞ جَبَرَمْ لَيْ دࣷونْ بَ تࣹيْ، تࣹ دُ ڪࣷ سَݧَ وَݖّ گِيرُ دُنْدَمْ؞
‏20 وَايࣹ سُ تُومَ جِ دࣹي دࣴگّ، تࣹ گِسࣹيسُلْ لࣹنّ ندَوْ ݖَ ندَوْ سَ، ‏21 نَݧُ يٗبُّ ندَوْ سَ بَ ݖَ بُنتّ ڪࣴرْ بَايَمْ، تࣹ گٗورِ دࣴڪّ بِ دٗورْ ڪࣷيْ دࣷجْ بَ مُ دࣹي، ندَخْ مࣷو دࣹفْ جࣴفْ جُ گࣵݖّࣹلُ ݖِ دِگّ إِسْرَاءِيلْ، دِ گࣵنݖّتࣷو ڪࣴرْ بَايَمْ؞ نࣷونُ نگࣹينْ دِ تࣹنقࣹي لُ بࣷنْ ݖِ سࣹينْ بِيرْ؞
‏22 بُ ݧُ فࣹڪّࣹي ڪُيْ تࣴدّࣹ ڪُ اَمْ جࣴڪّࣴرْ؛ ڪَ ڪࣷ تࣴدّࣹ اَڪْ ڪَ ݧُ تࣴدّࣹ، دࣹي مࣷويْ سࣹينْ اࣵتّࣹ، ݧࣷومْ ݧَارْ ݧࣺݒّ، نگࣹينْ تࣹنقࣹي ڪࣷ لُ بࣷنْ ݖِ إِسْرَاءِيلْ؞
‏23 سُ اَبْ جَنقّ دِگࣷو اَڪْ وَايَمْ اَبْ سࣺيْ، گَنَّاوْ گِ ڪࣹنࣹينْ دَجࣹيڪْ مࣷومْ ݖِ بِيرْ دࣴڪّ بِ، تࣴدّࣹ ڪࣷ، ‏24 گࣺنّࣹلࣹينْ لࣹينْ ݧࣷومْ ݧَارْ، بَ ݖَ بُنتّ دࣴڪّ بࣷوبَ، نگࣹينْ دٗورْ لࣹينْ اَيْ دࣷجْ بَ ݧُ دࣹي، ندَخْ ندَوْ سِ مࣷو يُوخُوُلْ وࣷونْ وَلّ، تࣹ مُيْ ݖَ بِيرْ دࣴڪّ بَ، تࣹ وَا جِ اِتْ مࣷو جَبَرࣷو جَبَرُ مࣷرࣷومُ گٗورَمْ؞ نࣷونُ نگࣹينْ دِ تࣹنقࣹي لُ بࣷنْ ݖِ سࣹينْ بِيرْ؞
‏25 سُ فࣹڪّࣹي نࣹ ݖِ اࣵلّ بِ لَ وَا جِ دَجࣹيڪْ ندَوْ، سِ دِگࣷوڪْ ڪࣹنࣹينْ اَبْ سࣺيْ، مُ جࣵݒّ ڪࣷ، تࣴدّࣹ ڪࣷ، نَگْ، وَا جِ ڪࣷ تࣴدّࣹ دࣷݝّ لَ دࣹي دِ اࣵتّࣹيمْ؞ ‏26 وَايࣹ ندَوْ سِ دࣹيسُ ڪࣷ دࣹفْ دَرَ؞ بࣵڪَّارْ بُ دࣹي دِ اࣵتّࣹيمْ تࣷݒُّ ڪࣷ ݖَ، ندَخْ نِ ڪُ سࣷنگّ مࣷرࣷومَمْ، بٗومْ ڪࣷ رࣹڪّ، نِ لَ مبِرْ مࣷومُ دࣹمࣹ، ‏27 ندࣹ بِيرْ اࣵلّ بَ مُ دَجࣹيڪْ ندَوْ سَ، بُ ندَوْ سَ يُوخُ وࣷونْ اِتْ، دُ اَمْ ڪُ ڪࣷ وَلُّ؞
‏28 ڪُ دَجࣹيڪْ اَسْ ندَوْ، تࣹ مُ دِبْ جَنقّ بُ دِگࣷووُلْ اَڪْ ڪࣹنࣹينْ اَبْ سࣺيْ، سُ ڪࣷ جࣵݒّࣹي، تࣴدّࣹ ڪࣷ، بَ ݧُ گِسْ لࣹينْ، ‏29 ڪࣷوڪَ تࣴدّࣹ ندَوْ سَ، دَيْ جࣷخْ بَايُ ندَوْ سَ جُرٗومْ فُڪِّ دࣷنجِ خَالِسْ؞ جَبَرَمْ لَيْ دࣷونْ اِتْ، گَنَّاوْ مࣷو ڪࣷ جَبَرࣷو، تࣹ دُ ڪࣷ سَݧَ فَسࣹ، گِيرُ دُنْدَمْ؞
