مبِرُمْ سَنْتَانࣹ يُ سَرْخَلْڪَتْ يِ سَسࣷو
(سَارْ 6‏.1—7‏.21)
لُ جࣴمْ ݖِ سَرَخِ رࣹنْدِ-دٗومَلْ
6
‏1 اَجِ سَخْ جِ وَخَاتْ مُوسَىٰ نࣹ ڪࣷ:‏ ‏2 «سَنْتَلْ هَـٰرُونَ اَڪْ دࣷومَمْ يُ گٗورْ، نࣹ لࣹينْ لِ ݧُ دࣷگَلْ ݖِ سَرَخُ رࣹنْدِ-دٗومَلْ مࣷويْ لِي:‏ اَبْ سَرَخُ رࣹنْدِ-دٗومَلْ دَيْ فَنَانْ فِ ݧُ ڪࣷيْ لَڪّࣹ ݖِ ڪَوْ سَرْخَلُڪَايْ بِ، بَ بࣴتْ سࣹتْ، سَوَرَ وَيْ وࣺيࣹ تࣵڪّ ݖَ ڪَوْ سَرْخَلُڪَايْ بَ؞ ‏3 سَرْخَلْڪَتْ بِ دَيْ سࣷلْ مبُبَّمْ مِ ݧُ رࣵبّࣹ وࣴݧْ گُ اࣴݖُّ¤ بَ سࣹوْ اَڪْ تُبࣺيِ نجِيتْلاَيَمْ جِ ݧُ رࣵبّࣹ وࣴݧْ گُ اࣴݖُّ بَ سࣹوْ؞ نَ تࣷنّࣹي ݖَ سَرْخَلُڪَايْ بَ دٗومُ سَرَخُ رࣹنْدِ-دٗومَلْ بِ، دَلْدِ تࣹگْ دٗومْ بِ ݖِ وࣹتُ سَرْخَلُڪَايْ بِ؞ ‏4 سُ ڪࣷ دࣹفࣹي مُ سُمِّ يࣺرࣹيمْ يࣷويُ، سࣷلْ يࣹنࣹينْ تࣹ گࣺنّࣹ دٗومْ بِ دَلْ بِ، يٗبُّ ڪࣷ فُ مُݖّ سࣷبࣹ؞ ‏5 نَ سَوَرَسْ سَرْخَلُڪَايْ بِ وࣺيࣹ تࣵڪّ تࣹ دُ فࣹيْ؞ سُبَ سُ نࣹڪّ نَ ڪࣷ سَرْخَلْڪَتْ بِ خَمبّ، بَ نࣷݒِّ تࣹگْ ݖِ سَرَخُ رࣹنْدِ-دٗومَلْ بِ، تࣹ نَ لَڪّ نࣹبّࣷنِ سَرَخِ ݖَنتّ گِ ݖِ بِيرْ جࣵمّ؞ ‏6 نَ سَوَرَ وࣺيࣹ تࣵڪّ فࣵوّ ݖِ ڪَوْ سَرْخَلُڪَايْ بِ، تࣹ بُمُ فࣹيْ؞
لُ جࣴمْ ݖِ سَرَخِ ݒࣹݒّ
‏7 «لِ ݧُ دࣷگَلْ ݖِ سَرَخِ ݒࣹݒّ مࣷويْ لِي:‏ نَ ڪࣷ دࣷومِ هَـٰرُونَ يُ گٗورْ جࣷخࣹ فَ ڪَنَمْ اَجِ سَخْ جِ ݖَ سَرْخَلُڪَايْ بَ؞ ‏8 بَرْݖِ-لࣷخࣷبْ سُنگُفْ سُ مُݖّ اَيِبْ اࣵندّ اَڪْ دِوَمْ لࣹيسْ ݖِيْ تِبّࣹ، بࣷولࣹ ڪࣷوڪْ مبࣷولࣹمْ ݖُورَايْ لَ ݖَ ڪَوْ سَرَخُ ݒࣹݒّ مَ، دَلْدِ ڪࣷيْ لَڪَّلْ اَجِ سَخْ جِ ݖِ ڪَوْ سَرْخَلُڪَايْ بِ، مُيْ خࣹيݧْ خࣹتُگْ جࣵمّ، دِ سَرَخُ بَاخَنْتَلْ؞ ‏9 لَ ݖَ دࣹسْ هَـٰرُونَ اَڪِ دࣷومَمْ يُ گٗورْ ݧࣷو ڪࣷيْ لࣹڪّ، تࣹ بُݧُ ݖِ بࣷولࣹ لَوِيرْ؞ نَݧُ ڪࣷ لࣹڪّࣹ ݖِ بࣺرَبْ بُ سࣹلّ؞ ݖِ بِيرْ اࣴتُّ خَيْمَبْ ندَجࣹ مِ لَݧُ ڪࣷيْ لࣹڪّࣹ؞ ‏10 دࣹيسُ ڪࣷ لَڪَّالࣹيڪْ لَوِيرْ؞ سࣹينْ ݖࣺرْ لَ بُ لࣹينْ اَجِ سَخْ جِ سࣺدّ ݖِ سَرَخِ سَوَرَامْ؞ لُ سࣹلَّ سࣹلّ لَ، مࣹلْ نِ سَرَخُ ݒٗوتُمْ بࣵڪَّارْ اَڪْ سَرَخُ ݒࣹيُگْ تࣷوݧْ؞ ‏11 ڪࣺݒّ ڪُيْ گٗورْ ݖِ دࣷومِ هَـٰرُونَ يِ مَنْ نَ ݖࣹي لࣹڪّ؞ سࣹينْ ݖࣺرْ لَ بُ ݧُ لࣹينْ سࣺدّࣹ ݖِ سَرَخِ سَوَرَ يِ ݧࣹيلْ اَجِ سَخْ جِ، تࣹ دࣷگَلَلْ لࣹينْ ڪࣷ فࣵوّ؞ لࣺݒّ لُ ݖِ لاَلْ دَيْ دࣷونْ لُ سࣹلَّ سࣹلّ؞»
‏12 اَجِ سَخْ جِ وَخْ نَ مُوسَىٰ بَ تࣹيْ نࣹ ڪࣷ:‏ ‏13 «لِ هَـٰرُونَ اَڪِ دࣷومَمْ يُ گٗورْ دِ سَرْخَلَلْ اَجِ سَخْ جِ، ڪࣹرࣷوگْ بُ ݧُ ڪࣷيْ دِوْ، فَلْ ڪࣷ، مࣷويْ لِي:‏ ݧࣹتِّ ڪِلࣷيْ سُنگُفْ سُ مُݖّ اَيِبْ، تࣷلّࣷوڪْ سَرَخُ ݒࣹݒّ بِ ݧُ سَخࣷو بࣺسْ بُ نࣹڪّ، تࣹ گࣹنّ-وࣵلّ گِ دِ سُبَ، گࣹنّ-وࣵلّ گِ نگࣷونْ؞ ‏14 سَافُڪَايُ وࣹݧْ لَݧُ ڪࣷيْ لَڪّࣹ، خِيوَالࣹ ڪࣷ دِوْ، اِنْدِ خِيوْ بَ، مُيْ سَرَخُ ݒࣹݒّ؞ نَݧُ ڪࣷ دࣹفْ اَيْ دࣷگْ يُ ݧُيْ لَڪَّلْ اَجِ سَخْ جِ، مُيْ خࣹيݧْ خࣹتُگْ جࣵمّ؞ ‏15 سَرْخَلْڪَتْ بِ ݧُيْ فَلْ ݖِ دࣷومِ هَـٰرُونَ يُ گٗورْ يِ، مُ وَرْ ڪࣷو وُوتُ، دَ ڪࣷيْ دࣹفْ مࣷومْ اِتَمْ؞ اَجِ سَخْ جِ مࣷو جَگࣷو سَرَخْ بࣷوبُ فࣵوّ، تࣹ دَݧُ ڪࣷيْ لَڪّ بَ مُ جࣹيخْ؞ ‏16 تࣹ اِتْ مبࣷولࣹمْ سَرَخُ ݒࣹݒّ بُ سَرْخَلْڪَتْ دِ دࣹفَلْ بࣷݒَّمْ، دࣹيسْ ڪࣷيْ لَڪّ بَ مُ جࣹيخْ؞ دࣹيسُ ݖِ لࣹڪّ؞»
لُ جࣴمْ ݖِ سَرَخْ سُ بࣵڪَّارْ وَرَلْ
‏17 اَجِ سَخْ جِ وَخَتِ مُوسَىٰ نࣹ ڪࣷ:‏ ‏18 «وَخَلْ هَـٰرُونَ اَڪْ دࣷومَمْ يُ گٗورْ نࣹ لࣹينْ:‏ دࣷگَلْ بِ ݖِ سَرَخُ ݒٗوتُمْ بࣵڪَّارْ مࣷويْ لِي:‏ جُرُ سَرَخَسْ ݒٗوتُمْ بࣵڪَّارْ دَݧُ ڪࣷيْ رࣹنْدِ فَ ݧُ وَرَ رࣹنْدِ جُرُ سَرَخُ دٗومَلْ، فِ ڪَنَمْ اَجِ سَخْ جِ؞ لُ سࣹلَّ سࣹلّ لَ؞ ‏19 سَرْخَلْڪَتْ بِ ڪࣷيْ رࣹنْدِ، مُيْ سَرَخُ ݒٗوتُمْ بࣵڪَّارْ دَ ݖِيْ لࣹڪّ، تࣹ فُ سࣹلّ لَݧُ ڪࣷيْ لࣹڪّࣹ، ݖِ بِيرْ اࣴتُّ خَيْمَبْ ندَجࣹ مِ؞ ‏20 لࣺݒّ لُ لاَلْ ݖِ يࣵݒّ وِ، دَيْ دࣷونْ لُ سࣹلّ، تࣹ بُ لࣹنّ ݖِ دࣹرࣹتُ سَرَخْ سِ تِسࣹي ݖِ ڪَوِ يࣺرࣹ، دࣹيسْ ڪࣷيْ خَجْ فَ مُ تِسْ، ݖِ بࣺرَبْ بُ سࣹلّ؞ ‏21 سُ فࣹڪّࣹي نࣹ ندَبَلْ خَنْدࣹيرْ لَݧُ تࣷگّࣹ وࣷونْ يࣵݒُّ سَرَخْ سَ، نَݧُ ڪࣷ تࣷجْ؞ بُ دࣷونْ ندَبَلْ خࣵنجَرْ، ݧُ جࣷنجّ ڪࣷ، رَخَسْ ڪࣷ؞ ‏22 ڪࣺݒّ ڪُيْ گٗورْ ݖِ اَسْڪَنُ سَرْخَلْڪَتْ سَݧْ نَ ݖࣹي لࣹڪّ؞ لُ سࣹلَّ سࣹلّ لَ؞ ‏23 وَايࣹ جُرُ سَرَخَسْ ݒٗوتُمْ بࣵڪَّارْ دࣹيسُ ݖِ لࣹڪّ لࣹنّ تࣹ فࣹڪّ دࣹرࣹتَمْ دُگّ ݖَ بِيرْ خَيْمَبْ ندَجࣹ مَ، ݧُ دِ ݖَ اَمَلْ اَگْ نجࣷتْلاَيْ ݖَ بِيرْ بࣺرَبْ بُ سࣹلّ بَ؞ لࣷولُ دُ لُ ݧُيْ لࣹڪّ؞ دࣹيسْ ڪࣷيْ لَڪّ، بَ مُ دِبْ دٗومْ؞
