لُ جࣴمْ ݖِ اَتُمْ نࣷفْلاَيْ
25
‏1 گَنَّاوْ لࣷولُ اَجِ سَخْ جِ وَخْ نَ مُوسَىٰ ݖَ تُنْدُ سِنَيِ نࣹ ڪࣷ:‏ ‏2 «وَخَلْ بࣵنِّ إِسْرَاءِيلْ نࣹ لࣹينْ:‏ بُ نگࣹينْ دُگّࣹي ݖَ رࣺيوْ مَ مَ لࣹينْ جࣷخْ، سُوفْ سَ ݖَ بࣷݒَّمْ وَرْ نَا اَمْ اࣵݒُّ نࣷفْلاَيَمْ نگِرْ اَجِ سَخْ جِ؞ ‏3 دِيرُ جُرٗومْ بࣹنِّ اَتْ نگࣹينْ دِ جِ سࣹينْ تࣷولْ؛ جُرٗومْ بࣹنِّ اَتْ يࣷويُ اِتَمْ نگࣹينْ دِ وࣷلِّ سࣹينْ تٗوڪࣴرُ رࣹسࣹݧْ، دَجَلࣹ مࣹݧّࣹيفْ مَ؞ ‏4 وَايࣹ اَتُمْ جُرٗومْ ݧَارࣹيلْ بَ اَتُمْ نࣷفْلاَيْ لَ وَرَ دࣷونْ:‏ سُوفْ سَ ݖَ بࣷݒَّمْ دَيْ نࣷݒّلُ دࣷݝّ نگِرْ اَجِ سَخْ جِ؞ بُلࣹينْ ݖِ جِ سࣹينْ تࣷولْ، بُلࣹينْ ݖِ وࣷلِّ سࣹينْ تٗوڪࣴرُ رࣹسࣹݧْ؞ ‏5 لَ سَخࣹ ݖَ سࣹينُمْ نگٗوبْ سَخْ، بُلࣹينْ ڪࣷ گٗوبَاتْ تࣹ اِتْ رࣹسࣹݧْ گُ ݧُ وࣷلِّوُلࣷونْ بُلࣹينْ ڪࣷ رَاسَاتُ؞ اَتُمْ نࣷفْلاَيْ لَ وَرَ دࣷونْ:‏ سُوفْ سَ ݖَ بࣷݒَّمْ نَ نࣷݒّلُ اَتُمْ لࣴمّ؞ ‏6 وَايࣹ لࣺݒّ لُيْ سَخَيَايْ ݖِ اَتُمْ نࣷفْلاَيْ سَݧْ نگࣹينْ ڪࣷو لࣹڪّ، يࣹينْ اَڪْ سࣹينْ جَامْ يُ گٗورْ اَڪْ يُ جِگࣺينْ، اَڪْ سࣹينِ سُرْگَ اَڪْ دࣷخَنْدࣺيمْ بُ دَلْ اَڪْ يࣹينْ؞ ‏7 سࣹينْ جُرْ اِتْ ݖَ لَ، بَ ݖَ رَبِ اࣵلّ يَ ݖَ سࣹينْ رࣺيوْ؞ مبࣷولࣹمْ لُ سُوفْ سَ مࣹݧّ، مَنࣹيسْ نَ ڪࣷو لࣹڪّ؞
‏8 «گَنَّاوْ لࣷولُ نَنگࣹينْ وَݧّ بَ ݖِ جُرٗومْ ݧَارِ اَتِ نࣷفْلاَيْ، لࣷولُ دِ جُرٗومْ ݧَارِ اَتْ بَ مُيْ جُرٗومْ ݧَارِ يࣷونْ؞ مُ دࣷونْ سࣹينْ جُرٗومْ ݧَارِ اَتِ نࣷفْلاَيْ، تࣷلُّ ݖِ ݧࣹينتّ فُڪِّ اَتْ اَڪْ جُرٗومْ ݧࣹينتّ؞ ‏9 سُ بࣷوبَا ݖِ فُڪِّ فَنْ ݖَ جُرٗومْ ݧَارࣹيلُ وࣹيرْ وَ نَنگࣹينْ وَلْ-لُ اَگْ لِيتْ فࣺݒّ؞ بࣺسُبْ نجࣷتْلاَيْ بࣷوبُ نَنگࣹينْ وَلْ اَگْ لِيتْ ݖِ سࣹينُمْ رࣺيوْ بَ مُ دَجْ؞ ‏10 تࣹ نَنگࣹينْ سࣹلَّلْ اَتْ مَ ݖَ تࣷݒّ دِ جُرٗومْ فُڪّࣹيلُ اَتْ مَ، نگࣹينْ يࣺينࣹ ݖَ مبࣷولࣹمْ وَا رࣺيوْ مَ بَ مُ دَجْ، نࣹ نگࣷرࣹيلْ تَخَوْ نَ؞ سࣹينْ اَتُمْ يِوِڪُ لَ، ڪُ نࣹڪّ وَرْ ݖࣹي دࣹلُّ ݖَ سُوفُ وَا ڪࣴرَمْ، ڪُ نࣹڪّ اِتْ وَرَ دࣹلُّ ݖَ لࣵنگّ گَ مُ بࣷڪّ؞ ‏11 سࣹينْ اَتُمْ يِوِڪُ لَ اَتُمْ جُرٗومْ فُڪّࣹيلْ بࣷوبُ دِ دࣷونْ؞ دُنگࣹينْ ݖِ جِ، دُنگࣹينْ ݖِ گٗوبَاتْ لُ سَخࣹ ݖَ سࣹينُمْ نگٗوبْ تࣹ دُنگࣹينْ ݖِ وࣷلِّ اَگْ رࣹسࣹݧْ، دِ ڪࣷ وِتّ؞ ‏12 گَنَّاوْ اَتُمْ يِوِڪُ لَ، نَ دࣷونْ اَتْ مُ نگࣹينْ سࣹلَّلْ؞ سَخَيَايُبْ تࣷولْ نگࣹينْ سَݧَ لࣹڪّ؞
‏13 «اَتُمْ يِوِڪُ مࣷومُ ݖِ نگࣹينْ دِ دࣹلّࣷو ڪُ نࣹڪّ سُوفَمْ؞ ‏14 بُ نگࣹينْ دِ جَايْ سࣹينْ وَا رࣺيوْ سُوفْ نَگْ مبَا نگࣹينْ دِ جࣴنْدࣹ سُوفْ ݖِ سࣹينْ وَا رࣺيوْ، بُلࣹينْ دِ نَخَنْتࣹ؞ ‏15 لِمُ اَتْ يِ دࣷخْ سࣹينْ دِگَّنْتࣹيڪْ اَتُمْ يِوِڪُ مَ مُجّ، نَ دࣴݒّࣷوڪْ نجࣺگْ گِ نگࣹينْ دِ جࣴنْدࣹ سُوفْ ݖِ سࣹينْ وَا رࣺيوْ؞ لِمُ اَتِ مبࣹيْ يِ دࣷخْ سࣹينْ دِگَّنْتࣹيڪْ اَتُمْ يِوِڪُ مَ ݖَ تࣷݒَّاتْ اِتْ، وَرْ نَا دࣴݒّࣷوڪْ نجࣺگْ گَ ݧُ لࣹينْ جَايࣹ؞ ‏16 لُ اَتْ يࣷويَ گࣴنَ بَرࣹ، نجࣺگْ گَ گࣴنَ جَفࣹ؛ لُ اَتْ يَ گࣴنَ نࣺيوْ، نجࣺگْ گَ گࣴنَ يࣷمبّ، ندَخْتࣹ لِمُ مࣹݧّࣹيفُمْ سُوفْ سَ لࣹيسْ لࣹينْ ڪࣷيْ جَايࣹ؞ ‏17 بُلࣹينْ دِ نَخَنْتࣹ، وَايࣹ رَگَلْ-لࣹينْ سࣹينْ يࣵلَّ، ندَخْ مَنْ اَجِ سَخْ جِ مَايْ سࣹينْ يࣵلَّ؞
‏18 «سَخْلࣹينْ ݖِ سَمَيْ دࣷگَلْ، سࣵمّ سَمَيْ ندِگَلِ يࣷونْ، دِ ڪࣷ جࣴفࣹ، ندَخْ نگࣹينْ دࣴڪّ اَڪْ جࣵمّ ݖِ رࣺيوْ مِ؞ ‏19 ڪࣷنْ سُوفْ سِ دِنَ نَنگُ، نگࣹينْ دِ لࣹڪّ بَ سُورْ تࣹ دࣴڪّ فَ ݖِ جࣵمّ؞ ‏20 جࣷمْبُلْ نگࣹينْ دِ وَخْ نَانْ:‏ ‹لُ نُيْ لࣹڪّ نَگْ ݖِ اَتُمْ جُرٗومْ ݧَارࣹيلْ بَ، گَنَّاوْ دُنُ ݖِ جِ تࣹ دُنُ ݖِ دَجَلࣹ سُنُ مࣹݧّࣹيفْ؟› ‏21 وَايࣹ دِنَا لࣹينْ يٗنّࣹي سَمَ بَرْڪࣹ ݖَ اَتُمْ جُرٗومْ بࣹنّࣹيلْ بَ، سُوفْ سِ نَنگُلْ لࣹينْ مࣹݧّࣹيفُمْ ݧࣹتِّ اَتْ؞ ‏22 بُ نگࣹينْ تࣷلّࣷو ݖِ جِ ݖِ اَتُمْ جُرٗومْ ݧࣹتّࣹيلْ بَ، دِنَ لࣹينْ فࣹڪّ نگࣹينْ دِ لࣹڪّ مࣹݧّࣹيفُمْ دَاوْ-جࣺيگْ، تࣹ مࣹݧّࣹيفُمْ اَتُمْ جُرٗومْ ݧࣹينْتࣹيلْ بَ دِنَ اَگْسِ، فࣹڪّ نگࣹينْ دِ لࣹڪّ ݖَ مُ دَاوْ-جࣺيگْ مَ؞
لِيْ سَݧْ-سَݧْ ݖِ مبِرُمْ سُوفْ سُ ݧُيْ جࣷتَاتْ
‏23 «سُوفْ سِ دࣹيسُ ڪࣷ جَايْ، مُ وࣹلْ فࣵوّ، ندَخْ مَايْ بࣷرࣷومْ سُوفْ سِ، يࣹينْ اَيْ دࣷخَنْدࣺيمْ نگࣹينْ فِ، يُ مَ فِ دࣴڪَّلْ؞ ‏24 فࣺݒّ فُ بࣷڪّ ݖِ سࣹينُمْ رࣺيوْ، نَنگࣹينْ فَ يࣷونَلْ سَݧْ-سَݧُ جࣷتَاتْ ݖَ سُوفْ سَ؞
‏25 «بُ سَ مبࣷڪّ دࣹمࣹي بَ ݧࣵڪّ تَخْ ڪࣷو جَايْ ݖِ سُوفَمْ، مبࣷڪَّمْ مِ ڪࣷ گࣴنَ جࣹگࣹ تࣹ اَمْ سَݧْ-سَݧُ جࣷتَاتْ سُوفَمْ، نَ جࣴنْدَاتْ لَ مبࣷڪّ مَ جَايࣷونْ؞ ‏26 ڪُ اَمُلْ ڪُ ڪࣷ ڪࣷ جࣷتَاتَلْ تࣹ مࣷومْ ݖِ بࣷݒَّمْ مُ مُجّ وࣷومْلࣹ، بَ اَمْ لُ مُ ڪࣷ جࣷتَاتࣹ، ‏27 دَيْ وَݧّ اَتْ يَ نجَايْ مَ اَمْ، دࣹلّࣷو ڪِ مُ جَايࣷونْ اَتِ مبࣹيْ يَ ݖَ ڪࣷوڪَ دࣹسࣹ، دَلْدِ مَنَ جࣷتَاتْ ݖِ سُوفَمْ؞ ‏28 وَايࣹ سُ اَمُلْ لُ مُ جࣴنْدَاتࣹ سُوفَمْ، سُوفْ سَ دَيْ دࣹسْ ݖَ لࣷخࣷلْ ڪَ مُ جَايࣷونْ، بَ ڪࣹرࣷوگْ اَتُمْ يِوِڪُ مَ دࣺلْسِ؞ بُ اَتُمْ يِوِڪُ مَ دࣺلْسࣹي، سَݧْ نَا دࣹلُّ ݖَ سُوفَمْ؞
‏29 «سُ دࣹي وَا جُ جَايْ اَبْ دࣴڪُّوَايْ بُ نࣹڪّ ݖِ دࣴڪّ بُ مَگْ بُ ݧُ تَبَخْ تَتَ وࣴرَلࣹي ڪࣷ ڪࣷ، دِنَ ڪࣷ سَݧَ جࣷتَاتْ لِ فࣹيڪْ نجَايْ مِ اَمُلْ اَتْ؞ ݖِ دِيرْ بࣷوبُ لَ ڪࣷ سَݧَ جࣷتَاتࣹ؞ ‏30 سُ ݧُ ڪࣷ جࣷتَاتُلْ بَ نجَايْ مَ اَمْ اَتُمْ لࣴمّ، دࣴڪُّوَايْ بࣷوبُ ݖِ بِيرْ دࣴڪّ بُ مَگْ بَ ݧُ وࣴرَلࣹي تَتَ دَيْ وࣹلْ فࣵوّ، ݧࣹيلْ ڪَ ڪࣷ جࣴندّ، مࣷوڪْ ڪُ اَسْڪَنࣷو ݖِ مࣷومْ؞ دُ لࣹيسْ دِ دࣹلّࣷو ݖِ اَتُمْ يِوِڪُ؞ ‏31 سُ دࣹي اَيْ دࣴڪُّوَايْ ݖِ دࣴڪّ يُ ندَوْ يُ ݧُ تَبَخُلْ تَتَ، وࣴرَلࣹي ڪࣷ ڪࣷ، دࣹيسْ ڪࣷيْ يࣹمَلࣹيڪْ اَلَلِ سُوفْ؞ سَݧࣹيسْ نَ ڪࣷو جࣷتَاتْ تࣹ دࣹيسْ ڪࣷيْ دࣹلّࣷو ݖِ اَتُمْ يِوِڪُ؞
‏32 «سُ دࣹي دࣴڪِّ لࣹوࣹينْ ݧِ اَڪْ سࣹينْ دࣴڪُّوَايْ يَ ݖَ بِيرْ نَگْ، لࣹوࣹينْ ݧِ سَݧْ نَݧُ ݖࣹي جࣷتَاتْ فࣵوّ؞ ‏33 ڪࣷنْ اَلَلُ لࣹوࣹينْ مَنࣹيسْ نَ ڪࣷو جࣷتَاتْ —دࣴڪُّوَايْ لَيْ دࣷونْ بُ ݧُ جَايْ ݖِ دࣴڪِّ لࣹوࣹينْ ݧِ— دࣹيسْ ڪࣷيْ دࣹلّࣷو بُ اَتُمْ يِوِڪُ دِڪّࣹي، ندَخْ دࣴڪُّوَايْ يِ ݖِ دࣴڪِّ لࣹوࣹينْ ݧِ، سࣹينْ اَلَلْ لَ جُ لࣹينْ لࣹوْ فࣵوّ ݖِ سࣹينْ بِيرْ بࣷڪِّ بࣵنِّ إِسْرَاءِيلْ؞ ‏34 وَايࣹ سُوفَسْ ݒَرْلُ يِ بࣷڪّ ݖِ دࣴڪِّ لࣹوࣹينْ ݧِ، دࣹيسُ ڪࣷ جَايْ، ندَخْ ݧࣷو ڪࣷيْ مࣷومْ فࣵوّ؞
بَاخࣹلࣹينْ اَجِ ݧࣵڪّ
‏35 «بُ سࣹينْ مبࣷڪُّمْ بࣵنِّ إِسْرَاءِيلْ ݧࣵڪّࣹي بَ مَنَتُلَ فࣹيْ لُ مُ لࣹينْ اَمࣹيلْ، نَنگࣹينْ ڪࣷ دِمْبَلِ، نِ بُ دࣷونْ دࣷخَنْدࣺيمْ مبَا مࣵنگَانْ، ندَخْ مُ مَنَ دُندّ ݖِ سࣹينْ بِيرْ؞ ‏36 بُلࣹينْ تࣹگْ لࣹنّ لُ مُ مَنَ دࣷونْ ݖِ بࣷرْ بُ لࣹينْ سࣹينْ مبࣷڪّ مࣷومُ اَمࣹيلْ، وَايࣹ رَگَلْ-لࣹينْ ݖِ سࣹينْ يࣵلَّ، ندَخْ مُ مَنَ دُندّ ݖِ سࣹينْ بِيرْ؞ ‏37 بُلࣹينْ ڪࣷ لࣹبَلْ سࣹينْ خَالِسْ، دِ ݖَ تࣹگْ لࣹنّ، تࣹ بُلࣹينْ ڪࣷ جࣷخْ دُندّ، دِ سࣵڪُّ لُ مُ ݖَ يࣷڪّ نگِرْ دࣹلّࣷو لࣹينْ ڪࣷ؞ ‏38 مَنْ اَجِ سَخْ جِ مَايْ سࣹينْ يࣵلَّ جِ لࣹينْ گࣺنّࣹ رࣺيوُمْ مِسْرَ، جࣷخْ لࣹينْ رࣺيوُ ڪَنَانْ، تࣹ دِ سࣹينْ يࣵلَّ؞
لِيْ سَݧْ-سَݧْ ݖِ جࣷتَاتْ نِتْ
‏39 «بُ سࣹينْ مبࣷڪّ ݧࣵڪّࣹي ݖِ سࣹينْ بِيرْ، بَ جَايْ لࣹينْ بࣷݒَّمْ، بُلࣹينْ ڪࣷ جَامْلࣷو؞ ‏40 دࣹفْلࣹينْ ڪࣷ نِ لِگّࣺييْڪَتْ بُيْ فࣹيࣹيڪُ مبَا مࣵنگَانْ بُ نگࣹينْ دࣴڪَّلْ؞ نَ لࣹينْ لِگّࣺييَلْ بَ ڪࣹرࣷوگْ اَتُمْ يِوِڪُ مَ، ‏41 مُ گࣺنّ ݖِ سࣹينِ لࣷخࣷ، مࣷوڪْ اَيْ دࣷومَمْ، دَلْدِ دࣹلُّ ݖِ لࣵنگُ بࣷݒَّمْ، فَ اَيْ مَامَمْ سࣺدّࣷو؞ ‏42 لِ وَرَلْ لࣷولُ مࣷو دِ بࣵنِّ إِسْرَاءِيلْ دَݧُ دِ سَمَ جَامْ يِ مَ گࣺنّࣹ رࣺيوُمْ مِسْرَ؞ دࣹيسُ لࣹينْ جَايْ نجَامْ؞ ‏43 بُلࣹينْ لࣹينْ سࣷخࣷرࣹ ݖِ بِيرْ ڪِلِفْتࣹيفْ، وَايࣹ رَگَلْ-لࣹينْ ݖِ سࣹينْ يࣵلَّ؞ ‏44 سࣹينْ جَامْ يُ گٗورْ اَڪْ سࣹينْ جَامْ يُ جِگࣺينْ، نَنگࣹينْ لࣹينْ جࣴلࣹ ݖِ خࣹيتْ يِ لࣹينْ وࣴرْ؞ ݖِ ݧࣷوݧُ نگࣹينْ دِ جࣴنْدࣹ اَيْ جَامْ، گٗورْ اَڪْ جِگࣺينْ؞ ‏45 وَلَّ نگࣹينْ جࣴنْدࣹ بَ تࣹيْ ݖِ دࣷخَنْدࣺيمْ يِ ݖِ سࣹينْ بِيرْ، مبَا ݖِ نجَبࣷوتْ يِ دࣷخَنْدࣺيمْ يِ اَمْ ݖِ رࣺيوْ مِ تࣹ ݧُ نࣹڪّ ݖِ سࣹينْ بِيرْ؞ سُ ڪࣷ دࣹفࣹي ݧُ دࣷونْ سࣹينْ نگࣵلّࣷ؞ ‏46 سَݧْ نگࣹينْ لࣹينَ دࣷنَلࣹ سࣹينْ دࣷومْ يِ لࣹينْ دِ وُوتُ، ݧُ دࣷونْ سࣹينْ نگࣵلّࣷ سࣹينْ گِيرُ دُندّ؞ ݧࣷوݧُ سَݧْ نگࣹينْ لࣹينَ دࣹفْ اَيْ جَامْ، وَايࣹ سُ دࣹي ݖِ سࣹينْ دِگَّنْتࣹ، يࣹينْ بࣵنِّ إِسْرَاءِيلْ، بُلࣹينْ دِ سࣷخࣷرَنْتࣹ ݖِ بِيرْ ڪِلِفْتࣹيفْ؞
‏47 «سُ فࣹڪّࣹي نࣹ اَبْ دࣷخَنْدࣺيمْ بُ نگࣹينْ دࣴڪَّلْ مبَا اَبْ مࣵنگَانْ دَفَ وࣷومْلࣹ تࣹ ڪࣹنّ ݖِ سࣹينِ بࣷڪّ ݧࣵڪّ، بَ جَايْ بࣷݒَّمْ دࣷخَنْدࣺيمْ بِ نگࣹينْ دࣴڪَّلْ مبَا ڪࣹنّ ڪُ بࣷڪّ اَڪْ دࣷخَنْدࣺيمْ بِ، ‏48 گَنَّاوْ بُ جَايࣹي بࣷݒَّمْ، مَنࣹيسْ نَ ڪࣷو جࣷتَاتْ بَ تࣹيْ؞ ڪࣹنّ ݖِيْ بࣷڪَّمْ سَݧْ نَ ڪࣷو جࣷتَاتْ، ‏49 يَارْ مُيْ بَايَمْ بُ ندَوْ مبَا دࣷومُ بَايَمْ بُ ندَوْ؞ ڪُ مُ بࣷڪَّلْ لࣵنگّ دِ مبࣷڪَّمْ لࣹنقࣹ سَخْ مَنْ نَ ڪࣷو جࣷتَاتْ، تࣹ مࣷومْ ݖِ بࣷݒَّمْ مَنْ نَا وࣷومْلࣹ بَ جࣷتَاتْ بࣷݒَّمْ؞ ‏50 نَ بࣷڪّ اَڪْ ڪَ ڪࣷ جࣴندّ وَݧّ، لَ دَلࣹ ݖَ اَتْ مَ مُ جَايࣹ بࣷݒَّمْ، بَ ݖَ اَتُمْ يِوِڪُ مَيْ ݧࣴوْ، مُيْ نجࣺگُ گࣷرࣹيلَمْ، دࣴݒّࣷوڪْ اَتْ يࣷويَ؞ تࣹ اِتْ نَݧُ ݖِ سࣹيتْ اࣵقُ ݧَقُ لِگّࣺييْڪَتْ؞ ‏51 سُ دࣹسࣹي اَيْ اَتْ، لُ اَتْ يَ گࣴنَ بَرࣹي، نجࣺگْ گَ مُيْ جࣷتَاتࣹ بࣷݒَّمْ گࣴنَ سࣷرࣹ ݖِ نجࣺگْ، گَ ݧُ ڪࣷ جࣴنْدࣹ وࣷونْ؞ ‏52 تࣹ سُ اَتْ يُ نࣺيوْ دࣷخࣹي سࣹينْ دِگَّنْتࣹيڪْ اَتُمْ يِوِڪُ مَ، نَ ڪࣷ وَݧّ تࣹ فࣹيْ لُ نِ نࣺيوࣹ، نگِرْ جࣷتَاتْ بࣷݒَّمْ؞ ‏53 اَبْ سُرْگَ لَيْ دࣷونْ ݖِ نجَاتِگࣹيمْ اَتْ مُ نࣹڪّ، وَايࣹ وَرُ ڪࣷو سࣷخࣷرࣹ ݖِ بِيرْ ڪِلِفْتࣹيفْ، نگࣹينْ دِ سࣹيتَانْ؞ ‏54 سُ فࣹڪّࣹي نࣹ جࣷتَاتُݧُ ڪࣷ ݖِ جࣷتِنْ يࣷويُ سَخْ، وَرْ نَا مࣷومْ بࣷݒَّمْ مࣷوڪِ دࣷومَمْ ݖَ اَتُمْ يِوِڪُ مَ؞ ‏55 ندَخْ ڪَتْ مَنْ لَ بࣵنِّ إِسْرَاءِيلْ دِ سَمَيْ جَامْ؞ ݧࣷو دِ سَمَ جَامْ يَ مَ گࣺنّࣹ رࣺيوُمْ مِسْرَ؛ يࣹينْ، مَايْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ؞
