2
‏1 بࣴگّ نَا نَگْ، نگࣹينْ خَمْ خَرࣹ بُ مࣹتِّ، بِ مَ لࣹينْ دِ خَرࣹيلْ، يࣹينْ اَڪْ وَا لَوْدِسࣹ اَڪْ مبࣷولࣹمْ گࣴمْڪَتْ ݧِ مَ مَسُلَ تࣹگْ بࣴتْ، ‏2 نگِرْ نگࣹينْ گࣴنَ اَمْ فِتْ، مࣵنڪࣷو ݖِ بِيرْ ݖࣷفࣹيلْ، تࣹ گࣺيجُ ݖِ گِسْ-گِسُ خࣷلْ بُ لࣹيرْ نࣵݧّ تࣹ وٗورْ، بَ مَنَ دَجْ دࣴگّ، مبٗوتُمْ يࣵلَّ، مِيْ اَلْمَسِ ݖِ بࣷݒَّمْ؞ ‏3 ݖِ مࣷومْ لَ مبࣷولࣹمْ اَلَلْ جِ ݖِ خࣹلْ اَڪْ خَمْ-خَمْ، دࣴخࣴݧࣹ؞ ‏4 لࣷولُ لاَ لࣹينْ دِ وَخْ، نگِرْ ڪࣹنّ بَݧْ لࣹينَ رࣺيرَلࣹ اَيْ ڪࣵدُّيْ نَخࣹ يُ مبُبّࣷو اَگْ دࣴگّ؞ ‏5 ندَخْ جࣴمّ جِ ڪَتْ تࣹيوُلْ، وَايࣹ خࣷلْ بَا نگࣹيڪْ يࣹينْ، دِ بࣺگࣹي گِسْ سࣹينْ دࣷخَلِنْ وُ جࣹڪّ، اَڪْ سࣹينْ نگࣴمْ گُ وࣺرْ گُ نگࣹينْ گࣴمْ اَلْمَسِ؞
جࣴفࣹلࣹينْ نِ ݧُ بࣷڪّ اَڪْ اَلْمَسِ
‏6 ڪࣷنْ نَگْ، نِ نگࣹينْ نَنگࣷو اَلْمَسِ عِيسَىٰ دِ سَنگّ بِ، اࣵندّلࣹينْ اَڪْ مࣷومْ، جࣴفࣹ نࣷونُ؞ ‏7 سَمݒّلࣹينْ سࣹينِ رࣹينْ ݖِ مࣷومْ، تَبَخُلࣹينْ ݖِ مࣷومْ، فࣹدّلِڪُلࣹينْ ݖِ وࣵلُّ نگࣴمْ، نِ ݧُ لࣹينْ ڪࣷ جࣵنگَلࣹ، بَ سࣹينُبْ خࣷلْ بَاوَانَلْ ڪࣵدُّيْ ݖَنتّ؞
‏8 مࣷيْتُلࣹينْ بَ ڪࣹنّ بَݧْ لࣹينَ ڪࣴفࣹ ڪࣵدُّيْ نَخِ نࣹينْ يُ ݧُ جࣵݒّࣹ خَمْ-خَمْ بُ خࣹلُ، ڪࣵدُّ يُ سُڪَّنْدِڪࣷووُلْ اَلْمَسِ، خَنَا آدَيْ نِتْ، اَڪْ ݖࣴسْلاَيِ اࣵدِّنَ؞
‏9 ندَخْ ڪَتْ ݖِ اَلْمَسِ، ݖِ يَرَمُ نِتَمْ لَ لࣺݒِّ يࣵلَّ دࣴڪّࣹ، بَ مَتْ سࣴڪّ ݖِ مࣷومْ، ‏10 ݖِ مࣷومْ نگࣹينْ مَتࣹ اِتْ، تࣹ مࣷويْ ڪِلِفَيْ مبࣷولࣹمْ بࣷرࣷومْ نگُورْ يِ، اَڪْ مبࣷولࣹمْ بࣷرࣷومْ سَݧْ-سَݧْ يِ؞ ‏11 ݖِ مࣷومْ نگࣹينْ خَرَفࣹ خَرَفْ گُ لࣷخࣷلْ نِتْ دࣹفُلْ، ندَخْ نࣷونُ ڪُيْ خَرْفَلْ نِتْ ڪِ دِ دࣷگࣹ اَسْ لࣴفْ ݖِ يَرَمُ نِتْ ڪِ، نِ لَ اَلْمَسِ دِ دࣷگࣹ بَ مُ سࣹتْ، بُومُ نجَامْ گِ لࣹينْ دࣷولࣹيْ بࣵڪَّارْ تࣴنڪࣹ وࣷونْ؞ نࣷونُ لَ اَلْمَسِ دِ خَرْفَلࣹي؞ ‏12 نِ ݧُ لࣹينْ بࣷولࣹيڪْ مࣷومْ سُولْ، ڪࣹرࣷوگْ بَ ݧُ لࣹينْ سٗوبࣹي ݖِمْ ندࣷخْ، نࣷونُ نگࣹينْ دࣹڪّࣹينْدࣷو اَڪْ مࣷومْ، ندَخْ يࣹينَ وٗولُ دࣷولࣹيْ يࣵلَّ مِ ڪࣷ دࣹڪَّلْ؞
‏13 يࣹينْ ݧِ دࣷونࣷونْ ندࣹي فَ ڪَنَمْ يࣵلَّ ندَخْ سࣹينِ مࣷيْ، اَڪْ سࣹينْ ݧࣵڪَّ خَرَفْ، ندَخْ لِ نگࣹينْ دُلْ يَوُتْ، يࣵلاَّ لࣹينْ بࣷولࣹيڪْ اَلْمَسِ دُنْدَلاَتْ لࣹينْ؞ مࣷو نُ بَالْ سُنُ مࣷيْ يࣺݒّ؞ ‏14 ڪَيِتُ بࣷرْ بِ نُ تِيݧَلࣷونْ، تࣹ دࣷگَلِ يࣷونْ تࣷݒّࣹ نُ ڪࣷ، مࣷومْ مࣷو فَرْ مبِندّ مَ، يٗبُّ ڪَيِتْ گَ، دَاجَالࣹ ڪࣷ اَڪْ اَلْمَسِ ݖَ بَنتّ بَ؞ ‏15 مࣷو دࣷگْ سِدِّتِ بࣷرࣷومْ نگُورْ يࣹيڪْ بࣷرࣷومْ سَݧْ-سَݧْ يِ ݖِ جَوّ جِ، ݧُ دࣷونْ اَيْ جَامْ يُ مُ جࣵݒّࣹ خَرࣹ؞ مࣷو لࣹينْ وࣹيرْ اِتْ ݖِ ڪَنَمْ ݧࣺݒّ، دِرِوَالࣹ لࣹينْ ݖَ بِيرْ نجَابْ مَ تࣷݒّ ݖَ ندَمْ، لَ اَلْمَسِ اَمࣹ ݖَ بَنتّ بَ؞
‏16 ڪࣷنْ نَگْ بُ لࣹينْ ڪࣹنّ سِڪَّلْ ݖِ ݧَمْ وُ نگࣹينْ لࣹڪّ، مبَا لُ نگࣹينْ نَانْ، مبَا مࣵگَّلْ بࣺسْ بُ رَفࣹتْ، مبَا تࣹرُتࣹلْ وࣹيرْ، اَڪْ بࣺسُبْ نࣷفْلاَيْ؞ ‏17 لࣷولُ لࣺݒّ تَڪّندࣹيرُ يَرَمْ وُيْ جࣴمَّلْ لُ دِڪَّگُلْ، مࣷومْ رࣹڪّ لَ وࣷونْ؞ وَايࣹ يَرَمْ وِ دࣴگّ مࣷو دِ اَلْمَسِ؞ ‏18 بُلࣹينْ نَنگُ ڪࣹنّ ڪُ لࣹينْ سِڪَّلْ، تࣹ دِ ڪُيْ تࣹگْ بࣷݒَّمْ جࣴفِ تࣷرࣷخْلُ، دِ جَامُ اَيْ مَلاَڪَ، تࣷݒّ سࣹينْ گِسْ-گِسِ ݒࣹيݧُيْ بࣷݒّ؞ ݧُ نِ مࣹلْ ݧࣷويْ رࣺيْ-رࣺيْلࣷو خَلاَتِ نࣹينْ يُ ݧُ فࣹنْتࣹ سࣹينْ يࣺگْ-يࣺگِ بَڪَّنْ؞ ‏19 ݧࣷوݧُ جࣷڪُّوُݧُ ݖِ اَلْمَسِ، مِيْ بࣷݒّ بِ، تࣹ ݖِ بࣷݒّ بِ لَ يَرَمْ وِ وࣺݒّ دِ جࣴفَنْدِڪࣷو سِدِّتْ يࣹيڪْ تࣹنقࣷ يِ، بَ سُوخَتُ، نگِرْ لࣺݒّ مࣵنڪࣷو بَ مَنَ مࣵگّ، نِ ڪࣷ يࣵلَّ دِ مࣵگَّلࣹ؞
‏20 ندࣹگَمْ نَگْ يࣹينَ بࣷڪّ اَڪْ اَلْمَسِ دࣹي، بَ تࣹقْلِڪࣷو اَڪْ لِ اࣵدِّنَ دࣹفْ ݖࣴسْلاَيْ، اَنَ لُ تَخْ نگࣹينْ مࣹلْ نِ ݧُ بࣷڪّ ݖِ اࣵدِّنَ بَ تࣹيْ، دِ تࣹگࣷو اَيْ دࣷگَلِ يࣷونْ يُ مࣹلْ نِي:‏ ‏21 «بُلْ جࣴلْ لِي، بُلْ ݧَمْ لࣹي، بُلْ لاَلْ لَ ݖَ دࣹسْ؞» ‏22 تࣹ مبࣷولࣹمْ يࣴفْ يࣷويُ دࣷگَلِ يࣷونْ يِ جࣴمْ دَيْ جࣹيخْ دࣷݝّ بُ ݧُ لࣹينْ جࣴفَنْدِڪࣷو؞ لࣷولُ اَيْ سَنْتَانࣹ اَڪِ اࣵلُّوَيْ نِتْ دࣷݝّ لَ؞ ‏23 دࣴگّ لَ، خࣹلْ مَنْ نَا نَنگُ دࣷگَلِ يࣷونْ يُ نِ مࣹلْ، ندَخْ لاَجْ نَ نِتْ ڪِ ݒَسْتࣺيفُ ݖِ دِينࣹ جُ مُ تࣹگࣷولْ بࣷݒَّمْ، لاَجْ نَ اِتْ تࣷرࣷخْلُ، اَڪْ ݖࣷونࣷيْ يَرَمْ يُ تُڪّࣹي ݖِ يَرْ بُ مࣹتِّ بُ نِتْ ڪِيْ تࣹگࣷو، وَايࣹ جَرِݧُلْ دَرَ ڪُ بࣴگَّ مࣷومْ بَڪَّنَمْ؞
