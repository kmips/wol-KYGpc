بَتَاخَلْ بُ جࣴڪّ بِ يࣵلَّ مَيْ ݒٗولْ، مُ بِندّ ڪࣷ وَا
تࣹسَلࣷنِگْ
1
‏1 مَنْ لَ ݒٗولْ، مِ اࣵندّ اَڪْ سِلْوَنْ اَڪْ تِمࣷتࣹ ݖِ ڪࣵدُّ گِي مَ يࣺڪَّتِ، ݧࣹيلْ مبࣷولࣷومْ گࣴمْڪَتِ وَا تࣹسَلࣷنِگْ، ݧِ بࣷڪّ ݖِ يࣵلَّ بَايْ بِ اَڪْ سَنگّ عِيسَىٰ اَلْمَسِ؞ يِوْ اَڪْ جࣵمّ ݧࣹيلْ نَ لࣹينْ؞
وَا تࣹسَلࣷنِگْ اَمْ نَݧُ اَوْ يِوْ
‏2 نُ نگِ سَنتّ يࣵلَّ فُ نُ تࣷلُّ ݖِ يࣹينْ ݧࣺݒّ، تࣹ نࣷݒِّوُنُ لࣹينَ فࣵتّلِڪُ ݖِ سُنُيْ ݧَانْ، ‏3 اَڪْ دِ بَامْتُ فِ ڪَنَمْ يࣵلَّ سُنُ بَايْ، جࣴفْ جِ جُدّࣷو ݖِ سࣹينْ نگࣴمْ، اَڪْ لِگّࣺييْ بِ سࣷتّࣹي ݖِ سࣹينْ ݖࣷفࣹيلْ، اَڪْ يَاڪَارْ جِ نگࣹينْ سَخࣷو دِ سࣺينْتُ سُنُ سَنگّ عِيسَىٰ اَلْمَسِ؞
‏4 يࣹينْ بࣷڪِّ گࣴمْڪَتْ ݧِ يࣵلَّ سࣷݒّ، خَمْ نَنُ نࣹ مࣷومْ مࣷو لࣹينْ تَامُ، ‏5 ندَخْ سُنُ خِبَارُ جࣵمّ، بِ نُ لࣹينْ اࣵگّࣹ، دِڪّࣹوُ لࣹينْ اَيْ وَخْ رࣹڪّ، وَايࣹ مَنࣷورࣹ لَ لࣹينْ دِڪَّالࣹ، اَڪْ ندِمْبَلَلْ نࣷو گُ سࣹلّ گِ، اَڪْ ڪٗولُتࣹ گُ مَتْ سࣴڪّ گُ نگࣹينْ اَمْ؞ خَمْ نگࣹينْ مࣷوسْ نَ نُ نࣹڪّࣹ وࣷونْ ݖَ سࣹينْ بِيرْ، دِ خِنْتࣹوࣷو سࣹينْ نجَرِݧْ؞ ‏6 اَوْ نگࣹينْ سُنُيْ تࣵنڪّ، اَوْ اِتْ ݖِ تࣵنڪِ سَنگّ بِ، ندࣹ يࣹينَ جࣷتْ ݖِ ڪࣵدُّ گِ، ݖِ بِيرْ نجࣵقَرࣹ جُ بَرࣹ، تࣹ دِ ڪࣷ بࣺگࣹ ݖِ ندِمْبَلَلْ نࣷو گُ سࣹلّ گِ؞ ‏7 مࣷو لࣹينْ تَخَ دࣷونْ رࣷيُڪَايْ ݖِ مبࣷولࣹمْ گࣴمْڪَتِ دِيوَانِ مَسࣹدُوَنْ اَڪْ اَڪَيِ؞ ‏8 ݖِ سࣹينْ بِيرْ ڪَيْ لَ ڪࣵدُّگْ بࣷرࣷومْ بِ جࣷلّࣹي، تࣹ يࣹمُلْ ݖِ مَسࣹدُوَنْ اَڪْ اَڪَيِ رࣹڪّ، وَايࣹ سࣹينْ نگࣴمْ يࣵلَّ فࣺݒّ لَ سِيوࣹ، بَ سࣷخْلَوُنُ ݖࣹي وَخَتِ فࣹنّ دَرَ؞ ‏9 نِتْ ݧِ ݖِ سࣹينْ بࣷݒّ ݧࣷو نُيْ وَخَلْ فࣺݒّ، نَ سُنُ نگَنْ اَڪْ يࣹينْ دࣹمࣹ، اَڪْ نِ نگࣹينْ وَلْبَتِڪࣷو ݖِ يࣵلَّ، دࣴدُّ تُورْ يِ، بَيْ جَامُ يࣵلَّ جُ دࣴگُّ جِيْ دُندّ، ‏10 تࣹيْ سࣺينْتُ دࣷومَمْ عِيسَىٰ مِ مُ دࣹڪَّلْ، مُ دِڪّ بࣵيِّڪࣷو اَسَمَانْ، مࣷومْ مِ نُ جࣷتْ ݖِ سࣵنجّ مِيْ دِڪّ؞
