لُ جࣴمْ ݖِ سَرَخُ ݒࣹݒّ
2
‏1 «بُ نِتْ دࣹي اِنْدِلْ اَجِ سَخْ جِ سَرَخُ ݒࣹݒّ، سُنگُفْ¤ سُ مُݖّ اَيِبْ لَ سَرَخَمْ وَرَ دࣷونْ، مُ سࣷتِّ ݖِ دِوْ، دࣹفْ ݖِ ݖُورَايْ، ‏2 تࣹ يٗتْ ڪࣷ دࣷومِ هَـٰرُونَ يُ گٗورْ يِيْ سَرْخَلْڪَتْ يِ؞ نَ سَرْخَلْڪَتْ بِ سࣵڪّ بَرْݖِ-لࣷخࣷبْ سُنگُفْ سُ مُݖّ اَيِبْ سࣷوسُ اࣵندّ اَڪْ دِوْ گِ، تࣹ بࣷولࣹ ڪࣷوڪْ ݖُورَايْ لِ لࣺݒّ؞ نَ سَرْخَلْڪَتْ بِ دࣹفْ ݖࣺرْ بࣷوبُ سَرَخُ بَاخَنْتَلْ، لَڪّ ڪࣷ ݖَ ڪَوْ سَرْخَلُڪَايْ بَ، مُيْ سَرَخُ سَوَرَ نگِرْ خࣹتُگْ جࣵمّ، ݧࣹيلْ اَجِ سَخْ جِ؞ ‏3 لِ دࣹسْ ݖِ سَرَخُ ݒࣹݒّ مِ، هَـٰرُونَ اَڪِ دࣷومَمْ ݧࣷو ڪࣷيْ فࣺيتࣹوࣷو، مُيْ ݖࣺرْ بُ سࣹلَّ سࣹلّ، بَوࣷو ݖِ سَرَخِ سَوَرَ يِ ݧࣹيلْ اَجِ سَخْ جِ؞
‏4 «سُ دࣹي سَرْخَلْ لُ ݧُ دࣹفَرࣹ ݒࣹݒّ، دِ لُ ݧُ لَڪّ ݖِبْ تَالْ بَ مُ ݧࣷرْ، نَ دࣷونْ سُنگُفْ سُ مُݖّ اَيِبْ، دِ مبُرُ يُ اَمُلْ لَوِيرْ يُ ݧُ خِيوَالࣹ دِوْ، مبَا مبُرُ يُ سࣹوْ يُ اَمُلْ لَوِيرْ، ݧُ سࣷتِّ دِوْ ݖَ ڪَوَمْ؞ ‏5 بُ سَرَخَمْ دࣹي لُ ݧُ دࣹفَرࣹ ݒࣹݒّ، لَڪّࣹ ڪࣷ ݖِ سَافُڪَايُ وࣹݧْ، نَ دࣷونْ سُنگُفْ سُ مُݖّ اَيِبْ سُ ݧُ خِيوَالࣹ دِوْ تࣹ بُمُ اَمْ لَوِيرْ؞ ‏6 نَ ڪࣷ دَمَّتْ تࣹ سࣷتِّ دِوْ ݖَ ڪَوْ دࣷگْ يَ؞ سَرَخُ ݒࣹݒّ لَ؞ ‏7 بُ سَرَخَمْ دࣹي لُ ݧُ دࣹفَرࣹ ݒࣹݒّ، دِ لُ ݧُ سَافْ ݖِ ݖِنْ بَ مُ ݧࣷرْ، نَ دࣷونْ سُنگُفْ سُ مُݖّ اَيِبْ سُ ݧُ خِيوَالࣹ دِوْ؞
‏8 «نَ يٗبُّلْ اَجِ سَخْ جِ سَرَخُ ݒࣹݒّ سُ ݧُ دࣹفَرْ ݖِ تࣹگْتَلْ يࣷويُ، يٗتْ ڪࣷ سَرْخَلْڪَتْ بِ، مُ يٗبُّ ڪࣷ ݖَ سَرْخَلُڪَايْ بَ؞ ‏9 سُ ڪࣷ دࣹفࣹي سَرْخَلْڪَتْ بِ گࣺنّࣹ ݖِ سَرَخُ ݒࣹݒّ مِ، اَبْ سَرَخُ بَاخَنْتَلْ، دَلْدِ ڪࣷيْ لَڪّ ݖِ ڪَوْ سَرْخَلُڪَايْ بِ، مُيْ سَرَخُ سَوَرَ نگِرْ خࣹتُگْ جࣵمّ ݧࣹيلْ اَجِ سَخْ جِ؞ ‏10 لِ دࣹسْ ݖِ سَرَخُ ݒࣹݒّ مِ، هَـٰرُونَ اَڪِ دࣷومَمْ ݧࣷو ڪࣷيْ فࣺيتࣹوࣷو، مُيْ ݖࣺرْ بُ سࣹلَّ سࣹلّ، بَوࣷو ݖِ سَرَخِ سَوَرَ يِ ݧࣹيلْ اَجِ سَخْ جِ؞
‏11 «دُ بࣹنّ سَرَخُ ݒࣹݒّ بُ ݧُيْ دࣹفَلْ اَجِ سَخْ جِ مُيْ اَمْ لَوِيرْ، ندَخْ دُ لࣹنّ ݖِ لَوِيرْ مبَا لࣹمْ، لُ ݧُ وَرَ لَڪّ، نگِرْ سَرَخُ سَوَرَ سُ ݧࣹيلْ اَجِ سَخْ جِ؞ ‏12 سَرَخِ ندࣷورْتࣹلْ مࣹݧّࣹيفْ مَنْ نَا اࣵندّ اَڪْ يࣷويُ، وَايࣹ دُ بࣷڪّ ݖِ سَرَخْ سُ جࣴمْ ݖَ سَرْخَلُڪَايْ بَ، ݧُ دِ ڪࣷ تَالْ، نگِرْ خࣹتُگْ جࣵمّ؞ ‏13 سࣹينْ مبࣷولࣹمْ سَرَخُ ݒࣹݒّ نَنگࣹينْ ڪࣷ خࣷرࣷمْ؞ بُ سࣹينْ سَرَخُ ݒࣹݒّ ݧࣵڪّ مُڪّ خࣷرࣷمْ، ندَخْ مࣷويْ مࣵنْدَرْگَالْ سࣹينْ ڪٗلّࣴرࣹيڪْ سࣹينْ يࣵلَّ؞ سࣹينْ سَرَخْ يࣺݒّ نگࣹينْ دِ خࣷرࣷمْ؞
‏14 «بُ نِتْ دࣹي اِنْدِلْ اَجِ سَخْ جِ سَرَخُ ندࣷورْتࣹلْ مࣹݧّࣹيفْ، نَ ڪࣷ اِنْدِلْ مبࣷولْ¤ مُ ݧُ سࣺنْدَلْ، دࣹفْ ڪࣷ سࣵنقَلْ، مُيْ سَرَخُ ندࣷورْتࣹلْ مࣹݧّࣹيفَمْ؞ ‏15 نَ ݖِ سࣷتِّ دِوْ اَڪْ ݖُورَايْ؞ سَرَخُ ݒࣹݒّ لَ؞ ‏16 نَ سَرْخَلْڪَتْ بِ لَڪّ لࣹنّ ݖِ مبࣷولْ مُ مࣷڪّ مِ اࣵندّ اَڪْ دِوْ، بࣷولࣹ ڪࣷوڪْ ݖُورَايْ لِ لࣺݒّ، مُيْ سَرَخُ بَاخَنْتَلْ، تࣹ دِ سَرَخُ سَوَرَ؞
