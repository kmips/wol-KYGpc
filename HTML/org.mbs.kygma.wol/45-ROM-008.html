نࣷووُگْ يࣵلَّ مࣷو جِيتࣹ گࣴمْڪَتْ بِ
8
‏1 ڪࣷنْ نَگْ لࣺيگِ، اَمُلْ بࣹنّ دَانْ بُ ݧࣹيلْ ݧِ بࣷڪّ ݖِ اَلْمَسِ عِيسَىٰ؞ ‏2 ندَخْ ڪَتْ، دࣷولࣹيْ نࣷو گِيْ مَيࣹ بَڪَّنْ تࣹ نࣹڪّ ݖِ اَلْمَسِ عِيسَىٰ مࣷو لَ يِوِ، بَ نگَ مُݖّ ݖِ دࣷولࣹيْ بࣵڪَّارْ اَڪْ ندࣹي؞ ‏3 لِ يࣷونُ مُوسَىٰ تࣴلࣹ وࣷونْ ندَخْ نࣺيوْ دࣷولࣹيْ نِتُ سُوخُ نࣹينْ، مࣷويْ لِي يࣵلَّ دࣹفْ:‏ دࣷومَمْ لَ يٗنِّ، مُ دِڪّࣹ مࣹلࣷوْ نِتْ، تࣹ نِتْ دِ بࣵڪَّارْڪَتْ، دِ سُوخُ نࣹينْ، نگِرْ دࣷومَمْ فَجْتَلْ مبِرُمْ بࣵڪَّارْ؞ يَرَمُ سُوخْ وُ نِتْ ڪࣷوڪُ لَ يࣵلَّ سࣷتَّلࣹ مبُگَلْ گِ وَرْ ݖِ بࣵڪَّارْ، ‏4 نگِرْ نجُبْ گِ يࣷونُ مُوسَىٰ لاَجْ مَنَ سࣷتِّ بَ مَتْ سࣴڪّ ݖِ نُنْ، ݧِ تࣷݒَّتُلْ سُنُ بِنْدُ سُوخْ، خَنَا نࣷووُگْ يࣵلَّ؞
‏5 ندَخْ ڪَتْ ݧِ تࣷݒّ سࣹينْ بِنْدُ سُوخْ، مبِرِ بَڪَّنْ دࣷݝّ لَݧُ خِنْتࣹ، وَايࣹ ݧِ تࣷݒّ نࣷووُگْ يࣵلَّ ݧࣷومْ، مبِرِ نࣷووُگْ يࣵلَّ مࣷويْ سࣹينْ خِنْتࣹ؞ ‏6 خِنْتࣹيْ بَڪَّنْ نَگْ ݖِ ندࣹي لَيْ جࣴمࣹ، وَايࣹ خِنْتࣹيْ نࣷووُگْ يࣵلَّ ندُندّ اَڪْ جࣵمّ لَيْ جُرْ؞ ‏7 ندَخْتࣹ بࣷرࣷومْ خِنْتࣹيْ بَڪَّنْ اَبْ نࣷونُ يࣵلَّ لَ، ندࣹ دࣺگَّلُلْ يࣷونُ يࣵلَّ، تࣹ مَنُ ڪࣷ سَخْ؞ ‏8 مࣷو تَخْ ݧِ وࣺيࣹ سࣹينْ بِنْدُ سُوخُ نࣹينْ دُݧُ مَنَ نࣹيخْ يࣵلَّ؞
‏9 يࣹينْ نَگْ نࣹڪَّتُلࣹينْ ݖِ ڪِلِفْتࣹيفُ بِنْدُ سُوخُ نࣹينْ، وَايࣹ ݖِ ڪِلِفْتࣹيفُ نࣷو گِ نگࣹينْ نࣹڪّ، ندࣹگَمْ ڪَيْ نࣷووُگْ يࣵلاَّ نگِ مࣵڪَّانࣷو ݖِ يࣹينْ؞ ڪُ اَمُلْ نࣷووُگْ اَلْمَسِ، ڪࣷوڪُ دُ نِتَمْ؞ ‏10 سُ اَلْمَسِ نࣹڪّࣹي ݖِ يࣹينْ، سࣹينْ يَرَمْ دِنَ دࣹي مࣷوسْ ندَخْ بࣵڪَّارْ، وَايࣹ دُ تࣹي سࣹينُگْ نࣷو دِ دُندّ، ندَخْ اࣵتّࣹبْ نجُبْ بِ نگࣹينْ اَمْ بَ نࣷݒِّ؞ ‏11 نࣷووُگْ ڪِ دࣹڪَّلْ عِيسَىٰ مࣷو مࣵڪَّانࣷو ݖِ يࣹينْ؞ ڪࣷنْ نَگْ ڪِ دࣹڪَّلْ اَلْمَسِ مࣷويْ دࣹڪَّلْ اِتَمْ سࣹينْ يَرَمِ ندࣹي، ندَخْ نࣷووَمْ گࣷوگُ مࣵڪَّانࣷو ݖِ يࣹينْ؞
‏12 ڪࣷنْ نَگْ بࣷڪّ يِ، بࣷرْ تࣷݒّ نَ نُ، وَايࣹ دُ سُنُ بِنْدُ سُوخْ لَنُ اَمࣹيلْ لࣹنّ لُ نُ تَخَ تࣷݒّ سُنُ بَڪَّنْ؛ ‏13 ڪُ تࣷݒّ بَڪَّنَمْ، فࣵوّ مُ دࣹي، وَايࣹ ڪُ رࣹيْ جࣴفِ يَرَمَمْ ݖِ ندِمْبَلَلْ نࣷووُگْ يࣵلَّ، دِنَ تࣹخࣹ؞ ‏14 ندَخْ ڪَتْ، مبࣷولࣹمْ ݧِيْ دࣷخْ ݖِ نجِيتَلْ نࣷو گِ، ݧࣷوݧُ اَيْ دࣷومِ يࣵلَّ لَݧُ؞ ‏15 مَيࣹيسُ لࣹينْ نࣷو گُ لࣹينْ دِ دࣹسْلࣷو ݖِگْ نجَامْ، بَ تَخْ لࣹينَ دࣹلُّ ݖِگْ رَگَلْ، وَايࣹ نࣷووُگْ دࣷومْ يُ ݧُ دࣷومࣷو، مࣷومْ لࣹيسْ لࣹينْ مَيْ، تࣹ لࣷولࣷو نُ تَخْ دِ وࣷويࣹ يࣵلَّ:‏ «اَبَّ¤»، مُ تࣹڪِّ «بَايْ؞» ‏16 تࣹ نࣷووُگْ يࣵلَّ، مࣷومْ ݖِ بࣷݒَّمْ، مࣷو مࣵنڪࣷو اَڪْ سُنُ نࣷووُگْ بࣷݒّ، دِ سࣹيدࣹ نࣹ اَيْ دࣷومِ يࣵلَّ لَنُ؞ ‏17 گَنَّاوْ اَيْ دࣷومْ لَنُ نَگْ، اَيْ بࣷرࣷومْ ݖࣺرْ لَنُ، اَيْ بࣷرࣷومْ ݖࣺرْ يُ يࣵلَّ سࣺدّ، ݖࣺرْ يُ نُ بࣷڪّ اَڪْ اَلْمَسِ، گَنَّاوْ نࣷو بࣷڪّ اَڪْ مࣷومْ اَبْ ݖࣷونࣷ تࣹيْ، نگِرْ بࣷڪّ اَڪْ مࣷومْ تࣹرَانگَيْ اࣴلّࣴگْ؞
تࣹرَانگَيْ اࣴلّࣴگْ ݧࣹيلْ نَ گࣴمْڪَتْ ݧِ
‏18 جࣵݒّ نَا نࣹ سُنُ ݖࣷونࣷبْ تࣹيْ دُ دَرَ، سࣷو ڪࣷ خࣷولࣹي ݖِ تࣹرَانگَ جِ ݧُ نُيْ فࣹيݧَلَلْ اࣴلّࣴگْ؞ ‏19 ݖࣵڪّࣺيفُ اࣵدِّنَا نگِ سࣺينْتُ، ندَخْ يࣵڪَّمْتِ بࣺسْ بَ دࣷومِ يࣵلَّ دِ فࣹيݧْ؞ ‏20 ندَخْ ڪَتْ، ݖࣵڪّࣺيفْ گࣺݒّ، لِ ݧُيْ دࣺگَّلْ نࣹينُگْ نࣹينْ لَ، تࣹ دُ سࣹينْ ݖࣷوبَرࣹ، وَايࣹ ڪِ لࣹينْ ݖِ تࣴنڪّ لَ نࣹيخْ، بَ تَخْ ݧُيْ يَاڪَارْ، ‏21 ݧࣷومْ ݧِ بࣷڪّ ݖِ ݖࣵڪّࣺيفْ گࣷوگُ اِتَمْ، نࣹ دࣹيسْ نَ رࣹنگِ سࣹينْ بُومِ نجَامْ گِ لࣹينْ ندࣹي اَڪْ يࣵقُتࣹيْ يَرَمْ رࣵنگّ، نگِرْ ݧُ اَمْ ݖࣺرْ ݖِ گࣷرࣹيلُگْ دࣷومِ يࣵلَّ يِ، اَمْ ݖࣺرْ ݖِ سࣹينْ تࣹرَانگَ؞
‏22 خَمْ نَنُ بَ فِي نُ تࣷلُّ نࣹ مبࣷولࣹمْ ݖࣵڪّࣺيفْ گَا نگِ بِنِّ ندَخْ مِتِتْ، نِ ڪُيْ مَتُ؞ ‏23 تࣹ دُ ݖࣵڪّࣺيفْ گِ رࣹڪّ، وَايࣹ نُنْ ݖِ سُنُ بࣷݒّ اِتَمْ، ݖِ لَنُ، نُنْ ݧِ جࣷتْ اَبْ دَوَلْ ݖِ نࣷو گِ، نُنْ ݖِ سُنُ بࣷݒّ سَخْ نُ نگِ بِنِّ سُنُ بِيرْ خࣷلْ، ندَخْ يࣵڪَّمْتࣹي جࣷتْ ݖࣺرُ دࣷومِ يࣵلَّ، بَ سُنُ نجࣷتُگْ يَرَمْ مَتْ؞ ‏24 ندَخْ ڪَتْ سُنُ مُݖّ، يَاڪَارْ لَ تَخَوࣹيگُمْ، تࣹ يَاڪَارْ جࣷو تࣹگْ سَ بࣴتْ، دࣷوتُ يَاڪَارْ؞ لِ نِتْ ڪِ تࣹگْ بࣴتَمْ، نُ مُ ڪࣷيْ يَاڪَارَتࣹي؟ ‏25 لِ نُ گِسُلْ ڪَيْ لَنُيْ يَاڪَارْ، دِ ڪࣷ سࣺينْتࣷوڪَ مُݧࣹ يَاڪَارْ جِ؞
‏26 تࣹ اِتَمْ نࣷو گِ مࣷو نُيْ دِمْبَلِ ندَخْ سُنُگْ نࣺيوْ دࣷولࣹ، ندࣹ خَمُنُ سَخْ لِ جَادࣷو سࣵڪُّ ݖِ سُنُيْ ݧَانْ، وَايࣹ نࣷووُگْ يࣵلَّ مࣷو نُيْ تِنُلࣹ اَيْ اࣷنڪّ يُ وࣹيسْ ڪࣵدُّگْ لࣵمِّݧْ؞ ‏27 تࣹ ڪِيْ خࣷولْ لِ ݖِ بِيرْ خࣷلْ يِ، خَمْ نَ لِيْ خِنْتࣹيْ نࣷو گِ، ندࣹ نِ ڪࣷ يࣵلَّ بࣴگّࣹ، نِ لَ نࣷو گِيْ تِنࣷولْ ݧُ سࣹلّ ݧِ؞
‏28 تࣹ اِتَمْ نَگْ خَمْ نَنُ نࣹ يࣵلَّ مࣷويْ دࣹفْ مبࣷولࣹمْ لُ خࣹوْ، دَجࣹ بَ جُرْ نجَرِݧَلْ ݧِ سࣷݒّ يࣵلَّ، ݧِ مُ وࣷو، نࣷونࣹي ڪࣷ اَبْ دࣷگَلَمْ نَمّࣹ؞ ‏29 ندَخْ ݧࣷوݧُ يࣵلاَّ لࣹينْ رࣵݧّࣹي وࣷونْ لُ جِيتُ، تࣹ اِتْ لَ مُ دࣷگَلࣷونْ لُ جِيتُ ݖِ ݧࣷومْ مࣷو دِ ݧُ نِرُ دࣷومَمْ، نگِرْ دࣷومَمْ دِ تَاوْ، ݖِ بِيرْ رَڪّ يُ بَرࣹ؞ ‏30 ݧِ يࣵلَّ تُدَّلࣷونْ لࣷولُ لُ جِيتُ، مࣷو لࣹينْ وࣷو اِتَمْ؛ ݧࣷوݧُ مُ وࣷو، مࣷو لࣹينْ جࣷخْ اِتَمْ اࣵتّࣹبْ ݧُ جُبْ؛ تࣹ ݧࣷوݧُ مُ جࣷخْ اࣵتّࣹبْ ݧُ جُبْ، مࣷو لࣹينْ دَرَجَالْ اِتَمْ؞
ݖࣷفࣹيلُ يࣵلاَّ نگِ ݖِ اَلْمَسِ
‏31 اَنَ لُ نُيْ وَخَتِ نَگْ؟ گَنَّاوْ يࣵلاَّ اࣵندّ اَڪْ نُنْ، ڪُ نُ مَنَلْ دَرَ؟ ‏32 مࣷومْ مِ نُ بَݧَلُلْ دࣷومُ بࣷݒَّمْ، خَنَا جࣷوخࣹيلْ نُ ڪࣷ، نُنْ ݧࣺݒّ، اَنَ نُ مُ نُيْ بَݧَ بَاخࣹي گࣺݒّ يِوْ، بࣷولࣹ ݖِ؟ ‏33 ݧِ يࣵلَّ تَامُ، اَنَ ڪُ لࣹينْ دِ تُومَالْ؟ يࣵلَّ دࣹ مࣷويْ جࣷخْ نِتْ اࣵتّࣹبْ ڪُ جُبْ؞ ‏34 اَنَ ڪُيْ تࣹگࣹبْ دَانْ؟ اَلْمَسِ عِيسَىٰ دࣹ مࣷو دࣹي، رَخْ ݖِ دࣷلِّ، مُ دࣹڪِّ، تࣹ مࣷو تࣷوگْ فَ ندِجࣷورُ يࣵلَّ، دِ نُ فَ ݧَانَلْ!‏ ‏35 اَنَ ڪُ نُيْ تࣵگَّلࣹيڪْ ݖࣷفࣹيلُ اَلْمَسِ؟ دُ اَيْ نَتُّ، تࣹ دُ نجࣵقَرࣹ، دُ اِتْ ݖࣷونࣷيْ بُندّخَتَلْ، مبَا اَبْ خِيفْ، مبَا رَفْلࣹ، تࣹ دُ گࣹنّ آيْتࣺيفْ، بَ ݖِبْ سَامَرْ؞ ‏36 نࣷونُ لَ مبِندّ مِ اِنْدࣹي نࣹ:‏
«يَا تَخْ ݧُ يࣹنْدُ نࣷو بٗومْ،
دࣹيسْ نࣷو جࣵݒّࣹ نِ خَرْ يُ ݧُيْ تࣴرْ¤؞»
‏37 وَايࣹ ݖِ مبࣷولࣹمْ لُ نُ دِڪَّلْ، نࣷويْ اَمْ ندَمْ لُ مَتْ سࣴڪّ، ݖِ ندِمْبَلَلْ ڪِ نُيْ سࣷݒّࣹيمْ؞ ‏38 ندَخْ ڪَتْ بِرْ نَ مَ نࣹ دُ اَگْ ندࣹي، دُ اَگْ ندُندّ، دُ اَيْ مَلاَڪَ، دُيْ بࣷرࣷومْ نگُورْ، دُ تࣹيْ، دُ اࣴلّࣴگْ، دُيْ بࣷرࣷومْ دࣷولࣹ¤، ‏39 دُ لُ ڪَوࣹ، دُ لُ خٗوتْ، مبَا مࣹنࣹينْ مبِنْدࣹيفْ مُ مُ دࣷونْ؛ دَرَ مَنُ نࣷو تࣵگَّلࣹ اَڪْ ݖࣷفࣹيلْ، گِ نُ يࣵلَّ وࣷنْ ݖِ اَلْمَسِ عِيسَىٰ سُنُ سَنگّ؞
