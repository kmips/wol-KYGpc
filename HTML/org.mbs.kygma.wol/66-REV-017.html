مبُگَلْ ݧࣹيلْ نَ جِگࣺينُ مࣷيْڪَتْ جُ مَگْ جِ
17
‏1 ڪࣹنّ ݖِ جُرٗومْ ݧَارِ مَلاَڪَ، يِ تَاوُ جُرٗومْ ݧَارِ ندَبْ يِ نَگْ دِڪّ، نࣹ مَ:‏ «ڪَايْ، مَ وࣷنْ لَ مبُگَلُ جِگࣺينُ مࣷيْڪَتْ جُ مَگْ، جِ تࣷوگْ ݖِ ندࣷخْ مُ بَرࣹ؞ ‏2 مࣷومْ لَ بُورِ اࣵدِّنَ اࣵنْدَلْ دِ مࣷيْ، تࣹ بِيݧُ مࣷيَمْ لَ وَا اࣵدِّنَ دِ مࣵنْدࣹي؞»
‏3 ݖِ ڪَوْ لࣷولُ مَلاَڪَ مِ يٗبّࣷو مَ نࣷووُگْ يࣵلَّ بَ ݖَ مࣵنْدِݝْ مَ؞ مَ گِسْ جِگࣺينْ جُ وَرْ رَبْ وُ خࣷنقّ وُ اَمْ جُرٗومْ ݧَارِ بࣷݒّ اَڪْ فُڪِّ بࣺجّࣺنْ، تࣹ يَرَمْ وِ فࣹيسْ اَڪْ تُرْ يُيْ سَاگَ يࣵلَّ؞ ‏4 جِگࣺينْ جَا نگِ سࣷلْ لُ يࣷلࣹتْ تࣹ خࣷنقّ، تَڪّ وُرُسْ اَڪْ جَمَا اَڪِ ݒࣹرْ؞ مُ نگِ يࣷرْ ڪَاسُ وُرُسْ بُ فࣹيسْ اَڪْ ݧَاوْتࣺيفْ اَڪْ سࣷبࣹيْ مࣷيَمْ؞ ‏5 جࣴ بِ نَگْ، تُرْ وِي لَݧُ ݖِ بِندّ، مُيْ ڪُمْݒَ:‏ بَبِلࣷنْ گُ مَگْ گِ، ندࣹيُ جِگࣺينِ مࣷيْڪَتْ يࣹيڪْ ݧَاوْتࣺيفِ اࣵدِّنَ يِ؞ ‏6 مَ گِسْ نࣹ جِگࣺينْ جِ دَا نَانْ بَ مࣵنْدِ ݖِ دࣹرࣹتُ ݧُ سࣹلّ ݧِ اَڪْ دࣹرࣹتُ ݧِ دࣷونْ سࣹيدࣹيلْ عِيسَىٰ؞
بَ مَ ڪࣷ گِسࣹي، جَاخْلࣹ نَا لࣷولْ؞ ‏7 مَلاَڪَ مَ نࣹ مَ:‏ «لࣷويْ جَاخْلࣹ؟ مَنْ مَا لَيْ خَمَلْ ڪُمْݒَگْ جِگࣺينْ جِ اَڪْ رَبْ، وِ مُ وَرْ، بࣷرࣷومْ جُرٗومْ ݧَارِ بࣷݒّ يࣹيڪْ فُڪِّ بࣺجّࣺنْ يِ؞ ‏8 رَبْ وِ نگَ گِسْ مَسْ نَا نࣹڪّ وَايࣹ نࣹڪَّتُلْ؞ دِنَ گࣺنّࣹ ݖَ خٗوتࣹ بَ، وَايࣹ دَيْ سࣵنڪُجِ؞ وَا اࣵدِّنَ، ݧِ ݧُ بِنْدُلْ وࣷونْ سࣹينِ تُرْ ݖِ تࣺيرࣹبْ دُندّ بِ بَ اࣵدِّنَيْ دࣷورْ نَگْ، بُ ݧُ گِسࣹي رَبْ وِ، دِنَݧُ يࣺيمُ، ندَخْ لَ مُ مَسࣷونَ نࣹڪّ، تࣹيْ نࣹڪَّتُلْ، تࣹ اࣴلّࣴگْ مُيْ نࣹڪَّاتْ؞
‏9 «فِي نَگْ خࣹلْ مُ ݧَوْ لَ لاَجْ؞ جُرٗومْ ݧَارِ بࣷݒّ يِ ݧࣷويْ جُرٗومْ ݧَارِ تُندّ، يِ جِگࣺينْ جِ تࣷوگْ، تࣹ جُرٗومْ ݧَارِ بُورْ لَݧُ اِتَمْ:‏ ‏10 جُرٗومْ دَانُ نَݧُ، ڪࣹنّ ڪَا نگِ فِ، ڪِ ݖِ دࣹسْ نَگْ دِڪَّگُلْ، وَايࣹ بُ دِڪّࣹي، اَبْ دِيرْ رࣹڪّ لَ فِيْ اَمْ؞ ‏11 رَبْ وِ مَسࣷونَ نࣹڪّ تࣹ نࣹڪَّتُلْ نَگْ، مࣷومْ ݖِ بࣷݒَّمْ مࣷو دِ جُرٗومْ ݧࣹتّࣹيلُ بُورْ بِ، ڪࣹنّ لَ ݖِ جُرٗومْ ݧَارِ بُورْ يِ جِيتُ اِتَمْ، تࣹ دَيْ سࣵنڪُجِ؞
‏12 «فُڪِّ بࣺجّࣺنْ يِ نگَ گِسْ، فُڪِّ بُورْ لَݧُ يُ جࣷتَگُلْ ݖِ سࣹينْ نگُورْ؞ وَايࣹ دِيرُبْ وࣹنّ وَخْتُ، لَݧُيْ جࣷتْ سَݧْ-سَݧُ نگُورْ، ݧࣷوڪْ رَبْ وِ؞ ‏13 مࣹنّ مࣺبࣺتْ لَݧُ بࣷڪّ، تࣹ دِنَݧُ يࣺڪَّتِ سࣹينْ دࣷولࣹ اَڪْ سࣹينْ سَݧْ-سَݧْ، جࣷخْ ڪࣷ رَبْ وِ؞ ‏14 دِنَݧُ خَرࣹ اَڪْ گࣵتّ بِ، وَايࣹ گࣵتّ بࣹي لࣹينْ دِ دَانْ، ندَخْ مࣷويْ سَنگُ سَنگّ يِ، دِ بُورُ بُورْ يِ؞ مࣷويْ دَانْ مࣷوڪْ ݧࣷݧَمْ، ݧِ مُ وࣷو، تࣵنّ لࣹينْ تࣹ ݧُ سَخࣷو تࣷݒّ ڪࣷ؞»
‏15 مُ نࣹيتِ مَ:‏ «ندࣷخْ يِ نگَ گِسْ جِگࣺينُ مࣷيْڪَتْ جِ تࣷوگْ ݖِ نَگْ، مࣷويْ نِتِ خࣹيتْ يࣹيڪْ مبࣷولࣷو يࣹيڪْ اَسْڪَنْ يࣹيڪْ لࣵمِّݧْ يِ؞ ‏16 فُڪِّ بࣺجّࣺنْ يِ نگَ گِسْ، ݧࣷوڪْ رَبْ وِ، دِنَݧُ وࣷنْ جِگࣺينُ مࣷيْڪَتْ جِ مبَݧࣹيلْ؞ دَݧُ ڪࣷيْ فُتِّ بَ مُ نࣹ دُݝّ، ݧُ لࣹڪّ اَوْ سُوخَمْ، بَ نࣷݒِّ لَڪّ ڪࣷ؞ ‏17 ندَخْتࣹ يࣵلاَّ دࣹفْ ݖِ سࣹينْ خࣷلْ ݒَسْتࣺيفُ سࣷتَّلْ مࣺبࣺتَمْ، بَ ݧُ بࣷڪّ سࣷتَّلْ مࣹنّ مࣺبࣺتْ مِ، دَلْدِ جࣺبَّلْ رَبْ وِ سࣹينْ نگُورْ، بَ ڪࣹرَ ڪࣵدُّگْ يࣵلَّ مَتْ سࣴڪّ؞ ‏18 جِگࣺينْ جِي نگَ گِسْ، مࣷو دِ دࣴڪّ بُ مَگْ، بِ يِلِفْ بُورِ اࣵدِّنَ؞»
