تِمࣷتࣹ فࣹڪّسِ نَ ݒٗولْ اَڪْ سِلَسْ
16
‏1 ݖِ ڪَوْ لࣷولُ ݒٗولْ اَگْسِ دࣹرْبࣹ، تࣹگْ ݖِ دࣴڪّ بَ ݧُيْ وَخْ لِسْتَرْ؞ ندࣹڪࣹ اَبْ تَالِبࣹي نگَ فَ بُ ݧُيْ وَخْ تِمࣷتࣹ؛ ندࣹيَمْ دِ گࣴمْڪَتُبْ اَلْمَسِ بُ بࣷڪّ ݖِ اَسْڪَنُ يَوُتْ، بَايَمْ دِ گࣹرࣹگْ؞ ‏2 مُيْ ڪُ رَفࣹتْ سࣹيدࣹ ݖِ بِيرْ بࣷڪِّ دࣴڪّ يࣷويُ دِ لِسْتَرْ اَڪْ اِڪࣷݧُمْ؞ ‏3 ݒٗولْ نَگْ بࣴگّ مُ اࣵندّ اَڪْ مࣷومْ، بَ تَخْ مُ يٗبُّ ڪࣷ، خَرْفَلْ¤ ڪࣷ ندَخْ يَوُتْ يَ دࣴڪّࣹ وࣷونْ گࣷخْ بَ، نگِرْ ݧࣺݒَّ خَمࣷونْ نࣹ بَايَمْ اَبْ گࣹرࣹگْ لَ؞ ‏4 بَ لࣷولُ اَمࣹي ݧُيْ وࣴرْ دࣴڪّ يَ، دِ جࣷتّلِ دࣷگَلْ يِ تُڪّࣹي ݖَ ندَوِ عِيسَىٰ يَاڪْ مَگِ يࣹرُسَلࣹمْ، نگِرْ ݧُ سࣵمّ ڪࣷ؞ ‏5 مبࣷولࣷويْ گࣴمْڪَتْ ݧِ دِ گࣴنَ فࣹدّلِڪُ ݖِ وࣵلُّ نگࣴمْ، سࣹينِ لِمْ دِ يࣷڪُّ بࣺسْ اَڪْ بࣺسْ؞
يࣵلَّ فࣹيݧُ نَ ݒٗولْ
‏6 خࣹلْ مُ سࣹلّ مِ نَگْ تࣹرࣹ لࣹينَ يࣺگْلࣹجِ ڪࣵدُّ گِ ݖَ دِيوَانُ اَسِ، بَ تَخْ ݧُ جࣵلِّ بِيرْ دِيوَانُ فِرِسِ اَڪْ گَلَسِ؞ ‏7 ݧُ دࣹمْ بَ ݖَ ڪࣹمُ مِسِ، نَرَ جࣵلِّ بِيرْ دِيوَانُ بِتِنِ، خࣹلُمْ عِيسَىٰ مَيُ لࣹينْ ڪࣷ؞ ‏8 لࣷولُ تَخْ ݧُ جࣵلّࣹ بِيرْ مِسِ، اࣵڪِّ دࣴڪّ بَ ݧُيْ وَخْ تࣷرࣷوَسْ؞ ‏9 ݖِ بِيرْ لࣷولُ اَمْ ݒࣹيݧُ دِڪَّلْ ݒٗولْ ݖَ گُدِّ:‏ مُ گِسْ اَوْ نِتُ مَسࣹدُوَنْ تَخَوْ، دِ ڪࣷ تِنُ، نࣹ ڪࣷ:‏ «جࣵلّسِلْ مَسࣹدُوَنْ، وَلُّسِ نُ!‏» ‏10 نَڪَ لَ جࣷتْ ݒࣹيݧُ مَ، مُ دَلْدِ نُ¤ بِرْ نࣹ يࣵلاَّ نُ يࣹبَلْ نگِرْ نُ خَمْلࣹجِ فࣷوفَ خِبَارُ جࣵمّ بِ؞ ݖَ سَا سَ نُ فࣹخࣹي جࣵلّ بَ مَسࣹدُوَنْ؞
لِدِ گࣴمْ نَ اَلْمَسِ ݖَ فِلِݒِ
‏11 ݖِ ڪَوْ لࣷولُ نُ دُگّࣹ تࣷرࣷوَسْ گَالْ، جُبَلْ دُنُ سَمࣷتَرَسْ، جٗگࣹيتِ فَ، تࣹيرࣹجِ نࣹيَݒࣷلِسْ ݖَ اࣴلّࣴگْ سَ؞ ‏12 نُ بَوࣷو فَ جࣵلّ بَ فِلِݒِ، دࣴڪُّ گࣷخْ بَ جࣴڪّ ݖَ دِيوَانُ مَسࣹدُوَنْ تࣹ دِ مࣷومࣹيلُ رࣷومْ؞ نُ تࣷوگْ فَ اَيْ فَنْ؞
‏13 بَ بࣺسُبْ نࣷفْلاَيْ بَ تَخَوࣹي، نُ گࣺنّ ݖَ بُنْتُ دࣴڪّ بَ، ݖَ وࣹتُ دࣹخْ گَ، دَلْدِ فَيْ گِسْ بࣺرَبْ بُ نُ فࣷوگْ نࣹ اَبْ جُلِّڪَايْ لَ؞ نُ تࣷوگْ فَ نَگْ، دِ وَخْتَانْ اَڪْ جِگࣺينْ ݧَ فَ دَجࣹ؞ ‏14 اَسْ ندَوَ نگَ ݖَ، دِ دࣺگْلُ؛ ڪُ ݧُيْ وَخْ لِدِ، دࣴڪّࣹ ݖَتِرْ، دِ جَايْڪَتُبْ ݖُوبِ تࣵنّࣺيفْ يُ خࣷنقّ ݖُرّ، تࣹ دِ جَامُڪَتُ يࣵلَّ؞ بࣷرࣷومْ بِ نَگْ اُبِّ خࣷلَمْ، بَ مُ تࣹيوْلُ بُ بَاخْ ڪࣵدُّيْ ݒٗولْ؞ ‏15 ݖِ ڪَوْ لࣷولُ ݧُ سٗوبْ ڪࣷ ݖِ ندࣷخْ، مࣷومْ اَڪْ وَا ڪࣴرَمْ؞ مُ دَلْدِ نُ ݧَانْ، نࣹ نُ:‏ «ندࣹگَمْ اَبْ گࣴمْڪَتُبْ سَنگّ بِ نگࣹينْ مَ جࣵݒّࣹ، اَگْسِلࣹينْ، دَلْ اَڪْ مَنْ ݖَ سَمَ ڪࣴرْ؞» مُ سࣷݧّ نُ بَ نُ نَنگُ؞
تࣴجْ نَݧُ ڪَسࣷ ݒٗولْ اَڪْ سِلَسْ
‏16 نࣷو دࣷونْ دࣹمْ جُلِّڪَايْ بَ، اَبْ سُرْگَ بُ جِگࣺينْ بُ رَبُ گِسَانࣹ جࣵݒّ، دَجࣹيڪْ نُنْ؞ اَلَلْ جُ بَرࣹ لَ اَيْ نجَاتِگࣹيمْ دَا جࣴلࣹ ݖِ نگِسَانࣹيمْ؞ ‏17 ندَوْ سَ تࣷݒّ ݖِ ݒٗولْ اَڪْ نُنْ، دِ خَاݖُ، نَانْ:‏ «نِتْ ݧِي جَامِ يࣵلَّ اَجِ ڪَوࣹ جِ لَݧُ؛ دَݧُ لࣹينْ دِ يࣺگَلْ يࣷونُ مُݖّ؞» ‏18 نࣷونُ لَ جࣵݒّࣷو اَيْ فَنِ فَنْ، بَ ݒٗولْ مُجّ خَجࣷوتُ ڪࣷ؞ مُ وَلْبَتِڪُ نࣹ رَبْ وَ:‏ «مَا لَ سَنتّ ݖِ تُرُ عِيسَىٰ اَلْمَسِ، گࣺنَّلْ ݖِ ندَوْ سِ؞» رَبْ وَ دَلْدِ گࣺنّ ݖَ سَا سَ؞
‏19 بَ لࣷولُ اَمࣹي اَلَلْ جَ نجَاتِگࣹيْ ندَوْ سَ يَاڪَارࣷونْ، رࣺيرْ لࣹينْ، ݧُ جࣵݒّ ݒٗولْ اَڪْ سِلَسْ، دِرِ، يٗبُّ ݖَ ݒࣺنݖّ مَ، فَ ڪَنَمْ نجِيتْ يَ؞ ‏20 ݧُ تَخَوَلْ لࣹينْ فَ اࣵتّࣹڪَتْ يَ، نࣹ:‏ «ݧِي ݧࣷو يࣴنگَلْ سُنُ دࣴڪّ بِ، دِ اَيْ يَوُتْ ‏21 يُيْ خَمْلࣹ آدَ يࣷيْ، نُنْ ݧِيْ وَا رࣷومْ، سَݧُنُ ڪࣷو نَنگُ، سَݧُنُ ڪࣷو جࣴفࣹ؞»
‏22 ݖِ ڪَوْ لࣷولُ مبࣷولࣷو مَ اِتْ دَلْ ݖِ سࣹينْ ڪَوْ؛ اࣵتّࣹڪَتْ يَ فُتِّ سࣹينِ يࣺرࣹ، دَلْدِ سَنْتَانࣹ ݧُ دٗورْ لࣹينْ اَيْ يَرْ؞ ‏23 ݧُ دٗورْ لࣹينْ اَيْ يَرْ يُ بَرࣹ، سࣵنِّ لࣹينْ ݖَ ڪَسࣷ بَ، بࣷولࣹ ݖَ جࣷخْ بࣷرࣷومْ ڪَسࣷ بَ ندِگَلَلْ وَتُّ لࣹينْ بُ بَاخْ؞ ‏24 ندِگَلْ لࣷولُ نَگْ تَخْ مُ سࣵنِّ لࣹينْ ݖَ نࣺيگْ بَ گࣴنَ رُقࣹ ݖَ ڪَسࣷ بَ، جࣺنگّ لࣹينْ؞
‏25 بَ گُدِّ گَيْ خَاجْ، ݒٗولْ اَڪْ سِلَسَ نگَيْ ݧَانْ، دِ سࣵبَّالْ يࣵلَّ، ݧࣹنࣹينْ ݧَ ݧُ فَ تࣴجْ دِ لࣹينْ دࣺگْلُ؞ ‏26 ݖِ ڪَوْ لࣷولُ سُوفْ جࣹڪِّ يࣴنگُ يࣴنگُ بُ مَگْ بَ ڪࣹنُگْ ڪَسࣷ بَ دِ رࣴنگّ-رࣴنگِ؛ بُنتّ يࣺݒّ جࣹڪِّ اُبِّڪُ، جࣺنگِ ݧࣺݒّ دࣵجِّڪُ؞ ‏27 بࣷرࣷومْ ڪَسࣷ بَ تِيتࣹ ݖِيْ نࣹلَوْ، گِسْ بُنْتِ ڪَسࣷ بِ اُبِّڪُ؞ مُ بࣷݖِّ سَامَرَمْ، نَرَ خَرُ، يَاڪَارْ نࣹ ݧَ ݧُ تࣴجْ دَݧࣷو رࣴݖّ؞ ‏28 ݒٗولْ نَگْ اࣵدُّ ݖَ ڪَوْ نࣹ:‏ «بُلْ لࣷرْ سَ بࣷݒّ، نُنْ ݧࣺݒَّ نگِ فِ؞»
‏29 بَ لࣷولُ اَمࣹي بࣷرࣷومْ ڪَسࣷ بَ اِنْدِلُ اَبْ لࣵمݒّ، دَلْدِ نࣹ سَرَاخْ دُگّ، نࣹ گُرُبْ فَ ڪَنَمْ ݒٗولْ اَڪْ سِلَسْ، دِ لࣷخْ؞ ‏30 گَنَّاوْ گِ مُ گࣺنّࣹ لࣹينْ ݖَ بِتِ نࣹ لࣹينْ:‏ «سَنگّ يِ، اَنَ لُ مَ وَرَ دࣹفْ، بَ مُݖّ؟» ‏31 ݧُ نࣹ ڪࣷ:‏ «گࣴمَلْ سَنگّ عِيسَىٰ، دَلْدِ مُݖّ، يَاڪْ سَ وَا ڪࣴرْ؞» ‏32 بَ لࣷولُ اَمࣹي ݧُ خَمَلْ ڪࣷ ڪࣵدُّگْ سَنگّ بِ، مࣷومْ اَڪْ وَا ڪࣴرَمْ گࣺݒّ؞ ‏33 مُ يٗبُّ لࣹينْ نَگْ ݖِ وَخْتُوْ گُدِّ وࣷووُ، رَخَسْ سࣹينِ گَاݧُ-گَاݧُ، ݧُ سٗوبْ ڪࣷ ݖِ ندࣷخْ ݖَ سَا سَ، مࣷومْ اَڪْ وَا ڪࣴرَمْ گࣺݒّ؞ ‏34 گَنَّاوْ لࣷولُ مُ يٗبُّ لࣹينْ ڪࣴرَمْ، جࣷخْ لࣹينْ ݧُ لࣹڪّ، تࣹيْ بࣺگࣹينْدࣷو اَڪْ وَا ڪࣴرَمْ گࣺݒّ ݖِ نگࣴمْ گِ ݧُ گࣴمْ يࣵلَّ؞
‏35 بَ بࣴتْ سࣹتࣹي، اࣵتّࣹڪَتْ يَ يࣹبَلْ اَلْڪَاتِ يَ، ݧُ نࣹ بࣷرࣷومْ ڪَسࣷ بَ:‏ «ݧِي، يِوِ لࣹينْ؞» ‏36 بࣷرࣷومْ ڪَسࣷ بَ اࣵگّࣹ ݒٗولْ ندِگَلْ لَ، نࣹ ڪࣷ:‏ «اࣵتّࣹڪَتْ يِ ڪَتْ ݧࣷو فِ يࣹبْلࣹ، نࣹ ݧُ يِوِ لࣹينْ؞ ڪࣷنْ نَگْ مَنْ نگࣹينَ گࣺنّ تࣹ دࣹمْ اَڪْ جࣵمّ؞»
‏37 ݒٗولْ نَگْ نࣹ اَلْڪَاتِ يَ:‏ «ݧࣷومْ ݧُ دٗورْ نُ ݖِ ڪَنَمْ نِتْ ݧِ، اࣵتّࣹيسُ نُ سَخْ تࣹ نُ دِيْ گࣷرْ ݖِ رࣷومْ گِي، رَخْ ݖِ دࣷلِّ ݧُ سࣵنِّ نُ ݖِ ڪَسࣷ بِ؞ لࣺيگِ ݧُ بࣴگّ نࣷو دࣵقْ ندࣵنڪّ ݖِ بِتِ؟ مُڪّ!‏ نَݧُ ݧࣴوْ، ݧࣷومْ ݖِ سࣹينْ بࣷݒّ، گࣺنّࣹ نُ؞»
‏38 بَ مُ ڪࣷ دࣹفࣹي، اَلْڪَاتِ يَ اࣵگّࣹ اࣵتّࣹڪَتْ يَ ڪࣵدُّ گࣷوگُ، ݧُ خَمْ نࣹ اَيْ گࣷرِ وَا رࣷومْ لَݧُ، تِيتَانگࣹ جࣵݒّ لࣹينْ؞ ‏39 ݧُ دِڪّ، جࣺگَّلُ، گࣺنّࣹ لࣹينْ ݖِ بِتِ، ݧَانْ لࣹينْ، ݧُ گࣺنّ دࣴڪّ بَ؞ ‏40 نࣷونُ لَݧُ گࣺنّࣹ ڪَسࣷ بَ، دࣹلُّ ڪࣴرْ لِدِ؞ ݧُ گِسࣹ فَاڪْ بࣷڪِّ گࣴمْڪَتْ ݧَ، ݧَاخْ لࣹينْ، دࣷورَ دࣹمْ؞
