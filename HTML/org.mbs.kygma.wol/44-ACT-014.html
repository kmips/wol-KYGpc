ݒٗولْ اَڪْ بَرْنَبَ وَارࣹ نَݧُ ݖَ اِڪࣷݧُمْ
14
‏1 بَ ݒٗولْ اَڪْ بَرْنَبَ نࣹڪّࣹي اِڪࣷݧُمْ، نࣷونُ لَݧُ دُگّࣹ اِتَمْ ݖَ جࣵنگُبْ يَوُتْ يَ؞ ݧُ وَارࣹ، بَ مبࣷولࣷو مُ مَگْ گࣴمْ، اَيْ يَوُتْ اَڪِ گࣹرࣹگْ؞ ‏2 يَوُتْ يِ گࣴمُلْ نَگْ دِ خِيرْتَلْ جَامْبُرْ ݧِ دُلْ يَوُتْ، دِ يࣵقْ سࣹينْ خࣹلْ ݖِ بࣷڪِّ تَالِبࣹ يِ؞ ‏3 تࣹيوُلْ ݒٗولْ اَڪْ بَرْنَبَ تࣷوگْ اِڪࣷݧُمْ دِيرْ بُ يࣵگّ لࣷولْ؞ فِتْ لَݧُ دࣷونْ وَارࣹي نگِرْ بࣷرࣷومْ بِ، بࣷرࣷومْ بِ مَيࣹ اَيْ فِرْندࣹ اَڪِ ڪࣺيمَانْ يُ جَارࣹ ݖِ ݧࣷومْ، نگِرْ سࣹيرࣹي ڪࣷ ڪࣵدُّگْ يِوَمْ گِ ݧُيْ بِرَلْ؞ ‏4 بَ لࣷولُ اَمࣹي وَا دࣴڪّ بِ خࣵجَّلِڪࣷو، ݧِي فَرْ اَڪْ يَوُتْ يِ، ݧَلࣹ فَرْ اَڪْ ندَوْ يَ؞ ‏5 جَامْبُرْ ݧࣹيڪْ يَوُتْ يِ نَگْ اࣵندّ اَڪْ سࣹينِ ڪِلِفَ، دِ لࣹينْ وَاجَ مِتِتَلْ اَڪَ دٗورِ دࣷجْ؞ ‏6 ݧُ يࣺگْ ڪࣷ، دَوْ لࣵقُجِ دِيوَانُ لِڪَوْنِ، ݖَ دࣴڪّ يَ ݧُيْ وَخْ لِسْتَرْ، اَڪْ دࣹرْبࣹ اَڪْ لَ ڪࣷ وࣴرْ؞ ‏7 فࣷوفَ اِتَمْ ݧُ دِ فَ خَمْلࣹ خِبَارُ جࣵمّ بِ؞
ݒٗولْ اَڪْ بَرْنَبَ وَارࣹ نَݧُ ݖَ لِسْتَرْ
‏8 جࣹنّ وَايَ نگَ دَانْ تࣷوگْ ݖَ لِسْتَرْ، تࣵنڪّ يَ بَاخُلْ وࣷونْ؛ لاَگࣷ لَ جُدُّوَالࣹ، تࣹ مَسُلَ دࣷخْ؞ ‏9 ڪࣷوڪࣷو دࣺگّ ݒٗولْ مُيْ وَارࣹ، ݒٗولْ خࣷولْ ڪࣷ جࣵڪّ، گِسْ نࣹ اَمْ نَ نگࣴمْ گُ مُݖَّمْ مَنَ جَارࣹ؞ ‏10 مُ اࣵدُّ ݖَ ڪَوْ، نࣹ وَا جِ:‏ «جٗگَلْ تَخَوْ جࣷنّ!‏» وَا جِ نࣹ بࣴرࣺتْ، جٗگْ، دَلْدِ دࣷخْ؞
‏11 بَ مبࣷولࣷو مَ گِسࣹي لَ ݒٗولْ دࣹفْ، ݖَ ڪَوْ لَݧُ اࣵدُّ ݖِ لࣵڪُّ لِڪَوْنِ، نࣹ:‏ «يࣵلَّ يࣹي سࣷݒّلِڪُ مࣹلࣷوْ نِتْ، وࣵݖّسِ ݖِ سُنُ بِيرْ!‏» ‏12 ݖِ ڪَوْ لࣷولُ ݧُ تُدّࣹ بَرْنَبَ سࣴسْ، ڪِلِفَيْ يࣵلَّيْ گࣹرࣹگْ يِ، ݒٗولْ نَگْ گَنَّاوْ مࣷو فَرࣹوࣷو وࣷونْ ڪࣵدُّگْ وَارࣹ گَ، ݧُ تُدّࣹ ڪࣷ اࣹرْمࣹسْ، ندَوَلْ يࣵلَّيْ گࣹرࣹگْ يِ؞ ‏13 جَامࣷوڪَايُ سࣴسْ مَ نگَ وࣷونْ ݖَ بُنْتُ دࣴڪّ بَ؞ سَرْخَلْڪَتْ بَ دَلْدِ اِنْدِ ݖَ بُنْتِ دࣴڪّ بَ اَيْ يࣴڪّ يُ ݧُ تَڪَّلْ ݖَقِ تٗورْ-تٗورْ؞ سَرْخَلْڪَتْ بَ اࣵندّ اَڪْ مبࣷولࣷو مَ، نَرَ رࣹنْدِ جُرْ گَ، سَرْخَلْ ڪࣷ ݒٗولْ اَڪْ بَرْنَبَ؞
‏14 ݖِ ڪَوْ لࣷولُ ݒٗولْ اَڪْ بَرْنَبَ يࣺگْ لَ خࣹوْ؞ ݧُ خࣷتِّ سࣹينِ يࣺرࣹ ندَخْ ݧَاوْلُ ڪࣷ؞ ݧُ بُورْ نَگْ بَ ݖَ بِيرْ مبࣷولࣷو مَ، دَلْدِ اࣵدُّ ݖَ ڪَوْ نࣹ:‏ ‏15 «يࣹينْ، لُ لࣹينْ خِيرْ ݖِ لِي؟ نُنْ اِتَمْ اَيْ نِتْ لَنُ، بࣷڪّ اَڪْ يࣹينْ بࣹنّ بِندّ؞ خِبَارُ جࣵمّ بِ لَنُ لࣹينْ دِ خَمَلْ، نگِرْ نگࣹينْ وَݖّ يࣴفِ نࣹينْ يِي تࣹ وَلْبَتِڪُ ݖِ يࣵلَّ، ڪِيْ دُندّ تࣹ سࣵڪّ اَسَمَانْ اَڪْ سُوفْ اَڪْ گࣺيجْ اَڪْ مبࣷولࣹمْ لُ نࣹڪّ ݖِ بِيرْ؞ ‏16 مࣷو مَيࣷونْ دࣺمبّ خࣹيتْ يࣺݒّ، ݧُ اَوࣹ سࣹينْ يࣷونِ بࣷݒّ؞ ‏17 مࣷونَ نࣷݒِّوُلَ سࣹيرࣹيلْ بࣷݒَّمْ ݖِ مبَاخَمْ گِ مُيْ جࣴفࣹ، دِ لࣹينْ وࣵݖّࣹيلْ فَ اَسَمَانْ، تَوْ اَڪْ نَاتَانگࣹ، رࣹگَّلْ لࣹينْ، بࣺگَلْ لࣹينْ بَ سࣹينْ خࣷلْ سࣹدّ؞»
‏18 لࣷولُ لࣺݒّ ندَوْ يَ وَخْ نَݧُ ڪࣷ تࣹ تُوتِ ڪࣷنْ ݧُ تࣴلࣹي تࣹرࣹ مبࣷولࣷو مَ ݧُ رࣹنْدِلْ لࣹينْ سَرَخْ؞
‏19 ݖِ ڪَوْ لࣷولُ اَيْ يَوُتْ يُ جٗگࣹ اࣵنݖࣷسْ گُ ݒِسِدِ اَڪْ اِڪࣷݧُمْ، فࣹخࣹ بَ فࣵبِّ مبࣷولࣷو مَ، دَلْدِ دٗورْ ݒٗولْ اَيْ دࣷجْ، دِرِ ڪࣷ بَ گࣺنّࣹ ڪࣷ دࣴڪّ بَ، يَاڪَارْ نࣹ دࣹي نَ؞ ‏20 نَڪَ لَ ڪࣷ تَالِبࣹ يَ يࣺيوْ، مُ جٗگْ، دَلْدِ دࣹلُّ ݖَ بِيرْ دࣴڪّ بَ؞ ݖَ اࣴلّࣴگْ سَ، مُ اࣵندّ اَڪْ بَرْنَبَ، دࣹمْ دࣹرْبࣹ؞
دࣺلْسِ نَݧُ اࣵنݖࣷسْ گُ سِرِ
‏21 گَنَّاوْ بَ ݧُ خَمْلࣹي فَ دࣹرْبࣹ خِبَارُ جࣵمّ بِ، بَ اَمْ فَ تَالِبࣹ يُ بَرࣹ، دَݧࣷو دࣹلُّ لِسْتَرْ اَڪْ اِڪࣷݧُمْ اَڪْ اࣵنݖࣷسْ گُ ݒِسِدِ؞ ‏22 ݧُيْ فࣹدّلِ ݒَسْتࣺيفُ تَالِبࣹ يِ، دِ لࣹينْ ݧَاخْ نگِرْ ݧُ سَخࣷو نگࣴمْ، نࣹ لࣹينْ:‏ «ݖࣷونࣷ يُ بَرࣹي نگِي، فࣵوّ نُ جَارࣹ ݖِ، نگِرْ تࣵبِّ ݖِ نگُورُگْ يࣵلَّ؞» ‏23 مبࣷولࣷومْ گࣴمْڪَتْ مُ نࣹڪّ نَگْ، ݧُ تَبّ ݖِ اَيْ مَگْ؛ ݖِ بِيرْ لࣷولُ ݧُ ݧَانْ، بࣷولࣹ ڪࣷوڪْ ڪࣷورْ، دَلْدِ دࣺنڪّ مَگْ ݧَ سَنگّ بِ ݧُ گࣴمْ؞ ‏24 گَنَّاوْ لࣷولُ ݧُ جَارࣹ بِيرْ دِيوَانُ ݒِسِدِ، بَ اࣵگّ ݒَمْفِلِ؞ ‏25 ݧُ دَلْدِ سِيوْتَانࣹ ڪࣵدُّ گِ ݖَ ݒࣹرْگࣹ، دࣷورَ وࣵݖِّ بَ اَتَلِ؞
‏26 فࣷوفَ لَݧُ دُگّࣹ گَالْ، دࣹلُّ اࣵنݖࣷسْ گُ سِرِ، گَ ݧُ لࣹينْ دࣺنڪࣹ وࣷونْ يࣵلَّ ݖِوْ يِوَمْ، نگِرْ لِگّࣺييْ بَ ݧُ سࣷتَّلْ، بَ دࣺلْسِ نࣷونُ؞ ‏27 بَ ݧُ اَگْسࣹي، دَݧࣷو وࣷو مبࣷولࣷومْ گࣴمْڪَتْ ݧَ، اࣵگّࣹ لࣹينْ مبࣷولࣹمْ نَ يࣵلَّ دࣹفْ اَڪْ ݧࣷومْ، اَڪْ نَ مُ اُبِّلࣹي جَامْبُرْ ݧِ دُلْ يَوُتْ بُنْتُ نگࣴمْ؞ ‏28 ݧُ تࣷوگْ نَگْ اَڪْ تَالِبࣹ يَ دِيرْ بُ خَوَ يࣵگّ؞
