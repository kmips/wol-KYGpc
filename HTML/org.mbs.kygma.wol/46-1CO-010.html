جَارْ-جَارِ مَامْ يࣹدّ نَ سࣴتْ يَ
10
‏1 بࣷڪّ يِ، بُگُّمَ نگࣹينْ اُمݒّلࣹ ڪࣷ، سُنُ مَامْ يࣺݒّ لَ نِيرْ وَ يِيرࣷونْ، تࣹ ݧࣷومْ ݧࣺݒَّ جࣵلّ گࣺيجْ گَ¤؞ ‏2 ݧࣷومْ ݧࣺݒّ لَݧُ سٗوبْ سٗوبࣹبْ مُوسَىٰ، ݖَ بِيرْ نِيرْ وَ، اَڪْ ݖَ بِيرْ گࣺيجْ گَ؞ ‏3 ݧࣷومْ ݧࣺݒّ، وࣹنّ ݧَمُ ڪࣺيمَانُ يࣵلَّ وَ، مࣷومْ لَݧُ لࣹڪّ، ‏4 تࣹ ݧࣷومْ ݧࣺݒّ، مࣹنّ ندࣷخُمْ ڪࣺيمَانُ يࣵلَّ مَ لَݧُ نَانْ، ندَخْ ݧࣷو دࣷونْ نَانࣹ ݖَ دࣷجُ ڪࣺيمَانُ يࣵلَّ گَ، دࣷجْ وَ لࣹينْ دࣷونْ گُنگࣹ، تࣹ دࣷجْ وَ اَلْمَسِ لَ وࣷونْ؞ ‏5 تࣹيوُلْ ݧِ اࣴݒّ ݖِ ݧࣷومْ، يࣵلَّ اَمُلْ وࣷونْ لُ ڪࣷ نࣹيخْ ݖِ ݧࣷومْ، بَ تَخْ ݧُ دَانُ، دࣹي ݖَ مࣵنْدِݝْ مَ؞
‏6 يࣷويُ خࣹوْ-خࣹوْ اَيْ مِسَالْ لَ، نگِرْ نُ بَݧَ خࣹمّࣹمْ يࣴفْ يُ بࣷنْ، نَ ڪࣷ مَامْ يَ دَانْ خࣹمّࣹمࣹ؞ ‏7 بُلࣹينْ دࣷونْ اَيْ جَامُڪَتِ تُورْ، نِ ڪࣷ ݧࣹنّ ݖِ ݧࣷومْ دࣷونࣹ وࣷونْ؞ نࣷونُ لَݧُ بِنْدࣹ نࣹ:‏ «مبࣷولࣷو مَ تࣷوگْ، دِ لࣹڪّ اَڪَ نَانْ، نࣹسْ تُوتْ ݧُ جٗگْ، دِ مبُمْبَايْ¤؞» ‏8 بُنُ جࣴفࣹ ݒࣷوُمْ سࣺيْ، نِ ڪࣷ ݧࣹنّ ݖِ ݧࣷومْ دَا دࣹفࣹ، بَ تَخْ ݧَارْ فُڪِّ جُنِّ اَڪْ ݧࣹتّ (23000) ݖَ مبࣷولࣷو مَ بࣷڪّ دࣹي ݖِ بࣹنّ بࣺسْ؞ ‏9 بُنُ وُتْ اِتَمْ اَلْمَسِ، نِ ڪࣷ ݧࣹنّ ݖِ ݧࣷومْ وُتࣹ وࣷونْ، بَ تَخْ اَيْ جَانْ رࣹيْ لࣹينْ؞ ‏10 بُلࣹينْ جࣵمْبَتْ، نِ ݧࣹنّ ݖِ ݧࣷومْ دࣷونْ جࣵمْبَتࣹ، بَ تَخْ مَلاَڪَمْ رࣹيْڪَتْ بَ رࣹيْ لࣹينْ؞
‏11 خࣹوْ-خࣹوْ يࣷويُ دَلْ مَامْ يَ اَيْ مِسَالْ لَ، تࣹ دࣹيسْ ڪࣷو بِندّ نگِرْ نُ جࣵنگࣹ ݖِ، نُنْ ݧِ مُجُگْ جَمࣷنࣷ يࣹمࣷولْ؞ ‏12 مࣷو تَخْ ڪُ يَاڪَارْ نࣹ تَخَوْ نگَ، نَنگَ مࣷيْتُ بَلَ نگَيْ دَانُ؞ ‏13 بࣹنّ نَتُّ دَبُ لࣹينْ بُ مࣷيْ نَتُّ يِيْ دَلࣹ؞ تࣹ اِتْ يࣵلاَّ جَرَ وٗولُ؞ يࣵلَّ دُ نَنگُ نگࣹينْ تࣹگࣷو نَتُّ بُ نگࣹينْ اࣵتَّنُلْ، وَايࣹ نَتُّ بَ لَيْ بࣷولࣹيڪْ ݒࣹخࣹ، مَ لࣹينْ ݖَيْ گࣺنّࣹ، بَ نگࣹينْ مَنْ ڪࣷو دࣺڪُّ؞
دَوْلࣹينْ بࣷڪَّالࣹ
‏14 ڪࣷنْ ڪَتْ، سَمَيْ سࣷݒّࣹ، دَوْلࣹينْ بࣷڪَّالࣹ؞ ‏15 نجࣷرتّ نَا نࣹ بࣷرࣷومِ خࣹلْ لاَيْ وَخَلْ؞ ڪࣷنْ اࣵتّࣹيلْ-لࣹينْ سࣹينْ بࣷݒّ لِ مَ لࣹينْ وَخْ؞ ‏16 ڪَاسُ بَرْڪࣹ بِ نُيْ بَرْڪࣹيلࣷو، خَنَا دُ مࣷويْ بࣷڪّ گِ نُ بࣷڪّ ݖِ دࣹرࣹتُ اَلْمَسِ؟ تࣹ مبُرُ مِ نُيْ دَگّࣷو، دُ مࣷويْ بࣷڪّ گِ نُ بࣷڪّ ݖِ يَرَمُ اَلْمَسِ؟ ‏17 گَنَّاوْ مࣹنّ مبُرُ لَ نَگْ، دࣷونْتࣹ ݧُ بَرࣹ لَنُ، ڪࣷنْ وࣹنّ يَرَمْ لَنُ، ندَخْ نُنْ ݧࣺݒّ، مࣹنّ مبُرُ مِ لَنُ بࣷڪّ؞
‏18 خࣷولْ-لࣹينْ بࣵنِّ إِسْرَاءِيلْ ݖِ وࣵلُّ دࣹرࣹتْ؞ ݧِيْ لࣹڪّ يࣵݒُّ جُرُ سَرَخْ گُ ݧُ رࣹنْدِ، خَنَا بࣷڪُّݧُ اَڪْ سَرْخَلُڪَايْ بِ اَبْ ݖࣺرَمْ؟ ‏19 اَنَ ڪࣷنْ لُ سَمَ وَخْ جِيْ وُندّ؟ ندَخْ دَا وُندّ نࣹ يࣵݒّ وُ ݧُ سَرْخَلْ تُورْ يِ تࣹڪِّ نَ لࣹنّ؟ اَمْ دَا وُندّ نࣹ تُورْ يِ تࣹڪِّ نَ لࣹنّ؟ ‏20 لِ مَيْ وَخْ ڪَيْ مࣷو دِ يࣷويُ سَرَخْ، رَبْ يِ لَ ݧࣹيلْ، وَايࣹ دُ يࣵلَّ، تࣹ بُگُّمَ نگࣹينْ بࣷڪّ اَڪْ رَبْ يِ لࣹنّ؞ ‏21 دُنگࣹينْ مَنَ نَانْ ݖِ ڪَاسُ سَنگّ بِ، نَانْ ݖِ ڪَاسُ رَبْ يِ؞ دُنگࣹينْ مَنَ بࣷڪّ لࣹڪّ ݖِ ندَبَلْ سَنگّ بِ، بࣷڪّ لࣹڪّ ݖِ ندَبَلْ رَبْ يِ؞ ‏22 اَمْ دَنُيْ يࣺڪَّتِ فِيرَانگࣹيْ سَنگّ بِ؟ دَنُ ڪࣷو اࣴݒّ دࣷولࣹ؟
جࣴفࣹيلْ سَيْ سَݧْ-سَݧْ تࣹيْ سٗورَالࣹ نِتْ ݧِ
‏23 «سَݧࣹيسْ نَ لࣺݒّ،» لَݧُ وَخْ؞ وَايࣹ دُ لࣺݒّ اَيْ جَرِݧࣹ؞ «سَݧࣹيسْ نَ لࣺݒّ،» وَايࣹ دُ لࣺݒّ اَيْ يࣷڪّࣹ؞ ‏24 بُ ڪࣹنّ سࣵڪُّ نجَرِݧَلْ بࣷݒَّمْ، وَايࣹ سࣵڪُّلࣹينْ سࣹينْ نجَرِݧَلْ مࣷرࣷومْ؞
‏25 مبࣷولࣹمْ لُيْ جَايْ ݖَ جَ بَ، سَݧْ نگࣹينْ ڪࣷو لࣹڪّ تࣹ بَݧْ ݖࣹي لُقَتُ لࣹنّ لُيْ اِنْدِ نگࣵنتّ ݖِمْ خࣹلْ، ‏26 ندَخْ بࣷرࣷومْ بࣹي مࣷومْ اࣵدِّنَ، اَڪْ لِ ݖِ بِيرَمْ؞
‏27 بُ لࣹينْ ڪُ گࣴمُلْ وࣷويࣹي لࣹڪّ، سُ نگࣹينْ نَنگࣷو، مبࣷولࣹمْ لُ مُ لࣹينْ تَاجَلْ، لࣹڪّلࣹينْ، تࣹ بَݧْ ݖࣹي لُقَتُ لُيْ اِنْدِ نگࣵنتّ ݖِمْ خࣹلْ؞ ‏28 وَايࣹ بُ لࣹينْ ڪࣹنّ نࣹي:‏ «لِي ݧَمْ وُ ݧُ سَرْخَلْ تُورْ يِ لَ،» بُلࣹينْ لࣹڪّ، ندَخْ ڪَ لࣹينْ ڪࣷ خَمَلْ، ندَخْ اِتَمْ، نگࣵنتّ لُ وَخْ جِيْ اِنْدِ ݖِمْ خࣹلْ؞ ‏29 وَخُمَ سࣹينُمْ خࣹلْ يࣹينْ، وَايࣹ خࣹلُ ڪࣹنࣹينْ ڪَ لاَيْ وَخْ؞
اَنَ سَخْ لُ وَرَلْ سَمَ سَݧْ-سَݧُ بࣷݒّ دِ اَجُ ݖِ اࣵتّࣹبْ ڪࣹنࣹينْ ڪُ اَمْ خࣹلَمْ اَمَلْ نگࣵنتّ؟ ‏30 بُ مَ يࣺڪَّتࣹي ڪࣵدُّگْ ݖَنتّ بَ لࣹڪّ سَمَ وࣵلّ، اَنَ لࣹيسْ مَيْ سِڪّࣹ ݖِ اَوْ ݧَمْ وُ مَ سَنْتࣹ يࣵلَّ؟
‏31 ڪࣷنْ نَگْ سُ نگࣹينْ دِ لࣹڪّ، مبَا نگࣹينْ دِ نَانْ، مبَا نگࣹينْ دِ دࣹفْ لࣹنࣹينْ، لࣺݒّ لُ نگࣹينْ دِ دࣹفْ، وُتْلࣹينْ ڪࣷو تࣹرَلࣹ يࣵلَّ؞ ‏32 بُلࣹينْ مَيْ ڪࣹنّ لُ مُيْ دࣹفْ نجࣷمْبࣹ:‏ دُ يَوُتْ يِ، دُ جَامْبُرْ ݧِ، دُ مبࣷولࣷومْ يࣵلَّ مِ؞ ‏33 نࣷونُ لاَيْ دࣹفْ مَنْ ݖِ سَمَ بࣷݒّ:‏ ݧࣺݒّ لاَيْ وُتَ نࣹيخْ، ݖِ لࣺݒّ؞ سࣵڪُّوُمَ سَمَ نجَرِݧَلْ بࣷݒّ، وَايࣹ نجَرِݧَلْ لِمْ بُ گࣴنَ تَڪُّ، مࣷومْ لاَيْ سࣵڪُّ، نگِرْ ݧُ تࣹخࣹ؞
