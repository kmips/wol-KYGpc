لُ جࣴمْ ݖِ سࣷرِ خَيْمَبْ جَامࣷوڪَايْ بِ
26
‏1 «خَيْمَبْ جَامࣷوڪَايْ بِ نَگْ، دࣹفَرࣹ ڪࣷ فُڪِّ سࣷرِ رَبَلِ لࣹي، مُ اࣵندّ اَڪْ وࣴݧْ گُ بَخَ تࣹ لاَلْ يࣷلࣹتْ اَڪْ وࣴݧْ گُ خࣹوَرْ اَڪْ گُ خࣷنقّ ݖُرّ، تࣹ نگَ ݧَوَالࣹ ݖَ نَتَالِ سࣹرُبْ؞ ‏2 گُدَّايُ سࣷرْ بُ نࣹڪّ نَ دࣷونْ ݧَارْ فُڪِّ خَسَبْ اَڪْ جُرٗومْ ݧࣹتّ، يَاتُوَايْ بَ دِ ݧࣹينْتِ خَسَبْ، تࣹ سࣷرْ يࣺݒّ اَيْ تࣷلّࣷو؞ ‏3 يِي جُرٗومِ سࣷرْ دَيْ تَقْلࣷو ݖَ گُدَّايْ بَ، جُرٗومِ سࣷرْ يَ ݖَ دࣹسْ تَقْلࣷو نࣷونُ؞ ‏4 ݧَوْلُلْ اَيْ تَڪُّ يُ بَخَ تࣹ لاَلْ يࣷلࣹتْ ݖَ ݖَتَلْ سࣷرْ، بَ مُجّ ݖَ بࣹنّ تَقَلَنْتࣹ بَ، تࣹ ݧَوࣹ ڪࣷ نࣷونُ اِتَمْ ݖَ ݖَتَلْ سࣷرْ، بَ جࣹيخَلْ تَقَلَنْتࣹ بَ ݖَ دࣹسْ؞ ‏5 جُرٗومْ فُڪِّ تَڪُّ نگَيْ دࣹفْ ݖِ ݖَتَلْ بࣹنّ سࣷرْ بِ، اَڪْ جُرٗومْ فُڪِّ تَڪُّ ݖِ ݖَتَلْ سࣷرْ، بِ جࣹيخَلْ تَقَلَنْتࣹ بِ ݖِ دࣹسْ، تࣹ نَ تَڪُّ يِ يࣹمࣷو؞ ‏6 سُ ڪࣷ دࣹفࣹي نگَ دࣹفَرْلُ جُرٗومْ فُڪِّ لࣷنڪُ يُ وُرُسْ يُ ݧُيْ جࣷڪّلࣹي سࣷرْ يِ، بَ جَامࣷوڪَايْ بِ دࣷونْ بࣹنّ؞
‏7 «تࣹ اِتَمْ رࣵبّلُلْ اَيْ سࣷرِ ڪَوَرِ بࣴيْ، نگِرْ دࣹفْ ڪࣷ ݧَارࣹيلُ خَيْمَ، مُيْ ڪِيرَايْ ݖِ ڪَوْ جَامࣷوڪَايْ بِ؞ فُڪّ اَڪْ بࣹنّ لَ سࣷرْ يِيْ دࣷونْ؞ ‏8 گُدَّايُ سࣷرْ بُ نࣹڪّ دِ فَنْوࣹيرِ خَسَبْ، يَاتُوَايْ بِ دِ ݧࣹينْتِ خَسَبْ، تࣹ فُڪِّ سࣷرْ يࣹيڪْ بࣹنّ يࣺݒّ اَيْ تࣷلّࣷو دَيࣷ؞ ‏9 تَقْلࣹيلْ يِي جُرٗومِ سࣷرْ، جُرٗومْ بࣹنِّ سࣷرْ يِ ݖِ دࣹسْ اِتْ نࣷونُ، تࣹ جُرٗومْ بࣹنّࣹيلُ سࣷرْ بِ، نَنگَ ڪࣷ لࣹمْ، فِ فࣺيتࣹيڪْ ڪَنَمْ خَيْمَ بِ؞ ‏10 سࣵڪَّلْ جُرٗومْ فُڪِّ تَڪُّ ݖِ ݖَتَلْ سࣷرْ، بِ مُجّ ݖِ بࣹنّ تَقَلَنْتࣹ بِ، اَڪْ جُرٗومْ فُڪِّ تَڪُّ ݖِ ݖَتَلْ سࣷرْ، بِ جࣹيخَلْ تَقَلَنْتࣹ بِ ݖِ دࣹسْ؞ ‏11 سࣵڪَّلْ جُرٗومْ فُڪِّ لࣷنڪُ يُ خࣵنجَرْ تࣹ نگَ وࣹ لࣷنڪُ يِ ݖِ تَڪُّ يِ، دَلْدِ جࣷڪّلࣹ ݧَارِ تَقَلَنْتࣹ يِ، مُيْ بࣹنّ خَيْمَ؞ ‏12 لِيْ لَنگّ تࣹ اࣴݒّࣹ ݖِ سࣷرِ خَيْمَ بِ نَگْ، دِ بࣹنࣹينْ گࣹنّ-وࣵلُّ سࣷرْ بِ، نَ لَنگࣹ ݖِ گَنَّاوْ جَامࣷوڪَايْ بِ؞ ‏13 وَايࣹ بࣹنّ خَسَبْ بِ اࣴݒّࣹ وࣹتْ گُ نࣹڪّ ݖِ گُدَّايُ سࣷرِ خَيْمَ بِ، نَ لَنگࣹ ݖِ ݧَارِ وࣹتِ جَامࣷوڪَايْ بِ، نگِرْ يِيرࣹ ڪࣷ ڪࣷ؞ ‏14 نَنگَ دࣹفَرْلُلْ خَيْمَ بِ مَلاَنُمْ دࣹرِ ڪُويْ يُ ݧُ سٗوبْ خࣷنقّ، اَڪْ مࣹنࣹينْ مَلاَنُمْ دࣹرِ ݒِيݒِ مُيْ تࣹگࣹ ݖَ ڪَوَمْ؞
لُ جࣴمْ ݖِ ڪࣹنُيْ جَامࣷوڪَايْ بِ
‏15‏-17 «سࣵڪَّلَلْ جَامࣷوڪَايْ بِ اَيْ لࣵݖِّ بَنْتُ سࣺنگّ يُيْ سَمْݒࣹ؞ نَ لࣵݖّ وُ نࣹڪّ اَمْ ݧَارِ ڪࣹنُ يُ ݧُ لࣴڪّلࣹي گَلَنْ يُ ندَوْ، تࣹ نگَ دࣹفْ لࣵݖِّ جَامࣷوڪَايْ بِ يࣺݒّ نࣷونُ؞ نَ تَخَوَايُ لࣵݖّ وُ نࣹڪّ دِ فُڪِّ خَسَبْ، يَاتُوَايْ بِ دِ خَسَبْ اَڪْ گࣹنّ-وࣵلّ؞ ‏18 ݖِ وࣵلُّ لࣵݖِّ جَامࣷوڪَايْ بِ، سࣴسَلࣹيلْ ݧَارْ فُڪِّ لࣵݖّ فِ فࣺيتࣹ بࣴجْ-سَالُمْ، ‏19 بࣷولࣹ ڪࣷوڪْ ݧࣹينتّ فُڪِّ تࣹگُڪَايِ خَالِسْ يُيْ رࣷنُ ڪࣹنُيْ ݧَارْ فُڪِّ لࣵݖّ يِ، ݧَارِ تࣹگُڪَايْ اَيْ رࣷنُ لࣵݖّ وِ جِيتُ نگِرْ ݧَارِ ڪࣹنࣷومْ يِ مُيْ اࣵنْدَلْ، ݧَارِ تࣹگُ تࣷݒّ ݖِ، رࣷنُ وࣺݒّ لࣵݖّ وُ ݖِ تࣷݒّ، نگِرْ ݧَارِ ڪࣹنࣷومْ يِ مُيْ اࣵنْدَلْ؞ ‏20 گࣹنࣹينْ وࣹتُ جَامࣷوڪَايْ گِ فࣺيتࣹ بࣴجْ-گࣵنَّارْ اِتَمْ، نَ دࣷونْ ݧَارْ فُڪِّ لࣵݖّ ‏21 اَڪْ سࣹينِ ݧࣹينتّ فُڪِّ تࣹگُيْ خَالِسْ، ݧَارِ تࣹگُ رࣷنُ لࣵݖّ وِ جِيتُ، ݧَارِ تࣹگُ تࣷݒّ ݖِ، رࣷنُ وࣺݒّ لࣵݖّ وُ ݖِ تࣷݒّ؞ ‏22 گَنَّاوُ جَامࣷوڪَايْ گِ فࣺيتࣹ سࣷوُ، سࣵڪَّلْ ڪࣷ جُرٗومْ بࣹنِّ لࣵݖّ؛ ‏23 نگَ تࣹگْ ݖِ ݧَارِ لࣵݖّ نگِرْ ݖࣷلِّ جَامࣷوڪَايْ بِ ݖِ گَنَّاوْ؞ ‏24 نَݧُ اࣵنْدَنْدࣷو ݖِ سُوفْ تࣹ لࣴڪّࣷو ݖَ ڪَوْ ݖِ بࣹنّ لَمْ ڪࣹݒّ؞ نࣷونُ لَݧُيْ بِنْدࣷو ݧࣷومْ ݧَارْ؛ ݧࣷويْ دࣷونْ ݧَارِ ݖࣷلّ يِ؞ ‏25 مُيْ جُرٗومْ ݧࣹتِّ لࣵݖّ اَڪْ سࣹينْ تࣹگُيْ خَالِسْ، دِ فُڪِّ تࣹگُ اَڪْ جُرٗومْ بࣹنّ، ݧَارِ تࣹگُ رࣷنُ لࣵݖّ وِ جِيتُ، ݧَارِ تࣹگُ تࣷݒّ ݖِ، رࣷنُ وࣺݒّ لࣵݖّ وُ ݖِ تࣷݒّ؞
‏26 «نَنگَ سࣵڪّ اِتَمْ اَيْ گَلَنِ بَنْتُ سࣺنگّ:‏ جُرٗومِ گَلَنْ يُيْ اࣵندّ اَڪْ لࣵݖّ يِ ݖِ گࣹنّ وࣹتُ جَامࣷوڪَايْ بِ، ‏27 اَڪْ جُرٗومِ گَلَنْ نگِرْ لࣵݖِّ وࣹتْ گِ ݖِ دࣹسْ، اَڪْ جُرٗومِ گَلَنْ نگِرْ لࣵݖّ، يِ ݖِ گَنَّاوْ جَامࣷوڪَايْ بِ، فِ فࣺيتࣹ سࣷوُ؞ ‏28 گَلَنُ دِگّ بِ، ݖِ دِگّ لࣵݖّ يِ لَيْ نࣹڪّ، دَلࣹ ڪࣷ ݖَتْ بَ ݖَتْ؞ ‏29 خࣷوبَلْ لࣵݖّ يِ وُرُسْ تࣹ نگَ سࣵڪَّلْ لࣵݖّ وُ ݖِ نࣹڪّ اَيْ لَمِ وُرُسْ يُيْ تࣺيࣹ گَلَنْ يِ، تࣹ نگَ خࣷوبْ گَلَنْ يِ اِتْ وُرُسْ؞ ‏30 سُ ڪࣷ دࣹفࣹي نگَ تَخَوَلْ جَامࣷوڪَايْ بِ، نِ ݧُ لَ ڪࣷ وࣷنࣹ ݖِ ڪَوْ تُندّ وِ؞
لُ جࣴمْ ݖِ رِدࣷ بِ
‏31 «سࣵڪَّلْ رِدࣷ بُ ݧُ رࣵبّࣹ وࣴݧْ گُ بَخَ تࣹ لاَلْ يࣷلࣹتْ، اَڪْ وࣴݧْ گُ خࣹوَرْ اَڪْ گُ خࣷنقّ ݖُرّ، اَڪْ وࣴݧُ لࣹي، ݧُ ݧَوَالࣹ ݖَ نَتَالِ سࣹرُبْ، ‏32 تࣹ نگَ وࣺڪّ ڪࣷ ݖِ ݧࣹينْتِ جࣴنِ سࣺنگّ يُ ݧُ خࣷوبْ وُرُسْ، جࣴنْ يِ اࣵندّ اَڪْ سࣹينْ وࣺڪُّڪَايِ وُرُسْ، تࣹ نگَ مٗولَلْ جࣴنْ يِ ݧࣹينْتِ تࣹگُيْ خَالِسْ؞ ‏33 وࣹوَلْ رِدࣷ بِ ݖِ رࣷنْ لࣷنڪُ يِ، سُ ڪࣷ دࣹفࣹي نگَ يٗبُّ گَالُ اࣵلُّوَيْ سࣹيدࣹ سِ فࣷوفُ ݖِ گَنَّاوْ رِدࣷ بِ، رِدࣷ بِ خࣵجَّلࣹيلْ لࣹينْ بࣺرَبْ بُ سࣹلّ بِ، اَڪْ بࣺرَبْ بُ سࣹلّ بࣹي سࣹلّ؞ ‏34 تࣹگَلْ ڪُبࣹيرْ گَ ݧُيْ اَمَلࣹ اَگْ نجࣷتْلاَيْ ݖَ ڪَوْ گَالْ گَ اࣵلُّوَيْ سࣹيدࣹ سَ دࣹنݖࣹ، ݖَ بِيرْ بࣺرَبْ بُ سࣹلّ بَا سࣹلّ؞ ‏35 تࣹگَلْ تَابَلْ جَ ݖَ بِتِ، ݖَ ڪَنَمْ رِدࣷ بَ، تࣹ نگَ تࣹگْ تࣹگُڪَايُ لࣵمݒّ بَ ݖَ وࣹتُ بࣴجْ-سَالُمُ جَامࣷوڪَايْ بَ، مُ جࣵڪَّارْلࣷوڪْ تَابَلْ جَ، تَابَلْ جَ نَگْ، نگَ تࣹگࣹ ڪࣷ وࣹتُ بࣴجْ-گࣵنَّارْ؞
لُ جࣴمْ ݖِ ڪِيرَايَلْ بُنْتُ خَيْمَ بِ
‏36 «سࣵڪَّلَلْ بُنْتُ خَيْمَ بِ ڪِيرَايْ لُ ݧُ رࣵبّࣹ وࣴݧْ گُ بَخَ تࣹ لاَلْ يࣷلࣹتْ، اَڪْ وࣴݧْ گُ خࣹوَرْ اَڪْ گُ خࣷنقّ ݖُرّ اَڪْ وࣴݧُ لࣹي، مُيْ لِگّࣺييُ رࣵبّ بُ رَفࣹتْ؞ ‏37 نگَ سࣵڪَّلْ نَگْ ڪِيرَايْ لِ جُرٗومِ جࣴنِ سࣺنگّ، خࣷوبْ وُ ݖِ نࣹڪّ وُرُسْ، سࣹينِ وࣺڪُّڪَايْ دِ وُرُسْ، تࣹ نگَ مٗولَلْ جࣴنْ يِ جُرٗومِ تࣹگُيْ خࣵنجَرْ؞
