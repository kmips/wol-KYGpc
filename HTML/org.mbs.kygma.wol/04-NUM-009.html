لُ جࣴمْ ݖِ مࣵگَّلُ بࣺسُبْ مُݖّ
9
‏1 اَجِ سَخْ جِ وَخْ نَ مُوسَىٰ ݖَ مࣵنْدِݝُ سِنَيِ، ݖَ سࣹينْ اَتُمْ ݧَارࣹيلْ، گَنَّاوْ بَ ݧُ گࣺنّࣹي رࣺيوُمْ مِسْرَ ݖَ وࣹيرْ وَ جࣴڪّ؞ مُ نࣹ ڪࣷ:‏ ‏2 «نَ بࣵنِّ إِسْرَاءِيلْ دِ مࣵگَّلْ بࣺسُبْ مُݖّ بَ، سَا يُ جࣷتࣹي؞
‏3 «فُڪِّ فَنْ اَڪْ ݧࣹينتّ ݖِ وࣹيرْ وِي، دِگَّنْتࣹ نگࣷونْ اَڪْ جَنتّ بُ سࣷ، ݖِ نگࣹينْ ڪࣷيْ مࣵگَّلْ، سَا يُ جࣷتࣹي؞ نَ مُ دࣴݒّࣷوڪْ مبࣷولࣹمْ دࣷگَلِ يࣷونَمْ اَڪْ اࣵتّࣹيْ يࣷونَمْ، نَنگࣹينْ ڪࣷ مࣵگَّلࣹ نࣷونُ؞»
‏4 مُوسَىٰ نَگْ سَنتّ بࣵنِّ إِسْرَاءِيلْ نࣹ لࣹينْ ݧُ مࣵگَّلْ بࣺسُبْ مُݖّ بِ؞ ‏5 ݧُ دَلْدِ مࣵگَّلْ بࣺسُبْ مُݖّ بِ ݖِ وࣹيرْ وِ جࣴڪّ، فُڪِّ فَنَمْ اَڪْ ݧࣹينتّ، دِگَّنْتࣹ نگࣷونْ اَڪْ جَنتّ بُ سࣷ، ݖَ مࣵنْدِݝُ سِنَيِ؞ نࣷونࣹي ڪࣷ اَجِ سَخْ جِ سَنْتࣹ وࣷونْ مُوسَىٰ، نَ لَ ڪࣷ بࣵنِّ إِسْرَاءِيلْ دࣹفࣹ؞
‏6 ݖِ بِيرْ لࣷولُ مُ اَمْ اَيْ نِتْ ݧُ اَبْ نࣺيوْ تَخْ ݧُ سࣷبࣹوُ، بَ مَنُݧࣷو مࣵگَّلْ بࣺسُبْ مُݖّ بِ بࣺسُبْ ڪࣹرࣷوگْ؞ ݧُ دِڪّ بَ ݖِ مُوسَىٰ اَڪْ هَـٰرُونَ ݖَ بࣺسْ بَ؞ ‏7 ݧࣷوݧُ نَگْ نࣹ مُوسَىٰ:‏ «نُنْ دࣹ، دَنࣷو سࣷبࣹوُ ندَخْ اَبْ نࣺيوْ، وَايࣹ لُ نُ تࣹرࣹي جࣷخࣹسِ سَرَخُ اَجِ سَخْ جِ ݖِ وَخْتࣷومْ، ݖِ دِگّ بࣵنِّ إِسْرَاءِيلْ؟» ‏8 مُوسَىٰ نࣹ لࣹينْ:‏ «تَخَوْلࣹينْ، بَ مَ دࣺگّ لُ اَجِ سَخْ جِ دِ سَنْتَانࣹ ݖِ سࣹينْ مبِرْ؞»
‏9 اَجِ سَخْ جِ نࣹ مُوسَىٰ:‏ ‏10 «وَخَلْ بࣵنِّ إِسْرَاءِيلْ نࣹ لࣹينْ، ڪࣺݒّ ڪُ بࣷڪّ ݖِ يࣹينْ تࣹيْ، مبَا مُ بࣷڪّ ݖِ سࣹينْ ڪُوتَايْ اࣴلّࣴگْ، سُ سࣷبࣹوࣷو ندَخْ اَبْ نࣺيوْ، مبَا مُ نࣹڪّ ݖِ يࣷونْ وُ سࣷرࣹ، دُ تࣹرࣹ مُ مࣵگَّلْ بࣺسُبْ مُݖّ بِ نگِرْ اَجِ سَخْ جِ؞ ‏11 وَايࣹ ݧَارࣹيلُ وࣹيرْ وَ، ݖَ فُڪِّ فَنَمْ اَڪْ ݧࣹينتّ، دِگَّنْتࣹ نگࣷونْ اَڪْ جَنتّ بُ سࣷ، ݖَ لَ ڪࣷيْ مࣵگَّلْ؞ مبُرُ مُ اَمُلْ لَوِيرْ اَڪْ خࣷبْ يُ وࣹخْ لَيْ بࣷولࣹيڪْ گࣵتُّ سَرَخْ بِ، لࣹڪّ ڪࣷ؞ ‏12 بُمُ ݖِ وَݖّ دَرَ مُ فَنَانْ، تࣹ بࣹنّ يَخْ، بُمُ ڪࣷ ݖِ دَمّ؞ نَ مُ دࣴݒّࣷوڪْ مبࣷولࣹمْ دࣷگَلِ يࣷونِ بࣺسُبْ مُݖّ، نَ ڪࣷ مࣵگَّلࣹ نࣷونُ؞ ‏13 ڪُ سࣹتْ نَگْ تࣹ نࣹڪُّلْ ݖِوْ يࣷونْ، سُ ݧࣵڪّࣹي مࣵگَّلْ بࣺسُبْ مُݖّ بِ، ڪࣷوڪُ دࣹيسْ ڪࣷيْ دَگّࣹ ݖِوْ اَسْڪَنَمْ، ندَخْ سَرَخُ اَجِ سَخْ جِ مُ جࣷخࣹوُلْ ݖِ بࣺسَمْ؞ ڪࣷوڪُ مࣷو گࣵدُّ بࣵڪَّارُ بࣷݒَّمْ؞
‏14 «سُ دࣹي اَبْ دࣷخَنْدࣺيمْ بُ نگࣹينْ دࣴڪَّلْ، سَݧْ نَا مࣵگَّلْ بࣺسُبْ مُݖّ بِ ݧࣹيلْ اَجِ سَخْ جِ؞ نَ مُ دࣴݒّࣷوڪْ مبࣷولࣹمْ دࣷگَلِ يࣷونِ بࣺسُبْ مُݖّ، اَڪْ اࣵتّࣹيْ يࣷونَمْ، نِ لَ ڪࣷيْ مࣵگَّلࣹ؛ بࣹنّ دࣷگَلْ بِ لَيْ دࣷونْ ݖِ يࣹينْ، مُيْ دࣷخَنْدࣺيمْ بِ، دِ نجُدُّ-جِ-رࣺيوْ؞»
اَجِ سَخْ جِ يِيرْ نَ خَيْمَبْ ندَجࣹ مَ
‏15 ڪࣹرࣷوگْ بࣺسْ بَ ݧُ سَمْݒࣹي مࣵڪَّانْ مَ، نِيرْ وَ دَا مُورْ مࣵڪَّانُ خَيْمَبْ سࣹيدࣹ سَ؞ ݖَ نگࣷونْ سَ لُ مࣹلْ نِ سَوَرَ تِيمْ مࣵڪَّانْ مَ، بَ ݖَ اࣴلّࣴگْ سَ؞ ‏16 نࣷونُ لَ ڪࣷ نِيرْ وَ جࣵݒّࣷو مُورࣹ، بُ گُدّࣹي مُ مࣹلْ نِ سَوَرَ؞ ‏17 فُ نِيرْ وَ مَسَ بࣵيِّڪࣷو فَ تِيمْ خَيْمَ بَ، بࣵنِّ إِسْرَاءِيلْ دَيْ فَبُ تࣷݒّ ڪࣷ، تࣹ فࣺݒّ فُ نِيرْ وَ مَسَ دَلࣹ، فَ لَ بࣵنِّ إِسْرَاءِيلْ دِ فࣵڪّ اَبْ دَلْ؞ ‏18 ݖِ ڪࣵدُّگْ اَجِ سَخْ جِ لَ بࣵنِّ إِسْرَاءِيلْ دَانْ فَبࣷو، ݖِ ڪࣵدُّگْ اَجِ سَخْ جِ لَݧُ دَانْ دَلࣹ، تࣹ مبࣷولࣹمْ فَنْ يَ نِيرْ وَيْ دَلْ، تِيمْ مࣵڪَّانْ مَ، ݧࣷومْ اِتْ دَݧُيْ دَلْ؞ ‏19 سُ نِيرْ وَ يࣵگّࣹي ݖَ ڪَوْ مࣵڪَّانْ مَ اِتْ، اَڪْ لُ فَنْ يَ بَرࣹ بَرࣹ، بࣵنِّ إِسْرَاءِيلَ نگَيْ دࣺنڪࣷو دࣺنڪَانࣹيْ اَجِ سَخْ جِ، وَايࣹ دُݧُ دࣹمْ؞ ‏20 لࣹيگْ-لࣹيگَا فَنْ يُ نࣺيوْ لَ نِيرْ وَيْ دَلْ-لُ فَ تِيمْ مࣵڪَّانْ مَ؞ وَايࣹ ݖِ ڪࣵدُّگْ اَجِ سَخْ جِ دَالْ، لَݧُيْ دَلࣹ تࣹ ݖِ ڪࣵدُّگْ اَجِ سَخْ جِ لَݧُيْ فَبࣷو؞ ‏21 لࣹيگْ-لࣹيگَا اِتْ نِيرْ وَ دَلْ-لُ اَگْ نگࣷونْ، فَنَانْ، بَ بࣴتْ سࣹتْ، مُ سࣵقِ، ݧُ فَبُ اࣵندّ اَڪْ مࣷومْ؞ مُيْ گُدِّ، دِ بࣴݖّࣴگْ، بُ نِيرْ وَ جٗگَا، رࣹڪّ، ݧُ فَبُ؞ ‏22 سُ دࣷونْ ݧَارِ فَنْ، سُ دࣷونْ وࣹيرْ، مبَا اَيْ فَنِ فَنْ لَ نِيرْ وَ تَخَوْ تِيمْ مࣵڪَّانْ مَ، دࣹسْ فَ اِتْ، بࣵنِّ إِسْرَاءِيلْ دَݧُيْ دَلْ، وَايࣹ دُݧُ دࣹمْ؞ بُ جٗگࣹي رࣹڪّ لَݧُيْ فَبُ؞ ‏23 ݖِ ڪࣵدُّگْ اَجِ سَخْ جِ لَݧُ دَا دَلࣹ، ݖِ ڪࣵدُّگْ اَجِ سَخْ جِ لَݧُ دَا فَبࣷو؞ دࣺنڪَانࣹيْ اَجِ سَخْ جِ دَالْ لَݧُيْ دࣺنڪࣷو، ݖِ ڪࣵدُّگْ اَجِ سَخْ، جِ مُوسَىٰ دِ جࣷتّلِ؞
