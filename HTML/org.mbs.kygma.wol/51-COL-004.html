4
‏1 يࣹينْ ݧِيْ سَنگّ يِ، يࣷرْلࣹينْ سࣹينِ جَامْ، يࣷرِنْ وُ جُبْ تࣹ اَوْ يࣷونْ، تࣹ نگࣹينْ خَمْ نࣹ يࣹينْ اِتَمْ اَمْ نگࣹينْ اَبْ سَنگّ فَ اَسَمَانْ؞
يࣹنࣹينْ وَرُگَرَ نگِ
‏2 اَگْ ݧَانْ نَگْ، سَخࣷولࣹينْ ڪࣷ، تࣹ نگࣹينْ تࣹيوْلُ ݖِ، بࣷولࣹ ݖِ اَگْ ݖَنتّ؞ ‏3 تࣹ نگࣹينْ ݧَانَالࣹ نُ ݖِ اِتَمْ، نگِرْ يࣵلَّ اُبِّلْ نُ بُنْتُ خَمْلࣹ ڪࣵدُّ گِ، نگِرْ سِيوْتَانࣹ مبٗوتْ مِ ݖِ اَلْمَسِ، لِ وَرَلْ ݧُ جࣺنگّ مَ؞ ‏4 ݧَانْلࣹينْ مَ مَنْ ڪࣷو وَخࣹ بَ مُ لࣹيرْ نࣵݧّ نِ مُ مَ وَرࣹ؞
‏5 جࣴفࣹلࣹينْ خࣹلْ مُ رَفࣹتْ سࣹينْ دِگَّنْتࣹ اَڪْ اَجِ گࣴمَدِ ݧِ ݖِ بِتِ، تࣹ جࣷتْ گُ نگࣹينْ اَمْ، فࣹخࣹلࣹينْ بَ جَرِݧࣷو ڪࣷ؞ ‏6 نَ سࣹينْ ڪࣵدُّ اࣵندّ اَڪْ يِوْ تࣹ اَمْ سࣷلࣷ، تࣹ نگࣹينْ خَمْ نُ نگࣹينْ دِ تࣷنْتࣷو ڪُ نࣹڪّ، تࣷنتّ لِ وَرْ؞
ݒٗولْ تࣵگُّ نَ وَا ڪࣷلࣷسْ
‏7 مبࣷولࣹمْ لُ جࣴمْ ݖِ مَنْ نَگْ، تِسِگْ دِنَ لࣹينْ ڪࣷ خَمَلْ؞ مبࣷڪّ لَ، سࣷݒّࣹ لَ، جَوْرِݧْ جُ وٗورْ لَ، تࣹ اِتْ سَمَ مبࣷڪُّمْ جَامْ لَ، ݖِ لِگّࣺييُ بࣷرࣷومْ بِ؞ ‏8 لِي سَخْ مࣷو تَخْ مَ يٗنِّ ڪࣷ ݖِ يࣹينْ:‏ خَمَلْ لࣹينْ لُ جࣴمْ ݖِ نُنْ، اَڪْ اِتَمْ، تࣹيوَايَمْ بُ دࣴفَلْ سࣹينْ خࣷلْ؞ ‏9 اࣷنࣹسِمْ لَيْ اࣵنْدَلْ، سُنُ مبࣷڪّ مُ وٗورْ، دِ سُنُ سࣷݒّࣹ، تࣹ مُ بࣷڪّ ݖِ يࣹينْ؞ ݧࣷو لࣹينْ دِ خَمَلْ مبࣷولࣹمْ لُ فِ خࣹوْ؞
‏10 اَرِسْتࣵرڪّ مِ مَ بࣷڪَّلْ ڪَسࣷ بِ، مُ نگِ لࣹينْ دِ نُيُ، مࣷومْ اَڪْ مࣵرڪّ رَڪُّ بَرْنَبَ؞ ݖِ مبِرُ مࣵرڪّ مࣷومُ نگࣹينْ جࣷتࣷونْ ندِگَلْ، نگِرْ بُ دِڪّࣹي، نگࣹينْ تࣹرَلْ ڪࣷ؞ ‏11 عِيسَىٰ مِ ݧُيْ وَخْ يُسْتُسْ اِتَمْ، مُ نگِ نُيࣷو؞ لِ بࣷڪّ ݖِ اَسْڪَنُ يَوُتْ، ݧࣷومْ رࣹڪّ اَيْ لِگّࣺييَنْدࣷو اَڪْ مَنْ نگِرْ نگُورُگْ يࣵلَّ، تࣹ ݧࣷو سࣹرَلْ سَمَ خࣷلْ؞ ‏12 اَبْ نُيࣷو اِتْ ݧࣹيلْ نَ لࣹينْ، تُڪّࣹي ݖِ جَامُبْ اَلْمَسِ عِيسَىٰ، اࣹݒَفْرَسْ، مِ بࣷڪّ ݖِ يࣹينْ؞ ڪُ جࣵمْبَارࣹ ݖِ سَخࣷو لࣹينَ ݧَانَلْ، نگِرْ سࣹينُبْ تَخَوَايْ سࣷتِّ تࣹ مَتْ سࣴڪّ ݖِ جࣴفࣹ مبࣷولࣹمْ لُيْ ݖࣷوبَرࣹيْ يࣵلَّ؞ ‏13 مَا ڪࣷ سࣹيدࣹيلْ نࣹ ݖࣷونࣷيْ ݧَانْ بُ رࣺيْ لَ نࣹڪّࣹ، نگِرْ يࣹينْ اَڪْ وَا لَوْدِسࣹ اَڪْ وَا يࣹرَݒࣷلِسْ؞ ‏14 اَبْ نُيࣷو اِتْ ݧࣹيلْ نَ لࣹينْ، تُڪّࣹي ݖِ سُنُ سࣷݒّࣹ، لُوگْ، دࣷڪْتࣷورْ بِ، اَڪْ دࣹمَسْ؞ ‏15 نُيُلْ-لࣹينْ نُ بࣷڪّ يِ دࣴڪّࣹ لَوْدِسࣹ، اَڪْ سُنُبْ جِگࣺينْ نِمْفَ، اَڪْ مبࣷولࣷومْ گࣴمْڪَتْ، مِيْ دَجࣹ ڪࣴرَمْ؞
‏16 گَنَّاوْ بُ نگࣹينْ جࣵنگࣹي بِرَلْ، بَتَاخَلْ بِي ݖِ سࣹينْ بِيرْ، فࣹخࣹلࣹينْ اِتْ بَ ݧُ جࣵنگّ ڪࣷ، بِرَلْ، فَ مبࣷولࣷومْ گࣴمْڪَتْ مَ فَ لَوْدِسࣹ، تࣹ يࣹينْ اِتْ بَتَاخَلْ بَ نگࣹينْ دِ جࣷتࣹ لَوْدِسࣹ، نَنگࣹينْ ڪࣷ جࣵنگّ بِرَلْ؞
‏17 وَخَلْ-لࣹينْ مَ اَرْسِݒّ نࣹ ڪࣷ:‏ «نࣹيلْ جࣵڪّ ݖِ لِگّࣺييْ بِ نگَ سَسࣷو ݖِ بࣷرࣷومْ بِ، نگِرْ نگَ سࣷتَّلْ ڪࣷ؞»
‏18 مَنْ ݒٗولْ نَگْ مَا لࣹينْ بِندّ نُيࣷو بِي ݖِ سَمَ لࣷخࣷلْ بࣷݒّ؞ فࣵتّلِڪُلࣹينْ نࣹ دَݧُ مَا تࣴجْ ڪَسࣷ؞ يَلْ نَ اَوْ يِوْ اࣵندّ اَڪْ يࣹينْ!‏
