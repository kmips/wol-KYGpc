إِسْحَاقَ جࣴلْ نَ رࣹبࣹڪَ سࣷخْنَ
24
‏1 إِبْرَاهِيمَ گُدّࣷونْ نَ فَنْ، بَ مࣵگَّتْ لࣷولْ، تࣹ اَجِ سَخْ جِ بَرْڪࣹيلْ ڪࣷ ݖِ لࣺݒّ؞
‏2 مُ اَمْ بࣺسْ إِبْرَاهِيمَ وَخْ سُرْگَامْ، بِ گࣴنَ يࣵگّ ݖِ ڪࣴرَمْ، تࣹيْ سَيْتُ اَلَلَمْ جࣺݒّ، نࣹ ڪࣷ:‏ «دُگَلَلْ سَ لࣷخࣷ ݖِ سَمَ سُوفُ لُݒّ¤، ‏3 مَ گِݧْلࣷو لَ ݖِ اَجِ سَخْ جِيْ يࣵلَّيْ اَسَمَانْ اَڪْ سُوفْ نࣹ دࣷو جࣴلَلْ سَمَ دࣷومْ سࣷخْنَ ݖِ بِيرْ جِگࣺينِ وَا ڪَنَانْ، گِ مَ دࣴڪّ ݖِ سࣹينْ بِيرْ؞ ‏4 وَايࣹ دِنگَ دࣹمْ سَمَ رࣺيوْ، مَ مَ ݖࣷسَانࣷو، جࣴلَلْ فَ سَمَ دࣷومْ إِسْحَاقَ سࣷخْنَ؞»
‏5 سُرْگَ بَ نࣹ ڪࣷ:‏ «سُ ندَوْ سَ نَنگُوُلَ اࣵندّ اَڪْ مَنْ ݖِ رࣺيوْ مِي نَگْ؟ ندَخْ ڪࣷنْ دَمَا وَرَ يٗبُّ سَ دࣷومْ بَ ݖَ رࣺيوْ مَ نگَ جٗگࣹ؟» ‏6 إِبْرَاهِيمَ اِتَمْ نࣹ ڪࣷ:‏ «بُ فَ يٗبُّ سَمَ دࣷومْ مُڪّ؞ ‏7 اَجِ سَخْ جِ، يࣵلَّيْ اَسَمَانْ، مࣷومْ مِ مَ جࣴلࣹ سَمَ ڪࣴرْ بَايْ اَڪْ فَ مَ جُدّࣷو، تࣹ وَخْ اَڪْ مَنْ، گِݧَلْ مَ نࣹ مَ، دِنَ جࣷخْ سَمَ اَسْڪَنْ رࣺيوْ مِي، مࣷومْ ݖِ بࣷݒَّمْ دِنَ يࣹبَلْ مَلاَڪَامْ، مُ جِيتُ لَ؞ فࣷوفَ نگَيْ جࣴلَلࣹ سَمَ دࣷومْ سࣷخْنَ؞ ‏8 سُ لَ ندَوْ سِ بُگُّلَ تࣷݒّ، ڪࣷنْ وࣵݖّࣷو نگَ اَڪْ مَنْ؞ وَايࣹ رِڪّ بُلْ يٗبُّ سَمَ دࣷومْ فࣷوفَ؞» ‏9 سُرْگَ بَ دُگَلْ لࣷخࣷومْ ݖَ سُوفُ لُݒُّ إِبْرَاهِيمَ سَنگَمْ بَ، دَلْدِ گِݧْ ݖَ لࣷولَ؞
‏10 بَ لࣷولُ اَمࣹي سُرْگَ بَ جࣴلࣹ فُڪِّ گِلࣺيمْ ݖَ گِلࣺيمِ سَنگَمْ، دࣹمْ، يٗبَّالࣹ لِ گࣴنْ ݖِ مبࣷولࣹمْ خࣹيتِ اَلَلُ سَنگَمْ؞ مَ نگَ جࣴمْ رࣺيوُ مࣹسࣷݒࣷتَمِ، گَ نَخࣷرْ دࣴڪّ؞ ‏11 بَ مُيْ جُبْ دࣴڪّ بَ، دَا يࣺسَلْ گِلࣺيمْ يَ فُ دࣹندّ اَڪُبْ سࣹيَانْ وَخْتُوْ نگࣷونْ، فࣹڪّ جِگࣺينْ ݧَيْ گࣺنّ دِ رࣷوتْسِ؞ ‏12 مُ دَلْدِ نࣹ:‏ «يَوْ اَجِ سَخْ جِ، سَمَ يࣵلَّيْ سَنگّ إِبْرَاهِيمَ، نگَلَّ نَنگُلْ مَ تࣹيْ جِي تࣹ لاَيࣹ بِيرْ سَمَ سَنگّ إِبْرَاهِيمَ؞ ‏13 مَا نگِ نِي تَخَوْ ݖِ بࣴتُ ندࣷخْ مِ، جَنقِ دࣴڪّ بِ دِ رࣷوتْسِ؞ ‏14 جَنقّ بُ مَ وَخْ نࣹ ڪࣷ:‏ ‹دࣷو يࣹنِّڪُ سَ نجَقْ، مَيْ مَ، مَ نَانْ،› مُ نࣹ مَ:‏ ‹نَانَلْ بَ نࣷݒِّ، مَ وࣴگَّلْ لَ سَيْ گِلࣺيمْ،› نَ ڪࣷوڪَ دِ ڪِ نگَ جَگْلࣹيلْ سَ جَامْ إِسْحَاقَ؞ نࣷونُ لاَيْ خَمࣹ نࣹ لاَيࣹ نگَ بِيرْ سَمَ سَنگّ؞»
‏15 لاَتَ مُيْ ݧَانْ بَ نࣷݒِّ سَخْ، گِسْ نَ رࣹبࣹڪَ گࣵدُّ نجَقَمْ، دِ ݧࣴوْ؞ رࣹبࣹڪَ مࣷومُ دِ دࣷومُ بࣹتُوࣹلْ تࣹ دِ سࣴتُ مِلْڪَ اَڪْ نَخࣷرْ، مِيْ مَگُ إِبْرَاهِيمَ؞ ‏16 ندَوْ سُ تَارُ لَ وࣷونْ، دِ اَبْ جَنقّ بُ سࣺيَگُلْ؞ نَڪَ لَ وࣵݖّ بَ ݖَ سࣹيَانْ بَ، دُيْ نجَقَمْ، يࣺيگَاتْ؞ ‏17 سُرْگَ بَ دَوْ، دَجࣹ اَڪْ مࣷومْ نࣹ ڪࣷ:‏ «دࣷو مَ مَيْ تُوتِ ݖِ سَ نجَقْ لِ، مَ نَانْ؞» ‏18 مُ نࣹ ڪࣷ:‏ «اَهَڪَيْ، سَنگّ بِ،» دَلْدِ گَاوْ، يࣹنِّڪُ نجَقْ لَ، مَيْ ڪࣷ مُ نَانْ؞ ‏19 بَ ڪࣷ ندَوْ سِ مَيࣹي، مُ نَانْ بَ نࣷݒِّ، دَ نࣹ ڪࣷ:‏ «خَارَلْ، مَ وࣴگَّلْ لَ گِلࣺيمْ يِ، ݧُ نَانْ بَ نࣷݒِّ؞» ‏20 مُ دَلْدِ گَاوْ، سࣷتِّ نجَقْ لَ ݖَ مبَلْڪَ مَ، دَوَاتْ رࣷوتَلِ گِلࣺيمْ يِ ݖَ سࣹيَانْ بَ؞ ‏21 وَا جَ نࣹ ڪࣷ جࣵڪّ، نࣹ تࣹڪّ، دِ سࣹيتْ، بَ خَمْ ندَخْ اَجِ سَخْ جِ اࣵنْتَلْ نَ يࣷونَمْ اَمْ دࣺيتْ؞
‏22 بَ گِلࣺيمْ يَ نَانࣹي بَ سࣷتَّلْ، وَا جِ جࣴلْ جَارࣷبْ بَڪَّنْ بُ وُرُسْ بُ تࣷلّࣷوڪْ جُرٗومْ بࣹنِّ گَرَامْ، اَڪْ ݧَارِ لَمِ وُرُسْ يُ تࣷلُّ ݖِ تࣺيمࣺيرِ گَرَامْ، تَڪَّلْ ڪࣷ؞ ‏23 مُ نࣹ ڪࣷ:‏ «وَخْ مَ، ڪَنْ مࣷو لَ جُرْ؟» اَڪْ اِتَمْ:‏ «ندَخْ مَنْ نَنࣷو دَلْ سَ ڪࣴرْ بَايْ؟» ‏24 مُ نࣹ ڪࣷ:‏ «بࣹتُوࣹلَ مَ جُرْ، دࣷومُ نَخࣷرْ اَڪْ مِلْڪَ،» ‏25 تࣹگْ ݖَ نࣹ ڪࣷ:‏ «وَاوَاوْ، ݧَخْ مَاڪْ نگࣷوݧْ مَا نگَ فَ، نࣹ گࣵݧّ، تࣹ اِتْ مَنࣹيسْ نَ فَا دَلْ؞» ‏26 وَا جِ نَگْ سُجّٗوتْ، سَنتّ اَجِ سَخْ جِ، ‏27 دَلْدِ نࣹ:‏ «ݖَنتّ ݧࣹيلْ نَ اَجِ سَخْ جِ، سَمَ يࣵلَّيْ سَنگّ، إِبْرَاهِيمَ، مࣷومْ مِ مَسُلَ نࣷݒّࣹي لاَيࣹ بِيرْ سَمَ سَنگّ، تࣹ مَسُ ڪࣷو ݧࣵڪّࣹ وࣷرْمَ؞ مَنْ اِتْ، اَجِ سَخْ جِ وࣷمَّتْ نَ مَ بَ سَمَ ڪࣴرْ بࣷڪِّ سَنگّ!‏»
‏28 جَنقّ بِ نَگْ دَوْ دࣹمْ، نࣹتَّلِ وَا ڪࣴرْ يَايَمْ لَ خࣹوْ لࣺݒّ؞ ‏29 رࣹبࣹڪَ اَمࣷونْ نَ ݖࣵمِّݧْ لُ تُدّ لَبَنْ؞ لَبَنْ دَلْدِ گَاوْتُ گࣺنّ، فࣹڪِّ وَا جَ ݖَ سࣹيَانْ بَ؞ ‏30 بࣷوبَ گِسْ نَ جَارࣷ بَ اَڪْ لَمْ، يَ ݖَ لࣷخࣷلْ جِگࣺينَمْ رࣹبࣹڪَ، تࣹ دࣺگّ نَ جِگࣺينَمْ دِ نࣹتَّلِ لَ ڪࣷ وَا جَ وَخْ؞ لࣷولُ تَخْ مُ دࣹمْ ݖَ وَا جَ، فࣹڪّ ڪࣷ ݖَ گِلࣺيمْ يَ ݖَ سࣹيَانْ بَ؞ ‏31 مُ نࣹ ڪࣷ:‏ «يَوْ مِ اَجِ سَخْ جِ بَرْڪࣹيلْ، اَگْسِلْ بࣷوگْ!‏ لࣷويْ تࣷوگْ ݖِ اࣵلّ بِ؛ وَاجَلَلْ نَا لَ اَبْ دَلْ، تࣹ گِلࣺيمْ يِ اَمْ نَݧُ فُ ݧُ نࣹڪّ؞» ‏32 وَا جَ دَلْدِ اࣵگّ ݖَ ڪࣴرْ گَ، ݧُ سِݒِّ گِلࣺيمْ يَ، مَيْ لࣹينْ ݧَخْ اَڪْ نگࣷوݧْ، مَيْ وَا جَ ندࣷخْ، مُ جࣵنگُ، مࣷومْ اَڪْ ݧِ مُ اࣵنْدَلْ، ‏33 بَ نࣷݒِّ ݧُ تَاجَلْ ڪࣷ لُ مُ لࣹڪّ؞ مُ نࣹ:‏ «دࣺيدࣺيتْ، دُمَ لࣹڪّ مُڪّ ڪَيْ، تࣹ وَخُمَ لِ مَ تَخَ جٗگْ؞» لَبَنْ نࣹ ڪࣷ:‏ «ڪࣷنْ نَگْ وَخَلْ؞»
‏34 مُ نࣹ:‏ «مَنْ سُرْگَبْ إِبْرَاهِيمَ لاَ؞ ‏35 اَجِ سَخْ جِ دَفَ بَرْڪࣹيلْ سَمَ سَنگّ لࣷولْ، بَ مُيْ بࣷرࣷومْ دَرَجَ؞ مَيْ نَ ڪࣷ جُرْ گُ گࣵتّ اَڪْ گُ گُدّ، خَالِسْ اَڪْ وُرُسْ اَڪْ اَيْ سُرْگَ يُ گٗورْ اَڪْ يُ جِگࣺينْ اَڪْ اَيْ گِلࣺيمْ اَڪِ مبَامْ؞ ‏36 سَمَ سࣷخْنَسْ سَنگّ سَارَتَ نَگْ دࣹمْ نَ بَ مࣵگَّتْ، دࣷورَ اَمَلْ سَمَ سَنگّ دࣷومْ جُ گٗورْ؞ دࣷومَمْ جࣷوجُ لَ جࣷخْ لِ مُ مࣷومْ لࣺݒّ؞ ‏37 سَمَ سَنگّ نَگْ گِݧْلࣷو مَ نࣹ:‏ ‹بُلْ جࣴلَلْ مُڪّ سَمَ دࣷومْ سࣷخْنَ ݖِ بِيرْ جِگࣺينِ وَا رࣺيوُ ڪَنَانْ، گِ مَ دࣴڪّ؞ ‏38 وَايࣹ دࣹمَلْ ݖَ سَمَ ڪࣴرْ بَايْ، ݖَ سَمَيْ بࣷڪّ، نگَ جࣴلَلْ فَ سَمَ دࣷومْ سࣷخْنَ؞› ‏39 مَ نࣹ سَمَ سَنگّ:‏ ‹بُ ندَوْ سَ نَنگُوُلَ اࣵندّ اَڪْ مَنْ نَگْ؟› ‏40 مُ نࣹ مَ:‏ ‹اَجِ سَخْ، جِ مَ تࣷݒّ، دِنَ يٗنِّ مَلاَڪَامْ، مُ اࣵندّ اَڪْ يَوْ، بَ اࣵنْتَلَلْ لَ سَ يࣷونْ، نگَ جࣴلَلْ سَمَ دࣷومْ سࣷخْنَ ݖَ سَمَ ڪࣴرْ بَايْ، ݖَ سَمَ بِيرِ بࣷڪّ؞ ‏41 بࣷو دࣹمࣹي بَ ݖَ سَمَيْ بࣷڪّ، وࣵݖّ نگَ، ندَخْ بُ ݧُ لَ مَيُلْ سࣷخْنَ، وࣵݖّࣷو نگَ اَڪْ مَنْ؞› ‏42 بَ مَ اَگْسࣹي تࣹيْ جِي ݖَ سࣹيَانْ بَ، دَمَا وَخْ نࣹ:‏ ‹اَجِ سَخْ جِ، سَمَ يࣵلَّيْ سَنگّ إِبْرَاهِيمَ، ندࣹگَمْ نَرْ نگَا اࣵنْتَلْ يࣷونْ وِ مَيْ دࣷخْ، نگَلَّ مَيْ مَ لِي:‏ ‏43 مَا نگِي تَخَوْ ݖِ سࣹيَانْ بِ؛ جَنقّ بَيْ ݧࣴوْ رࣷوتْسِ، مَ نࣹ ڪࣷ:‏ «دࣷو مَ مَيْ تُوتِ ݖِ سَ نجَقْ لِ، مَ نَانْ،» ‏44 مُ نࣹ مَ:‏ «نَانَلْ بَ نࣷݒِّ، مَ وࣴگَّلْ لَ سَيْ گِلࣺيمْ،» ڪࣷوڪُ نَ دِ سࣷخْنَ، سِ اَجِ سَخْ جِ جَگْلࣹيلْ سَمَ دࣷومُ سَنگّ؞›
‏45 «بَلَ مَا ݧَانْ بَ نࣷݒِّ، رࣹبࣹڪَ نࣹ جِمࣹيتْ، گࣵدُّ نجَقَمْ، وࣵݖّ ݖَ سࣹيَانْ بَ، دِ رࣷوتْ؞ مَ نࣹ ڪࣷ:‏ ‹مَيْ مَ، مَ نَانْ؞› ‏46 مُ گَاوْ يࣹنِّڪُ نجَقْ لِ، دَلْدِ نࣹ مَ:‏ ‹نَانَلْ بَ نࣷݒِّ مَ وࣴگَّلْ لَ سَيْ گِلࣺيمْ؞› مَ نَانْ بَ نࣷݒِّ، مُ وࣴگَّلْ مَ گِلࣺيمْ يِ؞ ‏47 مَ لاَجْ ڪࣷ نࣹ ڪࣷ:‏ ‹ڪَنْ مࣷو لَ جُرْ؟› مُ نࣹ:‏ ‹بࣹتُوࣹلْ، دࣷومُ نَخࣷرْ اَڪْ مِلْڪَ؞›
«ݖَ لاَ ڪࣷ تَڪَّلْ جَارࣷ بِ ݖِ بَڪَّنَمْ، اَڪْ لَمْ يِ ݖِيْ لࣷخࣷومْ؞ ‏48 بَ مُ ڪࣷ دࣹفࣹي مَ سُجّٗوتَلْ اَجِ سَخْ جِ، سَنتّ اَجِ سَخْ جِيْ سَمَ يࣵلَّيْ سَنگّ إِبْرَاهِيمَ، ندَخْ جُبَلْ نَ سَمَ يࣷونْ، بَ مَ جࣴلَلْ دࣷومَمْ سࣷخْنَ ݖِيْ بࣷڪَّمْ؞ ‏49 لࣺيگِ نَگْ ندࣹگَمْ نَرْ نگࣹينَ رَفࣹتࣷو اَڪْ سَمَ سَنگّ تࣹيْ سࣵمّ ڪٗلّࣴرࣹ، خَمَلْ-لࣹينْ مَ ڪࣷ، بَ مُ لࣹيرْ؞ سُ نگࣹينْ ڪࣷ نَرُلْ اِتْ، وَخْلࣹينْ مَ، مَ خَمْ نُ مَيْ دࣹفْ؞»
‏50 لَبَنْ اَڪْ بࣹتُوࣹلْ نࣹ ڪࣷ:‏ «مبِرْ مࣷومُ دࣹ ݖِ اَجِ سَخْ جِ لَ جٗگࣹ، تࣹ اَمُنُ ݖِ دَرَ لُ نُيْ وَخْ؞ ‏51 رࣹبࣹڪَا نگࣷوگُ، جࣴلَلْ تࣹ دࣹمْ، مُيْ سَ سࣷخْنَسْ دࣷومُ سَنگّ، نِ لَ ڪࣷ اَجِ سَخْ جِ تࣹگْتَلࣹ؞» ‏52 نَڪَ لَ سُرْگَبْ إِبْرَاهِيمَ دࣺگّ يࣷويُ ڪࣵدُّ، دَلْدِ سُجّٗوتَلْ اَجِ سَخْ جِ؞
‏53 سُرْگَ بَ نَگْ گࣺنّࣹ اَيْ گࣵنجَرِ خَالِسْ اَڪْ يُ وُرُسْ بࣷولࣹيڪْ اَيْ يࣺرࣹ، جࣷخْ رࣹبࣹڪَ؞ ݖࣵمِّݧْ لَ اَڪْ يَايْ جَ اِتَمْ، مُ مَيْ لࣹينْ اَلَلْ جُ تَڪُّ؞ ‏54 بَ لࣷولُ وࣹيسࣹي مُ دࣷورَ لࣹڪّ، دَلْدِ نَانْ، مࣷومْ اَڪْ ݧِ مُ اࣵنْدَلْ؞ ݧُ فَنَانْ فَ ڪࣹرࣷوگْ؞
بَ ݧُ يࣹوّࣷو ݖَ اࣴلّࣴگْ سَ، سُرْگَ بَ نࣹ:‏ «بࣵيِّلࣹينْ مَ نَگْ، مَ دࣹلُّ ݖَ سَمَ سَنگّ؞» ‏55 ݖࣵمِّݧَلْ ندَوْ سَ اَڪْ يَايْ جَ دَلْدِ نࣹ ڪࣷ:‏ «خَنَا ڪَيْ جَنقّ بِ دِنَ تࣷوگَاتْ اَڪْ نُنْ اَبْ دِيرْ بࣷوگْ، مَانَامْ فِي اَڪْ سِبِّرْ، دࣷورَ دࣹمْ؞» ‏56 مُ نࣹ لࣹينْ:‏ «گَنَّاوْ اَجِ سَخْ جِ اࣵنْتَلْ نَ سَمَ يࣷونْ، تࣹي نگࣹينْ مَ بَݧَ يࣹيخَلْ، تࣹ بࣵيِّ مَ رࣹڪّ، مَ دࣹلُّ ݖَ سَمَ سَنگّ؟» ‏57 ݧُ نࣹ ڪࣷ:‏ «ڪࣷنْ نَگْ نَنُ وࣷو جَنقّ بِ، لاَجْ ڪࣷ مࣷومْ ݖِ بࣷݒَّمْ؞» ‏58 ݖِ ڪَوْ لࣷولُ ݧُ وࣷو رࣹبࣹڪَ، نࣹ ڪࣷ:‏ «ندَخْ دَنگَيْ اࣵندّ اَڪْ وَا جِ لࣺيگِ؟» مُ نࣹ:‏ «وَاوْ؞» ‏59 بَ لࣷولُ اَمࣹي ݖࣵمِّݧْ يَ بࣵيِّ سࣹينْ جِگࣺينْ رࣹبࣹڪَ، مࣷومْ اَڪْ سُرْگَ، بِ ڪࣷ يَرْ تࣹ دَانْ ڪࣷ نࣵمْݒَلْ، ݧُ اࣵندّ اَڪْ سُرْگَبْ إِبْرَاهِيمَ اَڪْ ݧِ ڪࣷ تࣷݒّ، ݧُ دࣹمْ؞ ‏60 ݧَ نگَيْ ݧَانَلْ رࣹبࣹڪَ، مُ بَرْڪࣹيلْ، تࣹ نَانْ:‏
«اࣺيْ جِگࣺينْ سُنُ!‏ گِيرَلْ، بَ سࣷسْ اَيْ جُنِّيْ جُنِّ،
تࣹ سَ اَسْڪَنْ تࣹگْ اَيْ بَݧَمْ تࣵنڪّ؞»
‏61 گَنَّاوْ لࣷولُ رࣹبࣹڪَ اࣵندّ اَڪْ خَلࣹ يُ جِگࣺينْ، يِيْ جَامَمْ، ݧُ وَرْ اَيْ گِلࣺيمْ، تࣷݒّ وَا جَ؞ سُرْگَ بَ دَلْدِ دࣹمْ، يٗبُّ رࣹبࣹڪَ؞
‏62 لࣷولُ فࣹڪّ نَ إِسْحَاقَ دࣴڪّ ݖَ اࣵلُّبْ نࣹگࣹوْ؞ مُ دࣹمࣷونْ نَگْ تࣹينُ لَخَيْ رࣷيْ، بَ ݧِبّسِ؞ ‏63 گࣹنّ نگࣷونْ إِسْحَاقَ گࣺنّ، دِ دࣷخَنْتُ ݖَ اࣵلّ بَ؞ نَڪَ لَ نࣹ سِگّࣹتْ خࣷولْ، سࣺينْ گِلࣺيمْ يُيْ ݧࣴوْ؞ ‏64 رࣹبࣹڪَ اِتْ نࣹ سِگّࣹتْ، گِسْ إِسْحَاقَ؞ مُ نࣹ ݖࣴݒّ نَگْ، وࣵݖّ گِلࣺيمْ گَ، ‏65 دَلْدِ نࣹ سُرْگَ بَ:‏ «وَا جࣹي ݖِ اࣵلّ بِيْ ݧࣴوْ، جࣴمْ ݖِ نُنْ، ڪَنَ؟» سُرْگَ بَ نࣹ ڪࣷ:‏ «سَمَ سَنگّ لَ دࣹ؞» رࣹبࣹڪَ دَلْدِ جࣴلْ مُورَايْ بَ، مُورࣷو؞ ‏66 سُرْگَ بَ نَگْ نࣹتَّلِ إِسْحَاقَ لِ مُ دࣹفْ لࣺݒّ؞ ‏67 بَ لࣷولُ وࣺيࣹي إِسْحَاقَ يٗبُّ رࣹبࣹڪَ ݖَ خَيْمَبْ يَايَمْ سَارَتَ، مُيْ سࣷخْنَامْ، دِ ڪُ مُ بࣴگّ، تࣹ دِ ڪࣷ مُݧࣹ يَايَمْ، ندَخْ فࣹڪّ نَ مُ فَاتُ؞
