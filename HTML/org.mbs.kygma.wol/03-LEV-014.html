لِ وَرْ ݖِ لاَبَلْ بࣷرࣷومْ جࣵنگࣷرࣷيْ دࣹرْ
14
‏1 اَجِ سَخْ جِ تࣹگْ نَ ݖَ نࣹ مُوسَىٰ:‏ ‏2 «لِ يࣷونْ دِگْلࣹ ݖِ مبِرُمْ نِتْ ڪُ اَمࣹ جࣵنگࣷرࣷيْ دࣹرْ جُيْ لَوْ، ڪࣹرࣷوگْ بࣺسْ بَ ݧُيْ لاَبَلْ نِتْ ڪِ، مࣷويْ لِي:‏ نَݧُ ڪࣷ يࣹگّࣹ سَرْخَلْڪَتْ بِ؞ ‏3 سَرْخَلْڪَتْ بِ دَيْ گࣺنّ دَلْ بِ، بُ ڪࣷيْ سࣹيتْ؞ سُ نِتْ ڪِ دࣹمࣹي بَ جࣵنگࣷرࣷيْ دࣹرْ جِ ڪࣷ دَلࣷونْ وࣺرْ، ‏4 سَرْخَلْڪَتْ بِ دَيْ سَنْتَانࣹ، ݧُ اِنْدِلْ نِتْ ڪࣷوڪُ ݧُيْ لاَبَلْ ݧَارِ ݒِݖّ يُ دَگَنْ تࣹيْ دُندّ، اَڪْ بَنْتُ گَرَبُ سࣹيدَرْ، اَڪْ وࣴݧْ گُ خࣷنقّ ݖُرّ، اَڪْ ݖَرُ گَرَبُ اِسࣷبْ¤؞ ‏5 نَ سَرْخَلْڪَتْ بِ سَنْتَانࣹ، ݧُ رࣹنْدِ مࣹنّ ݒِݖّ مِ، تِيمَلࣹ ڪࣷ ندَبَلْ خَنْدࣹيرْ لُ دࣹفْ ندࣷخْ مُ بَلّࣹ ݖِ بࣴتُ ندࣷخْ؞ ‏6 ݒِݖّ مِيْ دُندّ دَ ڪࣷيْ بࣷولࣹيڪْ بَنْتُ سࣹيدَرْ بِ اَڪْ وࣴݧْ گُ خࣷنقّ ݖُرّ گِ اَڪْ ݖَرُ اِسࣷبْ بِ، بࣷولࣹ لࣺݒّ سٗوبْ ݖِ دࣹرࣹتُ ݒِݖّ مَ ݧُ رࣹنْدِ وࣷونْ تِيمَلࣹ ڪࣷ ندࣷخْ مِ بَلّࣹ ݖِ بࣴتُ ندࣷخْ؞ ‏7 بُ لࣷولُ اَمࣹي سَرْخَلْڪَتْ بِ وِسْ-وِسَلْ جُرٗومْ ݧَارِ يࣷونْ ݖِ ڪَوْ ڪِ ݧُيْ لاَبَلْ، نگِرْ مُ تࣵگّࣷوڪْ سࣷبࣹيْ جࣵنگࣷرࣷ جِ؞ سُ ڪࣷ دࣹفࣹي سَرْخَلْڪَتْ بِ بِرَلْ نࣹ سࣹتْ نَ؞ بُ نࣷݒّࣹي نَ بࣵيِّ ݒِݖّ مِيْ دُندّ مُ نَاوْ ݖَ اࣵلّ بَ؞ ‏8 ڪُ ݧُيْ لاَبَلْ دَيْ فٗوتْ اَيْ يࣺرࣹيمْ، وَتُ بَ سࣹتْ تࣹ سَنگُ، دࣷورَ سࣹتْ؞ گَنَّاوْ لࣷولُ مَنْ نَا دُگّ ݖِ دَلْ بِ، وَايࣹ نَ يࣹمْ ݖَ بُنتّ خَيْمَامْ ݖَ بِتِ دِيرُ جُرٗومْ ݧَارِ فَنْ؞ ‏9 بࣺسُبْ جُرٗومْ ݧَارࣹيلْ بَ نَ وَتُ بَ سࣹتْ، بࣷݒّ بࣹيڪْ سِڪِّمْ بࣹيڪْ يࣹينْ يِ اَڪْ ڪَوَرْ گِ ݖِ دࣹسْ يࣺݒّ لَيْ وَتْ، فٗوتْ اَيْ يࣺرࣹيمْ، سَنگُ، دَلْدِ سࣹتْ؞ ‏10 بࣺسُبْ جُرٗومْ ݧࣹتّࣹيلْ بَ نَ اِنْدِ ݧَارِ ڪُويْ يُ اَمُلْ سِڪّ، اَڪْ مࣹنّ خَرْ مُ جِگࣺينْ مُ اَمْ اَتْ تࣹ اَمُلْ سِڪّ اَڪْ جُرٗومْ ݧࣹينْتِ ڪِلࣷيْ سُنگُفْ سُ مُݖّ اَيِبْ سُ ݧُ خِيوࣹ دِوْ، مُيْ سَرَخُ ݒࣹݒّ؞ مُ اِنْدَالࣹ گࣹنّ-وࣵلُّ لِيتَرُ دِوْ؞ ‏11 سَرْخَلْڪَتْ بِيْ لاَبَلْ نِتْ ڪِ دَيْ اِنْدِ نِتْ ڪِ، مُ اࣵندّ اَڪْ لࣷولُ لࣺݒّ، تَخَوْ فِ ڪَنَمْ اَجِ سَخْ جِ، ݖِ بُنتّ خَيْمَبْ ندَجࣹ مِ؞
‏12 «سَرْخَلْڪَتْ بِ تࣹگْ ݖَ جࣴلْ مࣹنّ ڪُويْ مِ، رࣹنْدِ ڪࣷ، مُيْ سَرَخُ ݒࣹيُگْ تࣷوݧْ، بࣷولࣹ ڪࣷوڪْ گࣹنّ-وࣵلّ لِيتَرُ دِوْ گَ، دࣹفْ ڪࣷ سَرَخْ بُ ݧُيْ يࣺڪَّتِ، جࣺبَّلْ اَجِ سَخْ جِ؞ ‏13 نَݧُ رࣹنْدِ ڪُويْ مِ ݖِ بࣺرَبْ بُ سࣹلّ بِ، فِ ݧُيْ رࣹنْدࣹي جُرُ سَرَخُ ݒٗوتُمْ بࣵڪَّارْ اَڪْ سَرَخُ رࣹنْدِ-دٗومَلْ، ندَخْ لِ نࣹڪّ ݖِ سَرَخُ ݒࣹيُگْ تࣷوݧْ مࣷو نࣹڪّ ݖِ سَرَخُ ݒٗوتُمْ بࣵڪَّارْ:‏ سَرْخَلْڪَتْ بࣹي ڪࣷ مࣷومْ؛ لُ سࣹلَّ سࣹلّ لَ؞ ‏14 نَ سَرْخَلْڪَتْ بِ سࣵڪّ ݖِ دࣹرࣹتُ جُرْ گِيْ سَرَخُ ݒࣹيُگْ تࣷوݧْ، تَقَلْ ڪࣷ ݖِ تَبَنُ نࣷݒُّ ندِجࣷورُ نِتْ ڪِ ݧُيْ لاَبَلْ اَڪْ بَارَامُ دࣺيُ ندِجࣷورَمْ اَڪْ بَارَامُ دࣺيُ تࣵنڪُ ندِجࣷورَمْ؞ ‏15 بُ سَرْخَلْڪَتْ بِ نࣷݒّࣹي، نَ سࣵڪّ ݖِ گࣹنّ-وࣵلّ لِيتَرُ دِوْ گِ، سࣷتِّ ڪࣷ ݖِ بِيرْ لࣷخࣷلْ ݖࣵمّࣷݧُ بࣷݒَّمْ؞ ‏16 نَ سَرْخَلْڪَتْ بِ ݖَݒّ بَارَامُ جࣷخࣷݧُ ندِجࣷورَمْ ݖِ دِوْ گِ ݖِ لࣷخࣷلْ ݖࣵمّࣷݧَمْ، دَلْدِ وِسْ-وِسَلࣹ دِوْ گِ بَارَامَمْ، بَ مُ مَتْ جُرٗومْ ݧَارِ يࣷونْ فِ ڪَنَمْ اَجِ سَخْ جِ؞ ‏17 نَ سَرْخَلْڪَتْ بِ سࣵڪّ ݖِ دِوْ گِ دࣹسْ ݖِ لࣷخࣷومْ، تࣷويَلْ ݖِ تَبَنُ نࣷݒُّ ندِجࣷورُ نِتْ ڪِ ݧُيْ لاَبَلْ اَڪْ بَارَامُ دࣺيُ ندِجࣷورَمْ اَڪْ بَارَامُ دࣺيُ تࣵنڪُ ندِجࣷورَمْ؛ نَݧُ ڪࣷ لاَلَلْ ݖِ دِوْ گِ فَ ݧُ ڪࣷ جࣴڪّࣷونَ تَقَلْ ݖِ دࣹرࣹتُ سَرَخْ سِيْ ݒࣹيُگْ تࣷوݧْ؞ ‏18 لِ دࣹسْ ݖِ دِوْ گِ ݖِ لࣷخࣷلْ سَرْخَلْڪَتْ بِ، نَ ڪࣷ دِوْ ݖِ بࣷݒُّ نِتْ ڪِ ݧُيْ لاَبَلْ؞ نࣷونُ لَ ڪࣷ سَرْخَلْڪَتْ بِ دࣹفَلࣹ نجࣷتْلاَيَمْ فِ ڪَنَمْ اَجِ سَخْ جِ؞ بَ لࣷولُ اَمࣹي سَرْخَلْڪَتْ بِ رࣹنْدِ سَرَخْ بِيْ ݒٗوتُمْ بࣵڪَّارْ، ‏19 دࣹفَلْ ڪࣷ ڪِ ݧُيْ لاَبَلْ نجࣷتْلاَيَمْ نگِرْ مُ تࣵگّࣷوڪْ سࣷبࣹيمْ؞ تࣹ اِتَمْ نَ سَرْخَلْڪَتْ بِ رࣹنْدِ سَرَخُ دٗومَلْ بِ، ‏20 بࣷولࣹ ڪࣷوڪْ سَرَخُ ݒࣹݒّ مِ، جࣷخࣹ لࣺݒّ ݖِ ڪَوْ سَرْخَلُڪَايْ بِ، دࣹفَلْ ڪࣷ نِتْ ڪِ نجࣷتْلاَيَمْ، مُ سࣹتْ؞
‏21 «سُ فࣹڪّࣹي نࣹ نِتْ ڪِ دَفَ ݧࣵڪّ بَ دࣷولࣹيمْ مَتࣹوُ ڪࣷ، نَ وُتْ مࣹنّ ڪُويْ، مُيْ سَرَخُ ݒࣹيُگْ تࣷوݧْ، دِ سَرَخُ يࣺڪَّتِ-جࣺبَّلࣹ، نگِرْ نجࣷتْلاَيَمْ، بࣷولࣹ ڪࣷوڪْ ݧࣹتِّ ڪِلࣷيْ سُنگُفْ سُ مُݖّ اَيِبْ، خِيوࣹ ڪࣷ دِوْ، مُيْ سَرَخُ ݒࣹݒّ؞ نَ اِنْدَالࣹ گࣹنّ-وࣵلّ لِيتَرُ دِوْ، ‏22 اَڪْ ݧَارِ ݒِتَخْ مبَا ݧَارِ خَتِ يُ ندَوْ، لُ مُ ݖِ گࣴنَ جࣹڪُّ رࣹڪّ؞ مࣹنّ مِ سَرَخُ ݒٗوتُمْ بࣵڪَّارْ لَ، مِ ݖِ دࣹسْ دِ سَرَخُ رࣹنْدِ-دٗومَلْ؞ ‏23 دَ ڪࣷيْ اِنْدِلْ سَرْخَلْڪَتْ بِ، نگِرْ مُ لاَبَلْ ڪࣷ بُ بࣺسْ بَ دࣺلْسࣹي، ݖَ بُنتّ خَيْمَبْ ندَجࣹ مَ، فِ ڪَنَمْ اَجِ سَخْ جِ؞ ‏24 سَرْخَلْڪَتْ بِ دَيْ جࣴلْ ڪُويُ سَرَخْ سِيْ ݒࣹيُگْ تࣷوݧْ اَڪْ گࣹنّ-وࣵلّ لِيتَرُ دِوْ گِ، بࣷولࣹ ڪࣷ، مُيْ سَرَخْ بُ ݧُيْ يࣺڪَّتِ، جࣺبَّلْ اَجِ سَخْ جِ؞ ‏25 گَنَّاوْ گِ ݧُ رࣹنْدِ ڪُويْ مِيْ سَرَخُ ݒࣹيُگْ تࣷوݧْ، سَرْخَلْڪَتْ بِ سࣵڪّ ݖِ دࣹرࣹتْ جِ، تَقَلْ ݖِ تَبَنُ نࣷݒُّ ندِجࣷورُ نِتْ ڪِ ݧُيْ لاَبَلْ اَڪْ بَارَامُ دࣺيُ ندِجࣷورَمْ اَڪْ بَارَامُ دࣺيُ تࣵنڪُ ندِجࣷورَمْ؞ ‏26 بُ سَرْخَلْڪَتْ بِ نࣷݒّࣹي، نَ سࣷتِّ تُوتِ ݖِ دِوْ گِ ݖِ لࣷخࣷلْ ݖࣵمّࣷݧُ بࣷݒَّمْ، ‏27 بَارَامُ جࣷخࣷݧُ ندِجࣷورَمْ لَيْ وِسْ-وِسَلࣹ دِوْ گِ ݖِ لࣷخࣷلْ ݖࣵمّࣷݧَمْ بَ مُيْ جُرٗومْ ݧَارِ يࣷونْ فِ ڪَنَمْ اَجِ سَخْ جِ؞ ‏28 لࣷولُ وࣺيْ سَرْخَلْڪَتْ بِ وَرَ سࣵڪّ ݖِ دِوْ گِ ݖِ لࣷخࣷومْ، تࣷويَلْ ݖَ تَبَنُ نࣷݒُّ ندِجࣷورُ ڪِ ݧُيْ لاَبَلْ اَڪْ بَارَامُ دࣺيُ ندِجࣷورَمْ اَڪْ بَارَامُ دࣺيُ تࣵنڪُ ندِجࣷورَمْ؛ نَ ڪࣷ دࣹفْ ݖَ بࣺرَبْ بࣷوبُ مُ تَاجْ دࣹرࣹتُ سَرَخْ سِيْ ݒࣹيُگْ تࣷوݧْ؞ ‏29 لِ دࣹسْ ݖِ دِوْ گِ ݖِ لࣷخࣷلْ سَرْخَلْڪَتْ بِ نَ ڪࣷ دِوْ ݖِ بࣷݒُّ ڪِ ݧُيْ لاَبَلْ، دࣹفَلْ ڪࣷ ڪࣷ نجࣷتْلاَيْ فِ ڪَنَمْ اَجِ سَخْ جِ؞ ‏30 بُ نࣷݒّࣹي نَ رࣹنْدِ مࣹنّ ݒِتَخْ مِ مبَا مࣹنّ خَتِ مُ ندَوْ مِ، لُ مُ ݖِ گࣴنَ جࣹڪُّ رࣹڪّ، ‏31 مࣹنّ مِ دِ سَرَخَسْ ݒࣹيُگْ تࣷوݧْ، مِ ݖِ دࣹسْ دِ سَرَخُ رࣹنْدِ-دٗومَلْ، تࣹ مُ بࣷولࣹ ڪࣷوڪْ سَرَخُ ݒࣹݒّ مِ؞ نࣷونُ لَ سَرْخَلْڪَتْ بِيْ دࣹفَلࣹ ڪِ ݧُيْ لاَبَلْ نجࣷتْلاَيَمْ فِ ڪَنَمْ اَجِ سَخْ جِ؞»
‏32 لࣷولُ لَ يࣷونْ دِگْلࣹ ݖِ مبِرُمْ نِتْ ڪُ اࣵندّ اَڪْ جࣵنگࣷرࣷيْ دࣹرْ جُيْ لَوْ تࣹ دࣷولࣹيمْ مَتࣹوُلْ سَرَخْ يِ ݧُ ڪࣷيْ لاَبَلࣹ؞
لِ وَرْ ݖِ نࣺيگْ بُ تࣹگْ لِيرْ
‏33 اَجِ سَخْ جِ وَخْ نَ مُوسَىٰ اَڪْ هَـٰرُونَ نࣹ لࣹينْ:‏ ‏34 «بُ نگࣹينْ دࣹمࣹي ݖَ رࣺيوُ ڪَنَانْ گَ مَ لࣹينْ جࣷخْ مُيْ سࣹينْ مࣷومࣹيلْ، بُ مَ تࣹگࣹي لِيرْ وُيْ لَوْ بَ خُورَلْ اَبْ نࣺيگْ فࣷوفَ ݖَ سࣹينْ رࣺيوُمْ بࣷݒّ، ‏35 بࣷرࣷومْ نࣺيگْ بِ دَفَ وَرَ دࣹمْ ݖَ سَرْخَلْڪَتْ بَ نࣹ ڪࣷ:‏ ‹لُ مࣹلْ نِ اَوْ لِيرْ لاَ گِسْ سَمَ نࣺيگْ؞› ‏36 سُ بࣷوبَا نَ سَرْخَلْڪَتْ بِ سَنْتَانࣹ، ݧُ گࣺنّࣹ لࣺݒّ لُ نࣹڪّ ݖَ نࣺيگْ بَ، بَلَ مُيْ دُگّ دِ سࣹيتْ لِيرْ وِ؞ سُ ڪࣷ دࣹفࣹي دُ اَمْ لࣹنّ لُ نࣹڪّ ݖَ نࣺيگْ بَ، لُ مُيْ بِرَلْ نࣹ سࣷبࣹ تࣷݒّ نَ ڪࣷ؞ گَنَّاوْ لࣷولُ سَرْخَلْڪَتْ بِ دُگّ، سࣹيتْ نࣺيگْ بِ؞ ‏37 بُ سَرْخَلْڪَتْ بِ سࣹيتࣹي لِيرْ وِ ݖِ مِيرِ نࣺيگْ بِ ݖَ بِيرْ، مُيْ اَيْ خٗوتْ-خٗوتْ يُ خَوَ نࣴتࣴخْ، مبَا مُ خَوَ خࣷنقّ، مࣹلْ نِ لُ دࣹفْ ݒَخْ ݖِ مِيرْ بِ، ‏38 نَ گࣺنّ نࣺيگْ بِ تࣹ تࣴجْ نࣺيگْ بِ دِيرُ جُرٗومْ ݧَارِ فَنْ؞ ‏39 بࣺسُبْ جُرٗومْ ݧَارࣹيلْ بَ نَ سَرْخَلْڪَتْ بِ دࣺلْسِ سࣹيتْ نࣺيگْ بِ؞ بُ لِيرْ وِ لَوࣹي ݖِ مِيرِ نࣺيگْ بِ، ‏40 نَ سَرْخَلْڪَتْ بِ سَنْتَانࣹ، ݧُ تࣹگِّ دࣷجْ يِ لِيرْ وِ خُورَلْ، بࣷولࣹ ڪࣷ گࣺنّࣹ دࣴڪّ بِ، سࣵنِّ ڪࣷ فُ سࣹتُلْ؞ ‏41 نَ خࣹيتْلُ بِيرْ نࣺيگْ بِ بَ مُ دَجْ، تࣹ ݒࣴندّ بَ ݧُ ݖَ خࣹيتࣹ دࣹيسْ ڪࣷيْ گࣺنّࣹ دࣴڪّ بَ، تُورْ ڪࣷ فُ سࣹتُلْ؞ ‏42 دَݧࣷو وَرَ سࣵڪَّاتْ اَيْ دࣷجْ، وُوتَلࣹ ڪࣷ دࣷجْ يࣷويُ تࣹ سࣵڪّ بَنْ بُ بࣹيسْ، رَاخࣹ ڪࣷ نࣺيگْ بِ؞
‏43 «سُ لِيرْ وِ دࣹلّࣷو تࣹگُ ݖِ بِيرْ نࣺيگْ بِ گَنَّاوْ بُ ݧُ تࣹگّࣹي دࣷجْ يِ، خࣹيتْ بِيرْ نࣺيگْ بِ، بَ رَاخْ ڪࣷ، ‏44 سَرْخَلْڪَتْ بِ دَيْ دِڪّ سࣹيتْ ڪࣷ؞ سُ لِيرْ وِ لَوࣹي ݖِ نࣺيگْ بِ، لِيرْ وُ آيْتَلْ لَ وُ خُورَلْ بِيرْ نࣺيگْ بِ، دِ سࣷبࣹ؞ ‏45 نࣺيگْ بَ دࣹيسْ ڪࣷيْ دَانࣹ، بࣷولࣹ دࣷجْ يَاڪْ بَنتّ يَاڪْ بَنُ رَاخْ بَ بࣺݒّ، گࣺنّࣹ ڪࣷ دࣴڪّ بِ، تُورْ ڪࣷ فُ سࣹتُلْ؞
‏46 «ڪُ دُگّ بِيرْ نࣺيگْ بِ دِيرُ فَنْ يِ مُ تࣴجࣹ دِنَ يࣹنْدࣷو سࣷبࣹ بَ جَنتّ سࣷ؞ ‏47 ڪُ تࣴدّ ݖَ بِيرْ نࣺيگْ بَ وَرْ نَا فٗوتْ اَيْ يࣺرࣹيمْ تࣹ ڪُ فَ لࣹڪّࣹ اِتْ وَرْ نَا فٗوتْ اَيْ يࣺرࣹيمْ؞
‏48 «سُ سَرْخَلْڪَتْ بَ دࣹلّࣷو ݖَ نࣺيگْ بَ، سࣹيتْ ڪࣷ، فࣹڪّ لِيرْ وَ لَوُلْ ݖَ نࣺيگْ بَ، گَنَّاوْ بَ ݧُ ڪࣷ رَاخࣹي، نَ بِرَلْ نࣹ نࣺيگْ بِ سࣹتْ نَ، ندَخْ ڪࣷنْ لِيرْ وَ رَافْ نَ؞ ‏49 لِيْ لاَبَلْ نࣺيگْ بِ مࣷو دِ مُ سࣵڪّ ݧَارِ ݒِݖّ اَڪْ بَنْتُ گَرَبُ سࣹيدَرْ اَڪْ وࣴݧْ گُ خࣷنقّ ݖُرّ اَڪْ ݖَرُ اِسࣷبْ¤؞ ‏50 دَيْ رࣹنْدِ مࣹنّ ݒِݖّ مِ، تِيمَلࣹ ڪࣷ ندَبَلْ خَنْدࣹيرْ مُ دࣹفْ ندࣷخْ مُ بَلّࣹ ݖِ بࣴتُ ندࣷخْ؞ ‏51 بُ نࣷݒّࣹي نَ جَخَسࣹ دࣹرࣹتُ ݒِݖّ مِ ݧُ رࣹنْدِ اَڪْ ندࣷخْ مُ بَلّࣹ ݖِ بࣴتُ ندࣷخْ، دَلْدِ جࣴلْ بَنْتُ سࣹيدَرْ بِ اَڪْ ݖَرُ اِسࣷبْ بِ اَڪْ وࣴݧْ گِ خࣷنقّ ݖُرّ اَڪْ ݒِݖّ مِيْ دُندّ، بࣷولࣹ لࣺݒّ سٗوبْ ݖَ؞ سُ ڪࣷ دࣹفࣹي مُ وِسْ-وِسَلْ ݖَ جُرٗومْ ݧَارِ يࣷونْ ݖِ نࣺيگْ بِ؞ ‏52 ڪࣷنْ لِ مُيْ لاَبَلࣹ نࣺيگْ بِ مࣷو دِ دࣹرࣹتُ ݒِݖّ مِ اَڪْ ندࣷخْ مࣷومُ اَڪْ ݒِݖّ مِيْ دُندّ اَڪْ بَنْتُ سࣹيدَرْ بِ اَڪْ ݖَرُ اِسࣷبْ بِ اَڪْ وࣴݧْ گِ خࣷنقّ ݖُرّ؞ ‏53 نَ بࣵيِّ ݒِݖّ مِيْ دُندّ فُ گࣺنّ دࣴڪّ بَ، مُ نَاوْ ݖَ اࣵلّ بَ؞ نࣷونُ لَيْ دࣹفࣹ نجࣷتْلاَيَلْ نࣺيگْ بِ، مُ سࣹتْ؞»
‏54 لࣷولُ لَ يࣷونْ دِگْلࣹ ݖِ لࣺݒّ لُيْ جࣵنگࣷرࣷيْ دࣹرْ جُيْ لَوْ اَڪْ وَتْ-وَتَانْ، ‏55 مبَا لِيرْ وُ تࣹگُ ݖِ ندِمࣷ مبَا ݖِ نࣺيگْ، ‏56 اَڪْ ݖِ وࣵلُّ تࣴرْگࣴنْ اِتَمْ مبَا مُ دِيْ اࣴرْ مبَا گࣵڪّ بُ وࣹيخْ؛ ‏57 مࣷويْ تࣹگْتَلْ بِيْ خَمْلࣹ لُ سࣹتْ اَڪْ لُ سࣹتُلْ؞ لࣷولُ لَ يࣷونْ دِگْلࣹ ݖِ جࣵنگࣷرࣷيْ دࣹرْ جُيْ لَوْ مبَا لِيرْ؞
