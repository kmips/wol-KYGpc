اَسْڪَنُ آدَمَ، بَ ݖِ نُوحٍ
(سَارْ 5‏.1—6‏.8)
5
‏1 نِ تࣺيرࣹ اِنْدࣹي اَسْڪَنُ آدَمَ نِي لَ؞ بَ يࣵلَّ دِ سࣵڪّ نِتْ، دَ ڪࣷو بِندّ ݖِ مࣹلࣷڪَانَمْ، ‏2 گٗورْ اَڪْ جِگࣺينْ لَ سࣵڪّ، بَرْڪࣹيلْ لࣹينْ، تُدّࣹ لࣹينْ نِتْ، ڪࣹرࣷوگْ بَ مُ لࣹينْ سࣵڪّࣹي؞
‏3 آدَمَ دُندّ تࣺيمࣺيرِ اَتْ اَڪْ فَنْوࣹيرْ (130)، جُرْ دࣷومْ جُ گٗورْ، مُ بِنْدࣷو مࣹلࣷڪَانَمْ تࣹ نِرࣷوڪْ مࣷومْ؞ مُ تُدّࣹ ڪࣷ سࣹتْ؞ ‏4 سࣹتْ جُدُّ نَ، آدَمَ دُنْدَاتْ جُرٗومْ ݧࣹتِّ تࣺيمࣺيرِ اَتْ (800)، اَمَاتْ اَيْ دࣷومْ يُ گٗورْ اَڪْ يُ جِگࣺينْ؞ ‏5 ڪࣷنْ آدَمَ دُندّ نَ جُرٗومْ ݧࣹينْتِ تࣺيمࣺيرِ اَتْ اَڪْ فَنْوࣹيرْ (930)، دࣷورَ فَاتُ؞
‏6 سࣹتْ مࣷومࣹي اَمْ نَ تࣺيمࣺيرِ اَتْ اَڪْ جُرٗومْ (105)، دࣷورَ جُرْ اࣹنࣷسْ؞ ‏7 اࣹنࣷسْ جُدُّ نَ، سࣹتْ دُنْدَاتْ جُرٗومْ ݧࣹتِّ تࣺيمࣺيرِ اَتْ اَڪْ جُرٗومْ ݧَارْ (807)، اَمَاتْ دࣷومْ يُ گٗورْ اَڪْ يُ جِگࣺينْ؞ ‏8 ڪࣷنْ سࣹتْ دُندّ نَ جُرٗومْ ݧࣹينْتِ تࣺيمࣺيرِ اَتْ اَڪْ فُڪّ اَڪْ ݧَارْ (912)، دࣷورَ فَاتُ؞
‏9 اࣹنࣷسْ نَگْ اَمْ نَ جُرٗومْ ݧࣹينتّ فُڪِّ اَتْ (90)، دࣷورَ جُرْ ڪࣹنَنْ؞ ‏10 ڪࣹنَنْ جُدُّ نَ، اࣹنࣷسْ دُنْدَاتْ جُرٗومْ ݧࣹتِّ تࣺيمࣺيرِ اَتْ اَڪْ فُڪّ اَڪْ جُرٗومْ (815)، اَمَاتْ دࣷومْ يُ گٗورْ اَڪْ يُ جِگࣺينْ؞ ‏11 ڪࣷنْ اࣹنࣷسْ دُندّ نَ جُرٗومْ ݧࣹينْتِ تࣺيمࣺيرِ اَتْ اَڪْ جُرٗومْ (905)، دࣷورَ فَاتُ؞
‏12 ڪࣹنَنْ مࣷومࣹي اَمْ نَ جُرٗومْ ݧَارْ فُڪِّ اَتْ (70)، دࣷورَ جُرْ مَلَلࣹيلْ؞ ‏13 مَلَلࣹيلْ جُدُّ نَ، ڪࣹنَنْ دُنْدَاتْ جُرٗومْ ݧࣹتِّ تࣺيمࣺيرِ اَتْ اَڪْ ݧࣹينتّ فُڪّ (840)، اَمَاتْ اَيْ دࣷومْ يُ گٗورْ اَڪْ يُ جِگࣺينْ؞ ‏14 ڪࣷنْ ڪࣹنَنْ دُندّ نَ جُرٗومْ ݧࣹينْتِ تࣺيمࣺيرِ اَتْ اَڪْ فُڪّ (910)، دࣷورَ فَاتُ؞
‏15 مَلَلࣹيلْ نَگْ اَمْ نَ جُرٗومْ بࣹنّ فُڪِّ اَتْ اَڪْ جُرٗومْ (65)، دࣷورَ جُرْ يࣹرࣹدّ؞ ‏16 يࣹرࣹدّ جُدُّ نَ، مَلَلࣹيلْ دُنْدَاتْ جُرٗومْ ݧࣹتِّ تࣺيمࣺيرِ اَتْ اَڪْ فَنْوࣹيرْ (830)، اَمَاتْ اَيْ دࣷومْ يُ گٗورْ اَڪْ يُ جِگࣺينْ؞ ‏17 ڪࣷنْ مَلَلࣹيلْ دُندّ نَ جُرٗومْ ݧࣹتِّ تࣺيمࣺيرِ اَتْ اَڪْ جُرٗومْ ݧࣹينتّ فُڪّ اَڪْ جُرٗومْ (895)، دࣷورَ فَاتُ؞
‏18 يࣹرࣹدّ مࣷومࣹي اَمْ نَ تࣺيمࣺيرِ اَتْ اَڪْ جُرٗومْ بࣹنّ فُڪّ اَڪْ ݧَارْ (162)، دࣷورَ جُرْ اࣹنࣷگْ؞ ‏19 اࣹنࣷگْ جُدُّ نَ، يࣹرࣹدّ دُنْدَاتْ جُرٗومْ ݧࣹتِّ تࣺيمࣺيرِ اَتْ (800)، اَمَاتْ اَيْ دࣷومْ يُ گٗورْ اَڪْ يُ جِگࣺينْ؞ ‏20 ڪࣷنْ يࣹرࣹدّ دُندّ نَ جُرٗومْ ݧࣹينْتِ تࣺيمࣺيرِ اَتْ اَڪْ جُرٗومْ بࣹنّ فُڪّ اَڪْ ݧَارْ (962)، دࣷورَ فَاتُ؞
‏21 اࣹنࣷگْ مࣷومࣹي، اَمْ نَ جُرٗومْ بࣹنّ فُڪِّ اَتْ اَڪْ جُرٗومْ (65)، دࣷورَ جُرْ مَتُسَلࣹمْ؞ ‏22 مَتُسَلࣹمْ جُدُّ نَ، بَ نࣷݒِّ اࣹنࣷگْ تࣷݒّ يࣵلَّ ݧࣹتِّ تࣺيمࣺيرِ اَتْ (300)، اَمَاتْ اَيْ دࣷومْ يُ گٗورْ اَڪْ يُ جِگࣺينْ؞ ‏23 ڪࣷنْ اࣹنࣷگْ دُندّ نَ ݧࣹتِّ تࣺيمࣺيرِ اَتْ اَڪْ جُرٗومْ بࣹنّ فُڪّ اَڪْ جُرٗومْ (365)؞ ‏24 اࣹنࣷگْ دَفَ تࣷݒّ يࣵلَّ، بَ جࣹڪِّ-جࣹڪِّ گِسࣹيسَتُ ڪࣷ، ندَخْتࣹ يࣵلاَّ ڪࣷ يࣺيگࣹ فَ مࣷومْ؞
‏25 مَتُسَلࣹمْ مࣷومࣹي اَمْ نَ تࣺيمࣺيرِ اَتْ اَڪْ جُرٗومْ ݧࣹتّ فُڪّ اَڪْ جُرٗومْ ݧَارْ (187)، دࣷورَ جُرْ لࣹمࣹگْ؞ ‏26 لࣹمࣹگْ جُدُّ نَ، مَتُسَلࣹمْ دُنْدَاتْ جُرٗومْ ݧَارْ تࣺيمࣺيرِ اَتْ اَڪْ جُرٗومْ ݧࣹتّ فُڪّ اَڪْ ݧَارْ (782)، اَمَاتْ اَيْ دࣷومْ يُ گٗورْ اَڪْ يُ جِگࣺينْ؞ ‏27 ڪࣷنْ مَتُسَلࣹمْ دُندّ نَ جُرٗومْ ݧࣹينْتِ تࣺيمࣺيرِ اَتْ اَڪْ جُرٗومْ بࣹنّ فُڪّ اَڪْ جُرٗومْ ݧࣹينتّ (969)، دࣷورَ فَاتُ؞
‏28 لࣹمࣹگْ نَگْ اَمْ نَ تࣺيمࣺيرِ اَتْ اَڪْ جُرٗومْ ݧࣹتّ فُڪّ اَڪْ ݧَارْ (182)، دࣷورَ جُرْ دࣷومْ جُ گٗورْ؞ ‏29 مُ تُدّࣹ ڪࣷ نُوحٍ (مُ فِرِ نࣷفْلاَيْ) ندَخْ دَفَ نࣹ:‏ «ڪِي مࣷو نُيْ تَخَ نࣷيِّوَاتْ، نࣷݒَّلْ نُ ݖِ لِگّࣺييْ يُ مࣹتِّ اَڪْ ݖࣷونࣷيْ مبࣹيُ سُوفْ، سِ اَجِ سَخْ جِ رࣴبّ؞» ‏30 نُوحٍ جُدُّ نَ، لࣹمࣹگْ دُنْدَاتْ جُرٗومِ تࣺيمࣺيرِ اَتْ اَڪْ جُرٗومْ ݧࣹينتّ فُڪّ اَڪْ جُرٗومْ (595)، اَمَاتْ اَيْ دࣷومْ يُ گٗورْ اَڪْ يُ جِگࣺينْ؞ ‏31 ڪࣷنْ لࣹمࣹگْ دُندّ نَ جُرٗومْ ݧَارْ تࣺيمࣺيرِ اَتْ اَڪْ جُرٗومْ ݧَارْ فُڪّ اَڪْ جُرٗومْ ݧَارْ (777)، دࣷورَ فَاتُ؞
‏32 نُوحٍ مࣷومࣹي اَمْ نَ جُرٗومِ تࣺيمࣺيرِ اَتْ (500)، دࣷورَ جُرْ سࣹمْ اَڪْ خَمْ اَڪْ يَفࣹتْ؞
