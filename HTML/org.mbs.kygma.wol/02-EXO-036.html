36
‏1 «ڪࣷنْ نَگْ بࣹسَلࣹلْ اَڪْ اࣷلِيَبْ اَڪْ مبࣷولࣹمْ ݧُ خَرࣹݧْ ݧِ اَجِ سَخْ جِ مَيْ خَرࣹݧْتࣹ اَڪْ دࣺگِّنْ نگِرْ ݧُ مَنَ جࣴفࣹ مبࣷولࣹمْ سَسْ يِ جࣴمْ ݖِ لِگّࣺييُ بࣺرَبْ بُ سࣹلّ بِ، ݧࣷومْ ݧࣺݒّ نَݧُ سࣷتَّلْ لࣺݒّ لِ اَجِ سَخْ جِ سَنْتَانࣹ؞»
‏2 بَ لࣷولُ اَمࣹي مُوسَىٰ وࣷو بࣹسَلࣹلْ اَڪْ اࣷلِيَبْ اَڪْ مبࣷولࣹمْ ݧِ خَرࣹݧْ ݧِ اَجِ سَخْ جِ خَرࣹݧَلْ، مبࣷولࣹمْ ݧِ اَمْ يࣺينࣹيْ وࣵلّسِ ݖِ لِگّࣺييْ بِ، نگِرْ سࣷتَّلْ ڪࣷ؞ ‏3 بَ مُ ڪࣷ دࣹفࣹي ݧُ نَنگࣷو ݖِ مُوسَىٰ مبࣷولࣹمْ جࣷوخࣹ بُ بࣵنِّ إِسْرَاءِيلْ اِنْدِ، نگِرْ سࣷتَّلْ لِگّࣺييُ بࣺرَبْ بُ سࣹلّ بِ؞ بࣵنِّ إِسْرَاءِيلْ نَگْ جࣵݒّࣷو سُبَ گُ جࣷتْ دِ اِنْدِلْ مُوسَىٰ اَيْ سَرَخِ يࣺينࣹ، ‏4 بَ مبࣷولࣹمْ لِگّࣺييْڪَتْ يُ خَرࣹݧْ يِ نࣹڪّࣷونْ ݖِ مبࣷولࣹمْ لِگّࣺييُ بࣺرَبْ بُ سࣹلّ بِ، ڪࣹنّ ڪُ ݖِ نࣹڪّ بࣵيِّ لَ نگَ جࣵݒّࣷونْ، دَلْدِ دِڪّ؞ ‏5 ݧُ نࣹ مُوسَىٰ:‏ «لِ مبࣷولࣷو مِ اِنْدِ وࣹيسْ نَ دَيࣷبْ لِگّࣺييْ بِ اَجِ سَخْ جِ سَنْتَانࣹ؞» ‏6 مُوسَىٰ نَگْ يࣺينࣹلُ ݖَ دَلْ بَ بࣺݒّ، ݧُ نࣹ، ݧࣺݒّ، گٗورْ اَڪْ جِگࣺينْ، بُ ݖِ ڪࣹنّ وَاجَلَتِ اَبْ جࣷوخࣹ، ݧࣹيلْ بࣺرَبْ بُ سࣹلّ بِ؞ فَ لَ مبࣷولࣷو مَ نࣷݒّࣹي اِنْدِ؞ ‏7 بࣷوبَ جُمْتُڪَايْ يَ دࣷيࣷونْ نَ سࣴڪّ، بَ اࣴݒّ سَخْ لَ ݧُ ݖَيْ لِگّࣺييْ لࣺݒّ؞
دࣹفَرْ نَݧُ بࣺرَبْ بُ سࣹلّ بَ
‏8 بَ لࣷولُ اَمࣹي مبࣷولࣹمْ ݧُ خَرࣹݧْ ݧِ بࣷڪّ ݖِ لِگّࣺييْ بِ دَلْدِ دࣹفَرْ جَامࣷوڪَايْ بِ؞ ݧُ دࣹفْ ڪࣷ فُڪِّ سࣷرِ رَبَلِ لࣹي، يُ اࣵندّ اَڪْ وࣴݧْ گُ بَخَ تࣹ لاَلْ يࣷلࣹتْ، اَڪْ وࣴݧْ گُ خࣹوَرْ اَڪْ گُ خࣷنقّ ݖُرّ؞ ݧُ ݧَوَالࣹ ݖِ اَيْ نَتَالِ سࣹرُبْ؞ ‏9 گُدَّايُ سࣷرْ بُ نࣹڪّ دِ ݧَارْ فُڪِّ خَسَبْ اَڪْ جُرٗومْ ݧࣹتّ، يَاتُوَايْ بِ دِ ݧࣹينْتِ خَسَبْ، سࣷرْ يࣺݒّ تࣷلّࣷو؞ ‏10 ݧُ تَقْلࣹ يِي جُرٗومِ سࣷرْ، تَقْلࣹ جُرٗومِ سࣷرْ يِ ݖِ دࣹسْ؞ ‏11 ݖِ ڪَوْ لࣷولُ ݧُ ݧَوْ اَيْ تَڪُّ يُ بَخَ تࣹ لاَلْ يࣷلࣹتْ ݖِ ݖَتَلْ سࣷرْ بِ مُجّ ݖِ بࣹنّ تَقَلَنْتࣹ بِ، سࣷرْ بِ جࣹيخَلْ تَقَلَنْتࣹ بِ ݖِ دࣹسْ اِتْ، ݧُ دࣹفْ ڪࣷ نࣷونُ؞ ‏12 جُرٗومْ فُڪِّ تَڪُّ لَݧُ ݧَوْ ݖِ بࣹنّ سࣷرْ بِ، اَڪْ جُرٗومْ فُڪِّ تَڪُّ ݖِ ݖَتَلْ سࣷرْ بِ جࣹيخَلْ تَقَلَنْتࣹ بِ ݖِ دࣹسْ، تَڪُّ يِ يࣹمࣷو؞ ‏13 ݧُ دࣹفَرْ جُرٗومْ فُڪِّ لࣷنڪُيْ وُرُسْ، دَلْدِ جࣷڪّلَنْتࣹي سࣷرْ يِ لࣷنڪُ يِ، بَ جَامࣷوڪَايْ بِ دِ بࣹنّ؞
‏14 بَ لࣷولُ وࣹيسࣹي ݧُ رࣵبّ اَيْ سࣷرِ ڪَوَرِ بࣴيْ، نگِرْ دࣹفْ ڪࣷ ݧَارࣹيلُ خَيْمَ، مُيْ ڪِيرَايْ ݖِ ڪَوْ جَامࣷوڪَايْ بِ؞ فُڪّ اَڪْ بࣹنّ لَݧُ دࣹفْ سࣷرْ يِ؞ ‏15 گُدَّايُ سࣷرْ بُ نࣹڪّ فَنْوࣹيرِ خَسَبْ لَ، يَاتُوَايْ بِ دِ ݧࣹينْتِ خَسَبْ، تࣹ فُڪِّ سࣷرْ يࣹيڪْ بࣹنّ يࣺݒَّ تࣷلّࣷو دَيࣷ؞ ‏16 ݧُ تَقْلࣹ يِي جُرٗومِ سࣷرْ، جُرٗومْ بࣹنِّ سࣷرْ يِ ݖِ دࣹسْ اِتْ نࣷونُ؞ ‏17 ݧُ سࣵڪّ جُرٗومْ فُڪِّ تَڪُّ ݖِ ݖَتَلْ سࣷرْ، بِ جࣹيخَلْ بࣹنّ تَقَلَنْتࣹ بِ، دَلْدِ سࣵڪّ جُرٗومْ فُڪِّ تَڪُّ اِتَمْ ݖِ ݖَتَلْ سࣷرْ، بِ جࣹيخَلْ تَقَلَنْتࣹ بِ ݖِ دࣹسْ؞ ‏18 ݧُ سࣵڪّ نَگْ جُرٗومْ فُڪِّ لࣷنڪُيْ خࣵنجَرْ يُ ݧُ جࣷڪّلَنْتࣹي ݧَارِ تَقَلَنْتࣹ يِ، مُيْ بࣹنّ خَيْمَ؞ ‏19 ݧُ دࣹفَرَلْ خَيْمَ بِ مَلاَنُمْ دࣹرِ ڪُويْ بُ ݧُ سٗوبْ خࣷنقّ، اَڪْ مࣹنࣹينْ مَلاَنُمْ دࣹرِ ݒِيݒِ مُيْ تࣹگࣹ ݖِ ڪَوَمْ؞
‏20‏-22 گَنَّاوْ گِ ݧُ سࣵڪَّلْ جَامࣷوڪَايْ بِ اَيْ لࣵݖِّ بَنْتُ سࣺنگّ يُ سَمْݒࣹ؞ لࣵݖّ وُ نࣹڪّ اَمْ ݧَارِ ڪࣹنُ يُ ݧُ لࣴڪّلࣹي گَلَنْ يُ ندَوْ، تࣹ نࣷونُ لَݧُ دࣹفْ لࣵݖِّ جَامࣷوڪَايْ بِ يࣺݒّ؞ فُڪِّ خَسَبْ لَݧُ دࣹفْ تَخَوَايُ لࣵݖّ وُ نࣹڪّ، دࣹفْ يَاتُوَايْ بِ خَسَبْ اَڪْ گࣹنّ-وࣵلّ؞ ‏23 ݖِ وࣵلُّ لࣵݖِّ جَامࣷوڪَايْ بِ، ݧُ سࣴسَلࣹ ݧَارْ فُڪِّ لࣵݖّ فِ فࣺيتࣹ بࣴجْ-سَالُمْ، ‏24 بࣷولࣹ ڪࣷوڪْ ݧࣹينتّ فُڪِّ تࣹگُڪَايِ خَالِسْ يُ رࣷنُ ڪࣹنُيْ ݧَارْ فُڪِّ لࣵݖّ يِ، ݧَارِ تࣹگُڪَايْ رࣷنُ لࣵݖّ وِ جِيتُ نگِرْ ݧَارِ ڪࣹنࣷومْ يِ مُ اࣵنْدَلْ، ݧَارِ تࣹگُ تࣷݒّ ݖِ، رࣷنُ وࣺݒّ لࣵݖّ وُ ݖِ تࣷݒّ، نگِرْ ݧَارِ ڪࣹنࣷومْ يِ مُ اࣵنْدَلْ؞ ‏25 گࣹنࣹينْ وࣹتُ جَامࣷوڪَايْ گِ فࣺيتࣹ بࣴجْ-گࣵنَّارْ اِتَمْ، ݧُ دࣹفْ ڪࣷ ݧَارْ فُڪِّ لࣵݖّ، ‏26 اَڪْ سࣹينْ ݧࣹينتّ فُڪِّ تࣹگُيْ خَالِسْ، ݧَارِ تࣹگُ رࣷنُ لࣵݖّ وِ جِيتُ، ݧَارِ تࣹگُ تࣷݒّ ݖِ، رࣷنُ وࣺݒّ لࣵݖّ وُ ݖِ تࣷݒّ؞ ‏27 گَنَّاوْ جَامࣷوڪَايْ گِ فࣺيتࣹ سࣷوُ، ݧُ سࣵڪَّلْ ڪࣷ جُرٗومْ بࣹنِّ لࣵݖّ، ‏28 تࣹگْ ݖِ ݧَارِ لࣵݖّ نگِرْ ݖࣷلِّ جَامࣷوڪَايْ بِ ݖِ گَنَّاوْ، ‏29 ݧُ اࣵنْدَنْدࣷو دِ بࣹنّ ݖِ سُوفْ تࣹ لࣴڪّلࣷو ݖِ ڪَوْ ݖِ بࣹنّ لَمْ ڪࣹݒّ؞ نࣷونُ لَݧُ لࣹينْ دࣹفْ ݧࣷومْ ݧَارْ، ݧُيْ ݧَارِ ݖࣷلّ يِ؞ ‏30 مُيْ جُرٗومْ ݧࣹتِّ لࣵݖّ اَڪْ سࣹينِ تࣹگُيْ خَالِسْ؛ دِ فُڪِّ تࣹگُ اَڪْ جُرٗومْ بࣹنّ، ݧَارِ تࣹگُ رࣷنُ لࣵݖّ وُ نࣹڪّ؞
‏31 گَنَّاوْ لࣷولُ ݧُ سࣵڪّ اَيْ گَلَنِ بَنْتِ سࣺنگّ، جُرٗومِ گَلَنْ يِيْ اࣵندّ اَڪْ لࣵݖّ، يِ ݖِ گࣹنّ وࣹتُ جَامࣷوڪَايْ بِ، ‏32 اَڪْ جُرٗومِ گَلَنْ نگِرْ لࣵݖِّ وࣹتْ گِ ݖِ دࣹسْ، اَڪْ جُرٗومِ گَلَنْ نگِرْ لࣵݖّ يِ ݖِ گَنَّاوْ جَامࣷوڪَايْ بِ، فِ فࣺيتࣹ سࣷوُ؞ ‏33 ݧُ دࣹفْ گَلَنُ دِگّ بِ ݖِ دِگّ لࣵݖّ يِ، دَلࣹ ڪࣷ ݖَتْ بَ ݖَتْ؞ ‏34 ݖِ بِيرْ لࣷولُ ݧُ خࣷوبْ لࣵݖّ يِ وُرُسْ، سࣵڪَّلْ لࣹينْ اَيْ لَمِ وُرُسْ يُيْ تࣺيࣹ گَلَنْ يِ، دَلْدِ خࣷوبْ گَلَنْ يِ اِتْ وُرُسْ؞
‏35 ݧُ سࣵڪّ اِتَمْ رِدࣷ بِ، رࣵبّࣹ ڪࣷ وࣴݧْ گُ بَخَ تࣹ لاَلْ يࣷلࣹتْ، اَڪْ وࣴݧْ گُ خࣹوَرْ، اَڪْ گُ خࣷنقّ ݖُرّ اَڪْ وࣴݧُ لࣹي، ݧَوَالࣹ ݖِ اَيْ نَتَالِ سࣹرُبْ؞ ‏36 ݧُ سࣵڪَّلْ رِدࣷ بِ ݧࣹينْتِ جࣴنِ سࣺنگّ يُ ݧُ خࣷوبْ وُرُسْ، جࣴنْ يِ اࣵندّ اَڪْ وࣺڪُّڪَايِ وُرُسْ، ݧُ مٗولَلْ جࣴنْ يِ ݧࣹينْتِ تࣹگُيْ خَالِسْ؞
‏37 ݧُ سࣵڪَّلْ بُنْتُ خَيْمَ بِ ڪِيرَايْ لُ ݧُ رࣵبّࣹ وࣴݧْ گُ بَخَ تࣹ لاَلْ يࣷلࣹتْ، اَڪْ وࣴݧْ گُ خࣹوَرْ، اَڪْ گُ خࣷنقّ ݖُرّ اَڪْ وࣴݧُ لࣹي، مُ دࣷونْ لِگّࣺييُ رࣵبّ بُ رَفࣹتْ، ‏38 اَڪْ جُرٗومِ جࣴنَمْ يُ اࣵندّ اَڪْ سࣹينِ وࣺڪُّڪَايْ؞ ݧُ خࣷوبْ نَگْ وُرُسْ بࣷݒِّ جࣴنْ يِ اَڪْ سࣹينِ گَلَنْ، جُرٗومِ تࣹگُيْ جࣴنْ يِ دِ خࣵنجَرْ؞
