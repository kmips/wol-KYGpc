17
‏1 بُلࣹينْ سَرْخَلَلْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ، اَوْ نَگْ مبَا اَبْ گࣵتّ بُ اَمْ سِڪّ، مبَا جࣹنّ لاَگࣷ، ندَخْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ سࣹيخْلُ نَ لࣷولُ؞
‏2 بُ نِتْ فࣹيݧࣹي ݖِ سࣹينْ بِيرْ، ݖِ بࣹنّ ݖِ دࣴڪّ يِ لࣹينْ سࣹينْ يࣵلَّ اَجِ سَخْ جِيْ جࣷخْ، مُيْ گٗورْ مبَا جِگࣺينْ جُيْ دࣹفْ لُ سࣹينْ يࣵلَّ اَجِ سَخْ جِ ݧَاوْلُ، دِ فࣹݖِّ ڪٗلّࣴرࣹيمْ، ‏3 بَيْ جَامُجِ يࣹنࣹينْ يࣵلَّ، دِ لࣹينْ سُجّٗوتَلْ، مبَا مُيْ سُجّٗوتَلْ جَنتّ بِ، مبَا وࣹيرْ وِ، مبَا مبࣷولࣹمْ گࣵنگࣷورُ اَسَمَانْ، تࣹ سَنْتَانࣹوُمَ ݖِ لࣹنّ، ‏4 بُݧُ لࣹينْ ڪࣷ يࣹگّࣹي بَ نگࣹينْ دࣺگّ ڪࣷ، سࣹيتْلࣹينْ مبِرْ مَ بُ بَاخْ؞ سُ دࣹي دࣴگّ گُ وࣺرْ، نࣹ دࣹفࣹيسْ نَ لُ جࣷمبّلُ لࣷولُ ݖِ بِيرْ إِسْرَاءِيلْ، ‏5 نَنگࣹينْ گࣺنّࣹ بَ ݖَ بُنْتُ دࣴڪّ بَ، وَا جِ، مبَا ندَوْ سِ دࣹفْ جࣷوجُ جࣴفْ جُ بࣷنْ، تࣹ نگࣹينْ دٗورْ اَيْ دࣷجْ، وَا جࣷوجُ مبَا ندَوْ سِ، بَ مُ دࣹي؞ ‏6 ڪࣵدُّگْ ݧَارِ سࣹيدࣹ بَ ݧࣹتّ نَگْ، ݖِ لࣹيسْ دِ رࣹيࣹ ڪِ ݧُ وَرَ رࣹيْ، وَايࣹ دࣹيسُ ڪࣷ رࣹيْ ݖِ ڪࣵدُّگْ بࣹنّ سࣹيدࣹ؞ ‏7 لࣷخࣷيْ سࣹيدࣹ يࣹي ڪࣷيْ جࣴڪَّ دٗورْ دࣷجْ نگِرْ رࣹيْ ڪࣷ، لࣷخࣷيْ مبࣷولࣷو مࣺݒّ دࣷورْ ڪࣷو دٗورْ؞ نࣷونُ نگࣹينْ دِ تࣹنقࣹي لُ بࣷنْ ݖِ سࣹينْ بِيرْ؞
‏8 سُ لࣹينْ مبِرْ جَفࣹي بَ وࣹيسْ لࣹينْ ݖِ اࣵتّࣹبْ بَڪَّنْ بُ ݧُ فَاتْ، اَڪْ اَيْ جࣷتࣹ، اَڪْ لࣷرَانگࣹيْ يَرَمْ يُ ݧُ تࣹگْ نِتْ، اَڪْ مبࣷولࣹمْ دࣴنگࣷو بُ لاَجْ اࣵتّࣹ ݖِ سࣹينْ بِيرْ دࣴڪّ، دَنگࣹينْ دِ دࣹمْ بَ ݖَ بࣺرَبْ بَ سࣹينْ يࣵلَّ اَجِ سَخْ جِ دِ تَامُجِ، ‏9 نگࣹينْ اࣵگّ ݖَ سَرْخَلْڪَتْ يَ، لࣹوࣹينْ ݧَ، اَڪْ اࣵتّࣹڪَتْ، بَ مبِرْ مَ يࣹمࣷولْ جَمࣷنࣷ، تࣹ نگࣹينْ سࣵڪُّ ݧُ بِرَلْ لࣹينْ اࣵتّࣹبْ مبِرْ مَ؞ ‏10 نَنگࣹينْ جࣴفࣹ نَگْ ڪࣵدُّ گَ ݧُ لࣹينْ ݖَ وَخْ، ݖَ بࣺرَبْ بࣷوبَ اَجِ سَخْ جِ تَامُ، تࣹ نگࣹينْ سࣵمّࣷو ڪࣷو جࣴفࣹ مبࣷولࣹمْ نࣷونࣹي ݧُ لࣹينْ ڪࣷ دِگَلࣹ؞ ‏11 ڪࣵدُّگْ ندِگَلْ لَ ݧُ لࣹينْ دِگَلْ، اَڪْ اࣵتّࣹ بَ ݧُ لࣹينْ ݖِ وَخْ، جࣴفࣹلࣹينْ ڪࣷ؞ بُلࣹينْ لَجّࣹ فࣹنّ ڪࣵدُّ گَ ݧُ لࣹينْ ݖَ وَخْ؛ دُ ندِجࣷورْ، دُ ݖࣵمّࣷݧْ؞ ‏12 ڪࣺݒّ ڪُ ݖِ دَݝِّيرَلْ، بَݧَ دࣺگَّلْ سَرْخَلْڪَتْ بَ فَ اَيࣹي لِگّࣺييَلْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ، بَݧَ دࣺگَّلْ اࣵتّࣹڪَتْ بَ، ڪࣷوڪُ دࣹيسْ ڪࣷيْ رࣹيْ؞ نࣷونُ نگࣹينْ دِ تࣹنقࣹي لُ بࣷنْ ݖِ إِسْرَاءِيلْ؞ ‏13 اَسْڪَنْ وࣺݒَّ ڪࣷيْ دࣺگّ، تِيتْ، بَ دُݧُ دَݝِّيرَلَتِ؞
اَيْ ندِگَلْ ݧࣹيلْ نَ بُورْ
‏14 بُ نگࣹينْ دࣹمࣹي نَگْ ݖَ رࣺيوْ مَ لࣹينْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ دِ جࣷخْ، بُ نگࣹينْ ڪࣷ نَنگࣷو، دࣴڪّࣹ، بَ نࣹ:‏ «نَنُ فَلْ بُورْ بُ نُ جِيتࣹ نِ مبࣷولࣹمْ خࣹيتْ يِ نُ سࣺقْ،» ‏15 سُ بࣷوبَا فࣹخࣹلࣹينْ بَ ڪَ نگࣹينْ دِ فَلْ بُورْ، دِ ڪَ سࣹينْ يࣵلَّ اَجِ سَخْ جِ تَامُ؞ ݖِ سࣹينْ بِيرِ بࣷڪّ نَگْ نگࣹينْ دِ فَلْ سࣹينْ بُورْ؞ دُنگࣹينْ مَنَ فَلْ بُورْ اَبْ جَامْبُرْ بُ دُلْ سࣹينْ مبࣷڪُّمْ بࣵنِّ إِسْرَاءِيلْ؞ ‏16 وَايࣹ بُ بُورْ بِ بَرࣹيْ فَسْ، تࣹ بُمُ دࣹلّࣷو اَسْڪَنْ وِ مِسْرَ، نگِرْ سࣵڪُّ فَسْ يُ بَرࣹ، تࣹ اَجِ سَخْ جِ وَخࣷونْ لࣹينْ نࣹ بُلࣹينْ دࣹلّࣷوتِ وࣷووُ يࣷونْ؞ ‏17 بُمُ بَرࣹيْ جَبَرْ، بَلاَ اَمْ خࣹلَمْ دِ لَجّ؞ خَالِسْ اَڪْ وُرُسْ اِتْ، بُمُ ڪࣷ بَرࣹلࣹ لࣷولْ؞
‏18 بُ بُورْ بِ تࣷوگࣹي ݖِ جَلُبْ نگُورَمْ، نَ بِندّلُ نگِرْ مࣷومْ ݖِ بࣷݒَّمْ، اَبْ سࣷتِّبْ تࣺيرࣹبْ يࣷونْ وِي ݖِبْ تَخَݧْ، ݖِ دࣺگْتَلُبْ سَرْخَلْڪَتْ يِ، لࣹوࣹينْ ݧِ؞ ‏19 نَ تࣺيرࣹ بࣷوبُ اࣵندّ اَڪْ مࣷومْ، مُ دِ ݖِ جࣵنگّ گِيرُ دُنْدَمْ، ندَخْ مُ تࣵگَّتُ ݖِ رَگَلْ يࣵلاَّمْ اَجِ سَخْ جِ، نگِرْ دِ سࣵمّ مبࣷولࣹمْ ڪࣵدُّيْ يࣷونْ وِي، اَڪْ دࣷگَلِ يࣷونْ يِي، نگِرْ دِ ڪࣷ جࣴفࣹ، ‏20 بَ دُ خࣷگّلِڪُ ݖِ ڪَوِ بࣷڪَّمْ، تࣹ دُ مࣷيࣹ سَنْتَانࣹ بِ فࣹنّ؛ دُ ندِجࣷورْ، دُ ݖࣵمّࣷݧْ، ندَخْ مُ گُدّ فَنْ ݖِ نگُورَمْ، مࣷومْ اَڪْ اَسْڪَنَمْ، ݖِ دِگّ إِسْرَاءِيلْ؞
