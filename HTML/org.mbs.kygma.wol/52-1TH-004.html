سࣹلَّلْ وَرْتࣺيفْ لَ
4
‏1 لِ ݖِ دࣹسْ نَگْ، بࣷڪّ يِ، مࣷو دِ يࣹينَ نَنگࣷو ݖِ سُنُ لࣵمِّݧْ، جࣴفِنْ وِ نگࣹينْ وَرَ جࣴفࣹ بَ نࣹيخْ يࣵلَّ، تࣹ اِتْ مࣷومْ نگࣹينْ نࣹڪّࣹ؞ وَايࣹ دَنُ لࣹينْ دِ تِنُ، دِ لࣹينْ ݧَاخْ، ݖِ تُرُ سَنگّ عِيسَىٰ، نگِرْ نگࣹينْ گࣴنَتِ ڪࣷو جࣴفࣹ؞ ‏2 خَمْ نگࣹينْ ندِگَلْ، يِ نُ لࣹينْ جࣷخْ ݖِ تُرُ سَنگّ عِيسَىٰ؞ ‏3 لِي مࣷو دِ ݖࣷوبَرࣹيْ يࣵلَّ:‏ نگࣹينْ سࣹلَّلْ تࣹ مُݖّ ݖِ ݒࣷوُمْ سࣺيْ؞ ‏4 سُ ڪࣷ دࣹفࣹي ڪࣹنّ ڪُ نࣹڪّ ݖِ يࣹينْ مَنَ مࣷومْ يَرَمُ بࣷݒَّمْ ݖِ بِيرْ سࣹلّنگَ اَڪْ تࣹدّنگَ، ‏5 تࣹ نگࣹينْ اࣵندّ اَڪْ سَگࣷ ݖِ سࣹينِ خࣹمّࣹمْتࣺيفْ، بَݧَ مࣹلْ نِ نِتِ خࣹيتْ يِ خَمُلْ يࣵلَّ؞ ‏6 ݖِ مࣷومُ مبِرْ، بُ ڪࣹنّ ݖِ يࣹينْ تࣷوݧْ مبࣷڪَّمْ، دِ آڪِمࣷو يࣹلّࣹيفَمْ، ندَخْ وَخࣷونْ نَنُ لࣹينْ ڪࣷ، بَ اَرْتࣷوتِ لࣹينْ نࣹ لࣹينْ جࣴفْ جُ نِ مࣹلْ، بࣷرࣷومْ بࣹي ڪࣷيْ فࣹيْ ݧَ ڪࣷ دࣹفْ؞ ‏7 يࣵلَّ ڪَتْ ݖِ سࣹلّنگَ لَ نُ وࣷو، وَايࣹ دُ ݖِ جࣴفِنْ وُ سࣹتُلْ؞ ‏8 ڪࣷنْ نَگْ ڪُ جَلَخْ ندِگَلْ لِي، دُ نِتْ نگَ جَلَخْ، خَنَا يࣵلَّ، مࣷومْ مِ لࣹينْ جࣷخْ نࣷووَمْ گُ سࣹلّ؞
‏9 ݖِ وࣵلُّ مبࣴگّࣹيلْ گِ وَرْ ݖِ بِيرِ بࣷڪّ، سࣷخْلَوُلࣹينْ ڪُ لࣹينْ ݖِ بِندّ دَرَ، ندَخْ يࣹينْ ݖِ سࣹينْ بࣷݒّ، يࣵلَّ تࣵگَّتْ نَ لࣹينْ ݖِ سࣷݒَّنْتࣹ، ‏10 تࣹ مࣷومْ نگࣹينْ نࣹڪّࣹ سࣹينْ دِگّ اَڪْ مبࣷولࣹمْ بࣷڪِّ دِيوَانُ مَسࣹدُوَنْ گࣺݒّ؞ تࣹيوُلْ نَگْ، بࣷڪّ يِ، نُ دِ لࣹينْ ݧَاخْ، نگِرْ نگࣹينْ گࣴنَتࣹ جࣴفࣹ نࣷونُ؞ ‏11 سࣵڪُّلࣹينْ دِ جࣴفࣹ جࣴفِنْ وُ دَلْ، تࣹ نگࣹينْ دِ تࣷݒّتࣷو لُ سࣹينْ يࣷونْ نࣹڪّ، تࣹيْ دُنْدࣹ سࣹينُ ݧَقْ، نَ نُ لࣹينْ ڪࣷ سَنْتࣹ وࣷونْ؞ ‏12 ڪࣷنْ دِنگࣹينْ جࣹڪّ نُ نگࣹينْ دِ جࣴفْلَنْتࣹي اَڪْ ݧُ گࣴمُلْ ݧَ ݖَ بِتِ، تࣹ دُنگࣹينْ سࣷخْلاَلْ ڪࣹنّ دَرَ؞
تِيسْلُ بُ اࣴݒّ وَرُلْ
‏13 بࣷڪّ يِ نَگْ، بُگُّنُ نگࣹينْ رࣺيرࣹ مبِرُمْ وَيْ دَوْلُ ݧِ، بَيْ تِيسْلُ نِ ݧࣹنࣹينْ ݧِ اَمُلْ جࣹنّ يَاڪَارْ، دِ تِيسْلࣷو؞ ‏14 ندࣹگَمْ گࣴمْ نَنُ نࣹ عِيسَىٰ دࣹي نَ، بَ دࣹڪِّ، نࣷونُ اِتْ لَنُ وَرَ گࣴمࣹ نࣹ يࣵلَّ مࣷويْ دࣹڪَّلْ وَيْ دَوْلُ ݧِ گࣴمࣷونْ عِيسَىٰ، بَ بࣷولࣹ لࣹينْ اَڪْ عِيسَىٰ، اِنْدَالࣹ؞
‏15 لِي نُ لࣹينْ دِ اࣵگّࣹ، تࣹ مُيْ ڪࣵدُّگْ بࣷرࣷومْ بِ دࣹ، مࣷو دِ نُنْ ݧِ فِيْ دࣹسࣹ بَڪَّنْ بَ ڪࣹرَ سَنگّ بِيْ دࣺلْسِ، دُنُ جِيتُ وَيْ دَوْلُ ݧِ فَ مࣷومْ؞ ‏16 سَنگّ بِ ݖِ بࣷݒَّمْ، وࣷوتࣹ بُ خُمبّ لَيْ وࣵݖَّالࣹي اَسَمَانْ، بَاتُ ڪِلِفَگْ مَلاَڪَ جِبْ، اࣵندّ اَڪْ ݖࣷووَلْ لِيتُ يࣵلَّ؞ سُ ڪࣷ دࣹفࣹي ݧَ سࣹينُگْ دࣹي فࣹڪّ ݧُ گࣴمْ اَلْمَسِ ݧࣷويْ جࣴڪَّ دࣹڪِّ؞ ‏17 گَنَّاوْ لࣷولُ نُنْ ݧِ فِ دࣹسࣹ بَڪَّنْ لࣹيسْ دِ ڪࣴفَنْدࣷو اَڪْ ݧࣷومْ، جࣴمࣹ ݖَ نِيرْ يَ، نگِرْ نُ دَجࣹجࣹيڪْ سَنگّ بِ فَ جَوّ جَ، نُ دَلْدِ نࣹڪّ اَڪْ مࣷومْ بَ فࣵوّ؞ ‏18 ڪࣷنْ نَگْ دࣹيلࣹينْ دࣴفَلَنْتࣹي يࣷويُ ڪࣵدُّ؞
