عِيسَىٰ يࣹبَلْ نَ فُڪِّ تَالِبࣹ يِ اَڪْ ݧَارْ
9
‏1 عِيسَىٰ مࣷو وࣷو وࣷونْ فُڪّ ݧَاڪْ ݧَارْ، جࣷخْ لࣹينْ خَمْ-خَمْ اَڪْ سَݧْ-سَݧُ دࣵقْ مبࣷولࣹمْ رَبْ تࣹيْ فَجْ اَيْ جࣵنگࣷرࣷ؞ ‏2 ݖِ ڪَوْ لࣷولُ مُ يࣹبَلْ لࣹينْ نگِرْ ݧُ يࣺگْلࣹجِ نگُورُگْ يࣵلَّ تࣹيْ فَجْ جَرَگْ جِ؞ ‏3 مُ نࣹ لࣹينْ:‏ «بُ نگࣹينْ دِ دࣹمْ، بُلࣹينْ يٗبَّالࣹ دَرَ نگِرْ فࣵگُّ يࣷونْ وِ؛ دُ يࣹتْ، دُ مبُوسْ، دُ اَوْ ݧَمْ، دُ خَالِسْ؞ بُ ݖِ ڪࣹنّ يࣷرْ ݧَارِ تُرْڪِ سَخْ؞ ‏4 ڪࣴرْ گُ نگࣹينْ دُگّ، فَ نگࣹينْ دِ دَلْ، بَ ڪࣹرَ نگࣹينْ دࣹمْ؞ ‏5 فࣺݒّ فُ ݧُ لࣹينْ دَلَلࣹوُلْ، گࣺنّلࣹينْ دࣴڪّ بࣷوبَ تࣹ نگࣹينْ فَخَسْ فَ سࣹينْ ݒࣴنْدُبْ ندࣴگُّيْ تࣵنڪّ، سࣹيدࣹيلࣹ لࣹينْ ڪࣷ سࣹينْ نگࣵنْتَلْ؞»
‏6 ݧُيْ دࣹمْ نَگْ، دِ وࣴرْ دࣴڪّ اَڪْ دࣴڪّ، دِ يࣺگْلࣹ خِبَارُ جࣵمّ بِ، دِ فَجْ جَرَگْ جِ؞
‏7 ݖِ ڪَوْ لࣷولُ اࣹرࣷدّ بࣷرࣷومْ گَلِلࣹ دࣺگّ مبࣷولࣹمْ لَ خࣹوْ، بࣷݒَّمْ نَگْ اُبُ، ندَخْ ݧࣹنّ ݧُ نࣷونْ يَحْيَىٰ مࣷو دࣹڪِّ، دِ عِيسَىٰ؞ ‏8 ݧࣹي نࣹ اِلْيَاسْ مࣷو دࣺلْسِ، ݧࣹنࣹينْ بَ تࣹيْ، نࣹ ڪࣹنّ ݖَ يࣷنࣹنتّ يُ جࣴڪّ يَا دࣹڪِّ؞ ‏9 اࣹرࣷدّ دَلْدِ نࣹ:‏ «يَحْيَىٰ، مَنْ مَا دࣷگْلُ بࣷݒَّمْ؞ ڪࣷنْ ڪِ مَ دࣺگَّلْ جࣴفْ جِي، مࣷويْ ڪَنْ؟» بَ لࣷولُ اَمࣹي مُ دِ ڪࣷ فࣹخࣹي گِسْ؞
عِيسَىٰ لࣹيلْ نَ مبࣷولࣷو
‏10 بَ ندَوْ يَ دࣺلْسࣹي، دَݧࣷو نࣹتَّلِ عِيسَىٰ مبࣷولࣹمْ لَ ݧُ دࣹفْ؞ مُ اࣵندّ اَڪْ ݧࣷومُ نࣹينْ بَ سࣷرࣹ، ݖَ وࣹتُ دࣴڪّ بَ ݧُيْ وَخْ بࣹݖَّيْدَ؞ ‏11 تࣹيوُلْ نِتْ ݧَ يࣺگْ نࣹ فَ لَ دࣹمْ، دَلْدِ ڪࣷ تࣷݒّ؞ عِيسَىٰ دَلَلْ لࣹينْ، دِ لࣹينْ وَخْ ݖِ مبِرُمْ نگُورُگْ يࣵلَّ، ݧَ آجࣷوࣷو اَمْ ݒَجْ، مُ وࣺرَلْ لࣹينْ؞
‏12 بَ جَنتّ دِ سࣷوَتْ، فُڪّ ݧَاڪْ ݧَارْ دِڪّ نࣹ عِيسَىٰ:‏ «يِوِلْ مبࣷولࣷو مِ، ݧُ دُگّ ݖِ دࣴڪّ يُ ندَوْ يِي اَڪْ دࣴڪّ-دࣴڪَّانْ يِ ڪࣷ وࣴرْ، نگِرْ ݧُ گِسْ لُ ݧُ لࣹڪّ اَڪْ فُ ݧُ فَنَانْ، ندَخْ فِي اࣵلُّبْ نࣹينْ لَ؞» ‏13 مُ نࣹ لࣹينْ:‏ «يࣹينْ، جࣷخْلࣹينْ لࣹينْ لُ ݧُ لࣹڪّ؞» ݧُ نࣹ ڪࣷ:‏ «نُنْ دࣹ اَمُنُ لُ دُلْ جُرٗومِ مبُرُ اَڪْ ݧَارِ جࣴنْ؞ خَنَا نُ دࣹمْ، نُنْ ݖِ سُنُ بࣷݒّ، جࣴنْدَلِ مبࣷولࣷو مِي مࣺݒّ لُ ݧُ لࣹڪّ!‏» ‏14 لُ وَرَ مَتْ جُرٗومِ جُنِّيْ گٗورَ نگَ فَ وࣷونْ؞ عِيسَىٰ نࣹ تَالِبࣹ يَ:‏ «دَنگࣹينْ لࣹينْ دِ تࣷوگَلْ، ݧُ دࣹفْ گࣺيوِ جُرٗومْ فُڪِّ نِتْ؞» ‏15 ݧُ دࣹفْ نࣷونُ، تࣷوگَلْ ݧࣺݒّ؞
‏16 بَ مُ ڪࣷ دࣹفࣹي عِيسَىٰ جࣴلْ جُرٗومِ مبُرُ يَ اَڪْ ݧَارِ جࣴنْ يَ، خࣷولْ ڪَوْ اَسَمَانْ، دَلْدِ يࣺڪَّتِ ڪࣵدُّگْ ݖَنتّ، بَ نࣷݒِّ دَگّ مبُرُ يَ، جࣷخْ تَالِبࣹ يَ، نگِرْ ݧُ تَاجَلْ ڪࣷ مبࣷولࣷو مَ؞ ‏17 ݧࣺݒّ لࣹڪّ بَ رࣹگّ، بَ ݧُ يٗبُّ دࣷگْ يَ ݖَ دࣹسْ، مُيْ فُڪِّ ݒَݧࣹ اَڪْ ݧَارْ؞
ݒِيࣹيرْ خَمْ نَ ڪُيْ عِيسَىٰ
‏18 عِيسَىٰ مࣷو بࣹرُ وࣷونْ، دِ جُلِّ، تَالِبࣹ يَ فࣹگّࣷوڪْ مࣷومْ؞ ݖِ ڪَوْ لࣷولُ مُ لاَجْ لࣹينْ نࣹ لࣹينْ:‏ «اَنَ ڪُ نِتْ ݧِ نࣹ مࣷومْ لاَ؟» ‏19 ݧُ نࣹ ڪࣷ:‏ «ݧِي نࣹي ݧُ يَا دِ يَحْيَىٰ؛ ݧࣹي نࣹ يَا دِ اِلْيَاسْ؛ ݧَلࣹ نࣹ ڪࣹنّ ݖِ يࣷنࣹنتّ يُ جࣴڪّ يَا دࣹڪِّ، دِ يَوْ؞»
‏20 عِيسَىٰ نࣹ لࣹينْ:‏ «يࣹينْ نَگْ، نگࣹينْ نࣹ مَايْ ڪَنْ؟» ݒِيࣹيرْ نࣹ ڪࣷ:‏ «يَا دِ اَلْمَسِبْ يࣵلَّ؞» ‏21 مُ فࣹمُّ لࣹينْ نَگْ، سَنتّ لࣹينْ ݧُ بَݧْ ڪࣷو وَخْ ڪࣹنّ؞ ‏22 مُ نࣹيتِ:‏ «فࣵوّ دࣷومُ نِتْ ڪِ سࣷنّ لُ بَرࣹ، بَ دَجࣹيڪْ مبَݧࣹيلُ مَگِ اَسْڪَنْ وِ اَڪْ سَرْخَلْڪَتْ يُ مَگْ يِ، اَڪْ فِرِڪَتِ يࣷونْ يِ، تࣹ دࣹيسْ نَ ڪࣷ رࣹيْ، بَ ݧࣹتّࣹيلُ فَنَمْ، مُ دࣹڪِّ؞»
‏23 عِيسَىٰ نَگْ وَخْ لࣹينْ ݧࣷومْ ݧࣺݒّ، نࣹ لࣹينْ:‏ «ڪࣺݒّ ڪُ بࣴگَّ دِڪّ تࣷݒّ مَ، فࣵتّࣹيلْ سَ بࣷݒّ، تࣹ نگَ گࣵدُّ بࣺسْ بُ نࣹڪّ، بَنتّ بِ ݧُ لَيْ دَاجْ، دࣷورْ مَا تࣷݒّ؞ ‏24 ندَخْ ڪࣺݒّ ڪُ نَمَّ رَوَلࣹ سَ بَڪَّنْ، يَايْ ݧࣵڪّ سَ بَڪَّنْ، وَايࣹ ڪࣺݒّ ڪُ ݧࣵڪّ سَ بَڪَّنْ نگِرْ مَنْ، يَا مُسَلْ سَ بَڪَّنْ؞ ‏25 ندَخْ نِتْ ڪِ، سُ جَگࣷو اࣵدِّنَ سࣺݒّ سَخْ، تࣹ نَرَ ݧࣵڪّ بَڪَّنَمْ مبَا مُ سࣵنڪُ، اَنَ لُ مُ ڪࣷيْ جَرِݧْ؟ ‏26 ڪࣺݒّ ڪُ ڪࣹرْسَوࣷو سَمَ جࣴمّ، ڪࣹرْسَوࣷو تࣷݒّ سَمَيْ وَخْ، يَوْ اِتَمْ دࣷومُ نِتْ ڪِ دِنَ لَ ڪࣹرْسَوࣷو، بࣺسْ بُ دِڪّࣹي ݖِ دَرَجَامْ اَڪْ دَرَجَيْ بَايْ بِ، اَڪْ دَرَجَيْ مَلاَڪَ يُ سࣹلّ يِ؞ ‏27 مَا لࣹينْ ڪࣷ وَخْ دࣺيْ، ݧࣹنَّ نگِ فِي تَخَوْ تࣹيْ، دُݧُ دࣹي تࣹ گِسُݧُ نگُورُگْ يࣵلَّ؞»
لࣹيرُ عِيسَىٰ جࣷلِّ نَ
‏28 بَ لُ وَرَ تࣷلّࣷوڪْ جُرٗومْ ݧࣹتِّ فَنْ وࣺيࣹي گَنَّاوْ ڪࣵدُّ يࣷويُ، عِيسَىٰ اࣵنْدࣷونْ نَ اَڪْ ݒِيࣹيرْ اَڪْ يࣷوَانْ اَڪْ يَعْقُوبَ، ݧُ يࣺيگْ ݖَ ڪَوْ تُندّ وَ نگِرْ جُلِّ؞ ‏29 نَڪَ لَيْ جُلِّ، خَرْ ڪَنَمَمْ سࣷݒِّڪُ، اَيْ يࣺرࣹيمْ وࣹيخْ تࣵلّ، دِ مࣹلَخْ؞ ‏30 ݖِ ڪَوْ لࣷولُ ݧَارِ نِتْ جࣹڪِّ فࣹيݧْ، دِ وَخْتَانْ اَڪْ مࣷومْ؛ ݧُيْ يࣷنࣹنتّ يࣵلَّ مُوسَىٰ اَڪْ يࣷنࣹنتّ يࣵلَّ اِلْيَاسْ؞ ‏31 ݧࣷو فࣹيݧْ ݖِ اَگْ لࣹيرْ، دِ وَخْتَانࣹ دࣴدُّگْ عِيسَىٰ، گَ مُ دࣷونْ وَاجَ سࣷتَّلࣹ فَ يࣹرُسَلࣹمْ؞ ‏32 بࣷوبَ نگࣴمّࣹنْتَ نگَيْ مَنْ ݒِيࣹيرْ اَڪْ ݧَ مُ اࣵنْدَلْ؞ بَ ݧُ تࣹيوْلࣷو نَگْ لَݧُ گِسْ لࣹيرُ عِيسَىٰ، اَڪْ ݧَارْ ݧَ مُ تَخَوَلْ؞ ‏33 بَ ݧُ دࣹمࣹي بَ ݧَارِ نِتْ ݧࣷوݧُ دِ بࣵيِّڪࣷو فَ عِيسَىٰ، ݒِيࣹيرْ نࣹ ڪࣷ:‏ «نجَاتِگࣹ، سُ نُ تࣷوگࣷونْ فِي دࣹ، مُ بَاخْ ݖِ نُنْ؛ نَنْ يࣺڪَّتِ ݧࣹتِّ مبَارْ، بࣹنّ يَوْ، بࣹنّ مُوسَىٰ، بࣹنّ اِلْيَاسْ؞» وَايࣹ بࣷوبَ ݒِيࣹيرْ خَمُلْ لَ مُيْ وَخْ؞ ‏34 نَڪَ لَيْ وَخْ لࣷولُ، اَوْ نِيرْ دِڪّ يِيرْ لࣹينْ، تَالِبࣹ يَ تࣵبِّ ݖَ بِيرْ نِيرْ وَ، تِيتَانگࣹ جࣵݒّ لࣹينْ؞ ‏35 ݖِ بِيرْ لࣷولُ بَاتْ جِبࣹ ݖَ نِيرْ وَ، نࣹ:‏ «ڪِي مࣷو دِ سَمَ دࣷومْ، ڪِ مَ تَامُ؛ دࣺگْلُلࣹينْ ڪࣷ؞»
‏36 بَ بَاتْ بَ سࣹلَوࣹي، عِيسَىٰ دࣷݝَّ فࣹيݧْ؞ تَالِبࣹ يَ نَگْ ݧࣷومْ نࣹ ݖࣹلّ، وَخُݧُ ڪࣹنّ لࣹنّ جَنتّ يࣷويُ، ݖَ لَ ݧُ گِسْ؞
عِيسَىٰ فَجَتِ نَ ڪُ رَبْ جࣵݒّ
‏37 ݖَ اࣴلّࣴگْ سَ بَ ݧُيْ وࣵݖّࣹ تُندّ وَ، مبࣷولࣷو مُ بَرࣹي دِڪّ دَجࣹيڪْ عِيسَىٰ؞ ‏38 جࣹنّ وَايْ جࣹڪِّ خَاݖࣷو ݖَ مبࣷولࣷو مَ، نࣹ:‏ «سࣴرِݧْ بِ، دَمَ لَيْ ݧَانْ، نگَ خࣷولَلْ مَ سَمَ دࣷومْ، ندَخْ مࣷومْ دࣷݝّ لاَ اَمْ؞ ‏39 رَبْ ڪَتْ مࣷو ڪࣷ جࣵݒّ، مُ جࣹڪِّ دِ يُوخُ، دِ سَيْ، گࣺمِّݧْ گِيْ فُورْ؞ دَانَڪَ دُ تࣹقْلِڪࣷوڪْ مࣷومْ تࣹ سࣷنَلْ نَ ڪࣷ؞ ‏40 ݧَانْ نَا سَيْ تَالِبࣹ نگِرْ ݧُ دࣵقَلْ مَ ڪࣷ، وَايࣹ مَنُݧُ ڪࣷ؞» ‏41 عِيسَىٰ نࣹ لࣹينْ:‏ «ݖࣺيْ نِتِ تࣹيْ، يࣹينَ گࣴمَدِ تࣹ دࣴنگّ!‏ مَاڪْ يࣹينْ بَ ڪَݧْ، تࣹ دِ لࣹينْ مُݧَلْ؟ اِنْدِ سَ دࣷومْ فِي؞»
‏42 نَڪَ لَ خَلࣹ بِ جُبْسِ عِيسَىٰ، رَبْ وِ سࣵنِّ ڪࣷ ݖِ سُوفْ، مُيْ سَيْ؞ عِيسَىٰ اِتَمْ گࣴدّ رَبْ وَ، وࣺرَلْ خَلࣹ بَ، دࣹلّࣷو ڪࣷ بَايَمْ؞ ‏43 ݧࣺݒّ نَگْ وَارُ ݖِ مࣵگُّگْ يࣵلَّ؞
نَڪَ لَ ݧࣺݒّ دِ يࣺيمُ ݖَ مبࣷولࣹمْ لَ عِيسَىٰ دِ دࣹفْ، مُ نࣹ تَالِبࣹيمْ يَ:‏ ‏44 «اُبِّلࣹينْ سࣹينِ نࣷݒّ تࣹ تࣹيوْلُ بُ بَاخْ ڪࣵدُّ يِي:‏ دࣷومُ نِتْ ڪِ لࣹيسْ دِ وࣷرْ، نگِرْ تࣹگْ ڪࣷ ݖِ لࣷخࣷيْ نِتْ ݧِ؞» ‏45 تَالِبࣹ يَ نَگْ دࣺگُّݧُ ڪࣵدُّ گࣷوگُ، ندَخْ فࣹڪّ نَ ݒِرِ مَ لࣵقُ لࣹينْ، نگِرْ ݧُ بَݧْ ڪࣷو دَجْ، تࣹ ݧࣹمࣹوُݧُ ڪࣷو لاَجْ عِيسَىٰ؞
دَرَجَ رࣺيرْ نَ ڪُ ڪࣷ سࣵڪُّ
‏46 وࣹرَنْتࣹ مࣷو دِڪَّلࣷونْ تَالِبࣹ يِ، ݧُيْ سࣹيتْ ڪَنْ مࣷو گࣴنَ ڪَوࣹ ݖِ ݧࣷومْ؞ ‏47 عِيسَىٰ نَگْ رࣵݧّࣹي سࣹينْ خَلاَتْ؞ مُ دَلْدِ اِنْدِ اَبْ خَلࣹ، تَخَوَلْ ڪࣷ ݖِ وࣹتَمْ؞ ‏48 مُ نࣹ لࣹينْ:‏ «خَلࣹ بِي، ڪࣺݒّ ڪُ ڪࣷ دَلَلْ ندَخْ سَمَ تُرْ، مَنْ نگَ دَلَلْ؛ تࣹ ڪࣺݒّ ڪُ مَ دَلَلْ، دَلَلْ نگَ ڪِ مَ يࣹبَلْ؛ ندَخْ ڪِ گࣴنَ تُوتْ ݖِ يࣹينْ ݧࣺݒّ، ڪࣷوڪُ مࣷو گࣴنَ مࣵگّ؞»
‏49 ݖِ ڪَوْ لࣷولُ يࣷوَانْ اࣵدُّ، نࣹ:‏ «نجَاتِگࣹ، دَنࣷو گِسْ نِتْ ڪُيْ دࣵقࣹ سَ تُرْ اَيْ رَبْ، نُ آيࣹ ڪࣷ، ندَخْ بࣷڪُّلْ ݖِ نُنْ؞» ‏50 عِيسَىٰ نࣹ ڪࣷ:‏ «بُلࣹينْ ڪࣷ آيࣹ، ندَخْ ڪُ لَ سࣷتُلْ، يَوْ لَ فَرَلْ؞»
سَمَرِ گࣵنْتَلْ نَ عِيسَىٰ
‏51 بَ ݧُ دࣹمࣹي بَ بࣺسْ بِ ݧُ وَرَ يࣺيگࣹ عِيسَىٰ دِ جُبْسِ، عِيسَىٰ مࣷو دࣷگُ نگِرْ دࣹمْ يࣹرُسَلࣹمْ؞ ‏52 مُ يࣹبَلْ اَيْ ندَوْ، ݧُ جِيتُ ڪࣷ؞ ݧُ دࣹمْ، بَ اَگْسِ اَبْ دࣴڪّ ݖِ دِيوَانُ سَمَرِ، نَرْ ڪࣷ فَا وَاجَلْ اَبْ دَلْ؞ ‏53 وَا سَمَرِ نَگْ نَنگُوُݧُ ڪࣷو دَلَلْ، ندَخْ يࣹرُسَلࣹمْ گَ مُ نَمّࣷونَ جࣴمْ¤؞ ‏54 تَالِبࣹ يَ ݧُيْ وَخْ يَعْقُوبَ اَڪْ يࣷوَانْ گِسْ لࣷولُ، دَلْدِ نࣹ:‏ «سَنگّ بِ، سࣷو بࣴگّࣹي نُ جࣷخْ سَوَرَ ندِگَلْ، مُ وࣵݖّࣹ اَسَمَانْ، فَاگَاگَلْ لࣹينْ؟» ‏55 تࣹيوُلْ عِيسَىٰ گࣹيسُ، فࣹمُّ لࣹينْ؞ ‏56 بَ لࣷولُ اَمࣹي ݧُ جࣵلّ جࣴمْ بࣹنࣹينْ دࣴڪّ؞
گࣴمْڪَتُ عِيسَىٰ دُ خࣷولْ گَنَّاوَمْ
‏57 ݖَ يࣷونْ وَ ݧُ اَوࣹ لَ جࣹنّ وَايْ نࣹ عِيسَىٰ:‏ «يَوْ لاَيْ تࣷݒّ فࣺݒّ فࣷو جࣴمْ؞» ‏58 عِيسَىٰ اِتَمْ نࣹ ڪࣷ:‏ «تِلّ اَڪْ ڪࣵمْبَمْ، نجَنَاوْ اَڪُبْ تࣵگَّمْ، وَايࣹ دࣷومُ نِتْ ڪِ دࣹ، اَمُلْ فُ مُ وࣺيرْ بࣷݒَّمْ؞» ‏59 مُ نࣹيتِ ڪࣹنࣹينْ:‏ «تࣷݒَّلْ ݖِ مَنْ؞» مُ نࣹ ڪࣷ:‏ «سَنگّ بِ، مَيْ مَ، مَ جࣴڪَّ دࣹمْ رࣷبِ سَمَ بَايْ؞» ‏60 مُ نࣹ ڪࣷ:‏ «بࣵيِّلْ نࣺيوْ يِ، ݧُ رࣷبْ سࣹينِ نࣺيوْ، وَايࣹ يَوْ دࣹمَلْ، يࣺگْلࣹجِ نگُورُگْ يࣵلَّ؞» ‏61 ڪࣹنࣹينْ نࣹيتِ ڪࣷ:‏ «سَنگّ بِ، يَوْ لاَيْ تࣷݒّ، وَايࣹ مَيْ مَ، مَ جࣴڪَّ تࣵگّتࣷوڪْ سَمَ وَا ڪࣴرْ؞» ‏62 عِيسَىٰ نࣹ ڪࣷ:‏ «ڪࣺݒّ ڪُ تࣵمْبَلࣹي گࣵبّ تࣹيْ خࣷولْ گَنَّاوَمْ، جࣹڪّࣷو ݖِ نگُورُگْ يࣵلَّ؞»
