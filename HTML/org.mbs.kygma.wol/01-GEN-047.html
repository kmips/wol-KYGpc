يُوسُفَ نُيࣷلࣹ نَ اَيْ بࣷڪَّمْ فِرْعَوْنَ
47
‏1 يُوسُفَ دَا دࣹمْ ݖَ فِرْعَوْنَ، وَخْ ڪࣷ نࣹ ڪࣷ:‏ «سَمَ بَايْ اَڪْ سَمَيْ بࣷڪّ، جٗگࣹ نَݧُ ڪَنَانْ، اࣵندّ اَڪْ سࣹينْ جُرْ گُ گࣵتّ اَڪْ گُ گُدّ اَڪْ مبࣷولࣹمْ لُ ݧُ مࣷومْ؞ ݧَ نگَ لࣺيگِ ݖَ دِيوَانُ گࣷسࣹنْ؞» ‏2 فࣹڪّ نَ يُوسُفَ تࣵنّ جُرٗومْ ݖِيْ دࣷومِ بَايَمْ؞ مُ نُيࣷلࣹ لࣹينْ اَڪْ فِرْعَوْنَ؞ ‏3 فِرْعَوْنَ لاَجْ دࣷومِ بَايْ يَ، نࣹ لࣹينْ:‏ «لَنْ مࣷويْ سࣹينْ لِگّࣺييْ؟» ݧُ نࣹ ڪࣷ:‏ «نُنْ دࣹ سَنگّ بِ، اَيْ سࣵمّ لَنُ نَڪَ جࣹڪّ ݖَ سُنُيْ مَامْ بَ تࣹيْ؞» ‏4 ݧُ تࣹگْ ݖَ نࣹ ڪࣷ:‏ «دَنࣷو ݧࣴوْ نگِرْ دَلْ ݖِ رࣺيوْ مِ، ندَخْتࣹ اَمَلَتُنُ سُنُ جُرْ ݒَرْلُ، تࣹ خِيفْ بِ مࣹتِّ نَ ݖَ رࣺيوُ ڪَنَانْ؞ مࣷو تَخْ سَنگّ بِ، نُ دِ لَ ݧَانْ، نگَ مَيْ نُ، نُ دَلْ فࣷوفَ ݖَ دِيوَانُ گࣷسࣹنْ؞» ‏5 فِرْعَوْنَ وَخْ اَڪْ يُوسُفَ نࣹ ڪࣷ:‏ «گَنَّاوْ سَ بَايْ اَڪْ سَيْ بࣷڪّ ݧࣴوْ نَݧُ فِ يَوْ، ‏6 رࣺيوُمْ مِسْرَا نگِي ݖِ سَ ڪَنَمْ؞ سࣹيتَلْ فِ گࣴنْ ݖِ رࣺيوْ مِ، نگَ جࣷخْ سَ بَايْ اَڪْ سَيْ بࣷڪّ، ݧُ سَنݖّ؞ بࣵيِّ لࣹينْ، ݧُ سَنݖّ گࣷسࣹنْ، تࣹ بُ ݖِ اَمࣹي ݧُ خَرࣹݧْ ݖِ ݧࣷومْ، نگَ دࣺنڪّ لࣹينْ سَمَيْ گࣺتّ؞»
‏7 گَنَّاوْ لࣷولُ يُوسُفَ اࣵندّ اَڪْ بَايَمْ بَ فَ فِرْعَوْنَ، نُيࣷلࣹ ڪࣷ اَڪْ مࣷومْ؞ يَعْقُوبَ نُيُ فِرْعَوْنَ؞ ‏8 فِرْعَوْنَ لاَجْ يَعْقُوبَ نࣹ ڪࣷ:‏ «آ، گٗورْ گِ، خَنَا وَرْ نگَا دࣷونْ مَگْ دࣹ!‏» ‏9 يَعْقُوبَ وَخْ فِرْعَوْنَ نࣹ ڪࣷ:‏ «سَمَيْ اَتِ مࣵنگَانْ، بَ مَ گَنࣹي اࣵدِّنَ بَ تࣹيْ دࣹ، مُ نگِ تࣷلُّ ݖِ تࣺيمࣺيرْ اَڪْ فَنْوࣹيرْ، مُيْ اَتْ يُ گࣵتّ تࣹ نَقَرِ، تࣹ اِتْ دࣹمُمَ بَيْ تࣷلّࣷو اَيْ اَتْ اَڪْ سَمَيْ مَامْ، بَ ݧُيْ مࣵنگَانْ ݖِ اࣵدِّنَ؞» ‏10 بَ مُ ڪࣷ دࣹفࣹي يَعْقُوبَ ݧَانَلْ فِرْعَوْنَ، بࣵيِّڪࣷو فَ، دَلْدِ دࣹمْ؞
‏11 يُوسُفَ نَگْ دَلَلْ بَايَمْ اَڪِ بࣷڪَّمْ، دࣷگَلْ لࣹينْ سُوفْ سَ گࣴنْ ݖَ رࣺيوُمْ مِسْرَ ݖَ دِيوَانُ رَمْسࣹسْ، نِ ڪࣷ ڪࣷ فِرْعَوْنَ سَنْتࣹ وࣷونْ؞ ‏12 بَ لࣷولُ اَمࣹي يُوسُفَ دِ دُنْدَلْ بَايَمْ اَڪِ بࣷڪَّمْ، اَڪْ مبࣷولࣹمْ وَا ڪࣴرْ بَايَمْ، بࣷولࣹ ݖَ ݧَ بࣷوتُ ݖَ ݧࣷومْ؞
لُ جࣴمْ ݖِ سَيْتُبْ يُوسُفَ ݖَ خِيفْ بَ
‏13 اَبْ دُندّ اَمُلࣷونْ فࣹنّ ݖَ رࣺيوْ مَ مࣺݒّ، ندَخْ خِيفْ بُ مࣹتِّ لࣷولْ؞ وَا رࣺيوُمْ مِسْرَ اَڪْ رࣺيوُ ڪَنَانْ جࣹيخْ تࣵڪّ ندَخْ خِيفْ بَ؞ ‏14 يُوسُفَ نَگْ دَجَلࣹي ݖَ ݒࣹݒّ، مَ مُ جَايْ وَا مِسْرَ اَڪْ وَا ڪَنَانْ، مبࣷولࣹمْ خَالِسُ ݧَارِ رࣺيوْ يَ، يٗبُّ نَگْ خَالِسْ بَ ڪࣴرْ فِرْعَوْنَ؞ ‏15 بَ ݧُ دࣹمࣹي، بَ خَالِسْ اَمَتُلْ ݖَ مِسْرَ اَڪْ ݖَ ڪَنَانْ، وَا مِسْرَ گࣺݒّ ݧࣴوْ فَ يُوسُفَ نࣹ ڪࣷ:‏ «جࣷخْ نُ اَبْ دُندّ ڪَيْ!‏ مبَا دُ دَنگَ نُيْ سࣹيتَانْ، نُيْ دࣹي؟ ندَخْ خَالِسْ اَمَتُلْ؞» ‏16 يُوسُفَ نࣹ لࣹينْ:‏ «اِنْدِلࣹينْ سࣹينِ جُرْ، مَ وࣹݖِّ لࣹينْ ڪࣷ دُندّ، ندࣹگَمْ اَمَتُلࣹينْ خَالِسْ؞» ‏17 گَنَّاوْ لࣷولُ ݧُ اِنْدِلْ يُوسُفَ سࣹينِ جُرْ، يُوسُفَ جࣷخْ لࣹينْ ڪࣷ ݒࣹݒّ، وࣹݖِّ لࣹينْ ڪࣷ سࣹينِ فَسْ اَڪْ سࣹينِ گࣺتِّ جُرْ گُ گࣵتّ اَڪْ گُ گُدّ اَڪْ سࣹينِ مبَامْ؞ اَتْ مࣷومَ جࣷخْ نَ لࣹينْ اَبْ دُندّ، مُ وࣹݖِّ سࣹينْ گࣺتّ گࣺݒّ؞ ‏18 بَ اَتْ مَ جࣹيخࣹي، ݧُ ݧࣴوَاتْ ݖَ مࣷومْ ݖَ دࣺوࣺنْ سَ نࣹ ڪࣷ:‏ «سَنگّ بِ، دُنُ لَ نࣴبّ نࣹ اَمَتُنُ دَرَ، ندَخْ خَالِسْ بِ جࣹيخْ نَ، تࣹ جُرْ گَا نگِ ݖِ يَوْ، دَرَ نࣹڪَّتُ فِ لُ نُ لَ مَنَ جࣷخْ لُ مࣷيْ سُنُيْ جࣴمّ اَڪْ سُنُيْ تࣷولْ؞ ‏19 مبَا دُ دَنگَ نُيْ سࣹيتَانْ، نُ بࣷڪّ سࣵنڪُ ݖِ سَ ڪَنَمْ، نࣷوڪْ سُنُ سُوفْ؟ نگَلَّ بࣷولࣹ نُ جࣴنْدࣹ اَبْ دُندّ، نُنْ اَڪْ سُنُيْ سُوفْ؞ سُ ڪࣷ دࣹفࣹي نُيْ مࣷومࣹيلِ فِرْعَوْنَ، نُنْ اَڪْ سُنُيْ سُوفْ؞ تࣹ نگَ جࣷخْ نُ ݒࣹݒّ مُ نُ دُنْدࣹ، لُ ڪࣷ مࣷيْ ڪَتْ نُنْ دِنَنُ دࣹي، سُوفْ سِ اِتْ دِنَ بࣷويْ؞»
‏20 يُوسُفَ نَگْ جࣴنْدَلْ فِرْعَوْنَ سُوفَسْ مِسْرَ گࣺݒّ، نگِرْ وَا مِسْرَ يَ جَايْ ڪُ ݖِ نࣹڪّ سَ تࣷولْ ندَخْ خِيفْ، بَ لࣹينْ جِيتَلْ، بَ فِرْعَوْنَ مُجّ مࣷومْ سُوفْ سِ؞ ‏21 وَا رࣺيوْ مَ ݧࣷومْ، يُوسُفَ دَ لࣹينَ دࣹفْ اَيْ جَامْ ݖَ مِسْرَ، گِي وࣹتْ بَ گَا؞ ‏22 بࣵيِّوُلْ لُ مࣷيْ سُوفُ نجِيتِ دِينࣹ يَ، ندَخْ فِرْعَوْنَ جࣷتࣷونْ نَ لࣹينَ دࣷگَلْ اَيْ ݖࣺرْ، جَگْلࣹيلْ لࣹينْ؞ مࣷو تَخْ جَايُݧُ سࣹينِ سُوفْ؞
‏23 يُوسُفَ نࣹ وَا رࣺيوْ مَ:‏ «لࣺيگِ گَنَّاوْ جࣴندّ نَا لࣹينْ، بَ فِرْعَوْنَ مࣷومْ لࣹينْ، يࣹينْ اَڪْ سࣹينِ سُوفْ، جِوࣷو نگِي، دࣹمْلࣹينْ جِيِ؞ ‏24 بُ نُ اࣵگّࣹي ݖِ نگٗوبْ، جُرٗومْ يُ نگࣹينْ اَمْ دِنگࣹينْ ݖِ جࣷخْ فِرْعَوْنَ بࣹنّ، تࣹ فࣺيتࣹوࣷو ݧࣹينْتِ ݖࣺرْ يِ دࣹسْ؞ سُ ڪࣷ دࣹفࣹي نگࣹينْ گࣺنّࣹ ݖِ جِوُيْ تࣷولْ يِ، لِ ݖِ دࣹسْ نگࣹينْ دُنْدࣹ، يࣹينْ اَڪْ سࣹينْ وَا ڪࣴرْ اَڪْ سࣹينِ دࣷومْ؞» ‏25 ݧُ نࣹ ڪࣷ:‏ «سَنگّ بِ، رَوَلࣹ نگَ نُ!‏ تࣹ گَنَّاوْ لاَيࣹ نگَ نُ بِيرْ، نَنگُ نَنࣷو دࣷونْ جَامِ فِرْعَوْنَ؞» ‏26 نࣷونُ لَ يُوسُفَ تࣴرَلࣹ لࣷولَ، دࣹفْ ڪࣷ دࣷگَلُ يࣷونْ ݖَ سُوفُ مِسْرَ، تࣹ لࣷولࣷويْ وࣺيْ بَ تࣹيْ جِي:‏ جُرٗومِ ݖࣺرِ مࣹݧّࣹيفْ مُ نࣹ، فِرَوْنَايْ مࣷومْ بࣹنّ بَ؞ وَايࣹ سُوفِ نجِيتِ دِينࣹ يَ ݖَ بࣷڪُّلْ، فِرْعَوْنَ اَمُ ݖَ دَرَ؞
لُ جࣴمْ ݖِ اَتِ يَعْقُوبَ يُ مُجّ يَ
‏27 بࣵنِّ إِسْرَاءِيلْ سَنݖّ نَݧُ رࣺيوُمْ مِسْرَ ݖَ دِيوَانُ گࣷسࣹنْ؞ اَمْ نَݧُ فَ اَلَلْ، گِيرْ فَ، بَ يࣷڪُّ لࣷولْ؞
‏28 يَعْقُوبَ نَگْ نࣹڪّ نَ فُڪِّ اَتْ اَڪْ جُرٗومْ ݧَارْ ݖَ رࣺيوُمْ مِسْرَ، اَيْ اَتَمْ يࣺݒّ تࣷلُّ ݖِ تࣺيمࣺيرْ اَڪْ ݧࣹينتّ فُڪّ اَڪْ جُرٗومْ ݧَارْ (147)؞ ‏29 بَ نگَ خَمࣹي نࣹ اࣵݒُّ إِسْرَاءِيلَ نگَيْ بࣴگَّ مَتْ، دَفَ وࣷولُ دࣷومَمْ يُوسُفَ نࣹ ڪࣷ:‏ «نگَلَّ وَايْ دُگَلَلْ سَ لࣷخࣷ، مُ رࣷنْ سَمَ لُݒّ¤، مُيْ نگِݧْ ݖِ يَوْ نࣹ دِنگَ جࣴفْلَنْتࣹيڪْ مَنْ ݖِ نگࣷرْ اَڪْ وࣷرْمَ:‏ رِڪّ، بُ مَ دࣹنݖّ مِسْرَ؞ ‏30 بࣺسْ بُ مَ فࣹڪِّجࣹي سَمَيْ مَامْ، نَنگَ مَ جࣴلࣹ مِسْرَ، دࣹنݖِ مَ ݖَ سࣹينْ خُنْتِمْ بࣵمّࣹيلْ؞» يُوسُفَ نࣹ ڪࣷ:‏ «دِنَا لَ دࣹفَلْ لِ نگَ وَخْ؞» ‏31 مُ نࣹ ڪࣷ:‏ «گِݧَلْ مَ ڪࣷ؞» يُوسُفَ گِݧْ ڪࣷ، إِسْرَاءِيلْ دࣷورَ اَمْ جࣵمّ، دَلْدِ وࣺيرُ ݖَ بࣷݒّ لَلْ بَ؞
