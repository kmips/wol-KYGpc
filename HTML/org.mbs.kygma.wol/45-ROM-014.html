بُلْ سِڪَّلْ سَ مࣷرࣷومْ
14
‏1 ڪُ نگࣴمَمْ دࣴگࣴرُلْ نَگْ، خَجࣷولࣹينْ ڪࣷ، تࣹ بَݧْ ݖࣹي ݖࣵمْبَرْ گِسْ-گِسْ يُ وُوتࣹ؞ ‏2 مُ مࣹلْ نِ تࣹيْ، ڪِي نگࣴمَمْ مࣷو ڪࣷ مَيْ مُ سَݧَ لࣹڪّ لُ نࣹڪّ؛ ڪࣹي، نگࣴمَمْ لُجُمْ دࣷݝّ لَ ڪࣷ نگࣴمَمْ گُ دࣴگࣴرُلْ مَيْ مُ لࣹڪّ؞ ‏3 ڪِيْ لࣹڪّ، بُمُ ڪࣷ خࣹيبࣹ ڪِ لࣹڪُّلْ؛ ڪِ لࣹڪُّلْ اِتْ، بُمُ سِڪّ ڪِيْ لࣹڪّ، ندَخْ يࣵلاَّ ڪࣷ خَجࣷو؞ ‏4 يَوْ مِيْ سِڪَّلْ جَامُبْ جَامْبُرْ، يَايْ ڪَنْ؟ سُ بَاخَلࣹي مبَا مُ بࣷنَلْ، سَنگَمَ ڪࣷيْ سࣹيتْ!‏ تࣹ دَيْ بَاخَلْ ندَخْ بࣷرࣷومْ بِ مَنْ نَ ڪࣷو بَاخَلْ؞
‏5 نَڪَ نࣷونُ ڪِي جࣵݒّ نَ نࣹ بࣺسْ گࣴنْ نَا رَفࣹتْ مࣷرࣷومْ مَ؛ ڪࣹي جࣵݒّ نࣹ بࣺسْ يࣺݒَّ يࣹمْ؞ نَ ڪُ نࣹڪّ سࣹيتْ لَ ڪࣷ دࣷيْ، تࣹ يࣹمْ ݖَ؞ ‏6 ندَخْ ڪِيْ تَامُ بࣺسْ، بࣷرࣷومْ بِ لَ ڪࣷيْ تَامُلْ؛ نَڪَ نࣷونُ ڪِيْ لࣹڪّ، بࣷرࣷومْ بِ لَ ڪࣷيْ گࣴرࣴمࣹ، ندࣹ مࣷومْ لَيْ سَنتّ دࣷورَ لࣹڪّ؛ ڪِ لࣹڪُّلْ اِتْ بࣷرࣷومْ بِ لَيْ سࣵڪّࣷو ݧࣵڪَّ لࣹڪَّمْ، تࣹ يࣵلَّ لَيْ سَنتّ؞ ‏7 ندَخْ ڪَتْ ڪࣹنّ ݖِ نُنْ دُنْدُلْ نگِرْ بࣷݒَّمْ، تࣹ ڪࣹنّ دࣹيوُلْ نگِرْ بࣷݒَّمْ؞ ‏8 سُ نُيْ دُندّ، بࣷرࣷومْ بِ لَنُيْ دُنْدَلْ، سُ نُيْ دࣹي اِتْ، بࣷرࣷومْ بِ لَنُيْ دࣹييَلْ؞
‏9 لࣷولُ سَخْ مࣷو وَرَلْ اَلْمَسِ دࣹي، بَ نࣷݒِّ دࣹڪِّ، دࣹلُّ دِ دُندّ، نگِرْ ݧِ دࣹي اَڪْ ݧِيْ دُندّ ݧࣺݒّ، مُيْ سࣹينْ بࣷرࣷومْ؞ ‏10 يَوْ نَگْ اَنَ لࣷويْ سِڪّࣹ سَ مبࣷڪُّمْ گࣴمْڪَتْ، اَڪْ اِتَمْ، اَنَ لࣷويْ خࣹيبࣹ سَ مبࣷڪُّمْ گࣴمْڪَتْ؟ ندَخْ ڪَتْ نُنْ ݧࣺݒّ اَيْ تَخَوِ فَ ڪَنَمْ جَلُبْ اࣵتّࣹڪَايُ يࣵلَّ؞ ‏11 ندَخْتࣹ بِنْدࣹيسْ نَ نࣹ:‏
«ندࣹگَمْ مَايْ ڪِيْ دُندّ، بࣷرࣷومْ بࣹي ڪࣷ وَخْ،
مَنْ لَ ݧࣺݒّ دِ سُڪَّلْ سࣹينْ بࣴتِ اٗومْ،
تࣹ مَنْ يࣵلَّ لَ لࣵمِّݧُ ݧࣺݒّ دِ سࣵبَّالْ¤؞»
‏12 ڪࣷنْ ڪُ نࣹڪّ ݖِ نُنْ، يَايْ لَيࣷولْ سَ بࣷݒّ فَ يࣵلَّ؞
بُلْ يٗبّࣹ سَ مࣷرࣷومْ بࣵڪَّارْ
‏13 ڪࣷنْ نَگْ بُنُ دِ سࣹيتَتِ لُ نُ سِڪَّنْتࣹي، لِ نگࣹينْ دِ سࣹيتْ ڪَيْ مࣷويْ نُ نگࣹينْ فࣹخࣹي بَ دُنگࣹينْ يٗبّࣹ مبࣷڪُّمْ گࣴمْڪَتْ لُ ڪࣷيْ فَڪّتَلْ، مبَا لُ ڪࣷيْ دَانࣹ ݖِ بࣵڪَّارْ؞ ‏14 خَمْ نَا نَگْ، وࣺرْ نَ مَ اِتْ تࣹ سَنگّ عِيسَىٰ تَخْ، نࣹ اَمُلْ لࣹنّ لُ سࣷبࣹوُ ݖِ وࣵلُّ بࣷݒَّمْ، بَ دَگَنُلَ لࣹڪّ؛ وَايࣹ ڪِ جࣵݒّࣹ لࣹنّ مُيْ لُ سࣷبࣹوُ، ݖِ مࣷومْ لَ لࣷولَ دِ لُ سࣷبࣹوُ؞ ‏15 سࣷو دࣹي لࣹڪّ لࣹنّ بَ تَخْ سَ مبࣷڪّ اَمْ ݖِ لُ ڪࣷ نَقَرِ، نࣷݒِّ نگَا دࣷخࣹ ݖࣷفࣹيلْ؞ بُ سَگْ لࣹڪّ سࣵنڪّ ڪُ اَلْمَسِ دࣹييَلْ؞ ‏16 لِ دَگَنْ ݖِ يَوْ، بُ ڪࣷ جَرِݧࣷو بَ مَيْ ڪࣹنّ لُ مُ ݝࣵݧّࣹي؞ ‏17 ندَخْ ڪَتْ سࣷلࣷبْ نگُورُگْ يࣵلَّ جࣷتࣹوُلْ دَرَ اَڪُگْ لࣹڪّ مبَا اَگْ نَانْ، وَايࣹ سࣷلࣷبْ نگُورُ يࣵلَّ مࣷو دِ نجُبْ اَڪْ جࣵمّ، اَڪْ مبࣺگْتࣹ ݖِ ندِمْبَلَلْ نࣷو گُ سࣹلّ گِ؞ ‏18 نࣷونُ ڪَتْ ڪُ ڪࣷيْ جَامࣷو اَلْمَسِ، يَا نࣹيخْ يࣵلَّ، اَمْ نگࣴرࣴمَلْ نِتْ ݧِ؞
‏19 ڪࣷنْ نَگْ نَنُ سࣵڪُّ لُيْ اِنْدِ جࣵمّ، اَڪْ لُ نُ فࣹدَّلَنْتࣹي؞ ‏20 اَوْ ݧَمْ، بُمُ لَ يࣵقْلࣷو لِگّࣺييُ يࣵلَّ؞ دࣴگّ لَ، وࣺݒّ ݧَمْ سࣹتْ نَ، وَايࣹ لࣹڪّ گَ نگَ ڪࣷيْ لࣹڪّ، سُ گࣵڪَّلࣹي نگࣴمُ ڪࣹنࣹينْ، دࣹفْ نگَ لُ آيْ؞ ‏21 لِ گࣴنْ سَخْ مࣷويْ نگَ بَݧَ لࣹڪّ يࣵݒّ، مبَا نگَ بَݧَ نَانْ بِيݧْ، مبَا لࣹنࣹينْ لُ مُ دࣷونْ لُ مَنَ فَڪّتَلْ سَ مبࣷڪُّمْ گࣴمْڪَتْ؞
‏22 يَوْ ݖِ سَ وࣵلُّ بࣷݒّ، نگࣴمْ گِ نگَ ݖِ اَمْ، يࣹمَلࣹ ڪࣷ ݖِ سَ بࣷݒّ، فِ ڪَنَمْ يࣵلَّ؞ ندࣷڪّلࣹي ڪِ سِڪُّلْ بࣷݒَّمْ لࣹنّ ݖِ جࣴفِنْ وُ مُ دࣷگُلْ بࣷݒَّمْ نࣹ نࣷونُ لَيْ جࣴفࣹ؞ ‏23 وَايࣹ ڪِ اَمْ خࣹلْ-يَارْ، سُ لࣹڪّࣹي، اَبْ دَانْ تࣷݒّ نَ ڪࣷ، ندࣹ جࣴفࣹوُلْ نگࣴمْ، تࣹ لُ نگࣴمْ لَلُلْ بࣵڪَّارْ لَ؞
