لُ جࣴمْ ݖِ بَرْڪࣹ يِ
28
‏1 سُ نگࣹينْ دࣺگَّلࣹي بَ دࣺگَّلْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ، نگِرْ دِ سࣵمّࣷو جࣴفࣹ مبࣷولࣹمْ سَنْتَانࣹيمْ، يِ مَ لࣹينْ دِ دࣺنڪّ تࣹيْ، ڪࣷنْ اَجِ سَخْ جِ سࣹينْ يࣵلَّ، اَجِ ڪَوࣹ جِ مࣷو لࣹينْ دِ اَجْ ݖِ ڪَوْ مبࣷولࣹمْ خࣹيتِ اࣵدِّنَ؞ ‏2 بَرْڪࣹ يِي يࣺݒَّ لࣹينْ دِ دَبْ، سࣷتِّڪُ ݖِ سࣹينْ ڪَوْ، بُ نگࣹينْ دࣺگَّلࣹي سࣹينْ يࣵلَّ اَجِ سَخْ جِ:‏
‏3 بَرْڪࣹيلَ لࣹينْ دِ ݧࣹيلْ ݖِ بِيرْ دࣴڪّ، بَرْڪࣹيلَ لࣹينْ دِ ݧࣹيلْ ݖِبْ اࣵلّ؞
‏4 بَرْڪࣹيلْ اَيْ ݧࣹيلْ سࣹينْ مࣹݧّࣹيفُمْ نجُرُڪَايْ، اَڪْ سࣹينْ مࣹݧّࣹيفُمْ سُوفْ، اَڪْ سࣹينُگْ جُرْ، سࣹينْ جُرْ گُ گُدّ، اَڪْ سࣹينْ جُرْ گُ گࣵتّ؞
‏5 بَرْڪࣹيلْ اَيْ ݧࣹيلْ سࣹينُمْ وِتّ، اَڪْ سࣹينُمْ خِيوْ؞
‏6 بَرْڪࣹيلْ اَيْ ݧࣹيلْ سࣹينْ يࣷونُ دࣹمْ، بَرْڪࣹيلْ اَيْ ݧࣹيلْ سࣹينْ يࣷونُ دِڪّ؞
‏7 اَجِ سَخْ جࣹييْ دࣹفْ، نࣷونْ يِ لࣹينْ دِ جٗگَلْ، ݧࣷويْ دَانُ فِ سࣹينْ ڪَنَمْ؞ وࣹنّ يࣷونْ لَݧُيْ دِڪّࣹ، دَلْ ݖِ سࣹينْ ڪَوْ، وَايࣹ جُرٗومْ ݧَارِ يࣷونْ لَݧُيْ اَوࣹ، دَوْ، جِيتُ لࣹينْ؞
‏8 اَجِ سَخْ جࣹي لࣹينْ دِ بࣷولࣹيڪْ بَرْڪࣹ بُ وࣵݖَّلْ سࣹينُمْ سࣵقْ، اَڪْ سࣹينْ مبࣷولࣹمْ ݧَقْ، تࣹ مࣷو لࣹينْ دِ بَرْڪࣹيلْ ݖَ رࣺيوْ مَ لࣹينْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ جࣷخْ؞ ‏9 مࣷو لࣹينْ دِ دࣹفَلْ بࣷݒَّمْ اَسْڪَنْ وُ سࣹلّ، نࣷونࣹي مُ لࣹينْ ڪࣷ گِݧَلࣹ وࣷونْ، ندࣹگَمْ سࣵمّ نگࣹينْ سࣹينْ سَنْتَانࣹيْ يࣵلَّ اَجِ سَخْ جِ، دِ دࣷخࣹيْ يࣷونَمْ؞ ‏10 سُ بࣷوبَا مبࣷولࣹمْ خࣹيتِ ڪَوْ سُوفْ دِنَݧُ گِسْ نࣹ تُرُ اَجِ سَخْ جِ لࣹيسْ لࣹينْ دِ وࣷويࣹ، تࣹ دِنَݧُ لࣹينْ رَگَلْ؞ ‏11 اَجِ سَخْ جࣹي لࣹينْ دِ خࣺيوَلࣹ، نگِرْ بَاخࣹ سࣹينْ مࣹݧّࣹيفُمْ نجُرُڪَايْ، اَڪْ سࣹينُگْ جُرْ، اَڪْ سࣹينْ مࣹݧّࣹيفُمْ سُوفْ، ݖَ رࣺيوْ مَ مُ گِݧَلࣷونْ سࣹينِ مَامْ، نࣹ مࣷومْ لَ لࣹينْ دِ جࣷخْ؞ ‏12 اَجِ سَخْ جࣹي لࣹينْ دِ اُبِّلْ مبࣵنْدَمْ مُ بَاخْ مَ فَ اَسَمَانْ، بَ بُ جࣷتࣹي مُ تَوَلْ سࣹينُمْ رࣺيوْ، بࣷولࣹ ݖِ بَرْڪࣹيلْ سࣹينْ مبࣷولࣹمْ ݧَقْ؞
يࣹينَيْ لࣹبَلْ خࣹيتْ يُ بَرࣹ، وَايࣹ يࣹينْ دُنگࣹينْ لࣹبْ؞ ‏13 اَجِ سَخْ جࣹي لࣹينْ دِ دࣹفْ بࣷݒّ، تࣹ دُ لࣹينْ دࣹفْ اُبْ گࣹينْ، تࣹ ڪَوْ رࣹڪّ نگࣹينْ دِ جࣴمْ، وَايࣹ دُنگࣹينْ جࣴمْ سُوفْ، ندࣹگَمْ يࣹينَ نگِ تࣷݒّ سࣹينْ سَنْتَانࣹيْ يࣵلَّ اَجِ سَخْ جِ مَ لࣹينْ دِ دࣺنڪَلْ سَمَ بࣷݒّ بࣺسُبْ تࣹيْ، دِ ڪࣷ سࣵمّࣷو جࣴفࣹ، ‏14 تࣹ مبࣷولࣹمْ ڪࣵدُّ يِي مَ لࣹينْ دِ دࣺنڪّ تࣹيْ، نگࣹينْ بَݧْ ڪࣷو مࣷيࣹ ندِجࣷورْ، مبَا ݖࣵمّࣷݧْ، دِ تࣷݒّ يࣹنࣹينْ يࣵلَّ يُ نگࣹينْ دِ جَامُ؞
دࣺگَّدِ، اَلْڪُ
‏15 وَايࣹ نَگْ سُ نگࣹينْ دࣺگَّلُلْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ، تࣹ بَݧَ سࣵمّࣷو جࣴفࣹ مبࣷولࣹمِ سَنْتَانࣹيمْ، اَڪْ دࣷگَلِ يࣷونَمْ، يِ مَ لࣹينْ دِ جࣷخْ تࣹيْ، مبࣷولࣹمْ اَلْڪࣵنْدࣹ يِي دِنَ لࣹينْ دَبْ، سࣷتِّڪُ ݖِ سࣹينْ ڪَوْ:‏
‏16 دِنگࣹينْ اَلْڪࣷو بِيرُبْ دࣴڪّ، اَلْڪࣷو اࣵلّ؞
‏17 دِنگࣹينْ اَلْڪُلࣹ اَمْ وِتّ، اَڪَمْ خِيوْ؞
‏18 دِنگࣹينْ اَلْڪُلࣹ مࣹݧّࣹيفُمْ نجُرُڪَايْ، اَڪْ مࣹݧّࣹيفُمْ سُوفْ، اَڪْ جُرْ گُ گُدّ اَڪْ گُ گࣵتّ؞
‏19 دِنگࣹينْ اَلْڪࣷو سࣹينْ يࣷونُ دࣹمْ، اَلْڪࣷو سࣹينْ يࣷونُ دِڪّ؞
‏20 اَجِ سَخْ جࣹييْ يࣹبَلْ فِ سࣹينْ بِيرْ، ڪُمْتࣹ اَڪْ تِيتَانگࣹ، اَڪْ مبࣹتّࣹيلْ ݖِ مبࣷولࣹمْ لُ نگࣹينْ ݧَقْ، بَ نگࣹينْ يࣵقُ، بَ جࣹڪِّ سࣵنڪُ ندَخْ سࣹينْ جࣴفْ جُ ݧَاوْ جُ نگࣹينْ ڪࣷ دࣴدّࣷو؞
‏21 اَجِ سَخْ جࣹي لࣹينْ دِ لَنگَلَلْ مبَسْ، بَ رَافَلْ لࣹينْ ݖَ رࣺيوْ مَ نگࣹينْ جࣴمْ، نگِرْ نَنگُجِ ڪࣷ؞ ‏22 اَجِ سَخْ جࣹي لࣹينْ دِ دُمَا وࣷݒِّ رࣵگّ، اَڪْ يَرَمْ وُ تࣵنگّ، اَڪْ لَڪّ-لَڪّ، اَڪْ جَنتّ بُ تࣵنگّ جࣺرّ، اَڪْ مَرَلْ، اَڪْ گࣵنݖَخْ گُ خُورْ، تࣹگِ لِيرْ، لࣺݒّ تࣷݒّ لࣹينْ بَ نگࣹينْ سࣵنڪُ؞
‏23 اَسَمَانْ سِ تِيمْ سࣹينْ بࣷݒّ نَگْ دَ نَانْ سࣹرࣹݝْ نِ خࣵنجَرْ، سُوفْ سِ نگࣹينْ تࣹگْ سࣹينِ تࣵنڪّ، وࣷوْ ڪࣷݝّ نِ وࣹݧْ گُ ݧُولْ؞ ‏24 اَجِ سَخْ جࣹييْ دࣹفْ سࣹينْ تَوُبْ رࣺيوْ دِ سُوفْ اَڪْ ݒࣴندّ بُ بَوࣷو اَسَمَانْ، دِ يُرِڪُ ݖِ سࣹينْ ڪَوْ، بَ نگࣹينْ سࣵنڪُ؞
‏25 اَجِ سَخْ جࣹييْ دࣹفْ نگࣹينْ دَانُ فِ سࣹينْ ڪَنَمِ نࣷونْ؞ وࣹنّ يࣷونْ نگࣹينْ دِ دِڪّࣹ، سࣷنگّ لࣹينْ، وَايࣹ جُرٗومْ ݧَارِ يࣷونْ نگࣹينْ دِ اَوࣹ، دَوْ، تࣹ دࣹيسْ نَ لࣹينْ سࣹيخْلُ ݖِ مبࣷولࣹمْ نگُورِ ڪَوْ سُوفْ؞ ‏26 سُ ڪࣷ دࣹفࣹي سࣹينِ نࣺيوْ دِ ݒࣷرَاتُمْ مبࣷولࣹمْ نجَنَاوْ اَڪْ رَبُ اࣵلّ، تࣹ ڪࣹنّ دُ لࣹينْ دࣵقْ؞
‏27 اَجِ سَخْ جࣹي لࣹينْ دِ دُمَا تَابِ مِسْرَ، اَڪْ نࣹيوࣷيْ ڪࣴرُگْ سُوفْ، اَڪْ يَرَمْ وُ خَسْ، اَڪْ اࣴرْ يُيْ خَسَنْ تࣹ دُنگࣹينْ مَنَ اَمْ ݒَجَمْ؞ ‏28 اَجِ سَخْ جࣹي لࣹينْ دِ دُمَا اَگْ ندࣷفْ اَڪُگْ ݖِلْمَخَ اَڪْ نگࣴلࣴمْتࣹ، ‏29 بَ نگࣹينْ دِ لࣵمْبَتࣷو دِگّ بࣴݖّࣴگْ نِ سِلْمَخَ دِ لࣵمْبَتࣷو ݖِگْ لࣴنْدࣴمْ؞ دُنگࣹينْ سࣷتَّلْ سࣹينِ مࣺبࣺتْ؞ دࣹيسْ لࣹينْ دِ نࣺيوَلْ دࣷولࣹ، دِ فࣷقَتِ سࣹينْ اَلَلْ رࣹڪّ، بࣺسْ بُ نࣹڪّ، تࣹ ڪࣹنّ دُ لࣹينْ وَلُّ؞
‏30 اَسْ ندَوْ، نگࣹينْ دِگࣷولْ اَبْ سࣺيْ، ڪࣹنࣹينْ تࣴدّࣹ ڪࣷ؛ اَگْ ڪࣴرْ، نگࣹينْ تَبَخْ، تࣹ دُنگࣹينْ ڪࣷ دࣴڪّࣹ؛ اَبْ تٗوڪࣴرُ رࣹسࣹݧْ، نگࣹينْ جࣴمْبَتْ، تࣹ دُنگࣹينْ ڪࣷ جَرِݧࣷو؞ ‏31 سࣹينُ يࣴڪّ، ݧُ رࣹنْدِ، نگࣹينْ گِسْ، تࣹ دُنگࣹينْ ݖِ لࣹڪّ؛ سࣹينْ مبَامْ، ݧُ نَنگُ ڪࣷ ݖِ سࣹينِ لࣷخࣷ، تࣹ دُ دࣺلْسِ ݖِ يࣹينْ؛ سࣹينِ گࣵتّ، ݧُ جࣺبَّلْ سࣹينِ نࣷونْ، تࣹ ڪࣹنّ دُ لࣹينْ وَلُّ؛ ‏32 سࣹينْ دࣷومْ يُ گٗورْ اَڪْ يُ جِگࣺينْ، ݧُ جࣺبَّلْ وࣹنࣹينْ خࣹيتْ ݖِ سࣹينْ ڪَنَمْ، نگࣹينْ دِ لࣹينْ سࣺينْتُ ݖِ نࣹينْ بࣺسْ بُ يࣵلَّ سࣵڪّ، تࣹ دُنگࣹينْ ݖِ مَنْ دَرَ؞ ‏33 سࣹينْ مࣹݧّࣹيفُمْ سُوفْ، اَڪْ مبࣷولࣹمْ سࣹينُ ݧَقْ، خࣹيتْ وُ نگࣹينْ خَمُلࣷونَ ڪࣷيْ لࣹڪّ، تࣹ دࣹيسْ لࣹينْ دِ نࣷتْ اَڪَ دࣴگَّاتࣹ رࣹڪّ، بࣺسْ بُ نࣹڪّ؞ ‏34 دِنگࣹينْ مُجّ دࣷفْ ندَخْ لُ بُورْ سࣹينِ بࣴتْ نگࣹينْ دِ گِسْ؞
‏35 اَجِ سَخْ جࣹي لࣹينْ دِ دُمَا تَابِ اٗومْ، اَڪْ تَابِ يࣹيلْ يُ بࣷنْ؞ دُنگࣹينْ مَنَ اَمْ ݒَجَمْ، تࣹ ݖِ سࣹينْ وࣹوُ تࣵنڪّ لَيْ لَوࣹ، بَ ݖِ سࣹينْ ڪَوْ بࣷݒّ؞
‏36 اَجِ سَخْ جࣹي لࣹينْ دِ دِرِ، يࣹينْ اَڪْ سࣹينْ بُورْ بِ نگࣹينْ فَلْ، بَ ݖَ بِيرْ خࣹيتْ وُ نگࣹينْ خَمُلࣷونْ، يࣹينْ اَڪْ سࣹينِ مَامْ، نگࣹينْ دِ فَ جَامُ يࣹنࣹينْ يࣵلَّيْ بَنتّ اَڪْ دࣷجْ؞ ‏37 دࣹيسْ نَ لࣹينْ مُجّ سࣹيخْلُ، دِ لࣹينْ لࣺيبࣷوڪَ ڪٗڪَّلࣹي فࣷوفَ ݖَ بِيرْ مبࣷولࣹمْ خࣹيتْ، يَ مُ لࣹينْ دࣵقْ جࣴمࣹ؞
‏38 جِوُ مُ بَرࣹ نگࣹينْ دِ يٗبُّ تࣷولْ، تࣹ لُ نࣺيوْ نگࣹينْ ݖَيْ اَمࣹ، ندَخْ نجࣺيرࣺيرْ يࣹي ڪࣷيْ وَتْ؞ ‏39 اَيْ تٗوڪࣴرِ رࣹسࣹݧْ، نگࣹينْ جࣴمْبَتْ، سࣷنّ ݖَ، تࣹ بِيݧْ، دُنگࣹينْ ڪࣷ نَانْ، دُنگࣹينْ وِتّ دࣷومْ يَ اِتْ، ندَخْ سَخْ يࣹي ڪࣷيْ لࣹڪّ؞ ‏40 اَيْ گَرَبِ اࣷلِوْ نگࣹينْ دِ اَمْ ݖِ سࣹينْ رࣺيوْ مࣺݒّ، وَايࣹ اَگْ دِوْ، دُنگࣹينْ ڪࣷ دِوࣷو، ندَخْ سࣹينْ دࣷومِ اࣷلِوْ يَ دَيْ يࣷلَخْ؞ ‏41 اَيْ دࣷومْ يُ گٗورْ اَڪْ يُ جِگࣺينْ نگࣹينْ دِ جُرْ، وَايࣹ دُݧُ لࣹينْ نࣹڪَّلْ، ندَخْ نجَامْ لَݧُيْ دُگِّ؞ ‏42 سࣹينْ گࣺݒّ گَرَبْ اَڪْ سࣹينْ مࣹݧّࣹيفُمْ سُوفْ، سࣷݖّࣹتْ يࣹي ڪࣷيْ نَنگُ؞
‏43 دࣷخَنْدࣺيمْ بُ نࣹڪّ ݖِ سࣹينْ بِيرْ مࣷويْ يࣺيگْ تِيمْ لࣹينْ، بَ ݖَ ڪَوَ ڪَوْ، يࣹينْ نَگْ نگࣹينْ دِ وࣵݖّ، بَ سُوفࣹي سُوفࣹ؞ ‏44 مࣷومْ مࣷو لࣹينْ دِ لࣹبَلْ، وَايࣹ دُنگࣹينْ ڪࣷ لࣹبَلْ؞ مࣷومْ مࣷويْ دࣷونْ بࣷݒّ، يࣹينْ، نگࣹينْ دِ گࣹينْ؞
‏45 مبࣷولࣹمْ اَلْڪࣵنْدࣹ يِي ݧࣷو لࣹينْ دِ دَبْ، دَلْ ݖِ سࣹينْ ڪَوْ، تࣷݒّ لࣹينْ، بَ نگࣹينْ سࣵنڪُ، ندَخْ يࣹينَ دࣺگَّلُلْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ، نگِرْ وࣷرْمَالْ اَيْ سَنْتَانࣹيمْ، اَڪْ دࣷگَلِ يࣷونَمْ، يِ مُ لࣹينْ دࣺنڪّ؞ ‏46 لࣷولُ مࣷويْ دࣷونْ، ݖِ سࣹينْ بِيرْ اَڪْ سࣹينْ بِيرْ اَسْڪَنْ، بَ فࣵوّ، اَيْ فِرْندࣹ اَڪِ ڪࣺيمَانْ يُ لࣹينْ دِ اࣵرْتُ؞
‏47 گَنَّاوْ يࣹينَ جَامُوُلْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ ݖِ مبࣺگْتࣹ اَڪْ خࣷلْ بُ سࣹدّ، ڪࣹرࣷوگْ بَ نگࣹينْ بَرࣹلࣹي لࣺݒّ، ‏48 سࣹينْ نࣷونْ يِ اَجِ سَخْ جِيْ يࣹبَلْ ݖِ سࣹينْ بِيرْ، ݧࣷومْ نگࣹينْ دِ جَامُجِ ݖِ بِيرْ خِيفْ، اَڪْ مَرْ اَڪْ رَفْلࣹ، اَڪْ گࣺݒّ ݧࣵڪّ؞ اَجِ سَخْ جࣹييْ تࣹگْ يࣹتُ وࣹݧْ ݖِ سࣹينْ ندࣷدّ، بَ سࣵنڪّ لࣹينْ؞ ‏49 اَجِ سَخْ جࣹييْ جٗگْلࣷو ݖِ سࣹينْ ڪَوْ، خࣹيتْ وُ سࣷرࣹ، ݧُ بࣵيِّڪࣷو ݖَ ݖَتِ اࣵدِّنَ، دِڪّ نࣹ مِلِبْ نِگْ جَخَايْ، تࣹ دࣺگُّلࣹينْ اَوْ لࣵڪَّمْ، ‏50 خࣹيتْ وُ ݧࣵنگّ ڪَنَمْ تࣹ رُسُݧُ مَگْ، يࣴرࣴمُݧُ گࣷنࣹ؞ ‏51 ݧࣷويْ يࣵݒّ سࣹينُگْ جُرْ، لࣹڪّ سࣹينْ گࣵنݖَخْ، بَ نگࣹينْ رَافْ؞ دُݧُ لࣹينْ وَݖّࣹ دَرَ؛ دُ ݒࣹݒّ، دُ بِيݧْ، دُگْ دِوْ، دُ دࣷومُ جُرْ گُ گُدّ اَڪْ گُ گࣵتّ، بَ نگࣹينْ فَرْ سࣵنڪُ؞ ‏52 دࣹيسْ نَ لࣹينْ گَوࣹ ݖِ سࣹينْ دࣴڪّ يࣺݒّ، بَ سࣹينْ تَتَ يُ ڪَوࣹ تࣹ دࣵبّلِڪُ، يَ نگࣹينْ يَاڪَارࣷونْ ݖِ سࣹينْ رࣺيوْ مࣺݒّ، مࣵبّ؞ دࣹيسْ نَ لࣹينْ گَوࣹ سࣹينْ دࣴڪّ يࣺݒّ، ݖِ مبࣷولࣹمْ رࣺيوْ، مِ لࣹينْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ جࣷخْ؞ ‏53 دِنگࣹينْ مُجّ لࣹڪّ سࣹينْ مࣹݧّࣹيفُمْ نجُرُڪَايْ، سࣹينْ سُوخُ دࣷومْ يُ گٗورْ اَڪْ يُ جِگࣺينْ، يِ لࣹينْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ مَيْ، ندَخْ گَوْ بَ، اَڪْ تَنݖّ بُ لࣹينْ سࣹينِ نࣷونْ تَنݖّ؞ ‏54 گٗورْ گِ گࣴنَ لࣹوࣹتْ ݖِ يࣹينْ تࣹ خَيْ لࣷولْ دِنَ خࣹيلُ مبࣷڪَّمْ، خࣹيلُ جَبَرْ جُ مُيْ نࣷيّࣹي، اَڪْ ندࣹسُ دࣷومَمْ جُ مُ دࣹسࣹ، ‏55 نگِرْ بَݧَ سࣺدّ ڪࣹنّ ݖِ ݧࣷومْ ݖِ يࣵݒُّ دࣷومَمْ يَ مُيْ لࣹڪّ، ندَخْ دُ دࣹسࣹ دَرَ ݖَ بِيرْ گَوْ بَ، اَڪْ تَنݖّ بَ لࣹينْ سࣹينِ نࣷونْ تَنݖّ، ݖِ سࣹينْ دࣴڪّ يࣺݒّ؞ ‏56 جِگࣺينْ جُ گࣴنَ لࣹوࣹتْ تࣹ خَيْ ݖِ يࣹينْ، بَ نࣺيگُلَ تࣹگْ ندࣴگُّلْ تࣵنڪَمْ ݖِ سُوفْ ندَخْ خَيْتࣹيمْ اَڪْ لࣹوࣹتَايَمْ، دِنَ خࣹيلُ جࣴڪّࣴرَمْ جُ مُيْ نࣷيّࣹي، اَڪْ دࣷومَمْ جُ گٗورْ اَڪْ جُ جِگࣺينْ؞ ‏57 دَيْ فࣹڪّ مُ نَمَّ لࣹڪّ ݖِ ڪُمْݒَ دࣷومْ جُ مُ دࣷورَ وَسِنْ، بࣷولࣹ ڪࣷوڪْ اࣵندّ بَ، ندَخْ لࣺݒّ اَيْ ݧࣵڪّ ݖَ بِيرْ گَوْ بَ اَڪْ تَنݖّ، بَ لࣹينْ سࣹينِ نࣷونْ تَنݖّ ݖَ سࣹينِ دࣴڪّ؞
‏58 سُ نگࣹينْ سࣵمّࣷووُلَ جࣴفࣹ مبࣷولࣹمْ ڪࣵدُّيْ يࣷونْ وِي بِنْدࣹ ݖِ تࣺيرࣹ بِي، نگِرْ وࣷرْمَالْ بࣷرࣷومْ تُرْ وُ مࣵگّ وِي تࣹ رَگْلُ، سࣹينْ يࣵلَّ اَجِ سَخْ جِ، ‏59 سُ بࣷوبَا اَجِ سَخْ جِ دِنَ رࣺيَلْ سࣹينِ دُمَ، اَڪْ سࣹينْ دُمَيْ اَسْڪَنْ، دُمَ يُ مَگْ تࣹ يࣵگّ، اَڪْ وࣷݒّ يُ آيْ تࣹ اࣴسْ؞ ‏60 مࣷويْ دࣺلّࣷوسِ فِ سࣹينْ ڪَوْ، مبࣷولࣹمْ جࣵنگࣷرࣷيْ مِسْرَ يُ نگࣹينْ رَگَلࣷونْ، ݧُ تَقْ لࣹينْ؞ ‏61 تࣹ اِتْ مبࣷولࣹمْ وࣷݒّ اَڪْ مبࣷولࣹمْ مبُگَلْ مُ بِنْدࣹوُلْ ݖِ تࣺيرࣹبْ يࣷونْ وِي، اَجِ سَخْ جࣹي لࣹينْ ڪࣷيْ يࣹنْ، بَ نگࣹينْ سࣵنڪُ؞ ‏62 نِتْ ݧُ نࣺيوْ نگࣹينْ دِ دࣹسࣹ، تࣹ بَرࣹ وࣷونْ نِ بِدِّيوِ اَسَمَانْ، ندَخْ ݧࣵڪَّ دࣺگَّلْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ؞ ‏63 سُ ڪࣷ دࣹفࣹي نࣷونࣹي اَجِ سَخْ جِ دَانْ بࣺگࣹ دِ لࣹينْ بَاخࣹ اَڪَ يࣷڪّ، نࣷونُ لَ اَجِ سَخْ جِيْ بࣺگࣹ دِ لࣹينْ سࣵنڪّ اَڪَ فࣵڪَّسْ، تࣹ دࣹيسْ نَ لࣹينْ دࣺجَّتࣹي ݖَ رࣺيوْ مَ نگࣹينْ جࣴمْ، نگِرْ نَنگُجِ ڪࣷ؞ ‏64 اَجِ سَخْ جࣹي لࣹينْ دِ وَسَارࣹ ݖِ بِيرْ مبࣷولࣹمْ خࣹيتِ اࣵدِّنَ، ݖَتْ بَ ݖَتْ، نگࣹينْ دِ فَ جَامُ يࣹنࣹينْ يࣵلَّ يُ نگࣹينْ خَمُلࣷونْ، دُ يࣹينْ دُ سࣹينِ مَامْ، يࣵلَّيْ بَنتّ اَڪْ دࣷجْ؞ ‏65 ݖَ بِيرْ خࣹيتْ يࣷويُ نَگْ، دُنگࣹينْ فَ اَسْتَنْدِڪࣷو، دُنگࣹينْ فَ اَمْ سَخْ فُ نگࣹينْ نࣷݒَّلࣹ سࣹينْ تࣺسْتࣴنْ؞ اَجِ سَخْ جࣹي لࣹينْ فَيْ تࣹگْ فِتْ وُ رࣴݖّ، اَڪْ گࣴتْ يُ گِيمْ، اَڪْ خࣷلْ بُ جࣹيخْ؞ ‏66 سࣹينْ بَڪَّنْ اَيْ نࣹڪّ ݖِ خࣷتّ، نگࣹينْ دࣴڪّࣹ تِيتَانگࣹيْ گُدِّ اَڪْ بࣴݖّࣴگْ، تࣹ سࣹينْ اࣴلّࣴگْ دࣷوتُ لࣹينْ وٗورْ؞ ‏67 اَگْ سُبَ نگࣹينْ نࣹ:‏ «اࣺيْ وَايْ، بُ دࣷونْ نگࣷونَتِ؞» بُ نگࣷونࣹي نگࣹينْ نࣹ:‏ «اࣺيْ وَايْ، بُ دࣷونْ سُبَاتِ!‏»، ندَخْ تِيتَانگࣹ لُ نگࣹينْ دِ تِيتْ، اَڪْ لِ نگࣹينْ دِ تࣹگْ سࣹينِ گࣴتْ؞ ‏68 اَجِ سَخْ جࣹي لࣹينْ دِ يࣹبْ ݖِيْ گَالْ، دࣹلّࣷو لࣹينْ مِسْرَ، يࣷونْ وُ مَ لࣹينْ نࣷونْ دُنگࣹينْ ڪࣷ گِسَتِ، نگࣹينْ دِ فَ جَايْ سࣹينْ بࣷݒّ سࣹينِ نࣷونْ، نگِرْ دࣷونْ سࣹينْ جَامْ يُ گٗورْ اَڪْ يُ جِگࣺينْ، تࣹ ڪࣹنّ دُ لࣹينْ جࣴندّ؞

‏69 لِي مࣷويْ سࣵرْتِ ڪٗلّࣴرࣹ، گِ اَجِ سَخْ جِ سَنْتࣷونْ مُوسَىٰ نگِرْ مُ فَسࣷو ڪࣷ اَڪْ بࣵنِّ إِسْرَاءِيلْ، ݖَ رࣺيوُمْ مࣷوَبْ، تࣹ بࣷڪّࣹوُلْ اَڪْ ڪٗلّࣴرࣹ گَ مُ فَسࣷو وࣷونْ اَڪْ ݧࣷومْ ݖَ تُنْدُ اࣷرࣹبْ؞
