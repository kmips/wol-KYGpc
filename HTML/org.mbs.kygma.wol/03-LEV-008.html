فَلْ نَݧُ سَرْخَلْڪَتْ يِ
(سَارْ 8—10)
لُ جࣴمْ ݖِ خࣹوُ ݖࣷلُگْ سَرْخَلْڪَتْ يَ
8
‏1 اَجِ سَخْ جِ وَخَتِ مُوسَىٰ نࣹ ڪࣷ:‏ ‏2 «اِنْدِلْ هَـٰرُونَ اَڪِ دࣷومَمْ يُ گٗورْ تࣹ اِنْدَالࣹ يࣺرࣹ يِ اَڪْ دِوْ گِ ݧُيْ فَلࣹ اَڪْ يࣴڪّ وِيْ سَرَخُ ݒٗوتُمْ بࣵڪَّارْ اَڪْ ݧَارِ ڪُويْ يِ، بࣷولࣹ ݖَ ݒَݧࣹبْ مبُرُ يَ اَمُلْ لَوِيرْ، ‏3 تࣹ نگَ وࣷو مبࣷولࣷو مِ مࣺݒّ، ݧُ دَجࣹ ݖِ بُنتّ خَيْمَبْ ندَجࣹ مِ؞»
‏4 مُوسَىٰ دࣹفْ لَ ڪࣷ اَجِ سَخْ جِ سَنتّ، مبࣷولࣷو مَ دَجࣹ ݖَ بُنتّ خَيْمَبْ ندَجࣹ مَ؞ ‏5 مُوسَىٰ وَخْ مبࣷولࣷو مَ نࣹ لࣹينْ:‏ «لِي مَيْ وَاجْ لَ اَجِ سَخْ جِ سَنْتَانࣹ وࣷونْ ݧُ دࣹفْ ڪࣷ؞»
‏6 ݖِ ڪَوْ لࣷولُ مُوسَىٰ اِنْدِ هَـٰرُونَ اَڪِ دࣷومَمْ يُ گٗورْ، ݧُ سَنگُ، ‏7 سࣷلَلْ هَـٰرُونَ مبُبّ مُ گُدّ مَ، تَڪَّلْ ڪࣷ لَخَسَايْ گَ، سࣷلَلْ ڪࣷ فࣴخْيَ بَ، تࣹگَلْ ڪࣷ ݖَ خَرْ-سࣵنِّ مَ، تَڪَّلْ ڪࣷ نگَݧَايْ لَ ݖَ ڪَوَمْ، جࣵݒّࣹ ڪࣷ ڪࣷ، مُ تَفُ ݖَ ڪَوَمْ؞ ‏8 مُ سࣷلَلْ ڪࣷ نَگْ ڪِيرَايَلْ دࣴنّ بَ، دَلْدِ يࣹبْ ݖَ بِيرْ ڪِيرَايْ لَ جُمْتُڪَايِ تࣹگْتَلْ يَ، دِ اُرِمْ بَ اَڪْ تُمِمْ بَ¤، ‏9 لَ ݖَ تࣹگُ مُ تࣹگَلْ ڪࣷ ڪَالَ گَ ݖَ بࣷݒّ بَ، تَڪَّلْ ڪࣷ دࣷگُ وُرُسْ وُ تࣹلّ وَ، مࣷويْ مࣹيتࣹيلْ گُ سࣹلّ گَ، مُ نࣹڪّ ݖَ ڪَوْ ڪَالَ گَ، فࣺيتࣹ ڪَنَمْ، مُيْ لَ اَجِ سَخْ جِ سَنْتࣷونْ مُوسَىٰ؞
‏10 بَ لࣷولُ وࣺيࣹي مُوسَىٰ جࣴلْ دِوُ ݒَلْ گَ، دِوْ ݖَ جَامࣷوڪَايْ بَاڪْ يَ ݖَ بِيرْ يࣺݒّ، سࣹلَّلࣹ ڪࣷ يࣷويَ، ‏11 بَ نࣷݒِّ وِسْ-وِسَلْ ݖَ جُرٗومْ ݧَارِ يࣷونْ ݖَ ڪَوْ سَرْخَلُڪَايْ بَ، دِوࣹ ڪࣷ سَرْخَلُڪَايْ بَ اَڪْ مبࣷولࣹمْ اَيْ ندَبَمْ، بࣷولࣹ ݖَ مبَلْڪَمْ نجࣵݒّ مَ، مࣷوڪْ اُبْ تࣹگࣷومْ نگِرْ يࣷويُ اِتْ سࣹلّ؞ ‏12 مُ سࣷتِّ نَگْ تُوتِ ݖِ دِوْ گِ ݧُيْ فَلࣹ ݖِ بࣷݒُّ هَـٰرُونَ، دِوْ ڪࣷ، سࣹلَّلࣹ ڪࣷ ڪࣷ؞ ‏13 بَ مُ ڪࣷ دࣹفࣹي مُوسَىٰ اِنْدِ دࣷومِ هَـٰرُونَ يُ گٗورْ يَ، سࣷلَلْ لࣹينْ مبُبّ يُ گُدّ، تَڪَّلْ لࣹينْ اَيْ لَخَسَايْ، سࣷلَلْ لࣹينْ مبَخَنَ، مُيْ لَ اَجِ سَخْ جِ سَنْتࣷونْ مُوسَىٰ؞
‏14 مُوسَىٰ تࣹگْ ݖَ اِنْدِ يࣴڪُّ سَرَخْ سِيْ ݒٗوتُمْ بࣵڪَّارْ، هَـٰرُونَ اَڪِ دࣷومَمْ تࣹگْ سࣹينِ لࣷخࣷ ݖِ ڪَوْ بࣷݒُّ يࣴڪُّ سَرَخْ سِيْ ݒٗوتُمْ بࣵڪَّارْ، ‏15 مُوسَىٰ رࣹنْدِ ڪࣷ، سࣵڪّ ݖِ دࣹرࣹتْ جِ، ݖَݒّ ݖِ بَارَامَمْ، دِوْ ݖِ ݧࣹينْتِ بࣺجّࣺنِ سَرْخَلُڪَايْ بِ يࣺݒّ، سࣹتَلࣹ ڪࣷ سَرْخَلُڪَايْ بِ؞ مُ دَلْدِ تُورْ لِ دࣹسْ ݖِ دࣹرࣹتْ جِ ݖِ تَاتُ سَرْخَلُڪَايْ بِ، سࣹلَّلࣹ ڪࣷ نࣷونُ، مُيْ نجࣷتْلاَيَلْ سَرْخَلُڪَايْ بِ؞
‏16 بَ لࣷولُ اَمࣹي مُ جࣴلْ نࣹبّࣷنْ بِ سࣵنگّ يࣺرࣹيْ بِيرْ يِ، اَڪْ بࣵجّࣷ بِ ݖِ رࣹسْ وِ اَڪْ ݧَارِ دࣴمْبࣺينْ يِ اࣵندّ اَڪْ سࣹينْ نࣹبّࣷنْ، لَڪّ ڪࣷ ݖِ ڪَوْ سَرْخَلُڪَايْ بِ؞ ‏17 لِ دࣹسْ ݖِ يࣴڪّ وِ، دِ دࣹرْ بࣹيڪْ يࣵݒّ وࣹيڪْ سࣺبّرِيتْ مِ، مُ گࣺنّࣹ ڪࣷ دَلْ بِ، لَڪّ ڪࣷ ݖَ بِتِ، مُيْ لَ اَجِ سَخْ جِ سَنْتࣷونْ مُوسَىٰ؞
‏18 لَ ݖَ تࣹگُ مُ اِنْدِ ڪُويُ سَرَخُ رࣹنْدِ-دٗومَلْ بِ، هَـٰرُونَ اَڪْ دࣷومَمْ يُ گٗورْ تࣹگْ سࣹينِ لࣷخࣷ ݖَ بࣷݒُّ ڪُويْ مَ؞ ‏19 مُوسَىٰ رࣹنْدِ ڪࣷ، خࣴݒّ دࣹرࣹتْ جِ ݖِ مبࣷولࣹمْ وࣹتِ سَرْخَلُڪَايْ بِ، ‏20 دَلْدِ دَگَّتْ ڪُويْ مِ، دࣹفْ ڪࣷ اَيْ دࣷگْ، بࣷولࣹ بࣷݒّ بࣹيڪْ دࣷگْ يࣹيڪْ نࣹبّࣷنْ بِ، لَڪّ ڪࣷ، بَ مُ دِبْ دٗومْ؞ ‏21 بَ لࣷولُ وࣹيسࣹي مُوسَىٰ بࣷولࣹ يࣺرࣹيْ بِيرْ يِ، اَڪْ يࣹيلْ يِ، رَخَسْ، بَ نࣷݒِّ لَڪّ ڪُويْ مِ يࣺݒّ ݖَ سَرْخَلُڪَايْ بَ؞ لࣷولُ سَرَخُ رࣹنْدِ-دٗومَلْ لَ، نگِرْ خࣹتُگْ جࣵمّ؞ سَرَخُ سَوَرَ لَ، ݧࣹيلْ اَجِ سَخْ جِ، نَ ڪࣷ اَجِ سَخْ جِ سَنْتࣹ وࣷونْ مُوسَىٰ؞
‏22 گَنَّاوْ لࣷولُ مُوسَىٰ اِنْدِ ڪُويْ مَ ݖَ دࣹسْ، دِ ڪُويُ خࣹوُ ݖࣷلُ؞ هَـٰرُونَ اَڪِ دࣷومَمْ تࣹگْ سࣹينِ لࣷخࣷ ݖِ بࣷݒُّ ڪُويْ مِ، ‏23 مُوسَىٰ رࣹنْدِ ڪࣷ، سࣵڪّ ݖَ دࣹرࣹتْ جَ، تَقَلْ ݖَ تَبَنُ نࣷݒُّ ندِجࣷورُ هَـٰرُونَ، تَقَلْ ݖَ بَارَامُ دࣺيُ ندِجࣷورَمْ اَڪْ بَارَامُ دࣺيُ تࣵنڪُ ندِجࣷورَمْ؞ ‏24 مُوسَىٰ دࣷورَ اِنْدِ دࣷومِ هَـٰرُونَ يُ گٗورْ يَ، سࣵڪّ ݖَ دࣹرࣹتْ جَ، تَقَلْ ݖَ سࣹينْ تَبَنِ نࣷݒِّ ندِجࣷورْ اَڪْ سࣹينْ بَارَامُ دࣺيُ ندِجࣷورْ اَڪْ سࣹينْ بَارَامُ دࣺيُ تࣵنڪِ ندِجࣷورْ؞ لَ دࣹسْ ݖَ دࣹرࣹتْ جَ، مُوسَىٰ خࣴݒّ ڪࣷ ݖَ مبࣷولࣹمْ وࣹتِ سَرْخَلُڪَايْ بِ؞ ‏25 بَ مُ ڪࣷ دࣹفࣹي مُ گࣺنّࣹ نࣹبّࣷنْ بِ، مُيْ ݖَلْگࣹينْ بِ اَڪْ نࣹبّࣷنْ بِ سࣵنگّ يࣺرࣹيْ بِيرْ يࣺݒّ اَڪْ بࣵجّࣷ بِ ݖِ رࣹسْ وِ اَڪْ ݧَارِ دࣴمْبࣺينْ يِ اࣵندّ اَڪْ سࣹينْ نࣹبّࣷنْ، اَڪْ تࣵنڪُبْ ندِجࣷورْ بِ؞ ‏26 لَيُ بِ دࣹفْ مبُرُ يِ اَمُلْ لَوِيرْ، تَاجࣹ فِ ڪَنَمْ اَجِ سَخْ جِ، مُ جࣴلࣹ ݖَ مࣹنّ مبُرُ ڪࣹسࣹ مُ ندَوْ اَڪْ مࣹنّ مبُرُ مُ ندَوْ تࣹ اَمْ دِوْ اَڪْ مࣹنّ مبُرُ مُ تࣵݒَّنْدَارْ، بࣷولࣹ ڪࣷ تࣹگْ ݖَ ڪَوْ نࣹبّࣷنْ بَ اَڪْ تࣵنڪُ ندِجࣷورْ بَ؞ ‏27 مُ بࣷولࣹ يࣷويُ يࣺݒّ تࣹگْ ݖِ لࣷخࣷلْ هَـٰرُونَ اَڪْ لࣷخࣷيْ دࣷومَمْ يُ گٗورْ، نگِرْ ݧُ دࣹفْ ڪࣷ سَرَخْ بُ ݧُيْ يࣺڪَّتِ، جࣺبَّلْ اَجِ سَخْ جِ؞ ‏28 بَ لࣷولُ اَمࣹي مُوسَىٰ نَنگࣷو ڪࣷ ݖَ سࣹينِ لࣷخࣷ، تࣹگْ ڪࣷ ݖَ ڪَوْ سَرَخُ رࣹنْدِ-دٗومَلْ بَ، لَڪّ ڪࣷ ݖَ سَرْخَلُڪَايْ بَ؞ لࣷولُ سَرَخَسْ خࣹوُ ݖࣷلُ لَ وࣷونْ، نگِرْ خࣹتُگْ جࣵمّ؞ سَرَخُ سَوَرَ لَ، ݧࣹيلْ اَجِ سَخْ جِ؞ ‏29 مُوسَىٰ دَلْدِ جࣴلْ دࣴنّ بِيْ ݖࣺرَمْ ݖِ ڪُويُ خࣹوُ ݖࣷلُ گِ، يࣺڪَّتِ ڪࣷ، جࣺبَّلْ اَجِ سَخْ جِ، مُيْ سَرَخُ يࣺڪَّتِ-جࣺبَّلࣹ، دِ لَ اَجِ سَخْ جِ سَنْتࣷونْ مُوسَىٰ؞
‏30 مُوسَىٰ تࣹگْ ݖَ سࣵڪّ ݖَ دِوُ ݒَلْ گَ، اَڪْ دࣹرࣹتْ جِ ݖِ سَرْخَلُڪَايْ بِ، وِسْ-وِسَلْ ڪࣷ ݖِ ڪَوْ يَرَمُ هَـٰرُونَ اَڪْ ݖَ ڪَوِ يࣺرࣹيمْ، وِسْ-وِسَالࣹ ڪࣷ ݖَ ڪَوْ يَرَمِ دࣷومَمْ يُ گٗورْ اَڪْ ݖَ ڪَوْ يࣺرࣹيْ دࣷومَمْ؞
‏31 مُوسَىٰ وَخْ اَڪْ هَـٰرُونَ اَڪِ دࣷومَمْ يُ گٗورْ نࣹ لࣹينْ:‏ «تࣷگّلࣹينْ يࣵݒّ وِ ݖِ بُنتّ خَيْمَبْ ندَجࣹ مِ تࣹ لࣹڪّ ڪࣷ فࣷوفَ، بࣷولࣹ ڪࣷوڪْ مبُرُ مِ ݖِ لَيُبْ سَرَخِ خࣹوُ ݖࣷلُ گِ، مُيْ لَ مَ سَنْتَانࣹ وࣷونْ نࣹ يَوْ هَـٰرُونَ يَاڪْ سَيْ دࣷومْ يُ گٗورْ يࣹينَ ڪࣷيْ لࣹڪّ؞ ‏32 لِ دࣹسْ ݖِ يࣵݒّ وِ اَڪْ مبُرُ مِ، لَڪّلࣹينْ ڪࣷ، بَ مُ دِبْ دٗومْ؞ ‏33 وَايࣹ بُنتّ خَيْمَبْ ندَجࣹ مِ بُلࣹينْ ڪࣷ وࣹيسْ دِيرُ جُرٗومْ ݧَارِ فَنْ، بَ بࣺسُبْ ڪࣹرࣷوگْ بَ سࣹينْ خࣹوُ ݖࣷلُ دِ مَتْ، ندَخْ جُرٗومْ ݧَارِ فَنْ لَيْ مَتْ؞ ‏34 لࣺݒّ لُ ݧُ دࣹفْ تࣹيْ، اَجِ سَخْ جِ دَا سَنْتَانࣹ ݧُ دࣹفْ ڪࣷ، مُ دَلْدِ مَتَلْ سࣹينْ نجࣷتْلاَيْ؞ ‏35 ݖِ بُنتّ خَيْمَبْ ندَجࣹ مِ نگࣹينْ دِ يࣹمْ گُدّࣹيڪْ بࣴݖّࣴگْ دِيرُ جُرٗومْ ݧَارِ فَنْ، بࣷولࣹ ݖِ دِ دࣺنڪࣷو دࣺنڪَانࣹيْ اَجِ سَخْ جِ، ندَخْ نگࣹينْ بَݧَ دࣹي، ندَخْتࣹ لࣷولُ مࣷو دِ لِ ݧُ مَ سَنْتࣷونْ؞»
‏36 هَـٰرُونَ اَڪْ دࣷومَمْ يُ گٗورْ نَگْ دࣹفْ لࣺݒّ لُ اَجِ سَخْ جِ سَنْتَانࣹ وࣷونْ، مُوسَىٰ جࣷتّلِ؞
