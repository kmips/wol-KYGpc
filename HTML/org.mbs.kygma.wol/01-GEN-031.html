يَعْقُوبَ تࣷخُ نَ، جࣴمْ ڪَنَانْ
31
‏1 بَ لࣷولُ اَمࣹي مُ اَمْ بࣺسْ يَعْقُوبَ دࣺگّ دࣷومِ لَبَنْ يَ نَانْ، يَعْقُوبَ نَنگُ نَ لِ سࣹينْ بَايْ اَمࣷونْ لࣺݒّ، تࣹ ݖِ سࣹينْ اَلَلُ بَايْ لَ سࣵڪّࣷو اَلَلَمْ جࣷوجُ جࣺݒّ؞ ‏2 نَڪَ لَ يَعْقُوبَ گِسْ ڪَنَمُ لَبَنْ، دَلْدِ خَمْ نࣹ مࣹلَتُلْ، نَ مُ مࣹلࣷونْ ݖِ مࣷومْ؞ ‏3 اَجِ سَخْ جِ وَخْ يَعْقُوبَ نࣹ ڪࣷ:‏ «دࣹلُّلْ ݖَ سَ رࣺيوُ مَامْ يَ، ݖَ سَيْ بࣷڪّ؞ دِنَا اࣵندّ اَڪْ يَوْ؞»
‏4 يَعْقُوبَ وࣷولُ رَسࣹلْ اَڪْ لࣹيَ ݖَ تࣷولْ يَ، فَ گࣵتَّمْ يَ نࣹڪّ؞ ‏5 مُ نࣹ لࣹينْ:‏ «گِسْ نَا سࣹينْ ڪَنَمُ بَايْ؞ خَمْ نَا نࣹ مࣹلَتُلْ نَ مُ مࣹلࣷونْ ݖِ مَنْ؞ وَايࣹ سَمَ يࣵلَّيْ بَايْ اَيْ اࣵندّ اَڪْ مَنْ؞ ‏6 يࣹينْ ݖِ سࣹينْ بࣷݒّ خَمْ نگࣹينْ نࣹ لِگّࣺييَلْ نَا سࣹينْ بَايْ سَمَ ڪࣹمْ ڪࣵتَّنْ؞ ‏7 تࣹيوُلْ سࣹينْ بَايْ وࣷرْ مَ، سࣷݒِّ سَمَ ݒࣹيࣷورْ فُڪِّ يࣷونْ اَڪْ جُرٗومْ، وَايࣹ يࣵلَّ سࣹيتَانُ ڪࣷ، مُ دِ مَ لࣷرْ؞ ‏8 بُ نࣹي:‏ ‹يِ تِݒَّنْتࣹييْ سَ ݒࣹيࣷورْ،› جُرْ گࣺݒّ جُرْ لُ تِݒَّنْتࣹ؛ مبَا مُ نࣹ:‏ ‹يِ سࣹيرࣹيرْ اَيْ سَ ݒࣹيࣷورْ،› جُرْ گࣺݒّ جُرْ لُ سࣹيرࣹيرْ؞ ‏9 نِي لَ يࣵلَّ نَنگࣷو سࣹينْ جُرُ بَايْ، جࣷخْ مَ؞
‏10 «بَ ݧُ دࣹمࣹي بَ جُرْ گِيْ وَرَ تࣴبُ، دَمَا جࣹنࣹيرْ سِڪّࣹتْ يِيْ تࣴبْ بࣴيْ يِ يࣺݒّ دِ لُ سࣹيرࣹيرْ اَڪْ لُ تِݒَّنْتࣹ مبَا ݧُ رَخࣹ؞ ‏11 ݖَ لَ مَ مَلاَڪَمْ يࣵلَّ مَ وَخْ ݖِ بِيرْ گࣺنتّ گِ نࣹ مَ:‏ ‹يَعْقُوبَ،› مَ نࣹ ڪࣷ:‏ ‹نَامْ؞› ‏12 مُ نࣹ مَ:‏ ‹سِگِّلْ، خࣷولْ سِڪّࣹتْ يِيْ تࣴبْ بࣴيْ يِ يࣺݒّ، يِي دَݧࣷو سࣹيرࣹيرْ، يࣹي تࣷدّࣹ، يِ ݖِ دࣹسْ تِݒَّنْتࣹ، ندَخْتࣹ گِسْ نَا لِ لَ لَبَنْ دِ دࣹفْ لࣺݒّ؞ ‏13 مَايْ يࣵلَّ، جَ لَ فࣹيݧُ وࣷونْ بࣹتࣹلْ، فَ نگَ سَمݒّ دࣷجْ، سࣷتِّ ݖِ دِوْ، سࣹلَّلࣹ ڪࣷ، تࣹ دࣷگُ وࣷونْ فَ ݖِ مَنْ؞ لࣺيگِ نَگْ جٗگَلْ، گࣺنّ رࣺيوْ مِي، دࣹلُّ ݖَ رࣺيوْ، مَ نگَ جُدّࣷو؞›»
‏14 رَسࣹلْ اَڪْ لࣹيَ نࣹ ڪࣷ:‏ «ندَخْ اَمَتِ نَنُ بࣹنّ ݖࣺرْ وَلَّ ندࣷنࣷ سُنُ ڪࣴرْ بَايْ؟ ‏15 دࣷونُنُ لُ مࣷيْ اَيْ دࣷخَنْدࣺيمْ ݖِ مࣷومْ دࣹ!‏ ندࣹ دَ نࣷو جَايْ، لࣹڪّ سُنُ نجࣺگْ، بَ مُ جࣹيخْ تࣵڪّ؞ ‏16 تࣹ مبࣷولࣹمْ اَلَلْ، جِ يࣵلَّ نَنگࣷو ݖِ سُنُ بَايْ، نࣷو ڪࣷ مࣷومْ، نُنْ اَڪْ سُنُيْ دࣷومْ؞ ڪࣷنْ نَگْ لࣺيگِ لُ لَ يࣵلَّ وَخْ رࣹڪّ، دࣹفْ ڪࣷ؞»
‏17 يَعْقُوبَ دَلْدِ دَجَلࣹ اَيْ دࣷومَمْ اَڪْ اَيْ سࣷخْنَامْ، وَرَلْ لࣹينْ گِلࣺيمْ، ‏18 ݧُ اࣵندّ تࣹگُ ݖَ يࣷونْ وَ، يٗبَّالࣹ گࣺتَّمْ گࣺݒّ اَڪْ مبࣷولࣹمْ اَلَلْ جُ مُ لِگّࣺييْ؞ مࣷويْ گࣺتّ، گَ مُ سࣵڪّࣷو ݒَدَنْ اَرَمْ؞ ݧَ نگَيْ ݧِبِّ ڪࣴرْ بَايَمْ إِسْحَاقَ ݖَ رࣺيوُ ڪَنَانْ؞
‏19 فࣹڪّ نَ لَبَنْ دࣹمْ وَتِ اَيْ خَرَمْ¤؞ ݖِ بِيرْ لࣷولُ رَسࣹلْ سࣵݖّ گࣵلاَّجِ ڪࣴرْ يُ بَايَمْ؞ ‏20 نࣷونُ لَ يَعْقُوبَ نࣴبّࣷو لَبَنْ، اَرَمࣹينْ بَ، بَݧْ ڪࣷو يࣺگَلْ نࣹ دَيْ دࣹمْ؞ ‏21 بَ مُ دࣹمࣹي، دَفَ يٗبَّالࣹ لَ مُ اَمࣷونْ لࣺݒّ، جٗگࣹ فَ، جࣵلّ دࣹخُ اࣹفْرَاتْ، وُتِ تُنْدِ گَلࣵدّ؞
لَبَنْ دَبِ نَ يَعْقُوبَ
‏22 يَعْقُوبَ دَوْ نَ، بَ مُ اَمْ ݧࣹتِّ فَنْ، ݧُ سࣷوگْ ڪࣷو يࣺگَلْ لَبَنْ؞ ‏23 مُ اࣵندّ نَگْ اَڪْ اَيْ بࣷڪَّمْ، تࣷݒّ ڪࣷ جُرٗومْ ݧَارِ فَنْ، دࣷورْ ڪࣷو دَبْ ݖَ تُنْدِ گَلࣵدّ؞ ‏24 ݖِ ڪَوْ لࣷولُ يࣵلَّ فࣹيݧُ لَبَنْ، اَرَمࣹينْ بَ، ݖِ گࣺنْتُ گُدِّ نࣹ ڪࣷ:‏ «مࣷيْتُلَ وَخْ يَعْقُوبَ لࣹنّ لُ ڪࣷيْ گَارْ؞» ‏25 بَ لَبَنْ دَبࣹي يَعْقُوبَ، دَفَ فࣹڪّ مُ سَمݒّ خَيْمَامْ ݖَ تُندّ وَ؞ لَبَنْ اِتْ سَمݒّ بࣷسَمْ ݖَ ڪَوْ تُنْدِ گَلࣵدّ، مࣷومْ اَڪِ بࣷڪَّمْ؞
‏26 لَبَنْ نࣹ يَعْقُوبَ:‏ «لࣷويْ دࣹفْ نِي، يَوْ، دِ مَ نࣴبُّيْ دَوْ، يٗبَّالࣹ سَمَ دࣷومْ يُ جِگࣺينْ، ݧُ مࣹلْ نِ ݧُ ݧُ لࣴلࣹ خَرࣹ؟ ‏27 لُ وَرَلْ نگَ دِ مَ نࣴبُّيْ دَوْ تࣹ بَݧْ مَا تࣵگُّ؟ ندࣹ ڪࣷنࣷونْ جࣷمْبُلْ مَ يِوِ لَ ݖِ خࣷلْ بُ سࣹدّ، بُ اࣵندّ اَڪِ وࣷيْ، بࣷولࣹ ݖِ خَلَمْ اَڪِ سَبَرْ؞ ‏28 مَيࣷو مَ سَخْ، مَ فٗونْ سَمَيْ سࣴتْ اَڪْ سَمَيْ دࣷومْ يُ جِگࣺينْ، تࣵگّتࣷوڪْ ݧࣷومْ؛ يࣴفِ دࣷفْ دࣹ نگَ دࣹفْ نِي!‏ ‏29 مَنࣷونْ نَا لاَ دࣹفْ لُ بࣷنْ، وَايࣹ سَ يࣵلَّيْ بَايْ دَ مَا فࣹيݧُ بِيگْ نࣹ مَ:‏ ‹مࣷيْتُلَ وَخْ يَعْقُوبَ لࣹنّ لُ ڪࣷيْ گَارْ؞› ‏30 بُ لَ نࣹيخࣹي، نگَ نَمّ سَ ڪࣴرْ بَايْ، بَ دࣷگُ ݖَا دࣹمْ، وَايࣹ لُ تَخْ نگَ سࣵݖّ سَمَ گࣵلاَّجِ ڪࣴرْ، تࣹگْ ݖِ؟»
‏31 يَعْقُوبَ تࣷنْتُ لَبَنْ نࣹ ڪࣷ:‏ «خَنَا رَگَلࣷونْ رࣹڪّ نࣹ جࣷمْبُلْ دِنگَ تࣺيࣹ سَيْ دࣷومْ يُ جِگࣺينْ؞ ‏32 وَايࣹ ڪࣺݒّ ڪࣷو گِسْ مُ يࣷرْ سَ گࣵلاَّجِ ڪࣴرْ، بࣷو يࣹبࣷو رࣹيَلْ؞ سࣹيتَلْ فِي ݖِ سُنُ ڪَنَمِ بࣷڪّ لࣺݒّ لُ مَ فِ يࣷرْ، لࣷو ݖِ مࣷومْ فَبَلْ، يٗبُّ؞» بࣷوبَ يَعْقُوبَ خَمُلْ وࣷونْ نࣹ رَسࣹلَ ڪࣷ جࣴلْ؞ ‏33 لَبَنْ دُگّ ݖِ خَيْمَبْ يَعْقُوبَ، تࣹگْ ݖَ خَيْمَبْ لࣹيَ اَڪْ يࣷسِ ݧَارِ جَامَمْ يُ جِگࣺينْ يَ، تࣹ گِسُلْ دَرَ؞ مُ گࣺنّࣹ خَيْمَبْ لࣹيَ، دُگّ ݖِ بُ رَسࣹلْ؞
‏34 فࣹڪّ نَ رَسࣹلْ فَبْ گࣵلاَّجِ ڪࣴرْ يَ، لࣵقْ ݖَ رࣷنْ تࣹگُ گِلࣺيمْ گَ، تࣷوگْ ݖَ؞ لَبَنْ تࣴݖّ، تࣴݖّ ݖَ بِيرْ خَيْمَ بَ، گِسُلْ دَرَ؞ ‏35 رَسࣹلْ وَخْ بَايَمْ نࣹ ڪࣷ:‏ «بَايْ، بَالْ مَ دࣹ، دَمَا مَنُلَ جٗگْ ݖِ سَ ڪَنَمْ، ندَخْتࣹ دَمَا اࣵندّ اَڪْ مبَاخْ؞» لَبَنْ سࣹيتْ نَ فُ نࣹ، وَايࣹ گِسُلْ گࣵلاَّجِ ڪࣴرَمْ يَ؞
‏36 يَعْقُوبَ مࣹرْ لࣷولْ، دَلْدِ گࣴدّ لَبَنْ نࣹ ڪࣷ:‏ «ندَخْ دَمَا تࣷوݧْ وَلَّ دَمَا مࣷيْ، بَ نگَ سَمْبَلْ مَ، دِ مَ تࣷݒّ نِي؟ ‏37 بِ نگَ جࣺيخࣹي سَمَ يࣴفْ يࣺݒّ بَ نࣷݒِّ، اَنَ لࣷو گِسْ لُ بࣷڪّ ݖِ سَ يࣴفِ ڪࣴرْ؟ تࣹگْ ڪࣷ فِي ݖِ سَمَ ڪَنَمِ بࣷڪّ اَڪْ سَيْ بࣷڪّ، ݧُ اࣵتّࣹ نُ؞ ‏38 خَنَا دُ ݧَارْ فُڪِّ اَتَ نگِي مَ نࣹڪّ اَڪْ يَوْ؟ سَيْ خَرْ اَڪْ سَيْ بࣴيْ اࣴمْبُݧُ سَخْ، مُ يࣵقُ، تࣹ مَسُمَا لࣹڪّ سَ ڪُويِ گࣺتّ؞ ‏39 جُرْ گُ رَبْ فࣵدّ سَخْ مَ فࣺيتࣹوࣷو، وَايࣹ مَسُمَ لَ ڪࣷو اِنْدِلْ؞ مبࣷولࣹمْ لُ ݧُ سࣵݖّ گُدِّ اَڪْ لُ ݧُ سࣵݖّ بࣴݖّࣴگْ، دَنگَ مَ ڪࣷ دَا فࣹيُ؞ ‏40 نَاجْ وِ لاَ دَانْ يࣹنْدࣷو يࣹنُ بࣴݖّࣴگْ، دِ فَنَانࣷو سࣹدّ گُدِّ، مُ خَوْ مَا رࣹيْ، بَ دُمَ گࣴمّ بࣴتْ؞ ‏41 ݧَارْ فُڪِّ اَتَ نگِي، مَ دِ لَ جَامُ ݖِ ڪࣴرْ گِ، لِگّࣺييَلْ نَا لَ فُڪِّ اَتْ اَڪْ ݧࣹينتّ نگِرْ سَ ݧَارِ دࣷومْ يُ جِگࣺينْ ݧِي، تࣹگْ ݖِ جُرٗومْ بࣹنِّ اَتْ يُ مَ لَ سࣵمَّلْ سَيْ گࣵتّ، نگَ سࣷݒِّ سَمَ ݒࣹيࣷورْ فُڪِّ يࣷونْ اَڪْ جُرٗومْ؞ ‏42 سُ دُلْ وࣷونْ اَڪْ سَمَ يࣵلَّيْ بَايْ، جِ اࣵندّ اَڪْ مَنْ، دِ يࣵلَّيْ إِبْرَاهِيمَ، جِ إِسْحَاقَ وࣷرْمَالࣷونْ، ڪࣷنࣷونْ نگَ دࣵقْ مَ تࣹيْ، مَ دࣹمْ لࣷخࣷيْ نࣹينْ؞ وَايࣹ سَمَ تࣷسْڪَرࣹ اَڪْ سَمَ اࣵقُ ݧَقْ لَ يࣵلَّ گِسْ، بَ تَخْ مُ اࣵتّࣹ مَ اَڪْ يَوْ بِيگْ؞»
‏43 بَ مُ ڪࣷ وَخࣹي لࣷولُ، لَبَنْ نࣹ ڪࣷ:‏ «جِگࣺينْ ݧِي سَمَيْ دࣷومْ لَݧُ، خَلࣹ يِي دِ سَمَيْ سࣴتْ، تࣹ گࣺتّ گِي سَمَ گࣺتَّ؛ لُ فِ نࣹڪّ مَا ڪࣷ مࣷومْ؞ ݧِي دِ سَمَ دࣷومْ يُ جِگࣺينْ، نُ مَ لࣹينْ دِ دࣹفْ تࣹيْ، ݧࣷومْ اَڪْ سࣹينِ دࣷومْ؟ ‏44 لࣺيگِ ݧࣴوَلْ رࣹڪّ، ݧُ فَسَنْتࣹ ڪٗلّࣴرࣹ مَاڪْ يَوْ، تࣹ سࣹيدࣹ دࣷخْ سُنُ دِگَّنْتࣹ مَاڪْ يَوْ؞» ‏45 بَ لࣷولُ اَمࣹي يَعْقُوبَ جࣴلْ اَوْ دࣷجْ، سَمݒّ ڪࣷ، مُيْ دࣷجُ خࣵمِّڪَايْ؞ ‏46 يَعْقُوبَ نࣹ اَيْ بࣷڪَّمْ:‏ «فࣷرْلࣹينْ اَيْ دࣷجْ؞» ݧُ فࣷرْ اَيْ دࣷجْ، دࣹفْ ڪࣷ اَبْ جَلْ، دَلْدِ لࣹڪّ فࣷوفَ ݖَ وࣹتُ جَلْ بَ؞ ‏47 لَبَنْ تُدّࣹ بࣺرَبْ بَ يࣹگَرْ سَادُتَ؛ يَعْقُوبَ تُدّࣹ ڪࣷ گَلࣹتْ¤؞
‏48 بࣷوبَ لَ لَبَنْ نࣹ:‏ «جَلْ بِي تࣹيْ، مࣷويْ سُنُ سࣹيدࣹ سَمَ دِگَّنْتࣹ اَڪْ يَوْ؞» مࣷو تَخْ ݧُ تُدّࣹ ڪࣷ گَلࣹتْ (مُ فِرِ جَلُبْ سࣹيدࣹ)؞ ‏49 ݧُ دِ ڪࣷ تُدّࣹ اِتَمْ مِسْݒَ (مُ فِرِ تَتَيْ وَتُّڪَايْ)، ندَخْ لَبَنْ دَ نࣹ:‏ «يَلْ نَ اَجِ سَخْ جِ وَتُّ سُنُ دِگَّنْتࣹ، ڪࣹرࣷوگْ بُ ݧُ سࣷرࣹي؞ ‏50 سࣷو سࣷخࣷرࣹي سَمَيْ دࣷومْ يُ جِگࣺينْ، مبَا نگَ تَڪّ يࣹنࣹينْ جَبَرْ، بُ فَ ڪࣹنّ نࣹڪُّلْ سَخْ، مࣷيْتُلْ، ندَخْ يࣵلاَّ نُيْ سࣹيدࣹ؞» ‏51 لَبَنْ نࣹيتِ يَعْقُوبَ:‏ «خࣷولَلْ ݖِ جَلْ بِي اَڪْ دࣷجُ خࣵمِّڪَايْ، وِ ݖِ سَمَ دِگَّنْتࣹيڪْ يَوْ؞ ‏52 جَلْ بِ سࣹيدࣹ لَ، دࣷجْ وِ دِ سࣹيدࣹ نࣹ دُمَ جࣺگِّ جَلْ بِ، جُبْلُ ݖِ يَوْ، تࣹ يَوْ اِتَمْ دࣷو جࣺگِّ جَلْ بِ مبَا دࣷجْ وِ، جُبْلُ ݖِ مَنْ، تࣹ نَرَلُلࣷو مَ جࣵمّ؞ ‏53 يࣵلَّيْ إِبْرَاهِيمَ اَڪْ يࣵلَّيْ نَخࣷرْ، ݧࣷومْ ݧࣷوݧࣹي دِ سࣹينْ يࣵلَّيْ بَايْ، يَلْ نَݧُ اࣵتّࣹ سُنُ دِگَّنْتࣹ؞» يَعْقُوبَ دَلْدِ گِݧْ ݖِ يࣵلَّ، جِ بَايَمْ إِسْحَاقَ وࣷرْمَالࣷونْ؞ ‏54 گَنَّاوْ لࣷولُ يَعْقُوبَ سࣵڪّ سَرَخْ ݖَ ڪَوْ تُندّ وَ، وࣷو اَيْ بࣷڪَّمْ، ݧُ بࣷڪّ لࣹڪّ؞ بَ ݧُ لࣹڪّࣹي بَ نࣷݒِّ، فَنَانْ ݖَ تُندّ وَ؞
