اࣵتّࣹ لاَجْ نَ نجُبْ اَڪْ يࣴرْمَنْدࣹ
23
‏1 «بُلࣹينْ سِيوَلْ اَيْ فࣹنْ؛ بُلࣹينْ لࣴنگࣷو اَڪْ تࣷوݧْڪَتْ، نگِرْ دࣷونْ سࣹيدࣹبْ نَافࣺقَمْ؞ ‏2 بُلࣹينْ سࣹيتْ فِ مبࣷولࣷو گࣴنَ بَرࣹي، نگࣹينْ تࣷݒّ لࣹينْ ݖِ جࣴفْ جُ بࣷنْ؛ تࣹ اِتْ بُلࣹينْ جࣹنگّ، دِ تࣷݒّ مبࣷولࣷو مُ گࣴنَ بَرࣹ ݖِ اَبْ لَيࣷو، دِ لࣹينْ سࣹيدࣹيلْ لُ تࣷخَلْ دࣴگّ؞ ‏3 نࣺيوْ-دࣷولࣹ بِ اِتْ، بُلࣹينْ ڪࣷ فَرْ ݖِبْ لَيࣷو؞
‏4 «سُ نگࣹينْ دَجࣹيڪْ سࣹينْ نَگُ نࣷونْ مبَا مبَامَمْ مُ رࣺيرْ، فࣹخࣹلࣹينْ بَ دࣺلّࣷوسِلْ ڪࣷ ڪࣷ؞ ‏5 سُ نگࣹينْ گِسࣹي سࣹينْ مبَامُ نࣷونْ مُ اَبْ سࣴفْ دَانࣹيلْ، بُلࣹينْ ڪࣷ رࣷمبّ؛ دِمْبَلِلࣹينْ سࣹينْ نࣷونْ، يࣺڪَّتِلْ-لࣹينْ ڪࣷ ڪࣷ؞
‏6 «بُلࣹينْ جَلْگَتِ اࣵقُ نࣺيوْ-جِ-دࣷولࣹ ݖِبْ لَيࣷومْ؞ ‏7 ڪࣵدُّ گُ دࣴگُّوُلْ، سࣷرࣹلࣹينْ ڪࣷ، تࣹ ڪُ دࣹفُلْ دَرَ اَڪْ ڪُيْ دࣹفْ نجࣹڪّ، بُلࣹينْ ڪࣷ رࣹيْلُ، ندَخْ نِتْ ڪُ بࣷنْ، دُمَ ڪࣷ اࣵتّࣹ مُڪّ اࣵتّࣹبْ ڪُيْ دࣹفْ نجࣹڪّ؞ ‏8 اَلَلِ گࣹرْ نَگْ بُلࣹينْ ڪࣷ نَنگُ، ندَخْ گࣹرْ دَيْ گُمْبَالْ نجࣵݖَّارْ، دِ سَفَانْ وَخِ ڪُ نࣹڪّ ݖِ دࣴگّ؞
‏9 «اَبْ دࣷخَنْدࣺيمْ، بُلࣹينْ ڪࣷ نࣷتْ؛ يࣹينْ ݖِ سࣹينْ بࣷݒّ، جَارْ نگࣹينْ ݖِ ندࣷخَنْدࣺيمْ، ندَخْ اَيْ دࣷخَنْدࣺيمْ نگࣹينْ وࣷونْ ݖَ رࣺيوُمْ مِسْرَ؞
لُ جࣴمْ ݖِ بࣺسِ نࣷفْلاَيْ
‏10 «دِيرُ جُرٗومْ بࣹنِّ اَتْ نگࣹينْ دِ جِ سࣹينْ تࣷولْ، فࣷرْ لِ ݖِ مࣹݧّࣹ؛ ‏11 وَايࣹ اَتُمْ جُرٗومْ ݧَارࣹيلْ مَ، نࣷݒَّلْ-لࣹينْ تࣷولْ بِ، بࣷويَلْ ڪࣷ؛ سࣹينْ نࣺيوْ-دࣷولࣹيْ اَسْڪَنْ وِ مَنْ ݖࣹي لࣹڪّ، لِ ݖِ دࣹسْ، رَبِ اࣵلّ يِ فࣷرْ ڪࣷ؞ سࣹينْ تࣷولُبْ رࣹسࣹݧْ اَڪْ سࣹينْ تٗوڪࣴرُ اࣷلِوْ اِتَمْ، نَڪَ نࣷونُ؞
‏12 «دِيرُ جُرٗومْ بࣹنِّ فَنْ نگࣹينْ دِ لِگّࣺييْ، وَايࣹ بࣺسُبْ جُرٗومْ ݧَارࣹيلْ بَ، نࣷݒّلُ لࣹينْ، ندَخْ سࣹينْ نَگْ اَڪْ سࣹينْ مبَامْ مَنَ نࣷݒّلُ، سࣹينْ سُرْگَ اَڪُبْ دࣷخَنْدࣺيمْ اِتْ مَنَ اَسْتَنْدِڪُ؞
‏13 «لِ مَ لࣹينْ وَخْ لࣺݒّ سࣵمّلࣹينْ ڪࣷ، تࣹ تُرِ يࣹنࣹينْ يࣵلَّ، بُلࣹينْ ڪࣷ تُدّ؛ بُمُ جِبࣹ سَخْ سࣹينْ گࣺمِّݧْ؞
لُ جࣴمْ ݖِ بࣺسِ مࣵگَّلِ إِسْرَاءِيلْ
‏14 «ݧࣹتِّ يࣷونْ ݖِمْ اَتْ نگࣹينْ مَيْ جَگْلࣹيلْ مࣵگَّلْ؞ ‏15 مࣵگَّلُ مبُرُ مُ اَمُلْ لَوِيرْ، وࣷرْمَالْ-لࣹينْ ڪࣷ؞ دِيرُ جُرٗومْ ݧَارِ فَنْ نگࣹينْ ݖِيْ لࣹڪّ مبُرُ مُ اَمُلْ لَوِيرْ نِ مَ لࣹينْ ڪࣷ سَنْتࣹ، سَا يُ جَمࣷنࣷومْ جࣷتࣹي ݖِ وࣹيرُ اَبِيبْ، ندَخْتࣹ ݖِ نگࣹينْ گࣺنّ مِسْرَ؞ تࣹ اِتَمْ بُلࣹينْ تَخَوࣹ لࣷخࣷيْ نࣹينْ ݖِ سَمَ ڪَنَمْ؞ ‏16 وࣷرْمَالْ-لࣹينْ مࣵگَّلُ بࣺسُ نگٗوبْ مِ جࣴڪَّ ݧࣷرࣹ ݖِ تࣷولْ، يِ نگࣹينْ دِ بࣹيْ، اَڪْ مࣵگَّلُ بࣺسُ فَرْمࣵنگُ مِ ݖِ ندࣹيتࣹلْ اَتْ مِ، گَنَّاوْ بُ نگࣹينْ دَجَلࣹي سࣹينْ مࣹݧّࣹيفْ بَ نࣷݒِّ؞ ‏17 ݧࣹتِّ يࣷونْ ݖِمْ اَتْ لَ سࣹينْ گٗورْ ݧࣺݒّ دِ تࣹيوْ فِ ڪَنَمْ بࣷرࣷومْ بِ اَجِ سَخْ جِ؞
‏18 «بُلࣹينْ مَ رࣹنْدِلْ بࣹنّ سَرَخْ، دِ بࣷولࣹ دࣹرࣹتْ جَاڪْ لُ اَمْ لَوِيرْ، تࣹ نࣹبّࣷنُ جُرْ گُ نگࣹينْ مَ سَرْخَلْ ݖِ سَمَ مࣵگَّلْ، بُمُ فَنَانْ بَ بࣴتْ سࣹتْ؞
‏19 «نگࣴنࣹيلِ مࣹݧّࣹيفْ مَ جࣴڪَّ ݧࣷرࣹ ݖَ سࣹينْ سُوفْ، مࣷومْ نگࣹينْ دِ يٗبُّ سࣹينْ ڪࣴرْ يࣵلَّ اَجِ سَخْ جِ؞
«اَبْ تࣹفْ، بُلࣹينْ ڪࣷ تࣷگّࣹ سࣷووُمْ يَايَمْ؞
يࣵلَّ يࣹبْ نَ بࣵنِّ إِسْرَاءِيلْ
‏20 «تࣹ نَگْ مَنْ ݖِ سَمَ بࣷݒّ، مَايْ يࣹبَلْ مَلاَڪَ، مُ دِ لࣹينْ سࣵمّ ݖِ يࣷونْ وِ، بَ يٗبُّ لࣹينْ بࣺرَبْ بَ مَ وَاجَلْ؞ ‏21 بࣵيِّلࣹينْ ڪࣷ خࣹلْ بُ بَاخْ، دࣺگَّلْ ڪࣷ تࣹ بَݧْ ڪࣷو مࣹرْلࣷو، ندَخْ دُ لࣹينْ بَالْ سࣹينْ مࣷيْ، نگِرْ سَمَ تُرْ لَ تَخَوࣹ؞ ‏22 وَايࣹ سُ نگࣹينْ ڪࣷ دࣺگَّلࣹي بَ دࣺگَّلْ ڪࣷ، دِ جࣴفࣹ لِ مَ لࣹينْ وَخْ لࣺݒّ، سُ بࣷوبَا مَ نࣷونࣷو سࣹينِ نࣷونْ، بَݧْ سࣹينِ بَݧَالࣹ؞
‏23 «بُ لࣹينْ سَمَ مَلاَڪَ جِيتࣷو، بَ يٗبُّ لࣹينْ ݖَ رࣺيوُمْ اَمࣷرࣹينْ ݧَ اَڪْ اࣹتّࣹينْ ݧَ اَڪْ ݒࣹرِسࣹينْ ݧَ اَڪْ ڪَنَانࣹينْ ݧَ اَڪْ اࣹوࣹينْ ݧَ اَڪْ يࣹبُسࣹينْ ݧَ، بَ مَ فَاگَاگَلْ لࣹينْ ݧࣷومْ ݧࣺݒّ، ‏24 سُ بࣷوبَا بُلࣹينْ سُجّٗوتَلْ مُڪّ سࣹينِ تُورْ، دِ لࣹينْ جَامُ، مبَا نگࣹينْ دِ رࣷيْ سࣹينِ آدَ، وَايࣹ يࣵقْتࣹلࣹينْ سࣹينِ تُورْ تࣹ تࣷجْ تَسَرْ سࣹينْ تُورِ دࣷجْ، يَ فَ سَمْݒࣹ؞ ‏25 جَامُلࣹينْ مَ، مَنْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ، مَ بَرْڪࣹيلْ سࣹينْ دُگُبْ اَڪْ سࣹينْ ندࣷخْ، بࣷولࣹ ݖِ جࣴلࣹ وࣷݒّ ݖِ سࣹينْ بِيرْ؞ ‏26 سُ ڪࣷ دࣹفࣹي دُ ڪࣹنّ ڪُ اࣴمْبَمْ دِ يࣵقُ، مبَا مُ تࣴلࣹي اَمْ دࣷومْ ݖِ سࣹينُمْ رࣺيوْ، تࣹ مَايْ مَتَلْ سࣴڪّ سࣹينْ لِمُ فَنْ؞
‏27 «تِيتَانگࣹ لاَيْ يࣹبَلْ، مُ جِيتُ لࣹينْ؞ مَايْ تِيتَلْ مبࣷولࣹمْ خࣹيتْ وُ نگࣹينْ فࣹڪِّ، تࣹ مَايْ دࣹفْ سࣹينْ نࣷونْ يࣺݒّ وࣷنْ لࣹينْ گَنَّاوْ؞ ‏28 مَايْ يࣹبَلْ نجࣵقَرࣹ لُ لࣹينْ جِيتُ، دࣵقَلْ لࣹينْ اࣹوࣹينْ ݧِ اَڪْ ڪَنَانࣹينْ ݧِ اَڪْ اࣹتّࣹينْ ݧِ؞ ‏29 وَايࣹ دُمَ لࣹينْ دࣵقَنْدࣷو ݖِ مࣹنّ اَتْ، لُ ڪࣷ مࣷيْ رࣺيوْ مِ گࣹنتّ، رَبِ اࣵلّ يِ مُجّ بَرࣹ لࣷولْ ݖِ سࣹينْ ڪَوْ؞ ‏30 ندࣵنڪّ-ندࣵنڪّ لاَ لࣹينْ لࣹينْ دِ دࣵقَلْ، بَ ڪࣹرَ نگࣹينْ دِ يࣷڪُّ، بَ نَنگُ رࣺيوْ مِ؞
‏31 «مَا لࣹينْ دِ دࣷگَلْ ڪࣹمُگْ سُوفْ گُ دَلࣹ گࣺيجُ بَرَخْ يِ، بَ گࣺيجُ وَا فِلِسْتِ، دَلࣹيتِ ݖَ مࣵنْدِݝْ مَ، بَ ݖَ دࣹخُ اࣹفْرَاتْ، ندَخْ مَايْ تࣹگْ وَا رࣺيوْ مِ ݖِ سࣹينِ لࣷخࣷ، نگࣹينْ دࣵقْ لࣹينْ؞ ‏32 بُلࣹينْ وٗلّࣴرَنْتࣹيڪْ ݧࣷومْ، بُلࣹينْ وٗلّࣴرَنْتࣹيڪْ سࣹينِ تُورْ؞ ‏33 بُݧُ دࣴڪّࣹ سࣹينُمْ رࣺيوْ، لُ ڪࣷ مࣷيْ مُيْ سࣹينْ ندࣴگّ-سࣴرࣴخْ؛ ندَخْ دَݧُ لࣹينْ دِ خِيرْ ݖِ مࣷيْ مَ، بَ نگࣹينْ جَامُ سࣹينِ تُورْ؞»
