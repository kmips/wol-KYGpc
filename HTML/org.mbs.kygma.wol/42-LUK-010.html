عِيسَىٰ يࣹبَلْ نَ جُرٗومْ ݧَارْ فُڪّ اَڪْ ݧَارْ
10
‏1 گَنَّاوْ لࣷولُ سَنگّ بِ تَبّ نَ يࣹنࣹينْ جُرٗومْ ݧَارْ فُڪِّ تَالِبࣹ اَڪْ ݧَارْ، يࣹبَلْ لࣹينْ، ݧَارْ، ݧَارْ، ݧُ جِيتُجِ ڪࣷ ݖَ مبࣷولࣹمْ دࣴڪّ يَاڪْ گࣷخْ، يَ مُ نَمّࣷونَ دࣹمْ مࣷومْ ݖِ بࣷݒَّمْ؞ ‏2 مُ نࣹ لࣹينْ:‏ «نگٗوبْ مࣹي يَاتُ، تࣹ سُرْگَ يِ نࣺيوْ؞ ڪࣷنْ نَگْ ݧَانْلࣹينْ بࣷرࣷومْ نگٗوبْ مِ، اَيْ سُرْگَ يُ مُ يࣹبَلْ ݖِمْ نگٗوبَمْ؞ ‏3 دࣹمْلࣹينْ!‏ خَمْلࣹينْ نࣹ مَا لࣹينْ يࣹبَلْ، مبࣹتࣹ اَيْ گࣵتّ ݖِ بِيرِ بُڪِّ؞ ‏4 بُلࣹينْ يٗبَّالࣹ دࣹنݖُڪَايُ خَالِسْ، مبَا مبُوسْ، مبَا اَيْ ݖَرَخْ تࣹ بُلࣹينْ تَخَوْ دِ سَافࣷونْتࣹ اَڪْ ڪࣹنّ ݖِ يࣷونْ وِ؞
‏5 «گࣺݒّ ڪࣴرْ گُ نگࣹينْ دُگّ، جࣴڪّلࣹينْ نࣹ:‏ ‹جࣵمّ ݧࣹيلْ نَ لࣹينْ؞› ‏6 سُ فَ نِتُ جࣵمّ نࣹڪّࣹي، سࣹينْ جࣵمّ دَلْ ڪࣷ؞ لُ ڪࣷ مࣷيْ، مُ دࣺلْسِ ݖِ يࣹينْ؞ ‏7 دَلْ-لࣹينْ ݖَ ڪࣴرْ گࣷوگَ، دِ لࣹڪّ اَڪَ نَانْ ݖَ لَ ݧُ لࣹينْ جࣷخْ، ندَخْ لِگّࣺييْڪَتْ يࣹلّࣷو نَ ݒࣹيࣷورَمْ؞ بُلࣹينْ دِ وࣴرْ نَگْ ڪࣴرْ اَڪْ ڪࣴرْ؞
‏8 «بࣺݒّ دࣴڪّ بُ نگࣹينْ دُگّ، ݧُ دَلَلْ لࣹينْ، لَ ݧُ لࣹينْ تَاجَلْ، لࣹڪّلࣹينْ ڪࣷ؞ ‏9 وࣺرَلْ لࣹينْ جَرَگِ دࣴڪّ بَ، تࣹ نگࣹينْ وَخْ وَا دࣴڪّ بَ، نࣹ لࣹينْ:‏ ‹نگُورُگْ يࣵلَّ دࣴگْمَلْ نَ لࣹينْ؞› ‏10 وَايࣹ بࣺݒّ دࣴڪّ بُ نگࣹينْ دُگّ تࣹ دَلَلُݧُ لࣹينْ، دࣹمْلࣹينْ ݖَ ݒࣺنݖّ مَ تࣹ نگࣹينْ نࣹ لࣹينْ:‏ ‏11 ‹سࣹينْ ݒࣴنْدُبْ دࣴڪّ بِ ݖِ سُنُ ندࣴگُّيْ تࣵنڪّ سَخْ، نُ نگِ ڪࣷيْ فَخَسْ، سࣹيدࣹيلࣹ لࣹينْ ڪࣷ سࣹينْ نگࣵنْتَلْ؞ وَايࣹ نَگْ خَمْلࣹينْ نࣹ نگُورُگْ يࣵلَّ دࣴگْمَلْ نَ؞› ‏12 مَا لࣹينْ ڪࣷ وَخْ، بُ بࣺسْ بَا، بࣷوبُ دࣴڪّ، سࣷدࣷمَ ڪࣷيْ گࣴنُ دࣹمِنْ؞
‏13 «وٗويْ نگَلَّ يࣹينْ وَا ڪࣷرَسِنْ، يࣹينْ وَا بࣹݖَّيْدَ اِتْ، وٗويْ نگَلَّ يࣹينْ، ندࣹ ڪࣺيمَانْ يِ سࣷتّࣹي فِ سࣹينْ بِيرْ، سُ دࣷونْ فَ وَا تِرْ مبَا فَ وَا سِدࣷنْ لَ سࣷتّࣹي، ڪࣷنْ بُ يࣵگّ لَݧُ دࣷونْ سࣷلْ اَيْ سَاڪُ، خࣴݒّࣷوبْ دٗومْ، نگِرْ تُوبْ؞ ‏14 مࣷو تَخْ اِتْ، ڪࣹرَ اࣵتّࣹ بَ، وَا تِرْ اَڪْ سِدࣷنَ لࣹينْ دِ گࣴنُ دࣹمِنْ؞ ‏15 يࣹينْ وَا ڪَݒࣹرْنَوُمْ نَگْ، دࣹيسْ نَ لࣹينْ يࣺڪَّتِ بَ اَسَمَانَ؟ خَنَا بَ فَ نجَنِيوْ ڪَيْ نگࣹينْ دِ وࣵݖِّ؞
‏16 «ڪُ لࣹينْ دࣺگَّلْ نَگْ، مَنْ نگَ دࣺگَّلْ؛ ڪُ لࣹينْ گࣵنْتَلْ، مَنْ نگَ گࣵنْتَلْ؞ تࣹ ڪُ مَ گࣵنْتَلْ، گࣵنْتَلْ نگَ ڪِ مَ يࣹبَلْ؞»
‏17 بَ جُرٗومْ ݧَارْ فُڪّ ݧَاڪْ ݧَارْ دࣹمࣹي بَ دࣺلْسِ، مبࣺگْتࣹ لَݧُ اࣵنْدَلْ؞ ݧُ نࣹ:‏ «سَنگّ بِ، رَبْ يِ سَخْ، دَنُ لࣹينْ دِ جࣷخْ ندِگَلْ ݖِ سَ تُرْ، ݧُ نَنگُلْ نُ؞» ‏18 عِيسَىٰ نࣹ لࣹينْ:‏ «سࣺينْ نَا سࣹيْتَانࣹ مُ فࣵقࣹ اَسَمَانْ نِ اَگْ مࣹلَخْ؞ ‏19 دَمَ نࣹ، مَا لࣹينْ جࣷخْ سَݧْ-سَݧُ جࣷگِّ ݖِ ڪَوِ جَانْ اَڪِ جِيتْ، جࣷگَّتِ مبࣷولࣹمْ دࣷولࣹيْ بَݧَالࣹ بِ، تࣹ دَرَ دُ لࣹينْ مَنَ گَاݧْ مُڪّ؞ ‏20 نࣷونُ لࣹينْ رَبْ يِ نَنگُلࣹ نَگْ، بُلࣹينْ ڪࣷ بࣺگࣹ، بࣺگࣹلࣹينْ ڪَيْ لَ سࣹينْ تُرْ بِنْدࣹ فَ اَسَمَانْ؞»
‏21 ݖَ وَخْتُ وࣷووَ تࣹمْبࣹ لَ ڪࣹمْتَلاَيُ مبࣺگْتࣹ تُڪّࣹي ݖِ نࣷو گُ سࣹلّ گِ، دُگّ عِيسَىٰ، مُ دَلْدِ نࣹ:‏ «بَايْ، بࣷرࣷومْ اَسَمَانْ اَڪْ سُوفْ، سَنتّ نَا لَ، ندࣹ يَا لࣵقْ لِي بࣷرࣷومْ خَمْ-خَمْ يࣹيڪْ بࣷرࣷومْ خࣹلْ يِ، خَمَلْ ڪࣷ تُوتْ-تࣵنڪّ يِ، ندࣹ لࣷولُ دࣹ، بَايْ، مࣷويْ لِ لَ سࣷوبْ؞ ‏22 لࣺݒّ لَ مَ سَمَ بَايْ دࣺنڪّ، تࣹ ڪࣹنّ خَمُلْ ڪُيْ دࣷومْ جِ، ڪُ دُلْ بَايْ بِ؛ ڪࣹنّ اِتْ خَمُلْ ڪُيْ بَايْ بِ، ڪُ دُلْ دࣷومْ جِ، اَڪْ ڪࣺݒّ ڪُ ڪࣷ دࣷومْ جِ نَمَّ خَمَلْ؞»
‏23 ݖِ ڪَوْ لࣷولُ عِيسَىٰ وَلْبَتِڪُ ݖَ تَالِبࣹ يَ، وَخْ لࣹينْ ݧࣷومُ نࣹينْ، نࣹ لࣹينْ:‏ «ندࣷڪّلࣹي ݧِ سࣹينِ گࣴتْ گِسْ لِ نگࣹينْ گِسْ؞ ‏24 ندَخْ مَا لࣹينْ ڪࣷ وَخْ، بَرࣹ نَ اَيْ يࣷنࣹنتّ اَڪْ بُورْ يُ بࣴگّࣷونَ گِسْ لِ نگࣹينْ گِسْ، تࣹ گِسُݧُ ڪࣷ، اَڪْ اِتَمْ دࣺگّ لِ نگࣹينْ دࣺگّ، تࣹ دࣺگُّݧُ ڪࣷ؞»
تࣹخࣹ اَمْ نَ يࣷونْ
‏25 بࣹنّ خَمْڪَتُ يࣷونُ مُوسَىٰ مࣷو جࣹڪِّ وࣷونْ جٗگْ، دِ فࣹخࣹي فِيرْ عِيسَىٰ؞ مُ نࣹ ڪࣷ:‏ «سࣴرِݧْ بِ، اَنَ لُ مَيْ دࣹفْ، بَ تࣹخࣹ بَ فࣵوّ؟» ‏26 عِيسَىٰ نࣹ ڪࣷ:‏ «يࣷونُ مُوسَىٰ، اَنَ لࣹيسْ ݖِ بِندّ؟ لࣷو ݖِ جࣵنگࣹ؟» ‏27 مُ نࣹ ڪࣷ:‏ «بࣷرࣷومْ بِ سَ يࣵلَّ، نَنگَ ڪࣷ سࣷݒّࣹ سَ لࣺݒِّ خࣷلْ اَڪْ سَ لࣺݒِّ بَڪَّنْ اَڪْ سَ لࣺݒِّ دࣷولࣹ اَڪْ سَ لࣺݒِّ خࣹلْ، تࣹ نِ نگَ سࣷݒّࣹ سَ بࣷݒّ، نَنگَ ڪࣷ سࣷݒّࣹ سَ مࣷرࣷومْ¤؞» ‏28 عِيسَىٰ نَگْ نࣹ ڪࣷ:‏ «وَخْ نگَ دࣴگّ؞ نࣷونُ نگَيْ دࣹفْ، دَلْدِ تࣹخࣹ؞» ‏29 وَا جَ نَگْ نَمَّ وُتُ لَيْ، نࣹ عِيسَىٰ:‏ «سَمَ مࣷرࣷومْ مࣷويْ ڪَنْ؟»
‏30 عِيسَىٰ نࣹيتِ ڪࣷ:‏ «جࣹنّ وَايَ جٗگࣹ وࣷونْ يࣹرُسَلࣹمْ، جࣴمْ يࣹرِڪࣷ؞ مُ دَجࣹيڪْ اَيْ سࣵݖّ، ݧُ فُتِّ ڪࣷ، دٗورْ ڪࣷ، دَلْدِ دࣹمْ، بࣵيِّ ڪࣷ، مُيْ وَاجَ دࣹي؞ ‏31 يࣵلَّ وࣷو يࣵلَّ اَبْ سَرْخَلْڪَتْ بُ جَارࣹ يࣷونْ وࣷووَ، سࣺينْ نِتْ ڪَ، دَلْدِ تࣹگِّ؞ ‏32 جࣹنّ وَايْ جُ سࣷقِڪࣷو ݖِ گِيرُگْ لࣹوِ اِتْ نࣷونُ؛ دَا دِڪّ بَ تࣷلُّ فࣷوفَ، سࣺينْ وَا جَ، دَلْدِ تࣹگِّ؞ ‏33 جࣹنّ وَايْ جُ دࣴڪّ سَمَرِ نَگْ دࣷونْ تُڪِّ، دِڪّ بَ جࣹگࣹ وَا جَ، خࣷولْ ڪࣷ، يࣴرࣴمْ ڪࣷ لࣷولْ؞ ‏34 مُ دِڪّ، سࣷتِّ دِوْ اَڪْ بِيݧْ ݖَ گَاݧُ-گَاݧُ يَ، لࣴمࣴسْ ڪࣷ؞ ݖِ ڪَوْ لࣷولُ مُ تࣹگْ ڪࣷ ݖِ مبَامَمْ، يٗبُّ ڪࣷ ݖِ اَبْ دَلُڪَايْ، دِ ڪࣷ تࣷݒّتࣷو؞ ‏35 ݖَ اࣴلّࣴگْ سَ، مُ يࣺڪَّتِ ݧَارِ ݒࣷسࣹتِ خَالِسْ¤، جࣷخْ ڪࣷ بࣷرࣷومْ دَلْ بَ، نࣹ ڪࣷ:‏ ‹تࣷݒّتࣷو ڪࣷ وَا جِ؞ لࣷو ݖِ دࣷلِّ، بُ مَ دࣴݒّࣹي، مَا لَ ڪࣷيْ دࣹلّࣷو؞›»
‏36 مُ تࣹگْ ݖَ نࣹ ڪࣷ:‏ «ݧࣹتّ ݧِي، ڪَنْ نگَ فࣷوگْ نࣹ مࣷو ݖِ تَخَوُ وَا جَ دَجࣹيڪْ سࣵݖّ يَ، تَخَوَايُ نِتْ اَڪْ مࣷرࣷومَمْ؟» ‏37 مُ نࣹ ڪࣷ:‏ «خَنَا ڪِ ڪࣷ دࣹفَلْ يࣴرْمَنْدࣹ؞» عِيسَىٰ نَگْ نࣹ ڪࣷ:‏ «دࣹمَلْ بࣷوگْ، دِ دࣹفࣹ نࣷونُ؞»
مَرْيَامَ رَوْ نَ مࣵرتّ
‏38 عِيسَىٰ اَڪْ اَيْ تَالِبࣹيمْ ݧࣷو تࣷݒّࣷونْ سࣹينُ يࣷونْ، بَ تࣷلّࣷوڪْ اَبْ دࣴڪّ؞ بَ ݧُ دِڪّࣹي، سࣹنّ اَسْ ندَوْ سُ ݧُيْ وَخْ مࣵرتّ، گَنَلࣹ ڪࣷ؞ ‏39 ݖِ ڪَوْ لࣷولُ رَڪَّمْ جُ ݧُيْ وَخْ مَرْيَامَ تࣷوگْ ݖَ وࣹتُ سَنگّ بِ، دِ دࣺگْلُ ڪࣵدّࣷومْ، ‏40 تࣹ فࣹڪّ مࣵرتّ جࣵݒّ لࣷولْ ݖَ لِگّࣺييُ وَاݧْ وَ؞ مُ دِڪّ نࣹ عِيسَىٰ:‏ «سَنگّ بِ، خَنَا گِسࣷو سَمَ رَڪّ جِ مَ بࣵيّࣹيڪْ لِگّࣺييُ وَاݧْ وِ، مَنْ دࣷݝّ؟ وَخْ ݖِ بࣷوگْ، مُ جࣵݒّلࣹ مَ؞» ‏41 سَنگّ بِ نࣹ ڪࣷ:‏ «مࣵرتّ، مࣵرتّ، يِتّࣹ يِ نگَ خِنْتࣹوࣷو، دِ ݖِ دࣷݧْ-دࣷݧِ مࣷو بَرࣹ؞ ‏42 ندَخَمْ لࣹنّ دࣷݝَّ مَنُلَ ݧࣵڪّ؞ مَرْيَامَ نَگْ، ݖࣺرْ بِ گࣴنْ لَ تَامُ، تࣹ دࣹيسُ ڪࣷ نَنگُ ݖِ مࣷومْ؞»
