23
‏1 بُ ڪࣹنّ جࣴلْ جَبَرُ بَايَمْ، دِ تࣹدَّدِلْ لَلُ بَايَمْ؞
اَمْ نَ ݧُ دُلْ بࣷڪّ ݖِ ندَجࣹمْ اَجِ سَخْ جِ
‏2 ڪُ ݧُ مࣷرْ مبَا ݧُ دَگّ نگٗورَامْ، دُ بࣷڪّ ݖِ ندَجࣹمْ گࣴمْڪَتِ اَجِ سَخْ جِ؞ ‏3 ڪُ دُلْ دࣷومُ يࣷونْ دُ بࣷڪّ ݖِ ندَجࣹمْ گࣴمْڪَتِ اَجِ سَخْ جِ؞ ڪُ اَسْڪَنࣷو ݖِ مࣷومْ اِتْ، بَ ݖَ فُڪّࣹيلُ مَاسْ گَ، دُ بࣷڪّ ݖِ ندَجࣹمْ گࣴمْڪَتِ اَجِ سَخْ جِ؞ ‏4 اَبْ اَمࣷنࣹينْ مبَا اَبْ مࣷوَبࣹينْ دُ بࣷڪّ ݖِ ندَجࣹمْ گࣴمْڪَتِ اَجِ سَخْ جِ؞ ڪُ اَسْڪَنࣷو ݖِ ݧࣷومْ اِتْ، بَ ݖَ فُڪّࣹيلُ مَاسْ گَ، دُ بࣷڪّ ݖِ ندَجࣹمْ گࣴمْڪَتِ اَجِ سَخْ جِ مُڪّ، ‏5 ندَخْ لَ ݧُ لࣹينْ دَبࣹوُلْ وࣷونْ اَوْ ݧَمْ، مبَا اَمْ ندࣷخْ، ݖَ يࣷونْ وَ، بَ نگࣹينْ گࣺنّࣹي مِسْرَ، خَنَا دِ فࣹيْ بَلاَمْ دࣷومُ بࣹوࣷرْ مَ بࣷڪّ ݒࣹتࣷرْ ݖَ اَرَمْ نَارِمْ، نگِرْ مُ سࣵبِّ لࣹينْ؞ ‏6 وَايࣹ سࣹينْ يࣵلَّ اَجِ سَخْ جِ مࣷو نَنگُوُلَ دࣺگْلُ بَلاَمْ؛ ڪࣵدُّگْ يࣵقࣹ گَ لَ لࣹينْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ سࣷݒِّلْ بَرْڪࣹ، ندَخْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ دَ لࣹينَ سࣷݒّ؞ ‏7 بُلࣹينْ سࣵڪُّ مُڪّ جࣵمُّ ݧࣷوݧُ مبَا سࣹينْ نجَرِݧْ، سࣹينْ گِيرُ دُندّ، بَ فࣵوّ؞
‏8 بُلࣹينْ بَݧْ اَبْ اࣹدࣷمࣹينْ، ندَخْ سࣹينْ مبࣷڪّ لَ؞ بُلࣹينْ بَݧْ نِتُ مِسْرَ، ندَخْ اَيْ دࣷخَنْدࣺيمْ نگࣹينْ وࣷونْ ݖَ سࣹينُمْ رࣺيوْ؞ ‏9 ݧِ سࣷقِڪࣷو ݖِ ݧَارِ خࣹيتْ يࣷويِي، ݖَ ݧࣹتّࣹيلُ مَاسْ گَ، تࣹ دِ سࣹينِ سࣴتَاتْ¤ نَگْ، سَݧْ نَݧࣷو بࣷڪّ ݖِ ندَجࣹمْ گࣴمْڪَتِ اَجِ سَخْ جِ؞
‏10 بُ نگࣹينْ گࣺنّࣹي بَ فࣵڪّ اَبْ دَلْ، نگِرْ جࣵنڪࣷونْتࣹ اَڪْ سࣹينِ نࣷونْ، نَنگࣹينْ سࣵمُّ ݖِ لࣺݒّ لُ آيْ؞ ‏11 سُ اَمࣹي ݖِ سࣹينْ بِيرْ ڪُ سࣹتُلْ، ندَخْ گࣺنْتُ گُدِّ گُ ڪࣷ دِڪَّلْ، نَ گࣺنّ دَلْ بِ، ݖَ بِتِ، تࣹ بُمُ دُگَّاتْ ݖِ بِيرْ دَلْ بِ؞ ‏12 بُ نگࣷونْ دِ جُبْسِ نَ سَنگُ، بَ جَنتّ بِ سࣷ، مُ دࣷورَ دࣺلْسِ ݖِ بِيرْ دَلْ بِ؞ ‏13 تࣹ سࣵڪّلࣹينْ بࣺرَبْ بُ گࣺنّ دَلْ بِ، نگࣹينْ دِ فَ دࣹمْ، دِ سُتُرْلُجِ؞ ‏14 نَ نگَسْڪَ بࣷڪّ ݖِ سࣹينِ فࣵگُّوَايْ، ندَخْ سُ نگࣹينْ سُتُرْلُجࣹي ݖَ بِتِ بَ نࣷݒِّ، مࣷومْ نگࣹينْ دِ گَسࣹ، بَ سُولْ سࣹينِ جࣷنڪَنْ؞ ‏15 ندَخْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ مࣷويْ دࣹمْ اَڪَ دِڪّ ݖِ سࣹينْ بِيرْ دَلْ بِ، نگِرْ خࣹتّلِ لࣹينْ، بَ جࣺبَّلْ لࣹينْ سࣹينِ نࣷونْ؞ ڪࣷنْ نَگْ سࣹينُبْ دَلْ دَا وَرَ سࣹلّ، نگِرْ اَجِ سَخْ جِ ݧࣵڪَّ گِسْ فِ سࣹينْ بِيرْ، لࣹنّ لُ تࣹگِّنࣹوُلْ، بَ دࣴدُّ لࣹينْ؞
‏16 گَنَّاوْ لࣷولُ بُلࣹينْ دࣹلّࣷو اَبْ جَامْ بُ لࣵقُ ݖِ يࣹينْ، تࣹ رࣴݖّࣹ ݖِ اَبْ سَنگَمْ؞ ‏17 ݖِ يࣹينْ لَيْ تࣷوگْ، ݖِ سࣹينْ بِيرْ، بࣺرَبْ بَ مُ تَامُ، ݖِ بࣹنّ بِ ڪࣷ گࣴنَلْ ݖِ سࣹينِ دࣴڪّ، تࣹ بُلࣹينْ ڪࣷ نࣺيوَلْ دࣷولࣹ؞
‏18 گࣵنݖّ بُ جِگࣺينْ بُيْ گࣵنݖّتُلْ اَيْ تُورْ، بُمُ اَمْ ݖِ بِيرْ جِگࣺينِ إِسْرَاءِيلْ، تࣹ گࣵنݖّ بُ گٗورْ بُيْ گࣵنݖّتُلْ اَيْ تُورْ¤، بُمُ اَمْ ݖِ گٗورِ إِسْرَاءِيلْ؞ ‏19 بُلࣹينْ اِنْدِ مُڪّ ݖِ سࣹينْ بِيرْ ڪࣴرْ يࣵلَّ اَجِ سَخْ جِ، ݒࣹيࣷورُ گࣵنݖّ بُ جِگࣺينْ، مبَا بُ گٗورْ، نگِرْ وࣺݒّ خَسْ وُ نگࣹينْ نَمَّ وࣵݖّࣷو گࣷوگُ ݒࣹيࣷورْ، ندَخْ يࣷويُ يَارْ يࣺݒّ، سࣹينْ يࣵلَّ اَجِ سَخْ جِ سࣹيخْلُ نَ ڪࣷ؞
دࣹيسُلْ تࣷݒّ بࣵنِّ إِسْرَاءِيلْ اَبْ تࣹگَنْدَايْ
‏20 بُلࣹينْ تࣹگْ سࣹينْ مبࣷڪُّمْ بࣵنِّ إِسْرَاءِيلْ تࣹگَنْدَايَلْ خَالِسْ، مبَا تࣹگَنْدَايَلْ ݧَمْ، مبَا تࣹگَنْدَايَلْ مبࣷولࣹمْ بࣷرْ بُ تࣹگَنْدَايْ تࣹگࣹ؞ ‏21 اَبْ دࣷخَنْدࣺيمْ، سَݧْ نگࣹينْ ڪࣷو سࣵڪُّ تࣹگَنْدَايْ، وَايࣹ سࣹينْ مبࣷڪُّمْ بࣵنِّ إِسْرَاءِيلْ، بُلࣹينْ ڪࣷ سࣵڪُّ تࣹگَنْدَايْ؞ سُ ڪࣷ دࣹفࣹي سࣹينْ يࣵلَّ اَجِ سَخْ جِ مࣷويْ بَرْڪࣹيلْ مبࣷولࣹمْ سࣹينُ ݧَقْ، ݖَ رࣺيوْ مَ نگࣹينْ جࣴمْ، نگِرْ نَنگُجِ ڪࣷ؞
وࣵݖّࣷوڪْ اَوْ خَسْ وَرْتࣺيفْ لَ
‏22 سُ نگࣹينْ خَسَلࣹي سࣹينْ يࣵلَّ اَجِ سَخْ جِ، اَوْ خَسْ، بُلࣹينْ ڪࣷ يࣹيخَ وࣵݖّࣷولْ، ندَخْ ڪࣷنْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ دُ لࣹينْ ڪࣷ ݧࣵڪَّ تࣷݒّ، تࣹ ڪࣷنْ مُيْ سࣹينْ بࣵڪَّارْ؞ ‏23 وَايࣹ بَݧَ خَسْ، اَمُلْ بࣵڪَّارْ؞ ‏24 لِ سࣹينْ لࣵمِّݧْ تُدّ، سࣵمّࣷولࣹينْ ڪࣷو جࣴفࣹ نَ نگࣹينْ ڪࣷ خَسَلࣹ وࣷونْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ، ݖِ سࣹينْ سَگࣷيْ بࣷݒّ، اَڪْ سࣹينْ ڪࣵدُّگْ بࣷݒّ؞
رَاسَاتُ مُ سُورْ بࣷرࣷومْ لࣹوْ نَ
‏25 بُ نگࣹينْ دُگّࣹي ݖِ سࣹينْ تٗوڪࣴرُ رࣹسࣹݧُ مࣷرࣷومْ، لࣹڪّلࣹينْ ݖَ رࣹسࣹݧْ جَ لُ لࣹينْ نࣹيخْ بَ سُورْ، وَايࣹ سࣹينْ ندَبْ، بُلࣹينْ ڪࣷ ݖَ يࣹبْ؞ ‏26 بُ نگࣹينْ دُگّࣹي ݖِ سࣹينْ تࣷولُ ݒࣹݒُّ مࣷرࣷومْ، فࣹڪّ مُ ݧࣷرْ، سࣵڪّࣹلࣹينْ سࣹينْ لࣷخࣷ ݖَ گُبْ يَ، وَايࣹ اَبْ سࣵرتّ، بُلࣹينْ ڪࣷ خࣵݖِّلْ سࣹينْ ݒࣹݒُّ مࣷرࣷومْ؞
