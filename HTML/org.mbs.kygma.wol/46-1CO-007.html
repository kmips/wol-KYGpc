لُ جࣴمْ ݖِ مبِرُمْ سࣺيْ
7
‏1 سَمَ تࣷنتّ ݖِ مبِرْ مِ نگࣹينْ مَ بِندّ نَگْ، مࣷو دِ لُ بَاخْ لَ ݖِ گٗورْ، مُ مُݖّ ݖِ جِگࣺينْ¤؞ ‏2 وَايࣹ مࣷيْتُ ݒࣷوُمْ سࣺيْ وَرْ نَا تَخْ بَ گٗورْ گُ نࣹڪّ اَمْ جَبَرَمْ، جِگࣺينْ جُ نࣹڪّ اِتْ، اَمْ جࣴڪّࣴرَمْ؞ ‏3 جࣴڪّࣴرْ جِ وَرْ نَا دࣹفْ وَرُگَرُ جࣴڪّࣴرْ ݖِ جَبَرَمْ؛ نَڪَ نࣷونُ اِتْ، جَبَرْ جِ وَرْ نَا دࣹفْ وَرُگَرُ جَبَرْ ݖِ جࣴڪّࣴرَمْ؞ ‏4 جِگࣺينْ جِ مࣷومُلْ يَرَمَمْ، مࣷومْ دࣷݝّ، وَايࣹ جࣴڪّࣴرْ جِ اِتْ اَمْ نَ ݖِ سَݧْ-سَݧْ؞ نَڪَ نࣷونُ اِتْ، جࣴڪّࣴرْ جِ مࣷومُلْ يَرَمَمْ، مࣷومْ دࣷݝّ، وَايࣹ جَبَرْ جِ اِتْ اَمْ نَ ݖِ سَݧْ-سَݧْ؞ ‏5 بُلࣹينْ دِ خَݧَنْتࣹ سࣹينْ بࣷݒّ، ندَرࣹ دُ يࣹينَ مࣵنڪࣷو اَبْ دِيرْ، نگِرْ بࣷڪّ تࣴنڪُ ݖِ ݧَانْ؞ وَايࣹ گَنَّاوْ لࣷولُ، دࣹلُّلࣹينْ دِ اࣵندّ، بَلَ لࣹينَ سࣹيْتَانࣹي خِيرْتَلْ ݖِ مࣷيْ، ندَخْ تࣴلࣹي تࣺيࣹ سࣹينْ بࣷݒّ؞ ‏6 لࣷولُ مَ لࣹينْ وَخْ نَگْ مࣷويْ لِ لࣹينْ يࣷونْ مَيْ، وَايࣹ دُ ندِگَلْ؞ ‏7 ندَخْتࣹ سُ مَ سَݧࣷونْ ڪَيْ ݧࣺݒّ تࣷوگْ نِ مَنْ؛ وَايࣹ ڪُ نࣹڪّ اَڪْ جَگْلࣹبْ بࣷݒَّمْ بُ ڪࣷ يࣵلَّ بَاخࣹ، ڪِي لِي، ڪࣹي لࣹي؞
‏8 لِ مَيْ وَخْ گࣴمْڪَتْ ݧِ سࣺقُلْ اَڪْ ڪࣹنّ اَبْ سࣺيْ، اَڪْ ݧِ سࣹينْ جࣴڪّࣴرْ فَاتُ دَالْ مࣷو دِ، سُ ݧُ وࣺيࣹي تࣷوگْ نࣷونُ نِ مَنْ مِي، لُ بَاخْ لَ ݖِ ݧࣷومْ؞ ‏9 وَايࣹ سُ ݧُ مَنُلَ مَانْدُ، نَݧُ سࣺيْ، ندَخْ سࣺيْ مࣷو گࣴنْ سࣹينْ خࣺرَانگࣹيْ بَڪَّنْ خَبْتَلْ لࣹينْ؞
‏10 ݧِ سࣺقْ اَبْ سࣺيْ نَگْ، مَا لࣹينْ دࣺنڪّ لِي، تࣹ دُ سَمَ ندِگَلْ، ندِگَلَلْ بࣷرࣷومْ بࣹي؞ جَبَرْ جِ، بُمُ تࣹقْلِڪࣷو اَڪْ جࣴڪّࣴرَمْ؞ ‏11 تࣹ بُ تࣹقْلِڪࣷو اَڪْ جࣴڪّࣴرَمْ اِتْ، نَ تࣷوگْ، بَݧَ سࣺيَاتْ، مبَا مُ جُبࣷو اَڪْ جࣴڪّࣴرَمْ؞ تࣹ جࣴڪّࣴرْ جِ اِتْ بُمُ فَسࣹ جَبَرَمْ؞
‏12 ݧِ ݖِ دࣹسْ نَگْ، لِ مَ خَلاَتْ تࣹ دُ بࣷرࣷومْ بࣹي ڪࣷ وَخْ، مَا ڪࣷ وَخَلْ سَمَ بࣷݒّ، مࣷو دِ:‏ اَبْ گࣴمْڪَتُبْ اَلْمَسِ، بُ اَمࣹي جَبَرْ جُ گࣴمُلْ، تࣹ تࣹيوُلْ مُ نَنگُ ڪࣷو سࣺيَلْ بَ تࣹيْ، بُمُ ڪࣷ فَسࣹ؞ ‏13 اَبْ گࣴمْڪَتُبْ اَلْمَسِ اِتْ، بُ اَمࣹي جࣴڪّࣴرْ جُ گࣴمُلْ، تࣹ تࣹيوُلْ مُ نَنگࣷو نࣹڪّ اَڪْ مࣷومْ، بُمُ تَسْ؞ ‏14 ندَخْتࣹ جࣴڪّࣴرْ جِ گࣴمُلْ دَيْ دَلْدِ سࣺدُّ سࣹلّنگَ ندَخْ جَبَرَمْ جَ دِبْ گࣴمْڪَتْ، تࣹ جِگࣺينْ جِ گࣴمُلْ اِتْ دَيْ سࣺدُّ سࣹلّنگَ ݖَ جࣴڪّࣴرَمْ جَيْ گࣴمْڪَتْ؞ سُ دُلْ وࣷونْ نࣷونُ دࣷومْ يُ نگࣹينْ ݖَ اَمْ دِ ݧُ سࣷبࣹوُ، تࣹ بِرْ نَ نࣹ دࣷومْ يُ نگࣹينْ ݖَ اَمْ، ݧُ سࣹلّ لَݧُيْ دࣷونْ؞ ‏15 ندࣹگَمْ نَگْ ڪِ گࣴمُلْ دَا بࣴگَّ دࣹمْ، نَ دࣹمْ؞ سُ دࣹمࣹي نࣷونُ، مبࣷڪُّمْ گࣴمْڪَتْ بُ گٗورْ بِ، مبَا ڪُ جِگࣺينْ ڪِ اَمَتُلْ بࣹنّ گࣵلَّنڪࣷورْ، وَايࣹ ݖِ جࣵمّ لَ لࣹينْ يࣵلَّ وࣷو؞ ‏16 يَوْ مِيْ جَبَرْ جِ نَگْ سَ تࣹخࣹيْ جࣴڪّࣴرْ مَنْ نَا جَارࣹ ݖِ يَوْ، جࣷمْبُلْ؛ يَوْ جࣴڪّࣴرْ جِ اِتْ، سَ تࣹخࣹيْ جَبَرْ جَارࣹ ݖِ يَوْ، جࣷمْبُلْ؞ ‏17 لُ ڪࣷ مࣷيْ نَگْ، ڪُ ݖِ نࣹڪّ، ݖࣺرْ بَ لَ بࣷرࣷومْ بِ سࣺدّࣷونْ، نࣷونࣹي نگَ نࣹڪّࣹ وࣷونْ بَ لَ يࣵلَّ دِ وࣷو، نَنگَ ڪࣷ وࣺيࣹ؞ لࣷولُ لاَ سَنْتَانࣹ ݖِ مبࣷولࣷويْ گࣴمْڪَتْ ݧࣺݒّ؞ ‏18 ڪِ خَرَفࣷونْ بَلَ ݧُ ڪࣷيْ وࣷو، بُمُ لࣵقْ مࣵنْدَرْگَمْ خَرَفْ مَ؞ ڪِ خَرَفُلْ وࣷونْ بَ ݧُ ڪࣷيْ وࣷو، بُمُ خَرَفْ؞ ‏19 خَرَفْ اَمُلْ سࣷلࣷ، ݧࣵڪَّ خَرَفْ اَمُلْ سࣷلࣷ؞ لِ اَمْ سࣷلࣷ مࣷو دِ تࣷݒّ ندِگَلِ يࣵلَّ؞ ‏20 ڪُ نࣹڪّ، نࣹڪِّنْ وِ نگَ نࣹڪّࣹ وࣷونْ بَ ݧُ لَيْ وࣷو، نَنگَ ڪࣷ وࣺيࣹ؞ ‏21 ندَخْ اَبْ جَامْ نگَ وࣷونْ، بَ ݧُ لَيْ وࣷو؟ اَمُلْ سࣷلࣷ؞ وَايࣹ سࣷو مَنࣹي يِوِڪُ، يِوِڪُلْ؞ ‏22 ڪِ ݧُ وࣷو ݖِ سَنگّ بِ، فࣹڪّ ڪࣷ مُ دِبْ جَامْ، مࣷويْ ڪِ سَنگّ بِ گࣷرࣹيلْ؞ نَڪَ نࣷونُ، گࣷرْ سِ ݧُ وࣷو، اَبْ جَامُ اَلْمَسِ لَيْ دࣷونْ؞ ‏23 گَنَّاوْ فࣹيࣹيسْ نَ سࣹينْ نجࣺگْ بَ جࣷتْ لࣹينْ نَگْ، بُلࣹينْ دࣷونْ جَامُبْ نِتْ؞ ‏24 بࣷڪّ يِ، ڪُ ݖِ نࣹڪّ دَالْ، نࣹڪِّنْ وَ نگَ نࣹڪّࣹ وࣷونْ بَ ݧُ لَيْ وࣷو، وࣺيࣹ ڪࣷ رࣹڪّ فِ ڪَنَمْ يࣵلَّ؞
‏25 ݖِ وࣵلُّ جَنقّ جِ سࣺيَگُلْ، اَوْمَ ݖِ ندِگَلُ بࣷرࣷومْ بِ، خَنَا مَ وَخْ ݖِ سَمَ خَلاَتْ بُ مَتَ گࣴمْ، ندَخْ يࣴرْمَنْدࣹ جِ مَ بࣷرࣷومْ بِ بَاخࣹ؞ ‏26 لِي مࣷويْ لِ مَ فࣷوگْ نࣹ مࣷويْ لِ بَاخْ، ندَخْ نجࣵقَرࣹيْ جَمࣷنࣷيْ تࣹيْ؞ لِ بَاخْ ݖِ نِتْ ڪِ دَالْ مࣷويْ مُ وࣺيࣹ لِ مُيْ دُندّ تࣹيْ؞ ‏27 دَنگَا اَمْ جَبَرْ؟ بُلْ سࣵڪُّ ݒَسࣹ؞ اَمْ اَمُلࣷو جَبَرْ؟ ڪࣷنْ بُلْ سࣵڪُّ جَبَرْ؞ ‏28 وَايࣹ نَگْ سࣷو تَڪّࣹي جَبَرْ، بࣵڪَّارُلࣷو؛ اَبْ جَنقّ اِتْ، بُ سࣺيࣹي، بࣵڪَّارُلْ؞ وَايࣹ ݧِ سࣺقْ اَبْ سࣺيْ دِنَݧُ اَمْ نَقَرُ اࣵدِّنَ، تࣹ نَقَرْ وࣷووُ لاَ لࣹينْ بࣴگّࣷونَ فَنقَلْ؞
‏29 لِ مَيْ وَخْ، بࣷڪّ يِ، مࣷو دِ، لِ دࣹسْ يࣵگَّتُلْ؞ ڪࣷنْ نَگْ بࣺسْ نِڪِ تࣹيْ، نَ ݧِ اَمْ جَبَرْ مࣹلْ نِ بُ ݧُ ڪࣷ اَمُلْ وࣷونْ؛ ‏30 ݧِيْ جࣷويْ مࣹلْ نِ ݧُ جࣷويُلْ؛ ݧِ بࣺگْ، مࣹلْ نِ ݧُ بࣺگُلْ؛ ݧِ جࣴندّ، مࣹلْ نِ ݧُ مࣷومُلْ لِ ݧُ جࣴندّ؛ ‏31 ݧِيْ جَرِݧࣷو اَلَلُ اࣵدِّنَ، بَݧْ ݖࣹي خࣺرْ، ندَخْ مࣹلࣷوْ اࣵدِّنَ سِي تࣹيْ، دَيْ وࣹيسْ؞
‏32 سَمَ ݖࣷوبَرࣹ مࣷويْ، يࣹينْ گࣴمْڪَتِ اَلْمَسِ، نگࣹينْ مُݖّ ݖِ خَلاَتْ يُ اُبْ سࣹينْ بࣷݒّ؞ تࣹ ڪُ اَمُلْ جَبَرْ، مبِرِ بࣷرࣷومْ بِ لَيْ خَلاَتْ، اَڪْ نُ مُيْ دَبࣹ بࣵنّࣹيخُ بࣷرࣷومْ بِ؞ ‏33 وَايࣹ ڪُ اَمْ جَبَرْ، مبِرِ اࣵدِّنَ لَيْ خَلاَتْ، اَڪْ نُ مُيْ دَبࣹ بࣵنّࣹيخُ جَبَرَمْ، ‏34 مُ دَلْدِ ݝَرَلࣹ؞ جِگࣺينْ جُ سࣺيُلْ، مبَا اَبْ جَنقّ، مبِرِ بࣷرࣷومْ بِ لَيْ خَلاَتْ، دِ نَمَّ سࣹلَّلْ يَرَمَمْ اَڪُمْ خࣹلَمْ؞ وَايࣹ ڪِيْ سࣺيْ، مبِرِ اࣵدِّنَ لَيْ خَلاَتْ، اَڪْ نُ مُيْ دَبࣹ بࣵنّࣹيخُ جࣴڪّࣴرَمْ؞ ‏35 لِي نَگْ، سࣹينْ نجَرِݧَ مَ ڪࣷ تَخَ وَخْ، نَمُّمَ لࣹينْ ڪࣷو تࣹگࣹ اَبْ گࣵلَّنڪࣷورْ، لِ مَ ݖِ نَمّ مࣷويْ نگࣹينْ دِ دࣹفْ لِ جࣹڪّ، تࣹ تَخَوَلْ تࣹمّ بࣷرࣷومْ بِ، تَخَوَايْ بُ لࣹينْ بࣹنّ خَلاَتُ اࣵدِّنَ دُلْ فࣺلࣹ؞
‏36 وَايࣹ سُ اَمࣹي ڪُ جࣵݒّ نࣹ جࣴفࣹوُلْ نُ جࣹڪّ، اَڪْ جَنقّ بُ مُ دِگࣷولْ اَبْ سࣺيْ، تࣹ جَنقّ بَ جࣴمْ مَگْ¤، بَ مُ خَلاَتْ نࣹ تَڪّ ڪࣷ رࣹڪّ مࣷو ݖِ وَرْ، نَ مَتَلْ يࣺينࣹيمْ، تَڪّ ڪࣷ؞ سُ دࣹفࣹي نࣷونُ، بࣵڪَّارُلْ؞ ‏37 وَايࣹ ڪِ دࣷگُ دࣴگّ ݖِ خࣷلَمْ، تࣹ دُ ندِگَلُ گࣵنْتَلْ لُ مُ سَݧُلَ بَݧْ، خَنَا ݖِ سَگࣷيْ بࣷݒَّمْ دࣷݝّ لَ دࣷگࣷو نࣹ دُ جࣴلْ جَنقّ بِ مُ دِگࣷولࣷونْ اَبْ سࣺيْ، ڪࣷوڪُ بَ تࣹيْ، دࣹفْ نَ لُ بَاخْ؞ ‏38 ڪࣷنْ ڪِ جࣴلْ اَبْ جَنقَمْ بُ مُ دِگࣷولْ اَبْ سࣺيْ، دࣹفْ نَ لُ بَاخْ، وَايࣹ ڪِ جࣴلُلْ اَبْ جَنقَمْ مࣷو دࣹفْ لِ گࣴنْ؞
‏39 جَبَرْ نَگْ، بُومُ جࣴڪّࣴرَمَ نگِ ݖِ بَاتَمْ، لِ فࣹيڪْ جࣴڪّࣴرَمَ نگِ دُندّ؞ وَايࣹ سُ جࣴڪّࣴرْ جِ فَاتࣷو، سَݧْ نَا سࣺيْ اَڪْ ڪُ ڪࣷ نࣹيخْ، ندࣹگَمْ گࣴمْڪَتُبْ اَلْمَسِ لَ سࣺيَلْ؞ ‏40 دࣷونْ نَ گࣴنَ بࣺگْ نَگْ، سُ تࣷوگࣷونْ، بَݧَ سࣺيَاتْ، ݖِ سَمَ خَلاَتْ، تࣹ مَنْ اِتْ دࣹفࣹ نَا نࣹ نࣷووُگْ يࣵلَّ مُ نگِ ݖِ مَنْ؞
