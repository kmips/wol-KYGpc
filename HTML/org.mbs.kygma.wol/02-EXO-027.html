لُ جࣴمْ ݖِ سَرْخَلُڪَايْ بِ
27
‏1 «نَنگَ دࣹفَرࣹ سَرْخَلُڪَايْ بِ بَنْتُ سࣺنگّ، گُدَّايْ بِ دِ جُرٗومِ خَسَبْ، يَاتُوَايْ بِ دِ جُرٗومِ خَسَبْ؛ اَبْ ڪَارࣹ لَ سَرْخَلُڪَايْ بِيْ دࣷونْ، تَخَوَايَمْ دِ ݧࣹتِّ خَسَبْ؞ ‏2 دࣹفَرَلْ ڪࣷ اَيْ بࣺجّࣺنَمْ ݖِ ݧࣹينْتِ ڪࣷݧْ يِ ݖِ ڪَوْ؞ بࣺجّࣺنْ يِ، دࣹيسْ ڪࣷيْ لِگّࣺييَالࣹيڪْ سَرْخَلُڪَايْ بَ، ݧُ اࣵندّ دِ بࣹنّ، تࣹ نَنگَ خࣷوبْ سَرْخَلُڪَايْ بِ خࣵنجَرْ؞ ‏3 سࣵڪَّلْ اَيْ ندَبِ تِبُّڪَايُ دٗومَمْ، اَڪْ اَيْ ݧِيتُڪَايَمْ، اَڪْ اَيْ ڪࣴلَّمْ، اَڪْ اَيْ ݖَݒُّڪَايَمْ اَڪِ اَنْدَمْ؞ دࣹفَلْ جُمْتُڪَايَمْ يࣺݒّ خࣵنجَرْ؞ ‏4 سࣵڪَّلْ ڪࣷ ݖَاخْ بُ ݧُ تࣴگّࣹ خࣵنجَرْ تࣹ نگَ سࣵڪَّلْ ݖَاخُ خࣵنجَرْ بِ ݧࣹينْتِ نࣷݒّ ݖِ ݧࣹينْتِ ݖࣷلَّمْ؞ ‏5 تࣹگَلْ ݖَاخْ بِ، مُ رࣷنُ ݒࣹگُّ سَرْخَلُڪَايْ بِ، تࣹ دَلࣹ ݖِ سُوفْ بَ ݖِ دِگّ سَرْخَلُڪَايْ بِ؞ ‏6 سࣵڪَّلَلْ سَرْخَلُڪَايْ بِ اَيْ نجࣵݒُّيْ بَنْتُ سࣺنگّ، تࣹ نگَ خࣷوبْ لُ ݖِ نࣹڪّ خࣵنجَرْ، ‏7 تࣹ نگَ رࣷوفْ نجࣵݒُّ يِ ݖِ نࣷݒّ يِ، سُ ڪࣷ دࣹفࣹي نجࣵݒُّ يِ دࣹسْ ݖِ ݧَارِ وࣹتِ سَرْخَلُڪَايْ بِ، نگِرْ بُ ݧُ ڪࣷيْ يٗبُّ؞ ‏8 ݒَخْ نگَيْ دࣹفْ سَرْخَلُڪَايْ بِ ݖِ بِيرْ؛ نگَ دࣹفَرࣹ ڪࣷ اَيْ خࣵنقّ يُ تࣵݒَّنْدَارْ؞ نَ ݧُ لَ ڪࣷ وࣷنࣹ ݖَ ڪَوْ تُندّ وَ رࣹڪّ لَݧُ ڪࣷيْ دࣹفࣹ؞
لُ جࣴمْ ݖِ اࣴتُّ جَامࣷوڪَايْ بِ
‏9 «نَنگَ سࣵڪَّلَلْ جَامࣷوڪَايْ بِ اَبْ اࣴتّ:‏ گُدَّايُ اࣴتّ بِ ݖِ وࣹتُ بࣴجْ-سَالُمْ، تࣺيمࣺيرِ خَسَبِ سࣷرِ رَبَلْ يُ ݧُ رࣵبّࣹ وࣴݧُ لࣹي مࣷو ڪࣷيْ اُبْ، ݖِ گࣹنّ وࣹتْ گࣷوگُ؞ ‏10 جࣴنِ جَامࣷوڪَايْ بِ ݧَارْ فُڪّ لَيْ دࣷونْ، سࣹينْ ݧَارْ فُڪِّ تࣹگُ دِ خࣵنجَرْ، وࣺڪُّڪَايْ يࣹيڪْ گَلَنِ جࣴنْ يِ دِ خَالِسْ؞
‏11 «نَڪَ نࣷونُ اِتْ ݖِ وࣹتُ بࣴجْ-گࣵنَّارْ، گُدَّايْ بِ دِ تࣺيمࣺيرِ خَسَبِ سࣷرِ رَبَلْ، اَڪْ سࣹينْ ݧَارْ فُڪِّ جࣴنْ يُ سࣹينْ ݧَارْ فُڪِّ تࣹگُيْ خࣵنجَرْ رࣷنُ، وࣺڪُّڪَايْ يࣹيڪْ گَلَنِ جࣴنْ يِ دِ خَالِسْ؞
‏12 «يَاتُوَايُ اࣴتّ بِ، فِ فࣺيتࣹ سࣷوُ، جُرٗومْ فُڪِّ خَسَبِ سࣷرِ رَبَلَ ڪࣷيْ اُبْ، سࣹينِ جࣴنْ دِ فُڪّ، سࣹينِ تࣹگُ دِ فُڪّ؞
‏13 «يَاتُوَايُ اࣴتّ بِ، فِ فࣺيتࣹ ݒࣹنڪُ ݖِ ڪَنَمْ، جُرٗومْ فُڪِّ خَسَبْ لَ:‏ ‏14 فُڪِّ خَسَبِ سࣷرِ رَبَلْ اَڪْ جُرٗومْ مࣷو اُبْ اࣴتّ بِ ݖِ گࣷوگُ وࣹتْ، سࣹينِ جࣴنْ دِ ݧࣹتّ، سࣹينِ تࣹگُ دِ ݧࣹتّ؞ ‏15 وࣹتْ گِ ݖِ دࣹسْ اِتْ، فُڪِّ خَسَبِ سࣷرِ رَبَلْ اَڪْ جُرٗومْ، سࣹينِ جࣴنْ دِ ݧࣹتّ، سࣹينِ تࣹگُ دِ ݧࣹتّ؞ ‏16 بُنْتُ اࣴتّ بِ نَگْ، نَ دࣷونْ لَفُ ݧَارْ فُڪِّ خَسَبْ، وُ ݧُ دࣹفَرࣹ وࣴݧْ گُ بَخَ تࣹ لاَلْ يࣷلࣹتْ، اَڪْ وࣴݧْ گُ خࣹوَرْ اَڪْ گُ خࣷنقّ ݖُرّ اَڪْ وࣴݧُ لࣹي، مُيْ لِگّࣺييُ رࣵبّ بُ رَفࣹتْ، اَيْ جࣴنَمْ دِ ݧࣹينتّ، سࣹينِ تࣹگُ دِ ݧࣹينتّ؞
‏17 «جࣴنْ يَ وࣴرْ اࣴتّ بَ يࣺݒّ، دࣹيسْ ڪࣷيْ لࣴڪّلࣹي گَلَنْ يُ خَالِسْ، وࣺڪُّڪَايِ جࣴنْ يِ دِ خَالِسْ، سࣹينِ تࣹگُ دِ خࣵنجَرْ؞
‏18 «گُدَّايُ اࣴتّ بِ تࣺيمࣺيرِ خَسَبْ لَيْ دࣷونْ، يَاتُوَايْ بِ دِ جُرٗومْ فُڪّ، تَخَوَايُ وࣹتْ يِ دِ جُرٗومِ خَسَبْ، سࣷرْ يِ ݧُ رࣵبّࣹ لࣺݒّ دِ وࣴݧُ لࣹي، تࣹگُ يِ دِ خࣵنجَرْ؞ ‏19 مبࣷولࣹمْ جُمْتُڪَايِ يِ ݧُيْ لِگّࣺييࣹ ݖِ جَامࣷوڪَايْ بِ، بَ ݖِ مبࣷولࣹمْ خࣹرِ خَيْمَامْ، اَڪْ مبࣷولࣹمْ خࣹرِ اࣴتّ بِ، لࣺݒّ خࣵنجَرْ لَيْ دࣷونْ؞
لُ جࣴمْ ݖِ سَنْتَانࣹ، يِ سَرْخَلْڪَتْ يِ سَسࣷو
لُ جࣴمْ ݖِ دِوُ لࣵمݒّ بِ
‏20 «يَوْ نَگْ، سَنْتَلْ بࣵنِّ إِسْرَاءِيلْ، ݧُ اِنْدِلْ لَ دِوُ سࣹگَلُ اࣷلِوْ گُ سࣹتْ گُ ݧُيْ تَالࣷو، نگِرْ لࣵمݒّ يِ دࣴڪّࣹ تࣵڪّ؞ ‏21 ݖِ بِيرْ خَيْمَبْ ندَجࣹ مِ¤ لَ هَـٰرُونَ اَڪْ دࣷومَمْ يُ گٗورْ دِ تࣹگْ لࣵمݒّ يِ، ݖِ ڪَنَمْ رِدࣷ بِ لࣵقْ گَالُ اࣵلُّوَيْ سࣹيدࣹ سِ، ݧُيْ فَنَانࣹي تࣵڪّ نگࣷونْ بَ بࣴتْ سࣹتْ، فِ ڪَنَمْ اَجِ سَخْ جِ؞ دࣷگَلَ نگࣷوگُ، وَرْ بࣵنِّ إِسْرَاءِيلْ فࣵوّ، ݧࣷومْ اَڪْ سࣹينْ اَسْڪَنْ؞
