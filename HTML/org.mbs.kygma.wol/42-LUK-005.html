عِيسَىٰ وࣷو نَ اَيْ تَالِبࣹ
5
‏1 عِيسَىٰ مࣷو تَخَوࣷونْ ݖَ تࣵڪَّلْ دࣹخُ سࣹنࣹسَرࣹتْ¤، مبࣷولࣷو مَ تَنݖَلࣷو ݖِ ڪَوَمْ، دِ دࣺگْلُ ڪࣵدُّگْ يࣵلَّ؞ ‏2 ݖِ بِيرْ لࣷولُ مُ گِسْ ݧَارِ گَالْ يُ تَخَوْ ݖِ تࣵڪّ گِ، نَݒّڪَتْ يَ ݖَ وࣵݖّࣹ، دِ فٗوتْ سࣹينِ مبَالْ؞ ‏3 مُ دُگّ ݖِ گࣹنّ گَالْ گِ، مُيْ گُ سِمࣷݝْ؞ مُ ݧَانْ ڪࣷ مُ بࣴمࣴخْ بَ دࣵندّ تࣹفࣹسْ گِ اَسْ لࣴفْ؞ مُ دࣹفَلْ ڪࣷ نࣷونُ، عِيسَىٰ تࣷوگْ، تࣵمْبَلࣹي جࣵنگَلْ مبࣷولࣷو مَ فَ مُ تࣷلُّ ݖَ بِيرْ گَالْ گَ؞
‏4 نَڪَ لَ جࣵنگّلࣹ بَ نࣷݒِّ، نࣹ سِمࣷݝْ:‏ «بࣴمࣴخَلْ بَ ݖَ خٗوتࣹ بَ، تࣹ نگࣹينْ سࣵنِّ سࣹينِ مبَالْ بَ جࣵݒّ؞» ‏5 سِمࣷݝْ نࣹ ڪࣷ:‏ «نجَاتِگࣹ، ݖࣷونࣷ لَنُ فَنَانࣹ بِيگْ تࣹ جࣵݒُّنُ دَرَ؞ خَنَا مَ سࣵنِّ ڪࣷ رࣹڪّ ݖِ سَ ندِگَلْ؞»
‏6 نَڪَ لَݧُ سࣵنِّ مبَالْ يِ، دَلْدِ جࣵݒّ جࣴنْ يُ بَرࣹي بَرࣹ، بَ مبَالْ يَ دِ خࣴتّ؞ ‏7 بَ لࣷولُ اَمࣹي ݧُ لِيَارْ سࣹينْ مࣷرࣷومْ، يَ ݖَ گࣹنࣹينْ گَالْ گَ، نگِرْ ݧُ دِمْبَلِسِ لࣹينْ؞ ݧُ دِڪّ، يࣹبَنْدࣷوڪْ ݧࣷومْ، فࣹيسَلْ ݧَارِ گَالْ يࣺݒّ، بَ ݧُيْ بࣴگَّ سُوخْ؞ ‏8 سِمࣷݝْ ݒِيࣹيرْ نَگْ گِسْ لࣷولُ، دَلْدِ سُڪّ فَ ڪَنَمْ عِيسَىٰ، نࣹ ڪࣷ:‏ «سَنگّ بِ، نگَلَّ سࣷرࣹ مَ، نگِرْ مَنْ نِتُ بࣵڪَّارْڪَتْ لاَ!‏»، ‏9 ندَخْ تِيتَانگࣹ لُ ڪࣷ جࣵݒّ، مࣷوڪْ مبࣷولࣹمْ ݧَ مُ اࣵنْدَلْ، ݖَ جࣴنْ يُ بَرࣹ يَ ݧُ جࣵݒّ؞ ‏10 يَعْقُوبَ اَڪْ يࣷوَانْ دࣷومِ سࣹبࣹدࣹ يَ مُ بࣷڪَّلࣷونْ لِگّࣺييْ بَ اِتَمْ نࣷونُ لَݧُ تِيتࣹ؞ عِيسَىٰ نࣹ سِمࣷݝْ:‏ «بُلْ رَگَلْ دَرَ؛ گَنَّاوْ-سِ-تࣹيْ، اَيْ نِتْ نگَيْ نَݒّ؞» ‏11 ݖِ ڪَوْ لࣷولُ ݧُ تࣹيرَلْ گَالْ يَ ݖَ تࣹفࣹسْ گَ، دَلْدِ بࣵيِّ لࣺݒّ، تࣷݒّ ݖَ مࣷومْ؞
عِيسَىٰ فَجْ نَ ڪُ گَانَ
‏12 عِيسَىٰ مࣷو نࣹڪّࣷونْ ݖِبْ دࣴڪّ، ندࣹڪࣹ جࣹنّ وَايَ نگَ فَ، جُ جࣵنگࣷرࣷيْ نگَانَ دَجَلْ يَرَمْ وَ؞ بَ مُ گِسࣹي عِيسَىٰ، دَا دَانُ، دࣴݒّ جࣴعࣴمْ ݖِ سُوفْ، سَرْخُ ڪࣷ، نࣹ:‏ «سَنگّ بِ، سُ لَ سࣷوبࣹي، مَنْ نگَ مَا سࣹتَلْ؞» ‏13 عِيسَىٰ تࣵلَّلْ لࣷخࣷومْ، لاَلْ ڪࣷ نࣹ:‏ «سࣷوبْ نَ مَ، سࣹتَلْ؞» ݖَ سَا سَ نگَانَ گَ تࣹقْلِڪࣷوڪْ مࣷومْ؞ ‏14 عِيسَىٰ دࣺنڪّ ڪࣷ نࣹ ڪࣷ:‏ «بُ ڪࣷ وَخْ ڪࣹنّ، دࣹمَلْ رࣹڪّ ݖَ سَرْخَلْڪَتْ بَ، نگِرْ مُ سࣹيتْ لَ، تࣹ نگَ گࣺنّࣹ سَ جࣷوخࣹبْ سࣹتْلُ، نِ ڪࣷ يࣷونُ مُوسَىٰ سَنْتَانࣹي، مُيْ فِرْندࣹ ݖِ ݧࣷومْ¤؞» ‏15 تࣹيوُلْ جࣴوُبْ عِيسَىٰ گࣴنَتࣹي لَوْ؞ مبࣷولࣷو يُ بَرࣹ دَجࣹ، دِ ڪࣷ دࣺگْلُ تࣹيْ فَجُ، بَ تࣵگّࣷوڪْ سࣹينِ جࣵنگࣷرࣷ؞ ‏16 مࣷومْ نَگْ مُيْ رࣷݖّࣹيڪُيْ دࣹمْ بَ ݖِ بࣺرَبْ يُ وࣺيتْ، دِ فَ ݧَانࣹجِ؞
عِيسَىٰ فَجْ نَ اَبْ لَفَݧْ
‏17 عِيسَىٰ مࣷو دࣷونْ جࣵنگّلࣹ بࣺسْ، فُ اَيْ فَرِسࣹنْ اَڪْ اَيْ جࣵنگّلࣹڪَتِ يࣷونُ مُوسَىٰ تࣷوگࣹ، جٗگࣹ ݖَ مبࣷولࣹمْ دࣴڪِّ گَلِلࣹ اَڪْ يُدࣹ اَڪْ يࣹرُسَلࣹمْ، تࣹ مَنࣷورࣹيْ بࣷرࣷومْ بِ تࣹيوْ فَاڪْ مࣷومْ، نگِرْ فَجْ جَرَگْ جَ؞
‏18 ݖِ بِيرْ لࣷولُ اَيْ نِتْ نࣹ جَلَݧْ، جࣵݒّࣷو اَبْ لَفَݧْ بُ ݧُ تࣴرَلْ ݖِگْ لَلْتُ؞ ݧُ دِ ڪࣷ وُتَ دُگَلْ، نگِرْ تࣴرَلْ ڪࣷ ݖَ ڪَنَمْ عِيسَىٰ؞ ‏19 وَايࣹ گِسُݧُ فُ ݧُ ڪࣷيْ اَوَلࣹ ندَخْ مبࣷولࣷو مَ؞ ݧُ يࣺيگْ ݖَ ڪَوْ نࣺيگْ بَ، خࣹدِّ لࣹنّ ݖَ خَدّ بَ، يࣷورْ وَا جَاڪْ لَلْتُ گَ، وࣵݖّࣹ ڪࣷ ݖَ دِگّ بَ، فَ ڪَنَمْ عِيسَىٰ؞ ‏20 عِيسَىٰ نَگْ گِسْ سࣹينْ نگࣴمْ، دَلْدِ نࣹ لَفَݧْ بَ:‏ «مبࣷڪّ مِ، جࣺگَّلࣹيسْ نَ لَ سَيْ بࣵڪَّارْ؞»
‏21 فِرِڪَتِ يࣷونُ مُوسَىٰ يِ اَڪْ فَرِسࣹنْ يِ تࣵمْبَلࣹي نࣵتَّبْلࣹ ݖِ سࣹينْ خࣹلْ، نَانْ:‏ «ڪِي دِ سَاگَ يࣵلَّ مࣷويْ ڪَنْ؟ اَنَ ڪُ مَنَ جࣺگَّلࣹيْ بࣵڪَّارْ ڪُ دُلْ يࣵلَّ دࣷݝّ؟»
‏22 عِيسَىٰ نَگْ خَمْ لَ ݧُيْ نࣵتَّبْلࣹ ݖِ سࣹينْ خࣹلْ، دَلْدِ اࣵدُّ، نࣹ لࣹينْ:‏ «لُ نگࣹينْ دِ نࣵتَّبْلࣹ ݖِ سࣹينْ خࣹلْ؟ ‏23 اَنَ لُ گࣴنَ يࣷمبّ ݖِ يِي يَارْ:‏ نࣹ ‹جࣺگَّلࣹيسْ نَ لَ سَيْ بࣵڪَّارْ،› اَمْ نࣹ:‏ ‹جٗگَلْ تࣹ دࣷخْ›؟ ‏24 ڪࣷنْ نَگْ دِنگࣹينْ خَمْ نࣹ دࣷومُ نِتْ ڪِ اَمْ نَ ݖِ ڪَوْ سُوفْ، سَݧْ-سَݧُ جࣺگَّلࣹ اَيْ بࣵڪَّارْ؞» مُ نࣹ لَفَݧْ بَ:‏ «دَمَ نࣹ، جٗگَلْ، جࣴلْ سَگْ لَلْتُ تࣹ دࣹمْ ݧِبِّ؞» ‏25 نِتْ ڪَ دَلْدِ نࣹ ݧࣷڪّࣹتْ ݖَ ڪَنَمْ ݧࣺݒّ، جࣴلْ لَلْتُ گَ مُ تࣴدّࣷونْ، جࣴمْ ڪࣴرَمْ، دِ سࣵبَّالْ يࣵلَّ؞ ‏26 بَ لࣷولُ اَمࣹي مبࣷولࣷو مࣺݒَّ نࣹييْ، دِ سࣵبَّالْ يࣵلَّ، تِيتَانگࣹ جࣵݒّ لࣹينْ، ݧُ نَانْ:‏ «گِسْ نَنُ تࣹيْ مبِرْ يُ تِيݧْ اَمْ خࣹلْ؞»
عِيسَىٰ وࣷو نَ لࣹوِ
‏27 گَنَّاوْ لࣷولُ عِيسَىٰ گࣺنّ، گِسْ اَبْ جُوتِڪَتْ بُ ݧُيْ وَخْ لࣹوِ، مُ تࣷوگْ ݖَ جُوتِ بَ؞ مُ نࣹ ڪࣷ:‏ «ڪَايْ، تࣷݒّ مَ؞» ‏28 لࣹوِ جٗگْ، بࣵيِّ لࣺݒّ، تࣷݒّ ݖَ مࣷومْ؞ ‏29 ݖِ ڪَوْ لࣷولُ لࣹوِ وَاجَلَلْ ڪࣷ بࣹرْندࣹ جُ رࣺيْ ݖَ ڪࣴرَمْ؞ مبࣷولࣷو مُ بَرࣹ تࣹيوْ، اَيْ جُوتِڪَتْ اَڪْ يࣹنࣹينْ گَنْ، ݧُ بࣷڪّ تࣷوگْ، دِ لࣹڪّ؞ ‏30 فَرِسࣹنْ يِ اَڪْ سࣹينِ فِرِڪَتْ نَگْ دِ خُلْتُ ݖِ ڪَوْ تَالِبࣹيْ عِيسَىٰ يِ، نࣹ لࣹينْ:‏ «لُ تَخْ نگࣹينْ دِ لࣹڪَّنْدࣷو اَڪَ نَانَنْدࣷو اَڪْ جُوتِڪَتْ يࣹيڪْ بࣵڪَّارْڪَتْ يِ؟» ‏31 عِيسَىٰ نࣹ لࣹينْ:‏ «دُ ݧِ وࣺرْ ݧࣷو سࣷخْلَ فَجْڪَتْ، ݧِ وࣺرَدِ ݧࣷو ڪࣷ سࣷخْلَ؞ ‏32 دُ ݧِ جُبْ نَگْ لاَ وࣷوسِ، بࣵڪَّارْڪَتْ يِ لاَيْ وࣷو، نگِرْ جࣴمࣹ لࣹينْ ݖِگْ تُوبࣹيلْ؞»
ڪࣷورْ دُ وَرْتࣺيفْ ݖِ گࣴمْڪَتُبْ عِيسَىٰ
‏33 دَݧࣷو وَخْ عِيسَىٰ، نࣹ ڪࣷ:‏ «تَالِبࣹيْ يَحْيَىٰ دࣹ، دِنَݧُ فَرَلْ دِ وࣷورْ، دِ جُلِّ؞ فَرِسࣹنْ يِ اِتْ نࣷونُ، وَايࣹ سَ ݧࣷݧْ، لࣹڪّ اَڪَ نَانْ لَݧُ وࣺيࣹ؞» ‏34 عِيسَىٰ نࣹ لࣹينْ:‏ «ندَخْ مَنْ نگࣹينَ وࣷورْلࣷو اَيْ گَنْ ݧُ وࣵلِّ اَگْ ݖࣺيتْ، لِ فࣹيڪْ بࣷرࣷومْ سࣺيتْ بِيْ تࣹيوْ ݖِ سࣹينْ بِيرْ؟ ‏35 وَايࣹ اَيْ بࣺسَ نگِ ݧࣴوْ، دࣹيسْ نَ جࣴلࣹ بࣷرࣷومْ سࣺيتْ بِ ݖِ سࣹينْ بِيرْ؞ ڪࣹرَ بࣺسْ يࣷويُ نَگْ، دِنَݧُ وࣷورْ؞»
‏36 عِيسَىٰ لࣺيبَتِ لࣹينْ، نࣹ لࣹينْ:‏ «مبُبّ مُ بࣹيسْ، دࣹيسُ ݖِ خࣷتِّ اَبْ سࣹڪِّتْ، نگِرْ دَاخࣹ ڪࣷ مبُبّ مُ رࣵݒّ؞ لُ ڪࣷ مࣷيْ مبُبّ مُ بࣹيسْ مِ خࣷتِّڪُ بَ نࣷݒِّ، تࣹ دُ تَخْ سࣹڪِّتْ وُ بࣹيسْ وِ اࣵندّ اَڪْ مبُبّ مِ رࣵݒّ؞ ‏37 تࣹ اِتْ مبُوسْ يُ رࣵݒّ، دࣹيسُ ݖِ دُيْ بِيݧْ بُ بࣹيسْ؞ لُ ڪࣷ مࣷيْ بِيݧْ بُ بࣹيسْ بِ فࣵݖّ مبُوسْ يِ؛ بِيݧْ بِ تُورُ، مبُوسْ يِ يࣵقُ؞ ‏38 بِيݧْ بُ بࣹيسْ دَالْ، ݖِ مبُوسْ مُ بࣹيسْ لࣹيسْ ڪࣷيْ دࣹنݖّ؞ ‏39 تࣹ اِتَمْ دࣹيسُلْ نَانْ بِيݧْ بُ يࣵگّ، دِ بࣴگَّتِ بُ بࣹيسْ، خَنَا نگَ نࣹ:‏ ‹بُ يࣵگّ بࣹي بَاخْ¤؞›»
