ݒٗولَ نگِ لَيࣷولْ سَسُبْ ندَوَمْ
3
‏1 دَنُيْ تࣵمْبَلࣹيتࣹي سࣹيدࣹيلْ سُنُ بࣷݒّ نࣹ نࣷو يࣹلّࣷو نگࣴرࣴمَ؟ اَمْ دَنࣷو سࣷخْلَ نِ ݧࣹنّ نِتْ ݧِ، اَيْ بَتَاخَلْ يُ جࣴمْ ݖِ يࣹينْ، مبَا مُ بَوࣷو ݖِ يࣹينْ، نگِرْ سࣹيدࣹيلْ نُ نگࣴرࣴمْ؟ ‏2 سُنُ بَتَاخَلْ ڪَيْ، يࣹينْ لَ، تࣹ دࣹيسْ ڪࣷو بِندّ ݖِ سُنُ خࣷلْ، بَ ݧࣺݒّ خَمْ ڪࣷ، جࣵنگّ ڪࣷ؞ ‏3 بِرْ نَ نࣹ بَتَاخَلُبْ اَلْمَسِ بُ جࣷتّلِڪࣷو ݖِ سُنُبْ لِگّࣺييْ، مࣷومْ نگࣹينْ، تࣹ دُ دَا لࣹيسْ ڪࣷ بِنْدࣹ، وَايࣹ نࣷووُگْ يࣵلَّ جِيْ دُندّ لࣹيسْ ڪࣷ بِنْدࣹ؞ دُ ݖِ اࣵلُّوَيْ دࣷجْ لࣹيسْ ڪࣷ رࣴدّ، وَايࣹ ݖِ اࣵلُّوَيْ خࣷلِ سُوخْ، لࣹيسْ ڪࣷ رࣴدّ؞
‏4 گࣷوگُ ڪٗولُتࣹ لَنُ اَمْ ݖِ يࣵلَّ ندَخْ اَلْمَسِ؞ ‏5 دُ ݖَاگࣹينُ دࣷيْلࣷو سُنُ مَنࣷورࣹيْ بࣷݒّ، بَ يَاڪَارْ نࣹ لࣹنّ مَنْ نَا سࣷتّࣹي ݖِ سُنُ وࣵلُّ بࣷݒّ، وَايࣹ سُنُ مَنࣷورࣹ ݖِ يࣵلَّ لَ جٗگࣹ؞ ‏6 مࣷو نُ مَيْ مَنࣷورࣹيْ دࣷونْ اَيْ لِگّࣺييْڪَتِ ڪٗلّࣴرࣹ گُ بࣹيسْ گُ اَجُوُلْ ݖِ مبِنْدُمْ ندِگَلِ يࣷونْ، خَنَا ڪٗلّࣴرࣹ گُ اَجُ ݖِ نࣷو گِ؞ ندَخْتࣹ مبِنْدُمْ ندِگَلِ يࣷونْ دَيْ رࣹيْ، تࣹ نࣷو گِ دِ دُندّلࣷو؞
‏7 مبِنْدُمْ ندِگَلِ يࣷونْ وِ، ݖِ اَوْ دࣷجْ لࣹيسْ ڪࣷ رࣴدّࣷونْ، تࣹ مُوسَىٰ مࣷو سَسࣷو وࣷونْ لِگّࣺييُ يࣷونْ وࣷووُيْ جࣴمࣹ ݖِ ندࣹي؞ تࣹيوُلْ مُيْ لِگّࣺييْ بُ لࣵمْبࣷو اَگْ لࣹيرْ گُ دࣷونْ لࣹرْخَتْ ݖَ ڪَنَمْ گَ، بَ بࣵنِّ إِسْرَاءِيلْ تࣴلࣹ ڪࣷو خࣷولْ جࣵڪّ، دࣷونْتࣹ لࣹيرْ گُيْ وࣺيْ لَ وࣷونْ؞ ‏8 وَايࣹ لِگّࣺييُ نࣷو گِ، لࣹيرْ گُ گࣴنَتࣹي رࣺيْ لَ لࣵمْبࣷو؞ ‏9 ندࣹگَمْ لِگّࣺييْ بࣷوبُيْ تࣹگࣹ اَبْ دَانْ مࣷو لࣵمْبࣷو اَگْ لࣹيرْ، ڪࣷنْ لِگّࣺييُ نࣷو، گِيْ مَيࣹ نجࣹڪّ مࣷو ڪࣷ گࣴنَتࣹي لࣵمْبࣷو اَگْ لࣹيرْ؞ ‏10 سࣷو ڪࣷ خࣷولࣹي نࣷونُ، لࣹيرْ گَ وࣷونْ، دُ وࣷونْ سَخْ اَگْ لࣹيرْ، ݖِ لࣹيرْ گِي ڪࣷ رَوْ تࣹيْ، ‏11 ندَخْ ندࣹگَمْ لِيْ وࣺيْ مࣷو لࣵمْبࣷو وࣷونْ اَگْ لࣹيرْ، اَسْتَمَاڪْ لِ فِ سَخْ؞
‏12 گَنَّاوْ اَمْ نَنُ جࣷوجُ يَاڪَارْ نَگْ، فِتْ وُ رࣺيْ لَنُيْ دࣹمࣹ، ‏13 تࣹ بَݧَ دࣹفْ نِ مُوسَىٰ مَ دَانْ مُورْ ڪَنَمْ گَ، ندَخْ بࣵنِّ إِسْرَاءِيلْ بَݧْ نࣹ جࣵڪّ ݖِ لࣹيرْ گُيْ وࣺيْ، دِ گِيمْ، دِ گِيمْ بَ جࣹيخْ؞ ‏14 وَايࣹ سࣹينُمْ خࣹلَ دࣴرْڪِيسْ؞ بَ تࣹيْ جِي، بُ ݧُيْ جࣵنگࣹ بِيرْ مبࣷولࣷو، تࣺيرࣹبْ ڪٗلّࣴرࣹ گُ جࣴڪّ گِ، مُورَايْ گࣷوگࣷو فِ دࣹسْ؛ مُرِڪُوُلْ، ندَخْ ݖِ اَلْمَسِ لَيْ دࣹݧࣹ؞ ‏15 وَاوْ، بَ تࣹيْ جِي ڪَيْ، سَا يُ ݧُيْ بِرَلْ يࣷونُ مُوسَىٰ، اَگْ مُورَايَ نگِ تࣹگࣹ ݖِ سࣹينْ خࣷلْ؞ ‏16 وَايࣹ سَا سُ نِتْ وَݧِّڪࣷو ݖِ بࣷرࣷومْ بِ، مُورَايْ گِ دَيْ مُرِڪُ؞ ‏17 بࣷرࣷومْ بِ مࣷويْ نࣷو گِ، تࣹ فُ نࣷو گِ نࣹڪّ، فَ لَ نِتْ دِ مࣷومࣹ بࣷݒَّمْ؞ ‏18 نُنْ ݧࣺݒّ نَگْ، ڪَنَمْ گُ مُرِڪُ لَنُيْ خࣷولࣹ نِڪِ ݖِبْ سࣹيتُ، لࣹيرُ بࣷرࣷومْ بِ، بَ تَخْ نُيْ سࣷݒِّڪُ، دࣷونْ تَڪّندࣹيرَمْ، تࣹيْ لࣵمْبࣷو لࣹيرْ گُيْ مࣵگّ، دِ گࣴنَ مࣵگّ؞ لࣷولࣷو دِ لِگّࣺييُ بࣷرࣷومْ بِ، مࣷومْ مِيْ نࣷو گِ؞
