لُ جࣴمْ ݖِ ندِمْبَلْ لُ ݧࣹيلْ گࣴمْڪَتِ يࣹرُسَلࣹمْ
16
‏1 ݖِ وࣵلُّ ندِمْبَلْ لِ جࣴمْ ݖِ گࣴمْڪَتْ ݧُ سࣹلّ ݧِ، نࣷونُ مَ ڪࣷ دِگَلࣹ مبࣷولࣷويْ گࣴمْڪَتِ دِيوَانُ گَلَسِ، يࣹينْ اِتَمْ دࣹفࣹلࣹينْ ڪࣷ نࣷونُ؞ ‏2 بࣺسْ بُ جࣴڪّ ݖِ اَيُ بࣺسْ يِ، نَ ڪُ نࣹڪّ گࣺنّࣹ اَسْ لࣴفْ ݖَ لَ مُ اَمْ، نَ مُ ڪࣷ اࣵتَّنࣹ، تࣹ مُ دࣹنݖَلْ ڪࣷ بࣷݒَّمْ، نگِرْ بُ مَ دِڪّࣹي، دࣷوتُ سࣷخْلاَ وࣴرْ، دِ دَجَلࣹ؞ ‏3 بُ مَ دِڪّࣹي، ݧِ نگࣹينْ تࣵنّ، ݧࣷومْ لاَيْ يࣹبَلࣹ اَبْ بَتَاخَلْ، نگِرْ ݧُ يٗبُّ يࣹرُسَلࣹمْ سࣹينْ ندِمْبَلْ؞ ‏4 سُ آجࣷو مَ دࣹمْ مَنْ اِتَمْ، ݧُ اࣵندّ اَڪْ مَنْ؞
ݒٗولْ مࣺبࣺتْ نَ، دࣺنڪَانࣹ نَ
‏5 دِنَا دِڪّ بَ ݖِ يࣹينْ، گَنَّاوْ بُ مَ جَارࣹي دِيوَانُ مَسࣹدُوَنْ، ندَخْ مَسࣹدُوَنْ لاَيْ جَارࣹ؞ ‏6 مَنْ نَا اَمْ مَ يࣵگّ ݖِ يࣹينْ، جࣷمْبُلْ سَخْ مَ لࣷلِّڪࣷو فِ يࣹينْ؞ سُ ڪࣷ دࣹفࣹي نگࣹينْ يِوِ مَ، مَ دࣹمْ، اَڪْ فُ مَ مَنَ جࣴمْ؞ ‏7 ندَخْتࣹ اِتَمْ، بُگُّمَا جَارࣹ فِ يࣹينْ رࣹڪّ، جࣵلّ، وَايࣹ يَاڪَارْ نَا تࣷوگْ ݖِ يࣹينْ اَبْ دِيرْ، سُ ڪࣷ بࣷرࣷومْ بِ مَيࣹي؞ ‏8 وَايࣹ ݖِ اࣹفࣹسْ گِي لاَيْ تࣷوگْ بَ مࣵگَّلُ ݒࣵنْتَڪࣷتْ، ‏9 ندَخْ بُنتّ اُبِّڪُلْ نَ مَ فِ ݝࣵݒّ، نگِرْ لِگّࣺييْ بُ اَمْ نجَرِݧْ، تࣹ نࣷونْ يِ بَرࣹ نَݧُ؞
‏10 بُ تِمࣷتࣹ دِڪّࣹي نَگْ، فࣹخࣹلࣹينْ بَ بُمُ اَمْ لُ مُيْ تِيتْ ݖِ سࣹينْ بِيرْ، ندَخْ لِگّࣺييُ بࣷرࣷومْ بِ لَيْ لِگّࣺييْ نِ مَنْ؞ ‏11 ڪࣷنْ بُ ڪࣷ ڪࣹنّ خࣹيبْ؞ گَنَّاوْ لࣷولُ نَنگࣹينْ ڪࣷ يِوِ اَڪْ جࣵمّ، بَ مُ دࣺلْسِ ݖِ مَنْ، ندَخْ مࣷومْ لاَيْ خَارْ، مࣷوڪْ بࣷڪّ يِ اِتَمْ؞
‏12 مبࣷڪّ مِ اَݒࣷلࣷسْ نَگْ مࣷومْ، سࣷݧّ نَا ڪࣷ لُ بَرࣹ، نگِرْ مُ دِڪّ بَ ݖِ يࣹينْ، اࣵندّ اَڪْ بࣷڪّ يِ، وَايࣹ فِي نُ تࣷلُّ يࣹبُوُ ݖِ دَرَ؞ تࣹيوُلْ بُ جࣹڪّࣷو، دِنَ دِڪّ؞
يࣹنࣹينْ ندِگَلَ نگِ اَڪِ تࣵگّتࣷو
‏13 تࣹيوْلُلࣹينْ تࣹ سَخࣷو نگࣴمْ، گٗورْ-گٗورْلُلࣹينْ تࣹ دࣴگࣴرْلُ؞ ‏14 لࣺݒّ لُ نگࣹينْ دِ دࣹفْ، دࣹفࣹلࣹينْ ڪࣷ ݖࣷفࣹيلْ؞
‏15 بࣷڪّ يِ، خَمْ نگࣹينْ نࣹ اࣹسْتࣹفَنَسْ اَڪْ نجَبࣷوتَمْ ݧࣷويْ جࣷوخࣹبْ ندࣷورْتࣹلْ مࣹݧّࣹيفْ ݖِ دِيوَانُ اَڪَيِ، تࣹ جࣵݒّلࣹ ݧُ سࣹلّ ݧِ لَݧُ سࣺدّࣷو؞ ڪࣷنْ دَمَ لࣹينْ دِ ݧَانْ، ‏16 نگࣹينْ نَنگُلْ نِتْ ݧُ مࣹلْ نࣷونُ، ݧࣷومْ اَڪْ مبࣷولࣹمْ ݧُ بࣷڪّ اَڪْ ݧࣷومْ لِگّࣺييْ بِ اَڪْ ݖࣷونࣷ بِ؞ ‏17 مَا نگِ بࣺگࣹ تࣹيوَايُ اࣹسْتࣹفَنَسْ اَڪْ فࣷرْتُنَتُسْ اَڪْ اَڪَيْڪُسْ، ندَخْ بِ نگࣹينْ وُوتࣹي تࣹيْ، ݧࣷو لࣹينْ تَخَوَلْ؞ ‏18 دَلَلْ نَݧُ سَمَ خࣹلْ، اَڪْ سࣹينْ مࣷسْ، يࣹينْ اِتَمْ؞ ڪࣷنْ نِتْ ݧُ مࣹلْ نࣷونُ، فࣷنڪّلࣹينْ لࣹينْ؞
‏19 مبࣷولࣷويْ گࣴمْڪَتِ دِيوَانُ اَسِ نَگْ ݧُ نگِ لࣹينْ دِ نُيُ؞ اَڪِلَسْ اَڪْ ݒِرْسِلْ، ݧُ نگِ لࣹينْ دِ نُيُ بُ بَاخْ ݖِ تُرُ سَنگّ بِ، ݧࣷومْ اَڪْ مبࣷولࣷومْ گࣴمْڪَتْ ݧِيْ دَجࣹ سࣹينْ ڪࣴرْ؞ ‏20 بࣷڪّ يࣺݒَّ نگِ لࣹينْ دِ نُيُ؞
سَافࣷونْتࣹيلࣹينْ فٗونَنْتࣹ يُ سࣹلّ؞
‏21 مَنْ ݒٗولْ مَا بِندّ نُيࣷو بِي ݖِ سَمَ لࣷخࣷلْ بࣷݒّ؞
‏22 ڪُ بُگُّلْ سَنگّ بِ، نَ اَلْڪُ؞ مَرَنَتَ (مُ فِرِ سُنُ سَنگّ، دِڪَّلْ!‏)
‏23 يَلْ نَ يِوُ سَنگّ عِيسَىٰ اࣵندّ اَڪْ يࣹينْ؞
‏24 سَمَ ݖࣷفࣹيلْ ݧࣹيلْ نَ لࣹينْ، يࣹينْ ݧࣺݒّ، ݖِ عِيسَىٰ اَلْمَسِ مِ نُ بࣷڪّ؞
