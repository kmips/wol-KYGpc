مࣵنْدِݝْ مَ
فَ بࣵنِّ إِسْرَاءِيلْ جَارࣹ وࣷونْ
دِ ݧࣹينْتࣹيلُ خَاجُ تَوْرࣹيتُ مُوسَىٰ
إِسْرَاءِيلْ جࣵنگࣹ نَ ݖَ مَامْ يَ جٗگࣹ وࣷونْ مِسْرَ
(سَارْ 1—21)
تَڪّ نَݧُ لِمُ بࣵنِّ إِسْرَاءِيلْ
1
‏1 اَجِ سَخْ جِ مࣷو وَخْ اَڪْ مُوسَىٰ ݖَ مࣵنْدِݝُ سِنَيِ ݖَ بِيرْ خَيْمَبْ ندَجࣹ مَ، ڪࣹرࣷوگْ بࣹنّ فَنُ ݧَارࣹيلُ وࣹيرْ وُ ݧَارࣹيلُ اَتْ، مَ تࣷݒّ ݖَ گࣵدَّايْ گَ بَوࣷو مِسْرَ؞ مُ نࣹ ڪࣷ:‏
‏2 «لِمْلࣹينْ مبࣷولࣷومْ بࣵنِّ إِسْرَاءِيلْ گࣺݒّ، نَ ݧُ بࣷڪّࣹ سࣹينِ لࣵنگّ اَڪْ سࣹينْ ڪࣴرِ مَامْ؞ وَݧّلࣹينْ گٗورْ ݧࣺݒّ، بࣷݒّ اَڪْ بࣷݒّ، ڪُ نࣹڪّ اَڪْ تُرَمْ؞ ‏3 لِ ڪࣷ دَلࣹ ݧَارْ فُڪِّ اَتْ، جࣴمْ ڪَوْ، مبࣷولࣹمْ ݧِ خَرࣹ وَرْ ݖِ إِسْرَاءِيلْ دَالْ، نَنگࣹينْ لࣹينْ لِمْ، گࣵنگࣷورْ اَڪْ گࣵنگࣷورْ، يَوْ اَڪْ هَـٰرُونَ؞ ‏4 گِيرْ گُ ݖِ نࣹڪّ، اࣵندّلࣹينْ ݖࣹيڪْ ڪِلِفَ يِ جِيتࣹ سࣹينْ ڪࣴرِ مَامْ، ݧُ جࣵݒّلࣹ لࣹينْ ݖِ؞
‏5 ݧِيْ وُيࣷو تُرْ يِي نَگْ، ݧࣷو لࣹينْ ݖِيْ تَخَوُ:‏
ݖِ وࣵلُّ رُبࣹنࣹينْ ݧِ، اࣹلِسُرْ دࣷومُ سࣹدࣹيُرْ؞
‏6 وࣵلُّ ݖِمْيࣷنࣹينْ ݧِ، سࣹلُمْيࣹلْ دࣷومُ سُرِسَدَايْ؞
‏7 وࣵلُّ يُدࣹينْ ݧِ، نَسࣷنْ دࣷومُ اَمِنَدَبْ؞
‏8 وࣵلُّ اِسَڪَرࣹينْ ݧِ، نࣹتَنࣹلْ دࣷومُ سُوَرْ؞
‏9 وࣵلُّ سَبُلࣷنࣹينْ ݧِ، اࣹلْيَبْ دࣷومُ اࣹلࣷنْ؞
‏10 ݧِ ݖِيْ اَسْڪَنُ يُوسُفَ ݧِي لَ:‏
ݖِ وࣵلُّ اࣹفْرَيْمࣹينْ ݧِ، اࣹلِسَمَ دࣷومُ اَمِيُتْ؞
وࣵلُّ مَنَسࣹينْ ݧِ، گَمَلْيࣹلْ دࣷومُ ݒࣹدَسُرْ؞
‏11 وࣵلُّ بࣹݧَمِنࣹينْ ݧِ، اَبِدَنْ دࣷومُ گِدࣷنِ؞
‏12 وࣵلُّ دَنࣹينْ ݧِ، اَيࣹسࣹرْ دࣷومُ اَمِسَدَايْ؞
‏13 وࣵلُّ اَسࣹرࣹينْ ݧِ، ݒَگْيࣹلْ دࣷومُ اࣷڪّرَنْ؞
‏14 وࣵلُّ گࣵدّࣹينْ ݧِ، اࣹلْيَسَفْ دࣷومُ دࣹوࣹلْ؞
‏15 وࣵلُّ نࣹفْتَلࣹينْ ݧِ، اَيْرَ دࣷومُ اࣹنَنْ؞»
‏16 ݧࣷوݧُ لَݧُ وࣷو ݖِ مبࣷولࣷو مِ، مُيْ ݧِ جِيتࣹ سࣹينْ گِيرِ مَامْ؞ ݧࣷويْ نجِيتِ گࣵنگࣷورِ إِسْرَاءِيلْ؞
‏17 بَ لࣷولُ اَمࣹي مُوسَىٰ اَڪْ هَـٰرُونَ تࣷفْلࣷو ݧࣷوݧُ ݧُ تُدّ سࣹينِ تُرْ، ‏18 ݧُ دَجَلࣹ مبࣷولࣷو مَ مࣺݒّ، ڪࣹرࣷوگْ ݖَ بࣹنّ فَنُ ݧَارࣹيلُ وࣹيرْ وَ؞ لَ ڪࣷ دَلࣹ ݖَ ݧَ اَمْ ݧَارْ فُڪِّ اَتْ، جࣴمْ ڪَوْ، ݧࣷومْ ݧࣺݒّ دَلْدِ بِنْدُ ݖَ لِمࣹيفُ تُرْ يَ، ڪُ نࣹڪّ اَڪْ ڪࣴرْ مَامَمْ، گِيرْ اَڪْ گِيرْ، لࣵنگّ اَڪْ لࣵنگّ، بࣷݒّ اَڪْ بࣷݒّ؞ ‏19 نࣷونࣹي ڪࣷ اَجِ سَخْ جِ سَنْتࣹ وࣷونْ مُوسَىٰ، نَ لَ مُوسَىٰ تَڪّࣹ سࣹينْ لِمْ ݖَ مࣵنْدِݝُ سِنَيِ؞
‏20 ݧُ لِمْ بࣷݒّ اَڪْ بࣷݒّ، تُرِ سࣴتِ رُبࣹنْ تَاوُبْ إِسْرَاءِيلْ سࣹينْ مَامْ¤، ڪُ نࣹڪّ اَڪْ لࣵنگُ ݖࣷسَانَمْ اَڪْ ڪࣴرْ مَامَمْ، مبࣷولࣹمْ گٗورْ ݧِ خَرࣹ وَرْ، تࣹ ݧُ اَمْ ݧَارْ فُڪِّ اَتْ، جࣴمْ ڪَوْ؞ ‏21 لِمُبْ رُبࣹنࣹينْ ݧِ دِ ݧࣹينتّ فُڪِّ جُنِّ اَڪْ جُرٗومْ بࣹنّ اَڪْ جُرٗومِ تࣺيمࣺيرْ (46500)؞
‏22 ݧُ لِمْ بࣷݒّ اَڪْ بࣷݒّ، تُرِ سࣴتِ سِمࣹيࣷنْ، ڪُ نࣹڪّ اَڪْ لࣵنگُ ݖࣷسَانَمْ اَڪْ ڪࣴرْ مَامَمْ، مبࣷولࣹمْ گٗورْ ݧِ خَرࣹ وَرْ، تࣹ ݧُ اَمْ ݧَارْ فُڪِّ اَتْ، جࣴمْ ڪَوْ؞ ‏23 لِمُبْ ݖِمْيࣷنࣹينْ ݧِ دِ جُرٗومْ فُڪِّ جُنّࣹيڪْ جُرٗومْ ݧࣹينتّ اَڪْ ݧࣹتِّ تࣺيمࣺيرْ (59300)؞
‏24 ݧُ لِمْ تُرِ سࣴتِ گࣵدّ، ڪُ نࣹڪّ اَڪْ لࣵنگُ ݖࣷسَانَمْ اَڪْ ڪࣴرْ مَامَمْ، مبࣷولࣹمْ گٗورْ ݧِ خَرࣹ وَرْ، تࣹ ݧُ اَمْ ݧَارْ فُڪِّ اَتْ، جࣴمْ ڪَوْ؞ ‏25 لِمُبْ گࣵدّࣹينْ ݧِ دِ ݧࣹينتّ فُڪِّ جُنّࣹيڪْ جُرٗومْ اَڪْ جُرٗومْ بࣹنِّ تࣺيمࣺيرْ اَڪْ جُرٗومْ فُڪّ (45650)؞
‏26 ݧُ لِمْ تُرِ سࣴتِ يُدَ، ڪُ نࣹڪّ اَڪْ لࣵنگُ ݖࣷسَانَمْ اَڪْ ڪࣴرْ مَامَمْ، مبࣷولࣹمْ گٗورْ ݧِ خَرࣹ وَرْ، تࣹ ݧُ اَمْ ݧَارْ فُڪِّ اَتْ، جࣴمْ ڪَوْ؞ ‏27 لِمُبْ يُدࣹينْ ݧِ دِ جُرٗومْ ݧَارْ فُڪِّ جُنّࣹيڪْ ݧࣹينتّ، اَڪْ جُرٗومْ بࣹنِّ تࣺيمࣺيرْ (74600)؞
‏28 ݧُ لِمْ تُرِ سࣴتِ اِسَاڪَرْ، ڪُ نࣹڪّ اَڪْ لࣵنگُ ݖࣷسَانَمْ اَڪْ ڪࣴرْ مَامَمْ، مبࣷولࣹمْ گٗورْ ݧِ خَرࣹ وَرْ، تࣹ ݧُ اَمْ ݧَارْ فُڪِّ اَتْ، جࣴمْ ڪَوْ؞ ‏29 لِمُبْ اِسَڪَرࣹينْ ݧِ دِ جُرٗومْ فُڪِّ جُنّࣹيڪْ ݧࣹينتّ اَڪْ ݧࣹينْتِ تࣺيمࣺيرْ (54400)؞
‏30 ݧُ لِمْ تُرِ سࣴتِ سَبُلࣷنْ، ڪُ نࣹڪّ اَڪْ لࣵنگُ ݖࣷسَانَمْ اَڪْ ڪࣴرْ مَامَمْ، مبࣷولࣹمْ گٗورْ ݧِ خَرࣹ وَرْ، تࣹ ݧُ اَمْ ݧَارْ فُڪِّ اَتْ، جࣴمْ ڪَوْ؞ ‏31 لِمُبْ سَبُلࣷنࣹينْ ݧِ دِ جُرٗومْ فُڪِّ جُنّࣹيڪْ جُرٗومْ ݧَارْ اَڪْ ݧࣹينْتِ تࣺيمࣺيرْ (57400)؞
‏32 ݧُ لِمْ تُرِ سࣴتِ يُوسُفَ يِ اَسْڪَنࣷو ݖِ دࣷومَمْ اࣹفْرَيِمْ، ڪُ نࣹڪّ اَڪْ لࣵنگُ ݖࣷسَانَمْ اَڪْ ڪࣴرْ مَامَمْ، مبࣷولࣹمْ گٗورْ ݧِ خَرࣹ وَرْ، تࣹ ݧُ اَمْ ݧَارْ فُڪِّ اَتْ، جࣴمْ ڪَوْ؞ ‏33 لِمُبْ اࣹفْرَيْمࣹينْ ݧِ دِ ݧࣹينتّ فُڪِّ جُنِّ اَڪْ جُرٗومِ تࣺيمࣺيرْ (40500)؞
‏34 ݧُ لِمْ تُرِ سࣴتِ يُوسُفَ يِ اَسْڪَنࣷو ݖِ دࣷومَمْ مَنَسࣹ، ڪُ نࣹڪّ اَڪْ لࣵنگُ ݖࣷسَانَمْ اَڪْ ڪࣴرْ مَامَمْ، مبࣷولࣹمْ گٗورْ ݧِ خَرࣹ وَرْ، تࣹ ݧُ اَمْ ݧَارْ فُڪِّ اَتْ، جࣴمْ ڪَوْ؞ ‏35 لِمُبْ مَنَسࣹينْ ݧِ دِ فَنْوࣹيرِ جُنّࣹيڪْ ݧَارْ اَڪْ ݧَارْ تࣺيمࣺيرْ (32200)؞
‏36 ݧُ لِمْ تُرِ سࣴتِ بࣹݧَمِنْ، ڪُ نࣹڪّ اَڪْ لࣵنگُ ݖࣷسَانَمْ اَڪْ ڪࣴرْ مَامَمْ، مبࣷولࣹمْ گٗورْ ݧِ خَرࣹ وَرْ، تࣹ ݧُ اَمْ ݧَارْ فُڪِّ اَتْ، جࣴمْ ڪَوْ؞ ‏37 لِمُبْ بࣹݧَمِنࣹينْ ݧِ دِ فَنْوࣹيرِ جُنّࣹيڪْ جُرٗومْ اَڪْ ݧࣹينْتِ تࣺيمࣺيرْ (35400)؞ ‏38 ݧُ لِمْ تُرِ سࣴتِ دَنْ، ڪُ نࣹڪّ اَڪْ لࣵنگُ ݖࣷسَانَمْ اَڪْ ڪࣴرْ مَامَمْ، مبࣷولࣹمْ گٗورْ ݧِ خَرࣹ وَرْ، تࣹ ݧُ اَمْ ݧَارْ فُڪِّ اَتْ، جࣴمْ ڪَوْ؞ ‏39 لِمُبْ دَنࣹينْ ݧِ دِ جُرٗومْ بࣹنّ فُڪِّ جُنّࣹيڪْ ݧَارْ اَڪْ جُرٗومْ ݧَارْ تࣺيمࣺيرْ (62700)؞
‏40 ݧُ لِمْ تُرِ سࣴتِ اَسࣹرْ، ڪُ نࣹڪّ اَڪْ لࣵنگَمْ اَڪْ ڪࣴرْ مَامَمْ، مبࣷولࣹمْ گٗورْ ݧِ خَرࣹ وَرْ، تࣹ ݧُ اَمْ ݧَارْ فُڪِّ اَتْ، جࣴمْ ڪَوْ؞ ‏41 لِمُبْ اَسࣹرࣹينْ ݧِ دِ ݧࣹينتّ فُڪِّ جُنّࣹيڪْ بࣹنّ اَڪْ جُرٗومِ تࣺيمࣺيرْ (41500)؞
‏42 ݧُ لِمْ تُرِ سࣴتِ نࣹفْتَلِ، ڪُ نࣹڪّ اَڪْ لࣵنگُ ݖࣷسَانَمْ اَڪْ ڪࣴرْ مَامَمْ، مبࣷولࣹمْ ݧِ خَرࣹ وَرْ، تࣹ ݧُ اَمْ ݧَارْ فُڪِّ اَتْ، جࣴمْ ڪَوْ؞ ‏43 لِمُبْ نࣹفْتَلࣹينْ ݧِ دِ جُرٗومْ فُڪِّ جُنّࣹيڪْ ݧࣹتّ اَڪْ ݧࣹينْتِ تࣺيمࣺيرْ (53400)؞
‏44 يࣷويُ لِمࣹيفْ لَ مُوسَىٰ اَڪْ هَـٰرُونَ تࣴنڪّ، ݧَ لࣹينْ ݖَ جࣵݒّلࣹ دِ فُڪِّ ڪِلِفَيْ إِسْرَاءِيلْ اَڪْ ݧَارْ، ڪُ ݖِ نࣹڪّ تَخَوَلْ ڪࣴرْ مَامَمْ؞ ‏45 مبࣷولࣹمْ بࣵنِّ إِسْرَاءِيلْ يِ خَرࣹ وَرْ، تࣹ ݧُ اَمْ ݧَارْ فُڪِّ اَتْ، جࣴمْ ڪَوْ، لِمࣹيسْ نَ لࣹينْ، ڪُ ݖِ نࣹڪّ اَڪْ ڪࣴرْ مَامَمْ؞ ‏46 ݧَ ݧُ لِمْ ݧࣺݒّ دِ جُرٗومْ بࣹنّ تࣺيمࣺيرِ جُنّࣹيڪْ ݧࣹتّ اَڪْ جُرٗومِ تࣺيمࣺيرْ اَڪْ جُرٗومْ فُڪّ (603550)؞
لِمَالࣹيسُلْ لࣹوࣹينْ ݧِ
‏47 گِيرُگْ لࣹوࣹينْ ݧِ نَگْ ݖِ بࣵنِّ إِسْرَاءِيلْ لَݧُ بࣷڪّ، وَايࣹ لِمَالࣹيسُ لࣹينْ اَڪْ ݧࣷومْ؞ ‏48 اَجِ سَخْ جِ مࣷو وَخࣷونْ مُوسَىٰ، نࣹ ڪࣷ:‏ ‏49 «گِيرُگْ لࣹوِ ݧࣷومْ، بُ لࣹينْ لِمْ، بُ لࣹينْ وَݧَّالࣹيڪْ بࣵنِّ إِسْرَاءِيلْ؞ ‏50 وَايࣹ يَوْ ݖِ سَ بࣷݒّ نَنگَ سَسْ لࣹوࣹينْ ݧِ جَامࣷوڪَايْ بِ سࣹيدࣹسْ ڪٗلّࣴرࣹ گِ دࣹنݖࣹ، مࣷوڪْ لَ مُيْ اࣵنْدَلْ لࣺݒّ، اَڪْ مبࣷولࣹمْ لُ ݖَ بࣷڪّ؞ ݧࣷويْ گࣵدُّ جَامࣷوڪَايْ بِ، اَڪْ لَ مُيْ اࣵنْدَلْ لࣺݒّ، دِ ڪࣷ تࣷݒّتࣷو، تࣹ اِتْ فِ وࣴرْ جَامࣷوڪَايْ بِ لَݧُيْ دَلْ؞ ‏51 بُ جَامࣷوڪَايْ بِ دࣹي دࣹمْ، لࣹوࣹينْ ݧࣹي ڪࣷيْ وࣺڪِّ، بُ جَامࣷوڪَايْ بِ دࣹي دَلࣹ بࣺرَبْ، لࣹوࣹينْ ݧࣹي ڪࣷيْ وࣹ؞ ڪُ ݖِ بࣷڪُّلْ بَ لاَلْ ݖِ نَگْ، دࣹيسْ ڪࣷيْ رࣹيْ؞ ‏52 بࣵنِّ إِسْرَاءِيلْ گِ ݖِ دࣹسْ، نَ ڪُ ݖِ نࣹڪّ تࣷݒّ دَلَمْ، اَڪْ رَايَامْ اَڪْ ڪُرࣺلَمْ؞ ‏53 لࣹوࣹينْ ݧِ نَگْ نَݧُ دَلْ فِ وࣴرْ جَامࣷوڪَايْ بِ سࣹيدࣹسْ ڪٗلّࣴرࣹ گِ دࣹنݖࣹ؞ سُ ڪࣷ دࣹفࣹي اَمْ سࣵنجّ دُ دَلْ ݖِ ڪَوْ مبࣷولࣷومْ بࣵنِّ إِسْرَاءِيلْ؞ تࣹ اِتَمْ نَ لࣹوࣹينْ ݧِ دࣺنڪࣷو دࣺنڪَانࣹيْ جَامࣷوڪَايْ بِ سࣹيدࣹسْ ڪٗلّࣴرࣹ گِ دࣹنݖࣹ؞»
‏54 بࣵنِّ إِسْرَاءِيلْ دَلْدِ دࣹفْ نࣷونُ؞ مبࣷولࣹمْ لُ اَجِ سَخْ جِ سَنْتࣷونْ مُوسَىٰ، نࣷونࣹي لَݧُ ڪࣷ دࣹفࣹ؞
