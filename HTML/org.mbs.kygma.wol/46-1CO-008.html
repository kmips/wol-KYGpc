لُ جࣴمْ ݖِ ݧَمْ وُ ݧُ سَرْخَلْ اَيْ تُورْ
8
‏1 سُ دࣹي وࣵلُّ ݧَمْ وُ ݧُ سَرْخَلْ اَيْ تُورْ، دࣴگّ لَ، نُنْ ݧࣺݒّ اَمْ نَنُ خَمْ-خَمْ ݖِ مࣷومُ مبِرْ؞ خَمْ-خَمْ نَگْ دَيْ تَخْ نِتَ رࣺيْ-رࣺيْلُ، وَايࣹ ݖࣷفࣹيلْ دَيْ يࣷڪّࣹ؞ ‏2 ڪُ خَلاَتْ نࣹ يَا خَمْ لࣹنّ، خَمَگُلࣷو نِ نگَ وَرَ خَمࣹ؞ ‏3 وَايࣹ ڪُ سࣷݒّ يࣵلَّ، يࣵلَّ خَمْ لَ؞
‏4 ڪࣷنْ سُ دࣹي وࣵلُّ ݧَمْ وُ ݧُ سَرْخَلْ اَيْ تُورْ، خَمْ نَنُ نࣹ اَيْ تُورْ دُ دَرَ ݖِ اࣵدِّنَ، تࣹ يࣵلَّ ڪࣹنّ لَ؞ ‏5 دࣷونْتࣹ اَمْ نَ يُ ݧُيْ وࣷويࣹ اَيْ يࣵلَّ، تࣹ مُيْ لُ نࣹڪّ ݖِ اَسَمَانْ مبَا ݖِ سُوفْ، تࣹ دࣴگّ لَ، بَرࣹ نَ ݧُ ݧُ نࣹييْ اَيْ يࣵلَّ لَݧُ، بَرࣹ نَ اِتْ ݧُ ݧُ نࣹييْ اَيْ سَنگّ لَݧُ؞ ‏6 تࣹيوُلْ ݖِ سُنُ وࣵلُّ بࣷݒّ، جࣹنّ يࣵلاَّ اَمْ، مُيْ بَايْ بِ لࣺݒّ جٗگࣹ ݖِ مࣷومْ، تࣹ سُنُ بَڪَّنْ ݧࣹيلْ ڪࣷ، بࣹنّ سَنگّ دࣷݝَّ اَمْ، مُيْ عِيسَىٰ اَلْمَسِ، بِ لࣺݒّ اَمࣹ ݖِ مࣷومْ، تࣹ سُنُ بَڪَّنْ، ݖِ مࣷومْ لَ اَمࣹ؞
‏7 وَايࣹ دُ ݧࣺݒَّ خَمْ لࣷولُ؞ اَمْ نَ گࣴمْڪَتْ ݧُ سࣹينْ مِينࣹيلُ تُورْ يِ بࣵيّࣹيگُلْ، بَ لࣺيگِ، تࣹ مࣷو تَخْ بُ ݧُ لࣹڪّࣹي ݧَمْ وُ ݧُ فࣷوگْ نࣹ سَرَخِ تُورْ لَ، سࣹينْ خࣹلْ مُ دَلَدِ مࣷو لࣹينْ دِ فࣷوگْلࣷو نࣹ سࣷبࣹ تࣷݒّࣹ نَ لࣹينْ ݖَ ݧَمْ وَ؞ ‏8 وَايࣹ دُ وࣹنّ ݧَمْ وُ نُيْ جࣹگࣹلࣹ اَڪْ يࣵلَّ؞ نُ لࣹڪّ ڪࣷ، اَڪْ نُ بَݧْ ڪࣷو لࣹڪّ، دُ نُ يࣷڪّ، دُ نُ وࣵݧِّ؞
‏9 وَايࣹ نِ نگࣹينْ دِ دُنْدࣹ سࣹينْ سَݧْ-سَݧُ بࣷݒّ، دَنگࣹينْ ݖِيْ خࣷولْ ݧِ سࣹينْ نگࣴمْ دࣴگࣴرُلْ، نگِرْ نگࣹينْ بَݧْ لࣹينَ يٗبّࣹ بࣵڪَّارْ؞ ‏10 يَوْ مِ رࣺيْ خَمْ-خَمْ، ڪُ لَ گِسْ ݖَ ندَبْ لَ ݖَ جَامࣷوڪَايُ تُورْ يَ، سُ دࣹي اَبْ گࣴمْڪَتْ بُ خࣹلَمْ دَلَدِ ندَخْ نگࣴمْ گُ دࣹسࣹي دࣴگࣴرْ، مبَا دُ ڪࣷ خِيرْ ݖِ رࣷيْ لَ، بَ مُ لࣹڪّ ݖَ ݧَمُ سَرَخْ سَ؟ ‏11 تࣹ سُ دࣹمࣹي نࣷونُ نَگْ، سُ بࣷوبَا سَ خَمْ-خَمْ بُ رࣺيْ نگَيْ سࣵنڪࣹ سَ مبࣷڪُّمْ گࣴمْڪَتْ بُ نࣺيوْ دࣷولࣹ بࣷوبُ اَلْمَسِ دࣹييَلْ!‏ ‏12 سُ نگࣹينْ تࣷوݧࣹي نࣷونُ بࣷڪِّ گࣴمْڪَتْ ݧِ، بَ يࣵقْ سࣹينْ خࣹلْ مُ دَلَدِ، ڪࣷنْ دࣹ اَلْمَسِ نگࣹينْ تࣷوݧْ؞ ‏13 مࣷو تَخْ، سُ دࣹي اَوْ ݧَمْ وُ مَ لࣹڪّ مࣷويْ تَخْ مَ بࣵڪَّارْلࣷو سَمَ مبࣷڪُّمْ گࣴمْڪَتْ، دࣷوتُمَ لࣹڪّ مُڪّ اَوْ يࣵݒّ ݖِ اࣵدِّنَ، نگِرْ بَݧَ بࣵڪَّارْلࣷو سَمَ مبࣷڪُّمْ گࣴمْڪَتْ؞
