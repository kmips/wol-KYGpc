لُ جࣴمْ ݖِ نجࣺگَّلُگْ بࣷرْ
15
‏1 جُرٗومْ ݧَارِ اَتْ يُ مَتْ يُ نࣹڪّ، نَنگࣹينْ اَمَلْ نجࣺگَّلُگْ بࣷرْ؞ ‏2 مبِرُمْ نجࣺگَّلࣹگْ بࣷرْ گِ، نِي لَيْ دࣹمࣹ:‏ مبࣷولࣹمْ ڪُ لࣹبَلࣷونْ دࣴڪَّنْدࣷومْ، بُمُ ڪࣷ ݖَ فࣹيࣹيڪࣷوتِ؞ نَ جࣺگَّلࣹ لَ مُ لࣹبَلࣷونْ دࣴڪَّنْدࣷومْ، مبࣷڪُّمْ بࣵنِّ إِسْرَاءِيلَمْ، گَنَّاوْ بُ ݧُ بِرَلࣹي نجࣺگَّلࣹگْ بࣷرْ، ݧࣹيلْ اَجِ سَخْ جِ؞ ‏3 جَامْبُرْ بِ بࣷڪُّلْ ݖِ بࣵنِّ إِسْرَاءِيلْ، سَݧْ نگࣹينْ ڪࣷو فࣹيࣹيڪُ، وَايࣹ لَ لࣹينْ سࣹينْ مبࣷڪُّمْ بࣵنِّ إِسْرَاءِيلْ اَمࣹيلْ، جࣺگَّلࣹلࣹينْ ڪࣷ؞
‏4 وَايࣹ اَبْ ندٗولْ دُ اَمْ ݖِ سࣹينْ بِيرْ، ندَخْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ مࣷو لࣹينْ دِ بَرْڪࣹيلْ بُ بَاخْ ݖَ رࣺيوْ مَ مُ لࣹينْ دِ جࣷخْ، مُيْ سࣹينْ ݖࣺرْ، نگࣹينْ جَگࣷو، ‏5 ندࣹگَمْ دࣺگَّلْ نگࣹينْ بُ بَاخْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ، دِ سࣵمّࣷو جࣴفࣹ مبࣷولࣹمْ سَنْتَانࣹ بِي مَ لࣹينْ دِ دࣺنڪّ تࣹيْ؞ ‏6 بُ لࣹينْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ بَرْڪࣹيلࣹي نِ مُ ڪࣷ وَخࣹ دࣺيْ، يࣹينَيْ لࣹبَلْ خࣹيتْ يُ بَرࣹ، وَايࣹ يࣹينْ دُنگࣹينْ لࣹبْ؞ تࣹ اِتْ يࣹينَيْ يِلِفْ خࣹيتْ يُ بَرࣹ، وَايࣹ يࣹينْ، دࣹيسُ لࣹينْ يِلِفْ؞
‏7 سُ اَمࣹي ݖِ سࣹينْ بِيرْ اَبْ ندٗولْ، دِ ڪࣹنّ ݖِ سࣹينْ بࣷڪِّ بࣵنِّ إِسْرَاءِيلْ، ݖِ بࣹنّ ݖِ سࣹينْ دࣴڪِّ رࣺيوْ، مِ لࣹينْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ دِ جࣷخْ، بُلࣹينْ دࣴگࣴرْ خࣷلْ، دِ تࣴجَلْ سࣹينْ لࣷخࣷ، سࣹينْ مبࣷڪُّمْ ندٗولْ مࣷومُ؞ ‏8 نࣵدِّلْ-لࣹينْ ڪࣷو نࣵدِّلْ، اَبَلْ ڪࣷو اَبَلْ دَيࣷبْ سࣷخْلاَمْ؞ ‏9 وَتُّلࣹينْ سࣹينْ بࣷݒّ، بَلَ لࣹينَ جِڪّࣷ جُ سࣹوْ يٗبّࣹ نگࣹينْ نَانْ ݖِ سࣹينْ خࣹلْ:‏ «اَتُمْ جُرٗومْ ݧَارࣹيلْ مِ، اَتُمْ نجࣺگَّلُگْ بࣷرْ دࣴگْمَلْ نَ،» بَ تَخْ نگࣹينْ سَجّࣹ خࣹفْ، سࣹينْ مبࣷڪُّمْ ندٗولْ، جࣷخُلࣹينْ ڪࣷ دَرَ؞ دَ لࣹينْ دِ بࣷولࣹيڪْ اَجِ سَخْ جِ، نگࣹينْ گࣵدُّ بࣵڪَّارْ؞ ‏10 نَنگࣹينْ ڪࣷ جࣷخَ جࣷخْ تࣹ بُ نگࣹينْ ڪࣷيْ جࣷخْ اِتْ، بُ ݖِ سࣹينْ خࣷلْ بࣷنْ، ندَخْ لࣷولُ مࣷويْ تَخْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ بَرْڪࣹيلْ مبࣷولࣹمْ سࣹينْ لِگّࣺييْ، اَڪْ مبࣷولࣹمْ سࣹينُ ݧَقْ؞ ‏11 دُ ݧࣵڪّ نَگْ اَبْ ندٗولْ ݖِ رࣺيوْ مِ؞ مࣷو تَخْ مَ سَنتّ لࣹينْ نࣹ لࣹينْ نگࣹينْ نࣵدِّلَ نࣵدِّلْ سࣹينْ مبࣷڪُّمْ ندٗولْ، اَڪْ سࣹينْ نࣺيوْ-جِ-دࣷولࣹ جِ ݖِ سࣹينُمْ رࣺيوْ؞
سَنگّ اَمْ نَ لُ ڪࣷ وَرْ ݖِ جَامَمْ
‏12 سُ فࣹڪّࣹي نࣹ سࣹينْ مبࣷڪُّمْ بࣵنِّ إِسْرَاءِيلْ، گٗورْ مبَا جِگࣺينْ دَ لࣹينَ جَايْ بࣷݒَّمْ¤، بُ لࣹينْ لِگّࣺييَلࣹي جُرٗومْ بࣹنِّ اَتْ، اَتُمْ جُرٗومْ ݧَارࣹيلْ مَ، يِوِلࣹينْ ڪࣷ، مُ مࣷومْ بࣷݒَّمْ؞ ‏13 تࣹ بُ نگࣹينْ ڪࣷ يِوࣹي، مُ مࣷومْ بࣷݒَّمْ اِتْ، بُلࣹينْ ڪࣷ يِوࣹي لࣷخࣷيْ نࣹينْ؞ ‏14 سࣵڪَّلْ-لࣹينْ ڪࣷو سࣵڪَّلْ ݖِ سࣹينِ گࣵتّ، اَڪْ سࣹينُمْ ݒࣹݒّ، اَڪْ سࣹينْ بِيݧْ؞ ڪࣹمُ اَلَلْ جَ لࣹينْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ بَرْڪࣹيلࣹ، مࣷومْ نگࣹينْ ڪࣷيْ جࣷخْ؞ ‏15 فࣵتّلِڪُلࣹينْ نࣹ اَيْ جَامْ نگࣹينْ وࣷونْ ݖَ رࣺيوُمْ مِسْرَ، سࣹينْ يࣵلَّ اَجِ سَخْ جِ مࣷو لࣹينْ جࣷتْ؞ مࣷو تَخْ مَ سَنتّ لࣹينْ لِي تࣹيْ؞
‏16 ندࣹگَمْ نَگْ سࣹينْ جَامَ لࣹينْ نࣹ:‏ «دُمَ دࣹمْ، بَ لَ،» ندَخْ ݖࣷفࣹيلْ گُ مُ اَمْ ݖِ يࣹينْ اَڪْ سࣹينْ وَا ڪࣴرْ، تࣹ بࣵنّࣹيخَمْ دِ نࣹڪّ اَڪْ يࣹينْ، ‏17 سُ بࣷوبَا فَبْلࣹينْ ݒُسࣷ، نگࣹينْ سࣴسَلࣹ نࣷݒُّ جَامْ بِ سࣹينْ لَفُ بُنتّ، تࣹ نگࣹينْ بࣴنّࣹ ݒُسࣷ بِ نࣷݒَّمْ، بَ مُ جَمْ ݖِ لَفُ بُنتّ بِ، مُ دَلْدِ دࣷونْ سࣹينْ جَامْ بَ فࣵوّ؞ سࣹينْ جَامْ بُ جِگࣺينْ اِتْ، نࣷونُ نگࣹينْ دِ دࣹفْ اَڪْ مࣷومْ؞
‏18 بُلࣹينْ مِتِتْلُ لُ نگࣹينْ يِوِ سࣹينُبْ جَامْ، مُ مࣷومْ بࣷݒَّمْ، ندَخْ جَرِݧْ نَ لࣹينْ لُ فࣹيْ ݧَارِ يࣷونْ سُرْگَ بُيْ فࣹيࣹيڪُ، ݖِ جُرٗومْ بࣹنِّ اَتْ يِ مُ لࣹينْ لِگّࣺييَلْ؞ تࣹ سࣹينْ يࣵلَّ اَجِ سَخْ جِ دِنَ لࣹينْ بَرْڪࣹيلْ ݖِ مبࣷولࣹمْ لُ نگࣹينْ دࣹفْ؞
لِ وَرْ ݖِ تَاوُگْ جُرْ
‏19 مبࣷولࣹمْ لُيْ تَاوْ ݖِ سࣹينْ جُرْ گُ گُدّ، اَڪْ گُ گࣵتّ، تࣹ دِ گٗورْ، دࣹفَلْ لࣹينْ ڪࣷ سࣹينْ يࣵلَّ اَجِ سَخْ جِ ݖࣺرْ بُ سࣹلّ؞ بُلࣹينْ لِگّࣺييْلࣷو مُڪّ سࣹينْ يࣴڪّ وُ دِبْ تَاوْ، تࣹ بُلࣹينْ وَتْ سࣹينْ گࣵتّ بُ دِبْ تَاوْ؞ ‏20 فِ سࣹينْ ڪَنَمْ يࣵلَّ اَجِ سَخْ جِ نگࣹينْ ڪࣷ وَرَ خࣺيوْلࣷو اَتْ مُ جࣷتْ، يࣹينْ اَڪْ سࣹينْ نجَبࣷوتْ، فِ بࣺرَبْ بِ اَجِ سَخْ جِ تَامُ؞ ‏21 جُرْ گُ ݖِ اَمْ سِڪّ نَگْ، مُيْ سࣷوخْ، مبَا مُ گُمْبَ، مبَا مُ اَمْ سِڪّ سُ رࣺيْ سُ مُ مَنَ دࣷونْ، بُلࣹينْ ڪࣷ سَرْخَلَلْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ؞ ‏22 ݖِ سࣹينْ بِيرْ دࣴڪّ نگࣹينْ ڪࣷ وَرَ خࣺيوْلࣷو، تࣹ ڪُ سࣷبࣹوُ اَڪْ ڪُ سࣹتْ يࣺݒّ مَنْ نَ ݖَا لࣹڪّ، مࣹلْ نِ بُ دࣷونْ ڪࣺوࣺلْ مبَا ڪࣷوبَ؞ ‏23 وَايࣹ دࣹرࣹتَمْ، بُلࣹينْ ڪࣷ لࣹڪّ؞ ݖِ سُوفْ نگࣹينْ ڪࣷيْ سࣷتِّ نِمْ ندࣷخْ؞
