بࣹرْ دࣴڪِّ رَوْتُ ندِگَلْ لَ
19
‏1 گَنَّاوْ بُ لࣹينْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ وࣷلِّلࣹي خࣹيتْ يِ مُ لࣹينْ دِ جࣷخْ سࣹينُمْ رࣺيوْ، نگࣹينْ دࣵقْ لࣹينْ، بَ دࣴڪّࣹ سࣹينِ دࣴڪّ اَڪْ سࣹينِ ڪࣴرْ، ‏2 ݧࣹتِّ دࣴڪّ يُ مَگْ نگࣹينْ دِ بࣹرْ ݖَ بِيرْ رࣺيوْ مࣷومَ لࣹينْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ جࣷخْ، نگِرْ نگࣹينْ جَگࣷو ڪࣷ؞ ‏3 نگࣹينْ وَاجَلْ نَگْ يࣷونْ وُ جࣴمْ ݖِ دࣴڪّ بُ ݖِ نࣹڪّ، تࣹ نگࣹينْ سࣺدَّلࣹ ݧࣹتِّ ݒࣵݖّ رࣺيوْ مࣷومَ لࣹينْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ سࣺدّ، نگِرْ ڪࣺݒّ ڪُ رࣹيْ نِتْ، مَنْ ݖَا لࣵقُجِ؞
‏4 مبِرُمْ ڪِ رࣹيْ نِتْ تࣹ دِ فَ لࣵقُجِ بَ مُݖّ، نِي لَيْ تࣴدّࣹ:‏ دَيْ فࣹڪّ مُ رࣹيْ مࣷرࣷومَمْ تࣹ دُ تࣹيࣹيفَمْ، وَايࣹ بَݧُ ڪࣷ وࣷونْ دࣺمبّ اَڪْ بࣴڪّ-دࣺمبّ؞ ‏5 مُ مࣹلْ نِ ڪُ اࣵندّ اَڪْ مࣷرࣷومَمْ ݖَ گࣷتّ بَ، دِ گࣷرْ مَتّ؞ مُ دࣷونْ خࣵݖِّ سࣺمِّݧَمْ دِ گࣷرْ، وࣹݧْ گَ فࣹتَّخࣹ ݖَ نجࣵݒُّ لَ، دَلْ ݖَ مࣷرࣷومْ مَ، مُ دࣹي؞ ڪࣷوڪُ دَيْ لࣵقُجِ ݖِ دࣴڪّ يࣷويُ بࣹنّ، بَ مُݖّ؞ ‏6 تࣷݒّڪَتُ بَڪَّنْ بَ نَگْ وَرُلْ مَنَ دَبِ ݖِ بِيرْ خࣷلَمْ بُ تࣵنگّ، ڪِ رࣹيْ، بَ رࣹيْ ڪࣷ نگِرْ سࣷرࣹوَايُ يࣷونْ وَ جࣴمْ ݖَ دࣴڪُّبْ رَوْتُ بَ، تࣹ اࣵتّࣹبْ يࣷونْ رࣹيْلُوُ ڪࣷ، گَنَّاوْ فࣹڪُّلْ وࣷونْ مُ بَݧْ ڪَ مُ رࣹيْ، دࣺمبّ اَڪْ بࣴڪّ-دࣺمبّ؞ ‏7 لࣷولࣷو وَرَلْ مَ سَنتّ لࣹينْ، نࣹ لࣹينْ ݧࣹتِّ دࣴڪِّ رَوْتُ نگࣹينْ دِ بࣹرْ؞
‏8 بُ سࣹينْ يࣵلَّ اَجِ سَخْ جِ يَاتَلࣹي سࣹينْ سُوفْ، نَ مُ ڪࣷ گِݧَلࣹ وࣷونْ سࣹينِ مَامْ نَگْ، بَ جࣷخْ لࣹينْ مبࣷولࣹمْ رࣺيوْ مَ مُ دِگࣹ وࣷونْ نࣹ دِنَ ڪࣷ جࣷخْ سࣹينِ مَامْ، ‏9 ندࣹگَمْ سࣵمّ نگࣹينْ مبࣷولࣹمْ سَنْتَانࣹ بِي مَ لࣹينْ دࣺنڪّ تࣹيْ، نگِرْ دِ ڪࣷ جࣴفࣹ، نگِرْ سࣷݒّ سࣹينْ يࣵلَّ اَجِ سَخْ جِ، دِ اَوࣹ اَيْ يࣷونَمْ بࣺسْ بُ نࣹڪّ، سُ بࣷوبَا نَنگࣹينْ يࣷڪَّاتْ ݧࣹتِّ دࣴڪّ يُ مَگْ، مُ دࣷلِّڪُ ݖِ ݧࣹتِّ دࣴڪّ يِ نگࣹينْ جࣴڪَّ بࣹرْ؞ ‏10 سُ ڪࣷ دࣹفࣹي دࣹيسُلْ تُورْ دࣹرࣹتُ جَامْبُرْ بُ دࣹفُلْ دَرَ ݖِ سُوفَسْ رࣺيوْ، مِ لࣹينْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ سࣺدّ، بَ اࣵقُ دࣹرࣹتْ دِ لࣹينْ تࣷݒّ؞
‏11 سُ فࣹڪّࣹي نࣹ نِتْ ڪِ دَفَ بَݧْ مࣷرࣷومَمْ، سُ ڪࣷ تࣴرࣷو، دَلْ ݖِ ڪَوَمْ، بَ فࣵدّ ڪࣷ، مُ دࣹي، سُ لࣵقُجࣹي ݖِ دࣴڪّ يࣷويُ بࣹنّ، ‏12 سُ بࣷوبَا نَ مَگِ دࣴڪَّمْ يࣹبْلࣹ، ݧُ جࣴلࣹ ڪࣷ فࣷوفَ، تࣹگْ ڪࣷ ݖِ لࣷخࣷلْ تࣷݒّڪَتُ بَڪَّنْ بَ، نگِرْ مُ دࣹي؞ ‏13 بُلࣹينْ ڪࣷ خࣷولْ بࣴتُ يࣴرْمَنْدࣹ؞ نࣷونُ نگࣹينْ دِ دِنْدࣹي ݖِ إِسْرَاءِيلْ، اࣵقُ دࣹرࣹتُ جَامْبُرْ بُ دࣹفُلْ دَرَ، ندَخْ نگࣹينْ بَاخْلࣹ؞
لُ جࣴمْ ݖِ ڪࣹمُ سُوفْ اَڪْ لِمُ سࣹيدࣹ ݖِبْ اࣵتّࣹ
‏14 تࣹ اِتْ بُلࣹينْ تࣷخَلْ سࣹينْ يࣹمُڪَايُ سُوفَسْ مࣷرࣷومْ يُ مَامْ يَ يࣹمَلࣹي وࣷونْ ݖِ سࣹينْ بِيرْ ݖࣺرُ سُوفْ بَ نگࣹينْ جَگࣷو، ݖَ رࣺيوْ مَ لࣹينْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ دِ مࣷومَلࣹ؞
‏15 بࣹنُّبْ سࣹيدࣹ نَگْ دُ تَخَوْ، دِ تُومَالْ ڪࣹنّ، ݖِ گࣹنّ تࣷوݧْ مبَا بࣵڪَّارْ، اَڪْ بࣺݒّ بࣵڪَّارْ بُ مُ مَنَ بࣵڪَّارْ؞ ݖِ ڪَوْ ڪࣵدُّگْ ݧَارِ سࣹيدࣹ مبَا ݧࣹتّ لَ مبِرْ مَ دِ وࣺرࣹ؞
‏16 سُ اَبْ سࣹيدࣹبْ سَفَانْ تَخَوࣹي، تُومَالْ ڪࣹنّ، نࣹ مࣷو لَجّ يࣷونْ، ‏17 نَ ݧَارِ نِتْ، ݧَ سࣺقْ مبِرْ مَ تَخَوِ فَ ڪَنَمْ اَجِ سَخْ جِ، فَ ڪَنَمْ سَرْخَلْڪَتْ يَ، اَڪْ اࣵتّࣹڪَتْ يَ ݖَ يࣷويُ جَنتّ؞ ‏18 سُ ڪࣷ دࣹفࣹي اࣵتّࣹڪَتْ يِ سࣹيتْ بُ بَاخْ، بَ مُ وࣺرْ نࣹ سࣹيدࣹ بَ ڪَتْ، سࣹيدࣹبْ سࣷسْڪَتْ لَ بُيْ سࣹيدࣹيلْ مبࣷڪَّمْ اَوْ سࣷسْ، ‏19 تࣹ نگࣹينْ دࣹفْ اَڪْ مࣷومْ، لَ مُ نَرࣷونَ يٗبّࣹ ڪَ مُ سࣷسَلْ؞ نࣷونُ نگࣹينْ دِ تࣹنقࣹي لُ بࣷنْ ݖِ سࣹينْ بِيرْ؞ ‏20 ݧِ ݖِ دࣹسْ دِنَݧُ ڪࣷ دࣺگّ، تࣹ دِنَݧُ رَگَلْ بَ دُݧُ دࣹفَتِ لُ بࣷنْ لُ نِ مࣹلْ ݖِ سࣹينْ بِيرْ؞ ‏21 بُلࣹينْ ݖِ خࣷولْ بࣴتُ يࣴرْمَنْدࣹ؞ بَڪَّنْ، بَڪَّنَ ڪࣷيْ فࣹيْ؛ بࣴتْ، بࣴتْ؛ بࣴݧْ، بࣴݧْ؛ لࣷخࣷ، لࣷخࣷ؛ تࣵنڪّ، تࣵنڪّ؞
