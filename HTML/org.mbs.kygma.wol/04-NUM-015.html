لُ جࣴمْ ݖِ سَرَخِ ݒࣹݒّ اَڪْ دِوْ اَڪْ بِيݧْ
15
‏1 اَجِ سَخْ جِ وَخَتِ مُوسَىٰ، نࣹ:‏ ‏2 «وَخَلْ بࣵنِّ إِسْرَاءِيلْ نࣹ لࣹينْ:‏ بُ نگࣹينْ دࣹمࣹي ݖَ رࣺيوْ مَ نگࣹينْ دِ سَنݖِ، رࣺيوْ مَ مَ لࣹينْ جࣷخْ، ‏3 بُ نگࣹينْ دِ دࣹفَلْ اَجِ سَخْ جِ سَرَخُ سَوَرَ، مُيْ سَرَخُ رࣹنْدِ-دٗومَلْ، مبَا سَرَخْ سُ نِتْ دِ وࣵݖّࣷو اَوْ خَسْ، مبَا سَرَخُ يࣺينࣹ، مبَا سࣹينْ سَرَخِ مࣵگَّلْ، نگِرْ خࣹيݧَلْ خࣹتُگْ جࣵمّ ݧࣹيلْ اَجِ سَخْ جِ، تࣹ مُيْ لُ جٗگࣹ ݖِ جُرْ گُ گُدّ مبَا گُ گࣵتّ، دࣹفࣹ لࣹينْ ڪࣷ نِي:‏
‏4 «ڪِيْ سَرْخَلْ اَجِ سَخْ جِ اَبْ سَرَخَمْ، نَ سَرْخࣹوَالࣹ سَرَخُ ݒࣹݒّ بُ بࣹنّ فُڪّࣹيلُ اࣹفَبْ سُنگُفْ سُ مُݖّ اَيِبْ، دِ ݧࣹتِّ ڪِلࣷ، سُ ݧُ خِيوࣹ ݧࣹينْتࣹيلُ خَاجُبْ خِينُ دِوْ، دِ لِيتَرْ اَڪْ گࣹنّ-وࣵلّ، ‏5 اَڪْ بِيݧْ نگِرْ سَرَخُ تُورُ، ݧࣹينْتࣹيلُ خَاجُبْ خِينْ، دِ لِيتَرْ اَڪْ گࣹنّ-وࣵلّ، مُ اࣵندّ اَڪْ سَرَخُ رࣹنْدِ-دٗومَلْ مبَا بࣹنࣹينْ سَرَخْ، سُ دࣹي خَرْ مُ ندَوْ؞
‏6 «سُ دࣹي اَمْ ڪُويْ، سَرَخُ ݒࣹݒّ بِ مُيْ اࣵنْدَلْ، ݧَارِ فُڪّࣹيلِ اࣹفَيْ سُنگُفْ سُ مُݖّ اَيِبْ لَيْ دࣷونْ، دِ جُرٗومْ بࣹنِّ ڪِلࣷ، ݧُ خِيوࣹ ڪࣷ ݧࣹتّࣹيلُ خَاجُبْ خِينُ دِوْ، دِ ݧَارِ لِيتَرِ دِوْ، ‏7 اَڪْ بِيݧْ نگِرْ سَرَخُ تُورُ، ݧࣹتّࣹيلُ خَاجُبْ خِينْ، دِ ݧَارِ لِيتَرْ؞ نَ جࣷخࣹ لࣷولُ، دࣹفْ ڪࣷ خࣹتُگْ جࣵمّ، ݧࣹيلْ اَجِ سَخْ جِ؞
‏8 «سُ دࣹي سَرْخࣹ جُرْ گُ گُدّ، مُيْ سَرَخُ رࣹنْدِ-دٗومَلْ، مبَا سَرَخْ بُ مُيْ وࣵݖّࣷو اَوْ خَسْ، مبَا سَرَخُ ݖَنتّ ݖِ بِيرْ جࣵمّ، ݧࣹيلْ اَجِ سَخْ جِ، ‏9 نَ سَرْخࣹوَالࣹيڪْ جُرْ گُ گُدّ گِ، اَبْ سَرَخُ ݒࣹݒّ بُ ݧࣹتِّ فُڪّࣹيلِ اࣹفَيْ سُنگُفْ سُ مُݖّ اَيِبْ، دِ جُرٗومْ ݧࣹينْتِ ڪِلࣷ، ݧُ خِيوࣹ ڪࣷ گࣹنّ-وࣵلُّ خِينُ دِوْ، دِ ݧࣹتِّ لِيتَرْ، ‏10 اَڪْ بِيݧْ بُ مُيْ جࣷخࣹوَالࣹ، نگِرْ سَرَخُ تُورُ، دِ گࣹنّ-وࣵلُّ خِينْ، مُيْ ݧࣹتِّ لِيتَرْ؞ لࣷولُ سَرَخُ سَوَرَ لَ، دِ خࣹتُگْ جࣵمّ، ݧࣹيلْ اَجِ سَخْ جِ؞
‏11 «نࣷونُ لࣹيسْ دِ دࣹفْ بُ دࣹي سَرَخُ يࣴڪّ، مبَا اَمْ ڪُويْ، مبَا اَمْ خَرْ مُ ندَوْ، مبَا بࣴيْ؞ ‏12 نُ لِمُ جُرِ سَرَخْ يِ تࣷلُّ رࣹڪّ، نگࣹينْ دࣹفْ مُ ݖِ نࣹڪّ نࣷونُ، اࣵندّ اَڪْ لِمْ بَ؞ ‏13 مبࣷولࣹمْ نجُدُّ-جِ-رࣺيوْ، نࣷونُ لَيْ دࣹفࣹ سَرَخْ يࣷويِي، بُيْ اِنْدِ سَرَخُ سَوَرَ نگِرْ خࣹتُگْ جࣵمّ، ݧࣹيلْ اَجِ سَخْ جِ؞
‏14 «سُ دࣹي اَبْ دࣷخَنْدࣺيمْ بُ دَلْ اَڪْ يࣹينْ، مبَا ڪُ نࣹڪّ اَڪْ يࣹينْ ݖِ سࣹينْ مَاسْ يِيْ ݧࣴوْ، تࣹ مُيْ جࣷخࣹ سَرَخُ سَوَرَ نگِرْ خࣹتُگْ جࣵمّ، ݧࣹيلْ اَجِ سَخْ جِ، نِ نگࣹينْ ڪࣷيْ دࣹفࣹ رࣹڪّ لَ ڪࣷيْ دࣹفࣹ؞ ‏15 مبࣷولࣷو مِ مࣺݒّ، دࣷگَلُ يࣷونْ بࣷوبُ بࣹنّ لَ ݖِيْ دࣷونْ، مُيْ يࣹينْ، دِ دࣷخَنْدࣺيمْ بِ؞ دࣷگَلْ لَ بُيْ سَخْ فࣵوّ، ݧࣹيلْ سࣹينْ مَاسࣷو مَاسْ، تࣹ دِ بࣹنّ ݖِ يࣹينْ اَڪْ ݖِ دࣷخَنْدࣺيمْ بِ، فِ ڪَنَمْ اَجِ سَخْ جِ؞ ‏16 وࣹنّ يࣷونْ وࣹي، دِ بࣹنّ اࣵتّࣹ بِ، ݧࣹيلْ لࣹينْ، ݧࣹيلْ دࣷخَنْدࣺيمْ بِ دَلْ اَڪْ يࣹينْ؞»
اَبْ جࣷوخࣹ وَرْ نَ ݖِ ݒࣹݒّ مُ جࣴڪّ
‏17 اَجِ سَخْ جِ وَخَتِ مُوسَىٰ، نࣹ ‏18 «وَخَلْ بࣵنِّ إِسْرَاءِيلْ، نࣹ لࣹينْ، بُ نگࣹينْ دࣹمࣹي ݖَ رࣺيوْ، مَ مَ لࣹينْ جࣴمࣹ، ‏19 بَيْ لࣹڪّ ݖَ ݧَمُ رࣺيوْ مَ، نَنگࣹينْ ݖِ جࣷوخࣹ اَبْ جࣷوخࣹ، ݧࣹيلْ اَجِ سَخْ جِ؛ ‏20 سࣹينْ نࣷتُ سُنگُفْ بُ جࣴڪّ، ݖِ نگࣹينْ دِ جࣷوخࣹ مࣹنّ مبُرُمْ جࣷوخࣹ؞ نِ نگࣹينْ دِ سࣵڪّࣹ جࣷوخࣹ بُ جٗگࣹ ݖَ دࣵگَّ جَ رࣹڪّ، نِ نگࣹينْ ڪࣷيْ جࣷوخࣹي؞ ‏21 سࣹينْ نࣷتُ سُنگُفْ بُ جࣴڪّ، ݖِ نگࣹينْ دِ سࣵڪّࣹ اَبْ جࣷوخࣹ، ݧࣹيلْ اَجِ سَخْ جِ، يࣹينْ اَڪْ سࣹينْ اَسْڪَنْ؞»
سَرَخْ وَرْ نَ ݖِ مࣷيْ گُ دُلْ تࣹيࣹيفْ
‏22 سُ دُلْ ݖِ تࣹيࣹيفْ نگࣹينْ ݧࣵڪَّ جࣴفࣹ لࣹنّ ݖِ مبࣷولࣹمْ سَنْتَانࣹ يِي اَجِ سَخْ جِ دࣺنڪّ مُوسَىٰ، ‏23 ݖِ مبࣷولࣹمْ لُ لࣹينْ اَجِ سَخْ جِ سَنتّ، مُوسَىٰ جࣷتّلِ، دَلࣹ ڪࣷ ڪࣹرࣷوگْ بَ اَجِ سَخْ جِ سَنْتَانࣹي لࣷولَ، اَڪْ گَنَّاوْ گِ، بَ ݖَ سࣹينْ مَاسࣷو مَاسْ، ‏24 سُ دࣹي مࣷيْ گُ دُلْ تࣹيࣹيفْ، تࣹ مبࣷولࣷو مِ يࣺگُ ڪࣷ وࣷونْ، سُ بࣷوبَا مبࣷولࣷو مࣺݒّ اَيْ سَرْخَلْ اَوْ يࣴڪّ وُ ندَوْ وُيْ دࣷونْ رࣹنْدِ-دٗومَلْ، نگِرْ خࣹتُگْ جࣵمّ، ݧࣹيلْ اَجِ سَخْ جِ، اَڪْ سَرَخُ ݒࣹݒَّمْ، اَڪْ سَرَخُ تُورࣷومْ، نِ ڪࣷ يࣷونْ اࣵتّࣹي، اَڪْ بࣹنّ سِڪّࣹتْ نگِرْ سَرَخُ ݒٗوتُمْ بࣵڪَّارْ؞ ‏25 سَرْخَلْڪَتْ بِ دَيْ دࣹفْ لࣷولُ نجࣷتْلاَيَلْ مبࣷولࣷومْ بࣵنِّ إِسْرَاءِيلْ گࣺݒّ، نجࣺگَّلْ دَلْدِ لࣹينْ ݧࣹيلْ، گَنَّاوْ مࣷيْ گَ دُگْ تࣹيࣹيفْ، تࣹ ݧࣷومْ اِتْ اِنْدِ نَݧُ سࣹينْ سَرَخُ سَوَرَ، ݧࣹيلْ اَجِ سَخْ جِ، بࣷولࣹيڪْ سࣹينْ سَرَخُ ݒٗوتُمْ بࣵڪَّارْ فِ ڪَنَمْ اَجِ سَخْ جِ، ندَخْ سࣹينْ مࣷيْ گَ دُلْ تࣹيࣹيفْ؞ ‏26 نجࣺگَّلْ دَيْ دَلْدِ ݧࣹيلْ مبࣷولࣷومْ بࣵنِّ إِسْرَاءِيلْ گࣺݒّ، اَڪْ بࣺݒّ دࣷخَنْدࣺيمْ بُ دَلْ اَڪْ ݧࣷومْ، ندَخْ مبࣷولࣷو مࣺݒّ لَيْ سࣹينْ مࣷيْ گُ دُلْ تࣹيࣹيفْ؞
‏27 سُ دࣹي ڪࣹنّ نِتَ بࣵڪَّارْ تࣹ دُ تࣹيࣹيفَمْ، نَ سَرْخَلْ اَوْ بࣴيْ وُ جِگࣺينْ وُ مࣹنّ اَتْ، نگِرْ سَرَخُ ݒٗوتُمْ بࣵڪَّارْ؞ ‏28 سَرْخَلْڪَتْ بِ دَلْدِ ڪࣷيْ دࣹفْ فِ ڪَنَمْ اَجِ سَخْ جِ، نجࣷتْلاَيَلْ ڪِ مࣷيْ ݖِ اَگْ ݧࣵڪَّ تࣹيْ، ݖَ بࣵڪَّارَمْ بَ دُلْ تࣹيࣹيفْ؞ لࣷولُ لَ ڪࣷيْ جࣷتࣹ، نجࣺگَّلْ ݧࣹيلْ ڪࣷ؞ ‏29 مُيْ نجُدُّ-جِ-رࣺيوْ جُ بࣷڪّ ݖِ بࣵنِّ إِسْرَاءِيلْ، دِ دࣷخَنْدࣺيمْ بُ دَلْ اَڪْ ݧࣷومْ، يࣷونْ وِ وࣹنّ لَيْ دࣷونْ، ݧࣹيلْ ڪِ مࣷيْ تࣹ دُ تࣹيࣹيفَمْ؞
‏30 نِتْ ڪِ تࣷوݧْ ݖِگْ تࣹيࣹيفَمْ نَگْ، مُيْ نجُدُّ-جِ-رࣺيوْ، دِ دࣷخَنْدࣺيمْ، ڪࣷوڪُ اَجِ سَخْ جِ لَ ݧࣵڪّࣹ ڪࣹرْسَ؞ نِتْ ڪࣷوڪُ، دࣹيسْ ڪࣷيْ دَگّࣹ ݖِ بِيرِ بࣷڪَّمْ:‏ ‏31 گَنَّاوْ ڪࣵدُّگْ اَجِ سَخْ جِ لَ تࣹدَّدِلْ تࣹ ندِگَلَمْ لَ خࣴتّ، نِتْ ڪࣷوڪُ دࣹيسْ ڪࣷيْ دَگّ رࣹڪّ، مُ گࣵدُّ بࣵڪَّارَمْ؞
بࣺسُبْ نࣷفْلاَيْ وࣷرْمَالْ لَ لاَجْ
‏32 بَ بࣵنِّ إِسْرَاءِيلْ نࣹڪّࣹي ݖَ مࣵنْدِݝْ مَ، دَݧࣷو فࣹڪّ جࣹنّ وَايْ، مُيْ تَخَنْ بࣺسُبْ نࣷفْلاَيْ؞ ‏33 ݧَ ڪࣷ فࣹڪّ مُيْ تَخَنْ، يٗبُّ ڪࣷ بَ ݖَ مُوسَىٰ اَڪْ هَـٰرُونَ، اَڪْ مبࣷولࣷو مَ مࣺݒّ؞ ‏34 بَ لࣷولُ اَمࣹي ݧُ تࣹگْ ڪࣷ ݖِيْ لࣷخࣷ، ندَخْ دࣷگَلࣹيسَگُلْ وࣷونْ نࣹيسْ دِ دࣹفْ اَڪْ مࣷومْ؞ ‏35 ݖِ ڪَوْ لࣷولُ اَجِ سَخْ جِ نࣹ مُوسَىٰ:‏ «دࣹي رࣹڪّ مࣷويْ اࣵتّࣹبْ وَا جِ:‏ دࣹيسْ ڪࣷيْ گࣺنّࣹ دَلْ بِ، مبࣷولࣷو مࣺݒّ دٗورْ ڪࣷيْ خࣹيرْ، بَ مُ دࣹي؞»
‏36 مبࣷولࣷو مَ مࣺݒّ دَلْدِ ڪࣷيْ گࣺنّࣹ دَلْ بِ، دٗورْ ڪࣷيْ خࣹيرْ بَ مُ دࣹي، نَ ڪࣷ اَجِ سَخْ جِ سَنْتࣹ مُوسَىٰ؞
يࣷونْ سࣵرْتَلْ نَ وࣵلُّ ݖࣷلْ
‏37 اَجِ سَخْ جِ وَخَتِ مُوسَىٰ، نࣹ ڪࣷ:‏ ‏38 «وَخَلْ بࣵنِّ إِسْرَاءِيلْ، نࣹ لࣹينْ، ݖِ سࣹينْ مَاسࣷو مَاسْ، نَݧُ دࣹفْ اَيْ لَنگَتْ ݖِ سࣹينْ ݖَتِ مبُبّ، لَنگَتْ وُ ݖِ نࣹڪّ ݧُ نَسْ ݖِ بُومْ گُ بَخَ؞ ‏39 مُ دࣷونْ سࣹينْ لَنگَتْ يُ نگࣹينْ دِ خࣷولْ، دِ ڪࣷ فࣵتّلِڪࣷو مبࣷولࣹمْ سَنْتَانࣹيْ اَجِ سَخْ جِ، نگِرْ دِ ڪࣷ جࣴفࣹ، تࣹ بَݧَ تࣷݒّ سࣹينْ بࣵنّࣹيخُ بَڪَّنْ اَڪْ سࣹينِ گࣴتْ، بَيْ مࣷيْ؞ ‏40 سُ ڪࣷ دࣹفࣹي نگࣹينْ دِ بࣵيِّ خࣹلْ تࣹيْ جࣴفࣹ سَمَ سَنْتَانࣹ يࣺݒّ، دَلْدِ دࣷونْ ݧُ سࣹلّ، ݧࣹيلْ سࣹينْ يࣵلَّ؞ ‏41 مَنْ مَايْ اَجِ سَخْ جِ، سࣹينْ يࣵلَّ، جِ لࣹينْ گࣺنّࣹي رࣺيوُمْ مِسْرَ نگِرْ دࣷونْ سࣹينْ يࣵلَّ؞ مَايْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ؞»
