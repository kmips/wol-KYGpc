عِيسَىٰ جࣵنگَلْ نَ فَرِسࣹنْ يَ
14
‏1 بࣹنّ بࣺسُبْ نࣷفْلاَيْ لَ عِيسَىٰ رࣹيرِ وࣷونْ ڪࣴرْ گࣹنّ ڪِلِفَگْ فَرِسࣹنْ يَ؞ ݧࣷومْ نَگْ ݧُ دِ ڪࣷ بࣵيِّ خࣹلْ بُ بَاخْ؞ ‏2 ندࣹڪࣹ جࣹنّ وَايَ نگَ فَ جَنࣷوڪْ مࣷومْ، يَرَمْ وࣺݒَّ سَدّ؞ ‏3 عِيسَىٰ اࣵدُّ، نࣹ خَمْڪَتِ يࣷونْ يَاڪْ فَرِسࣹنْ يَ:‏ «وَاوْ، ݒَجُمْ بࣺسُبْ نࣷفْلاَيْ، يࣷونْ مَيࣹ نَ ڪࣷ، اَمْ دࣺيتْ؟»
‏4 ݧُ نࣹ ݖࣹلّ؞ عِيسَىٰ تࣹگْ وَا جَ لࣷخࣷ، وࣺرَلْ ڪࣷ، دَلْدِ ڪࣷ يِوِ؞ ‏5 مُ نࣹ لࣹينْ:‏ «اَنَ ڪَنْ ݖِ يࣹينْ لَ دࣷومَمْ مبَا اَوْ يࣴڪَّمْ دَانُ ݖِبْ تࣹينْ، مُ بَݧْ ڪࣷو دَلْدِ گࣺنّࣹ ݖِ بࣺسُبْ نࣷفْلاَيْ؟» ‏6 لُ ݧُ ݖَ تࣹگْ نَگْ تࣴ لࣹينْ، ݧُ نࣹ خࣹرࣹمْ؞
‏7 ݖِ بِيرْ لࣷولُ عِيسَىٰ گِسْ نَ گَنْ ݧَيْ تَامࣷو تࣷوگُ يُ ڪَنَمࣹ، مُ لࣺيبْ لࣹينْ، نࣹ لࣹينْ:‏ ‏8 «بُ لَ بࣷرࣷومْ خࣹوْ وࣷويࣹي ݖِ بࣹرْندࣹگْ ݖࣺيتْ، بُلْ وࣵلِّسِ، بَ تࣷوگࣹ تࣷوگُ بَ گࣴنَ ڪَنَمࣹ، خَمࣷو فُ ڪُ لَ سُتُ دَرَجَ دِ وُيْسࣹي بࣷرࣷومْ خࣹوْ مَ؞ ‏9 سُ بࣷوبَا مࣷومْ مِ لَ وࣷو، وࣷو ڪࣷ، مࣷويْ دِڪّ، نࣹ لَ:‏ ‹مَيَلْ ڪِي، مُ تࣷوگْ؞› نگَ اَمْ نَگْ گࣵݖّࣹيْ وَلْبَتِڪُ تࣷوگِ فَ گࣴنَ گَنَّاوࣹ؞ ‏10 بُ ݧُ لَ وࣷويࣹي اَمْ خࣹوْ ڪَيْ، دࣹمَلْ تࣷوگࣹ فَ گࣴنَ گَنَّاوࣹ، بَ ڪَ لَ وࣷو دِڪّ نࣹ لَ:‏ ‹خَرِتْ، اَگْسِلْ ݖِ ڪَنَمْ؞› سُ بࣷوبَا مُيْ سَ تࣹرَانگَ ݖِ ڪَنَمْ مبࣷولࣹمْ ݧَ نگَ بࣷڪَّلْ بࣹرْندࣹ جَ؞ ‏11 ندَخْتࣹ ڪࣺݒّ ڪُيْ رࣺيْلُ، دࣹيسْ نَ لَ تࣷرࣷخَلْ، ڪَ تࣷرࣷخْلُ لࣹيسْ دِ يࣺڪَّتِ؞»
‏12 ݖِ ڪَوْ لࣷولُ عِيسَىٰ نࣹ وَا جَ ڪࣷ وࣷو بࣹرْندࣹ:‏ «بࣷو دࣹي وَاجَلْ بࣹرْندࣹ، اَݧْ مبَا رࣹيرْ، بُلْ وࣷو سَيْ خَرِتْ، مبَا سَيْ دࣷومِ ندࣹيْ، مبَا سَيْ بࣷڪّ، مبَا سَيْ دࣴڪّ يُ اَمْ اَلَلْ، ندࣹ ݧࣷومْ اِتَمْ دَݧُيْ وَلْبَتِڪُ، وࣷو لَ رࣹڪّ، مُ دࣷونْ سَگْ ݒࣹيْ؞ ‏13 بࣷو دࣹي وَاجَلْ بࣹرْندࣹ ڪَيْ، وࣷولْ نࣺيوْ-دࣷولࣹ يِ اَڪْ لࣵگِّ يِ، اَڪْ ݧِيْ سࣷوخْ اَڪْ سِلْمَخَ يِ؞ ‏14 سُ بࣷوبَا ندࣷڪّلࣹي يَوْ، ندࣹ اَمُݧُ سَگْ ݒࣹيْ، وَايࣹ دࣹيسْ نَ لَ فࣹيْ، ڪࣹرَ ندࣹڪِّتࣹلْ اَجِ جُبْ ݧِ؞»
دࣵقْ سَ تࣹرَانگَ، رࣷمبّ سَ نجَرِݧْ
‏15 ڪࣹنّ ݖَ گَنْ ݧَ نَگْ دࣺگّ لࣷولُ، نࣹ عِيسَىٰ:‏ «ندࣷڪّلࣹي ڪُ لࣹڪّ ݖَ بࣹرْندࣹ جَ ݖَ نگُورُگْ يࣵلَّ!‏» ‏16 عِيسَىٰ نࣹ ڪࣷ:‏ «جࣹنّ وَايْ مࣷو وَاجَلࣷونْ بࣹرْندࣹ جُ مَگْ، وࣷو ݖَ ݧُ بَرࣹ؞ ‏17 بَ وَخْتُوْ لࣹڪّ جࣷتࣹي، مُ يࣹبَلْ اَبْ سُرْگَامْ ݖَ ݧَ مُ وࣷو وࣷونْ، نࣹ لࣹينْ:‏ ‹ݧࣴوْلࣹينْ، لࣺݒّ سࣷتِّ نَ لࣺيگِ؞›
‏18 «ݧُ تࣵمْبَلࣹي گࣵنْتَلْ نَگْ ݧࣷومْ ݧࣺݒّ؞ ڪِي نࣹ ڪࣷ:‏ ‹اَبْ تࣷولْ لاَ جࣴندّ، تࣹ فࣵوّ مَ خࣷولِ ڪࣷ؞ دَمَ لَيْ ݧَانْ رࣹڪّ، نگَ جࣺگَّلْ مَ؞› ‏19 ڪࣹي نࣹ:‏ ‹اَيْ لࣴڪّࣹيْ يࣴڪّ، ݖِ لاَ جࣴندّ جُرٗومْ، دَمَيْ دࣹمْ نِي، جࣺيمْلࣷو لࣹينْ بَ خَمْ؞ دَمَ لَيْ ݧَانْ رࣹڪّ، نگَ جࣺگَّلْ مَ؞› ‏20 ڪَلࣹ نࣹ ڪࣷ:‏ ‹اَبْ سࣺيتْ لاَيْ سࣷوگَ تَڪّ، لࣷولࣷو تَخْ دُمَ مَنَ ݧࣴوْ؞›
‏21 «سُرْگَ بَ دࣹلُّ، دَلْدِ ڪࣷيْ اࣵگّࣹ بࣷرࣷومْ ڪࣴرْ گَ، نجَاتِگࣹيمْ؞ مُ مࣹرْ، نࣹ سُرْگَ بَ:‏ ‹دࣹمَلْ گَاوْ ݖَ ݒࣺنݖّ يَاڪْ مبࣹدِّ دࣴڪّ بِ، تࣹ نگَ اِنْدِ فِي نࣺيوْ-دࣷولࣹ يِ، اَڪْ لࣵگِّ يِ اَڪْ سِلْمَخَ يِ اَڪْ ݧِيْ سࣷوخْ؞›
‏22 «سُرْگَ بَ دࣹمْ بَ دࣺلْسِ، نࣹ ڪࣷ:‏ ‹سَنگّ بِ، لِ نگَ سَنْتَانࣹ دࣹ دࣹفْ نَا ڪࣷ، وَايࣹ ڪࣴرْ گِ فࣹيسَگُلْ؞› ‏23 نجَاتِگࣹ لِ نࣹ ڪࣷ:‏ ‹دࣹمَلْ تࣷݒّ يࣷونْ يِ، لࣹرُ ݧَگْ يِ، تࣹ نگَ سࣷݧّ نِتْ ݧِ، ݧُ دِڪّ بَ سَمَ ڪࣴرْ فࣹيسْ؞› ‏24 مَا لࣹينْ ڪࣷ وَخْ، ݧِ مَ وࣷو وࣷونْ ݖِ سَمَ بࣹرْندࣹ جِ، ڪࣹنّ دُ ݖِ مࣷسْ؞»
تࣷݒّ عِيسَىٰ اَمْ نَ لُ مُ لاَجْ
‏25 عِيسَىٰ مࣷو دࣷونْ دࣹمْ، مبࣷولࣷو مُ رࣺيْ دَرْ ڪࣷ؞ مُ گࣹيسْتُ نࣹ لࣹينْ:‏ ‏26 «ڪࣺݒّ ڪُ ݧࣴوْ ݖِ مَنْ، تࣹ بَݧُلࣷو سَ بَايْ اَڪْ سَ ندࣹيْ، اَڪْ سَ جَبَرْ، اَڪْ سَ دࣷومْ، اَڪْ سَ بࣷڪّ يُ گٗورْ، اَڪْ سَ بࣷڪّ يُ جِگࣺينْ، بَݧُلࣷو سَ بَڪَّنُ بࣷݒّ سَخْ، دࣷو مَنَ دࣷونْ سَمَ تَالِبࣹ؞ ‏27 ڪࣺݒّ ڪُ گࣵدُّوُلْ سَ بَنْتُ بࣷݒّ بُ ݧُ لَيْ دَاجْ، تࣹ تࣷݒّ ݖِ مَنْ، دࣷو مَنَ دࣷونْ سَمَ تَالِبࣹ؞ ‏28 اَنَ ڪَنْ ݖِ يࣹينْ سَخْ مࣷويْ نَمَّ تَبَخْ تَاخُمْ ڪَوْ، تࣹ جࣴڪُّلَ تࣷوگْ، وَݧّ لَ مُ ڪࣷيْ دِڪّࣹ، نگِرْ سࣹيتْ بَ خَمْ ندَخْ اَمْ نَ لُ ڪࣷ اࣵگَّلࣹ، ‏29 بَلَ مࣷو تࣹگْ ڪࣹنُگْ تَاخْ مَ، تࣹ اَمُلْ دࣷولࣹيْ سࣷتَّلْ تَبَخْ بࣺݒّ، بَ ݧَ ڪࣷيْ سࣹيتَانْ دِ ڪࣷ ݧَاوَلْ، ‏30 نَانْ:‏ ‹وَا جِي مࣷو تࣵمْبَلࣹي تَبَخْ، تࣹ مَنُلَ اࣵگَّلࣹ!‏› ‏31 اَمْ بَنْ بُورْ مࣷويْ خَرࣹجࣹيڪْ بࣹنࣹينْ بُورْ تࣹ جࣴڪُّلَ تࣷوگْ، سࣹيتْ بَ خَمْ ندَخْ مَنْ نَا اࣵندّ اَڪْ فُڪِّ جُنِّيْ خَرࣹڪَتَمْ، نگِرْ خَرࣹجࣹيڪْ ڪُ ڪࣷ سࣷنگّسࣹيڪْ ݧَارْ فُڪِّ جُنِّيْ خَرࣹڪَتْ؟ ‏32 سُ ڪࣷ مَنُلْ نَگْ، بَ بُورْ بࣹي دِ سࣷرࣹيگُمْ لَيْ يࣹبَلْ ڪُرࣺلْ ݖَ مࣷومْ، نگِرْ سࣹيتْ نُ مُ مَنَ سࣵرْتࣷو اَڪْ مࣷومْ جࣵمّ؞ ‏33 نَڪَ نࣷونُ اِتَمْ، ڪࣹنّ ݖِ يࣹينْ دُ مَنَ دࣷونْ سَمَ تَالِبࣹ تࣹ دࣴدُّوُلْ مبࣷولࣹمْ لُ مُ اَمْ؞
‏34 «خࣷرࣷمْ دࣺيْ بَاخْ نَ، وَايࣹ بُ سࣵݒّࣹي، اَنَ لࣹيسْ ڪࣷيْ سَفَلࣹيتࣹي؟ ‏35 جَرِݧَتُلْ سُوفْ، جَرِݧَتُلْ اَبْ تࣷسْ؞ ݖَ بِتِ لࣹيسْ ڪࣷيْ تُورِ؞ ڪُ اَمِ نࣷݒّ ڪَتْ، دࣺگْلُلْ؞»
