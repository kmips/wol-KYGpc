11
‏1 سࣷݒّلࣹينْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ، تࣹ نگࣹينْ دࣺنڪࣷو ندࣺنڪَانࣹيمْ، اَڪْ دࣷگَلِ يࣷونَمْ، اَڪْ اࣵتّࣹيْ يࣷونَمْ، اَڪِ سَنْتَانࣹيمْ، بࣺسْ بُ نࣹڪّ؞ ‏2 خَمْلࣹينْ بࣺسُبْ تࣹيْ نࣹ سࣹينِ دࣷومْ تَخُلْ مَيْ وَخْ، ندَخْ ݧࣷومْ خَمُݧُ، گِسُݧُ سࣹينْ نجࣵنگّلࣹمْ يࣵلَّ اَجِ سَخْ جِ:‏ مࣵگَّايَمْ، اَڪْ دࣷولࣹيْ لࣷخࣷومْ، اَڪْ ڪࣵتَّنَمْ گُ رࣺيْ، ‏3 اَڪِ فِرْندࣹيمْ، اَڪْ جَلࣷورࣹيمْ جَ مُ جࣴفْ ݖَ دِگّ مِسْرَ، ݖِ ڪَوْ فِرْعَوْنَ بُورُ مِسْرَ اَڪْ ݖِ بِيرْ رࣺيوَمْ مࣺݒّ، ‏4 جࣴفࣹ ڪࣷوڪْ گࣵنگࣷورُ مِسْرَ، اَڪْ سࣹينِ فَسْ اَڪْ سࣹينْ وَتِيرِ خَرࣹ، بَ مُ لࣹينْ مࣴدّࣹي ݖَ گࣺيجُ بَرَخْ يَ، فࣹڪّ ݧُ نࣹ لࣹينْ دَنّ؞ اَجِ سَخْ جࣹي لࣹينْ سࣵنڪّ بَ بࣺسُبْ تࣹيْ جِي، ‏5 اَڪْ لَ مُ لࣹينْ دࣹفْ ݖَ مࣵنْدِݝْ مَ، بَ نگࣹينْ اَگْسِ بࣺرَبْ بِي، ‏6 اَڪْ لَ مُ دࣹفْ دَتَنْ اَڪْ اَبِرَمْ، دࣷومِ اࣹلْيَبْ رُبࣹنࣹينْ بَ، بَ سُوفْ ݝَيࣹي، وࣷنّ لࣹينْ، ݧࣷوڪْ سࣹينْ وَا ڪࣴرْ، اَڪْ سࣹينِ خَيْمَ، اَڪْ مبࣷولࣹمْ لُيْ دُندّ لُ لࣹينْ تࣷݒّ، ݖِ دِگّ إِسْرَاءِيلْ گࣺݒّ؞ ‏7 سࣹينِ گࣴتْ دࣺيْ نگࣹينْ تࣹگْ ݖِ مبࣷولࣹمْ جَلࣷورࣹ جُ مࣵگّ جࣷوجُ اَجِ سَخْ جِ دࣹفْ؞
‏8 ڪࣷنْ نَگْ سࣵمّلࣹينْ مبࣷولࣹمْ سَنْتَانࣹ بِ مَ لࣹينْ دِ دࣺنڪّ تࣹيْ، ندَخْ نگࣹينْ اَمْ دࣷولࣹ بَ نَنگُجِ رࣺيوْ مَ نگࣹينْ جࣵلّ جࣴمْ، نگِرْ نَنگُجِ ڪࣷ، ‏9 ندَخْ اِتَمْ نگࣹينْ گُدّ فَنْ ݖَ رࣺيوْ مَ اَجِ سَخْ جِ گِݧࣷونَ جࣷخْ سࣹينِ مَامْ، ݧࣷوڪْ سࣹينْ اَسْڪَنْ؛ رࣺيوْ مُ مࣹيوْ مَاڪْ لࣹمْ جَ تُورࣷو؞ ‏10 رࣺيوْ مࣷومَ نگࣹينْ جࣴمْ، دِ ڪࣷ نَنگُجِ، دࣺيْ، نِرࣷووُلْ اَڪْ رࣺيوُمْ مِسْرَ گَ نگࣹينْ جٗگࣹ، فَ نگࣹينْ دَانْ جِوࣹ، بَ نࣷݒِّ تࣷݒّ سࣹينُمْ جِوُ، دِ ڪࣷ سُوخَتࣹ دࣷولࣹيْ تࣵنڪّ نِ مبࣹيُمْ دࣴرْ؞ ‏11 رࣺيوْ مࣷومَ نگࣹينْ جࣵلّ جࣴمْ، نگِرْ نَنگُجِ ڪࣷ، رࣺيوُمْ تُندّ اَڪِ خُرْ لَ، تࣹ ندࣷخُمْ تَوْ لَيْ نَانْ؞ ‏12 رࣺيوْ لَ مُ سࣹينْ يࣵلَّ اَجِ سَخْ جِ يࣺگْ، فُ مُ تࣷلُّ سࣹينْ گࣴتِ يࣵلَّ اَجِ سَخْ جَا نگِ نࣹ ڪࣷ جࣵڪّ، تࣹرُتࣹلْ اَتْ بَ ندࣹيتࣹلَمْ؞
‏13 سُ فࣹڪّࣹي نࣹ نَگْ سࣵمّ نگࣹينْ بَ سࣵمّ سَمَ سَنْتَانࣹ يِ مَ لࣹينْ دࣺنڪّ تࣹيْ جِي، نگِرْ سࣷݒّ سࣹينْ يࣵلَّ اَجِ سَخْ جِ، دِ ڪࣷ جَامࣷو سࣹينْ لࣺݒِّ خࣷلْ، اَڪْ سࣹينْ لࣺݒِّ بَڪَّنْ، ‏14 سُ بࣷوبَا، يࣵلَّ نࣹي:‏ «مَا لࣹينْ دِ مَيْ سࣹينْ تَوُبْ رࣺيوْ، سَا يُ جࣷتࣹي:‏ لࣷلِّبْ رࣺيوْ مَ دَيْ تَوْ¤، ݖࣷورࣷونَمْ تَوْ، نگࣹينْ دَجَلࣹ سࣹينُمْ ݒࣹݒّ، اَڪْ سࣹينْ بِيݧْ بُ بࣹيسْ، اَڪْ سࣹينْ دِوْ گُ يࣹيسْ؞ ‏15 سُ بࣷوبَا مَايْ مَيࣹمْ ݧَخْ ݖِ سࣹينُبْ اࣵلّ نگِرْ سࣹينُگْ جُرْ، يࣹينْ اِتْ نگࣹينْ دِ لࣹڪّ بَ رࣹگّ؞»
‏16 وَتُّلࣹينْ سࣹينْ بࣷݒّ نَگْ، بَلَ لࣹينْ سࣹينْ خࣹلَ نَخْ، نگࣹينْ لَجّ، دِ جَامُ يࣹنࣹينْ يࣵلَّ، دِ لࣹينْ سُجّٗوتَلْ، ‏17 بَ سࣵنجُمْ اَجِ سَخْ جِ تࣵڪَّلْ لࣹينْ، مُ نࣹ رࣵݒّ اَسَمَانْ، تَوْ تَخَوْ، سُوفْ نࣷݒّࣹي مࣹݧَّلْ، بَ نگࣹينْ گَاوَ سࣵنڪࣷو ݖَ رࣺيوْ مُ بَاخْ مَ لࣹينْ اَجِ سَخْ جِيْ جࣷخْ؞
‏18 دࣹفْلࣹينْ سَمَ ڪࣵدُّ يِي ݖِ سࣹينُمْ خࣹلْ اَڪْ ݖِ سࣹينْ خࣷلْ؞ تَڪّلࣹينْ ڪࣷ، مُيْ فِرْندࣹ ݖِ سࣹينْ لࣷخࣷ، تࣹ نَ دࣷونْ اَمْ مبَنْدَلْ ݖِ سࣹينْ دِگَّنْتࣹيْ گࣴتْ؞ ‏19 نَنگࣹينْ ڪࣷ جࣵنگَلْ سࣹينِ دࣷومْ، دِ ڪࣷ وَخࣹ بُ نگࣹينْ تࣷوگࣹي سࣹينْ بِيرْ نࣺيگْ، اَڪْ بُ نگࣹينْ دِ دࣷخْ ݖِوْ يࣷونْ، اَڪْ بُ نگࣹينْ تࣴدّࣹي، اَڪْ بُ نگࣹينْ جٗگࣹي؞ ‏20 تࣹ اِتَمْ بِندّلࣹينْ ڪࣷ ݖِ سࣹينْ ڪَوْ جࣴنِ بُنْتُ نࣺيگْ، اَڪْ ݖِ سࣹينْ بُنْتِ دࣴڪّ، ‏21 ندَخْ سࣹينِ فَنْ، اَڪْ سࣹينْ فَنِ دࣷومْ، بَرࣹي ݖَ رࣺيوْ مَ اَجِ سَخْ جِ گِݧَلࣷونْ سࣹينِ مَامْ، نِ ڪࣹمُ فَنْ يِ اَسَمَانْ تِيمْ سُوفْ؞
‏22 ندࣹگَمْ دࣺيْ سࣵمّ نگࣹينْ بَ سࣵمّ مبࣷولࣹمْ سَنْتَانࣹ بِي مَ لࣹينْ دࣺنڪّ نگِرْ دِ ڪࣷ جࣴفࣹ، بَ سࣷݒّ سࣹينْ يࣵلَّ اَجِ سَخْ جِ، دِ دࣷخࣹ مبࣷولࣹمْ اَيْ يࣷونَمْ تࣹ ݝࣷيْ ݖِ مࣷومْ، ‏23 سُ بࣷوبَا اَجِ سَخْ جࣹي لࣹينْ دِ دࣵقَلْ مبࣷولࣹمْ خࣹيتْ يࣷويُ، تࣹ يࣹينَيْ نَنگُ رࣺيوِ خࣹيتْ يُ لࣹينْ اࣴݒّ، اࣴݒّ لࣹينْ دࣷولࣹ؞ ‏24 فࣺݒّ فُ نگࣹينْ تࣹگْ سࣹينْ تࣵنڪّ، يࣹينَ ڪࣷيْ مࣷومْ؞ لَ ڪࣷ دَلࣹ ݖَ مࣵنْدِݝْ مَ بَ ݖَ لِبَݝْ، دَلࣹيتِ دࣹخُ اࣹفْرَاتْ بَ ݖَ گࣺيجْ گَ ݖَ سࣷوُ، لࣺݒّ سࣹينْ سُوفْ لَيْ دࣷونْ؞ ‏25 دُ ڪࣹنّ ڪُيْ تَخَوْ فِ سࣹينْ ڪَنَمْ؞ سࣹينْ يࣵلَّ اَجِ سَخْ جِ مࣷويْ دࣹفْ نَ مُ لࣹينْ ڪࣷ وَخࣹ وࣷونْ، بَ وَا دࣺنْدُ رࣺيوْ مُ نگࣹينْ تࣹگْ سࣹينْ تࣵنڪّ، ݧࣷومْ ݧࣺݒّ، يࣹينْ لَݧُيْ تِيتْ، يࣹينْ لَݧُيْ رَگَلْ؞
بَرْڪࣹ اَڪْ اَلْڪࣵنْدࣹ دࣴگْمَلْ نَ
‏26 مَا نگِي نَگْ دِ لࣹينْ تࣹگَلْ بࣺسُبْ تࣹيْ بَرْڪࣹ اَڪْ اَلْڪࣵنْدࣹ؞ ‏27 بَرْڪࣹي نگِ، ندࣹگَمْ دࣺگَّلْ نگࣹينْ سࣹينْ سَنْتَانࣹيْ يࣵلَّ اَجِ سَخْ، جِ مَ لࣹينْ دِ دࣺنڪّ تࣹيْ؞ ‏28 اَلْڪࣵنْدࣹي نگِ، ندࣹگَمْ دࣺگَّلُلࣹينْ سࣹينْ سَنْتَانࣹيْ يࣵلَّ اَجِ سَخْ جِ، خَنَا وَݖّ يࣷونْ وِ مَ لࣹينْ دِ دࣺنڪّ بࣺسُبْ تࣹيْ، نگِرْ تࣷݒّ يࣹنࣹينْ يࣵلَّ يُ نگࣹينْ خَمُلࣷونْ؞
‏29 بُ لࣹينْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ يٗبّࣷو ݖَ رࣺيوْ مَ نگࣹينْ جࣴمْ، نگِرْ نَنگُجِ ڪࣷ، نَنگࣹينْ تُدّ بَرْڪࣹ بِ ݖَ ڪَوْ تُنْدُ گَرِسِمْ، تࣹ نگࣹينْ تُدّ اَلْڪࣵنْدࣹ جِ ݖَ ڪَوْ تُنْدُ اࣹبَلْ؞ ‏30 ݧَارِ تُندّ يࣷويُ، ݧَ نگَ مࣷوسْ ݖَ وࣵلاَّ دࣹخُ يُرْدَنْ، ݖَ وࣵلاَّ يࣷونُ سࣷوُ بَ، ݖَ رࣺيوُمْ ڪَنَانࣹينْ، ݧَ دࣴڪّࣹ جࣷورْ گَ جࣵڪَّارْلࣷوڪْ گِلْگَلْ، ݖَ وࣹتِ گَرَبِ مࣷرࣹ يُ مَگْ يَ؞ ‏31 لࣺيگِ نَگْ يࣹينَ نگِ جࣵلِّ دࣹخُ يُرْدَنْ، نگِرْ نَنگُجِ رࣺيوْ مِ لࣹينْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ جࣷخْ؞ بُ نگࣹينْ ڪࣷ نَنگࣷو بَ دࣴڪّ فَ نَگْ، ‏32 سࣵمّࣷولࣹينَ جࣴفࣹ مبࣷولࣹمْ دࣷگَلْ يࣹيڪْ اࣵتّࣹيْ يࣷونْ يِ مَ تࣹگْ فِ سࣹينْ ڪَنَمْ، بࣺسُبْ تࣹيْ جِي؞
