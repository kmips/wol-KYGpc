لُ جࣴمْ ݖِ سࣷبࣹيْ گٗورْ
15
‏1 اَجِ سَخْ جِ وَخْ نَ اِتْ مُوسَىٰ اَڪْ هَـٰرُونَ نࣹ لࣹينْ:‏ ‏2 «وَخْلࣹينْ اَڪْ بࣵنِّ إِسْرَاءِيلْ نࣹ لࣹينْ:‏ گٗورْ گُ اࣵندّ اَڪْ جࣵنگࣷرࣷ جُ سَبَبْ نگٗورَامْ دِ خࣹلِّ مبࣺرْ، ڪࣷوڪُ سࣷبࣹوُ نَ؞ ‏3 مبࣺرْ مِ مُيْ خࣹلِّ مࣷو ڪࣷ سࣷبࣹيلْ؛ سُ نگٗورَ لِيْ خࣹلِّ مبࣺرْ مبَا سُ ڪࣷ مبࣺرْ مِ فَتّࣹي، وَا جِ سࣷبࣹوُ نَ؞ ‏4 لَلْ بَ مُ تࣴدّ سࣷبࣹوُ نَ، تࣹ لࣺݒّ لُ مُ تࣷوگْ لࣷولَ سࣷبࣹوُ نَ؞ ‏5 ڪࣺݒّ ڪُ لاَلْ لَلَمْ، نَ فٗوتْ اَيْ يࣺرࣹيمْ، سَنگُ، تࣹ دُ تࣹي مُ يࣹنْدࣷو سࣷبࣹ بَ جَنتّ سࣷ؞ ‏6 ڪُ تࣷوگْ ݖِ لࣹنّ لُ بࣷرࣷومْ جࣵنگࣷرࣷيْ خࣹلِّ تࣷوگࣷونْ، نَ فٗوتْ اَيْ يࣺرࣹيمْ، سَنگُ، تࣹ دُ تࣹي مُ يࣹنْدࣷو سࣷبࣹ بَ جَنتّ سࣷ؞ ‏7 ڪُ لاَلْ بࣷرࣷومْ جࣵنگࣷرࣷيْ خࣹلِّ وَرْ نَا فٗوتْ اَيْ يࣺرࣹيمْ، سَنگُ، تࣹ دُ تࣹي مُ اࣵندّ اَڪْ سࣷبࣹ بَ جَنتّ سࣷ؞ ‏8 سُ ڪُ اࣵندّ اَڪْ جࣵنگࣷرࣷيْ خࣹلِّ تِفْلࣹي ڪُ تَقُلْ سࣷبࣹ، نَ ڪࣷوڪُ فٗوتْ اَيْ يࣺرࣹيمْ، سَنگُ، تࣹ دُ تࣹرࣹ مُ يࣹنْدࣷو سࣷبࣹ بَ جَنتّ سࣷ؞ ‏9 گࣺݒّ تࣹگُ دَامَرْ گُ بࣷرࣷومْ جࣵنگࣷرࣷيْ خࣹلِّ وَرࣷونْ، سࣷبࣹوُ نَ؞ ‏10 ڪُ لاَلْ لࣹنّ لُ مُ تࣷوگࣷونْ لُ ڪࣷ فࣺيتࣹ وࣷونْ سُوفْ دِنَ يࣹنْدࣷو سࣷبࣹ بَ جَنتّ سࣷ، تࣹ ڪࣺݒّ ڪُ يࣺڪَّتِ لࣷولَ نَ فٗوتْ اَيْ يࣺرࣹيمْ، سَنگُ، تࣹ دُ تࣹرࣹ مُ يࣹنْدࣷو سࣷبࣹ بَ جَنتّ سࣷ؞ ‏11 بࣷرࣷومْ جࣵنگࣷرࣷيْ خࣹلِّ بُ رَخَسُلْ اَيْ لࣷخࣷومْ، ڪُ مُ لاَلْ، ڪࣷوڪُ، نَ فٗوتْ اَيْ يࣺرࣹيمْ، سَنگُ، تࣹ دُ تࣹرࣹ مُ يࣹنْدࣷو سࣷبࣹ بَ جَنتّ سࣷ؞ ‏12 ندَبَلْ خَنْدࣹيرْ لُ بࣷرࣷومْ جࣵنگࣷرࣷيْ خࣹلِّ لاَلْ، دࣹيسْ ڪࣷو وَرَ تࣷجْ؞ سُ دࣹي مبࣷولࣹمْ ندَبَلْ بَنتّ، دࣹيسْ ڪࣷيْ رَخَسْ؞
‏13 «گَنَّاوْ لࣷولُ بُ بࣷرࣷومْ جࣵنگࣷرࣷيْ خࣹلِّ جِ تࣵگّࣷوڪْ سࣷبࣹيْ جࣵنگࣷرࣷ جَ، دَفَ وَرَ وَݧّ جُرٗومْ ݧَارِ فَنْ نگِرْ سࣹتْلُ، بَ نࣷݒِّ فٗوتْ اَيْ يࣺرࣹيمْ، سَنگࣷو ندࣷخْ مُ بَلّࣹ ݖِ بࣴتُ ندࣷخْ، دࣷورَ سࣹتْ؞ ‏14 بࣺسُبْ جُرٗومْ ݧࣹتّࣹيلْ بَ، نَ وُتْ ݧَارِ ݒِتَخْ مبَا ݧَارِ خَتِ يُ ندَوْ تࣹ تࣹيوِ فِ ڪَنَمْ اَجِ سَخْ جِ ݖَ بُنتّ خَيْمَبْ ندَجࣹ مَ، دَلْدِ ڪࣷيْ جࣷخْ سَرْخَلْڪَتْ بِ؞ ‏15 سَرْخَلْڪَتْ بِ دَيْ رࣹنْدِ يࣷويُ، مࣹنّ مِ دِ سَرَخُ ݒٗوتُمْ بࣵڪَّارْ، مِ ݖِ دࣹسْ دِ سَرَخُ رࣹنْدِ-دٗومَلْ؞ نࣷونُ لَ ڪࣷ سَرْخَلْڪَتْ بِيْ دࣹفَلࣹ نجࣷتْلاَيَمْ فِ ڪَنَمْ اَجِ سَخْ جِ ندَخْ جࣵنگࣷرࣷيْ خࣹلّࣹيمْ جࣷوجُ؞
‏16 «سُ گٗورْ گࣺنّࣹي وَسَلْ، وَرْ نَا رَگّ يَرَمَمْ وࣺݒّ، تࣹ دُ تࣹرࣹ مُ يࣹنْدࣷو سࣷبࣹ بَ جَنتّ سࣷ؞ ‏17 لࣺݒّ لُيْ يࣺرࣹ مبَا دࣹرْ بُ وَسَلْ تَقْ، دࣹيسْ ڪࣷيْ فٗوتْ تࣹ دُ تࣹرࣹ مُ يࣹنْدࣷو سࣷبࣹ بَ جَنتّ سࣷ؞ ‏18 سُ دࣹي جِگࣺينْ جُ گٗورْ اࣵنْدَلْ، تࣹ وَسَلْ گࣺنّ، نَݧُ سَنگُ، تࣹ دُ تࣹرࣹ ݧُ يࣹنْدࣷو سࣷبࣹ بَ جَنتّ سࣷ؞
لُ جࣴمْ ݖِ سࣷبࣹيْ جِگࣺينْ
‏19 «سُ جِگࣺينْ گِسࣹي مبࣴرࣴگْ، سࣷبࣹيْ مبࣴرࣴگْ مَ دَيْ اࣵندّ اَڪْ مࣷومْ جُرٗومْ ݧَارِ فَنْ تࣹ ڪࣺݒّ ڪُ لاَلْ ڪُ جِگࣺينْ ڪَ دِنَ يࣹنْدࣷو سࣷبࣹ بَ جَنتّ سࣷ؞ ‏20 لࣺݒّ لُ مُ تࣴدّ تࣹ فࣹڪّ مُ گِسْ بَاخْ، لࣷولَ سࣷبࣹوُ نَ تࣹ لࣺݒّ لُ مُ تࣷوگْ، لࣷولَ سࣷبࣹوُ نَ؞ ‏21 تࣹ اِتْ ڪࣺݒّ ڪُ لاَلْ اَبْ لَلَمْ، وَرْ نَا فٗوتْ اَيْ يࣺرࣹيمْ، سَنگُ، تࣹ دُ تࣹرࣹ مُ يࣹنْدࣷو سࣷبࣹ بَ جَنتّ سࣷ؞ ‏22 ڪࣺݒّ ڪُ لاَلْ لࣹنّ لُ مُ تࣷوگࣷونْ وَرْ نَا فٗوتْ اَيْ يࣺرࣹيمْ، سَنگُ، تࣹ دُ تࣹي مُ يࣹنْدࣷو سࣷبࣹ بَ جَنتّ سࣷ؞ ‏23 سُ تࣷوگࣷونْ ݖِ لَلْ مبَا لࣹنࣹينْ لُ مُ تࣷوگࣷونْ، ڪُ لاَلْ لࣷولَ دِنَ يࣹنْدࣷو سࣷبࣹ بَ جَنتّ سࣷ، ‏24 تࣹ ڪُ ڪࣷ دࣴڪّࣷو، بَ مبࣴرࣴگْ تَقْ ڪࣷ، ڪࣷوڪَ سࣷبࣹوُ نَ دِيرُ جُرٗومْ ݧَارِ فَنْ، تࣹ ڪَ ڪࣷ دࣴڪّࣷو، لَلْ بُ مُ تࣴدّ سࣷبࣹوُ نَ؞
‏25 «ڪُيْ خࣴݒّ دࣹرࣹتْ اَيْ فَنْ يُ بَرࣹ تࣹ دُ جَمࣷنࣷيْ بَاخَمْ، مبَا مُ گِسْ مبࣴرࣴگْ مُ وࣹيسْ اࣵݒّ بِ مُ ڪࣷ بَاخࣷو گِسْ، ڪࣷوڪَ سࣷبࣹوُ نَ دِيرُ فَنْ يِ مُيْ خࣴݒّ؞ دَيْ مࣹلْ نِ بُ نࣹڪّࣹي ݖِ جَمࣷنࣷيْ بَاخَمْ؞ ‏26 لَلْ بُ مُ تࣴدّ ݖِ دِيرْ بِ مُيْ خࣴݒّ يࣺݒّ، دَيْ مࣹلْ نِ لَلَمْ بُ ڪࣷ فࣹڪّ ݖِ جَمࣷنࣷيْ بَاخْ تࣹ لࣺݒّ لُ مُ تࣷوگْ دَيْ سࣷبࣹوُ، نِ بُ نࣹڪّࣷونْ ݖِ جَمࣷنࣷيْ بَاخَمْ؞ ‏27 ڪُ لاَلْ لࣹنّ ݖِ يࣷويُ يࣴفْ، ڪࣷوڪَ سࣷبࣹوُ نَ؞ وَرْ نَا فٗوتْ اَيْ يࣺرࣹيمْ، سَنگُ، تࣹ دُ تࣹرࣹ مُ يࣹنْدࣷو سࣷبࣹ بَ جَنتّ سࣷ؞
‏28 «بُ خࣴݒّࣹي بَ جࣹيخَلْ، دِنَ وَݧّ جُرٗومْ ݧَارِ فَنْ، دࣷورَ تࣵگّࣷوڪْ سࣷبࣹ سَ؞ ‏29 بࣺسُبْ جُرٗومْ ݧࣹتّࣹيلْ بَ نَ وُتْ ݧَارِ ݒِتَخْ مبَا ݧَارِ خَتِ يُ ندَوْ، يٗبُّ ݖَ سَرْخَلْڪَتْ بَ، ݖَ بُنتّ خَيْمَبْ ندَجࣹ مَ؞ ‏30 سَرْخَلْڪَتْ بِ دَيْ رࣹنْدِ مࣹنّ مِ مُيْ سَرَخُ ݒٗوتُمْ بࣵڪَّارْ، مِ ݖِ دࣹسْ دࣷونْ سَرَخُ رࣹنْدِ-دٗومَلْ؞ سُ ڪࣷ دࣹفࣹي سَرْخَلْڪَتْ بِ دࣹفَلْ ڪࣷ ڪࣷ نجࣷتْلاَيْ فِ ڪَنَمْ اَجِ سَخْ جِ ندَخْ سࣷبࣹيْ خࣴݒّ جَ؞
‏31 «فࣹخࣹلࣹينْ بَ بࣵنِّ إِسْرَاءِيلْ دِ تࣵگّࣷوڪْ سࣹينِ سࣷبࣹ، لُ ڪࣷ مࣷيْ دِنَݧُ دࣹي ندَخْ سࣹينْ سࣷبࣹ، نگِرْ دَݧُيْ تَقَلْ سَمَ مࣵڪَّانْ مِ ݖِ سࣹينْ بِيرْ؞»
‏32 لࣷولُ لَ يࣷونْ دِگْلࣹ ݖِ مبِرِ گٗورْ گُ اࣵندّ اَڪْ جࣵنگࣷرࣷيْ خࣹلِّ مبَا مُ گࣺنّࣹ وَسَلْ، بَ سࣷبࣹوُ؛ ‏33 اَڪْ جِگࣺينْ جُ نࣹڪّ ݖِ جَمࣷنࣷيْ بَاخَمْ؛ ݖِ گࣵتَّلْ ڪࣺݒّ ڪُ تَقْ سࣷبࣹ سُيْ گࣺنّࣹ ݖِ مࣷومْ، مُيْ گٗورْ مبَا جِگࣺينْ، اَڪْ اِتَمْ گٗورْ گُ دࣴڪّࣷو جِگࣺينْ جُ سࣷبࣹوُ؞
