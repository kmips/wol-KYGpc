دࣹي مࣷويْ اࣵتّࣹبْ ڪُيْ خِيرْتَلࣹ ݖِ بࣷڪَّالࣹ
13
‏1 مبࣷولࣹمْ ڪࣵدُّ گِ مَ لࣹينْ دࣺنڪّ نَگْ، سࣵمّࣷولࣹينْ ڪࣷو جࣴفࣹ؞ بُلࣹينْ ݖِ يࣷڪّ لࣹنّ، بُلࣹينْ ݖِ وࣵݧِّ لࣹنّ؞ ‏2 گَنَّاوْ لࣷولُ سُ اَبْ يࣷنࣹنتّ فࣹيݧࣹي ݖِ سࣹينْ بِيرْ، مبَا بࣷرࣷومْ گࣺنتّ، بُ لࣹينْ دِگࣹي فِرْندࣹ مبَا ڪِرَامَ، ‏3 بَ فِرْندࣹ لَ مبَا ڪِرَامَ گَ مُ لࣹينْ وَخࣷونْ دِڪّ، سُ لࣹينْ نࣹي:‏ «نَنُ تࣷݒّ يࣹنࣹينْ يࣵلَّ، جَامُ لࣹينْ،» يࣵلَّ يُ نگࣹينْ خَمُلْ، ‏4 بُلࣹينْ دࣺگْلُ وَخُ يࣷنࣹنتّ بࣷوبُ، مبَا بࣷرࣷومْ گࣺنتّ بࣷوبُ، ندَخْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ مࣷو لࣹينْ دِ نَتُّ نگِرْ خَمْ ندَخْ يࣹينَ نگِ ڪࣷيْ سࣷݒّࣹ سࣹينْ لࣺݒِّ خࣷلْ اَڪْ سࣹينْ لࣺݒِّ بَڪَّنْ؞ ‏5 سࣹينْ يࣵلَّ اَجِ سَخْ جِ مࣷومْ نگࣹينْ دِ تࣷݒّ، مࣷومْ نگࣹينْ دِ رَگَلْ؞ اَيْ سَنْتَانࣹيمْ نگࣹينْ دِ سࣵمّ، بَاتَمْ نگࣹينْ دِ دࣺگْلُ، مࣷومْ نگࣹينْ دِ جَامُ، تࣹ ݖِ مࣷومْ نگࣹينْ وَرَ ݝࣷيْ؞ ‏6 يࣷنࣹنتّ بࣷوبُ نَگْ مبَا بࣷرࣷومْ گࣺنتّ بࣷوبُ، دࣹي مࣷويْ اࣵتّࣹيمْ، ندَخْ مࣷو وࣷوتࣹ اَگْ فِݒُّ ݖِ سࣹينْ ڪَوْ يࣵلَّ اَجِ سَخْ، جِ لࣹينْ جࣴلࣹ ݖَ رࣺيوُمْ مِسْرَ، جࣷتࣹ لࣹينْ ڪࣴرُگْ نجَامْ گَ؞ مُ نَرْ لࣹينَ فࣺلࣹ ݖِ يࣷونْ وِ لࣹينْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ سَنتّ نگࣹينْ دࣷخࣹ ڪࣷ؞ نࣷونُ نگࣹينْ دِ تࣹنقࣹي لُ بࣷنْ ݖِ سࣹينْ بِيرْ؞
‏7 سࣹينْ مبࣷڪّ مُ نگࣹينْ بࣷڪَّلْ جࣹنّ ندࣹيْ، مبَا سࣹينْ دࣷومْ جُ گٗورْ، مبَا جُ جِگࣺينْ، مبَا سࣹينْ جَبَرْ جُ نگࣹينْ خࣹجّࣷو، مبَا سࣹينْ خَرِتُ بࣹنّ بَڪَّنْ، سُ لࣹينْ دࣹي خِيرْتَلْ، نَانْ لࣹينْ ݖِ ڪُمْݒَ:‏ «نَنُ جَامُجِ يࣹنࣹينْ يࣵلَّ،» يࣵلَّ يُ نگࣹينْ خَمُلْ، دُ يࣹينْ دُ سࣹينِ مَامْ، ‏8 ݖِ يࣵلَّيْ خࣹيتْ يِ لࣹينْ سࣺقْ، ݧِ لࣹينْ جࣹگࣹ، اَڪْ ݧِ لࣹينْ سࣷرࣹ، ݖِ اࣵدِّنَ سࣺݒّ، ݖَتْ بَ ݖَتْ، ‏9 سُ بࣷوبَا بُلࣹينْ ڪࣷ نَنگُلْ، بُلࣹينْ ڪࣷ دࣺگْلُ، بُلࣹينْ ڪࣷ خࣷولْ بࣴتِ يࣴرْمَنْدࣹ، بُلࣹينْ ڪࣷ بَالْ، بُلࣹينْ ڪࣷ نࣴبَّلْ؞ ‏10 رࣹيْلࣹينْ ڪࣷ رࣹڪّ؞ يࣹينْ ݧِ مُ دࣷونْ جࣺيمَ يٗبَّالࣹ، نَ ڪࣷ سࣹينْ لࣷخࣷ جࣴڪَّ دَلْ، نگِرْ رࣹيْ ڪࣷ، بَلَ ݖࣹي لࣷخࣷلْ مبࣷولࣷو مࣺݒّ تࣷݒّ؞ ‏11 دٗورْلࣹينْ ڪࣷيْ دࣷجْ بَ مُ دࣹي، ندَخْ مࣷو لࣹينْ دࣷونْ فࣹخࣹي سࣷرࣹلࣹيڪْ سࣹينْ يࣵلَّ اَجِ سَخْ، جِ لࣹينْ جࣴلࣹ رࣺيوُمْ مِسْرَ، ڪࣴرُگْ نجَامْ گَ؞ ‏12 إِسْرَاءِيلْ گࣺݒَّ ڪࣷيْ دࣺگّ، دَلْدِ رَگَلْ بَ دࣹيسَتُلْ دࣹفْ مبࣷنْ گُ نِ مࣹلْ ݖِ سࣹينْ بِيرْ؞
‏13 بُ نگࣹينْ دࣺگّࣹي، ݖِ بࣹنّ دࣴڪّ، ݖِ دࣴڪّ يِ لࣹينْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ جࣷخْ، نگِرْ نگࣹينْ دࣴڪّࣹ، ݧُ نࣹ ‏14 اَيْ نِتْ ݧُ تࣹڪِّوُلْ دَرَ، تࣹ بࣷڪّ ݖِ يࣹينْ، ݧࣷو جٗگْ دِ وَݧِّ سࣹينْ وَا دࣴڪّ، نَانْ لࣹينْ:‏ «نَنُ جَامُجِ يࣹنࣹينْ يࣵلَّ،» يࣵلَّ يُ نگࣹينْ خَمُلْ، ‏15 سُ بࣷوبَا نَنگࣹينْ گࣴسْتُ، سࣹيتْ تࣹ لاَجْتࣹ بُ بَاخْ، بَ دࣴگّ گُ وࣺرْ فࣹيݧࣹ ݖَ مبِرْ مَ، نࣹ دࣹفࣹيسْ نَ مࣷومُ مبِرْ مُ جࣷمبّلُ ݖِ سࣹينْ بِيرْ، ‏16 نگࣹينْ دَلْدِ جَمْ رࣹڪّ وَا دࣴڪّ بࣷوبُ، لࣹيلْ لࣹينْ ݧَوْڪَيْ سَامَرْ، فَاگَاگَلْ دࣴڪّ بَ، اَڪْ مبࣷولࣹمْ لَ ݖَ بِيرَمْ، جُرْ گَ اِتْ، نگࣹينْ لࣹيلْ لࣹينْ ݧَوْڪَيْ سَامَرْ؞ ‏17 مبࣷولࣹمْ اَلَلُ سࣴخࣴتࣷو جُ جٗگࣹ ݖَ دࣴڪّ بَ، دَجَلࣹلࣹينْ ڪࣷ ݖَ دِگّ ݒࣺنݖّ مَ، تࣹ نگࣹينْ لَڪّ دࣴڪّ بَ، بࣷولࣹيڪْ ݖࣴخࣴتࣷو مَ، دࣹفْ ڪࣷ دٗومَلْ بُ مَتْ سࣴڪّ، ݧࣹيلْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ؞ دَيْ دࣷونْ اَبْ گࣹنتّ بَ فࣵوّ، دࣹيسُ ڪࣷ تَبَخَاتْ؞ ‏18 بُ دَرَ دࣹسْ ݖِ سࣹينْ لࣷخࣷ، لُ بࣷڪّ ݖِ اَلَلْ جࣷوجُ ݧُ وَرَ فَاگَاگَلْ؞ سُ ڪࣷ دࣹفࣹي اَجِ سَخْ جِ دࣴدُّ سࣵنجَمْ مُ تࣵنگّ، يࣴرࣴمْ لࣹينْ، ݧࣹيوَنْتࣹ لࣹينْ، يࣷڪّ لࣹينْ نَ مُ ڪࣷ گِݧَلࣹ وࣷونْ سࣹينِ مَامْ، ‏19 ندࣹگَمْ يࣹينَ نگِ دࣺگّ سࣹينْ ڪࣵدُّگْ يࣵلَّ اَجِ سَخْ جِ، نگِرْ سࣵمّ مبࣷولࣹمْ سَنْتَانࣹيمْ يِ مَ لࣹينْ دِ دࣺنڪّ بࣺسُبْ تࣹيْ، نگِرْ دِ جࣴفࣹ لِ سࣹينْ يࣵلَّ اَجِ سَخْ جِ رَفࣹتْلُ؞
