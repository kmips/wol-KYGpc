بࣵنِّ إِسْرَاءِيلْ يࣺڪَّتِ نَ جَامࣷوڪَايْ بَ
بࣺسُبْ نࣷفْلاَيْ دُ بࣺسُ لِگّࣺييْ
35
‏1 مُوسَىٰ نَگْ دَجَلࣹ مبࣷولࣷومْ بࣵنِّ إِسْرَاءِيلْ مࣺݒّ؞ مُ وَخْ لࣹينْ نࣹ لࣹينْ:‏ «لِي مࣷويْ لِ اَجِ سَخْ جِ سَنْتَانࣹ، نگِرْ ݧُ دࣹفْ ڪࣷ:‏ ‏2 جُرٗومْ بࣹنِّ فَنْ لࣹيسْ دِ لِگّࣺييْ، وَايࣹ بࣺسُبْ جُرٗومْ ݧَارࣹيلْ بَ، سࣹينْ بࣺسُبْ نࣷفْلاَيْ بُ سࣹلّ بُ مَتْ سࣴڪّ لَ، ݧࣹيلْ اَجِ سَخْ جِ؞ ڪࣺݒّ ڪُ ݖِ لِگّࣺييْ لࣹنّ، دࣹيسْ ڪࣷيْ رࣹيْ؞ ‏3 بُلࣹينْ جَفَلْ سَخْ سَوَرَ فࣹنّ ݖِ سࣹينِ دࣴڪُّوَايْ، ݖِ بࣺسُبْ نࣷفْلاَيْ؞»
اَيْ سَرَخْ وَرْ نَ ݖِ جَامࣷوڪَايْ بِ
‏4 مُوسَىٰ دࣹلُّ وَخْ اَڪْ مبࣷولࣷومْ بࣵنِّ إِسْرَاءِيلْ مࣺݒّ نࣹ لࣹينْ:‏ «لِي مࣷويْ لِ اَجِ سَخْ جِ سَنْتَانࣹ؞
‏5 «سࣵڪّلࣹينْ ݖِ سࣹينْ اَلَلْ اَبْ جࣷوخࣹ بُ ݧࣹيلْ اَجِ سَخْ جِ؞ ڪࣺݒّ ڪُ اَمْ يࣺينࣹبْ خࣷلَمْ، نَ اِنْدِ جࣷوخࣹبْ اَجِ سَخْ جِ:‏ وُرُسْ اَڪْ خَالِسْ اَڪْ خࣵنجَرْ، ‏6 اَڪْ وࣴݧْ گُ بَخَ تࣹ لاَلْ يࣷلࣹتْ، اَڪْ وࣴݧْ گُ خࣹوَرْ، اَڪْ گُ خࣷنقّ ݖُرّ، اَڪْ وࣴݧُ لࣹي، اَڪْ ڪَوَرِ بࣴيْ ‏7 اَڪْ دࣹرِ ڪُويْ يُ ݧُ سُوبْ خࣷنقّ، اَڪْ دࣹرِ ݒِيݒِ، اَڪْ بَنْتِ سࣺنگّ، ‏8 اَڪْ دِوُ لࣵمݒّ، اَڪْ اَيْ خࣹيݧَلْ نگِرْ دِوُ ݒَلْ گࣹيڪْ ݖُورَايْ لُ خࣹيݧْ لِ، ‏9 اَڪْ اَيْ ݒࣹرِ اࣷنِگْسࣴ اَڪْ يࣹنࣹينْ ݒࣹرْ يُيْ تَݒّࣹ ݖِ خَرْ-سࣵنِّ مࣹيڪْ ڪِيرَايَلْ دࣴنّ بِ؞
‏10 «مبࣷولࣹمْ ڪُ خَرࣹݧْ ݖِ يࣹينْ، نَ ݧࣴوْ، نگࣹينْ دࣹفْ لِ اَجِ سَخْ جِ سَنْتَانࣹ لࣺݒّ:‏ ‏11 جَامࣷوڪَايْ بِ اَڪْ خَيْمَامْ اَڪْ ڪِيرَايَمْ اَڪِ ڪࣹݒّࣷومْ اَڪِ لࣵݖَّمْ، اَڪْ گَلَنْ يࣹيڪْ جࣴنْ يࣹيڪْ تࣹگُ يِ، ‏12 اَڪْ گَالْ گࣹيڪْ اَيْ نجࣵݒّࣷومْ اَڪْ ڪُبࣹيرْ گِ ݧُيْ اَمَلࣹ اَگْ نجࣷتْلاَيْ، اَڪْ رِدࣷ بِ ڪࣷ لࣵقْ، ‏13 اَڪْ تَابَلْ جࣹيڪِ نجࣵݒّࣷومْ اَڪْ ندَبَمْ يࣺݒّ اَڪْ مبُرُمْ تࣹيوَلْ مِ، ‏14 اَڪْ تࣹگُڪَايُ لࣵمݒّ بِ، اَڪِ جُمْتُڪَايَمْ اَڪِ لࣵمْݒَمْ اَڪْ دِوُ لࣵمݒّ بِ، ‏15 اَڪْ سَرْخَلُڪَايُ ݖُورَايْ بِ اَڪِ نجࣵݒّࣷومْ، اَڪْ دِوُ ݒَلْ گُ سࣹلّ گِ، اَڪْ ݖُورَايْ لُ خࣹيݧْ لِ؛ اَڪْ ڪِيرَايَلْ بُنْتُ جَامࣷوڪَايْ بِ، ‏16 اَڪْ سَرْخَلُڪَايُ رࣹنْدِ-دٗومَلْ بِ، اَڪْ ݖَاخُبْ خࣵنجَرَمْ اَڪِ نجࣵݒّࣷومْ اَڪْ لِ مُيْ اࣵنْدَلْ لࣺݒّ؛ مبَلْڪَمْ نجࣵݒّ مِ اَڪُبْ تࣹگࣷومْ، ‏17 اَڪْ سࣷرْ يِ يࣺيوْ جَامࣷوڪَايْ بِ اَڪِ جࣴنَمْ اَڪِ تࣹگࣷومْ، اَڪْ لَفُ بُنْتُ اࣴتّ بِ، ‏18 اَڪْ خࣹرِ جَامࣷوڪَايْ بِ، اَڪْ جࣴنْ يِ يࣺيوْ اࣴتّ بِ اَڪْ سࣹينِ بُومْ، ‏19 اَڪْ يࣺرࣹ يُ يࣵنجّ يِ ݧُيْ لِگّࣺييࣹ ݖِ بࣺرَبْ بُ سࣹلّ بِ، يࣺرࣹ يُ سࣹلّ يُ هَـٰرُونَ، سَرْخَلْڪَتْ بِ، اَڪْ يࣺرࣹيْ دࣷومَمْ، يِ ݧُيْ سَرْخَلࣹي؞»
‏20 ݖِ ڪَوْ لࣷولُ مبࣷولࣷومْ بࣵنِّ إِسْرَاءِيلْ گࣺݒّ بࣵيِّڪࣷو فَ مُوسَىٰ؞ ‏21 بَ مُ ڪࣷ دࣹفࣹي مبࣷولࣹمْ ݧَ ݖَ اَمْ ݒَسْتࣺيفُ خࣷلْ اَڪْ مبࣷولࣹمْ ݧَ ݖَ سࣹينْ خࣹلْ خِيرْ، ݧࣺݒّ دِڪّ، اِنْدِ جࣷوخࣹبْ اَجِ سَخْ جِ نگِرْ لِگّࣺييُ خَيْمَبْ ندَجࣹ مࣹيڪْ بࣺݒّ لِگّࣺييْ بُ مُ لاَجْ، نگِرْ اِتَمْ يࣺرࣹ يُ سࣹلّ يِ؞ ‏22 گٗورْ اَڪْ جِگࣺينْ بࣷڪّ دِڪّ، مبࣷولࣹمْ ݧِ اَمْ يࣺينࣹيْ خࣷلَمْ، مبࣷولࣹمْ ݧِ بࣴگَّ اِنْدِ وُرُسْ، دࣹفَلْ ڪࣷ اَجِ سَخْ جِ سَرَخُ يࣺڪَّتِ-جࣺبَّلࣹ؛ ݧِي اِنْدِ اَيْ لَمْ، ݧِي جَارࣷيْ نࣷݒّ، ݧِي جَارࣷيْ لࣷخࣷ، ݧِي اَيْ ݖَقْ، گࣵنجَرِ وُرُسْ يُ نࣹڪّ رࣹڪّ؞ ‏23 تࣹ اِتْ مبࣷولࣹمْ ݧِ دࣹنݖࣷونْ وࣴݧْ گُ بَخَ تࣹ لاَلْ يࣷلࣹتْ، اَڪْ وࣴݧْ گُ خࣹوَرْ، اَڪْ گُ خࣷنقّ ݖُرّ اَڪْ وࣴݧُ لࣹي، اَڪْ ڪَوَرِ بࣴيْ مبَا دࣹرِ خَرْ يُ ݧُ سُوبْ خࣷنقّ، مبَا دࣹرِ ݒِيݒِ، ݧࣷومْ ݧࣺݒَّ اِنْدِ لَ ݧُ دࣹنݖࣷونْ؞ ‏24 ڪࣺݒّ ڪُ نَمَّ جࣷوخࣹ خَالِسْ مبَا خࣵنجَرْ، جࣷوخࣹيلْ نگَ ڪࣷ اَجِ سَخْ جِ، تࣹ اِتَمْ ڪࣺݒّ ڪُ دࣹنݖࣷونْ بَنْتُ سࣺنگّ بُ جࣴمْ ݖِ مبࣷولࣹمْ لِگّࣺييُ سَرْخَلْ بِ، اِنْدِ نگَ بَنتّ بِ؞ ‏25 ݖِ بِيرْ لࣷولُ جِگࣺينْ ݧُ خَرࣹݧْ ݧࣺݒّ اࣴݖّࣹ سࣹينِ لࣷخࣷ، نگِرْ وَاجَلْ وࣴݧْ گُ بَخَ تࣹ لاَلْ يࣷلࣹتْ، اَڪْ وࣴݧْ گُ خࣹوَرْ، اَڪْ گُ خࣷنقّ ݖُرّ اَڪْ وࣴݧُ لࣹي، ݧُ بࣷولࣹ، اِنْدِ؞ ‏26 مبࣷولࣹمْ جِگࣺينْ ݧِ ڪࣷ مَنࣹ تࣹ اَمْ يࣺينࣹيمْ اِتَمْ، دَلْدِ اࣴݖّ ڪَوَرِ بࣴيْ؞ ‏27 نجِيتْ يِ نَگْ اِنْدِ اَيْ ݒࣹرِ اࣷنِگْسࣴ اَڪْ يࣹنࣹينْ ݒࣹرْ يِيْ تَݒّࣹ ݖِ خَرْ-سࣵنِّ مࣹيڪْ ڪِيرَايَلْ دࣴنّ بِ، ‏28 اَڪِ خࣹيݧَلْ، اَڪُگْ دِوْ نگِرْ لࣵمݒّ بࣹيڪْ دِوُ ݒَلْ گࣹيڪْ ݖُورَايْ لُ خࣹيݧْ لِ؞ ‏29 مبࣷولࣹمْ بࣵنِّ إِسْرَاءِيلْ، گٗورْ ݧࣹيڪْ جِگࣺينْ ݧِ اَمْ يࣺينࣹيْ اِنْدِ لُ جࣴمْ ݖِ لِگّࣺييْ بِ اَجِ سَخْ جِ سَنْتَانࣹ، مُوسَىٰ جࣷتّلِ، ݧࣷومْ ݧࣺݒَّ اِنْدِلْ اَجِ سَخْ جِ اَيْ سَرَخِ يࣺينࣹ؞
بࣹسَلࣹلْ اَڪْ اࣷلِيَبْ تࣵمْبَلِ نَݧُ لِگّࣺييْ بِ
‏30 مُوسَىٰ نَگْ نࣹ بࣵنِّ إِسْرَاءِيلْ:‏ «گِسْلࣹينْ، اَجِ سَخْ جِ مࣷو وࣷو بࣹسَلࣹلْ ݖِ تُرَمْ؞ بࣹسَلࣹلْ، اُرِ مࣷويْ بَايَمْ؛ اُرِ، اُرْ يُدࣹينْ بِ دِ بَايَمْ؞ ‏31 مࣷو ڪࣷ فࣹيسَلࣹ خࣹلُمْ يࣵلَّ اَڪْ خَرࣹݧْتࣹ اَڪْ دࣺگِّنْ اَڪْ خَمْ-خَمْ اَڪْ مَنࣷورࣹيْ بࣺݒّ خࣹيتُ لِگّࣺييْ، ‏32 نگِرْ مُيْ نَسْ اَيْ لِگّࣺييِ وُرُسْ اَڪْ خَالِسْ اَڪْ خࣵنجَرْ، ‏33 دِ يࣹتّ ݒࣹرْ يُ مُيْ تَݒّ، دِ يࣹتّ بَنتّ، دِ لِگّࣺييْ لُ نࣹڪّ؞ ‏34 مَيُ جࣵنگّلࣹ اِتَمْ يࣹبْ نَ ڪࣷ ݖِبْ دࣴنَّمْ، مࣷومْ اَڪْ اࣷلِيَبْ، دَنࣹينْ بِ، دࣷومُ اَيِسَمَاگْ؞ ‏35 مࣷو لࣹينْ فࣹيسَلْ خَرࣹݧْتࣹ، نگِرْ بࣺݒّ لِگّࣺييْ، ݖِ وࣵلُّ يࣹتّ اَڪْ نَتَالْ، اَڪْ لِگّࣺييُ وࣴݧْ گُ بَخَ تࣹ لاَلْ يࣷلࣹتْ، اَڪْ وࣴݧْ گُ خࣹوَرْ، اَڪْ گُ خࣷنقّ ݖُرّ اَڪْ وࣴݧُ لࣹي؞ مَنࣹ نَݧُ رࣵبّ تࣹ اِتَمْ دَݧُيْ سࣷسْ سࣹينْ تَارَلِ بࣷݒّ؞
