5
‏1 وَايࣹ نَگْ بࣷڪّ يِ، لُ جࣴمْ ݖِ اَيْ اࣵݒّ، اَڪْ بࣺسْ يَ لࣷولُ دِ اَمْ، سࣷخْلَوُلࣹينْ ڪُ لࣹينْ ݖِ بِندّ، ‏2 ندَخْ يࣹينْ ݖِ سࣹينْ بࣷݒّ، خَمْ نگࣹينْ خࣺلّ نࣹ سَنگّ بِ، نِ سࣵݖُّبْ گُدِّ لَيْ دِڪّࣹ؞ ‏3 فَ نِتْ ݧِ نَانْ:‏ «جࣵمّ اَڪْ ڪَارَانگࣹ دِڪّ نَ،» فَ لَ سࣵنڪُتࣹ دِ دَلࣹ ݖِ سࣹينْ ڪَوْ، نِ اَوْ مَتْ دِ جࣹڪّࣹي دَلْ ݖِ ڪَوْ جِگࣺينُ وࣺرُلْ، تࣹ ڪࣹنّ دُ ݖَ رࣴݖّࣹ؞ ‏4 وَايࣹ يࣹينْ بࣷڪّ يِ، نࣹڪُّلࣹينْ ݖِگْ لࣴنْدࣴمْ، بَ بࣺسْ بࣷوبُ دِ لࣹينْ بࣹتّ نِ اَبْ سࣵݖّ دِ بࣹتّࣹي؞ ‏5 يࣹينْ ݧࣺݒّ اَيْ دࣷومِ لࣹيرْ نگࣹينْ، دࣷومِ بࣴݖّࣴگْ ندَرَڪࣵمّ؛ نُنْ بࣷڪُّنُ دَرَ اَڪْ گُدِّ، مبَا اَگْ لࣴنْدࣴمْ؞ ‏6 ڪࣷنْ نَگْ بُنُ نࣹلَوْ نِ ݧِ ݖِ دࣹسْ، وَايࣹ نَنُ تࣹيوْلُ، تࣹ اࣵندّ اَڪْ سُنُ سَگࣷ؞ ‏7 ندَخْتࣹ ݧِيْ نࣹلَوْ، گُدِّ لَݧُيْ نࣹلَوْ؛ ݧِيْ مࣵنْدِ اِتْ، گُدِّ لَݧُيْ مࣵنْدِ؞ ‏8 وَايࣹ نُنْ ݧِيْ وَا بࣴݖّࣴگْ، نَنُ اࣵندّ اَڪْ سُنُ سَگࣷ، تࣹ دࣹفْ نگࣴمْ اَڪْ ݖࣷفࣹيلْ، سُنُ ڪِيرَايَلْ دࣴنّ؛ يَاڪَارُ تࣹخࣹ دِ سُنُ مبَخَنَمْ خَرࣹ؞ ‏9 ندَخْ ڪَتْ يࣵلَّ سࣺدُّ نُ اَمْ سࣵنجّ، وَايࣹ اَگْ مُݖّ لَ نُ مࣷومَلࣹ ندَخْ سُنُ سَنگّ عِيسَىٰ اَلْمَسِ؞ ‏10 سُ نُيْ دُندّ، اَڪْ سُ نُيْ دࣹي، مࣷومْ مࣷو نُ دࣹييَلْ، نگِرْ نُ مَنَ اࣵندّ اَڪْ مࣷومْ، دُندّ؞ ‏11 ڪࣷنْ نَگْ نࣷونُ نگࣹينْ دِ ݧَاخَنْتࣹي، اَڪَ يࣷڪَّنْتࣹ، نَنگࣹينْ ڪࣷ وࣺيࣹ؞
دࣺنڪَانࣹ يِي لَ ݒٗولْ تࣴجࣹ
‏12 گَنَّاوْ لࣷولُ بࣷڪّ يِ، دَنُ لࣹينْ دِ ݧَانْ، ݧِ فࣺيتࣹوࣷو ݖࣷونࣷبْ لِگّࣺييْ ݖِ سࣹينْ بِيرْ تࣹ جِيتࣹ لࣹينْ ݖِ يࣷونُ بࣷرࣷومْ بِ، دِ لࣹينْ يࣹي، نَنگࣹينْ لࣹينْ يࣺگْ؞ ‏13 نَنگࣹينْ لࣹينْ نَوࣹ سࣹينْ ڪࣹمْ ڪࣵتَّنْ تࣹ سࣷݒّࣹ لࣹينْ نࣷونُ ندَخْ سࣹينْ لِگّࣺييْ؞ گَنَّاوْ لࣷولُ، نَنگࣹينْ جࣵمّࣷو؞ ‏14 بࣷڪّ يِ، دَنُ لࣹينْ دِ دࣺنڪُ؛ يَافُسْ يِ، فࣹمُّلࣹينْ لࣹينْ؛ ݧِ نࣺيوُ فِتْ، ݧَاخْلࣹينْ لࣹينْ؛ نࣺيوْ-دࣷولࣹ يِ، دِمْبَلِلࣹينْ لࣹينْ، تࣹ اِتْ مُݧَلْ-لࣹينْ ݧࣺݒّ؞ ‏15 فࣹخࣹلࣹينْ بَ ڪࣹنّ دُ فࣹيَنْتࣷو؞ وَايࣹ لُ بَاخْ، دࣹيلࣹينْ ڪࣷ دࣹفْلَنْتࣹ ݖِ سࣹينْ بِيرْ، تࣹ دِ ڪࣷ دࣹفَلْ ݧࣺݒّ؞
‏16 فُ نگࣹينْ تࣷلُّ بࣺگْلࣹينْ، ‏17 سَخࣷولࣹينْ ݧَانْ، ‏18 تࣹ دࣹيلࣹينْ سَنتّ، اَڪْ نُ مُ مَنَ دࣹمࣹ، ندَخْ لࣷولُ مࣷويْ ݖࣷوبَرࣹيْ يࣵلَّ ݖِ يࣹينْ ݧِ اࣵندّ اَڪْ اَلْمَسِ عِيسَىٰ؞
‏19 بُلࣹينْ گِيمَلْ تَالُبْ نࣷووُگْ يࣵلَّ گِ، ‏20 بُلࣹينْ گࣵتَّݧْلُ اَيْ ڪࣵدُّيْ وَخْيُ؞ ‏21 وَايࣹ لُ مُ مَنَ دࣷونْ، سࣹتَّنْتَلْ-لࣹينْ ڪࣷ، لِ بَاخْ، نگࣹينْ جࣵݒّ ݖِ، ‏22 تࣹ لُ بࣷنْ، اَڪْ نُ مُ مَنَ بِنْدࣷو، دࣵندّلࣹينْ ڪࣷ؞
‏23 يࣵلَّ ݖِ بࣷݒَّمْ، مࣷومْ مِيْ بࣷرࣷومْ جࣵمّ، يَلْ نَ لࣹينْ سࣹلَّلْ، بَ نگࣹينْ مَتْ سࣴڪّ، تࣹ سࣹينْ لࣺݒِّ خࣹلْ، اَڪْ سࣹينْ لࣺݒِّ خࣷلْ، اَڪْ سࣹينْ لࣺݒِّ يَرَمْ سࣵمُّ، بَ مُݖّ سِڪّ، بَ ڪࣹرَ بࣺسْ بُ سُنُ سَنگّ عِيسَىٰ اَلْمَسِ فࣹيݧࣹي؞ ‏24 ڪِ لࣹينْ وࣷو دࣹ ڪُ وٗورْ لَ، تࣹ مࣷو ڪࣷيْ دࣹفْ اِتْ؞
ݒٗولْ تࣵگّتࣷو نَ اَڪْ وَا تࣹسَلࣷنِگْ
‏25 بࣷڪّ يِ، نُنْ اِتَمْ، ݧَانَلْ-لࣹينْ نُ؞
‏26 نُيُلْ-لࣹينْ نُ بࣷڪّ يࣺݒّ، تࣹ فٗونَلْ-لࣹينْ نُ لࣹينْ نَ مُ گࣴنَ سࣹلّࣹ؞ ‏27 مَا نگِ لࣹينْ دِ دࣺنڪُ نَگْ، نگِرْ بࣷرࣷومْ بِ، جࣵنگّلࣹينْ بِرَلْ بَتَاخَلْ بِي نگِرْ بࣷڪّ يࣺݒّ؞
‏28 يَلْ نَ سُنُ يِوُ بࣷرࣷومْ عِيسَىٰ اَلْمَسِ اࣵندّ اَڪْ يࣹينْ؞
