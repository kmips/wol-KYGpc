مࣹلْڪِسࣹدࣹگْ مࣷو سُتْ إِبْرَاهِيمَ اَڪْ هَـٰرُونَ
7
‏1 مࣹلْڪِسࣹدࣹگْ مࣷومُ دࣷونْ بُورُبْ سَلࣹمْ تࣹ نࣹڪّࣷونْ سَرْخَلْڪَتُبْ يࣵلَّ اَجِ ڪَوࣹ جِ، بَ إِبْرَاهِيمَ دࣺلْسࣹي جٗگࣹ خَرࣹيمْ بَاڪْ بُورْ يَ مُ دُمَ، ڪࣷوڪُ مࣷو ڪࣷ گَتَنْدُ وࣷونْ، ݧَانَلْ ڪࣷ¤؞ ‏2 مࣷومْ لَ إِبْرَاهِيمَ جࣷخْ فُڪّࣹيلُ ݖࣺرُبْ مبࣷولࣹمْ ݖࣴخࣴتࣷو مَ؞ نِ ݧُ جࣴڪَّ دࣺگّࣹ اَوْ تُرَمْ مࣹلْڪِسࣹدࣹگْ، ݖِ تࣹڪِّ، مࣷويْ «بُورْ بُ جُبْ»؞ بُ لࣷولُ وࣹيسࣹي اِتَمْ مࣷومْ مࣷويْ بُورُبْ سَلࣹمْ، مُيْ «بُورُ جࣵمّ»؞ ‏3 مࣹلْڪِسࣹدࣹگْ نَگْ دُ بَايْ، دُ ندࣹيْ، دُ اَسْڪَنُ مَامْ وࣹيسْ ڪࣷ خَمَلْ؞ دُ ندࣷورْتࣹلْ گَنࣹ اࣵدِّنَ، دُ نجࣹيخْتࣹلْ دُندّ وࣹيسْ ڪࣷ خَمَلْ؞ دࣹيسْ ڪࣷو دࣹفْ مُ نِرُ دࣷومُ يࣵلَّ، تࣹ سَخࣷو دِ اَبْ سَرْخَلْڪَتْ بَ فࣵوّ؞
‏4 سࣹيتْلࣹينْ نِ مُ رࣵݧِّڪࣷو، مࣷومْ لَ إِبْرَاهِيمَ مَامُ ݖࣷسَانْ مَ ݖِ بࣷݒَّمْ جࣺبَّلْ فُڪّࣹيلُ ݖࣺرُبْ ݖࣴخࣴتࣷو مَ!‏ ‏5 تࣹ سَرْخَلْڪَتْ يِ بࣷڪّ ݖِ گِيرُگْ لࣹوِ¤ نِ ڪࣷ يࣷونْ وِ تࣴرَلࣹ، ݧࣷومْ لَݧُ سَنتّ ݧُ نَنگُ ݖࣺرِ فُڪّࣹيلِ اَلَلِ اَسْڪَنْ وِيْ سࣹينِ بࣷڪّ، تࣹ سࣷقِڪࣷو ݧࣷومْ اِتَمْ ݖِ گࣹݧࣷگْ إِبْرَاهِيمَ؞ ‏6 وَايࣹ مࣹلْڪِسࣹدࣹگْ مِ ݧُ اَسْڪَنَلࣹوُلْ گِيرُگْ لࣹوِ، ݖِ إِبْرَاهِيمَ لَ نَنگࣷو ݖࣺرُبْ فُڪّࣹيلْ؞ تࣹ اِتْ مࣷو ݧَانَلْ بَرْڪࣹ إِبْرَاهِيمَ مِ جَگࣷو دِگࣹيْ يࣵلَّ؞ ‏7 تࣹ اَمُلْ بࣹنّ وࣹرَنْتࣹ نࣹ ڪِ فࣺيتࣹ سُوفْ، مࣷومْ لَ ڪِ فࣺيتࣹ ڪَوْ دِ ݧَانَلْ بَرْڪࣹ؞ ‏8 ڪࣷنْ ݖِ گِي وࣹتْ اَيْ نِتْ ݧُيْ دࣹي دࣷݝّ ݧࣷويْ نَنگُ اَيْ ݖࣺرِ فُڪّࣹيلْ، ݖَ گࣹي وࣹتْ نَگْ ڪُ ݧُ سࣹيدࣹيلْ نࣹ مُ نگِ دُندّ بَ تࣹيْ مࣷو جَگࣷو ݖࣺرُبْ فُڪّࣹيلْ؞ ‏9 لࣺيگِ نَگْ ݖِ وَخِنْ، لࣹوِ مِيْ نَنگُ ݖࣺرُبْ فُڪّࣹيلْ، مࣷو گࣺنّࣹ ݖࣺرُبْ فُڪّࣹيلْ بُ مُ فࣹيࣹ ݖِ لࣷخࣷلْ إِبْرَاهِيمَ؞ ‏10 ندَخْ بَ مࣹلْڪِسࣹدࣹگْ دِ گَتَنْدُ إِبْرَاهِيمَ، لࣹوِ مَ نگَ وࣷونْ ݖَ گࣹݧࣷگْ إِبْرَاهِيمَ مَامَمْ؞
مࣹلْڪِسࣹدࣹگْ مِسَالْ نَ اَلْمَسِ
‏11 ڪࣷنْ لِگّࣺييُ ݖَرْخَلْ گُ لࣹوࣹينْ، ݧِ يࣷرࣷونْ يࣷونُ مُوسَىٰ، وِ ݧُ دࣺنڪࣷونْ بࣵنِّ إِسْرَاءِيلْ، سُ دࣷونْ لِگّࣺييْ بُيْ تَخَ مَتْ سࣴڪّ، ندَخْ دࣹيسْ نَ سࣷخْلاَتِ بࣹنࣹينْ سَرْخَلْڪَتْ بُ جٗگْ تࣷݒّ دَيࣷبْ مࣹلْڪِسࣹدࣹگْ، تࣹ تࣷݒُّلْ دَيࣷبْ هَـٰرُونَ؟ ‏12 ݖَرْخَلْ گِ نَگْ سُ سࣷݒِّڪࣷو، فࣵوّ يࣷونْ وِ اِتَمْ سࣷݒِّڪُ، اࣵندّ اَڪْ مࣷومْ؞ ‏13 ڪِ ڪࣵدُّ يِي جࣴمْ ݖِ مࣷومْ، فَ مُ بࣷڪّ، گࣹنࣹينْ گِيرْ لَ، تࣹ ڪࣹنّ ݖِ گِيرَمْ مَسُ ݖࣹي لِگّࣺييࣹ ݖَ سَرْخَلُڪَايْ بَ؞ ‏14 لِ وࣺرْ تࣹ وٗورْ اِتْ مࣷو دِ سُنُ سَنگّ ݖِ يُدَ لَ سࣷقِڪࣷو، تࣹ گࣷوگُ گِيرْ، مُوسَىٰ وَخُ ݖِ لࣹنّ لُ جࣴمْ ݖِ سَرْخَلْڪَتْ يِ؞ ‏15 مبِرْ مِ گࣴنَ لࣹيرَتِ گَنَّاوْ بَ بࣹنࣹينْ سَرْخَلْڪَتْ فࣹيݧࣹي تࣹ مࣹلْ نِ مࣹلْڪِسࣹدࣹگْ؞ ‏16 تࣹ ڪࣷوڪُ، دُ سَنْتَانࣹبْ ندࣷنࣷلْ دࣹرࣹتْ بُ مُ وࣵݖّࣷولْ مࣷو ڪࣷ تَخَ دࣷونْ اَبْ سَرْخَلْڪَتْ، وَايࣹ مَنࣷورࣹيْ دُنْدَمْ گُ اَمُلْ اࣵݒّ مࣷو ڪࣷ دࣹفْ اَبْ سَرْخَلْڪَتْ؞ ‏17 ندَخْ ڪَتْ، گِي ڪࣵدُّ، سࣹيدࣹيلࣹيسْ نَ ڪࣷ ڪࣷ:‏
«يَا دِبْ سَرْخَلْڪَتْ بَ فࣵوّ،
تࣷݒّ دَيࣷبْ مࣹلْڪِسࣹدࣹگْ؞»
‏18 ڪࣷنْ ݖِ گࣹنّ وࣹتْ، سَنْتَانࣹ بَ وࣷونْ، مࣷومْ لَݧُ نࣹينَلْ، ندَخْ مَتَدِگْ سَنْتَانࣹ بَ، اَڪْ ݧࣵڪَّ جَرِݧࣹيمْ، ‏19 ندَخْ يࣷونُ مُوسَىٰ سࣷتَّلُلْ دَرَ بَ مُ مَتْ سࣴڪّ؞ ݖِ گࣹنࣹينْ وࣹتْ، يَاڪَارْ جُ گࣴنْ مࣷو نُيࣷو، دِ يࣷونْ وُ نُيْ جَارࣹ بَ اࣵگّ فَ يࣵلَّ؞
‏20 تࣹ اِتْ ݒَلُگْ سَرْخَلْڪَتْ بِي ݧࣵڪُّلَ اࣵندّ اَڪْ نگِݧْ، ندࣹ سَرْخَلْڪَتْ يࣹي، ݧࣷومْ لࣹيسْ دَانْ فَلْ ݒَلْ گُ اࣵنْدُلْ اَڪْ نگِݧْ؞ ‏21 وَايࣹ عِيسَىٰ مِي لَ اَگْ ݒَلَمْ اࣵندّ اَڪْ نگِݧْ لُ جِبࣹ ݖَ ڪَ ڪࣷ وَخْ نࣹ ڪࣷ:‏
«بࣷرࣷومْ بࣹي ڪࣷ گِݧْ
تࣹ دُ دࣹلُّ گَنَّاوْ:‏
‹يَا دِبْ سَرْخَلْڪَتْ بَ فࣵوّ؞›»
‏22 ݖِ ڪَوْ نگِݧْ لُ رࣺيْ لࣷولُ لَ عِيسَىٰ نࣹڪّࣹ ڪِ وࣺرَلࣹ جࣴمَّمْ، نࣹ ڪٗلّࣴرࣹ گُ گࣴنْ تَخَوْ نَ؞
‏23 ݧِ ݧُ مَسَ فَلْ سَرْخَلْڪَتْ نَگْ بَرࣹ نَݧُ، تࣹ تࣴلࣹي وࣺيࣹ فࣵوّ سࣹينْ سَسْ مࣷو تَخْ، گَنَّاوْ ݧࣷومْ ݧࣺݒّ اَيْ دࣹي؞ ‏24 وَايࣹ عِيسَىٰ مِ سَخْ بَ فࣵوّ، مࣷو سَسࣷو ݖَرْخَلْ گُ سَخْ دࣵڪّ؞ ‏25 لࣷولࣷو وَرَلْ مُ مَنَ مُسَلْ مُݖّ گُ مَتْ سࣴڪّ، ݧِيْ اَوࣹ فِ مࣷومْ، دِڪّ فَ يࣵلَّ، ندَخْ مࣷويْ دُندّ بَ فࣵوّ، نگِرْ دِ لࣹينْ تِنُلْ؞
‏26 بࣷوبُ سَرْخَلْڪَتْ بُ مَگْ لَنُ سࣷخْلَ؛ مࣷو سࣹلّ، مُݖّ سِڪّ، مُݖّ سࣷبࣹ، مࣷومْ لَݧُ تࣹقَلࣹ اَڪْ بࣵڪَّارْڪَتْ يِ، تࣹ مࣷومْ لَݧُ ڪَوࣹيلْ بَ فَ تِيمْ اَسَمَانْ؞ ‏27 بࣷڪُّلْ اَڪْ يࣹنࣹينْ سَرْخَلْڪَتْ يُ مَگْ يِ؛ سࣷخْلَوُلَ جِيتَلْ بࣺسْ بُ نࣹڪّ، سَرَخِ جُرْ نگِرْ بࣵڪَّارِ بࣷݒَّمْ، دࣷورْ ݖࣹي تࣹگْ اَيْ سَرَخْ نگِرْ بࣵڪَّارِ اَسْڪَنْ وِ؞ بࣹنّ يࣷونْ لَ عِيسَىٰ سَرْخَلْ، مُ دࣷيْ، تࣹ مࣷويْ بَ مُ سَرْخࣹي بࣷݒَّمْ؞ ‏28 ندَخْ ڪَتْ يࣷونُ مُوسَىٰ، اَيْ نِتْ اَڪْ سࣹينُگْ نࣺيوْ دࣷولࣹ لَ فَلْ اَيْ سَرْخَلْڪَتْ يُ مَگْ؞ وَايࣹ ڪࣵدُّگْ نگِݧْ لِ تࣷݒّ ݖِ يࣷونْ وِ، مࣷو فَلْ سَرْخَلْڪَتْ بُ مَگْ، دࣷومْ جِ ݧُ مَتَلْ سࣴڪّ بَ فࣵوّ؞
