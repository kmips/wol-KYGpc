لُ جࣴمْ ݖِ سَرَخْ سُ بࣵڪَّارْ سَبَبْ
4
‏1 اَجِ سَخْ جِ وَخْ نَ اِتَمْ مُوسَىٰ نࣹ ڪࣷ:‏ ‏2 «وَخَلْ بࣵنِّ إِسْرَاءِيلْ نࣹ لࣹينْ:‏ سُ فࣹڪّࣹي نࣹ نِتْ ڪِ دَفَ مࣷيْ لࣹنّ ݖِ سَنْتَانࣹيْ اَجِ سَخْ جِ، بَ دࣹفْ لࣹنّ لُ وَرُلْ تࣹ دُ تࣹيࣹيفَمْ، لِ ݖِ وَرْ لِي لَ؞
لِ وَرْ سَرْخَلْڪَتْ بُ مَگْ بُ بࣵڪَّارْ
‏3 «ندࣹگَمْ سَرْخَلْڪَتْ بِ ݧُ دِوْ، فَلْ ڪࣷ، مࣷو مࣷيْ، بَ يٗبّࣹ مبࣷولࣷو مِ تࣷوݧْ، نَ اِنْدِلْ اَجِ سَخْ جِ يࣴڪّ وُ ندَوْ وُ اَمُلْ سِڪّ، مُيْ سَرَخُ ݒٗوتُمْ بࣵڪَّارْ ندَخْ بࣵڪَّارْ بَ مُ دࣹفْ؞ ‏4 دَيْ يٗبُّ يࣴڪّ وِ ݖِ بُنتّ خَيْمَبْ ندَجࣹ مِ، ݖِ ڪَنَمْ اَجِ سَخْ جِ، دَلْدِ تࣹگْ لࣷخࣷومْ ݖِ بࣷݒُّ يࣴڪّ وِ، ݧُ رࣹنْدِ يࣴڪّ وِ ݖِ ڪَنَمْ اَجِ سَخْ جِ؞ ‏5 سَرْخَلْڪَتْ بِ ݧُ دِوْ، فَلْ ڪࣷ، نَ سࣵڪّ ݖِ دࣹرࣹتُ يࣴڪّ وِ، يٗبُّ ݖِ بِيرْ خَيْمَبْ ندَجࣹ مِ؞ ‏6 سُ ڪࣷ دࣹفࣹي سَرْخَلْڪَتْ بِ ݖَݒّ بَارَامَمْ ݖِ دࣹرࣹتْ جِ، وِسْ-وِسَلْ دࣹرࣹتْ جِ جُرٗومْ ݧَارِ يࣷونْ فِ ڪَنَمْ اَجِ سَخْ جِ، فَ جَنࣷوڪْ رِدࣷبْ بࣺرَبْ بُ سࣹلّ بَ؞ ‏7 نَ سَرْخَلْڪَتْ بِ سࣵڪّ ݖِ دࣹرࣹتْ جِ، تَقَلْ ݖِ بࣺجّࣺنِ سَرْخَلُڪَايُ ݖُورَايْ لُ خࣹيݧْ لَ ݖَ بِيرْ خَيْمَبْ ندَجࣹ مَ ݖَ ڪَنَمْ اَجِ سَخْ جِ؞ لِ دࣹسْ ݖِ دࣹرࣹتُ يࣴڪّ وِ، نَ ڪࣷ تُورْ ݖِ تَاتُ سَرْخَلُڪَايْ بِ ݧُيْ لَڪّ سَرَخُ رࣹنْدِ-دٗومَلْ، فࣷوفَ ݖَ بُنتّ خَيْمَبْ ندَجࣹ مَ؞ ‏8 نَ گࣺنّࣹ لࣺݒّ لُيْ نࣹبّࣷنْ ݖِ يࣴڪّ وِ ݧُيْ سَرْخَلْ مُيْ ݒٗوتُمْ بࣵڪَّارْ:‏ نࣹبّࣷنْ بِ سࣵنگّ يࣺرࣹيْ بِيرْ يِ، اَڪْ لࣺݒّ لُيْ نࣹبّࣷنْ تࣹ جࣵݒّ ݖِ يࣺرࣹيْ بِيرْ يِ، ‏9 اَڪْ ݧَارِ دࣴمْبࣺينْ يِ اࣵندّ اَڪْ سࣹينْ نࣹبّࣷنْ ݖَ ڪَوْ، جࣵݒّ ݖَ فࣵلَّرࣹ جَ، اَڪْ بࣵجّࣷ بِ ݖِ رࣹسْ وِ تࣹ مُ دِ ڪࣷ بࣷولࣹيڪْ دࣴمْبࣺينْ يِ، گࣺنّࣹ ڪࣷ؞ ‏10 مࣷويْ نِ ݧُ ڪࣷيْ گࣺنّࣹي رࣹڪّ ݖِ يࣴڪُّ سَرَخُ ݖَنتّ ݖِ بِيرْ جࣵمّ؞ سَرْخَلْڪَتْ بِ دَ ڪࣷيْ بࣷولࣹ لَڪّ ݖَ ڪَوْ سَرْخَلُڪَايُ رࣹنْدِ-دٗومَلْ بَ؞ ‏11 وَايࣹ دࣹرُ يࣴڪّ وِ اَڪْ يࣵݒَّمْ وࣺݒّ دࣹيسْ ڪࣷيْ بࣷولࣹيڪْ بࣷݒّ بِ اَڪْ يࣹيلْ يِ اَڪْ يࣺرࣹيْ بِيرْ يِ اَڪْ سࣺبّرِيتْ مِ؛ ‏12 مبࣷولࣹمْ لِ دࣹسْ ݖِ يࣴڪّ وِ دَالْ لَيْ گࣺنّࣹ دَلْ بِ، يٗبُّ ڪࣷ ݖِ بࣺرَبْ بُ مُݖّ سࣷبࣹ، ݖَ جَلُبْ دٗومْ بَ، دَلْدِ تَالْ مَتّ، لَڪّ ڪࣷ ݖَ، بَ مُ دِبْ دٗومْ؞ فࣷوفَ دٗومْ بَ جَلࣹ لࣹيسْ ڪࣷيْ لَڪّ؞
لُ جࣴمْ ݖِ سَرَخْ يِ وَرْ مبࣷولࣷو مِ
‏13 «ندࣹگَمْ مبࣷولࣷومْ إِسْرَاءِيلْ مࣺݒَّ مࣷيْ لࣹنّ ݖِ سَنْتَانࣹيْ اَجِ سَخْ جِ، بَ دࣹفْ لࣹنّ لُ وَرُلْ تࣹ دُ سࣹينْ تࣹيࣹيفْ، مُيْ سࣹينُگْ تࣷوݧْ تࣹ سࣹيتْلُوُݧُ ڪࣷ، ‏14 بُ مبࣷولࣷو مِ نࣹمِّڪࣷو مࣷيْ گِ رࣹڪّ، وَرْ نَݧࣷو گࣺنّࣹ ݖِ جُرْ گُ گُدّ گِ يࣴڪّ وُ ندَوْ وُيْ دࣷونْ سَرَخُ ݒٗوتُمْ بࣵڪَّارْ، يٗبُّ ڪࣷ ݖَ بُنتّ خَيْمَبْ ندَجࣹ مَ؞ ‏15 مَگِ مبࣷولࣷو مِ دَݧُيْ تࣹگْ سࣹينِ لࣷخࣷ ݖِ بࣷݒُّ يࣴڪّ وِ، فِ ڪَنَمْ اَجِ سَخْ جِ، بَ نࣷݒِّ ݧُ رࣹنْدِ يࣴڪّ وِ فِ ڪَنَمْ اَجِ سَخْ جِ؞ ‏16 گَنَّاوْ لࣷولُ نَ سَرْخَلْڪَتْ بِ ݧُ دِوْ، فَلْ ڪࣷ، سࣵڪّ ݖِ دࣹرࣹتُ يࣴڪّ وِ، يٗبُّ ݖِ بِيرْ خَيْمَبْ ندَجࣹ مِ؞ ‏17 نَ ݖَݒّ بَارَامَمْ ݖِ دࣹرࣹتْ جِ، وِسْ-وِسَلْ دࣹرࣹتْ جِ جُرٗومْ ݧَارِ يࣷونْ فِ ڪَنَمْ اَجِ سَخْ جِ، فࣷوفُ ݖِ ڪَنَمْ رِدࣷ بِ؞ ‏18 بُ نࣷݒّࣹي نَ سࣵڪّ ݖِ دࣹرࣹتْ جِ، تَقَلْ ݖِ بࣺجّࣺنِ سَرْخَلُڪَايْ بِ ݖِ ڪَنَمْ اَجِ سَخْ جِ، ݖِ بِيرْ خَيْمَبْ ندَجࣹ مِ؞ لِ دࣹسْ ݖِ دࣹرࣹتْ جِ، نَ ڪࣷ تُورْ ݖَ تَاتُ سَرْخَلُڪَايُ رࣹنْدِ-دٗومَلْ بَ، ݖَ بُنتّ خَيْمَبْ ندَجࣹ مَ؞ ‏19 نَ گࣺنّࣹ لࣺݒّ لِ ݖِيْ نࣹبّࣷنْ، لَڪّ ڪࣷ ݖِ ڪَوْ سَرْخَلُڪَايْ بِ؞ ‏20 نَ دࣹفْ اَڪْ يࣴڪّ وࣷووُ نَ مُيْ دࣹفْ اَڪْ يࣴڪُّ سَرَخُ ݒٗوتُمْ بࣵڪَّارْ رࣹڪّ؞ نࣷونُ لَيْ دࣹفْ اَڪْ مࣷومْ؞ نِ لَ ڪࣷ سَرْخَلْڪَتْ بِيْ دࣹفَلࣹ سࣹينْ نجࣷتْلاَيْ، ݧُ اَمْ نجࣺگَّلْ؞ ‏21 نَ گࣺنّࣹ يࣴڪّ وِ دَلْ بِ، تَالْ ڪࣷ، نَ مُيْ تَالࣹ يࣴڪّ وِ ݧُ جࣴڪَّ وَخْ؞ سَرَخُ ݒٗوتُمْ بࣵڪَّارُ مبࣷولࣷو مِ لَ؞
لِ وَرْ ڪِلِفَ گُ بࣵڪَّارْ
‏22 «سُ فࣹڪّࣹي نࣹ ڪِلِفَ مࣷو مࣷيْ لࣹنّ ݖِ سَنْتَانࣹيْ يࣵلاَّمْ اَجِ سَخْ جِ، بَ دࣹفْ لࣹنّ لُ وَرُلْ تࣹ دُ تࣹيࣹيفَمْ، مُ دِگْ تࣷوݧَمْ، ‏23 بُ نࣹمِّڪࣷو بࣵڪَّارْ بِ مُ دࣹفْ رࣹڪّ، نَ سَرْخࣹ سِڪّࣹتْ بُ اَمُلْ سِڪّ؞ ‏24 دَيْ تࣹگْ لࣷخࣷومْ ݖِ بࣷݒُّ سِڪّࣹتْ بِ، رࣹنْدِ ڪࣷ فَ ݧُيْ رࣹنْدِ سَرَخُ رࣹنْدِ-دٗومَلْ، فَ ڪَنَمْ اَجِ سَخْ جِ؞ سَرَخُ ݒٗوتُمْ بࣵڪَّارْ لَ؞ ‏25 نَ سَرْخَلْڪَتْ بِ ݖَݒّ بَارَامَمْ ݖِ دࣹرࣹتُ سَرَخْ سِيْ ݒٗوتُمْ بࣵڪَّارْ، تَقَلْ ڪࣷ ݖِ بࣺجّࣺنِ سَرْخَلُڪَايُ رࣹنْدِ-دٗومَلْ بِ، بَ نࣷݒِّ تُورْ دࣹرࣹتْ جِ ݖِ تَاتُ سَرْخَلُڪَايُ رࣹنْدِ-دٗومَلْ بِ؞ ‏26 نَ لَڪّ نࣹبّࣷنْ بِ يࣺݒّ ݖِ ڪَوْ سَرْخَلُڪَايْ بِ، دࣹفْ ڪࣷ نِ نࣹبّࣷنُ سَرَخُ ݖَنتّ ݖِ بِيرْ جࣵمّ؞ نࣷونُ لَ ڪࣷ سَرْخَلْڪَتْ بِيْ دࣹفَلࣹ نجࣷتْلاَيَلْ بࣵڪَّارَمْ، مُ اَمْ نجࣺگَّلْ؞
لِ وَرْ بَادࣷولࣷ بُ بࣵڪَّارْ
‏27 «سُ دࣹي ڪࣹنّ ݖِ بَادࣷولࣷ يࣹي مࣷيْ لࣹنّ ݖِ سَنْتَانࣹيْ اَجِ سَخْ جِ، بَ دࣹفْ لࣹنّ لُ وَرُلْ تࣹ دُ تࣹيࣹيفَمْ، مُ دِگْ تࣷوݧَمْ، ‏28 بُ نࣹمِّڪࣷو بࣵڪَّارْ بِ مُ دࣹفْ رࣹڪّ، نَ سَرْخࣹ اَوْ بࣴيْ وُ جِگࣺينْ وُ اَمُلْ سِڪّ، ندَخْ بࣵڪَّارْ بِ مُ دࣹفْ؞ ‏29 دَيْ تࣹگْ لࣷخࣷومْ ݖِ بࣷݒُّ گࣵتُّ سَرَخْ سࣷوسُ دِ ݒٗوتُمْ بࣵڪَّارْ، رࣹنْدِ ڪࣷ فَ ݧُيْ لَڪّ سَرَخُ رࣹنْدِ-دٗومَلْ؞ ‏30 نَ سَرْخَلْڪَتْ بِ ݖَݒّ بَارَامَمْ ݖِ دࣹرࣹتْ جِ، تَقَلْ ݖِ بࣺجّࣺنِ سَرْخَلُڪَايُ رࣹنْدِ-دٗومَلْ بِ؛ لِ دࣹسْ ݖِ دࣹرࣹتْ جِ دَ ڪࣷيْ تُورْ ݖِ تَاتُ سَرْخَلُڪَايْ بِ؞ ‏31 تࣹ اِتْ نَ گࣺنّࣹ نࣹبّࣷنْ بِ يࣺݒّ نِ مُيْ گࣺنّࣹي نࣹبّࣷنُ جُرْ گُ ݧُ دࣹفْ سَرَخُ ݖَنتّ ݖِ بِيرْ جࣵمّ؞ نَ ڪࣷ سَرْخَلْڪَتْ بِ لَڪَّلْ اَجِ سَخْ جِ ݖِ ڪَوْ سَرْخَلُڪَايْ بِ، مُيْ خࣹيݧْ خࣹتُگْ جࣵمّ؞ نࣷونُ لَ ڪࣷ سَرْخَلْڪَتْ بِيْ دࣹفَلࣹ نجࣷتْلاَيَلْ بࣵڪَّارَمْ، مُ اَمْ نجࣺگَّلْ؞
‏32 «سُ فࣹڪّࣹي نࣹ اَمْ خَرْ لَيْ سَرْخࣹ، مُيْ سَرَخُ ݒٗوتُمْ بࣵڪَّارَمْ، نَ دࣷونْ خَرْ مُ جِگࣺينْ، مُ اَمُلْ سِڪّ؞ ‏33 دَيْ تࣹگْ لࣷخࣷومْ ݖِ بࣷݒُّ خَرْ مࣷومُ دِ سَرَخُ ݒٗوتُمْ بࣵڪَّارْ، رࣹنْدِ ڪࣷ فَ ݧُيْ رࣹنْدࣹي سَرَخُ رࣹنْدِ-دٗومَلْ، مُيْ سَرَخُ ݒٗوتُمْ بࣵڪَّارْ؞ ‏34 نَ سَرْخَلْڪَتْ بِ ݖَݒّ بَارَامَمْ ݖِ دࣹرࣹتُ خَرْ مِيْ سَرَخُ ݒٗوتُمْ بࣵڪَّارْ، تࣹ تَقَلْ ݖِ بࣺجّࣺنِ سَرْخَلُڪَايُ رࣹنْدِ-دٗومَلْ بِ؛ لِ دࣹسْ ݖِ دࣹرࣹتْ جِ دَ ڪࣷيْ تُورْ ݖِ تَاتُ سَرْخَلُڪَايْ بِ؞ ‏35 تࣹ اِتَمْ نَ گࣺنّࣹ نࣹبّࣷنْ بِ يࣺݒّ، نِ مُيْ گࣺنّࣹي نࣹبّࣷنُ خَرْ مُ ݧُيْ دࣹفْ سَرَخُ ݖَنتّ ݖِ بِيرْ جࣵمّ؞ نَ سَرْخَلْڪَتْ بِ لَڪّ لࣷولُ ݖِ ڪَوْ سَرْخَلُڪَايْ بِ، مُ دࣷلِّڪُ ݖِ سَرَخُ سَوَرَ يِ ݧࣹيلْ اَجِ سَخْ جِ؞ نࣷونُ لَ سَرْخَلْڪَتْ بِيْ دࣹفَلࣹ نِتْ ڪِ نجࣷتْلاَيَلْ بࣵڪَّارْ بِ مُ دࣹفْ، مُ اَمْ نجࣺگَّلْ؞
