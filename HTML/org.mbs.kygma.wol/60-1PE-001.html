بَتَاخَلْ بُ جࣴڪّ بِ يࣵلَّ مَيْ
ݒِيࣹيرْ
مُ بِندّ ڪࣷ
1
‏1 ݖِ مَنْ ݒِيࣹيرْ، ندَوَلْ عِيسَىٰ اَلْمَسِ لَ بَتَاخَلْ بِي بَوࣷو، ݧࣹيلْ اَسْڪَنْ وِ يࣵلَّ تَامُ، تࣹ ݧُ تࣷخُجِ دِيوَانِ ݒࣷنتّ اَڪْ گَلَسِ اَڪْ ڪَݒَدࣷسْ اَڪْ اَسِ اَڪْ بِتِنِ¤؞ ‏2 يࣹينْ لَ يࣵلَّ بَايْ بِ رࣵݧّࣹي لُ جِيتُ، ݖِ نِ مُ لࣹينْ سࣹلَّلࣹ نࣷو گُ سࣹلّ گِ، نگِرْ نگࣹينْ مَنَ دࣺگَّلْ عِيسَىٰ اَلْمَسِ، بࣷولࣹ ݖِ تࣷويْلࣷو دࣹرࣹتَمْ، مُ سࣹتَلْ لࣹينْ؞ اَوْ يِوْ اَڪْ جࣵمّ جُ نࣹ خࣺيوْ ݧࣹيلْ نَ لࣹينْ؞
يَاڪَارْ جُيْ دُندّ لَنُ اَمْ ݖِ اَلْمَسِ
‏3 ݖَنتّ ݧࣹيلْ نَ يࣵلَّ، سُنُ بَايُ سَنگّ عِيسَىٰ اَلْمَسِ!‏ مࣷو نُ جُدُّوَاتَلْ ݖِ يࣴرْمَنْدࣹيمْ جُ يَا، ݖِ دࣷولࣹيْ ندࣹڪِّتࣹلْ عِيسَىٰ اَلْمَسِ، نگِرْ مَيْ نُ يَاڪَارْ جُيْ دُندّ؞ ‏4 لࣷولُ مࣷو اَمْ نگِرْ اَبْ ݖࣺرْ بُ دُلْ يࣵقُ، دُ سࣷبࣹوُ، دُ سࣹييْ، ݖࣺرْ بُ دࣹنݖࣹ فَ اَسَمَانْ ݧࣹيلْ لࣹينْ، ‏5 يࣹينْ ݧِ سࣹينْ نگࣴمْ تَخْ نگࣹينْ يِيرُ ݖِ مَنࣷورࣹيْ يࣵلَّ، مِ لࣹينْ دِ نࣺگࣹ مُݖّ گَ مُ وَاجَلْ تࣹ دِ ڪࣷ فࣹيݧَلْ ݖَ بࣺسْ بُ مُجّ بَ؞
‏6 لࣷولُ بࣺگࣹلࣹينْ ڪࣷو بࣺگࣹ، دࣷونْتࣹ تࣹيْ ݧࣵڪُّلْ نگࣹينْ نࣹڪّࣹ نجࣵقَرࣹ اَبْ دِيرْ، ندَخْ اَيْ نَتُّيْ نَتُّ؞ ‏7 يࣷويُ نَتُّ، نِ سَوَرَ دِ نَتّࣹ وُرُسْ، نِ لَيْ نَتّࣹ سࣹينْ دࣴگُّتࣹيْ نگࣴمْ، تࣹ نگࣴمَ گࣴنْ وُرُسْ فُوفْ، ندࣹ وُرُسْ دَيْ جࣹيخْ؞ سُ ڪࣷ دࣹفࣹي سࣹينْ نگࣴمْ گُ دࣴگُّ جُرَلْ لࣹينْ اَگْ ݖَنتّ، اَڪْ دَرَجَ اَڪْ تࣹرَانگَ، بࣺسْ بُ عِيسَىٰ اَلْمَسِ فࣹيݧࣹي؞ ‏8 مࣷومْ مࣷويْ ڪِ نگࣹينْ گِسُلْ تࣹ سࣷݒّ ڪࣷ؞ تࣹيْ فِ تࣹيْ گِسُلࣹينْ ڪࣷ، وَايࣹ گࣴمْ نگࣹينْ ڪࣷ، بَ تَخْ لࣹينَ بࣺگْ لࣷولْ، مبࣺگْ مُ مࣵگّ مُ لࣵمِّݧْ مَنُلَ تُدّ، ‏9 ندَخْ يࣹينَ نگِ جࣷتْ ݖِ يࣷولْ بِ نگࣴمْ دِ يٗوتُ، تࣹ مࣷويْ تࣹخࣹ؞
‏10 گࣷوگُ تࣹخࣹ، يࣷنࣹنتّ يَ جࣷتّلِ وࣷونْ وَخْيُبْ سࣹينْ يِوْ وِ لࣹينْ دِڪَّلْ، نِيرْ نَݧُ ڪࣷ، ݖࣵمْبَرْ ڪࣷ؞ ‏11 گࣴسْتُ نَݧُ جَنْ جَمࣷنࣷ، اَڪْ يَنْ خࣹوْ-خࣹوْ لَ نࣷووُگْ اَلْمَسِ گَ نࣹڪّࣷونْ ݖِ ݧࣷومْ، تࣹگْتَلࣹ وࣷونْ؞ مࣷويْ بَ نࣷو گَيْ سࣹيدࣹ لُ جِيتُ، ݖࣷونࣷيْ اَلْمَسِ، اَڪْ ندَمْ يَ ݖَيْ تࣷݒّ؞ ‏12 يࣷنࣹنتّ يࣷويُ، خَمَلࣹيسْ نَ لࣹينْ نࣹ دُ سࣹينْ نجَرِݧَلْ بࣷݒّ لَݧُيْ لِگّࣺييَلْ خِبَارْ بࣷوبُ، خَنَا يࣹينْ؞ ݧِ لࣹينْ اࣵگّࣹ خِبَارْ بِ تࣹيْ ݧࣷو دِ ݧِيْ وَارࣹ خِبَارُ جࣵمّ بِ ݖِ دࣷولࣹيْ نࣷو گُ سࣹلّ، گِ ݧُ يࣹبَلࣹ اَسَمَانْ، تࣹ مُيْ مبِرْ مُ مَلاَڪَ يِ سَخْ، نࣷونْ سِيوَ نࣹمِّڪُ؞
سࣹلَّلْ مࣷو وَرْ
‏13 ڪࣷنْ نَگْ تࣹيوْلࣷولࣹينْ سࣹينْ خࣹلْ، تࣹ اࣵندّ اَڪْ سࣹينْ سَگࣷ، تࣹ نگࣹينْ وࣺڪّ سࣹينْ جࣺݒّ يَاڪَارْ ݖِ يِوْ وَ ݧُ لࣹينْ دِ دِڪّࣹ ڪࣹرَ بُ عِيسَىٰ اَلْمَسِ فࣹيݧࣹي؞ ‏14 گَنَّاوْ گࣷنࣹ يُ دࣺگّ ندِگَلْ نگࣹينْ، بُلࣹينْ تࣷݒَّتِ سࣹينْ خࣹمّࣹمْتࣺيفِ بَڪَّنْ يَ وࣷونْ، بَ نگࣹينْ خَمُلࣹي؞ ‏15 وَايࣹ نِ ڪِ لࣹينْ وࣷو سࣹلّࣹ، يࣹينْ اِتَمْ نَنگࣹينْ سࣹلّࣹ نࣷونُ ݖِ سࣹينْ مبࣷولࣹمْ جࣴفِنْ؞ ‏16 ندَخْ ڪَتْ بِنْدࣹيسْ نَ نࣹ:‏ «سࣹلّ-لࣹينْ، ندَخْ مَنْ ڪُ سࣹلّ لاَ¤؞»
‏17 گَنَّاوْ نَگْ ڪِيْ اࣵتّࣹ نِتْ ڪُ نࣹڪّ ݖِ اَيْ جࣴفَمْ تࣹ اَمُ ݖِ خࣹجّ اَڪْ سࣹينْ، بَايْ نگࣹينْ ڪࣷيْ وَخْ، سࣵلّࣷولࣹينْ ڪࣷو رَگَلْ، سࣹينْ دِيرُبْ نگَنْ ݖِ ڪَوْ سُوفْ؞ ‏18 خَمْ نگࣹينْ نࣹ يࣴفْ يُيْ وࣺيْ يُ مࣹلْ نِ خَالِسْ مبَا وُرُسْ، دُ مࣷومْ لࣹيسْ لࣹينْ جࣷتࣹ ݖِ سࣹينْ دُنْدِنُ نࣹينْ وَ نگࣹينْ دࣷنّࣹ ݖَ سࣹينِ مَامْ، ‏19 وَايࣹ سࣹينُگْ نجࣷتْ مࣷو دِ دࣹرࣹتُ اَلْمَسِ جُ جَفࣹ، جِ تَخَوࣹ دࣹرࣹتُ خَرْ مُ سِڪُّلْ، گࣵڪُّلْ؞ ‏20 بُ يࣵگّ، بَ اࣵدِّنَ سࣷسࣷوگُلْ لَ يࣵلَّ تُدَّلࣷونْ اَلْمَسِ لࣷولُ، وَايࣹ لࣺيگِ ݖِ مُجُگْ جَمࣷنࣷ جِي لَ ڪࣷ يࣵلَّ فࣹيݧَلْ نگِرْ يࣹينْ، ‏21 ݧِ اَلْمَسِ تَخْ نگࣹينْ گࣴمْ يࣵلَّ؞ تࣹ يࣵلاَّ ڪࣷ دࣹڪَّلْ، مَيْ ڪࣷ ندَمْ، بَ تَخْ يࣵلَّ دِ ڪِ نگࣹينْ گࣴمْ، يَاڪَارْ ڪࣷ؞
‏22 ڪࣷنْ نَگْ گَنَّاوْ سࣹتَلْ نگࣹينْ سࣹينْ خࣷلْ، ندَخْ دࣴگّ گِ نگࣹينْ دࣺگَّلْ، بَ تَخْ نگࣹينْ اَمْ ݖࣷفࣹيلْ گُ مُݖّ جِنِگَلْ، ݖِ بࣷڪِّ گࣴمْڪَتْ ݧِ، نَنگࣹينْ سࣷݒَّنْتࣹي خࣷلْ بُ سࣹتْ، ݖࣷفࣹيلْ گُ مَتْ سࣴڪّ؞ ‏23 ندَخْ ڪَتْ دَنگࣹينَ جُدُّوَاتْ، تࣹ جِوُ وِ نگࣹينْ جُدُّوَاتࣹ مࣷويْ ڪࣵدُّگْ يࣵلَّ گِيْ دُندّ تࣹ سَخْ دࣵڪّ؛ دُ جِوُ وُيْ يࣵقُ، وَايࣹ مࣷويْ جِوُ وُ دُلْ يࣵقُ مُڪّ؞ ‏24 ندَخْ بِنْدࣹيسْ نَ نࣹ:‏
«بࣺݒّ بࣷرࣷومْ سُوخْ، نِمْ ݧَخْ لَ مࣹلْ،
مبࣷولࣹمْ دَرَجَامْ مࣹلْ نِ تٗورْ-تٗورُ اࣵلّ،
ݧَخْ مَ وࣷوْ، تٗورْ-تٗورْ مَ رُوسْ؞
‏25 وَايࣹ ڪࣵدُّگْ بࣷرࣷومْ بِ مࣷويْ سَخْ بَ فࣵوّ¤؞»
ڪࣵدُّ گࣷوگُ مࣷو دِ خِبَارُ جࣵمّ بِ ݧُ لࣹينْ اࣵگّࣹ؞
