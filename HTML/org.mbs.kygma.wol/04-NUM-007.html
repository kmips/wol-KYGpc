دَلࣷوبْ جَامࣷوڪَايْ بِ لاَجْ نَيْ سَرَخْ
7
‏1 ڪࣹرࣷوگْ بَ مُوسَىٰ سَمْݒࣹي جَامࣷوڪَايْ بِ بَ نࣷݒِّ، دِوْ نَ ڪࣷ نگِرْ سࣹلَّلْ ڪࣷ، مࣷوڪْ مبࣷولࣹمْ اَيْ جُمْتُڪَايَمْ، سَرْخَلُڪَايْ بࣹيڪْ مبࣷولࣹمْ ندَبَمْ اِتْ، مُ بࣷولࣹ دِوْ، سࣹلَّلࣹ ڪࣷ؞ ‏2 بَ لࣷولُ اَمࣹي ڪِلِفَيْ إِسْرَاءِيلْ يِ جِيتࣹ سࣹينِ ڪࣴرْ مَامْ دِڪّ؞ ݧࣷوݧࣷو دࣷونْ نجِيتِ گِيرْ يِ تࣹ ݧࣷو دࣷونْ سَيْتُ لِمࣹيفُ نِتْ ݧِ؞ ‏3 ݧُ اِنْدِلْ اَجِ سَخْ جِ سࣹينْ سَرَخْ، مُيْ جُرٗومْ بࣹنِّ وَتِيرْ اَڪْ فُڪِّ نَگْ اَڪْ ݧَارْ؛ ݧَارِ ڪِلِفَ يُ نࣹڪّ، بࣹنّ وَتِيرْ؛ ڪِلِفَ گُ نࣹڪّ، وࣹنّ نَگْ، ݧُ يٗبُّ لࣺݒّ فَ ڪَنَمْ جَامࣷوڪَايْ بَ؞ ‏4 اَجِ سَخْ جِ وَخْ مُوسَىٰ، ‏5 نࣹ ڪࣷ:‏ «نَنگُلْ لِي، مُ جࣴمْ ݖِ لِگّࣺييُ خَيْمَبْ ندَجࣹ مِ، تࣹ نگَ جࣷخْ ڪࣷ لࣹوࣹينْ ݧِ، ڪُ ݖِ نࣹڪّ اَڪْ لِ لِگّࣺييَمْ لاَجْ؞»
‏6 مُوسَىٰ نَنگُ وَتِيرْ يَاڪْ نَگْ يَ، جࣷخْ لࣹوࣹينْ ݧِ؞ ‏7 يِي ݧَارِ وَتِيرْ اَڪْ ݧࣹينْتِ نَگْ، مُ جࣷخْ گࣹرْسࣷنࣹينْ ݧَ، نَ ڪࣷ سࣹينْ لِگّࣺييْ لاَجࣹ، ‏8 يࣹي ݧࣹينتّ، مُ بࣷولࣹيڪْ جُرٗومْ ݧࣹتِّ نَگْ، جࣷخْ مࣹرَرࣹينْ ݧَ، نَ ڪࣷ سࣹينْ لِگّࣺييْ لاَجࣹ، ݖِ نجِيتَلْ سَرْخَلْڪَتْ بَ اِتَمَرْ دࣷومُ هَـٰرُونَ؞ ‏9 ڪࣹيَتࣹينْ ݧَ نَگْ، جࣷخُ لࣹينْ، ندَخْ يُ سࣹلّ يَ لَݧُ سَسࣷو وࣷونَ گࣵدُّ؞
‏10 ݖِ بِيرْ لࣷولُ ڪِلِفَ يِ اِنْدِ اِتَمْ سَرَخُ دَلࣷوبْ سَرْخَلُڪَايْ بِ، ڪࣹرࣷوگْ بࣺسْ بَ ݧُ ڪࣷيْ دِوْ؞ ݧُ دَلْدِ جࣷخࣹ سࣹينِ سَرَخْ فِ ڪَنَمْ سَرْخَلُڪَايْ بِ؞ ‏11 تࣹيوُلْ اَجِ سَخْ جِ وَخْ مُوسَىٰ، نࣹ ڪࣷ:‏ «ڪِلِفَ يِيْ جࣷخࣹسِ سࣹينْ سَرَخْ، يِ جࣴمْ ݖِ دَلࣷوبْ سَرْخَلُڪَايْ بِ، نَݧُ ڪࣷ دࣹفْ بࣺسْ بُ نࣹڪّ، ڪࣹنّ؞»
‏12 ڪِ جࣷخࣹ سَرَخَمْ ݖِ بࣺسْ بُ جࣴڪّ بِ، ݧࣹيلْ گِيرُگْ يُدَ، مࣷويْ نَسࣷنْ دࣷومُ اَمِنَدَبْ؞ ‏13 اَبْ سَرَخَمْ دِ بࣹنّ ندَبُ خَالِسْ بُ دِيسࣹ تࣺيمࣺيرِ سِيڪَلْ اَڪْ فَنْوࣹيرْ، مُيْ ڪِلࣷوڪْ لِيبَرْ، اَڪْ بࣹنّ ڪࣴلُّ خَالِسْ بُ دِيسࣹ جُرٗومْ ݧَارْ فُڪِّ سِيڪَلِ بࣺرَبْ بُ سࣹلّ بِ، مُيْ جُرٗومْ ݧࣹتِّ تࣺيمࣺيرِ گَرَامْ، لُ ݖِ نࣹڪّ فࣹيسْ اَڪْ سُنگُفْ سُ مُݖّ اَيِبْ سُ ݧُ خِيوَالࣹ اَگْ دِوْ، نگِرْ سَرَخُ ݒࣹݒّ بِ، ‏14 اَڪْ سࣹنّ مبࣵتُّسْ وُرُسْ سُ فࣹيسْ اَڪْ ݖُورَايْ تࣹ دِيسࣹ فُڪِّ سِيڪَلْ، دِ تࣺيمࣺيرِ گَرَامْ اَڪْ فُڪّ، ‏15 اَڪْ وࣹنّ يࣴڪّ وُ ندَوْ، اَڪْ مࣹنّ ڪُويْ، اَڪْ مࣹنّ خَرُمْ مࣹنّ اَتْ، نگِرْ سَرَخُ رࣹنْدِ-دٗومَلْ، ‏16 اَڪْ وࣹنّ سِڪّࣹتْ نگِرْ سَرَخُ ݒٗوتُمْ بࣵڪَّارْ، ‏17 اَڪْ سَرَخَسْ ݖَنتّ گِ ݖِ بِيرْ جࣵمّ، دِ ݧَارِ يࣴڪّ اَڪْ جُرٗومِ ڪُويْ اَڪْ جُرٗومِ سِڪّࣹتْ، اَڪْ جُرٗومِ خَرِ مࣹنّ اَتْ؞ لࣷولࣷو دࣷونْ سَرَخُ نَسࣷنْ دࣷومُ اَمِنَدَبْ؞
‏18 بࣺسُبْ ݧَارࣹيلْ بَ، ڪِلِفَگْ اِسَڪَرࣹينْ ݧَ، نࣹتَنࣹلْ دࣷومُ سُوَرْ مࣷو اِنْدِ سَرَخَمْ، ‏19 سَرَخْ بَ مُ اِنْدِ دِ بࣹنّ ندَبُ خَالِسْ بُ دِيسࣹ تࣺيمࣺيرِ سِيڪَلْ اَڪْ فَنْوࣹيرْ، دِ ڪِلࣷوڪْ لِيبَرْ، اَڪْ بࣹنّ ڪࣴلُّ خَالِسْ بُ دِيسࣹ جُرٗومْ ݧَارْ فُڪِّ سِيڪَلِ بࣺرَبْ بُ سࣹلّ بِ، مُيْ جُرٗومْ ݧࣹتِّ تࣺيمࣺيرِ گَرَامْ، لُ ݖِ نࣹڪّ فࣹيسْ اَڪْ سُنگُفْ سُ مُݖّ اَيِبْ سُ ݧُ خِيوَالࣹ اَگْ دِوْ، نگِرْ سَرَخُ ݒࣹݒّ، ‏20 اَڪْ سࣹنّ مبࣵتُّسْ وُرُسْ سُ فࣹيسْ اَڪْ ݖُورَايْ تࣹ دِيسࣹ فُڪِّ سِيڪَلْ، دِ تࣺيمࣺيرِ گَرَامْ اَڪْ فُڪّ، ‏21 اَڪْ وࣹنّ يࣴڪّ وُ ندَوْ، اَڪْ مࣹنّ ڪُويْ، اَڪْ مࣹنّ خَرُمْ مࣹنّ اَتْ، نگِرْ سَرَخُ رࣹنْدِ-دٗومَلْ، ‏22 اَڪْ وࣹنّ سِڪّࣹتْ نگِرْ سَرَخُ ݒٗوتُمْ بࣵڪَّارْ، ‏23 اَڪْ سَرَخَسْ ݖَنتّ گِ ݖِ بِيرْ جࣵمّ، دِ ݧَارِ يࣴڪّ اَڪْ جُرٗومِ ڪُويْ اَڪْ جُرٗومِ سِڪّࣹتْ، اَڪْ جُرٗومِ خَرِ مࣹنّ اَتْ؞ لࣷولࣷو دࣷونْ سَرَخُ نࣹتَنࣹلْ دࣷومُ سُوَرْ؞
‏24 بࣺسُبْ ݧࣹتّࣹيلْ بَ، ڪِلِفَگْ سَبُلࣷنࣹينْ ݧَ، اࣹلْيَبْ دࣷومُ اࣹلࣷنْ مࣷو اَيࣹ، ‏25 اَبْ سَرَخَمْ دِ بࣹنّ ندَبُ خَالِسْ بُ دِيسࣹ تࣺيمࣺيرِ سِيڪَلْ اَڪْ فَنْوࣹيرْ، مُيْ ڪِلࣷوڪْ لِيبَرْ، اَڪْ بࣹنّ ڪࣴلُّ خَالِسْ بُ دِيسࣹ جُرٗومْ ݧَارْ فُڪِّ سِيڪَلِ بࣺرَبْ بُ سࣹلّ بِ، دِ جُرٗومْ ݧࣹتِّ تࣺيمࣺيرِ گَرَامْ، لُ ݖِ نࣹڪّ فࣹيسْ اَڪْ سُنگُفْ سُ مُݖّ اَيِبْ سُ ݧُ خِيوَالࣹ اَگْ دِوْ، نگِرْ سَرَخُ ݒࣹݒّ، ‏26 اَڪْ سࣹنّ مبࣵتُّسْ وُرُسْ سُ فࣹيسْ اَڪْ ݖُورَايْ تࣹ دِيسࣹ فُڪِّ سِيڪَلْ، مُيْ تࣺيمࣺيرِ گَرَامْ اَڪْ فُڪّ، ‏27 اَڪْ وࣹنّ يࣴڪّ وُ ندَوْ، اَڪْ مࣹنّ ڪُويْ، اَڪْ مࣹنّ خَرُمْ مࣹنّ اَتْ، نگِرْ سَرَخُ رࣹنْدِ-دٗومَلْ، ‏28 اَڪْ وࣹنّ سِڪّࣹتْ نگِرْ سَرَخُ ݒٗوتُمْ بࣵڪَّارْ، ‏29 اَڪْ سَرَخَسْ ݖَنتّ ݖِ بِيرْ جࣵمّ، دِ ݧَارِ يࣴڪّ اَڪْ جُرٗومِ ڪُويْ اَڪْ جُرٗومِ سِڪّࣹتْ، اَڪْ جُرٗومِ خَرِ مࣹنّ اَتْ؞ لࣷولࣷو دࣷونْ سَرَخُ اࣹلْيَبْ دࣷومُ اࣹلࣷنْ؞
‏30 بࣺسُبْ ݧࣹينْتࣹيلْ بَ ڪِلِفَگْ رُبࣹنࣹينْ ݧَ، اࣹلِسُرْ دࣷومُ سࣹدࣹيُرْ مࣷو اَيࣹ، ‏31 اَبْ سَرَخَمْ دِ بࣹنّ ندَبُ خَالِسْ بُ دِيسࣹ تࣺيمࣺيرِ سِيڪَلْ اَڪْ فَنْوࣹيرْ، دِ ڪِلࣷوڪْ لِيبَرْ، اَڪْ بࣹنّ ڪࣴلُّ خَالِسْ بُ دِيسࣹ جُرٗومْ ݧَارْ فُڪِّ سِيڪَلِ بࣺرَبْ بُ سࣹلّ بِ، دِ جُرٗومْ ݧࣹتِّ تࣺيمࣺيرِ گَرَامْ، لُ ݖِ نࣹڪّ فࣹيسْ اَڪْ سُنگُفْ سُ مُݖّ اَيِبْ سُ ݧُ خِيوَالࣹ اَگْ دِوْ، نگِرْ سَرَخُ ݒࣹݒّ، ‏32 اَڪْ سࣹنّ مبࣵتُّسْ وُرُسْ سُ فࣹيسْ اَڪْ ݖُورَايْ تࣹ دِيسࣹ فُڪِّ سِيڪَلْ، مُيْ تࣺيمࣺيرِ گَرَامْ اَڪْ فُڪّ، ‏33 اَڪْ وࣹنّ يࣴڪّ وُ ندَوْ، اَڪْ مࣹنّ ڪُويْ، اَڪْ مࣹنّ خَرُمْ مࣹنّ اَتْ، نگِرْ سَرَخُ رࣹنْدِ-دٗومَلْ، ‏34 اَڪْ وࣹنّ سِڪّࣹتْ نگِرْ سَرَخُ ݒٗوتُمْ بࣵڪَّارْ، ‏35 اَڪْ سَرَخَسْ ݖَنتّ گِ ݖِ بِيرْ جࣵمّ، دِ ݧَارِ يࣴڪّ اَڪْ جُرٗومِ ڪُويْ اَڪْ جُرٗومِ سِڪّࣹتْ، اَڪْ جُرٗومِ خَرِ مࣹنّ اَتْ؞ لࣷولࣷو دࣷونْ سَرَخُ اࣹلِسُرْ دࣷومُ سࣹدࣹيُرْ؞
‏36 بࣺسُبْ جُرٗومࣹيلْ بَ ڪِلِفَگْ ݖِمْيࣷنࣹينْ ݧَ، سࣹلُمْيࣹلْ دࣷومُ سُرِسَدَايْ مࣷو اَيࣹ، ‏37 اَبْ سَرَخَمْ دِ بࣹنّ ندَبُ خَالِسْ بُ دِيسࣹ تࣺيمࣺيرِ سِيڪَلْ اَڪْ فَنْوࣹيرْ، دِ ڪِلࣷوڪْ لِيبَرْ، اَڪْ بࣹنّ ڪࣴلُّ خَالِسْ بُ دِيسࣹ جُرٗومْ ݧَارْ فُڪِّ سِيڪَلِ بࣺرَبْ بُ سࣹلّ بِ، مُيْ جُرٗومْ ݧࣹتِّ تࣺيمࣺيرِ گَرَامْ، لُ ݖِ نࣹڪّ فࣹيسْ اَڪْ سُنگُفْ سُ مُݖّ اَيِبْ سُ ݧُ خِيوَالࣹ اَگْ دِوْ، نگِرْ سَرَخُ ݒࣹݒّ، ‏38 اَڪْ سࣹنّ مبࣵتُّسْ وُرُسْ سُ فࣹيسْ اَڪْ ݖُورَايْ تࣹ دِيسࣹ فُڪِّ سِيڪَلْ، مُيْ تࣺيمࣺيرِ گَرَامْ اَڪْ فُڪّ، ‏39 اَڪْ وࣹنّ يࣴڪّ وُ ندَوْ، اَڪْ مࣹنّ ڪُويْ، اَڪْ مࣹنّ خَرُمْ مࣹنّ اَتْ، نگِرْ سَرَخُ رࣹنْدِ-دٗومَلْ، ‏40 اَڪْ وࣹنّ سِڪّࣹتْ نگِرْ سَرَخُ ݒٗوتُمْ بࣵڪَّارْ، ‏41 اَڪْ سَرَخَسْ ݖَنتّ گِ ݖِ بِيرْ جࣵمّ، دِ ݧَارِ يࣴڪّ اَڪْ جُرٗومِ ڪُويْ اَڪْ جُرٗومِ سِڪّࣹتْ، اَڪْ جُرٗومِ خَرِ مࣹنّ اَتْ؞ لࣷولࣷو دࣷونْ سَرَخُ سࣹلُمْيࣹلْ دࣷومُ سُرِسَدَايْ؞
‏42 بࣺسُبْ جُرٗومْ بࣹنّࣹيلْ بَ ڪِلِفَگْ گࣵدّࣹينْ ݧَ، اࣹلْيَسَفْ دࣷومُ دࣹوࣹلْ مࣷو اَيࣹ، ‏43 اَبْ سَرَخَمْ دِ بࣹنّ ندَبُ خَالِسْ بُ دِيسࣹ تࣺيمࣺيرِ سِيڪَلْ اَڪْ فَنْوࣹيرْ، مُيْ ڪِلࣷوڪْ لِيبَرْ، اَڪْ بࣹنّ ڪࣴلُّ خَالِسْ بُ دِيسࣹ جُرٗومْ ݧَارْ فُڪِّ سِيڪَلِ بࣺرَبْ بُ سࣹلّ بِ، مُيْ جُرٗومْ ݧࣹتِّ تࣺيمࣺيرِ گَرَامْ، لُ ݖِ نࣹڪّ فࣹيسْ اَڪْ سُنگُفْ سُ مُݖّ اَيِبْ سُ ݧُ خِيوَالࣹ اَگْ دِوْ، نگِرْ سَرَخُ ݒࣹݒّ، ‏44 اَڪْ سࣹنّ مبࣵتُّسْ وُرُسْ سُ فࣹيسْ اَڪْ ݖُورَايْ تࣹ دِيسࣹ فُڪِّ سِيڪَلْ، مُيْ تࣺيمࣺيرِ گَرَامْ اَڪْ فُڪّ، ‏45 اَڪْ وࣹنّ يࣴڪّ وُ ندَوْ، اَڪْ مࣹنّ ڪُويْ، اَڪْ مࣹنّ خَرُمْ مࣹنّ اَتْ، نگِرْ سَرَخُ رࣹنْدِ-دٗومَلْ، ‏46 اَڪْ وࣹنّ سِڪّࣹتْ نگِرْ سَرَخُ ݒٗوتُمْ بࣵڪَّارْ، ‏47 اَڪْ سَرَخَسْ ݖَنتّ گِ ݖِ بِيرْ جࣵمّ، دِ ݧَارِ يࣴڪّ اَڪْ جُرٗومِ ڪُويْ اَڪْ جُرٗومِ سِڪّࣹتْ، اَڪْ جُرٗومِ خَرِ مࣹنّ اَتْ؞ لࣷولࣷو دࣷونْ سَرَخُ اࣹلْيَسَفْ دࣷومُ دࣹوࣹلْ؞
‏48 بࣺسُبْ جُرٗومْ ݧَارࣹيلْ بَ ڪِلِفَگْ اࣹفْرَيْمࣹينْ ݧَ، اࣹلِسَمَ دࣷومُ اَمِيُتْ مࣷو اَيࣹ، ‏49 اَبْ سَرَخَمْ دِ بࣹنّ ندَبُ خَالِسْ بُ دِيسࣹ تࣺيمࣺيرِ سِيڪَلْ اَڪْ فَنْوࣹيرْ، مُيْ ڪِلࣷوڪْ لِيبَرْ، اَڪْ بࣹنّ ڪࣴلُّ خَالِسْ بُ دِيسࣹ جُرٗومْ ݧَارْ فُڪِّ سِيڪَلِ بࣺرَبْ بُ سࣹلّ بِ، مُيْ جُرٗومْ ݧࣹتِّ تࣺيمࣺيرِ گَرَامْ، لُ ݖِ نࣹڪّ فࣹيسْ اَڪْ سُنگُفْ سُ مُݖّ اَيِبْ سُ ݧُ خِيوَالࣹ اَگْ دِوْ، نگِرْ سَرَخُ ݒࣹݒّ، ‏50 اَڪْ سࣹنّ مبࣵتُّسْ وُرُسْ سُ فࣹيسْ اَڪْ ݖُورَايْ تࣹ دِيسࣹ فُڪِّ سِيڪَلْ، مُيْ تࣺيمࣺيرِ گَرَامْ اَڪْ فُڪّ، ‏51 اَڪْ وࣹنّ يࣴڪّ وُ ندَوْ، اَڪْ مࣹنّ ڪُويْ، اَڪْ مࣹنّ خَرُمْ مࣹنّ اَتْ، نگِرْ سَرَخُ رࣹنْدِ-دٗومَلْ، ‏52 اَڪْ وࣹنّ سِڪّࣹتْ نگِرْ سَرَخُ ݒٗوتُمْ بࣵڪَّارْ، ‏53 اَڪْ سَرَخَسْ ݖَنتّ گِ ݖِ بِيرْ جࣵمّ، دِ ݧَارِ يࣴڪّ اَڪْ جُرٗومِ ڪُويْ اَڪْ جُرٗومِ سِڪّࣹتْ، اَڪْ جُرٗومِ خَرِ مࣹنّ اَتْ؞ لࣷولࣷو دࣷونْ سَرَخُ اࣹلِسَمَ دࣷومُ اَمِيُتْ؞
‏54 بࣺسُبْ جُرٗومْ ݧࣹتّࣹيلْ بَ ڪِلِفَگْ مَنَسࣹينْ ݧَ، گَمَلْيࣹلْ دࣷومُ ݒࣹدَسُرْ مࣷو اَيࣹ، ‏55 اَبْ سَرَخَمْ دِ بࣹنّ ندَبُ خَالِسْ بُ دِيسࣹ تࣺيمࣺيرِ سِيڪَلْ اَڪْ فَنْوࣹيرْ، مُيْ ڪِلࣷوڪْ لِيبَرْ، اَڪْ بࣹنّ ڪࣴلُّ خَالِسْ بُ دِيسࣹ جُرٗومْ ݧَارْ فُڪِّ سِيڪَلِ بࣺرَبْ بُ سࣹلّ بِ، مُيْ جُرٗومْ ݧࣹتِّ تࣺيمࣺيرِ گَرَامْ، لُ ݖِ نࣹڪّ فࣹيسْ اَڪْ سُنگُفْ سُ مُݖّ اَيِبْ سُ ݧُ خِيوَالࣹ اَگْ دِوْ، نگِرْ سَرَخُ ݒࣹݒّ، ‏56 اَڪْ سࣹنّ مبࣵتُّسْ وُرُسْ سُ فࣹيسْ اَڪْ ݖُورَايْ تࣹ دِيسࣹ فُڪِّ سِيڪَلْ، دِ تࣺيمࣺيرِ گَرَامْ اَڪْ فُڪّ، ‏57 اَڪْ وࣹنّ يࣴڪّ وُ ندَوْ، اَڪْ مࣹنّ ڪُويْ، اَڪْ مࣹنّ خَرُمْ مࣹنّ اَتْ، نگِرْ سَرَخُ رࣹنْدِ-دٗومَلْ، ‏58 اَڪْ وࣹنّ سِڪّࣹتْ نگِرْ سَرَخُ ݒٗوتُمْ بࣵڪَّارْ، ‏59 اَڪْ سَرَخُ ݖَنتّ گِ ݖِ بِيرْ جࣵمّ، دِ ݧَارِ يࣴڪّ اَڪْ جُرٗومِ ڪُويْ اَڪْ جُرٗومِ سِڪّࣹتْ، اَڪْ جُرٗومِ خَرِ مࣹنّ اَتْ؞ لࣷولࣷو دࣷونْ سَرَخُ گَمَلْيࣹلْ دࣷومُ ݒࣹدَسُرْ؞
‏60 بࣺسُبْ جُرٗومْ ݧࣹينْتࣹيلْ بَ ڪِلِفَگْ بࣹݧَمِنࣹينْ ݧَ، اَبِدَنْ دࣷومُ گِدࣷنِ مࣷو اَيࣹ، ‏61 اَبْ سَرَخَمْ دِ بࣹنّ ندَبُ خَالِسْ بُ دِيسࣹ تࣺيمࣺيرِ سِيڪَلْ اَڪْ فَنْوࣹيرْ، مُيْ ڪِلࣷوڪْ لِيبَرْ، اَڪْ بࣹنّ ڪࣴلُّ خَالِسْ بُ دِيسࣹ جُرٗومْ ݧَارْ فُڪِّ سِيڪَلِ بࣺرَبْ بُ سࣹلّ بِ، مُيْ جُرٗومْ ݧࣹتِّ تࣺيمࣺيرِ گَرَامْ، لُ ݖِ نࣹڪّ فࣹيسْ اَڪْ سُنگُفْ سُ مُݖّ اَيِبْ سُ ݧُ خِيوَالࣹ اَگْ دِوْ، نگِرْ سَرَخُ ݒࣹݒّ، ‏62 اَڪْ سࣹنّ مبࣵتُّسْ وُرُسْ سُ فࣹيسْ اَڪْ ݖُورَايْ تࣹ دِيسࣹ فُڪِّ سِيڪَلْ، دِ تࣺيمࣺيرِ گَرَامْ اَڪْ فُڪّ، ‏63 اَڪْ وࣹنّ يࣴڪّ وُ ندَوْ، اَڪْ مࣹنّ ڪُويْ، اَڪْ مࣹنّ خَرُمْ مࣹنّ اَتْ، نگِرْ سَرَخُ رࣹنْدِ-دٗومَلْ، ‏64 اَڪْ وࣹنّ سِڪّࣹتْ نگِرْ سَرَخُ ݒٗوتُمْ بࣵڪَّارْ، ‏65 اَڪْ سَرَخُ ݖَنتّ گِ ݖِ بِيرْ جࣵمّ، دِ ݧَارِ يࣴڪّ اَڪْ جُرٗومِ ڪُويْ اَڪْ جُرٗومِ سِڪّࣹتْ، اَڪْ جُرٗومِ خَرِ مࣹنّ اَتْ؞ لࣷولࣷو دࣷونْ سَرَخُ اَبِدَنْ دࣷومُ گِدࣷنِ؞
‏66 بࣺسُبْ فُڪّࣹيلْ بَ ڪِلِفَگْ دَنࣹينْ ݧَ، اَيࣹسࣹرْ دࣷومُ اَمِسَدَايْ مࣷو اَيࣹ، ‏67 اَبْ سَرَخَمْ دِ بࣹنّ ندَبُ خَالِسْ بُ دِيسࣹ تࣺيمࣺيرِ سِيڪَلْ اَڪْ فَنْوࣹيرْ، مُيْ ڪِلࣷوڪْ لِيبَرْ، اَڪْ بࣹنّ ڪࣴلُّ خَالِسْ بُ دِيسࣹ جُرٗومْ ݧَارْ فُڪِّ سِيڪَلِ بࣺرَبْ بُ سࣹلّ بِ، مُيْ جُرٗومْ ݧࣹتِّ تࣺيمࣺيرِ گَرَامْ، لُ ݖِ نࣹڪّ فࣹيسْ اَڪْ سُنگُفْ سُ مُݖّ اَيِبْ سُ ݧُ خِيوَالࣹ اَگْ دِوْ، نگِرْ سَرَخُ ݒࣹݒّ، ‏68 اَڪْ سࣹنّ مبࣵتُّسْ وُرُسْ سُ فࣹيسْ اَڪْ ݖُورَايْ تࣹ دِيسࣹ فُڪِّ سِيڪَلْ، دِ تࣺيمࣺيرِ گَرَامْ اَڪْ فُڪّ، ‏69 اَڪْ وࣹنّ يࣴڪّ وُ ندَوْ، اَڪْ مࣹنّ ڪُويْ، اَڪْ مࣹنّ خَرُمْ مࣹنّ اَتْ، نگِرْ سَرَخُ رࣹنْدِ-دٗومَلْ، ‏70 اَڪْ وࣹنّ سِڪّࣹتْ نگِرْ سَرَخُ ݒٗوتُمْ بࣵڪَّارْ، ‏71 اَڪْ سَرَخُ ݖَنتّ گِ ݖِ بِيرْ جࣵمّ، دِ ݧَارِ يࣴڪّ اَڪْ جُرٗومِ ڪُويْ اَڪْ جُرٗومِ سِڪّࣹتْ، اَڪْ جُرٗومِ خَرِ مࣹنّ اَتْ؞ لࣷولࣷو دࣷونْ سَرَخُ اَيࣹسࣹرْ دࣷومُ اَمِسَدَايْ؞
‏72 بࣺسُبْ فُڪّࣹيلْ بَاڪْ بࣹنّ، ڪِلِفَگْ اَسࣹرࣹينْ ݧَ، ݒَگْيࣹلْ دࣷومُ اࣷڪّرَنْ مࣷو اَيࣹ، ‏73 اَبْ سَرَخَمْ دِ بࣹنّ ندَبُ خَالِسْ بُ دِيسࣹ تࣺيمࣺيرِ سِيڪَلْ اَڪْ فَنْوࣹيرْ، مُيْ ڪِلࣷوڪْ لِيبَرْ، اَڪْ بࣹنّ ڪࣴلُّ خَالِسْ بُ جُرٗومْ ݧَارْ فُڪِّ سِيڪَلِ بࣺرَبْ بُ سࣹلّ بِ، مُيْ جُرٗومْ ݧࣹتِّ تࣺيمࣺيرِ گَرَامْ، لُ ݖِ نࣹڪّ فࣹيسْ اَڪْ سُنگُفْ سُ مُݖّ اَيِبْ سُ ݧُ خِيوَالࣹ اَگْ دِوْ، نگِرْ سَرَخُ ݒࣹݒّ، ‏74 اَڪْ سࣹنّ مبࣵتُّسْ وُرُسْ سُ فࣹيسْ اَڪْ ݖُورَايْ تࣹ دِيسࣹ فُڪِّ سِيڪَلْ، دِ تࣺيمࣺيرِ گَرَامْ اَڪْ فُڪّ، ‏75 اَڪْ وࣹنّ يࣴڪّ وُ ندَوْ، اَڪْ مࣹنّ ڪُويْ، اَڪْ مࣹنّ خَرُمْ مࣹنّ اَتْ، نگِرْ سَرَخُ رࣹنْدِ-دٗومَلْ، ‏76 اَڪْ وࣹنّ سِڪّࣹتْ نگِرْ سَرَخُ ݒٗوتُمْ بࣵڪَّارْ، ‏77 اَڪْ سَرَخُ ݖَنتّ گِ ݖِ بِيرْ جࣵمّ، دِ ݧَارِ يࣴڪّ اَڪْ جُرٗومِ ڪُويْ اَڪْ جُرٗومِ سِڪّࣹتْ، اَڪْ جُرٗومِ خَرِ مࣹنّ اَتْ؞ لࣷولࣷو دࣷونْ سَرَخُ ݒَگْيࣹلْ دࣷومُ اࣷڪّرَنْ؞
‏78 بࣺسُبْ فُڪّࣹيلْ بَاڪْ ݧَارْ ڪِلِفَگْ نࣹفْتَلࣹينْ ݧَ، اَيْرَ دࣷومُ اࣹنَنْ مࣷو اَيࣹ، ‏79 اَبْ سَرَخَمْ دِ بࣹنّ ندَبُ خَالِسْ بُ دِيسࣹ تࣺيمࣺيرِ سِيڪَلْ اَڪْ فَنْوࣹيرْ، مُيْ ڪِلࣷوڪْ لِيبَرْ، اَڪْ بࣹنّ ڪࣴلُّ خَالِسْ بُ دِيسࣹ جُرٗومْ ݧَارْ فُڪِّ سِيڪَلِ بࣺرَبْ بُ سࣹلّ بِ، مُيْ جُرٗومْ ݧࣹتِّ تࣺيمࣺيرِ گَرَامْ، لُ ݖِ نࣹڪّ فࣹيسْ اَڪْ سُنگُفْ سُ مُݖّ اَيِبْ سُ ݧُ خِيوَالࣹ اَگْ دِوْ، نگِرْ سَرَخُ ݒࣹݒّ، ‏80 اَڪْ سࣹنّ مبࣵتُّسْ وُرُسْ سُ فࣹيسْ اَڪْ ݖُورَايْ تࣹ دِيسࣹ فُڪِّ سِيڪَلْ، دِ تࣺيمࣺيرِ گَرَامْ اَڪْ فُڪّ، ‏81 اَڪْ وࣹنّ يࣴڪّ وُ ندَوْ، اَڪْ مࣹنّ ڪُويْ، اَڪْ مࣹنّ خَرُمْ مࣹنّ اَتْ، نگِرْ سَرَخُ رࣹنْدِ-دٗومَلْ، ‏82 اَڪْ وࣹنّ سِڪّࣹتْ نگِرْ سَرَخُ ݒٗوتُمْ بࣵڪَّارْ، ‏83 اَڪْ سَرَخُ ݖَنتّ گِ ݖِ بِيرْ جࣵمّ، دِ ݧَارِ يࣴڪّ اَڪْ جُرٗومِ ڪُويْ اَڪْ جُرٗومِ سِڪّࣹتْ، اَڪْ جُرٗومِ خَرِ مࣹنّ اَتْ؞ لࣷولࣷو دࣷونْ سَرَخُ اَيْرَ دࣷومُ اࣹنَنْ؞
‏84 لࣷولࣷويْ سَرَخِ دَلࣷوبْ سَرْخَلُڪَايْ بَ، ݖَ بࣺسْ بَ ݧُ ڪࣷيْ دِوْ، سࣹلَّلْ ڪࣷ، لࣺݒّ بَوࣷو ݖَ ڪِلِفَيْ إِسْرَاءِيلْ؞ ندَبِ خَالِسْ يَ فُڪّ لاَڪْ ݧَارْ، ڪࣴلِّ خَالِسْ يَ دِ فُڪّ اَڪْ ݧَارْ، بࣵتُّيْ وُرُسْ يَ دِ فُڪّ اَڪْ ݧَارْ، ‏85 ندَبَلْ خَالِسْ لُ نࣹڪّ دِ تࣺيمࣺيرِ سِيڪَلْ اَڪْ فَنْوࣹيرْ، دِ ڪِلࣷوڪْ لِيبَرْ، ڪࣴلّ بُ نࣹڪّ دِ جُرٗومْ ݧَارْ فُڪِّ سِيڪَلْ، دِ جُرٗومْ ݧࣹتِّ تࣺيمࣺيرِ گَرَامْ، مبࣷولࣹمْ خَالِسُ جُمْتُڪَايْ يِ دِ ݧَارِ جُنِّيْ سِيڪَلِ بࣺرَبْ بُ سࣹلّ بِ اَڪْ ݧࣹينْتِ تࣺيمࣺيرْ (2400)، دِ ݧَارْ فُڪِّ ڪِلࣷوڪْ جُرٗومْ ݧَارْ اَڪْ جُرٗومِ تࣺيمࣺيرِ گَرَامْ (27500)؞ ‏86 فُڪِّ بࣵتُّيْ وُرُسْ يِ فࣹيسࣹ ݖُورَايْ، مبࣵتُّ سُ ݖِ نࣹڪّ فُڪِّ سِيڪَلِ بࣺرَبْ بُ سࣹلّ بِ لَ، دِ تࣺيمࣺيرِ گَرَامْ اَڪْ فُڪّ (110)، مبࣷولࣹمْ وُرُسُ بࣵتُّ يِ دِ تࣺيمࣺيرِ سِيڪَلْ اَڪْ ݧَارْ فُڪّ (120)، مُيْ ڪِلࣷ اَڪْ ݧࣹتِّ تࣺيمࣺيرِ گَرَامْ (1300)؞ ‏87 مبࣷولࣹمْ نَگِ سَرَخُ رࣹنْدِ-دٗومَلْ يِ، فُڪِّ يࣴڪّ يُ ندَوْ اَڪْ ݧَارْ لَ، ڪُويْ يِ دِ فُڪّ اَڪْ ݧَارْ، خَرِ مࣹنّ اَتْ يِ دِ فُڪّ اَڪْ ݧَارْ، اࣵندّ اَڪْ سࣹينْ سَرَخِ ݒࣹݒّ، سِڪّࣹتِ سَرَخُ ݒٗوتُمْ بࣵڪَّارْ بِ دِ فُڪّ اَڪْ ݧَارْ؞ ‏88 مبࣷولࣹمْ نَگِ سَرَخُ ݖَنْتُ بِيرْ جࣵمّ يِ، ݧَارْ فُڪِّ يࣴڪّ يُ ندَوْ اَڪْ ݧࣹينتّ لَ، ڪُويْ يِ دِ جُرٗومْ بࣹنّ فُڪّ، سِڪّࣹتْ يِ دِ جُرٗومْ بࣹنّ فُڪّ، خَرِ مࣹنّ اَتْ يِ دِ جُرٗومْ بࣹنّ فُڪّ؞ لࣷولࣷويْ سَرَخِ دَلࣷوبْ سَرْخَلُڪَايْ بَ، گَنَّاوْ بَ ݧُ ڪࣷ دِوࣹي، سࣹلَّلْ ڪࣷ؞
‏89 بُ مُوسَىٰ دُگَّانْ ݖَ بِيرْ خَيْمَبْ ندَجࣹ مَ، نگِرْ وَخْ اَڪْ اَجِ سَخْ جِ، دَيْ دࣺگّ بَاتْ بَ دِ وَخْ اَڪْ مࣷومْ، بَوࣷو فَ تِيمْ ڪُبࣹيرُ گَالُ سࣹيدࣹسْ ڪٗلّࣴرࣹ گَ، ݖَ دِگَّنْتࣹ ݧَارِ سࣹرُبْ يَ؞ نࣷونُ لَ دَانْ وَخࣹيڪْ مࣷومْ؞
