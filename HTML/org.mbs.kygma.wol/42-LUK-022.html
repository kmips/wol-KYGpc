نجِيتِ يࣷونْ يِ فࣹخࣹيلْ نَݧُ عِيسَىٰ
22
‏1 مࣵگَّلُ مبُرُ مُ اَمُلْ لَوِيرْ، تࣹ ݧُ دِ ڪࣷ وَخْ بࣺسُبْ مُݖّ، مࣷو جُبْسِ وࣷونْ؞ ‏2 ݖِ بِيرْ لࣷولُ سَرْخَلْڪَتْ يُ مَگْ يَ اَڪْ فِرِڪَتِ يࣷونْ يَ دِ سࣹيتْ نُ ݧُيْ فࣹخࣹي بَ رࣹيْ عِيسَىٰ، وَايࣹ دَݧࣷو رَگَلࣷونْ نِتْ ݧَ؞ ‏3 سࣹيْتَانࣹ نَگْ دُگّ يُدَ، مِ ݧُيْ وࣷويࣹ اِسْڪَرْيࣷ تࣹ مُ بࣷڪّࣷونْ ݖَ فُڪّ ݧَاڪْ ݧَارْ؞ ‏4 يُدَ دَلْدِ دِيسࣷوجِ اَڪْ سَرْخَلْڪَتْ يُ مَگْ يَ، اَڪْ ڪِلِفَيْ وَتُّڪَتِ ڪࣴرْ يࣵلَّ گَ، نگِرْ خَمْ نَنْ لَيْ تࣹگࣹ عِيسَىٰ ݖِ سࣹينِ لࣷخࣷ؞ ‏5 ݧُ اَمْ ݖَ مبࣺگْتࣹ، دِگࣷوڪْ مࣷومْ، نگِرْ جࣷخْ ڪࣷ خَالِسْ؞ ‏6 يُدَ نَنگُ، دِ وُتْ نَگْ يࣷونْ وُ مُ وࣷرࣹ عِيسَىٰ، فُ مبࣷولࣷو مَ تࣹيوُلْ؞
عِيسَىٰ تࣵگّࣷو نَ
‏7 ݖِ ڪَوْ لࣷولُ بࣺسُ مࣵگَّلُ مبُرُ مُ اَمُلْ لَوِيرْ اَگْسِ، ݧُ وَرْ ݖَا رࣹنْدِ گࣵتُّبْ بࣺسُبْ مُݖّ؞ ‏8 عِيسَىٰ يࣹبَلْ ݒِيࣹيرْ اَڪْ يࣷوَانْ نࣹ لࣹينْ:‏ «دࣹمْلࣹينْ، وَاجَلَلْ نُ بࣺسُبْ مُݖّ بِ، بَ نُ رࣹيرِ؞» ‏9 ݧُ نࣹ ڪࣷ:‏ «فࣷو بࣴگّ نُ وَاجࣹ ڪࣷ فَ؟» ‏10 مُ نࣹ لࣹينْ:‏ «بُ نگࣹينْ دِ دُگّ ݖَ دࣴڪّ بَ رࣹڪّ، دِنگࣹينْ دَجࣹيڪْ وَا جُ يࣹنُ نجَقَسْ ندࣷخْ؞ تࣷݒّلࣹينْ ݖِ مࣷومْ بَ ݖَ بِيرْ ڪࣴرْ گَ مُ جࣴمْ، ‏11 تࣹ نگࣹينْ نࣹ بࣷرࣷومْ ڪࣴرْ گَ:‏ ‹سࣴرِݧْ بِ نࣹي:‏ اَنَ نࣺيگُ گَنْ، بِ مَيْ رࣹيرࣹ رࣹيرُ بࣺسُبْ مُݖّ، مَاڪْ سَمَيْ تَالِبࣹ؟› ‏12 ڪࣷوڪُ مࣷو لࣹينْ دِ وࣷنْ نࣺيگُبْ ڪَوْ بُ يَا بُ ݧُ دࣹفَرْ بَ نࣷݒِّ؞ فࣷوفَ نگࣹينْ دِ وَاجَلࣹ؞»
‏13 بَ ݧُ دࣹمࣹي، نࣷونࣹي لࣹينْ ڪࣷ عِيسَىٰ وَخࣹ، نَ لَݧُ گِسࣹ لࣺݒّ؞ ݧُ دَلْدِ وَاجَلْ رࣹيرْ بَ؞
‏14 بَ وَخْتُوْ رࣹيرْ جࣷتࣹي، عِيسَىٰ تࣷوگْ اَڪْ ندَوْ يَ، ‏15 نࣹ لࣹينْ:‏ «يࣵڪَّمْتِ وࣷونْ نَا لࣷولْ بࣷڪّ اَڪْ يࣹينْ رࣹيرُ بࣺسُبْ مُݖّ بِي، بَلَ مَا تࣵبِّ ݖࣷونࣷ؞ ‏16 ندَخْ مَا لࣹينْ ڪࣷ وَخْ، دࣷوتُمَ ڪࣷ رࣹيرْ مُڪّ، بَ ڪࣹرࣷوگْ سࣷلࣷبْ رࣹيرْ بِ دِ سࣷتِّ ݖَ نگُورُگْ يࣵلَّ؞»
‏17 ݖِ ڪَوْ لࣷولُ مُ جࣴلْ اَبْ ڪَاسْ، يࣺڪَّتِ ڪࣵدُّگْ ݖَنتّ، دَلْدِ نࣹ:‏ «جࣴلْ-لࣹينْ ڪَاسْ بِي، تࣹ نگࣹينْ سࣺدّࣷو ڪࣷ؞ ‏18 ندَخْ مَا لࣹينْ ڪࣷ وَخْ، گَنَّاوْ-سِ-تࣹيْ دࣷوتُمَ نَانْ ندࣷخُمْ رࣹسࣹݧْ، بَ ڪࣹرَ نگُورُگْ يࣵلَّ دِڪّ؞»
‏19 گَنَّاوْ لࣷولُ مُ جࣴلْ مبُرُ، يࣺڪَّتِ ڪࣵدُّگْ ݖَنتّ، دَگّ ڪࣷ، جࣷخْ لࣹينْ ڪࣷ نࣹ:‏ «لِي مࣷو دِ سَمَ يَرَمْ وِ ݧُ جࣷوخࣹ نگِرْ يࣹينْ؞ دࣹيلࣹينْ دࣹفْ لِي، دِ مَ ڪࣷ فࣵتّلِڪࣷو؞»
‏20 ڪَاسْ بَ اِتَمْ نࣷونُ لَ دࣹفْ اَڪْ مࣷومْ گَنَّاوْ رࣹيرْ بَ، نࣹ لࣹينْ:‏ «ڪَاسْ بِي مࣷويْ ڪٗلّࣴرࣹ گُ يࣹيسْ گِ فَسࣷو ݖِ سَمَ دࣹرࣹتْ جِ لࣹينْ نَرَ تُورُلْ؞ ‏21 وَايࣹ نَگْ ڪِ مَيْ وࣷرْ، لࣷخࣷومَ نگِ ݖِ ندَبْ لِي نُ بࣷڪّ؞ ‏22 دࣷومُ نِتْ ڪِ دࣺيْ، نِ ݧُ ڪࣷ دࣷگَلࣹ، نࣷونُ لَيْ دࣹمࣹ، وَايࣹ وٗويْ نگَلَّ ڪِ ݒࣹخࣹ مِ ݧُ ڪࣷيْ وࣷرࣹ دِ جَارࣹ ݖِ مࣷومْ؞» ‏23 ݖِ ڪَوْ لࣷولُ تَالِبࣹ يِ دِ لاَجَنْتࣹ ݖِ سࣹينْ بِيرْ، ڪَنْ ݖِ ݧࣷومْ مࣷو نَرْ جࣷوجُ جࣴفْ؞
‏24 اَبْ وࣹرَنْتࣹ اِتَمْ دِڪَّلْ لࣹينْ، ݧُيْ خࣴݖّࣷو ݖِ سࣹينْ بِيرْ ڪُ وَرَ گࣴنَ ڪَوࣹ ݖِ ݧࣷومْ؞ ‏25 عِيسَىٰ نَگْ نࣹ لࣹينْ:‏ «بُورِ خࣹيتِ اࣵدِّنَ دࣷولࣹ لَݧُيْ نگُورࣷو ݖِ سࣹينْ ڪَوْ، تࣹيوُلْ ݧِ لࣹينْ دِ نࣺيوَلْ دࣷولࣹ، بَاخࣹڪَتِ نِتْ ݧِ لَݧُ لࣹينْ دِ وࣷويࣹ؞ ‏26 يࣹينْ نَگْ بُمُ دࣹمࣹ نࣷونُ ݖِ سࣹينْ بِيرْ؞ ڪِ گࣴنَ ڪَوࣹ ݖِ يࣹينْ، نَ مࣹلْ نِ مࣷو گࣴنْ دِ ندَوْ، تࣹ نجِيتْ لِ، نَ مࣹلْ نِ اَبْ سُرْگَ؞ ‏27 ڪِ ݧُ تَاجَلْ ندَبْ لِ، اَڪْ سُرْگَ بِ ڪࣷ تَاجَلْ، اَنَ ڪُ ݖِ گࣴنَ ڪَوࣹ؟ خَنَا دُ ڪِ ݧُ تَاجَلْ؟ مࣷونْتࣹ مَنْ، ݖِ سࣹينْ بِيرْ لاَ تࣷوگࣹ تࣷوگَايُ سُرْگَ!‏ ‏28 يࣹينْ نَگْ يࣹينَيْ ݧِ دࣹسْ اَڪْ مَنْ ݖِ سَمَ بِيرْ ݖࣷونࣷ؞ ‏29 مَنْ اِتَمْ نِ مَ سَمَ بَايْ دࣺنڪࣹ نگُورْ، نِ لاَ لࣹينْ ڪࣷيْ دࣺنڪࣹ؞ ‏30 يࣹينَيْ لࣹڪّ اَڪَ نَانࣹ ݖَ سَمَ ندَبْ، ݖَ سَمَ نگُورْ، تࣹ يࣹينَيْ تࣷوگࣹ اَيْ گࣵنگُنࣹ، دِ جِيتࣹ فُڪِّ گِيرِ بࣵنِّ إِسْرَاءِيلْ اَڪْ ݧَارْ؞»
‏31 عِيسَىٰ نࣹيتِ:‏ «سِمࣷݝْ، سِمࣷݝْ!‏ سࣹيْتَانࣹ ڪَتْ مُ نگِ سࣵڪُّ اَبْ سَݧْ-سَݧْ نگِرْ مَنْ لࣹينَ جࣺرِ نِ اَمْ ݒࣹݒّ؞ ‏32 وَايࣹ مَنْ تِنُلْ نَا لَ، نگِرْ سَ نگࣴمْ بَݧَ گِيمْ؞ يَوْ نَگْ بࣷو وَݧِّڪࣷو، نَنگَ دࣷولࣹيلْ سَ بࣷڪِّ گࣴمْڪَتْ؞» ‏33 ݒِيࣹيرْ نࣹ ڪࣷ:‏ «سَنگّ بِ، جࣹڪّ نَا ݖِ اࣵندّ اَڪْ يَوْ ڪَسࣷ، اَڪْ ݖِ اࣵندّ اَڪْ يَوْ دࣹي؞» ‏34 عِيسَىٰ نࣹ ڪࣷ:‏ «ݒِيࣹيرْ، مَا لَ ڪࣷ وَخْ، بَلَ گَنَارَ سَبْ تࣹيْ جِي، دِنگَ وࣹدِّ ݧࣹتِّ يࣷونْ نࣹ خَمْ نگَ مَ؞»
‏35 عِيسَىٰ نࣹيتِ لࣹينْ:‏ «بَ مَ لࣹينْ يࣹبَلࣹي تࣹ دُ دࣹنݖُڪَايُ خَالِسْ، دُ مبُوسْ، دُ ݖَرَخْ يُ نگࣹينْ يٗبَّالࣹ، ندَخْ ݧࣵڪّ نگࣹينْ دَرَ؟» ݧُ نࣹ ڪࣷ:‏ «دࣺيدࣺيتْ؞» ‏36 مُ نࣹ لࣹينْ:‏ «لࣺيگِ نَگْ ڪࣺݒّ ڪُ اَمْ دࣹنݖُڪَايُ خَالِسْ، نَ ڪࣷ جࣴلْ، ڪُ اَمْ مبُوسْ اِتْ، نَ ڪࣷ جࣴلْ؛ ڪُ اَمُلْ سَامَرْ، نَ جَايْ مبُبَّمْ، جࣴنْدࣹ سَامَرْ؞ ‏37 ندَخْ مَا لࣹينْ ڪࣷ وَخْ، لِ ݧُ بِندّ نࣹ، ‹مࣷوڪْ دࣹفْڪَتِ مبࣷنْ يِ لَݧُ بࣷولࣹ اَبْ اࣵتّࣹ،› فࣵوّ مُ اَمْ ݖِ مَنْ، تࣹ لࣷولُ جࣴمْ ݖِ سَمَ مبِرْ مُ نگِ سࣷتِّسِ؞» ‏38 ݧُ نࣹ ڪࣷ:‏ «سَنگّ بِ، ݧَارِ سَامَرَ نگِ فِ؞» مُ نࣹ لࣹينْ:‏ «دࣷيْ نَ؞»
عِيسَىٰ ݧَانْ نَ ݖَ تُنْدُ اࣷلِوْ يَ
‏39 ݖِ ڪَوْ لࣷولُ عِيسَىٰ گࣺنّ، جࣴمْ ݖَ تُنْدُ اࣷلِوْ يَ، نَ مُ ڪࣷ بَاخࣷو وࣷونْ، تَالِبࣹ يَ تࣷݒّ ݖِ مࣷومْ؞ ‏40 بَ مُ اَگْسࣹي ݖَ بࣺرَبْ بَ، دَ نࣹ لࣹينْ:‏ «ݧَانْلࣹينْ، نگِرْ بَݧَ تࣵبِّ ݖِگْ فِيرْ؞» ‏41 بَ لࣷولُ اَمࣹي مُ دࣵندّ لࣹينْ لُ تࣷلّࣷوڪْ سَانُ خࣹيرْ، دَلْدِ سُڪّ، دِ ݧَانْ؞ ‏42 مُ نࣹ:‏ «بَايْ، سُ دࣹي سَ ݖࣷوبَرࣹ، تࣹگِّلْ مَ ڪَاسُ نَقَرْ بِي؞ بُمُيْ سَمَ ݖࣷوبَرࣹ نَگْ، وَايࣹ يَوْ، نَ سَ ݖࣷوبَرࣹ اَمْ؞» ‏43 مَلاَڪَ مُ بَوࣷو اَسَمَانْ دَلْدِ ڪࣷيْ فࣹيݧُ، دࣷولࣹيلْ ڪࣷ؞ ‏44 نجࣵقَرࣹ نَگْ جࣵݒّ عِيسَىٰ، مُ گࣴنَتࣹي تࣴنڪُ ݖِ ڪࣵدُّيْ ݧَانَمْ، بَ اَوْ ݧَقَمْ مُجّ مࣹلْ نِ لُمْبِ دࣹرࣹتْ، دِ دَلْ ݖِ سُوفْ؞
‏45 مُ ݧَانْ بَ نࣷݒِّ، دَلْدِ جٗگْ، دࣺلْسِ ݖِ تَالِبࣹ يِ، فࣹڪّ لࣹينْ ݧُيْ نࣹلَوْ، سࣷنّ لࣷولْ ندَخْ اَوْ تِيسْ؞ ‏46 مُ نࣹ لࣹينْ:‏ «لُ نگࣹينْ دِ نࣹلَوْ؟ جٗگْلࣹينْ ݧَانْ، نگِرْ بَݧَ تࣵبِّ ݖِگْ فِيرْ؞»
جࣵݒّ نَݧُ عِيسَىٰ
‏47 نَڪَ لَ عِيسَىٰ دِ وَخْ، دَانࣹيلَگُلْ، مبࣷولࣷو نࣹ جِمࣹيتْ، ڪَ ݧُيْ وَخْ يُدَ، دِ ڪࣹنّ ݖَ فُڪّ ݧَاڪْ ݧَارْ جِيتࣹ لࣹينْ؞ يُدَ دِڪّ بَ ݖِ عِيسَىٰ، بࣴگّ ڪࣷو فٗونْ؞ ‏48 عِيسَىٰ نࣹ ڪࣷ:‏ «يُدَ، فٗونْ نگَيْ وࣷرࣹ دࣷومُ نِتْ ڪِ!‏» ‏49 ݧَ اࣵندّ اَڪْ عِيسَىٰ نَگْ گِسْ لَ ݖَ نَرَ تࣷݒّ، ݧُ نࣹ:‏ «سَنگّ بِ، نُ جَمࣹ سَامَرْ بࣷوگْ؟» ‏50 ڪࣹنّ ݖِ ݧࣷومْ دَلْدِ ݖَوْ سُرْگَبْ سَرْخَلْڪَتْ بُ مَگْ بَ، نجࣷوفْ نࣷݒُّ ندِجࣷورَمْ؞ ‏51 تࣹيوُلْ عِيسَىٰ نࣹ:‏ «بࣵيِّلࣹينْ، نَ فِ يࣹمْ؞» مُ لاَلْ نࣷݒُّ وَا جَ، وࣺرَلْ ڪࣷ؞
‏52 بَ لࣷولُ اَمࣹي عِيسَىٰ وَخْ اَڪْ ݧَ ڪࣷ جࣵݒّسِ:‏ سَرْخَلْڪَتْ يُ مَگْ يَ، اَڪْ ڪِلِفَيْ وَتُّڪَتِ ڪࣴرْ يࣵلَّ گَ، اَڪْ مَگْ ݧَ؞ مُ نࣹ لࣹينْ:‏ «مبࣹتࣹ دَمَ دِ اَبْ فِݒُّڪَتْ، بَ نگࣹينْ دِڪّࣹيْ سَامَرْ اَڪِ نجُرْ؞ ‏53 مࣷونَ بࣺسْ بُ نࣹڪّ، مَاڪْ يࣹينْ لَ وࣷونْ ݖَ ڪࣴرْ يࣵلَّ گَ، تࣹ تࣹگُلࣹينْ مَ لࣷخࣷ؞ وَايࣹ سࣹينْ وَخْتُ لَ لࣺيگِ، بِ ڪِلِفْتࣹيفُگْ لࣴنْدࣴمْ اَيࣹي نِي؞»
ݒِيࣹيرْ جࣵمْبُ نَ
‏54 بَ مُ ڪࣷ دࣹفࣹي ݧُ جࣵݒّ عِيسَىٰ، يٗبُّ ڪࣷ بَ ݖَ ڪࣴرْ سَرْخَلْڪَتْ بُ مَگْ بَ، ݒِيࣹيرْ دࣵندّ فُ سࣷرࣹ، تࣷݒّ ݖَ؞ ‏55 فࣹڪّ نَ ݧُ تَالْ سَوَرَ ݖَ دِگّ اࣴتّ بَ، تࣷوگْ يࣺيوْ سَوَرَ وَ، ݒِيࣹيرْ دِڪّ تࣷوگْ ݖَ سࣹينْ بِيرْ؞ ‏56 اَبْ سُرْگَ بُ جِگࣺينْ نَگْ گِسْ ڪࣷ، مُ تࣷوگْ فِ ڪَنَمْ تَالْ بِ، مُ نࣹ ڪࣷ جࣵڪّ، نࣹ:‏ «ڪِي اِتْ ݖِ لَ، مࣷومْ لَ اࣵنْدَلࣷونْ!‏» ‏57 ݒِيࣹيرْ نَگْ مِيمْ، نࣹ:‏ «سࣷخْنَ سِ، خَوْمَ ڪࣷوڪُ!‏» ‏58 نࣹسْ تُوتْ ڪࣹنࣹينْ گِسْ ڪࣷ نࣹ:‏ «يَوْ اِتْ ݖِ ݧࣷومْ نگَ!‏» ݒِيࣹيرْ نࣹ ڪࣷ:‏ «وَا جِي، دُ مَنْ!‏»
‏59 بَ ݧُ ݖَ تࣹگࣹي لُ مَتْ وَخْتُ، ڪࣹنࣹينْ جَارَاتْ ݖَ وَخْ جَ نࣹ:‏ «ڪِي ڪَيْ وَخْ-دࣴگّ، مࣷومْ لَ اࣵنْدَلࣷونْ، ندࣹ وَا گَلِلࣹ لَ؞» ‏60 تࣹيوُلْ ݒِيࣹيرْ نࣹ ڪࣷ:‏ «وَا جِي، لِ نگَيْ وَخْ، خَوْمَ مَ ڪࣷ دࣹ!‏» ݖَ سَا سَ، گَنَارْ سَبْ، تࣹ ڪࣵدُّگْ ݒِيࣹيرْ دَانࣷوگُلْ سَخْ؞ ‏61 ݖِ ڪَوْ لࣷولُ سَنگّ بِ گࣹيسُ، نࣹ جࣵڪّ ݒِيࣹيرْ، ݒِيࣹيرْ فࣵتّلِڪُ لَ ڪࣷ سَنگّ بِ وَخࣷونْ نࣹ ڪࣷ:‏ «بَلَ گَنَارَ سَبْ تࣹيْ جِي، دِنَ فࣹڪّ نگَ جࣵمْبُ مَ ݧࣹتِّ يࣷونْ؞» ‏62 ݒِيࣹيرْ نَگْ گࣺنّ، دِ جࣷويْ جࣷويْ يُ مࣹتِّ؞
مِتِتَلْ نَݧُ عِيسَىٰ
‏63 ݖِ ڪَوْ لࣷولُ ݧَ جࣵݒّ عِيسَىٰ، دِ ڪࣷ ڪٗڪَّلِ اَڪَ دٗورْ؞ ‏64 ݧُ مُورْ ڪَنَمْ گِ، نَانْ ڪࣷ:‏ «يَوْ، وَخَلْ نُ ڪࣵدُّگْ وَخْيُ بࣷوگْ، اَنَ ڪُ لَ دٗورْ؟» ‏65 يࣹنࣹينْ سَاگَ يُ بَرࣹ تِفَارْ نَݧُ ڪࣷ ڪࣷ؞
عِيسَىٰ جَنࣷو نَاڪْ ݒِلاَتْ اَڪْ اࣹرࣷدّ
‏66 بَ بࣴتْ سࣹتࣹي، مبࣷولࣷومْ مَگِ اَسْڪَنُ يَوُتْ، اَڪْ سَرْخَلْڪَتْ يُ مَگْ يَ، اَڪْ فِرِڪَتِ يࣷونْ يَ، ݧࣺݒّ دَجࣹ؞ ݧُ يٗبُّ عِيسَىٰ ݖَ سࣹينْ ڪُرࣺلُ اࣵتّࣹ بَ؞ ‏67 ݧُ نࣹ ڪࣷ:‏ «سࣷو دࣹي اَلْمَسِ، وَخْ نُ ڪࣷ؞» مُ نࣹ لࣹينْ:‏ «سُ مَ لࣹينْ ڪࣷ وَخࣹي، دُنگࣹينْ مَ گࣴمْ؞ ‏68 تࣹ سُ مَ لࣹينْ لاَجࣹي، دُنگࣹينْ تࣷنْتُ؞ ‏69 وَايࣹ گَنَّاوْ-سِ-تࣹيْ، دࣷومُ نِتْ ڪِ مࣷويْ تࣷوگْ ݖِ ندِجࣷورُ يࣵلَّ، اَجِ مَنْ جِ؞» ‏70 ݧࣺݒّ دَلْدِ نࣹ:‏ «ڪࣷنْ يَايْ دࣷومُ يࣵلَّ؟» مُ نࣹ لࣹينْ:‏ «يࣹينَ وَخَلْ سࣹينْ بࣷݒّ نࣹ مَنْ لَ؞» ‏71 ݧُ نࣹ نَگْ:‏ «لُ نُيْ دࣷيࣹيتِ سࣹيدࣹ، گَنَّاوْ نُنْ ݖِ سُنُ بࣷݒّ نࣷو ڪࣷ دࣺگّࣹ ݖِ گࣺمِّݧَمْ!‏»
