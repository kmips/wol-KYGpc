نجࣷتْلاَيَلْ اَجِ سَخْ جِ لاَجْ نَ اَبْ گَلَگْ
26
‏1 گَنَّاوْ لࣷولُ، بُ نگࣹينْ دࣹمࣹي ݖَ رࣺيوْ مَ لࣹينْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ سࣺدّ، بَ نگࣹينْ نَنگُ ڪࣷ، دࣴڪّࣹ، ‏2 نَ ڪُ نࣹڪّ ݖِ يࣹينْ سࣵڪّ ݖَ ندࣷورْتࣹلْ مبࣷولࣹمْ مࣹݧّࣹيفُمْ سُوفْ مُ مُ جࣴلࣹ ݖَ سࣹينْ رࣺيوْ، مَ لࣹينْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ دِ جࣷخْ، مُ دࣹفْ ڪࣷ ݖِ ݒَݧࣹ، يٗبُّ ݖَ بࣺرَبْ بَ سࣹينْ يࣵلَّ اَجِ سَخْ جِ تَامࣷو دࣴعࣴلْ تُرَمْ اࣴلّࣴگْ؞ ‏3 نَ اࣵگّ ݖَ سَرْخَلْڪَتْ بَ فَ نࣹڪّ ݖِ يࣷويُ جَنتّ، نࣹ ڪࣷ:‏ «مَا بِرَلْ تࣹيْ جِي فِ سَ ڪَنَمْ يࣵلَّ اَجِ سَخْ جِ، نࣹ اَگْسِ نَا ݖِ رࣺيوْ مِ اَجِ سَخْ جِ گِݧَلࣷونْ سَمَيْ مَامْ، نࣹ مࣷومْ لَ مَيْ جࣷخْ؞» ‏4 نَ سَرْخَلْڪَتْ بِ نَنگࣷو ݒَݧࣹ بِ ݖِ لࣷخࣷومْ، تَاجْ ڪࣷ فِ سࣹينْ ڪَنَمْ سَرْخَلُڪَايُ يࣵلَّ اَجِ سَخْ جِ؞ ‏5 مُ دࣹلُّ نَگْ وَخْ فِ سࣹينْ ڪَنَمْ يࣵلَّ اَجِ سَخْ جِ، نࣹ:‏ «سَمَ مَامْ اَبْ اَرَمࣹينْ بُيْ مࣵنگَانْ لَ وࣷونْ؞ مُ دࣹمْ مِسْرَ، سَنݖࣹ فَ مبࣷولࣷو سُ نࣺيوْ، تࣹ مُجّ دِ اَسْڪَنْ وُ رࣺيْ، اَمْ دࣷولࣹ تࣹ بَرࣹ؞ ‏6 وَا مِسْرَ نَگْ سࣷخࣷرࣹ نُ، مِتِتَلْ نُ، يࣹنْ نُ لِگّࣺييْ بُ دِيسْ؞ ‏7 بَ لࣷولُ اَمࣹي نُ يُوخُ، وࣷو سُنُ يࣵلَّيْ مَامْ اَجِ سَخْ جِ وَلّ، اَجِ سَخْ جِ دࣺگّ نُ، گِسْ سُنُ مِتِتْ، اَڪْ سُنُ ݖࣷونࣷ، اَڪْ سُنُ نࣷتَانگࣹ؞ ‏8 اَجِ سَخْ جِ نَگْ گࣺنّࣹ نُ مِسْرَ ݖِ دࣷولࣹيْ لࣷخࣷومْ اَڪْ ڪࣵتَّنَمْ، ݖِ بِيرْ مُسِبَ مُ رࣺيْ، اَڪِ فِرْندࣹ، اَڪِ ڪࣺيمَانْ؞ ‏9 بَ لࣷولُ وࣺيࣹي مُ اِنْدِ نُ فِ بࣺرَبْ بِي، جࣷخْ نُ رࣺيوْ مِي، رࣺيوْ مُ مࣹيوْ مࣹيڪْ لࣹمْ جِ تُورࣷو؞ ‏10 مࣷو مَ تَخَ اِنْدِ نِي ندࣷورْتࣹلْ مࣹݧّࣹيفُمْ سُوفْ، سِ نگَ مَ جࣷخْ، اَجِ سَخْ جِ؞» سُ ڪࣷ دࣹفࣹي مُ تَاجْ ݒَݧࣹ بِ فِ سࣹينْ ڪَنَمْ يࣵلَّ اَجِ سَخْ جِ، دَلْدِ سُجّٗوتْ فِ سࣹينْ ڪَنَمْ يࣵلَّ اَجِ سَخْ جِ؞ ‏11 مُ بࣺگࣹ نَگْ، مࣷومْ اَڪْ لࣹوࣹينْ ݧࣹيڪْ دࣷخَنْدࣺيمْ يِ دࣴڪّ ݖِ سࣹينْ بِيرْ، مبࣷولࣹمْ لُ بَاخْ لِ ڪࣷ سࣹينْ يࣵلَّ اَجِ سَخْ جِ جࣷخْ، مࣷومْ اَڪْ وَا ڪࣴرَمْ؞
‏12 گَنَّاوْ بُ نگࣹينْ گࣺنّࣹي مبࣷولࣹمْ ݖࣺرُ فُڪّࣹيلُبْ مࣹݧّࣹيفْ، ݖِ اَتُمْ ݧࣹتّࣹيلْ، يࣹمࣷوڪْ اَتُمْ گࣺنّࣹ ݖࣺرُبْ فُڪّࣹيلْ، گَنَّاوْ ݧَارِ اَتْ يُ نࣹڪّ، نَنگࣹينْ ڪࣷ جࣷخْ لࣹوࣹينْ ݧِ اَڪْ دࣷخَنْدࣺيمْ يِ اَڪْ جِرِمْ يِ اَڪْ جࣴتُنْ ݧِ، نگِرْ ݧُ لࣹڪّ بَ رࣹگّ، فَ ݧُ فࣺيتࣹ ݖَ سࣹينِ دࣴڪّ، ‏13 تࣹ نَ ڪُ نࣹڪّ ݖِ يࣹينْ وَخْ فِ سࣹينْ ڪَنَمْ يࣵلَّ اَجِ سَخْ جِ، نࣹ:‏ «جࣴلࣹ نَا سَمَ ڪࣴرْ ݖࣺرُ سࣹلّنگَ بِ، بَ جࣷخْ ڪࣷ لࣹوࣹينْ ݧࣹيڪْ دࣷخَنْدࣺيمْ يࣹيڪْ جِرِمْ يࣹيڪْ جࣴتُنْ يِ، نࣷونࣹي مُ دࣴݒّࣷوڪْ سَ سَنْتَانࣹ بِ نگَ نُ دࣺنڪّ؞ مࣷيُمَ لࣹنّ ݖِ سَيْ سَنْتَانࣹ، تࣹ فࣵتّࣹوُمَ ݖِ لࣹنّ؞ ‏14 لࣹڪُّمَ ݖِ دَرَ ݖِ بِيرْ اَبْ دࣴجْ، جࣴلࣹوُمَ ݖِ دَرَ تࣹ فࣹڪّ مَ سࣷبࣹوُ، تࣹ اِتْ سَرْخَلُمَ ݖِ دَرَ ڪُ دࣹي؞ سَمَ ڪࣵدُّگْ يࣵلَّ اَجِ سَخْ جِ لاَ تࣷݒّ، نࣷونࣹي نگَ مَ ڪࣷ سَنْتࣹ وࣷونْ؞ ‏15 نگَلَّ گࣹيسࣷولْ فَ سَ دࣴڪُّوَايُ سࣹلّنگَ، فَ اَسَمَانْ، تࣹ نگَ بَرْڪࣹيلْ إِسْرَاءِيلْ، سَ ݧࣷݧْ، بَرْڪࣹيلْ سُوفْ سِ نگَ نُ جࣷخْ، نࣷونࣹي نگَ ڪࣷ گِݧَلࣹ وࣷونْ سُنُيْ مَامْ، رࣺيوْ مُ مࣹيوْ مࣹيڪْ لࣹمْ جِ تُورࣷو؞»
ڪٗلّࣴرࣹ فࣹدّلِڪُ نَ
‏16 بࣺسُبْ تࣹيْ جِي سࣹينْ يࣵلَّ اَجِ سَخْ جِ مࣷو لࣹينْ سَنتّ نگࣹينْ جࣴفࣹ دࣷگَلْ يِي، اَڪْ اࣵتّࣹيْ يࣷونْ يِ؞ سࣵمّࣷولࣹينْ ڪࣷو جࣴفࣹ سࣹينْ لࣺݒِّ خࣷلْ، اَڪْ سࣹينْ لࣺݒِّ بَڪَّنْ؞ ‏17 اَجِ سَخْ جِ نگࣹينْ بِرَلْ بࣺسُبْ تࣹيْ، نࣹ مࣷويْ دࣷونْ سࣹينْ يࣵلَّ، تࣹ دِنگࣹينْ دࣷخࣹ اَيْ يࣷونَمْ، دِ سࣵمّ دࣷگَلِ يࣷونَمْ، اَڪِ سَنْتَانࣹيمْ، اَڪْ اࣵتّࣹيْ يࣷونَمْ، تࣹيْ دࣺگّ وَخَمْ؞ ‏18 اَجِ سَخْ جِ اِتْ بِرَلْ نَ بࣺسُبْ تࣹيْ، نࣹ يࣹينَيْ دࣷونْ ݧࣷݧَمْ ݧِ مُ سࣺدّࣷو، نࣷونࣹي مُ لࣹينْ ڪࣷ دِگࣹ وࣷونْ، تࣹ نگࣹينْ دِ سࣵمّ مبࣷولࣹمِ سَنْتَانࣹيمْ؞ ‏19 مࣷو لࣹينْ دِ اَجْ بَ نگࣹينْ تِيمْ مبࣷولࣹمْ خࣹيتْ يِ مُ سࣵڪّ، ݖِ وࣵلُّ بَيْرࣹ، اَڪْ تُرْ اَڪْ تࣹدّنگَ، تࣹ مࣷو لࣹينْ دِ دࣹفْ اَسْڪَنْ وُ سࣹلّ، ݧࣹيلْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ، نࣷونࣹي مُ ڪࣷ دِگࣹي وࣷونْ؞
