اࣹݖّࣹنْ لَيࣷو نَ
7
‏1 سَرْخَلْڪَتْ بُ مَگْ بَ نࣹ ڪࣷ:‏ «ندَخْ لࣷولُ دࣴگّ لَ؟» ‏2 اࣹݖّࣹنْ نࣹ ڪࣷ:‏ «يࣹينْ بࣷڪّ يِ، يࣹينْ سَمَيْ بَايْ، دࣺگْلُلࣹينْ!‏ يࣵلَّ، مِيْ بࣷرࣷومْ دَرَجَ مࣷو فࣹيݧُ وࣷونْ سُنُ مَامْ إِبْرَاهِيمَ، بَ مُ نࣹڪّࣹي رࣺيوُمْ مࣹسࣷݒࣷتَمِ، بَلَ مࣷو سَنݖّ ڪَرَنْ؞ ‏3 مُ نࣹ ڪࣷ:‏ ‹جٗگࣹيلْ سَمْ رࣺيوْ، جٗگࣹ ݖِ سَيْ بࣷڪّ، تࣹ نگَ دࣹمْ ݖَ رࣺيوْ، مِ مَ لَيْ وࣷنْ؞›
‏4 «مُ دَلْدِ جٗگࣹ رࣺيوُمْ وَا ڪَلْدࣹ، سَنݖِ ڪَرَنْ؞ فَ لَ ڪࣷ يࣵلَّ تࣷخَلࣹ گَنَّاوْ بَ بَايَمْ فَاتࣷو، سَنݖَلْ ڪࣷ ݖِ رࣺيوْ مِي نگࣹينْ دࣴڪّ تࣹيْ؞ ‏5 تَخُلْ مُ مُورَلࣹ ڪࣷ لࣹنّ ݖِ رࣺيوْ مِي، دُ سَخْ فُ مُ تࣹگْ تࣵنڪَمْ؞ تࣹيوُلْ مُ دِگْ ڪࣷ، نࣹ مࣷومْ لَيْ مࣷومَلࣹ رࣺيوْ مِ، مࣷوڪْ اَسْڪَنَمْ وَ ڪࣷيْ وُوتُجِ، تࣹ فࣹڪُّلْ سَخْ مُ اَمْ دࣷومْ؞ ‏6 يࣵلَّ نࣹ ڪࣷ:‏ ‹سَوْ اَسْڪَنْ دِنَݧُ تࣷخُجِ بِتِمْ رࣺيوْ، تࣹ دࣹيسْ نَ لࣹينْ دࣹفْ اَيْ جَامْ، مِتِتَلْ لࣹينْ دِيرُ ݧࣹينْتِ تࣺيمࣺيرِ اَتْ (400)؞› ‏7 يࣵلَّ نࣹيتِ ڪࣷ:‏ ‹خࣹيتْ وِ لࣹينْ دِ جَامْلࣷو نَگْ، مَنْ مَا لࣹينْ دِ مبُگَلْ؞ گَنَّاوْ لࣷولُ دِنَݧُ فَ گࣺنّࣹ، جَامُسِ مَ ݖِ بࣺرَبْ بِي؞› ‏8 يࣵلَّ دَلْدِ فَسࣷوڪْ إِبْرَاهِيمَ ڪٗلّࣴرࣹ گُ مُ مࣵنْدَرْگَلࣹ خَرَفْ؞ مࣷو تَخْ إِبْرَاهِيمَ خَرْفَلْ دࣷومَمْ إِسْحَاقَ، بَ مُ جُدّࣷو بَ اَمْ جُرٗومْ ݧࣹتِّ فَنْ؛ إِسْحَاقَ، يَعْقُوبَ، يَعْقُوبَ، فُڪِّ مَامْ يَاڪْ ݧَارْ؞
‏9 «مَامْ يࣷويُ نَگْ اِݧَانࣹ سࣹينْ دࣷومُ بَايْ يُوسُفَ، بَ جَايْ ڪࣷ نجَامْ، ݧُ يٗبُّ ڪࣷ مِسْرَ؞ تࣹيوُلْ يࣵلَّ اࣵندّ اَڪْ يُوسُفَ، ‏10 خࣹتّلِ ڪࣷ ݖِ ݖࣷونࣷومْ يࣺݒّ، رَفࣹتَلْ خࣹلَمْ، مَيْ ڪࣷ اِتَمْ فِرْعَوْنَ بُورُ مِسْرَ بَاخࣹ ڪࣷ؛ مُ دࣹفْ ڪࣷ ڪِلِفَ ݖِ مِسْرَ اَڪْ ݖِ بِيرْ اࣴتُّ ڪࣴرَمْ گࣺݒّ؞
‏11 «ݖِ ڪَوْ لࣷولُ اَبْ خِيفْ دِڪَّلْ مِسْرَ گࣺݒّ اَڪْ رࣺيوُمْ ڪَنَانْ؞ مُ مࣹتِّ لࣷولْ؞ سُنُيْ مَامْ نَگْ اَمَتُݧُ اَبْ دُندّ؞ ‏12 يَعْقُوبَ دࣺگّ نࣹ اَبْ دُندّ اَمْ نَ مِسْرَ؞ ݖَ لَ فَ جࣴڪَّ يࣹبَلْ سُنُ مَامْ يَ؞ ‏13 سࣹينْ ݧَارࣹيلْ بِ يࣷونْ لَ يُوسُفَ خࣵمِّڪُ اَيْ دࣷومِ بَايَمْ، فِرْعَوْنَ نَگْ دࣷخࣹ ݖَ خَمْ ݖࣷسَانُ يُوسُفَ؞ ‏14 ݖِ ڪَوْ لࣷولُ يُوسُفَ يࣹبْلࣹ، اِنْدِ بَايَمْ اَڪْ بࣷڪَّمْ يࣺݒّ، ݧُيْ جُرٗومْ ݧَارْ فُڪِّ نِتْ اَڪْ جُرٗومْ؞ ‏15 نࣷونُ لَ يَعْقُوبَ دࣹمࣹ مِسْرَ، بَ فَاتࣷو فَ، مࣷوڪْ سُنُيْ مَامْ؞ ‏16 گَنَّاوْ گِ لَݧُ تࣷخَلْ سࣹينِ يَخْ ݖَ سِڪࣹمْ، دࣹنݖّ لࣹينْ ݖَ بࣵمّࣹيلْ، بَ إِبْرَاهِيمَ وࣹݖِّڪࣷو وࣷونْ خَالِسْ ݖَ دࣷومِ اَمࣷرْ ݖَ سِڪࣹمْ؞
‏17 «ݧُ دࣹمْ بَ دِگࣹ بَ يࣵلَّ گِݧَلࣷونْ إِبْرَاهِيمَ دِ وَاجَ سࣷتِّ، يࣹمࣷوڪْ خࣹيتْ وِ يࣷڪُّ، دِ گࣴنَ بَرࣹ ݖِ مِسْرَ، ‏18 بَ بࣹنࣹينْ بُورْ بُ خَمُلْ يُوسُفَ فَلُ ݖِ مِسْرَ؞ ‏19 بُورْ بࣷوبُ نَگْ دِ مُوسَاتُ سُنُ خࣹيتْ وِ، دِ مِتِتَلْ سُنُيْ مَامْ، دِ لࣹينْ سࣵنِّلࣷو سࣹينِ لِيرْ، نگِرْ ݧُ بَݧَ دُندّ؞
‏20 «بࣷوبَ لَ مُوسَىٰ جُدُّ، رَفࣹتْ بَ فَ ڪَنَمْ يࣵلَّ، ݧُ يࣷرْ ڪࣷ ݧࣹتِّ وࣹيرْ ݖِ بِيرْ ڪࣴرْ بَايَمْ؞ ‏21 بَ ݧُ ڪࣷ سࣵنّࣹي، دࣷومُ فِرْعَوْنَ جُ جِگࣺينَ ڪࣷ فࣷرَلْ بࣷݒَّمْ، يَرْ ڪࣷ نِ دࣷومَمْ جُ مُ جُرْ؞ ‏22 مُوسَىٰ نَگْ دِ ڪُ مࣷڪَّلْ مبࣷولࣹمْ خَمْ-خَمُ مِسْرَ، تࣹ رࣵݧِّڪُ لࣷولْ ݖِ وَخْ اَڪْ ݖِ جࣴفْ؞
‏23 «بَ ݧࣹينتّ فُڪِّ اَتَمْ مَتࣹي، خࣹلُمْ سࣹيتِ بࣷڪِّ بࣵنِّ إِسْرَاءِيلَمْ دِڪَّلْ ڪࣷ؞ ‏24 ݖِ ڪَوْ لࣷولُ مُ گِسْ ݖَ ڪُ ݧُيْ نࣺيوَلْ دࣷولࣹ، مُ سࣷتْلࣹ ڪࣷ، دَلْدِ رࣹيْ وَايِ مِسْرَ جَ، فࣹيُلْ ڪَ مُ نࣺيوَلْ؞ ‏25 بࣷوبَ مُوسَىٰ فࣷوگْ نَ نࣹ اَيْ بࣷڪَّمْ خَمْ نَݧُ نࣹ ݖِيْ لࣷخࣷومْ لَ لࣹينْ يࣵلَّ دِ اِنْدِلࣹ وَلّ، ندࣹڪࣹ خَمُݧُ ڪࣷ؞ ‏26 ݖَ اࣴلّࣴگْ سَ مُ جُوخْ ݖِ اَيْ يَوُتْ يُيْ خࣹيخْ، دِ لࣹينْ جࣺيمَ جُبَلࣹ، نࣹ لࣹينْ:‏ ‹يࣹينْ اَيْ بࣷڪّ نگࣹينْ، لُ تَخْ نگࣹينْ دِ لࣷرَنْتࣹ؟› ‏27 ڪَ دࣷونْ نࣺيوَلْ دࣷولࣹ مࣷرࣷومْ مَ بࣴمࣴخْ مُوسَىٰ، نࣹ ڪࣷ:‏ ‹يَوْ، ڪُ لَ دࣹفْ نجِيتْ مبَا اࣵتّࣹڪَتْ ݖِ سُنُ ڪَوْ؟ ‏28 خَنَا دَنگَ مَا نَرَ رࣹيْ، نَ نگَ رࣹيࣹ وَايِ مِسْرَ جَ دࣺمبّ؟› ‏29 مُوسَىٰ دࣺگّ ڪࣵدُّ گࣷوگُ، دَلْدِ گࣵدَّايْ، دࣹمْ دَلِ رࣺيوُمْ مَجَنْ، بَ اَمْ فَ ݧَارِ دࣷومْ يُ گٗورْ؞
‏30 «مُ تࣹگْ ݖَ ݧࣹينتّ فُڪِّ اَتْ، مَلاَڪَ فࣹيݧُ ڪࣷ ݖَ مࣵنْدِݝُ تُنْدُ سِنَيِ، ݖِ بِيرْ تࣵڪّ-تࣵڪُّ سَوَرَ وُ جَفَلْ اَبْ گَجّ؞ ‏31 مُوسَىٰ گِسْ ݒࣹيݧُ مࣷومُ، يࣺيمُ؛ مُ دِڪّ بَ جࣹگࣹ ڪࣷ نگِرْ نِيرْ ڪࣷ، ڪࣵدُّگْ بࣷرࣷومْ بِ دَلْدِ جِبْ؞ مُ نࣹ:‏ ‏32 ‹مَنْ مَايْ سَ يࣵلَّيْ مَامْ، مَايْ يࣵلَّيْ إِبْرَاهِيمَ اَڪْ إِسْحَاقَ اَڪْ يَعْقُوبَ؞› مُوسَىٰ تِيتْ بَيْ لࣷخْ، ݧࣹمࣹيتُلَ خࣷولْ؞ ‏33 بࣷرࣷومْ بِ نࣹ ڪࣷ:‏ ‹سُمِّلْ سَيْ دࣵلّ، ندَخْ فِي نگَ تَخَوْ، سُوفْ سُ سࣹلّ لَ؞ ‏34 گِسْ نَا بُ بَاخْ سَمَ ݖࣷونࣷبْ ݧࣷݧْ ݖِ مِسْرَ؛ مَا دࣺگّ سࣹينِ اࣷنڪّ، بَ وࣵݖّ نگِرْ وَلُّسِ لࣹينْ؞ لࣺيگِ نَگْ دِڪَّلْ، مَ يࣹبَلْ لَ مِسْرَ؞›
‏35 «مُوسَىٰ مࣷومُ ݧُ وࣹدِّ وࣷونْ، نࣹ ڪࣷ:‏ ‹ڪُ لَ دࣹفْ ڪِلِفَ اَڪْ اَبْ اࣵتّࣹڪَتْ؟› مࣷومْ نَگْ لَ يࣵلَّ دࣹفْ ڪِلِفَ اَڪْ اَبْ جࣷتْڪَتْ، يࣹبَلْ ڪࣷ ݖِ ݧࣷومْ، مَلاَڪَ مَ فࣹيݧُ مُوسَىٰ ݖَ گَجّ بَ، دَلْدِ ڪࣷيْ جࣷتّلِ يٗبَّنْتࣹ بَ؞ ‏36 مࣷو لࣹينْ گࣺنّࣹ رࣺيوُمْ مِسْرَ، مࣷو دࣹفْ اَيْ ڪࣺيمَانْ اَڪِ فِرْندࣹ ݖَ رࣺيوْ مَ، اَڪْ ݖَ گࣺيجُ بَرَخْ يَ، اَڪْ ݖَ مࣵنْدِݝْ مَ، دِيرُ ݧࣹينتّ فُڪِّ اَتْ؞ ‏37 مُوسَىٰ مࣷومُ مࣷو نࣷونْ بࣵنِّ إِسْرَاءِيلْ:‏ ‹يࣷنࣹنتّ بُ مࣹلْ نِ مَنْ لَ لࣹينْ يࣵلَّ دِ فࣹيݧَلَلْ ݖِ سࣹينْ بِيرِ بࣷڪّ؞› ‏38 مࣷومْ اِتْ مࣷو نࣹڪّࣷونْ اَڪْ مبࣷولࣷو مَ ݖَ مࣵنْدِݝْ مَ، مࣷوڪْ مَلاَڪَ مَ دࣷونْ وَخْ اَڪْ مࣷومْ ݖَ ڪَوْ تُنْدُ سِنَيِ ݖَ سُنُ وࣹتُ مَامْ يَ؞ مࣷو جࣷتْ ݖِ ڪࣵدُّ يُيْ دُندّ، نگِرْ جࣷتّلِ نُ ڪࣷ؞ ‏39 ڪࣷوڪُ لَ سُنُيْ مَامْ نَنگُوُلْ وࣷونَ دࣺگَّلْ؞ دَݧُ ڪࣷو فَرْ خَرَبْ، نَمَّ وَلْبَتِڪُ دࣹلُّ مِسْرَ؞ ‏40 ݧُ نࣹ هَـٰرُونَ:‏ ‹سࣵڪَّلْ نُ اَيْ يࣵلَّ يُ نُ جِيتࣹ، ندَخْ مُوسَىٰ مِي نُ گࣺنّࣹ رࣺيوُمْ مِسْرَ، خَمُنُ لُ ڪࣷ دَلْ؞› ‏41 جَنتّ يࣷويُ لَݧُ سࣵڪّ جࣴمُّ سࣴلُّ، دࣹفَلْ جࣴمّ جَ اَبْ سَرَخْ، دِ بࣵنّࣹيخࣷو لُ ݧُ سࣵڪّࣹ سࣹينِ لࣷخࣷ؞ ‏42 يࣵلَّ نَگْ دࣴدُّ لࣹينْ، بࣵيِّ لࣹينْ، ݧُيْ جَامُ بِدِّيوْ يِ، نِ ݧُ ڪࣷ بِنْدࣹ ݖِ تࣺيرࣹبْ يࣷنࣹنتّ يِ، نࣹ:‏
‹يࣹينْ وَا ڪࣴرْ إِسْرَاءِيلْ، سَرَخُ جُرْ اَڪْ يࣹنࣹينْ سَرَخْ،
ندَخْ مَنْ نگࣹينْ ڪࣷ دَانْ اِنْدِلْ دِيرُ ݧࣹينتّ فُڪِّ اَتْ ݖَ مࣵنْدِݝْ مَ؟
‏43 مࣷلࣷگْ سࣹينْ تُورْ مِ، نگࣹينْ يٗبَّالࣹ خَيْمَبْ جَامࣷوڪَايَمْ،
اَڪْ سࣹينْ بِدِّيوُ يࣵلَّ جِ ݧُ نَا رࣹفَنْ،
جࣴمّ يࣷويُ نگࣹينْ سࣵڪّ، دِ لࣹينْ سُجّٗوتَلْ!‏
ڪࣷنْ نَگْ مَا لࣹينْ دِ تࣷخَلْ ݖَ وࣵلاَّ بَبِلࣷنْ؞›
‏44 «خَيْمَبْ سࣹيدࣹ بَا نگَ وࣷونْ اَڪْ سُنُيْ مَامْ ݖَ مࣵنْدِݝْ مَ، ݧُ سࣵڪّࣹ ڪࣷ نَ ڪࣷ ڪَ دࣷونْ وَخْ اَڪْ مُوسَىٰ سَنْتࣹ، دࣴݒَّلࣹ ڪࣷوڪْ مِسَالْ مَ مُوسَىٰ گِسࣷونْ؞ ‏45 خَيْمَ بَ لَ سُنُيْ مَامْ جࣷتْ، دُگَلْ ڪࣷ ݖِ ڪِلِفْتࣹيفُ يࣷسُوࣹ ݖَ بِيرْ رࣺيوُمْ خࣹيتْ، يَ لࣹينْ يࣵلَّ دࣵقَلْ؞ خَيْمَ بَ نࣹڪّ فَ، بَ ݖَ جَنْتِ دَاوُدَ؞ ‏46 دَاوُدَ، مَ يࣵلَّ بَاخࣹ وࣷونْ، سࣵڪُّ وࣷونْ نَ اَمْ مࣵڪَّانْ نگِرْ يࣵلَّيْ يَعْقُوبَ؞ ‏47 وَايࣹ سُلࣹيْمَانْ مࣷو ڪࣷ مُجّ تَبَخَلْ ڪࣴرْ؞
‏48 «مࣷونَ اَجِ ڪَوࣹ جِ دُ دࣴڪّࣹ لُ لࣷخࣷ دࣹفَرْ، مࣷويْ لَ يࣷنࣹنتّ بَ نࣹ:‏
‏49 ‹اَسَمَانْ سَمَ نگࣵنگُنࣹي،
سُوفْ دِ سَمَ ندࣴگَّسْتَلْ؞
اَنَ ڪࣴرْ گُ نگࣹينْ مَيْ تَبَخَلْ؟
بࣷرࣷومْ بࣹي ڪࣷ وَخْ؞
مُ نࣹ:‏ اَنَ بَنْ بࣺرَبْ لاَيْ نࣷݒّلࣷو؟
‏50 لࣷولُ لࣺݒّ، خَنَا دُ سَمَ لࣷخࣷو ڪࣷ سࣵڪّ؟›
‏51 «يࣹينَ دࣴگࣴرْ بࣷݒّ، يࣹينَيْ بࣷرࣷومِ خࣷلْ اَڪْ نࣷݒِّ يࣺيفَرْ!‏ دُنگࣹينْ نࣷݒّࣹي تࣴ نࣷو گُ سࣹلّ گِ؟ مَامْ يَ، دࣷومْ يَ رࣹڪّ!‏ ‏52 ڪَنْ ݖِ يࣷنࣹنتّ يِ لَ سࣹينِ مَامْ فِتْنَالُلْ؟ ݧࣷو رࣹيْ دࣺمبّ ݧَ يࣺگْلࣹ وࣷونْ ݧࣴوُگْ اَجِ جُبْ، جِ نگࣹينْ وࣷرْ تࣹيْ، بٗومْ ڪࣷ؞ ‏53 يࣹينْ دࣹ يࣹينَ جࣷتࣷونْ ݖِ يࣷونُ مُوسَىٰ، وِ اَيْ مَلاَڪَ جࣷتّلِ، تࣹ تَخُلْ نگࣹينْ جࣴفࣹ ڪࣷ؞»
بٗومْ نَݧُ اࣹݖّࣹنْ
‏54 نَڪَ لَݧُ دࣺگّ لࣷولُ، سࣹينْ خࣷلْ فُدُّ، ݧُيْ يࣺيُ ݖِ ڪَوَمْ؞ ‏55 اࣹݖّࣹنْ نَگْ فࣹيسࣹ نࣷو گُ سࣹلّ گِ، نࣹ جࣵڪّ اَسَمَانْ، دَلْدِ گِسْ لࣹيرُ يࣵلَّ، گِسْ عِيسَىٰ تَخَوࣹ فَ ندِجࣷورُ يࣵلَّ؞ ‏56 مُ نࣹ:‏ «اَسَمَانَ نگِي مَ گِسْ، مُ اُبِّڪُ، دࣷومُ نِتْ ڪَا نگِي تَخَوْ فِ ندِجࣷورُ يࣵلَّ!‏»
‏57 بَ لࣷولُ اَمࣹي ݧُ يُوخُ ݖَ ڪَوْ، دَرْ سࣹينِ نࣷݒّ، دَلْدِ اࣵندّ، نࣹ مِلِبْ ݖَ ڪَوَمْ؛ ‏58 ݧُ وَتْ-وَتࣹي ڪࣷ، گࣺنّࣹ ڪࣷ دࣴڪّ بَ، دِ ڪࣷ دٗورْ اَيْ دࣷجْ؞ فࣹڪّ نَ سࣹيدࣹ يَ سُمِّ سࣹينِ مبُبّ، جُوگْ ݖَ تࣵنڪِ بࣹنّ خَلࣹلُ گٗورْ بُ ݧُيْ وَخْ سٗولْ؞
‏59 ݧُيْ دٗورْ اࣹݖّࣹنْ اَيْ دࣷجْ، مࣷومْ مُيْ ݧَانْ يࣵلَّ، نَانْ:‏ «سَنگّ عِيسَىٰ، نَنگُلْ سَمَ بَڪَّنْ!‏» ‏60 مُ نࣹ گُرُبْ، سَمݒّ بࣴتِ اٗومَمْ، دَلْدِ اࣵدُّ ݖَ ڪَوْ، نࣹ:‏ «سَنگّ بِ، بُ لࣹينْ تࣷݒّ بِي بࣵڪَّارْ؞» نَڪَ لَ وَخْ لࣷولُ، دَلْدِ نࣹلَوْ؞
