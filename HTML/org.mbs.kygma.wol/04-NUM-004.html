يࣹنࣹينْ ڪُرࣺلِ لࣹوࣹينَ نگِ اَڪْ سࣹينِ سَسْ
4
‏1 اَجِ سَخْ جِ وَخَتِ مُوسَىٰ اَڪْ هَـٰرُونَ، نࣹ:‏
‏2 «لࣹوࣹينْ ݧِ اَسْڪَنࣷو ݖِ ڪࣹيَتْ، وَݧّلࣹينْ لࣹينْ بࣷݒّ اَڪْ بࣷݒّ، نَ ݧُ بࣷڪّࣹ سࣹينِ لࣵنگّ، اَڪْ سࣹينْ ڪࣴرِ مَامْ؞ ‏3 دَلࣹلࣹينْ ݖِ گٗورْ ݧِ اَمْ فَنْوࣹيرِ اَتْ، بَ جُرٗومْ فُڪِّ اَتْ، مُيْ مبࣷولࣹمْ ݧِ يࣹلّࣷو بࣷڪّ ݖِ ڪُرࣺلْ، گِ لِگّࣺييُ بِيرْ خَيْمَبْ ندَجࣹ مِ وَرْ؞ ‏4 لِگّࣺييُ ڪࣹيَتࣹينْ ݧِ ݖِ بِيرْ خَيْمَبْ ندَجࣹ مِ مࣷو دِ وࣵلُّ جُمْتُڪَايْ يُ سࣹلّ يࣹي سࣹلّ؞
‏5 «بُ دَلْ بِ دࣹي جٗگْ، نَ هَـٰرُونَ اَڪْ دࣷومَمْ يُ گٗورْ دُگّ، وࣺڪِّ ڪِيرَايْ لِ، دِ رِدࣷبْ بِيرْ بِ، تࣹ ݧُ مُورࣹ ڪࣷ گَالُ سࣹيدࣹسْ ڪٗلّࣴرࣹ گِ؞ ‏6 نَݧُ لَلاَتْ ݖَ ڪَوْ ڪِيرَايَلْ دࣹرُ ݒِيݒِ، تࣹ ݧُ لَلَتِ ݖَ ڪَوْ، ندِمࣷ لُ لࣺݒّ بَخَ، دࣷورَ رࣷوفْ نجࣵݒُّيْ گَالْ گِ؞
‏7 «تَابَلُ مبُرُمْ تࣹيوَلْ مِ، نَݧُ ݖِ لَلْ ندِمࣷ لُ بَخَ، تࣹگْ ݖِ ندَبْ يࣹيڪْ بࣵتُّ يࣹيڪْ نجَقْ يࣹيڪْ تَخْ يِ ݧُيْ سࣷتّࣹي سَرَخِ تُورُ، دࣷورْ ݖࣹي تࣹگْ مبُرُ مࣷومُ دُلْ وُوتࣹ؞ ‏8 نَݧُ مُورࣹ يࣷويُ ندِمࣷ لُ خࣷنقّ ݖُرّ، تࣹ ݧُ لَلاَتْ ݖَ ڪَوْ مَلاَنُمْ دࣹرُ ݒِيݒِ، دࣷورَ رࣷوفْ نجࣵݒُّيْ تَابَلْ جِ؞
‏9 «سُ ڪࣷ دࣹفࣹي ݧُ جࣴلْ ندِمࣷ لُ بَخَ، مُورࣹ ڪࣷ تࣹگُڪَايُ لࣵمݒّ بِ، اَڪِ لࣵمْݒَمْ اَڪِ ݧِيمَمْ اَڪِ اَنْدَمْ، اَڪْ مبࣷولࣹمْ ندَبِ دِوَمْ يَ ݧُ ݖَيْ جࣴفࣷو؞ ‏10 نَݧُ بࣷولࣹ تࣹگُڪَايُ لࣵمݒّ بࣹيڪْ مبࣷولࣹمْ لَ مُيْ اࣵنْدَلْ، ݧُ اࣴمْبࣹ لࣺݒّ مَلاَنُمْ دࣹرُ ݒِيݒِ، تࣹ تࣹگْ ڪࣷ ݖِ ڪَوْ تَگَرْ بِ ݧُ ڪࣷيْ گࣵدّࣷو؞ ‏11 سَرْخَلُڪَايُ وُرُسْ بِ¤، نَݧُ لَلْ ݖِ ڪَوَمْ ندِمࣷ لُ بَخَ، لَلاَتْ ݖَ ڪَوْ مَلاَنُمْ دࣹرُ ݒِيݒِ، دࣷورَ رࣷوفْ نجࣵݒُّيْ سَرْخَلُڪَايْ بِ؞ ‏12 نَݧُ جࣴلْ مبࣷولࣹمْ جُمْتُڪَايْ يِ ݧُيْ لِگّࣺييࣹ ݖِ بِيرْ بࣺرَبْ بُ سࣹلّ بِ، ݧُ اࣴمْبࣹ ڪࣷ ندِمࣷ لُ بَخَ، مُورࣹ ڪࣷ مَلاَنُمْ دࣹرُ ݒِيݒِ، دࣷورْ ڪࣷو تࣹگْ ݖِ ڪَوْ تَگَرْ بِ ݧُ ڪࣷيْ گࣵدّࣷو؞ ‏13 نَݧُ دِنْدِ نِينِتِ سَرْخَلُڪَايُ خࣵنجَرْ بِ، تࣹ ݧُ لَلْ ݖِ ڪَوْ سَرْخَلُڪَايْ بِ، ندِمࣷ لُ خࣷنقّ ݖُرّ؞ ‏14 نَݧُ تࣹگْ ݖَ ڪَوْ ندِمࣷ لَ مبࣷولࣹمْ ندَبْ، يَ ݧُيْ لِگّࣺييࣹ ݖَ سَرْخَلُڪَايْ بَ:‏ اَندّ يࣹيڪْ ݖَݒُّڪَايِ يࣵݒّ يࣹيڪْ ݧِيتُڪَايْ يࣹيڪْ ڪࣴلّ يِ؛ جُمْتُڪَايِ سَرْخَلُڪَايْ بِ يࣺݒّ دَالْ، تࣹ ݧُ لَلْ ݖَ ڪَوْ، مَلاَنُمْ دࣹرُ ݒِيݒِ، دࣷورَ رࣷوفْ نجࣵݒُّيْ سَرْخَلُڪَايْ بِ؞
‏15 «بُ دَلْ بِ وَرࣹي جٗگْ، نَ هَـٰرُونَ اَڪْ دࣷومَمْ يُ گٗورْ مُورْ يࣴفْ يُ سࣹلّ يِ، اَڪْ مبࣷولࣹمْ جُمْتُڪَايِ بࣺرَبْ بُ سࣹلّ بِ بَ نࣷݒِّ؞ گَنَّاوْ لࣷولُ لَ اَسْڪَنُ ڪࣹيَتْ دِ دِڪّ، نگِرْ گࣵدُّ ڪࣷ؞ سُ ڪࣷ دࣹفࣹي دُݧُ لاَلْ دَرَ لُ سࣹلّ، بَيْ دࣹي؞ لࣷولُ لَ ڪࣹيَتࣹينْ ݧِ دِ گࣵدُّ ݖِ وࣵلُّ خَيْمَبْ ندَجࣹ مِ؞
‏16 «اࣹلَسَرْ دࣷومُ هَـٰرُونَ سَرْخَلْڪَتْ بِ نَگْ، اَبْ سَيْتࣷومْ مࣷو دِ دِوُ لࣵمݒّ بِ، اَڪْ ݖُورَايْ لِ، اَڪْ سَرَخُ ݒࣹݒّ بِ دُلْ جَاسْ، اَڪْ دِوُ ݒَلْ گِ؞ مࣷويْ فࣺيتࣹوࣷو جَامࣷوڪَايْ بِ بࣺݒّ، اَڪْ مبࣷولࣹمْ لِ ݖِ بِيرَمْ، بَ ݖِ يࣴفْ يُ سࣹلّ يِ، اَڪِ جُمْتُڪَايَمْ؞»
‏17 اَجِ سَخْ جِ وَخَتِ بَ تࣹيْ اَڪْ مُوسَىٰ اَڪْ هَـٰرُونَ، نࣹ لࣹينْ:‏ ‏18 «بُلࣹينْ يٗبّࣹ ڪࣹيَتࣹينْ ݧࣹيڪْ سࣹينِ لࣵنگّ، لُ ݧُ لࣹينْ دِ دَگّࣹ ݖِ بِيرْ لࣹوࣹينْ ݧِ؞ ‏19 نِي نگࣹينْ دِ دࣹفْ اَڪْ ݧࣷومْ، بَ بُ ݧُ جࣹگࣹي يࣴفْ يُ سࣹلّ يࣹي سࣹلّ، دِ دُندّ تࣹ دُݧُ ݖَ دࣹي:‏ هَـٰرُونَ اَڪِ دࣷومَمْ ݧࣷويْ دُگّ، دَلْدِ سࣺدّ ڪُ نࣹڪّ ݖِ ݧࣷومْ لِگّࣺييَمْ اَڪْ لَ مُيْ گࣵدُّ؞ ‏20 بُ ڪࣹيَتࣹينْ ݧِ دُگّ، دِ خࣷولْ لُ مُ گࣵتّ گࣵتّ، يࣴفْ يُ سࣹلّ يِ؞ لُ ڪࣷ مࣷيْ ݧُ دࣹي؞»
‏21 گَنَّاوْ لࣷولُ اَجِ سَخْ جِ وَخْ مُوسَىٰ، نࣹ ڪࣷ:‏ ‏22 «ݧِ اَسْڪَنࣷو ݖِ گࣹرْسࣷنْ اِتَمْ، نَنگَ لࣹينْ وَݧّ بࣷݒّ اَڪْ بࣷݒّ، نَ ݧُ بࣷڪّࣹ سࣹينْ ڪࣴرِ مَامْ اَڪْ سࣹينِ لࣵنگّ؞ ‏23 لِ دَلࣹ ݖِ گٗورْ ݧِ اَمْ فَنْوࣹيرِ اَتْ، بَ جُرٗومْ فُڪِّ اَتْ نگَيْ لِمْ، ݧُيْ مبࣷولࣹمْ ݧِ لِگّࣺييُ خَيْمَبْ ندَجࣹ مِ وَرْ؞ ‏24 گࣹرْسࣷنࣹينْ ݧِ نَگْ، لِي مࣷو دِ سࣹينْ ݖࣺرْ ݖِ وࣵلُّ لِگّࣺييْ بِ، اَڪْ لِ ݧُ وَرَ گࣵدُّ:‏ ‏25 ݧࣷويْ گࣵدُّ ڪِيرَايِ جَامࣷوڪَايْ بِ، اَڪْ خَيْمَبْ ندَجࣹ مࣹيڪْ مَلاَنَمْ، اَڪْ مَلاَنُمْ دࣹرُ ݒِيݒِ بِ ݖِ ڪَوْ، اَڪْ رِدࣷبْ بُنتّ خَيْمَبْ ندَجࣹ مِ؛ ‏26 اَڪْ سࣷرِ اࣴتّ بِ، اَڪْ ڪِيرَايَلْ بُنتّ اࣴتّ بِ وࣴرْ جَامࣷوڪَايْ بࣹيڪْ سَرْخَلُڪَايْ بِ، اَڪْ سࣹينِ بُومْ اَڪْ مبࣷولࣹمْ جُمْتُڪَايْ يَ ݧُ ݖَيْ لِگّࣺييࣹ؞ مبࣷولࣹمْ لࣹيسْ ݖَيْ دࣹفْ، ݧࣷو ڪࣷيْ لِگّࣺييْ؞ ‏27 مبࣷولࣹمْ سَسُ گࣹرْسࣷنࣹينْ ݧِ ݖِ مبࣷولࣹمْ لُ ݧُيْ گࣵدُّ اَڪْ مبࣷولࣹمْ لُ ݧُيْ لِگّࣺييْ، ݖِ ندِگَلَلْ هَـٰرُونَ اَڪِ دࣷومَمْ لَيْ دࣷونْ؞ ݧࣷو لࣹينْ دِ سَسْ مبࣷولࣹمْ لُ ݧُ وَرَ گࣵدُّ؞ ‏28 لࣷولࣷويْ ݖࣺرُ گࣹرْسࣷنࣹينْ ݧِ، ݖِ وࣵلُّ خَيْمَبْ ندَجࣹ مِ، تࣹ لࣷولُ لَݧُ لࣹينْ دࣺنڪّ، اِتَمَرْ دࣷومُ هَـٰرُونَ سَرْخَلْڪَتْ بِ دِ لࣹينْ ݖَ سَيْتُ؞
‏29 «ݧِ اَسْڪَنࣷو ݖِ مࣹرَرِ اِتَمْ، نَ ݧُ بࣷڪّࣹ سࣹينِ لࣵنگّ اَڪْ سࣹينْ ڪࣴرِ مَامْ، نࣷونُ نگَ لࣹينْ دِ لِمࣹ، ‏30 دَلࣹ ڪࣷ ݖِ ݧِ اَمْ فَنْوࣹيرِ اَتْ، بَ جُرٗومْ فُڪِّ اَتْ، ݧُيْ مبࣷولࣹمْ ݧِ يࣹلّࣷو بࣷڪّ ݖِ ڪُرࣺلْ، گِ لِگّࣺييُ خَيْمَبْ ندَجࣹ مِ وَرْ؞ ‏31 لِي مࣷويْ لِ ݧُ سَسࣷو گࣵدُّ ݖِ سࣹينْ مبࣷولࣹمْ لِگّࣺييْ ݖِ خَيْمَبْ ندَجࣹ مِ:‏ خࣵنقِ جَامࣷوڪَايْ بِ اَڪِ گَلَنَمْ اَڪِ جࣴنَمْ اَڪِ تࣹگࣷومْ؛ ‏32 مُيْ جࣴنْ يِ وࣴرْ اࣴتّ بِ، اَڪْ سࣹينِ تࣹگُ اَڪْ سࣹينِ بُومْ، مبࣷولࣹمْ سࣹينْ جُمْتُڪَايْ يُ ݧُ ݖَيْ لِگّࣺييࣹ؞ لِمَلْ-لࣹينْ لࣹينْ جُمْتُڪَايْ يِ ݧُ وَرَ گࣵدُّ؞ ‏33 لࣷولࣷويْ ݖࣺرُ مࣹرَرࣹينْ ݧِ، ݖِ وࣵلُّ خَيْمَبْ ندَجࣹ مِ ݖِ نجِيتَلْ سَرْخَلْڪَتْ بِ اِتَمَرْ دࣷومُ هَـٰرُونَ؞»
لِمْ نَݧُ لࣹوࣹينْ ݧِ لِگّࣺييْ وَرْ
‏34 ݖِ ڪَوْ لࣷولُ مُوسَىٰ اَڪْ هَـٰرُونَ اَڪْ ڪِلِفَگْ مبࣷولࣷو مَ لِمْ ڪࣹيَتࣹينْ ݧَ، نَ ݧُ بࣷڪّࣹ سࣹينِ لࣵنگّ اَڪْ سࣹينْ ڪࣴرِ مَامْ؞ ‏35 ݧُ دَلࣹ ڪࣷ ݖِ ݧِ اَمْ فَنْوࣹيرِ اَتْ، بَ جُرٗومْ فُڪِّ اَتْ، ݧُيْ مبࣷولࣹمْ ݧِ يࣹلّࣷو بࣷڪّ ݖِ ڪُرࣺلْ، گِ لِگّࣺييُ خَيْمَبْ ندَجࣹ مِ وَرْ؞ ‏36 ݧَ ݧُ لِمَلْ سࣹينِ لࣵنگّ دِ ݧَارِ جُنِّ اَڪْ جُرٗومْ ݧَارْ تࣺيمࣺيرْ اَڪْ جُرٗومْ فُڪِّ گٗورْ (2750)؞ ‏37 ݧࣷوݧُ لَݧُ لِمَلْ لࣵنگِ ڪࣹيَتࣹينْ ݧِ، ݧࣷومْ ݧࣺݒّ لَ لِگّࣺييُ خَيْمَبْ ندَجࣹ مِ وَرْ؞ مُوسَىٰ اَڪْ هَـٰرُونَ ݧࣷو لࣹينْ لِمࣷونْ، نَ ڪࣷ اَجِ سَخْ جِ سَنْتَانࣹي، مُوسَىٰ جࣷتّلِ؞
‏38 ݧَ ݧُ لِمَلْ گࣹرْسࣷنࣹينْ ݧِ، نَ ݧُ بࣷڪّࣹ سࣹينِ لࣵنگّ اَڪْ سࣹينْ ڪࣴرِ مَامْ، ‏39 دَلࣹ ڪࣷ ݖِ ݧِ اَمْ فَنْوࣹيرِ اَتْ، بَ جُرٗومْ فُڪِّ اَتْ، مبࣷولࣹمْ ݧَ يࣹلّࣷو بࣷڪّ ݖِ ڪُرࣺلْ، گِ لِگّࣺييُ خَيْمَبْ ندَجࣹ مِ وَرْ، ‏40 تࣹ ݧُ لِمَلْ لࣹينْ سࣹينِ لࣵنگّ اَڪْ سࣹينْ ڪࣴرِ مَامْ، ݧَارِ جُنّࣹيڪْ جُرٗومْ بࣹنِّ تࣺيمࣺيرْ اَڪْ فَنْوࣹيرِ گٗورْ لَݧُ (2630)؞ ‏41 ݧࣷوݧُ لَݧُ لِمَلْ لࣵنگِ گࣹرْسࣷنࣹينْ ݧِ، تࣹ ݧࣷومْ ݧࣺݒّ لَ لِگّࣺييُ خَيْمَبْ ندَجࣹ مِ وَرْ؞ مُوسَىٰ اَڪْ هَـٰرُونَ ݧࣷو لࣹينْ لِمࣷونْ ݖِ ندِگَلَلْ اَجِ سَخْ جِ؞
‏42 ݧَ ݧُ لِمَلْ لࣵنگِ مࣹرَرࣹينْ ݧِ، نَ ݧُ بࣷڪّࣹ سࣹينِ لࣵنگّ، اَڪْ سࣹينْ ڪࣴرِ مَامْ، ‏43 دَلࣹ ڪࣷ ݖِ ݧِ اَمْ فَنْوࣹيرِ اَتْ، بَ جُرٗومْ فُڪِّ اَتْ، مبࣷولࣹمْ ݧَ يࣹلّࣷو بࣷڪّ ݖِ ڪُرࣺلْ، گِ لِگّࣺييُ خَيْمَبْ ندَجࣹ مِ وَرْ، ‏44 تࣹ ݧُ لِمَلْ لࣹينْ سࣹينِ لࣵنگّ، ݧࣹتِّ جُنِّ اَڪْ ݧَارْ تࣺيمࣺيرِ گٗورْ (3200) لَݧُ؞ ‏45 ݧࣷوݧُ لَݧُ لِمَلْ لࣵنگِ مࣹرَرࣹينْ ݧِ، مُوسَىٰ اَڪْ هَـٰرُونَ ݧࣷو لࣹينْ لِمࣷونْ، نَ ڪࣷ اَجِ سَخْ جِ سَنْتَانࣹي، مُوسَىٰ جࣷتّلِ؞
‏46 مبࣷولࣹمْ لࣹوࣹينْ ݧَ مُوسَىٰ اَڪْ هَـٰرُونَ، اَڪْ ڪِلِفَيْ إِسْرَاءِيلْ لِمࣷونْ، نَ ݧُ بࣷڪّࣹ سࣹينِ لࣵنگّ، اَڪْ سࣹينْ ڪࣴرِ مَامْ، ‏47 دَلࣹ ڪࣷ ݖِ ݧِ اَمْ فَنْوࣹيرِ اَتْ، بَ جُرٗومْ فُڪِّ اَتْ، ݧُيْ مبࣷولࣹمْ ݧَ يࣹلّࣷو بࣷڪّ ݖِ ڪُرࣺلْ، گِ لِگّࣺييُ خَيْمَبْ ندَجࣹ مِ وَرْ، ‏48 سࣹينُبْ لِمْ نࣹڪّ نَ جُرٗومْ ݧࣹتِّ جُنِّ اَڪْ جُرٗومِ تࣺيمࣺيرْ اَڪْ جُرٗومْ ݧࣹتّ فُڪِّ گٗورْ (8580)؞ ‏49 ݖِ ندِگَلَلْ اَجِ سَخْ جِ مُوسَىٰ جࣷتّلِ، لَݧُ لࣹينْ لِمْ، بَ سَسْ ڪُ ݖِ نࣹڪّ اَبْ لِگّࣺييَمْ اَڪْ لَ مُيْ گࣵدُّ؞ نَ ڪࣷ اَجِ سَخْ جِ سَنْتࣹ مُوسَىٰ، نَ لَݧُ لࣹينْ لِمࣹ؞
