يَحْيَىٰ وَارࣹ نَ
3
‏1 ݖَ فُڪّࣹيلُ اَتْ اَڪْ جُرٗومْ ݖِ نگُورُگْ بُورْ بُ مَگْ بَ تِبࣹيرْ سࣹسَارْ، بَ ݒࣷݧْسࣴ ݒِلاَتْ يِلِفࣹي دِيوَانُ يُدࣹ، اࣹرࣷدّ يِلِفْ دِيوَانُ گَلِلࣹ، فِلِبْ مَگَمْ يِلِفْ دِيوَانُ اِتُرࣹ اَڪْ بُ تَرَڪࣷنِتْ، لِسَݧَسْ يِلِفْ اَبِلࣹنْ، ‏2 جَنتّ يࣷويُ اَنَسْ اَڪْ ڪَيِفْ دࣹي سَرْخَلْڪَتْ يُ مَگْ يَ، ݖَ لَ ڪࣵدُّگْ يࣵلَّ دِڪَّلْ يَحْيَىٰ دࣷومُ سࣵڪَّرْيَا، ݖَ مࣵنْدِݝْ مَ؞ ‏3 يَحْيَىٰ دُگّ مبࣷولࣹمْ اࣵلّ، بِ وࣴرْ دࣹخُ يُرْدَنْ، دِ وَارࣹ لُ جࣴمْ ݖِ تُوبࣹيلْ گُ نِتْ ݧِ وَرَ فࣺسَلࣹ اَگْ سٗوبُ ݖِمْ ندࣷخْ، نگِرْ ݧُ جࣺگَّلْ لࣹينْ سࣹينِ بࣵڪَّارْ؞
‏4 لࣷولُ لَݧُ بِندّ ݖِ تࣺيرࣹبْ وَخِ يࣷنࣹنتّ يࣵلَّ اࣹسَيِ نࣹ:‏
«بَاتَ نگِ خَاݖࣷو ݖَ مࣵنْدِݝْ مَ، نَانْ:‏
‹خࣵلّ-لࣹينْ يࣷونُ بࣷرࣷومْ بِ،
جُبَلْ-لࣹينْ اَيْ خࣵلَّمْ؞
‏5 خُرْ يࣺݒّ اَيْ سࣴڪّࣹ،
تُندّ وُ مَگࣷو وُ ندَوْ مَاسَلࣷو،
يࣷونْ يَ لُنڪࣷونْ، جُبَّنْتِڪُ،
يَ ݧَگَسࣷونْ، نࣷويْ،
‏6 بَ وࣺݒّ سُوخْ گِسْ مُݖّ گِ يࣵلَّ دࣷگَلْ؞›»
‏7 يَحْيَىٰ نَگْ دِ وَخْ مبࣷولࣷو مَيْ ݧࣴوْ نگِرْ مُ سٗوبْ لࣹينْ ݖِ ندࣷخْ، نَانْ لࣹينْ:‏ «يࣹينْ نجُرُمْ جَانْ مࣹي، ڪُ لࣹينْ اࣵرْتُ، بَ نگࣹينْ دِ دَوْ سࣵنجّ مِيْ دِڪّ؟ ‏8 ڪࣷنْ نَگْ جࣴفࣹلࣹينْ جࣴفْ جُ يࣹلّࣷو اَگْ تُوبࣹيلْ، تࣹ بَݧَ نَخْ سࣹينْ بࣷݒّ نَانْ:‏ ‹نُنْ، إِبْرَاهِيمَ مࣷويْ سُنُ مَامْ،› ندَخْ مَا لࣹينْ ڪࣷ وَخْ، دࣷجْ يِي، يࣵلَّ مَنْ نَ ݖࣹي سࣵڪَّلْ إِبْرَاهِيمَ اَوْ اَسْڪَنْ؞ ‏9 تࣹ اِتَمْ سࣺمِّݧْ خࣵݖِّلْ نَ رࣹينُ گَرَبْ بَ نࣷݒِّ؞ ڪࣷنْ نَگْ گࣺݒّ گَرَبْ گُ ݖِ مࣹݧُّلْ دࣷومْ يُ بَاخْ، دࣹيسْ ڪࣷيْ گࣷرْ، سࣵنِّ ݖَ سَوَرَ سَ؞»
‏10 مبࣷولࣷو مَ اِتَمْ لاَجْ ڪࣷ نࣹ ڪࣷ:‏ «اَنَ لُ نُ وَرْ نَگْ؟» ‏11 مُ نࣹ لࣹينْ:‏ «ڪُ اَمْ ݧَارِ تُرْڪِ، نَ سࣺدّ ڪِ اَمُلْ؞ ڪُ اَمْ اَوْ ݧَمْ اِتَمْ، نَ دࣹفࣹ نࣷونُ؞» ‏12 اَيْ جُوتِڪَتْ اِتَمْ دِڪّ، نگِرْ سٗوبُسِ؛ ݧُ نࣹ يَحْيَىٰ:‏ «سࣴرِݧْ بِ، اَنَ لُ نُ وَرْ؟» ‏13 مُ نࣹ لࣹينْ:‏ «خَنَا بَݧَ سࣵڪُّ لࣹنّ لُ مࣷيْ لِ جَادُ؞» ‏14 اَيْ تَڪّ-دࣹرْ اِتَمْ نࣹ ڪࣷ:‏ «نُنْ نَگْ، نُ نُيْ دࣹفْ؟» مُ نࣹ لࣹينْ:‏ «بُلࣹينْ نࣺيوَلْ ڪࣹنّ دࣷولࣹ، مبَا دِ دُورَلْ ڪࣹنّ نگِرْ جࣴلَلْ ڪࣷ دَرَ؛ دࣷيْلࣷو لࣹينْ سࣹينْ ݒࣹيࣷورْ؞»
‏15 ݖِ ڪَوْ لࣷولُ اَسْڪَنْ وَ دِ سࣺينْتُ، ݧࣺݒّ دِ سࣹيتَاتْ ݖِ سࣹينْ خࣹلْ، ندَخْ يَحْيَىٰ مࣷويْ اَلْمَسِ؞ ‏16 يَحْيَىٰ وَخْ لࣹينْ ݧࣷومْ ݧࣺݒّ نࣹ لࣹينْ:‏ «مَنْ دࣹ ݖِ ندࣷخْ لاَ لࣹينْ دِ سٗوبْ؞ وَايࣹ ڪُ مَ سُتَ نگِ ݧࣴوْ، ڪࣷوڪُ، بُومِ ݖَرَخَمْ سَخْ، يࣹيࣷووُمَ ڪࣷ ڪࣷو فࣹݖِّلْ؞ مࣷومْ، ݖِ نࣷو گُ سࣹلّ گِ اَڪْ سَوَرَ لَ لࣹينْ دِ سٗوبْ¤؞ ‏17 اَبْ لَيࣷومَ نگِ ݖِ لࣷخࣷومْ، مࣷويْ جࣺرِ دࣵگَّامْ، بَ مُ سࣹتْ وࣹݖّ؛ ݒࣹݒّ مِ لَيْ فَتْ ݖِمْ سࣵقَمْ، وَايࣹ خَتَخْ بِ، دَ ڪࣷيْ لَڪّ ݖِ سَوَرَ سُ دُلْ فࣹيْ مُڪّ؞»
‏18 بَرࣹ نَ يࣹنࣹينْ دࣺنڪَانࣹ يُ يَحْيَىٰ دࣷونْ خَمَلࣹ نِتْ ݧِ خِبَارُ جࣵمّ بِ؞ ‏19 اࣹرࣷدّ بࣷرࣷومْ دِيوَانْ بَ نَگْ، يَحْيَىٰ سِڪّࣷونْ نَ ڪࣷ ندَخْ اࣹرࣷجࣵدّ جَبَرُ مَگَمْ جَ مُ نࣹڪَّلࣷونْ، اَڪْ مبࣷولࣹمْ يࣹنࣹينْ ݧَاوْتࣺيفْ يُ اࣹرࣷدّ دࣹفْ؞ ‏20 مُ تࣹگَتِ لࣹنࣹينْ ݖَ لࣷولُ لࣺݒّ:‏ جࣵݒّ يَحْيَىٰ، تࣴجْ ݖَ ڪَسࣷ بَ؞
يَحْيَىٰ سٗوبْ نَ عِيسَىٰ ݖِ ندࣷخْ
‏21 نࣷونُ يَحْيَىٰ دࣷونْ سٗوبࣹ ݧࣺݒّ ݖِمْ ندࣷخْ، نِ لَ دࣹمࣹ بَ عِيسَىٰ دِڪّ، مُ سٗوبْ ڪࣷ ݖِ ندࣷخْ؞ ݖِ بِيرْ لࣷولُ عِيسَىٰ يࣺڪَّتِ ڪࣵدُّگْ ݧَانْ، اَسَمَانْ اُبِّڪُ؞ ‏22 بَ لࣷولُ اَمࣹي نࣷو گُ سࣹلّ گِ مࣷو بِنْدࣷو نِ ݒِتَخْ، وࣵݖّ ݖِ ڪَوَمْ، بَاتْ جِبࣹ اَسَمَانْ، نࣹ:‏ «يَا دِ سَمَ دࣷومْ، سَمَ سࣷݒّࣹ؛ يَا مَيْ بࣵنّࣹيخَلْ؞»
لُ جࣴمْ ݖِ ݖࣷسَانُ عِيسَىٰ اَلْمَسِ
‏23 عِيسَىٰ ݖِ بࣷݒَّمْ بَ مُيْ دࣷورْ، اَمࣷونْ نَ لُ تࣷلُّ ݖِ فَنْوࣹيرِ اَتْ، دِ دࣷومُ يُوسُفَ، نِ ڪࣷ خࣹلْ مَنَ جࣵݒّࣹ؞
يُوسُفَ، اࣹلِ مࣷويْ بَايَمْ، ‏24 اࣹلِ، مَتَتْ؞
مَتَتْ، لࣹوِ؛ لࣹوِ، مࣹلْڪِ؛
مࣹلْڪِ، يَنَيِ؛ يَنَيِ، يُوسُفَ؛
‏25 يُوسُفَ، مَتَݖَسْ؛ مَتَݖَسْ، اَمࣷسْ؛ اَمࣷسْ، نَوُمْ؛ نَوُمْ، اࣹسْلِ؛ اࣹسْلِ، نَگَيِ؞
‏26 نَگَيِ، مَاتْ؛ مَاتْ، مَتَݖَسْ؛ مَتَݖَسْ، سࣹمࣹيِنْ؛ سࣹمࣹيِنْ، يࣷسࣹگْ؛ يࣷسࣹگْ، يࣷدَ؞
‏27 يࣷدَ، يࣷوَنَنْ؛ يࣷوَنَنْ، رࣹسَ؛
رࣹسَ، سࣷرࣷبَبࣹلْ؛ سࣷرࣷبَبࣹلْ، سࣹلْݖࣹلْ؛ سࣹلْݖࣹلْ، نࣹرِ؛
‏28 نࣹرِ، مࣹلْڪِ؛ مࣹلْڪِ، اَدِ؛ اَدِ، ڪࣷسَمْ؛ ڪࣷسَمْ، اࣹلْمَدَمْ؛ اࣹلْمَدَمْ، اࣹرْ؞
‏29 اࣹرْ، عِيسَىٰ؛ عِيسَىٰ، اࣹلْيࣹسࣹرْ؛ اࣹلْيࣹسࣹرْ، يࣷرِمْ؛ يࣷرِمْ، مَتَتْ؛ مَتَتْ، لࣹوِ؛
‏30 لࣹوِ، سِمࣹيࣷنْ؛ سِمࣹيࣷنْ، يُدَ؛ يُدَ، يُوسُفَ؛ يُوسُفَ، يࣷنَمْ؛ يࣷنَمْ، اࣹلْيَڪِمْ؞
‏31 اࣹلْيَڪِمْ، مࣹلࣹيَ؛ مࣹلࣹيَ، مࣹنَ؛
مࣹنَ، مَتَتَ؛ مَتَتَ، نَتَنْ؛
نَتَنْ، دَاوُدَ؞ ‏32 دَاوُدَ مࣷويْ دࣷومُ يࣹسࣹ؛
يࣹسࣹ، اࣷبࣹدّ؛ اࣷبࣹدّ، بࣷوَسْ؛ بࣷوَسْ، سَلَ؛ سَلَ، نَسࣷنْ؞
‏33 نَسࣷنْ، اَمِنَدَبْ؛ اَمِنَدَبْ، اَدْمِنْ؛ اَدْمِنْ، اَرْنِ؛
اَرْنِ، اࣹسْرࣷنْ؛ اࣹسْرࣷنْ، ݒࣹرࣹسْ؛ ݒࣹرࣹسْ، يُدَ؛ ‏34 يُدَ، يَعْقُوبَ؛ يَعْقُوبَ، إِسْحَاقَ؛ إِسْحَاقَ، إِبْرَاهِيمَ؛
إِبْرَاهِيمَ، تࣹرَا؛ تࣹرَا، نَخࣷرْ؛
‏35 نَخࣷرْ، سࣹرُگْ؛ سࣹرُگْ، رࣹوْ؛ رࣹوْ، ݒࣹلࣹگْ؛ ݒࣹلࣹگْ، اࣹبࣹرْ؛ اࣹبࣹرْ، سࣹلَ؛
‏36 سࣹلَ، ڪࣹنَنْ؛ ڪࣹنَنْ، اَرْݒَگْسࣵدّ؛
اَرْݒَگْسࣵدّ، سࣹمْ؛ سࣹمْ، نُوحٍ؛ نُوحٍ، لࣹمࣹگْ؛
‏37 لࣹمࣹگْ، مَتُسَلࣹمْ؛ مَتُسَلࣹمْ، اࣹنࣷگْ؛ اࣹنࣷگْ،
يࣹرࣹدّ؛ يࣹرࣹدّ، مَلَلࣹيلْ؛ مَلَلࣹيلْ، ڪࣹنَنْ؛
‏38 ڪࣹنَنْ، اࣹنࣷسْ؛ اࣹنࣷسْ، سࣹتْ؛ سࣹتْ، آدَمَ؛
آدَمَ، يࣵلَّ؞
