بࣵڪَّارْ يِلِفَتُلْ گࣴمْڪَتُبْ عِيسَىٰ
6
‏1 اَنَ لُ لࣷولُ دِ تࣹڪِّ نَگْ؟ ندَخْ ڪࣷنْ دَنُيْ وࣺيࣹ بࣵڪَّارْ، نگِرْ يِوُ يࣵلَّ گࣴنَ يࣷڪُّ؟ ‏2 مُڪّ ڪَيْ!‏ نُنْ ݧِ ندࣹي تࣵگَّلࣹ اَڪْ بࣵڪَّارْ بَ نࣷݒِّ، اَنَ نُ نُيْ سَخࣷوتࣹي بࣵڪَّارْ؟ ‏3 دَنگࣹينَ اُمݒّلࣹ نࣹ مبࣷولࣹمْ نُنْ ݧِ ݧُ سٗوبْ ݖِ عِيسَىٰ اَلْمَسِ، ݖِگْ دࣹييَمْ لَݧُ نُ سٗوبَالࣹ؟ ‏4 ݖِ سٗوبُ گࣷوگُ لَݧُ نُ بࣷولࣹ اَڪْ مࣷومْ سُولْ، نگِرْ نࣷونࣹي اَلْمَسِ دࣹڪّࣹي ݖِ مࣵگُّگْ بَايْ بِ، نُنْ اِتَمْ نُ دࣹڪِّ بَيْ دُندّ دُنْدِنْ وُ يࣹيسْ؞
‏5 گَنَّاوْ دࣹيسْ نࣷو بࣹنَّلࣹ اَڪْ مࣷومْ ݖِ دࣹي گُ ݧُ نِرُلࣹ اَڪْ دࣹييَمْ، دࣹيسْ نُيْ بࣹنَّلࣹ اِتَمْ اَڪْ مࣷومْ ݖِ ندࣹڪِّتࣹ لُ نِرُ ندࣹڪِّتࣹيمْ؞ ‏6 لِ نُ بِرْ مࣷو دِ نُنْ ݧَ وࣷونْ دࣺمبّ، ݖَ بَنْتُبْ عِيسَىٰ بَ لࣹيسْ نُ دَاجَالࣹ، نگِرْ خࣹوِلْ سُنُ يَرَمُ بࣵڪَّارْ، بَ نُ نࣷݒّࣹي دࣷونْ اَيْ جَامْ، بࣵڪَّارْ دِ سُنُبْ سَنگّ؞ ‏7 ندَخْ ڪِ دࣹي بَ نࣷݒِّ، جࣷخࣹيسْ نَ ڪࣷ اࣵتّࣹبْ ڪُ جُبْ، مُ تࣵگّࣷوڪْ نࣷتࣹيلُ بࣵڪَّارْ؞
‏8 گَنَّاوْ نࣷو دࣹييَنْدࣷو اَڪْ اَلْمَسِ نَگْ، گࣴمْ نَنُ نࣹ نࣷويْ دُنْدَنْدࣷو اَڪْ مࣷومْ؞ ‏9 ندَخْ ڪَتْ خَمْ نَنُ نࣹ اَلْمَسِ مِ دࣹڪِّ دࣷوتُ دࣹي، تࣹ ندࣹي مَنَلَتُ ڪࣷ دَرَ؞ ‏10 دࣹييَمْ گَ مُ دࣹي ݖَ لَ دࣹييَلْ بࣵڪَّارْ بࣹنّ يࣷونْ بَ فࣵوّ، تࣹ دُندّ گِ مُيْ دُندّ تࣹيْ، يࣵلَّ لَ ڪࣷيْ دُنْدَلْ؞
‏11 نَڪَ نࣷونُ، يࣹينْ اِتَمْ جࣵݒّࣹلࣹينْ سࣹينْ بࣷݒّ نِتْ ݧُ دࣹييَلْ بࣵڪَّارْ، تࣹ دِ دُنْدَلْ يࣵلَّ، ݖِ عِيسَىٰ اَلْمَسِ؞ ‏12 ڪࣷنْ نَگْ بُ بࣵڪَّارْ يِلِفْ سࣹينْ يَرَمُ ندࣹي وِي، بَ دِ لࣹينْ نَنگُلࣷو اَيْ خࣹمّࣹمْتࣺيفَمْ يُ بࣷنْ؞ ‏13 بُلࣹينْ جࣺبَّلْ سࣹينِ ݖࣺرْ بࣵڪَّارْ، مُ دࣹفْ لࣹينْ جُمْتُڪَايِ نجُبَدِ؛ گَنَّاوْ يࣹينَ دࣹڪِّ بَيْ دُنْدَاتْ، جࣺبَّلْ-لࣹينْ يࣵلَّ سࣹينْ بࣷݒّ تࣹ جࣺبَّلْ ڪࣷ سࣹينِ ݖࣺرْ، نگِرْ دࣹفْ لࣹينْ اَيْ جُمْتُڪَايِ نجُبْ؞ ‏14 ندَخْ ڪَتْ بࣵڪَّارْ يِلِفَتُ لࣹينْ، ندࣹ دُ ݖِ ڪِلِفْتࣹيفُ يࣷونُ مُوسَىٰ نگࣹينْ نࣹڪّ، وَايࣹ ݖِ اَوْ يِوْ نگࣹينْ تࣵبِّ؞
ڪُ گࣴمْ عِيسَىٰ، اَگْ نجُبْ نگَيْ سُرْگَوُ
‏15 بَ ݖِ لَنْ بࣷوگْ؟ ندَخْ ڪࣷنْ دَنُيْ بࣵڪَّارْ، گَنَّاوْ دُ ݖِ ڪِلِفْتࣹيفُ يࣷونُ مُوسَىٰ لَنُ نࣹڪّ، وَايࣹ ݖِ اَوْ يِوْ لَنُ تࣵبِّ؟ دࣺيتْ ڪَيْ!‏ ‏16 خَنَا خَمُلࣹينْ نࣹ ڪࣺݒّ ڪࣷو جࣺبَّلْ سَ بࣷݒّ، نَنگُلْ ڪࣷ، اَبْ جَامَمْ نگَ؟ يَارْ نگࣹينْ دِيْ جَامْ، بࣵڪَّارْ دِ سࣹينْ سَنگّ، جࣴمࣹ لࣹينْ ݖِ ندࣹي؛ مبَا نگࣹينْ دࣺگَّلْ يࣵلَّ، مُ جࣷخْ لࣹينْ اࣵتّࣹبْ ݧُ جُبْ؞ ‏17 وَايࣹ ݖَنتّ ݧࣹيلْ نَ يࣵلَّ، ندࣹ اَيْ جَامِ بࣵڪَّارْ نگࣹينْ وࣷونْ، وَايࣹ گَنَّاوْ گِ، سࣹينُبْ خࣷلْ نگࣹينْ دࣺگَّلࣹ مِسَالُبْ اࣵلُّوَ، جِ نگࣹينْ دِ ندࣺنڪَانࣹيمْ؞ ‏18 نࣷونُ لࣹيسْ لࣹينْ يِوࣹي، تࣵگَّلࣹ لࣹينْ اَڪْ نࣷتࣹيلُ بࣵڪَّارْ، نگࣹينْ دࣷونْ اَيْ سُرْگَيْ نجُبْ؞
‏19 نِ ڪࣷ نِتْ مَنَ دࣺگّࣹ رࣹڪّ، نِ لاَ لࣹينْ وࣷيࣷفَلَلࣹ ڪࣵدُّ گِي، ندَخْ سࣹينْ نࣺيوْ دࣷولࣹيْ نِتُ نࣹينْ؞ نࣷونࣹي نگࣹينْ جࣺبَّلࣹ وࣷونْ سࣹينِ ݖࣺرْ، دࣹفْ لࣹينْ ݧُيْ جَامِ سࣷبࣹ، اَڪْ ندࣴنگّتࣹ گُ جُرْ ندࣴنگّتࣹ، نࣷونُ نگࣹينْ وَرَ جࣺبَّلࣹ سࣹينِ ݖࣺرْ، دࣹفْ لࣹينْ ݧُيْ جَامِ نجُبْ گُ جُرْ سࣹلّنگَ؞ ‏20 بَ نگࣹينْ دࣹي جَامِ بࣵڪَّارْ، تࣹيوُلࣹينْ وࣷونَ مࣷومْ سࣹينْ بࣷݒّ، ݖِ سࣹينْ دِگَّنْتࣹيڪْ نجُبْ؞ ‏21 سࣹينْ جࣴفِ دࣺمبّ جَ لࣹينْ گࣵݖّࣹيلْ تࣹيْ، اَنَ نجَرِݧْ لُ مُ لࣹينْ جُرَلْ؟ مُجُگْ جࣴفْ جَ دࣹ، سࣹينُگْ دࣹي لَ دࣷونْ دࣷونْ؞ ‏22 وَايࣹ تࣹيْ بِ ݧُ لࣹينْ يِوࣹي، بَ نگࣹينْ دࣷونْ اَيْ سُرْگَيْ يࣵلَّ، اَگْ سࣹلّنگَ لَ لࣹينْ لࣷولُ جُرَلْ، تࣹ مُجْ گَ دِ تࣹخࣹ گُ سَخْ دࣵڪّ؞ ‏23 ندَخْتࣹ ݒࣹيࣷورُ بࣵڪَّارْ مࣷويْ دࣹي، وَايࣹ يِوْ وِ يࣵلَّ دِ مَيࣹ مࣷويْ تࣹخࣹ گُ سَخْ دࣵڪّ فِ اَلْمَسِ عِيسَىٰ، سُنُ بࣷرࣷومْ؞
