لُ جࣴمْ ݖِ اࣵلُّوَيْ دࣷجْ يُ بࣹيسْ يِ
34
‏1 اَجِ سَخْ جِ وَخَتِ مُوسَىٰ نࣹ ڪࣷ:‏ «يࣹتَّلْ ݧَارِ اࣵلُّوَيْ دࣷجْ يُ مࣹلْ نِ يُ جࣴڪّ يَ، مَ بِندّ ݖِ ڪࣵدُّ يَ وࣷونْ ݖَ اࣵلُّوَ يُ جࣴڪّ، يَ نگَ تࣷجْ؞ ‏2 نَنگَ وَاجَلْ اࣴلّࣴگْ، تࣹ نگَ تࣹيلَ يࣺيگْسِ ڪَوْ تُنْدُ سِنَيِ وِي، بَ تࣹيوْسِ فِ مَنْ ݖِ نجࣷبَّخْتَلُ تُندّ وِ؞ ‏3 بُ ڪࣹنّ اࣵندّ اَڪْ يَوْ، تࣹ بُ ڪࣹنّ خَندّ فࣹنّ ڪَوْ تُندّ وِ؞ جُرْ گِ سَخْ بُمُ فࣷرْسِ ݖِ سُوفُ تُندّ وِ، دُ گُ گُدّ، دُ گُ گࣵتّ؞» ‏4 مُوسَىٰ نَگْ يࣹتّ ݧَارِ اࣵلُّوَيْ دࣷجْ يُ مࣹلْ نِ يُ جࣴڪّ يَ، بَ نࣷݒِّ تࣹيلَ خࣴيْ، يࣺيگِ ݖَ تُنْدُ سِنَيِ، نَ ڪࣷ ڪࣷ اَجِ سَخْ جِ سَنْتࣹ؞ مَ نگَ ݝࣵبّ ݧَارِ اࣵلُّوَيْ دࣷجْ يَ؞
‏5 اَجِ سَخْ جِ نَگْ وࣵݖّ ݖِ بِيرْ اَوْ نِيرْ، تَخَوْ اَڪْ مُوسَىٰ فࣷوفَ، دَلْدِ بِرَلْ تُرَمْ، نࣹ:‏ «اَجِ سَخْ جِ؞» ‏6 ݖِ ڪَوْ لࣷولُ اَجِ سَخْ جِ رࣷمبّ فَ ڪَنَمْ مُوسَىٰ، دِ بِرَلْ نَانْ:‏ «اَجِ سَخْ جࣹي، اَجِ سَخْ جِ، يࣵلَّ جِيْ يࣴرࣴمْ، دِ يࣺوࣺنࣹ، مࣹرْ مَ دِيبْ، نگࣷرْ گَاڪْ ڪٗلّࣴرࣹ گَ بَرࣹ؞ ‏7 ڪࣹييْ سَخَلْ نگࣷرْ بَ ݖِ جُنِّيْ مَاسْ، دِ بَلࣹ تࣷوݧْ اَڪْ مࣷيْ اَڪْ بࣵڪَّارْ، تࣹ دُ ݧࣵڪَّ تࣷݒّ ڪُ تࣷوݧْ، اَگْ تࣷوݧَانگࣹيمْ، خَنَا دِ تࣷݒّ دࣷومْ تࣷوݧُ وَاجُرْ، بَ ݖَ سࣴتْ يَاڪْ سࣴتَاتْ يَ؞» ‏8 بَ لࣷولُ اَمࣹي مُوسَىٰ نࣹ گُرُبْ فَ سُوفْ، سُجّٗوتْ، ‏9 نࣹ ڪࣷ:‏ «اࣺييْ بࣷرࣷومْ بِ، ندࣹگَمْ دَجْ نَا سَوْ يِوْ، رِڪّ اࣵنْدَلْ اَڪْ نُنْ رࣹڪّ، بࣷرࣷومْ بِ، اَڪْ لِ مبࣷولࣷو مِيْ دࣴگࣴرْ بࣷݒّ لࣺݒّ!‏ جࣺگَّلْ نُ سُنُ ݧَاوْتࣺيفْ اَڪْ سُنُ بࣵڪَّارْ رࣹڪّ، تࣹ نگَ سࣺدّࣷو نُ؞»
اَجِ سَخْ جِ يࣹيسَلْ نَ ڪٗلّࣴرࣹ گِ
‏10 اَجِ سَخْ جِ نࣹ ڪࣷ:‏ «مَا نگِ نِي دِ فَسْ ڪٗلّࣴرࣹ اَڪْ يࣹينْ؞ ݖِ سَ ڪَنَمْ مبࣷولࣷو مࣺݒّ لاَيْ دࣹفْ اَيْ جَلࣷورࣹ يُ ڪࣹنّ مَسُلَ اَمَلْ ݖِ اࣵدِّنَ سࣺݒّ، اَڪْ ݖِ اَسْڪَنْ يࣺݒّ، تࣹ مبࣷولࣷو، مِ نگَ نࣹڪّ ݖِ سࣹينْ بِيرْ مࣺݒّ اَيْ گِسْ جَلࣷورࣹيْ اَجِ سَخْ جِ، جَلࣷورࣹ جُ رَگْلُ جِ مَ لَيْ دࣹفَلْ؞ ‏11 بࣵيِّلࣹينْ خࣹلْ بُ بَاخْ لِ مَ لࣹينْ سَنتّ بࣺسُبْ تࣹيْ جِي؞ مَا نگِي دِ لࣹينْ دࣵقَلْ اَمࣷرࣹينْ ݧِ اَڪْ ڪَنَانࣹينْ ݧِ اَڪْ اࣹتّࣹينْ ݧِ اَڪْ ݒࣹرِسࣹينْ ݧِ اَڪْ اࣹوࣹينْ ݧِ اَڪْ يࣹبُسࣹينْ ݧِ؞ ‏12 مࣷيْتُلࣹينَ وٗلّࣴرَنْتࣹيڪْ نِتِ رࣺيوْ مَ نگࣹينْ جࣴمْ، بَلَ ݧࣷو دࣷونْ ندࣴگّ-سࣴرࣴخْ ݖِ سࣹينْ بِيرْ؞ ‏13 سࣹينِ سَرْخَلُڪَايْ ڪَيْ نگࣹينْ دِ مࣵبّ، سࣹينْ تُورِ دࣷجْ، نگࣹينْ رَجَخࣹ، سࣹينْ خࣹرْ يِ ݧُيْ جَامࣷو اَسࣹرَ تُورْ مِ، نگࣹينْ گࣷرْ، ‏14 ندَخْ وَرُلࣹينَ جَامُ جࣹنࣹينْ يࣵلَّ، نگِرْ اَجِ سَخْ جِ، تُرُ ‹ڪُ فِيرْ ڪِ› لَيْ وُيࣷو، تࣹ يࣵلَّ جُ فِيرْ لَ؞ ‏15 بُلࣹينْ وٗلّࣴرَنْتࣹيڪْ نِتِ رࣺيوْ مَ، لُ ڪࣷ مࣷيْ دِنَݧُ گࣵنݖّتࣷوڪْ سࣹينِ تُورْ، رࣹنْدِلْ سࣹينِ تُورْ اَيْ سَرَخْ، وࣷو لࣹينْ ݖَ، نگࣹينْ لࣹڪّ ݖَ سࣹينِ سَرَخْ؞ ‏16 تࣹ اِتْ بُ نگࣹينْ جࣴلَلࣹي سࣹينِ دࣷومْ جَبَرْ ݖَ سࣹينْ دࣷومْ يُ جِگࣺينْ، سࣹينْ جِگࣺينْ ݧࣷوݧَ دِنَݧُ گࣵنݖّتࣷوڪْ سࣹينِ تُورْ، بَ خِيرْ سࣹينْ دࣷومْ يُ گٗورْ ݖِ گࣵنݖّتࣷوڪْ سࣹينْ تُورْ يࣷويُ؞
‏17 «اَيْ تُورْ يُ ݧُ مٗولࣹ وࣹݧْ، بُلࣹينْ ڪࣷ مٗولْ؞
‏18 «مࣵگَّلُگْ اَيُ بࣺسُ مبُرُ مُ اَمُلْ لَوِيرْ، وࣷرْمَالْ-لࣹينْ ڪࣷ؞ دِيرُ جُرٗومْ ݧَارِ فَنْ نگࣹينْ ݖِيْ لࣹڪّ مبُرُ مُ اَمُلْ لَوِيرْ، نِ مَ لَ ڪࣷ سَنْتࣹ بُ اࣵݒَّمْ تَخَوࣹي ݖِ وࣹيرُ اَبِيبْ، ندَخْتࣹ وࣹيرُ اَبِيبْ نگࣹينْ گࣺنّࣹ فَ مِسْرَ؞
‏19 «بࣺݒّ تَاوْ بُ گٗورْ مَايْ بࣷرࣷومْ، بَ ݖِ لࣺݒّ لُ گٗورْ لُ جُرْ تَاوْلࣷو، مُيْ نَگْ مبَا گࣵتّ؞ ‏20 وَايࣹ لُ گٗورْ لُ مبَامْ تَاوْلࣷو، جࣷتࣹلࣹينْ ڪࣷ اَبْ گࣵتّ، مُ وُوتُ ڪࣷ؞ سُ نگࣹينْ ڪࣷ جࣷتࣹوُلْ لُ ڪࣷ وُوتُ، دَمّلࣹينْ لࣷوسْ وِ؞ ڪࣺݒّ ڪُيْ تَاوْ بُ گٗورْ ݖِ سࣹينِ دࣷومْ اِتْ، نَنگࣹينْ ڪࣷ جࣷتْ؞ تࣹ اِتَمْ بُ ڪࣹنّ تَخَوࣹ لࣷخࣷيْ نࣹينْ فِ سَمَ ڪَنَمْ؞
‏21 «جُرٗومْ بࣹنِّ فَنْ نَگْ نگࣹينْ دِ لِگّࣺييْ، وَايࣹ بࣺسُبْ جُرٗومْ ݧَارࣹيلْ بَ، دَلْ-لُلࣹينْ؞ بُ نگࣹينْ دࣷونْ بࣹيْ، بُ نگࣹينْ دࣷونْ گٗوبْ، دَلْ-لُلࣹينْ؞
‏22 «مࣵگَّلُگْ اَيِ بࣺسِ نگٗوبْ، نَنگࣹينْ ڪࣷ وࣷرْمَالْ، بُ نگࣹينْ دِ گٗوبْ گُبِ بࣹلࣹ يِ جࣴڪَّ ݧࣷرْ، مࣵگَّلُ بࣺسُ فَرْمࣵنگُ مِ ݖِ ندࣹيتࣹلْ اَتْ مِ اِتَمْ، مࣵگَّلْ-لࣹينْ ڪࣷ؞ ‏23 ݧࣹتِّ يࣷونْ ݖِمْ اَتْ لَ سࣹينْ گٗورْ ݧࣺݒّ دِ تࣹيوْ فِ ڪَنَمْ بࣷرࣷومْ بِ اَجِ سَخْ جِ، يࣵلَّيْ إِسْرَاءِيلْ؞ ‏24 مَا لࣹينْ دِ دࣵقَلْ اَيْ خࣹيتْ، بَ يَاتَلْ سࣹينْ سُوفْ، تࣹ دِيرُ ݧࣹتِّ يࣷونْ يِ نگࣹينْ دِ تࣹيوِ ݖِمْ اَتْ فِ ڪَنَمْ يࣵلَّ اَجِ سَخْ جِ، ڪࣹنّ دُ ݖِ وُتَ نَنگُ سࣹينُمْ رࣺيوْ؞
‏25 «بُلࣹينْ رࣹنْدِ سَمَ سَرَخْ، دِ بࣷولࣹ دࣹرࣹتْ جࣹيڪْ لُ اَمْ لَوِيرْ، تࣹ گࣵتُّ سَرَخُ بࣺسُبْ مُݖّ، بُ ݖِ دَرَ فَنَانْ بَ بࣴتْ سࣹتْ؞
‏26 «نگࣴنࣹيلِ مࣹݧّࣹيفْ مِ جࣴڪَّ ݧࣷرࣹ ݖِ سࣹينْ سُوفْ، نَنگࣹينْ ڪࣷ يٗبُّ سࣹينْ ڪࣴرْ يࣵلَّ اَجِ سَخْ جِ؞
«بُلࣹينْ تࣷگّࣹ اَبْ تࣹفْ سࣷووُمْ يَايَمْ؞»
‏27 اَجِ سَخْ جِ تࣹگْ ݖَ نࣹ ڪࣷ:‏ «نَنگَ بِندّ يِي سَنْتَانࣹ، ندَخْ ݖِ ڪَوْ سَنْتَانࣹ يِي لاَ فَسࣷوڪْ يَوْ ڪٗلّࣴرࣹ، يَاڪْ بࣵنِّ إِسْرَاءِيلْ؞»
‏28 فࣷوفَ لَ مُوسَىٰ دࣹسْ اَڪْ اَجِ سَخْ جِ دِيرُ ݧࣹينتّ فُڪِّ گُدِّ اَڪْ ݧࣹينتّ فُڪِّ بࣴݖّࣴگْ تࣹ لࣹڪُّلْ، نَانُلْ؞ اَجِ سَخْ جِ نَگْ بِندّ ݖِ اࣵلُّوَ يِ، فُڪِّ سَنْتَانࣹ يِ لَلْ ڪٗلّࣴرࣹ گِ؞
مُوسَىٰ دࣹلُّ نَ دَلْ بَ
‏29 ݖِ ڪَوْ لࣷولُ مُوسَىٰ وࣵݖّ تُنْدُ سِنَيِ، ݝࣵبّ ݧَارِ اࣵلُّوَيْ سࣹيدࣹ سَ؞ نَڪَ لَ مُوسَىٰ دِ وࣵݖّࣹ ݖَ تُندّ وَ، دࣹرُ ڪَنَمْ گَ دِ لࣹرْخَتْ تࣹ يࣺگُ ڪࣷ؞ بࣷوبَ ݧَارࣷومْ بَاڪْ يࣵلَّ مࣷو وَرَلْ لࣷولَ؞ ‏30 نَڪَ لَ هَـٰرُونَ اَڪْ بࣵنِّ إِسْرَاءِيلْ گࣺݒّ خࣷولاَتْ مُوسَىٰ، گِسُݧُ لُ مࣷيْ دࣹرُ ڪَنَمْ گَيْ لࣹرْخَتْ، ݧُ دَلْدِ ڪࣷيْ رَگَلَ جࣹگࣹ؞ ‏31 مُوسَىٰ نَگْ وࣷو لࣹينْ، هَـٰرُونَ اَڪْ نجِيتِ مبࣷولࣷو مَ يࣺݒّ دِڪّ، مُ وَخْ اَڪْ ݧࣷومْ؞ ‏32 گَنَّاوْ لࣷولُ بࣵنِّ إِسْرَاءِيلْ گَ ݖَ دࣹسْ گࣺݒّ دࣷورَ جࣹگࣹسِ، مُ دࣺنڪّ لࣹينْ مبࣷولࣹمْ لَ اَجِ سَخْ جِ وَخࣷونْ اَڪْ مࣷومْ ݖَ ڪَوْ تُنْدُ سِنَيِ؞ ‏33 مُوسَىٰ وَخْ اَڪْ ݧࣷومْ بَ نࣷݒِّ، دَلْدِ مُورࣷو ڪَالَ، لࣵقࣹ ڪࣷ ڪَنَمَمْ؞
‏34 سَا يُ مُوسَىٰ تࣹيوِوَانْ فَ ڪَنَمْ اَجِ سَخْ جِ، نگِرْ وَخْ اَڪْ مࣷومْ، دَيْ مُرِ ڪَالَ گِ بَ بَ مُيْ گࣺنّ؛ بُ گࣺنّࣹي، يࣹگّࣹ بࣵنِّ إِسْرَاءِيلْ لَ ݧُ ڪࣷ سَنتّ؞ ‏35 بࣵنِّ إِسْرَاءِيلْ نَگْ دِ گِسْ ڪَنَمُ مُوسَىٰ اَڪْ نَ دࣹرُ ڪَنَمْ گَيْ لࣹرْخَتࣹ؞ سُ ڪࣷ دࣹفࣹي مُوسَىٰ دࣹلُّ مُورُوَاتْ، لࣵقْ ڪَنَمَمْ، بَ بَ مُيْ دࣹمَاتْ، نگِرْ وَخْ اَڪْ اَجِ سَخْ جِ؞
