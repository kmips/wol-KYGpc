ندࣴنگّتࣹ جُ مَگْ مࣷويْ جِيتُ بࣺسُ سَنگّ بِ
2
‏1 لࣺيگِ نَگْ، بࣷڪّ يِ، ݖِ وࣵلُّ دࣹلُّسِگْ سُنُ سَنگّ عِيسَىٰ اَلْمَسِ، اَڪْ سُنُ دَجࣹجِ فَ مࣷومْ، لِ نُ لࣹينْ دِ ݧَانْ مࣷو دِ:‏ ‏2 بُ لࣹينْ لࣹنّ گَاوَ خَݧْ سَگࣷ بَ نگࣹينْ ݧࣵڪَّ اࣵندّ اَڪْ سࣹينُمْ خࣹلْ، مبَا مُ دِ لࣹينْ جَاخَلْ؛ دُ لُ ݧُ نࣹ ڪࣵدُّگْ وَخْيُ لَ، دُ گࣹنّ ڪࣵدُّ مبَا بَتَاخَلْ بُ ݧُ نُ مࣷومَلࣹ، تࣹ دِ لࣹينْ گࣴمْلࣷو نࣹ بࣺسُ سَنگّ بِ دِڪّ نَ بَ نࣷݒِّ؞ ‏3 بُ لࣹينْ ڪࣹنّ نَخࣹ نࣹنّ، ندَخْ بࣺسْ بَ دُ دِڪَّگُمْ، تࣹ فِݒُّ گُ مَگْ گَ جࣴڪُّلَ دِڪّ، بَ ڪِ جࣴمَّلْ ندࣴنگّتࣹ تࣹ دࣴگْمَلْ اَلْڪُتࣹ¤، فࣹيݧْ؞ ‏4 ڪࣷوڪُ مࣷويْ جࣵنڪࣷونْتࣹيڪَ دَمُ ݖِ ڪَوْ مبࣷولࣹمْ لࣹيسْ دِ وࣷويࣹ يࣵلَّ، مبَا لࣹيسْ دِ جَامُ، تࣹ مࣷويْ مُجّ تࣷوگْ ݖَ ڪࣴرْ يࣵلَّ گَ، بَ فَرْ تُدّࣹ بࣷݒَّمْ يࣵلَّ؞
‏5 خَنَا فࣵتّلِڪُوُلࣹينْ نࣹ وَخࣷونْ نَا لࣹينْ ڪࣷ، بَ مَ نࣹڪّࣹي اَڪْ يࣹينْ؟ ‏6 تࣹ خَمْ نگࣹينْ لِ ڪࣷ تࣺيࣹ لࣺيگِ، بَ دُ مَنَ فࣹيݧْ لُ دُلْ ݖَ بࣺسَمْ؞ ‏7 ڪُمْݒَ لَ دࣷولࣹيْ ندࣴنگّتࣹ جِ بࣵدّࣷو تࣹ مُ نگِ دࣹفْ لِگّࣺييَمْ خَاتْ، خَنَا ڪَيْ ڪِ ڪࣷ تࣺيࣹ بَ تࣹيْ مࣷو ڪࣷيْ تࣺيࣹ رࣹڪّ بَ ڪࣹرࣷوگْ جࣴلࣹيسْ فِ ڪِ ڪࣷ تࣺيࣹ؞ ‏8 سُ بࣷوبَا لَ ڪُ دࣴنگّ ڪَ دِ فࣹيݧْ، تࣹ سَنگّ عِيسَىٰ مࣷو ڪࣷيْ جࣵلَّرْبࣹي نگࣹلَوَلْ گࣺمِّݧَمْ، تࣹ مࣷو ڪࣷيْ رَجَخࣹي ݖَ نَ مُيْ فࣹيݧࣹ بُ دࣺلْسࣹي؞ ‏9 ݒࣹيݧُمْ ڪُ دࣴنگّ ڪِ لِگّࣺييُ سࣹيْتَانࣹ لَيْ اࣵنْدَلْ، تࣹ دِنَ وࣷنࣹ وࣺݒّ خࣹيتُ مَنࣷورࣹ، اَڪِ فِرْندࣹ اَڪْ ڪࣺيمَانِ فࣹنِ نࣹينْ، ‏10 اَڪْ مبࣷولࣹمْ خࣹيتِ نَخࣹ يُ نجُبَدِ لَلْ، ݧࣹيلْ ݧِ نَرَ سࣵنڪُ ندَخْ ݧࣵڪَّ سࣷݒّ دࣴگّ گِ، نگِرْ نَنگُ ڪࣷ بَ مُݖّ؞ ‏11 لࣷولࣷو تَخْ يࣵلَّ يٗنّࣹي لࣹينْ جࣴفْ جُيْ سࣵنڪࣹ، نگِرْ ݧُ گࣴمْ اَوْ فࣹنْ؞ ‏12 سُ ڪࣷ دࣹفࣹي مبُگَلْ دَلْ مبࣷولࣹمْ ݧِ گࣴمُلْ دࣴگّ گِ، خَنَا دِ خࣹجّࣷو نجُبَدِ؞
تَخَوْلࣹينْ بُ دࣴگࣴرْ
‏13 نُنْ نَگْ بࣷڪّ يِ، مَنُنࣷو ݧࣵڪَّ سَنتّ يࣵلَّ فُ نُ تࣷلُّ، ݖِ يࣹينْ ݧِ بࣷرࣷومْ بِ سࣷݒّ، ندَخْ يࣹينْ لَ جࣴڪَّ تَامُ، نگِرْ سࣹلَّلࣹ لࣹينْ نࣷو گُ سࣹلّ گِ، اَڪْ اِتَمْ سࣹينْ نگࣴمْ گِ نگࣹينْ گࣴمْ دࣴگّ گِ، نگِرْ تࣹخࣹيلْ لࣹينْ؞ ‏14 ݖِ لࣷولُ اِتْ لَ لࣹينْ وࣷو بَ مُ جࣴفَنْدِڪࣷو خِبَارُ جࣵمّ بِ نُ لࣹينْ اࣵگّࣹ، نگِرْ نگࣹينْ سࣺدُّ ݖِ سُنُ ندَمَلْ بࣷرࣷومْ عِيسَىٰ اَلْمَسِ؞ ‏15 ڪࣷنْ نَگْ بࣷڪّ يِ، تَخَوْلࣹينْ بُ دࣴگࣴرْ تࣹ ݝࣷيْ ݖِ دࣺنڪَانࣹ يِ نُ لࣹينْ اࣵگّࣹ ݖِ سُنُ لࣵمِّݧُ بࣷݒّ، اَڪْ يِ نُ لࣹينْ بِندّ ݖِ سُنُيْ بَتَاخَلْ؞
‏16 سُنُ سَنگّ عِيسَىٰ اَلْمَسِ ݖِ بࣷݒَّمْ، اَڪْ يࣵلَّ سُنُ بَايْ، مِ نُ سࣷݒّ، تࣹ مَيْ نُ ݖِوْ يِوَمْ، جࣵمِّ خࣷلْ جُ سَخْ دࣵڪّ، اَڪْ يَاڪَارْ جُ وࣺرْ، ‏17 يَلْ نَݧُ لࣹينْ مَيْ جࣵمِّ خࣷلْ، تࣹ مَيْ لࣹينْ دࣷولࣹيْ وࣺيࣹ جࣺݒّ جࣴفْ جُ بَاخْ، اَڪْ گࣺݒّ ڪࣵدُّ گُ رَفࣹتْ؞
