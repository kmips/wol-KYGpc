يࣵلَّ مُسَلْ نَ نُوحٍ ݖِ مبࣴنّ مِ
7
‏1 گَنَّاوْ بَ لࣷولُ اَمࣹي اَجِ سَخْ جِ دَفَ وَخْ نُوحٍ نࣹ ڪࣷ:‏ «دُگَّلْ ݖِ گَالْ گُ مَگْ گِ، يَاڪْ سَ نجَبࣷوتْ گࣺݒّ، ندَخْتࣹ نِتِ جَمࣷنࣷ جِي، يَوْ رࣹڪّ لاَ ݖِ گِسْ، نگَ جُبْ نِ مَ ڪࣷ بࣴگّࣹ؞ ‏2 خࣹيتِ رَبْ يُ سࣹتْ¤ يࣺݒّ يٗبَّالࣹ ݖِ جُرٗومْ ݧَارْ يُ گٗورْ اَڪْ جُرٗومْ ݧَارْ يُ جِگࣺينْ، اَڪْ ݧَارْ ݖِ خࣹيتِ رَبْ يِ سࣹتُلْ، تࣹ ݧُيْ گٗورْ اَڪْ جِگࣺينْ؞ ‏3 تࣹ اِتْ خࣹيتُ ݒِݖّ مُ نࣹڪّ يٗبَّالࣹ ݖِ جُرٗومْ ݧَارْ يُ گٗورْ اَڪْ جُرٗومْ ݧَارْ يُ جِگࣺينْ، نگِرْ سࣹينْ خࣹيتْ بَݧَ فࣹيْ ݖِ ڪَوْ سُوفْ؞ ‏4 ندَخْ ڪَتْ لِ فࣹيڪْ جُرٗومْ ݧَارِ فَنْ دِنَا وࣵݖّࣹ تَوْ ݖِ ڪَوْ سُوفْ دِيرُ ݧࣹينتّ فُڪِّ گُدِّ اَڪْ ݧࣹينتّ فُڪِّ بࣴݖّࣴگْ، بَ فَرْ ݖِ اࣵدِّنَ مبࣷولࣹمْ مبِنْدࣹيفْ مُ مَ سࣵڪّ؞» ‏5 نُوحٍ دَلْدِ دࣹفْ لَ ڪࣷ اَجِ سَخْ جِ سَنتّ لࣺݒّ؞
‏6 بِ مبࣴنّ مِيْ ݧࣴوْ ݖِ اࣵدِّنَ، بࣷوبَ فࣹڪّ نَ نُوحٍ اَمْ جُرٗومْ بࣹنِّ تࣺيمࣺيرِ اَتْ؞ ‏7 نُوحٍ دُگّ نَ ݖَ گَالْ گَ، اࣵندّ اَڪْ اَيْ دࣷومَمْ اَڪْ سࣷخْنَامْ اَڪْ سࣷخْنَيْ دࣷومَمْ يَ، نگِرْ مُݖّ ݖَ ندࣷخُ مبࣴنّ مَ؞ ‏8 رَبْ يُ سࣹتْ يِ اَڪْ يِ سࣹتُلْ اَڪْ ݒِݖّ يِ اَڪْ يِيْ رَامْ ‏9 اࣵندّ نَݧُ اَڪْ نُوحٍ، دُگّ ݖَ گَالْ گَ، دࣹفْ ݧَارْ-ݧَارْ، گٗورْ اَڪْ جِگࣺينْ، نِ ڪࣷ يࣵلَّ سَنْتࣹ نُوحٍ؞ ‏10 بَ جُرٗومْ ݧَارِ فَنْ وࣺيࣹي نَگْ، ندࣷخُ مبࣴنّ مَ اَگْسِ ݖِ ڪَوْ سُوفْ؞
‏11 اَتْ مَ نُوحٍ اَمࣹي جُرٗومْ بࣹنِّ تࣺيمࣺيرِ اَتْ، ݖَ فُڪِّ فَنْ يَ اَڪْ جُرٗومْ ݧَارْ ݖَ ݧَارࣹيلُ وࣹيرْ وَ، ڪࣹرَ لَ بࣴتِ ندࣷخْ يَ نࣹ جࣵيّ، جٗگࣹ خٗوتࣹيْ گࣺيجْ، بُنْتِ ندࣷخِ اَسَمَانْ نࣹ ڪُلْبࣺتْ، مُ نࣹ يُرࣹيتْ، ‏12 تَوْ بِيْ سࣷتِّڪُ ݖِ سُوفْ دِيرُ ݧࣹينتّ فُڪِّ گُدِّ اَڪْ ݧࣹينتّ فُڪِّ بࣴݖّࣴگْ؞
‏13 ڪࣹرࣷوگْ بࣺسْ بࣷوبُ لَ نُوحٍ اࣵندّ اَڪِ دࣷومَمْ يُ گٗورْ، سࣹمْ اَڪْ خَمْ اَڪْ يَفࣹتْ، اَڪْ سࣷخْنَامْ اَڪْ ݧࣹتِّ سࣷخْنَيْ دࣷومَمْ يࣷويُ، ݧُ دُگّ ݖَ گَالْ گَ، ‏14 اࣵنْدَنْدࣷوڪْ رَبُ اࣵلّ يࣺݒّ، لُ ݖِ نࣹڪّ اَڪْ وِرْگࣷومْ، اَڪْ جُرْ گࣺݒّ، لُ ݖِ نࣹڪّ اَڪْ وِرْگࣷومْ، اَڪْ لࣺݒّ لُيْ رَامْ ݖِ ڪَوْ سُوفْ، لُ ݖِ نࣹڪّ اَڪْ وِرْگࣷومْ، اَڪْ لࣺݒّ لُيْ نَاوْ، لُ ݖِ نࣹڪّ اَڪْ وِرْگࣷومْ، مُيْ ݒِݖّ مبَا بࣷرࣷومِ لاَفْ؞ ‏15 لࣺݒّ لُيْ بࣷرࣷومْ بَڪَّنْ دࣹفْ نَ ݧَارْ-ݧَارْ، اࣵندّ اَڪْ نُوحٍ، دُگّ ݖَ گَالْ گَ، ‏16 مبࣷولࣹمْ خࣹيتُ مبِنْدࣹيفْ رࣹڪّ، دِ گٗورْ اَڪْ جِگࣺينْ، نِ ڪࣷ ڪࣷ يࣵلَّ سَنْتࣹ؞ ݖِ ڪَوْ لࣷولُ اَجِ سَخْ جِ نࣹ رࣵݒّ بُنْتُ گَالْ گِ، گَنَّاوْ بِ نُوحٍ دُگّࣹي؞
‏17 مبࣴنّ مَ نَگْ دِ وَلْ ݖِ اࣵدِّنَ دِيرُ ݧࣹينتّ فُڪِّ فَنْ؞ ندࣷخْ يَ دِ يࣷڪُّ، گَالْ گَيْ جٗگْ، بَ تࣴمبّ، تِيمْ سُوفْ لُ سࣷرࣹ؞ ‏18 مَ نگَيْ گࣴنْ دِ يࣷڪُّ، تࣹيْ بَاوَانْ ݖِ ڪَوْ سُوفْ، بَ گَالْ گَ تࣴمبّ، دِ دࣹمْ ݖَ ڪَوْ ندࣷخْ مَ؞ ‏19 ندࣷخْ مَ دِ وَلَنگَانْ، بَ مࣴدّ تُندّ يُ ڪَوࣹ يَ فࣺݒّ فُ اَسَمَانْ تِيمْ؞ ‏20 مُيْ يࣷڪُّ، دِ يࣷڪُّ، بَ گࣴنَ ڪَوࣹ تُندّ يَ لُ مَتْ فُڪِّ خَسَبْ اَڪْ جُرٗومْ؞
‏21 مبࣷولࣹمْ مبِنْدࣹيفْ مُ دَانْ دࣷخْ ݖِ ڪَوْ سُوفْ دࣹي نَ ݖَ؞ ݒِݖّ يِ اَڪْ جُرْ گِ اَڪْ رَبِ اࣵلّ يِ اَڪْ لࣺݒّ لُيْ رَامْ ݖِ ڪَوْ سُوفْ دࣹي نَ، ݧࣷومْ اَڪْ نِتْ ݧࣺݒّ؞ ‏22 مبࣷولࣹمْ مبِنْدࣹيفْ مُيْ بࣷرࣷومْ بَڪَّنْ ݖِ ڪَوْ سُوفَ بࣷڪّ دࣹي؞ ‏23 لُيْ دُندّ ݖِ اࣵدِّنَ دَلْدِ سࣵنڪُ، بَ نِتْ اَڪْ جُرْ اَڪْ لُيْ رَامْ اَڪْ لُيْ نَاوْ— لࣺݒّ لَ اَجِ سَخْ جِ رَافَلْ ݖِ اࣵدِّنَ؞ نُوحٍ رࣹڪَّ دࣹسْ، مࣷومْ اَڪْ ݧَ اࣵنْدࣷونْ اَڪْ مࣷومْ ݖَ گَالْ گَ؞
‏24 ندࣷخْ مِ نَگْ دِ وَلْ-وَلِ ݖِ ڪَوْ سُوفْ دِيرُبْ تࣺيمࣺيرِ فَنْ اَڪْ جُرٗومْ فُڪّ؞
