مَگِ يُوسُفَ يَ دࣹمْ نَݧُ مِسْرَ
42
‏1 بَ يَعْقُوبَ يࣺگࣹي نࣹ ݒࣹݒّ اَمْ نَ مِسْرَ، دَا وَخْ اَيْ دࣷومَمْ يُ گٗورْ، نࣹ لࣹينْ:‏ «لُ نگࣹينْ دِ تࣷوگْ، دِ خࣷولَنْتࣹ رࣹڪّ؟» ‏2 مُ نࣹ لࣹينْ:‏ «دࣺگّ نَا نࣹ ݒࣹݒّ اَمْ نَ ݖَ مِسْرَ؞ دࣹمْلࣹينْ جࣴنْدَلِ نُ ݖَ، ندَخْ نُ مَنَ دُندّ، بَ خِيفْ بَݧْ نࣷو رࣹيْ ڪَيْ!‏»
‏3 ݖِ ڪَوْ لࣷولُ فُڪِّ مَگِ يُوسُفَ يَ دࣹمْ مِسْرَ، جࣴنْدِ ݒࣹݒّ؞ ‏4 وَايࣹ يَعْقُوبَ يࣹبَلُلْ بࣹݧَمِنْ، رَڪُّ يُوسُفَ، ندَخْ دَفَ رَگَلْ نࣹ مُسِبَ مَنْ نَ ڪࣷو دَبْ؞ ‏5 دࣷومِ إِسْرَاءِيلْ ݧࣴوْ نَگْ ݖِ بِيرْ ݧِيْ جࣴندّسِ ݒࣹݒّ، ندَخْ خِيفْ بِ لَوࣷونْ نَ بَ ڪَنَانْ؞ ‏6 فࣹڪّ بࣷوبَ يُوسُفَ مࣷويْ دࣷگَلْ ݖَ رࣺيوْ مَ، دِ جَايْ ݒࣹݒّ وَا رࣺيوْ مࣺݒّ؞ مَگِ يُوسُفَ يَ ݧࣴوْ، سُجّٗوتَلْ يُوسُفَ؞
يُوسُفَ خُݒّ نَ مَگَمْ يَ
‏7 نَڪَ لَ يُوسُفَ گِسْ مَگَمْ يَ، خࣵمِّ لࣹينْ، تࣹ مࣹلْ نِ ڪُ لࣹينْ خَمُلْ، دِ وَخْ اَڪْ ݧࣷومْ ڪࣵدُّ يُ دࣴگࣴرْ نࣹ لࣹينْ:‏ «فُ نگࣹينْ جٗگࣹ؟» ݧُ نࣹ ڪࣷ:‏ «رࣺيوُ ڪَنَانْ لَݧُ جٗگࣹ، دِ جࣴندّسِ اَبْ دُندّ؞» ‏8 يُوسُفَ خࣵمِّ نَ اَيْ مَگَمْ، وَايࣹ ݧࣷومْ خࣵمِّوُݧُ ڪࣷ؞ ‏9 خࣹلُمْ يُوسُفَ نࣹ يَرّ ݖَ لَ مُ گࣺنْتࣷونْ ݖَ ݧࣷومْ، دَلْدِ نࣹ لࣹينْ:‏ «يࣹينْ اَيْ يࣴدُّڪَتْ نگࣹينْ، يُيْ نࣹمِّڪُسِ مبٗوتِ رࣺيوْ مِ؞» ‏10 ݧُ نࣹ ڪࣷ:‏ «مُڪّ سَنگّ بِ، دَنُ لَيْ سِيَارࣹ، دِ وُتْسِ اَبْ دُندّ رࣹڪّ، ‏11 تࣹ نُنْ ݧࣺݒَّ بࣷڪّ بَايْ، دِ نِتْ ݧُ مَانْدُ؛ نُنْ ڪَيْ، سَنگّ بِ، دُنُ اَيْ يࣴدُّڪَتْ؞» ‏12 مُ نࣹ لࣹينْ:‏ «اَهَڪَيْ، نࣹمِّڪُسِ مبٗوتِ رࣺيوْ مِ مࣷو لࣹينْ تَخَ جٗگْ؞» ‏13 ݧُ نࣹ ڪࣷ:‏ «نُنْ ݧِي دِ لَ سِيَارࣹ دࣹ، فُڪِّ گٗورْ اَڪْ ݧَارْ لَنُ وࣷونْ، بࣷڪّ سُنُ بَايْ، بَ ݖَ ڪَنَانْ؞ ڪَ ݖَ گࣴنْ دِ خَلࣹي نگَ ݖَ سُنُ بَايْ، تࣹ ڪَ ݖَ دࣹسْ نࣹڪَّتُلْ؞» ‏14 يُوسُفَ نࣹ لࣹينْ:‏ «لِ مَ لࣹينْ وَخْ مࣷو اَمْ، اَيْ يࣴدُّڪَتْ دࣷݝّ نگࣹينْ؞ ‏15 وَايࣹ نِي لاَ لࣹينْ دِ سࣹيتْلࣷو:‏ بُ سࣹينْ رَڪّ ݧࣴوُلْ، اَڪْ بَڪَّنُ فِرْعَوْنَ، دُنگࣹينْ فِ جٗگࣹ؞ ‏16 نَ ڪࣹنّ ݖِ يࣹينْ جࣴلِ سࣹينْ رَڪّ، ݧُ تࣴجْ ݧِ ݖِ دࣹسْ تࣹ تࣷݒّ سࣹينْ وَخْ، بَ خَمْ ندَخْ يࣹينَ نگِ ݖِ دࣴگّ؞ سُ دُلْ دࣴگّ، گِݧْ نَا ݖِ فِرْعَوْنَ نࣹ اَيْ يࣴدُّڪَتْ دࣷݝّ نگࣹينْ؞» ‏17 مُ تࣴجْ لࣹينْ ݧࣹتِّ فَنْ؞
‏18 ݖَ گَنَّاوْ اࣴلّࣴگْ سَ يُوسُفَ نࣹ لࣹينْ:‏ «بُ نگࣹينْ دࣹفࣹي لِي مَيْ وَخْ، دِنگࣹينْ دُندّ، ندَخْ مَنْ رَگَلْ نَا يࣵلَّ؞ ‏19 ندࣹگَمْ ݧُ مَانْدُ نگࣹينْ، نَ لࣹينْ ڪࣹنّ ݖِ سࣹينْ دࣷومِ بَايْ وُوتُ فِي، ݧِ ݖِ دࣹسْ دࣹمْ يٗبُّلْ ݒࣹݒّ سࣹينْ وَا ڪࣴرْ، يِ نࣹڪّ ݖِ خِيفْ؞ ‏20 وَايࣹ نَنگࣹينْ مَ اِنْدِلْ سࣹينْ رَڪّ، بَ مَ خَمْ نࣹ لِ نگࣹينْ وَخْ دࣴگّ لَ؛ سُ ڪࣷ دࣹفࣹي دُنگࣹينْ دࣹي؞» ݧُ دَلْدِ نࣹ:‏ «وَاوْ،» ‏21 تࣹ نَا ݖِ سࣹينْ بِيرْ:‏ «ݖࣺيْ سُ نُ يࣺگࣷونْ، سُنُ اࣵقُ رَڪَّ نُ تࣷݒّ نِي، ندَخْ دَنُ ڪࣷو سࣹيتَانࣷونْ، مُ نࣹڪّ ݖِ نجࣵقَرࣹيْ خࣷلْ، دِ نُ تِنُ، تࣹ فَالࣹوُنُ ڪࣷ؛ لࣷولࣷو وَرَلْ سُنُ نجࣵقَرࣹيْ تࣹيْ جِي؞» ‏22 رُبࣹنْ نَگْ اࣵدُّ نࣹ لࣹينْ:‏ «ندَخْ نࣹوُمَ لࣹينْ وࣷونْ:‏ ‹بُلࣹينْ گࣵدُّ بَڪَّنُ خَلࣹ بِ،› وَايࣹ فَالࣹوُلࣹينْ مَ؛ تࣹيْ رࣹڪّ لَ نُ يࣵلَّ دࣷونْ خَارࣹ بَڪَّنَمْ، بِ نُ گࣵدُّ؞» ‏23 بَ ݧُيْ وَخْ لࣷولُ، فࣷوگࣷونْ نَݧُ نࣹ يُوسُفَ دࣺگُّلْ سࣹينْ لࣵڪّ، ندَخْ لࣵݒّتࣷ، بَ دࣷخࣷونْ سࣹينْ دِگَّنْتࣹ، دِ تࣹڪِّ؞
‏24 ݖِ بِيرْ لࣷولُ مُ دُمّٗويُ لࣹينْ، دِ جࣷويْ؞
دࣷومِ يَعْقُوبَ يَ دࣹلُّ نَݧُ ڪَنَانْ
بَ مُ جࣷويࣹي بَ نࣷݒِّ، وَلْبَتِڪُ، وَخْ اَڪْ ݧࣷومْ، دࣷورَ جࣴلْ سِمࣹيࣷنْ، جࣺنگّ ڪࣷ ݖِ سࣹينْ ڪَنَمْ؞
‏25 گَنَّاوْ لࣷولُ يُوسُفَ جࣷخࣹ ندِگَلْ، ݧُ سࣷلْ سࣹينِ سَاڪُ ݒࣹݒّ، تࣹ ڪُ نࣹڪّ ݖَ ݧࣷومْ، ݧُ يࣹبَلْ ڪࣷ خَالِسَمْ ݖَ سَاڪࣷومْ، بَ نࣷݒِّ مَيَالࣹ لࣹينْ لُ ݧُ لࣹڪّ ݖَ يࣷونْ وَ؞ ݧُ دࣹفَلْ لࣹينْ نࣷونُ؞ ‏26 بَ مُ ڪࣷ دࣹفࣹي ݧُ سࣴفْ سࣹينِ مبَامْ ݒࣹݒّ مَ، دَلْدِ فَيْ بࣵيِّڪࣷو؞
‏27 فَ ݧُ دَلْ ݖَ گُدِّ، ڪࣹنّ ݖِ ݧࣷومْ بࣴگَّ خࣷنتّ مبَامَمْ، اُبِّ سَاڪࣷومْ رࣹڪّ، نࣹ ݒࣹمّ ݖِ خَالِسَمْ؞ ‏28 مُ نࣹ دࣷومِ بَايْ يَ:‏ «دࣹلّࣷو نَݧُ مَ سَمَ خَالِسْ، مُ نگِ نِي ݖِ سَمَ سَاڪُ؞» سࣹينْ فِتْ نࣹ تࣴرِتْ، ݧُيْ رَگْ-رَگِ، ڪُ نࣹ خࣷولْ سَ مࣷرࣷومْ، نَانْ:‏ «لُ نُ يࣵلَّ نَتّࣷو نِي؟»
‏29 بَ ݧُ يࣹگّࣹي ݖَ سࣹينْ بَايْ يَعْقُوبَ ݖَ رࣺيوُ ڪَنَانْ، دَلْدِ ڪࣷيْ نࣹتَّلِ لِ لࣹينْ دَلْ لࣺݒّ، نࣹ ڪࣷ:‏ ‏30 «وَا جَ يِلِفْ رࣺيوْ مَ دَفَ نࣷو رࣺسࣴتِ، فࣷوگْ نࣹ دَنُيْ يࣴدُّسِ رࣺيوْ مَ؞ ‏31 نُ نࣹ ڪࣷ:‏ ‹نُنْ اَيْ نِتْ ݧُ مَانْدُ لَنُ، دُنُ اَيْ يࣴدُّڪَتْ؛ ‏32 فُڪِّ گٗورْ اَڪْ ݧَارْ لَنُ وࣷونْ يُ بࣷڪّ سُنُ بَايْ، وَايࣹ ڪࣹنّ ڪَ نࣹڪَّتُلْ، تࣹ ڪَ گࣴنْ دِ خَلࣹي نگَ فَ سُنُ بَايْ، بَ ݖَ رࣺيوُ ڪَنَانْ؞› ‏33 وَا جَ يِلِفْ رࣺيوْ مَ نࣹ نُ:‏ ‹نِ مَيْ خَمࣹ نࣹ ݧُ مَانْدُ نگࣹينْ، نِي لَ:‏ دَنگࣹينْ مَيْ بࣵيّࣹيڪْ ڪࣹنّ ݖِ يࣹينْ، تࣹ دࣹمْ يٗبُّلْ دُندّ سࣹينْ وَا ڪࣴرْ، يِ نࣹڪّ ݖِ خِيفْ؛ ‏34 سُ ڪࣷ دࣹفࣹي نگࣹينْ اِنْدِلْ مَ سࣹينْ رَڪّ جُ ندَوْ، ڪࣷنْ دِنَا خَمْ نࣹ دُنگࣹينْ اَيْ يࣴدُّڪَتْ وَايࣹ نِتْ ݧُ مَانْدُ نگࣹينْ، تࣹ دِنَا لࣹينْ دࣹلّࣷو سࣹينْ مبࣷڪّ، نگࣹينْ مَنَ دࣹمْ فُ لࣹينْ نࣹيخْ ݖِ رࣺيوْ مِ؞›»
‏35 نَڪَ لَ ڪُ نࣹڪّ سࣷتِّ سَاڪࣷومْ، نࣹ جࣵڪّ ݖَ مبُوسُ خَالِسَمْ؞ بَ ݧُ گِسࣹي مبُوسِ خَالِسْ يَ، ݧࣷومْ اَڪْ سࣹينْ بَايْ، دَݧࣷو تِيتْ؞ ‏36 سࣹينْ بَايْ يَعْقُوبَ نࣹ لࣹينْ:‏ «خَݧْ نگࣹينْ مَ اَيْ دࣷومْ:‏ يُوسُفَ نࣹڪَّتُلْ، سِمࣹيࣷنْ نࣹڪَّتُلْ، لࣺيگِ نگࣹينْ بࣴگَّ جࣴلْ بࣹݧَمِنْ، يٗبُّ؞ لࣷولُ لࣺݒّ ݖِ سَمَ ڪَوْ؛ نگَلَّ مَنْ!‏» ‏37 رُبࣹنْ وَخْ اَڪْ بَايَمْ نࣹ ڪࣷ:‏ «سُ مَ لَ ڪࣷ دࣺلّࣷوسِلُلْ، بࣷو يࣹبࣷو رࣹيَلْ سَمَ ݧَارِ دࣷومْ يُ گٗورْ يِ؞ تࣹگَلْ بࣹݧَمِنْ ݖِ سَمَ لࣷخࣷ رࣹڪّ، مَنْ دِنَا لَ ڪࣷ دࣺلّࣷوسِلْ؞» ‏38 ݖِ ڪَوْ لࣷولُ يَعْقُوبَ نࣹ:‏ «سَمَ دࣷومْ دَالْ، دُ اࣵندّ اَڪْ يࣹينْ، ندَخْ مَگَمْ فَاتُ نَ، تࣹ مࣷومْ رࣹڪَّ دࣹسْ؞ سُ ڪࣷ مُسِبَ دَبࣹي ݖِ يࣷونْ وِ نگࣹينْ جࣴمْ، دِنگࣹينْ مَ يٗبّࣹ نَقَرْ وُ مَيْ جࣵݒّ، بَ ڪࣹرَ مَيْ اࣵندّ اَڪْ سَمَ بِجَّاوْ، تࣵبِّ نجَنِيوْ؞»
