ݒٗولْ جࣵنگّلࣹ نَ ݖَ اࣹفࣹسْ
19
‏1 دِيرْ بَ اَݒࣷلࣷسْ نࣹڪّࣹي ڪࣷرࣹنتّ، ݒٗولَ نگَ جَارࣹ ݖَ دِيوَانْ يُ تُنْدࣹ يَ، بَ وࣵݖّسِ اࣹفࣹسْ؞ مُ دَجࣹ فَاڪَيْ تَالِبࣹ، ‏2 نࣹ لࣹينْ:‏ «ندَخْ بَ نگࣹينْ گࣴمࣹي، جࣷتْ نگࣹينْ ݖِ نࣷو گُ سࣹلّ گِ؟» ݧُ نࣹ ڪࣷ:‏ «نُنْ دࣹ يࣺگُنُ وࣷونْ سَخْ نࣹ نࣷو گُ سࣹلّ گِ اَمْ نَ؞» ‏3 مُ نࣹ لࣹينْ:‏ «ڪࣷنْ گَنْ سٗوبُ نگࣹينْ سٗوبُ؟» ݧُ نࣹ ڪࣷ:‏ «خَنَا سٗوبࣹبْ يَحْيَىٰ¤؞» ‏4 ݒٗولْ نࣹ لࣹينْ:‏ «يَحْيَىٰ، سٗوبࣹبْ تُوبࣹيلْ لَ دَانْ سٗوبࣹ ݖِ ندࣷخْ، دِ وَخَالࣹ اَسْڪَنْ وِ نࣹ لࣹينْ ݧُ گࣴمْ ڪِيْ تࣷݒّ ݖِ مࣷومْ، تࣹ مࣷويْ عِيسَىٰ؞» ‏5 بَ ݧُ دࣺگّࣹي لࣷولُ لَݧُ سٗوبُ ݖِ ندࣷخْ ݖِ تُرُ سَنگّ عِيسَىٰ؞ ‏6 بَ لࣹينْ ݒٗولْ تࣹگࣹي لࣷخࣷومْ، نࣷو گُ سࣹلّ گِ دِڪَّلْ لࣹينْ، ݧُ تࣵمْبَلࣹي لࣵڪّ يࣹنࣹينْ لࣵڪّ، بࣷولࣹ ݖِ دِ بِرَلْ وَخْيُ؞ ‏7 تَالِبࣹ يࣷويُ يࣺݒَّ نگِ وَرَ تࣷلُّ ݖِ فُڪّ اَڪْ ݧَارْ؞
‏8 ݖِ ڪَوْ لࣷولُ ݒٗولْ دࣹمْ ݖَ جࣵنگُ بَ، دِيرُ ݧࣹتِّ وࣹيرْ مُيْ وَارࣹي اَوْ فِتْ، دِ وَخَلْ وَا جࣵنگُ بَ ڪࣵدُّ يُ مَتَ گࣴمْ ݖِ مبِرُمْ نگُورُگْ يࣵلَّ؞ ‏9 ݧࣹنّ ݧَ نَگْ تࣴ تِݖّ، بَݧَ گࣴمْ، دِ ݝࣵݧّ يࣷونْ وِ ݖَ ڪَنَمْ مبࣷولࣷو مَ؞ بَ لࣷولُ اَمࣹي ݒٗولْ بࣵيِّ لࣹينْ، دَلْدِ بࣹرࣷوڪْ تَالِبࣹ يَ، بࣺسْ بُ نࣹڪّ مُيْ دِيسࣷو اَڪْ ݧࣷومْ ݖَ ڪࣴرُگْ جࣵنگُڪَايْ گُ تِرَنُسْ؞ ‏10 لࣷولُ مࣷو وࣺيْ دِيرُ ݧَارِ اَتْ، بَ مبࣷولࣹمْ وَا دِيوَانُ اَسِ دࣺگّ ڪࣵدُّگْ سَنگّ بِ، مُيْ يَوُتْ، دِ گࣹرࣹگْ؞
وَا اࣹفࣹسْ وَارُ نَݧُ
‏11 بَ مُ ڪࣷ دࣹفࣹي يࣵلَّ دِ دࣹفْ اَيْ ڪࣺيمَانْ يُ مَگْ يُ مُ سࣷتَّلࣹ لࣷخࣷيْ ݒٗولْ؞ ‏12 مُجّ نَ سَخْ ݧُيْ يٗبُّلْ جَرَگْ يِ اَيْ ڪَالَ اَڪِ تَرَخْلاَيْ يُ لاَلࣷونْ يَرَمُ ݒٗولْ، بَ لُ لࣹينْ دَلࣷونْ تࣹقْلِڪࣷوڪْ ݧࣷومْ، رَبْ وُ لࣹينْ جࣵݒّࣷونْ، بَ لࣹينْ؞
‏13 ݖِ ڪَوْ لࣷولُ اَيْ يَوُتْ يُيْ وࣴرْ دِ فَجْ ڪُ رَبْ جࣵݒّ، تࣵمْبَلࣹي رࣷيْ، دِ تُدَّلْ ݧِ رَبْ جࣵݒّ تُرُ سَنگّ عِيسَىٰ، نَانْ رَبْ يِ:‏ «مَا لَ دࣵقْ ݖِ تُرُ عِيسَىٰ، مِ ݒٗولْ دِ خَمْلࣹ؞» ‏14 ݧَ دࣷونْ دࣹفْ جࣴفْ جࣷوجُ ݧࣷويْ جُرٗومْ ݧَارِ دࣷومْ يُ گٗورْ يُ سࣹوَ، سَرْخَلْڪَتُبْ يَوُتْ بُ مَگْ؞ ‏15 لࣷولُ لَݧُ دࣷونْ جࣺيمْ بࣺسْ، رَبْ وُ بࣷنْ وَ نࣹ لࣹينْ:‏ «عِيسَىٰ دࣺيْ، خَمْ نَا ڪࣷ؞ ݒٗولْ اِتْ خَمْ نَا ڪࣷ؞ وَايࣹ يࣹينْ، يࣹينَيْ ݧَنْ؟» ‏16 ݖِ ڪَوْ لࣷولُ نِتْ ڪَ رَبْ جࣵݒّ سࣷنگّ لࣹينْ، تࣹگْ لࣹينْ لࣷخࣷ ݧࣷومْ ݧࣺݒّ، ݒَمْتِ-ݒَمْتࣹي لࣹينْ، گَاݧْ لࣹينْ، بَ ݧُ دَوࣹ ڪࣴرْ گَ يَرَمُ نࣹينْ؞
‏17 مبࣷولࣹمْ يَوُتْ يَاڪْ گࣹرࣹگْ يَ دࣴڪّࣹ اࣹفࣹسْ نَگْ يࣺگْ مبِرْ مࣷومُ، تِيتَانگࣹ دِڪَّلْ لࣹينْ ݧࣷومْ ݧࣺݒّ، دَرَجَ جُ رࣺيْ دࣷخࣹ ݖَ ݧࣹيلْ تُرُ سَنگّ عِيسَىٰ؞ ‏18 بَ مُ ڪࣷ دࣹفࣹي ݧُ بَرࣹ ݖَ گࣴمْڪَتْ ݧَ دِڪّ، بِرَلْ سࣹينُگْ تࣷوݧْ، دَلْدِ خَمْلࣹ يُ بࣷنْ يَ ݧُ دَانْ جࣴفࣹ؞ ‏19 ݖِ بِيرْ لࣷولُ ݧُ بَرࣹ ݧُ دَانْ ݧࣹينگࣷ، بࣷولࣹ سࣹينِ تࣺيرࣹ، اِنْدِ، لَڪّ ڪࣷ فَ ڪَنَمْ ݧࣺݒّ؞ بَ ݧُ خَيْمَا نجࣺگْ لَ، تࣷلّࣷو نَاڪْ جُرٗومْ فُڪِّ جُنِّيْ ݒࣷسࣹتِ دَرَخْمَ (50000)؞
‏20 نࣷونُ لَ ڪࣵدُّ گِ لَوࣹ بَ فࣺيتࣹ ڪَوْ ݖِ ڪࣵتَّنُ بࣷرࣷومْ بِ؞
اࣹفࣹسْ يࣴنگُ نَ
‏21 گَنَّاوْ بَ لࣷولُ وࣺيࣹي، ݒٗولْ ݖِ ندِگَلَلْ نࣷو گُ سࣹلّ گِ اَمْ مࣺبࣺتُمْ جَارࣹ دِيوَانِ مَسࣹدُوَنْ اَڪْ اَڪَيِ، تࣹگْ ݖَ يࣹرُسَلࣹمْ؞ مُ نࣹ:‏ «گَنَّاوْ بُ مَ اࣵگّࣹي فࣷوفُ، وَرْ نَمَا جࣵلّ بَ رࣷومْ اِتَمْ؞» ‏22 مُ دَلْدِ يࣹبَلْ فَ مَسࣹدُوَنْ، ݧَارِ جَرَافَمْ، تِمࣷتࣹ اَڪْ اࣹرࣵستّ، مࣷومْ مُ تࣷوگَاتْ اَبْ دِيرْ ݖَ اَسِ؞
‏23 جَنتّ يࣷويُ نَگْ يࣹمࣷوڪْ يࣴنگُ-يࣴنگُ بُ رࣺيْ بُ اَمࣷونْ ݖِ مبِرُمْ يࣷونْ وِ؞ ‏24 ڪُ ݧُيْ وَخْ دࣹمࣹتِرُسْ، اَبْ تࣴگُّ خَالِسْ، مࣷو دَانْ دࣹفَرْ جࣴمِّ جَامࣷوڪَايْ يُ ندَوْ يُ تُورْ مِ ݧُ دِݒّࣹي اَرْتࣹمِسْ، مُيْ نجَرِݧْ لُ رࣺيْ ݖِ لِگّࣺييْڪَتْ يِيْ نَوْلࣹيمْ؞ ‏25 مُ وࣷو لࣹينْ، ݧࣷوڪْ ݧَ فَرࣹوࣷو لِگّࣺييْ يُ نِ مࣹلْ، نࣹ لࣹينْ:‏ «بࣷڪّ يِ، خَمْ نگࣹينْ نࣹ لِگّࣺييْ بِي، ݖِ لَ سُنُ تࣹرَانگَ نࣹڪّ؞ ‏26 يࣹينَ گِسْ نَگْ مبَا نگࣹينْ دࣺگَّلْ سࣹينْ بࣷݒّ نِ ݒٗولْ مࣷومُ دِ نَخࣹي بَ فࣵبِّ مبࣷولࣷو مُ بَرࣹ، تࣹ دُ ݖِ اࣹفࣹسْ گِي رࣹڪّ، دَانَڪَ اَسِ گࣺݒّ لَ:‏ مُ نَايْ:‏ ‹يࣵلَّ يُ لࣷخࣷلْ نِتْ سࣵڪّ دُ يࣵلَّ؞› ‏27 لࣷولُ، جࣺلَّلࣹ نَا نِ مُ مَنَ گࣵڪَّلࣹ سُنُ لِگّࣺييْ بِي، وَايࣹ دِنَ تَخْ ݧُ تࣹدَّدِلْ جَامࣷوڪَايُ اَرْتࣹمِسْ، يࣵلَّ جُ جِگࣺينْ جُ مَگْ جِ، بَ دَرَجَامْ مُجّ نࣹينْ، مࣷومْ مِ مبࣷولࣹمْ اَسِ اَڪْ اࣵدِّنَ سَخْ دِ جَامُ؞»
‏28 مبࣷولࣷو مَ دࣺگّ ڪࣵدُّ يࣷويُ، دَلْدِ مࣹرْ لࣷولْ، دِ خَاݖُ نَانْ:‏ «اَرْتࣹمِسْ مُ اࣹفࣹسْ مࣵگّ نَ!‏» ‏29 ݖِ ڪَوْ لࣷولُ دࣴڪّ بࣺݒّ نࣹ ڪࣴرْ-ڪࣴرْ، نࣹ ڪࣴݒّ؞ ݧُ جࣵݒّ گَيُسْ اَڪْ اَرِسْتࣵرڪّ، وَا مَسࣹدُوَنْ يَ اࣵنْدࣷونْ اَڪْ ݒٗولْ ݖَ تُڪِّ بَ؞ ݧُ دَلْدِ رِيرَنْدࣷو وُتِ اࣴتُّ جࣷݝَنْتࣹڪَايْ بَ؞ ‏30 ݒٗولْ نَگْ بࣴگَّ تࣹيوِ ݖَ ڪَنَمْ مبࣷولࣷو مَ، تَالِبࣹ يَ بَݧْ؞ ‏31 ݧࣹنّ ݧَيْ اَيْ سࣷݒّࣹيمْ ݖَ ڪࣵنگَمِ نگُورْ گَ سَخْ، يٗنّࣹي نَݧُ ݖَ مࣷومْ، دِ ڪࣷ اࣵرْتُ نگِرْ مُ بَݧَ فࣷيࣹ بَڪَّنَمْ، دِ دࣹمْ ݖَ اࣴتُّ جࣷݝَنْتࣹڪَايْ بَ؞ ‏32 ندَجࣹ مَا نگَ رࣴبْ لࣷولْ، ݧِي خَاݖُ نࣹ لِي، ݧࣹي خَاݖُ، نࣹ لࣹي، تࣹ ݧَ ݖَ اࣴݒّ سَخْ خَمُݧُ لُ وَرَلْ ندَجࣹ مَ؞ ‏33 ݖِ ڪَوْ لࣷولُ مبࣷولࣷو مَ اَمْ لُ ݧُ دِگَلْ ڪُ ݧُيْ وَخْ اَلࣹگْسࣵنْدَرْ، يَوُتْ يَ جَݧَخْ ڪࣷ ݖَ ڪَنَمْ، مُ دَلْدِ تࣵلَّلْ لࣷخࣷومْ، نگِرْ لَيࣷولْ بࣷݒَّمْ ݖَ دِگّ مبࣷولࣷو مَ؞ ‏34 نَڪَ لَݧُ خَمْ نࣹ اَبْ يَوُتْ لَ، ڪࣵدُّ گَ دِ بࣹنّ، جِبࣹ ݖَ مبࣷولࣷو مࣺݒّ، دِيرُ ݧَارِ وَخْتُ؞ ݧُ نَانْ:‏ «اَرْتࣹمِسْ مُ اࣹفࣹسْ مࣵگّ نَ!‏»
‏35 ندَوَلْ بُورْ لَ نَگْ مُجّ گِيفَلْ مبࣷولࣷو مَ، نࣹ لࣹينْ:‏ «يࣹينْ وَا اࣹفࣹسْ، اَنَ مࣷوسْ ڪُ خَمُلْ نࣹ اࣹفࣹسْ گِي مࣷويْ دࣴڪّ بِيْ وَتُّڪَتُبْ جَامࣷوڪَايُ اَرْتࣹمِسْ، اَڪْ جࣴمَّمْ جِ وࣵݖّࣹ اَسَمَانْ؟ ‏36 لࣷولُ مَنࣹيسُ ڪࣷو وࣹدِّ؞ ڪࣷنْ نَگْ، دَنگࣹينَ وَرَ تࣹييْ، بَݧَ سَݧَخُيْ دࣹفْ لࣹنّ؞ ‏37 ندَخْ نِتْ ݧِي نگࣹينْ فِ اِنْدِ، تࣹدَّدِلُݧُ اَيْ جَامࣷوڪَايْ تࣹ وَخُݧُ لُ ݧࣵڪّࣹ سُنُ يࣵلَّ ڪࣹرْسَ؞ ‏38 سُ دࣹي دࣹمࣹتِرُسْ اَڪْ لِگّࣺييْڪَتْ يِ اࣵندّ اَڪْ مࣷومْ ݧࣷو جࣷتࣹ اَڪْ ڪࣹنّ ݖِ ݧࣷومْ، اَيْ اࣴتِّ لَيࣷو اَمْ نَ فِ، اَيْ ڪࣵنگَمَ نگِ فِ دِ اࣵتّࣹ؞ نَݧُ فَ لَيࣷوجِ؞ ‏39 سُ دࣹي لࣹنࣹينْ لُ نگࣹينْ دِ سࣵڪُّ نَگْ، ݖَ ندَجࣹمْ اࣵتّࣹ بَ لࣹيسْ ڪࣷيْ لِجَّنْتࣹي؞ ‏40 ندَخْ ڪَتْ رࣹݒّ نَنُ ݧُ جِيݧْ نُ اَگْ فِݒُّ، تࣹ اَمُلْ لࣹنّ لُ نُ مَنَ لَيْ، بَ لࣹوَلْ مِي ندَجࣹ؞» نَڪَ لَ وَخْ لࣷولُ، دَلْدِ تَسْ مبࣷولࣷو مَ؞
