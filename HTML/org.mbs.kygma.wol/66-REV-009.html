9
‏1 جُرٗومࣹيلُ مَلاَڪَ مِ وَلْ لِيتَمْ، مَ گِسْ، بِدِّيوْ بُ خࣵوِّڪࣷو وࣷونْ اَسَمَانْ، بَ وَدّ ݖِ ڪَوْ سُوفْ؞ ݧُ جࣷخْ ڪࣷ ݖَابِيْ ڪࣵمْبُ خٗوتࣹ بَ؞ ‏2 مُ اُبِّ بُنْتُ ڪࣵمْبُ خٗوتࣹ بَ، سَخَرْ سُ مࣹلْ نِ سَخَرَسْ تَالْ بُ مَگْ جࣷلّࣹي ݖَ، جَنتّ بِ اَڪْ جَوّ جِ، لࣺݒّ لࣴنْدࣴمْ ندَخْ سَخَرَسْ ڪࣵمبّ گِ؞ ‏3 بَ لࣷولُ اَمࣹي اَيْ نجࣺيرࣺيرْ گࣺنّࣹ ݖِ سَخَرْ سِ، دَلْدِ تَسَارࣷو ݖِ بِيرْ اࣵدِّنَ، ݧُ مَيْ لࣹينْ نَگْ دَݝَرْ جُ مࣹلْ نِ دَݝَرُ جࣵنڪَلاَرْ؞ ‏4 ݧُ نࣹ لࣹينْ، بُݧُ لاَلْ مࣹنُّمْ ݧَخْ، مبَا گࣹنّ گࣵنݖَخْ، مبَا گࣹنّ گَرَبْ، خَنَا ݧُ جُبْلُ رࣹڪّ ݖِ نِتْ ݧِ اَمُلْ مࣵنْدَرْگَمْ تࣷرْلُوَايُ يࣵلَّ ݖِ سࣹينْ جࣴ؞ ‏5 مَيْ نَݧُ لࣹينْ ݧُ مِتِتَلْ لࣹينْ دِيرُ جُرٗومِ وࣹيرْ، تࣹ بَݧْ لࣹينَ رࣹيْ، خَنَا مِتِتَلْ لࣹينْ لُ تࣹمْبࣷوڪْ مِتِتُ فِتُّ جࣵنڪَلاَرْ؞ ‏6 بࣺسِ ڪࣹرࣷوگْ نِتْ ݧࣹييْ سࣵڪُّ دࣹي، تࣹ دُݧُ ڪࣷ دَجْ، ݧُ نࣹ سِيوَ دࣹي، ندࣹي دَوْ لࣹينْ؞
‏7 نجࣺيرࣺيرْ يَا نگِ نِرࣷوڪْ فَسْ يُ ݧُ وَاجَلْ نگِرْ خَرࣹ، ڪَالَوࣷو لُ مࣹلْ نِ ڪَالَيْ وُرُسْ، سࣹينْ ڪَنَمْ نِرࣷوڪْ ڪَنَمُ نِتْ، ‏8 سࣹينْ ڪَوَرْ مࣹلْ نِ ڪَوَرُ جِگࣺينْ، سࣹينِ سࣹلّ نِرࣷوڪْ سࣹلِّ گَيْندࣹ؞ ‏9 ݧُ نگِ سࣷلْ ڪِيرَايِ دࣴنّ يُ مࣹلْ نِ ڪِيرَايِ وࣹݧْ گُ ݧُولْ، سࣹينْ ݖࣷووَلْ لاَفْ يِ مࣹلْ نِ ݖࣷووَلْ وَتِيرِ خَرࣹ يُ بَرࣹ؞ ‏10 گࣹينْ گُ اَمْ فِتّ لَݧُ اَمْ نِ جࣵنڪَلاَرْ؞ دَݝَرُ گࣹينْ گࣷوگُ لَݧُيْ مِتِتَلࣹ نِتْ ݧِ دِيرُ جُرٗومِ وࣹيرْ؞ ‏11 سࣹينْ بُورَ لࣹينْ جِيتࣹ، دِ مَلاَڪَمْ خٗوتࣹ بِ، تُرَمْ ݖِ اࣹبْرࣴ دِ اَبَدࣷنْ، اَݒࣷلْيࣷنْ ݖِ لࣵڪُّ گࣹرࣹگْ، (مُ فِرِ يࣵقْڪَتْ بِ)؞
‏12 نَڪَ لَ مُسِبَ مُ جࣴڪّ مِ جࣵلّ رࣹڪّ، يࣹنࣹينْ ݧَارِ مُسِبَ دِڪّ، تࣷݒّ ݖِ؞
‏13 بَ مُ ڪࣷ دࣹفࣹي جُرٗومْ بࣹنّࣹيلُ مَلاَڪَ مِ وَلْ لِيتَمْ؞ مَ دࣺگّ بَاتْ بُ جٗگࣹ ݖِ ݧࣹينْتِ بࣺجّࣺنِ سَرْخَلُڪَايُ وُرُسْ، بِ ݖِ ڪَنَمْ يࣵلَّ؞ ‏14 بَاتْ بِ وَخْ اَڪْ جُرٗومْ بࣹنّࣹيلُ مَلاَڪَ مِ يࣷرْ لِيتْ، نࣹ ڪࣷ:‏ «يِوِلْ ݧࣹينْتِ مَلاَڪَ، يَ يࣹيوࣹ ݖَ اࣹفْرَاتْ، دࣹخْ گُ مَگْ گَ؞»
‏15 ݖَ لَݧُ يِوِ ݧࣹينْتِ مَلاَڪَ، يَ ݧُ وَاجَلࣷونْ ݧࣹيلْ وَخْتُ وࣷووُ، ݖِ بࣺسْ بࣷوبُ، وࣹيرْ وࣷووُ، اَتْ مࣷومُ، نگِرْ ݧُ رࣹيْ ݧࣹتّࣹيلُ خَاجُ نِتْ ݧِ؞ ‏16 لِمُ گَوَرِ مبࣷولࣷو مَ دࣷونْ خَرࣹجِ ݧَارْ تࣺيمࣺيرِ جُنِّيْ جُنِّ لَ (200000000)؞ مَا دࣺگّ لِمْ بِ؞
‏17 نِي لاَ گِسࣹ ݖِ ݒࣹيݧُ مِ فَسْ يِ اَڪْ ݧِ لࣹينْ وَرْ:‏ ݧُ نگِ سࣷلْ ڪِيرَايْ يُ خࣷنقّ نِ سَوَرَ، بَخَ نِ سَفِيرْ، تࣹ مبࣷقْ نِ تَمَرَخْ؞ بࣷݒِّ فَسْ يَا نگِ نِرࣷوڪْ بࣷݒِّ گَيْندࣹ، سَوَرَ اَڪْ سَخَرْ اَڪْ تَمَرَخْ دِ گࣺنّࣹ ݖِ سࣹينْ گࣺمِّݧْ؞ ‏18 ݧࣹتِّ مُسِبَ يࣷويِي، سَوَرَ سࣹيڪْ سَخَرْ سࣹيڪْ تَمَرَخْ بِيْ گࣺنّࣹ ݖِ سࣹينْ گࣺمِّݧْ، دَلْدِ رࣹيْ ݧࣹتّࣹيلُ خَاجُ نِتْ ݧِ؞ ‏19 دَݝَرُ فَسْ يَا نگِ ݖِ سࣹينْ گࣺمِّݧْ اَڪْ ݖِ سࣹينْ گࣹينْ، ندَخْ سࣹينْ گࣹينْ يَا نگِ مࣹلْ نِ اَيْ جَانْ، اَمِ بࣷݒّ يُ ݧُيْ مِتِتَلࣹي؞
‏20 ندࣹسُ نِتْ ݧِ مُسِبَ يࣷويِي رࣹيُلْ نَگْ، تَخُلْ سَخْ ݧُ تُوبْ سࣹينِ جࣴفْ، بَ بَݧَتࣹي جَامُ اَيْ تُورْ، اَڪْ جࣴمِّ خࣴرࣴمِ وُرُسْ اَڪْ خَالِسْ اَڪْ خࣵنجَرْ اَڪْ خࣹيرْ اَڪْ دࣺنڪّ؛ خࣴرࣴمْ يُ گِسُلْ، دࣺگُّلْ، دࣷخُلْ؞ ‏21 رَخْ ݖِ دࣷلِّ، تُوبُݧُ بٗومࣹ گِ اَڪْ ݧࣹينگࣷ گِ اَڪْ ݒࣷوُمْ سࣺيْ مࣹيڪْ ݖࣵݖّ گِ ݧُ نࣹڪّࣹ؞
