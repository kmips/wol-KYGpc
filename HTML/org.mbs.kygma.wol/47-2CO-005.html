5
‏1 خَمْ نَنُ نࣹ سُنُ يَرَمْ، سُنُ ڪࣴرُ ڪَوْ سُوفْ گِي دِ اَبْ خَيْمَ، بُ دَانࣷو اِتْ، اَمْ تَاخْ لَنُ اَمْ فَ يࣵلَّ، ڪࣴرْ گُ سَخْ دࣵڪّ، گُ لࣷخࣷلْ نِتْ تَبَخُلْ؞ ‏2 مࣷو نُ تَخْ دِ اࣷنڪّ ݖِ بِيرْ خَيْمَ بِي نُ نࣹڪّ، ندَخْ يࣵڪَّمْتࣹي سࣷلࣷو سُنُ دࣴڪُّوَايُ اَسَمَانْ، تࣹگْ ڪࣷ ݖِ ڪَوْ بِي نُ جِيتَلْ، ‏3 ندࣹ بُ نُ سࣷلࣷو نࣷونُ، دُنُ دࣹفْ يَرَمُ نࣹينْ؞ ‏4 نࣷو نگِ اࣷنڪّ ندَخْ دِيسْ، لِ فࣹيڪْ نُ نگࣹيڪْ سُنُوْ يَرَمْ، خَيْمَ بِ نُ سࣷلࣷو نِي؞ دُ سُمِّڪُ لَنُ نَمّ، وَايࣹ سࣷلُ تࣹگْ ڪࣷ ݖِ لَنُ نَمّ، نگِرْ لِ جࣴمْ ندࣹي، دَلْدِ سُوخْ ݖِ بِيرْ ندُندّ؞ ‏5 گࣷوگُ مُجْ ݖِ بࣷݒَّمْ لَنُ يࣵلَّ مࣷوݧࣹ وࣷونْ، تࣹ نࣷو گِ مُ نُ جࣷخْ لَنُ ݖِ دَوَلَلْ؞
‏6 مࣷو تَخْ نُ سَخࣷو ݧࣹمࣹ، تࣹ خَمْ نࣹ سُنُ دࣹسَگُمْ بِيرْ ڪࣴرْ، ݖِ يَرَمْ وِ، مࣷويْ سُنُ دࣹسَگُمْ فُ سࣷرࣹ سَنگّ بِ، ‏7 ندَخْ نگࣴمْ لَنُيْ دࣷخࣹ، وَايࣹ دࣷخࣹوُنُ اَبْ گِسْ-گِسْ؞ ‏8 ڪࣷنْ ݧࣹمࣹ لَنُ دࣹفْ، تࣹ لِ نُ گࣴنَلْ دِ گࣺنّ، دࣹسْ بِتِبْ سُنُ يَرَمُ تࣹيْ، نگِرْ دࣴڪِّ اَڪْ سَنگّ بِ؞ ‏9 ڪࣷنْ نَگْ، نُ دࣹسْ بِيرْ ڪࣴرْ گِي، مبَا نُ گࣺنّ، دࣴڪِّ بِتِبْ ڪࣴرْ گِي، لِ نُ خِنْتࣹوࣷو دَالْ مࣷو دِ نُ نࣹيخْ سَنگّ بِ؞ ‏10 ندَخْ ڪَتْ، نُنْ ݧࣺݒّ فࣵوّ نُ تَخَوِ فَ جَلُبْ اࣵتّࣹڪَايُ اَلْمَسِ، نگِرْ ڪُ نࣹڪّ يࣷولࣷو ڪࣹمُ جࣴفْ جَ مُ جࣴفࣹ يَرَمَمْ، مُيْ لُ بَاخْ، مبَا لُ بࣷنْ؞
جُبࣷولࣹينْ اَڪْ يࣵلَّ
‏11 دَنࣷو خَمْ لَنْ مࣷويْ رَگَلْ بࣷرࣷومْ بِ نَگْ، تࣹ مࣷو نُ تَخْ دِ وَخْ لُ گࣴمْلࣷو نِتْ ݧِ؞ يࣵلَّ دࣹ رࣵݧّࣹي نَ بُ بَاخْ نࣷويْ ڪَنْ؞ مَيْ يَاڪَارْ نࣹ يࣹينْ اِتَمْ، ݖِ سࣹينْ بِيرْ خࣷلْ، رࣵݧّࣹي نگࣹينْ بُ بَاخْ نࣷويْ ڪَنْ؞ ‏12 سࣹيدࣹيلَتُنُ سُنُ بࣷݒّ لُ جَرْ نگࣴرࣴمْ دࣹ، دَنُ لࣹينْ دِ مَيْ لُ نگࣹينْ نُ مَنَ سَگࣷو، ندَخْ نگࣹينْ اَمْ تࣷنْتَلْ ݧِ سَگࣷووُلْ لِ ݖِ بِيرْ خࣷلْ، خَنَا دِ سَگࣷو اَمْ ݒࣹيݧْ؞ ‏13 ندࣹگَمْ دَنࣷو تࣹقْلِڪࣷو اَڪْ سُنُ سَگࣷ، يࣵلَّ لَ ݧࣹيلْ؞ سُ نُ اࣵنْدࣹي اَڪْ سُنُ سَگࣷ نَگْ، يࣹينْ لَ ݧࣹيلْ؞ ‏14 ندَخْتࣹ ݖࣷفࣹيلُ اَلْمَسِ دَالْ مࣷو نُ جِيتࣹ، نُنْ ݧِ رࣵݧّࣹي لِي:‏ گَنَّاوْ ڪࣹنّ مࣷو دࣹييَلْ ݧࣺݒّ، ڪࣷنْ ݧࣺݒَّ دࣹي؞ ‏15 تࣹ ڪِ دࣹييَلْ ݧࣺݒّ، دَا دࣹييَلْ ݧࣺݒّ نگِرْ ݧِيْ دُندّ بَݧَتࣹي دُندّ نگِرْ سࣹينْ بࣷݒّ، خَنَا دِ دُندّ نگِرْ ڪِ لࣹينْ دࣹييَلْ، بَ دࣹڪِّ؞
‏16 ڪࣷنْ نَگْ گَنَّاوْ-سِ-تࣹيْ، نِ نِتُ سُوخُ نࣹينْ دِ خࣷولࣹ مࣷرࣷومَمْ، نُنْ خࣷولࣹيتُنُ ڪࣷ ڪࣹنّ؞ دࣷونْتࣹ نِ نِتُ سُوخُ نࣹينْ لَنُ خࣷولࣹ وࣷونْ اَلْمَسِ، لࣺيگِ خࣷولࣹيتُنُ ڪࣷ نࣷونُ؞ ‏17 ڪࣷنْ بُ نِتْ ڪِ بࣷڪّࣹي ݖِ اَلْمَسِ، مبِنْدࣹيفْ مُ بࣹيسْ لَيْ دࣷونْ؞ لَ وࣷونْ دَيْ وࣷنِّ، لِ تࣹيوْ نَگْ، دِ لُ يࣹيسْ؞ ‏18 لࣷولُ لࣺݒّ ݖِ يࣵلَّ لَ جٗگࣹ، مࣷومْ مِ نُ جُبَلࣹ اَڪْ بࣷݒَّمْ ݖِ اَلْمَسِ، تࣹ مࣷو نُ سَسْ، نگِرْ نُيْ وَارࣹ جُبࣷو گࣷوگُ؞ ‏19 لࣷولُ دِ فِرِ نࣹ ݖِ اَلْمَسِ لَ يࣵلَّ جُبَلࣹ وَا اࣵدِّنَ، اَڪْ جࣴمُّ بࣷݒَّمْ، بَ تࣷݒُّ لࣹينْ سࣹينِ تࣷوݧْ، تࣹ مࣷو نُ دࣺنڪّ ڪࣵدُّگْ جُبࣷو گِ؞ ‏20 ڪࣷنْ نَگْ اَلْمَسِ، نࣷو دِيْ ندَوَمْ، تࣹ ݖِ نُنْ لَ يࣵلَّ دِ وࣷوتࣹي؞ ڪࣷنْ ݖِ تُرُ اَلْمَسِ، دَنُ لࣹينْ دِ ݧَانْ:‏ نَنگُلࣹينَ جُبࣷو اَڪْ يࣵلَّ؞ ‏21 اَلْمَسِ مِ مَسُلَ بࣵڪَّارْ، مࣷومْ لَ يࣵلَّ سࣴفْ سُنُ بࣵڪَّارْ، نگِرْ نُ مَنَ اَمࣹ ݖِ مࣷومْ اࣵتّࣹبْ نِتْ ݧُ جُبْ فَ يࣵلَّ؞
