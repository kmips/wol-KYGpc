11
‏1 نِ مَيْ رࣷيࣹ اَلْمَسِ، مَنْ ݖِ سَمَ بࣷݒّ، رࣷيࣹلࣹينْ مَ نࣷونُ؞
ݒٗولْ دِگْلࣹ نَ ݧُ جࣹڪَّلْ جَتَايُ جُلِّ
‏2 مَا نگِ لࣹينْ دِ سَنتّ ݖِ نِ نگࣹينْ مَيْ فࣵتّلِڪࣷو ݖِ لࣺݒّ، اَڪْ نِ نگࣹينْ سَخࣷو سَمَيْ دࣺنڪَانࣹ، نِ مَ لࣹينْ ڪࣷ دࣺنڪࣹ؞
‏3 وَايࣹ نَگْ بࣴگّ نَا نگࣹينْ خَمْ نࣹ گࣺݒّ گٗورْ، اَلْمَسِ مࣷو دِ بࣷݒَّمْ؛ جِگࣺينْ، گٗورْ مࣷو دِ بࣷݒَّمْ، تࣹ اَلْمَسِ، يࣵلَّ مࣷو دِ بࣷݒَّمْ؞ ‏4 ڪࣷنْ گࣺݒّ گٗورْ گُيْ جُلِّ، مبَا مُيْ وَخࣹ ڪࣵدُّگْ وَخْيُ، بُ مُورࣹي بࣷݒَّمْ، سُوفࣹيلْ نَ اَلْمَسِ، مِيْ بࣷݒَّمْ؞ ‏5 وَايࣹ جࣺݒّ جِگࣺينْ جُيْ جُلِّ، مبَا مُيْ وَخࣹ ڪࣵدُّگْ وَخْيُ، تࣹ مُورُلْ بࣷݒَّمْ، سُوفࣹيلْ نَ جࣴڪّࣴرَمْ جِيْ بࣷݒَّمْ؞ دَيْ مࣹلْ نِ بُ نࣹلُ وࣷونْ رࣹڪّ؞ ‏6 جِگࣺينْ جُ نَنگُوُلَ مُورُ، بُ يࣹبࣷو خُوفْ ڪَوَرَمْ؞ ندࣹگَمْ اَمْ نَ ڪࣹرْسَيْ خُوفْ ڪَوَرَمْ، مبَا مُ نࣹلُ نَگْ، فࣵوّ مُ مُورْ بࣷݒَّمْ؞ ‏7 گٗورْ مࣷومْ دُ مُورْ بࣷݒَّمْ، ندَخْ مࣷويْ تَڪّندࣹيرُ يࣵلَّ تࣹ مࣷو تَخَوࣹ دَرَجَيْ يࣵلَّ؞ وَايࣹ جِگࣺينْ مࣷو تَخَوࣹ دَرَجَيْ گٗورْ، ‏8 ندَخْ گٗورْ ڪَتْ، جࣴلࣹيسُ ڪࣷ ݖِ جِگࣺينْ، وَايࣹ جِگࣺينْ ڪَيْ لَݧُ جࣴلࣹ ݖِ گٗورْ؞ ‏9 گٗورْ اِتَمْ، سࣵڪّࣹيسُ ڪࣷ نگِرْ جِگࣺينْ، وَايࣹ جِگࣺينْ ڪَيْ لَݧُ سࣵڪّ نگِرْ گٗورْ؞ ‏10 لࣷولࣷو تَخْ جِگࣺينْ وَرَ فࣺسَلْ ڪِلِفْتࣹيفْ ݖِ ڪَوْ بࣷݒَّمْ، ندَخْ مَلاَڪَ يِ تࣹيوْ¤؞ ‏11 تࣹيوُلْ نَگْ ݖِ يࣷونُ سَنگّ بِ، جِگࣺينْ مَنُلَ ݧࣵڪَّ وࣺيرُ ݖِ گٗورْ، گٗورْ اِتْ مَنُلَ ݧࣵڪَّ وࣺيرُ ݖِ جِگࣺينْ، ‏12 ندَخْ نِ جِگࣺينْ بࣵيِّڪࣷو ݖِ گٗورْ، نࣷونُ لَ گٗورْ جُدّࣷو ݖِ جِگࣺينْ، تࣹ لࣺݒّ فَ يࣵلَّ لَ بَوࣷو؞
‏13 سࣹيتَلْ-لࣹينْ ڪࣷ سࣹينْ بࣷݒّ رࣹڪّ، ندَخْ جࣹڪّ نَ ݖِ جِگࣺينْ، مُيْ جُلِّ ݖِ يࣵلَّ، تࣹ مُورُلْ بࣷݒَّمْ؟ ‏14 خَنَا نِ اࣵدِّنَ ݖِ بࣷݒَّمْ تࣴدّࣹ خَمَلُ لࣹينْ نࣹ يَرْ ڪَوَرْ گࣵݖّࣹ لَ ݖِ گٗورْ؟ ‏15 ندَخَمْ ڪَوَرْ گُ بَرࣹ سَگُ جِگࣺينْ لَ؞ جِگࣺينْ لَݧُ جَگْلࣹيلْ ڪَوَرْ گُ بَرࣹ، مُيْ مُورَايَمْ؞ ‏16 وَايࣹ ڪُ بࣴگَّ وࣹرَنْتࣹ ݖِ لࣹنࣹينْ اِتْ، نُنْ لࣷولُ دُ سُنُ بَاخْ، تࣹ دُ بَاخُ مبࣷولࣷومْ يࣵلَّ مِ؞
جࣹڪَّلْ-لࣹينْ رࣹيرُ سَنگّ بِ
‏17 وَايࣹ لِي مَ لࣹينْ دِگَلْ، اَمْ نَ ݖِ لُ مَ لࣹينْ گࣴرࣴمࣹوُلْ، ندَخْ دُ نجَرِݧْ لَ سࣹينِ ندَجࣹ دِ جُرْ، خَنَا لࣷرَانگࣹ؞ ‏18 بࣹنّ، دࣺگّ نَا اَمْ نَ اَيْ فࣺيوَلࣷو ݖِ سࣹينِ ندَجࣹ، تࣹ اَمْ نَ لُ مَ ݖِ گࣴمْ؞ ‏19 فࣵوّ سَخْ مُ اَمْ اَيْ خَاجَلࣷو ݖِ سࣹينْ بِيرْ، نگِرْ ݧِ سࣹينْ نگࣴمْ وࣺرْ مَنَ فࣺسْ ݖِ سࣹينْ بِيرْ؞ ‏20 بُ نگࣹينْ دَجࣹي بࣹنّ بࣺرَبْ، بࣷڪّ رࣹيرُ سَنگّ بِ تَخُلْ، ‏21 ندَخْ بُ نگࣹينْ دࣹي لࣹڪّ، ڪُ نࣹڪّ، رࣹيرُ بࣷݒَّمْ لَيْ جࣴڪَّنْتࣹي لࣹڪّ، بَ تَخْ ݧࣹنّ ݖِ يࣹينْ خِيفْ، فُ ݧࣹنࣹينْ نَانࣹ بَ مࣵنْدِ؞ ‏22 خَنَا اَمُلࣹينْ ڪࣴرْ يُ نگࣹينْ دِ لࣹڪّࣹ اَڪَ نَانࣹ؟ اَمْ مبࣷولࣷومْ يࣵلَّ مِ نگࣹينْ سࣷفࣹنْتَلْ، دِ رُسْلࣷو ݧِ اَمُلْ؟ اَنَ لُ مَ لࣹينْ دِ وَخْ؟ دَمَ لࣹينْ ڪࣷيْ گࣴرࣴمࣹ؟ لࣷولُ دࣹ دُمَ لࣹينْ ݖِ گࣴرࣴمْ؞
‏23 مَنْ، لِ مَ جࣴلࣹ ݖِ سَنگّ بِ، مࣷومْ لاَ لࣹينْ جࣷتّلِ:‏ سَنگّ عِيسَىٰ، گُدِّ گِ ݧُ ڪࣷ وࣷرࣹي، دَفَ جࣴلْ مبُرُ، ‏24 يࣺڪَّتِ ڪࣵدُّگْ ݖَنتّ، دَگّ ڪࣷ، نࣹ:‏ «لِي مࣷو دِ سَمَ يَرَمْ وِ لࣹينْ ݧࣹيلْ؞ دࣹيلࣹينْ دࣹفْ لِي، دِ مَ ڪࣷ فࣵتّلِڪࣷو؞» ‏25 نࣷونُ اِتَمْ لَ دࣹفْ اَڪْ ڪَاسْ بَ، گَنَّاوْ رࣹيرْ بَ، نࣹ لࣹينْ:‏ «ڪَاسْ بِي مࣷويْ ڪٗلّࣴرࣹ گُ يࣹيسْ گِ فَسࣷو ݖِ سَمَ دࣹرࣹتْ؞ دࣹيلࣹينْ دࣹفْ لِي، دِ مَ ڪࣷ فࣵتّلِڪࣷو، سَا يُ نگࣹينْ ݖِ نَانࣹي؞» ‏26 سَا سُ نگࣹينْ لࣹڪّࣹي مبُرُ مِي، مبَا نگࣹينْ نَانْ ݖِ ڪَاسْ بِ، يࣺگْلࣹ نگࣹينْ دࣹيوُگْ سَنگّ بِ، بَ ڪࣹرَ مُيْ ݧࣴوْ؞
‏27 مࣷو تَخْ، ڪࣺݒّ ڪُ جࣹڪَّدِ نُ مُ لࣹڪّࣹ ݖِ مبُرُ مِ، اَڪْ نُ مُ نَانࣹ ݖِ ڪَاسُ سَنگّ بِ، تࣷوݧْ نگَ يَرَمُ سَنگّ بِ، اَڪْ دࣹرࣹتَمْ؞ ‏28 ڪࣷنْ نَ ڪُ نࣹڪّ سࣹيتْلُ بࣷݒَّمْ، دࣷورَ لࣹڪّ ݖِ مبُرُ مِ، نَانْ ݖِ ڪَاسْ بِ؞ ‏29 ندَخْ ڪُ لࣹڪّ مبُرُ مِ، نَانْ ݖِ ڪَاسْ بِ، تࣹ گࣴدَّالُلْ يَرَمُ اَلْمَسِ، مبُگَلُ بࣷݒَّمْ لَ لࣹڪّ، نَانْ ڪࣷ؞ ‏30 لࣷولࣷو تَخْ نگࣹينْ بَرࣹ ݧُ نࣺيوْ دࣷولࣹ، اَڪْ ݧُ خِيبࣷنْ، بَ لِمْ بُ تَڪُّ دࣹي؞ ‏31 سُ نُ دࣷونْ اࣵتّࣹ سُنُ بࣷݒّ، دࣹيسُ نُ اࣵتّࣹي نࣷونُ؞ ‏32 وَايࣹ بࣷرࣷومْ بࣹي نُيْ اࣵتّࣹي اَيْ مبُگَلَمْ نگِرْ يَرْ نُ، بَ دࣹيسُ نُ مبُگَلاَلࣹ ݖِ بِيرْ اࣵتّࣹبْ اࣵدِّنَ؞
‏33 ڪࣷنْ نَگْ بࣷڪّ يِ، بُ نگࣹينْ دَجࣹي، نگِرْ بࣷڪّ لࣹڪّ، خَارَنْتࣹلࣹينْ؞ ‏34 ڪُ خِيفْ، نَ لࣹڪّࣹ ڪࣴرَمْ، نگِرْ سࣹينِ ندَجࣹ بَݧْ لࣹينَ يٗبّࣹ مبُگَلْ؞
يࣹنࣹينْ مبِرْ يِ نَگْ، بُ مَ دِڪّࣹي، دِنَا ڪࣷ لࣹيرَلْ؞
