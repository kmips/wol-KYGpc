ڪࣵدُّگْ جُلِّ اَمْ نَ يࣷونْ
11
‏1 عِيسَىٰ مࣷو دࣷونْ جُلِّ بࣹنّ بࣺرَبْ؞ بَ مُ نࣷݒّࣹي، ڪࣹنّ ݖَ تَالِبࣹيمْ يَ نࣹ ڪࣷ:‏ «سَنگّ بِ، جࣵنگَلْ نُ جُلِّ نࣷونࣹي يَحْيَىٰ جࣵنگَلࣹ جُلِّ اَيْ تَالِبࣹيمْ؞» ‏2 عِيسَىٰ نࣹ لࣹينْ:‏ «بُ نگࣹينْ دࣹي جُلِّ، نࣹ لࣹينْ:‏
بَايْ، سࣹلّنگَ ݧࣹيلْ نَ سَوْ تُرْ،
نَ سَ نگُورْ دِڪّ؞
‏3 مَيْ نُ بࣺسْ اَڪْ بࣺسْ سُنُبْ دُنْدُ بࣺسْ بُ نࣹڪّ؞
‏4 بَالْ نُ سُنُيْ بࣵڪَّارْ،
نِ نُيْ بَالࣹ نُنْ اِتَمْ، مبࣷولࣹمْ ݧِ نُ تࣷوݧْ،
تࣹ بُ نُ دُگَلْ ݖِ نَتُّ؞»
‏5 عِيسَىٰ تࣹگْ ݖَ نࣹ لࣹينْ:‏ «اَنَ ڪَنْ ݖِ يࣹينْ مࣷو سࣹيتِ اَبْ خَرِتَمْ خَاجُ گُدِّ، نࣹ ڪࣷ:‏ ‹خَرِتْ، لࣹبَلْ مَ ݖِ ݧࣹتِّ مبُرُ؞ ‏6 سَمَ خَرِتْ بُ نࣹڪّ ݖِوْ يࣷونْ مࣷو دِڪّ سَمَ ڪࣴرْ، تࣹ اَوْمَ دَرَ لُ مَ ڪࣷ تَاجَلْ،› ‏7 تࣹ خَرِتْ بَ اࣵدّࣷو ݖَ بِيرْ نࣺيگَمْ، نࣹ ڪࣷ:‏ ‹بُ مَ سࣷنَلْ، تࣴجْ نَا بَ تࣴدّ، مَاڪْ سَمَيْ دࣷومْ؛ مَنُمَا جٗگْ، دِ لَ جࣷخْ دَرَ›؟ ‏8 مَا لࣹينْ ڪࣷ وَخْ، سُ جٗگُلْ جࣷخْ ڪࣷ لَ مُ لاَجْ ندَخُگْ خَرِتࣷو اِتْ، لَ ڪࣷ ڪࣹي اَمُلْ ڪࣹرْسَيْ يࣹي نگِرْ سࣵڪُّ ندِمْبَلْ مࣷويْ تَخْ مُ جٗگْ، جࣷخْ ڪࣷ مبࣷولࣹمْ لُ مُ سࣷخْلَ؞ ‏9 مࣷو تَخْ مَ نࣹ لࣹينْ، ݧَانْلࣹينْ، ݧُ مَيْ لࣹينْ؛ سࣹيتْلࣹينْ، ندَخْ نگࣹينْ گِسْ؛ فࣴگّلࣹينْ، ݧُ اُبِّلْ لࣹينْ؞ ‏10 ندَخْتࣹ ڪࣺݒّ ڪُ ݧَانْ، دِنگَ اَمْ؛ ڪُ سࣹيتْ، دِنگَ گِسْ؛ ڪُ فࣴگّ اِتْ، دࣹيسْ نَ لَ اُبِّلْ؞ ‏11 اَنَ ڪَنْ ݖِ يࣹينْ مࣷويْ بَايْ بُ دࣷومَمْ ݧَانْ اَوْ جࣴنْ، مُ جࣷخْ ڪࣷ جَانْ؟ ‏12 مبَا مُ ݧَانْ ڪࣷ نࣹنْ، مُ جࣷخْ ڪࣷ اَوْ جِيتْ؟ ‏13 ندࣹگَمْ يࣹينْ ݧِ بࣷنْ يࣹينَ مَنَ مَيْ سࣹينِ گࣷنࣹ لُ بَاخْ نَگْ، اَسْتَمَاڪْ بَايْ بَ فَ اَسَمَانْ؞ مࣷو لࣹينْ ڪࣷ دࣵقْ فُوفْ، ندࣹ مࣷويْ مَيْ نࣷو گُ سࣹلّ گِ، ݧِ ڪࣷ ڪࣷ ݧَانْ!‏»
بࣹيلْسࣹبُلْ خࣹيخُلْ بࣷݒَّمْ
‏14 عِيسَىٰ مࣷو دࣷونْ دࣵقْ اَوْ رَبْ وُ دُگّࣷونْ نِتْ بَ مُ لُو؞ بَ رَبْ وَ گࣺنّࣹي، لُو بَ وَخْ، مبࣷولࣷو مَ وَارُ؞ ‏15 تࣹيوُلْ ݧࣹنّ ݖَ ݧࣷومْ نࣹ:‏ «ݖِ بࣹيلْسࣹبُلْ، بُورُ رَبْ يِ لَيْ دࣵقࣹ اَيْ رَبْ؞»
‏16 ݧࣹنࣹينْ نَگْ دِ ڪࣷ فِيرْ، دِ سࣵڪُّ ݖِ مࣷومْ فِرْندࣹ جُ بَوࣷو اَسَمَانْ؞ ‏17 عِيسَىٰ اِتَمْ رࣵݧّࣹي سࣹينْ خَلاَتْ، نࣹ لࣹينْ:‏ گࣺݒّ نگُورْ گُيْ خࣵجَّلِڪࣷو ݖِ بِيرَمْ، دَيْ تَسْ، يِي ڪࣴرْ مࣵبّ ݖَ ڪَوْ يࣹي؞ ‏18 سࣹيْتَانࣹ اِتَمْ، سُ خࣵجَّلِڪࣷو ݖِ بِيرَمْ، اَنَ نُ نگُورَمْ دِ يࣵگّࣹ، گَنَّاوْ يࣹينَ نࣹ بࣹيلْسࣹبُلْ لاَيْ دࣵقࣹ اَيْ رَبْ؟ ‏19 تࣹ اِتْ ندࣹگَمْ مَنْ بࣹيلْسࣹبُلْ لاَيْ دࣵقࣹ رَبْ يِ، سࣹينْ ݧࣷݧْ نَگْ ݧࣷومْ، ڪَنْ لَݧُيْ دࣵقࣹ اَيْ رَبْ؟ ڪࣷنْ ڪَيْ سࣹينْ ݧࣷݧَ لࣹينْ تِيݧَلْ؞ ‏20 وَايࣹ سُ دࣹي لࣷخࣷلْ يࣵلَّ لاَيْ دࣵقࣹ رَبْ يِ نَگْ، ڪࣷنْ نگُورُگْ يࣵلَّ اَگْسِ نَ بَ ݖِ يࣹينْ؞
‏21 «سُ ݒࣷنڪَلْ گࣵنَّايࣷو بَ دِيسْ، وَتُّ ڪࣴرَمْ، اَلَلَمْ دِنَ رَوْ؞ ‏22 وَايࣹ بُ ڪࣷ ڪُ ڪࣷ مَنْ سࣷنگࣹي، بَ دَانْ ڪࣷ، دَيْ نَنگُ مبࣷولࣹمْ نگࣵنَّايْ لُ مُ يَاڪَارࣷونْ، بَ نࣷݒِّ سࣺدَّلࣹ اَلَلَمْ؞ ‏23 ڪُ فَرُلْ اَڪْ مَنْ نَگْ، مَنْ نگَيْ خࣹيخْ، تࣹ ڪُ اࣵنْدُلْ اَڪْ مَنْ دِ دَجَلࣹ، يَا نگِ تَسَارࣹ؞
‏24 «اَوْ رَبْ وُ بࣷنْ، بُ جࣵݒّࣹي نِتْ بَ گࣺنّ ڪࣷ، دَيْ وࣴرْ بࣺرَبْ يُ فࣹنْدِ، دِ وُتْ فُ مُ نࣷݒّلࣷو، تࣹ دُ ڪࣷ گِسْ؞ سُ ڪࣷ دࣹفࣹي مُ نࣹ:‏ ‹نَا دࣹلُّ ݖَ سَمَ ڪࣴرْ، گَ مَ جٗگࣹ؞› ‏25 بُ دِڪّࣹي فࣹڪّ ڪࣴرْ گِ بُوبࣹ، دࣹفَرࣹ بَ جࣹڪّ، ‏26 دَيْ دࣹمْ اِنْدِ يࣹنࣹينْ جُرٗومْ ݧَارِ رَبْ يُ ڪࣷ يࣺيسْ، ݧُ بࣷڪّ سَنݖّسِ؞ سُ ڪࣷ دࣹفࣹي نِ نِتْ ڪِيْ مُجّࣹ مࣷويْ يࣺيسْ نَ مُ جࣴڪّࣹ؞»
‏27 نَڪَ لَ عِيسَىٰ دِ وَخْ لࣷولُ، جࣹنّ جِگࣺينْ ݖَ مبࣷولࣷو مَ اࣵدُّ ݖَ ڪَوْ، نࣹ ڪࣷ:‏ «ندࣷڪّلࣹي ڪِ لَ اࣴمبّ ݖِ بِيرَمْ، نࣵمْݒَلْ لَ ݖِ وࣹينَمْ!‏» ‏28 مࣷومْ نَگْ مُ نࣹ ڪࣷ:‏ «خَنَا ڪَيْ، ندࣷڪّلࣹي ݧِيْ دࣺگْلُ ڪࣵدُّگْ يࣵلَّ تࣹ دِ ڪࣷ جࣴفࣹ؞»
يࣷنࣹنتّ يࣵلَّ يُنُسْ دࣷيْ نَ فِرْندࣹ
‏29 نَڪَ لَ مبࣷولࣷو مَيْ گࣴنَ تَڪُّ، عِيسَىٰ نࣹ:‏ «نِتِ تࣹيْ جِي، نِتْ ݧُ بࣷنْ لَݧُ؞ فِرْندࣹ لَݧُيْ وُتْ، وَايࣹ دࣹيسُ لࣹينْ جࣷخْ جࣹنّ فِرْندࣹ جُ مࣷيْ فِرْندࣹيْ يُنُسْ؞ ‏30 ندَخْ نَ يُنُسْ نࣹڪّࣹ وࣷونْ فِرْندࣹ ݖَ وَا نِنِوْ، نࣷونُ لَ دࣷومُ نِتْ ڪِ نࣹڪّࣹ فِرْندࣹ ݖِ نِتِ تࣹيْ جِي؞ ‏31 ڪࣹرࣷوگْ اࣵتّࣹ بَ، لِنگࣹيرُ بࣴجْ-سَالُمْ دِنَ جٗگْ، جَنࣷوڪْ نِتِ تࣹيْ جِي، تِيݧَلْ لࣹينْ؛ ندَخْ مࣷو بࣵيِّڪࣷو وࣷونْ ݖَ ݖَتَلْ اࣵدِّنَ، نگِرْ دࣺگْلُسِ خَمْ-خَمُ سُلࣹيْمَانْ، تࣹ تࣹيْ جِي، ڪُ سُتْ سُلࣹيْمَانَ نگِ فِ؞ ‏32 وَا رࣺيوُمْ نِنِوْ اِتْ ڪࣹرࣷوگْ اࣵتّࣹ بَ دِنَݧُ تَخَوْ، لَيࣷوڪْ نِتِ تࣹيْ جِي، تِيݧَلْ لࣹينْ ندَخْ ݧࣷو تُوبࣹ وࣷونْ ݖَ وَارࣹبْ يُنُسْ، تࣹ تࣹيْ جِي، ڪُ سُتْ يُنُسَ نگِ فِ؞
بࣴتْ مࣷويْ لࣵمْݒُ يَرَمْ
‏33 «دࣹيسُلْ تَالْ اَبْ لࣵمݒّ، دِ ڪࣷ دُگَلْ فُ لࣵقُ، مبَا ݖِ سُوفُ لࣹگࣹتْ؞ دࣹيسْ ڪࣷيْ اَجْ ڪَيْ، بَ ڪُيْ دُگّسِ گِسْ مُيْ لࣹيرْ؞ ‏34 بࣴتْ مࣷويْ لࣵمْݒُ يَرَمْ؞ بُ سَ بࣴتْ وࣺرࣹي، سَ يَرَمْ وࣺݒَّ ݖِيْ لࣹيرْلࣷو، وَايࣹ بُ سَ بࣴتْ بࣷنࣹي، سَ يَرَمْ وࣺݒَّ نگِ ݖِ لࣴنْدࣴمْ؞ ‏35 ڪࣷنْ سࣹيتَلْ ندَخْ لࣹيرْ گِ نگَ فࣷوگْ نࣹ مُ نگِ ݖِ يَوْ دُ اَگْ لࣴنْدࣴمْ؞ ‏36 بُ سَ يَرَمْ وࣺݒّ دࣹي لࣹيرْلُ نَگْ تࣹ لࣹنّ نࣹڪُّ ݖِ ݖِگْ لࣴنْدࣴمْ، سُ بࣷوبَا لَ سَ يَرَمْ دِ مَنَ لࣹيرْلُ بَ نࣹ رࣵݧّ، مࣹلْ نِ بُ لَ اَبْ لࣵمݒّ نِيتࣹي، نࣹ رࣵݧّ ݖِ سَ ڪَوْ؞»
عِيسَىٰ خَسْ نَ نِتِ دِينࣹ ݧِ
‏37 بَ عِيسَىٰ وَخࣹي بَ نࣷݒِّ، اَبْ فَرِسࣹنْ دَ ڪࣷو وࣷو ڪࣴرَمْ، نگِرْ بࣷڪّ اَڪْ مࣷومْ لࣹڪّ؞ مُ نَنگُ، دࣹمْ بَ دُگّسِ، دَلْدِ تࣷوگْ ݖَ لࣹڪُّڪَايْ بَ؞ ‏38 فَرِسࣹنْ بِ نَگْ جَاخْلࣹ، ندَخْ لَ عِيسَىٰ سࣹتْلُوُلْ بَلَ مࣷو وَاجَ لࣹڪّ؞ ‏39 سَنگّ بِ نࣹ ڪࣷ:‏ «يࣹينْ فَرِسࣹنْ يِ، بِتِبْ ڪَاسْ بِ اَڪْ ندَبْ لِ دَالْ نگࣹينْ دِ رَخَسْ، تࣹ فࣹڪّ لࣹينْ فࣹيسࣹ جِڪّࣷيْ جَگࣷو لُ لࣹوُلْ، اَڪُگْ مبࣷنْ ݖِ بِيرْ!‏ ‏40 يࣹينَ ݧࣵڪّ خࣹلْ!‏ خَنَا دُ ڪِ سࣵڪّ بِتِ مࣷو سࣵڪّ بِيرْ اِتَمْ؟ ‏41 لِ ݖِ بِيرْ ڪَيْ نگࣹينْ دِ سَرْخࣹ!‏ سُ بࣷوبَا سࣹينْ لࣺݒّ اَيْ سࣹتْ؞ ‏42 وَايࣹ وٗويْ نگَلَّ يࣹينْ فَرِسࣹنْ يِ!‏ خࣷبِ نَانَ اَڪْ يࣹنࣹينْ سَفَلْ يُ مࣹلْ نِ رُ¤ اَڪْ وࣺݒّ خࣹيتُ لُجُمْ، يࣹينَ نگِ ݖِيْ گࣺنّࣹ ݖࣺرُبْ فُڪّࣹيلْ، نگِرْ اَسَڪَ، نجُبْتࣹ اَڪْ ݖࣷفࣹيلُ يࣵلَّ نَگْ، نگࣹينْ سࣵگَّنࣹ؞ لِي دࣹ وَرْ نَ لࣹينْ، وَايࣹ لࣹي اِتْ، وَرُلࣹينْ ڪࣷو سࣵگَّنࣹ؞ ‏43 وٗويْ نگَلَّ يࣹينْ فَرِسࣹنْ يِ!‏ يࣹينَيْ سࣵڪُّ ݒࣺيتࣹ يُ ڪَنَمࣹ ݖَ جࣵنگُ يَ، اَڪْ نُيࣷوبْ گࣴدَّالْ ݖَ ݒࣺنݖّ مَ؞ ‏44 وٗويْ نگَلَّ يࣹينْ، يࣹينَ نگِ مࣹلْ نِ اَيْ بࣵمّࣹيلْ يُ اَمُلْ خࣵمِّڪَايْ، نِتْ ݧِيْ دࣷخْ ݖَ ڪَوْ تࣹ خَمُݧُ ڪࣷ؞»
‏45 ڪࣹنّ ݖِ خَمْڪَتِ يࣷونْ يِ نَگْ اࣵدُّ نࣹ:‏ «سࣴرِݧْ بِ، لࣷولُ نگَ وَخْ دࣹ، نُنْ اِتَمْ، خَسَالࣹ نگَ نُ ݖِ؞» ‏46 عِيسَىٰ اِتَمْ نࣹ:‏ «يࣹينْ خَمْڪَتِ يࣷونْ يِ اِتَمْ، وٗويْ نگَلَّ يࣹينْ، يࣹينْ ݧِ سࣴفْ نِتْ ݧِ سࣴفْ يُ ݧُ اࣵتَّنُلْ، تࣹ يࣹينْ ݖِ سࣹينْ بࣷݒّ، سࣹينْ بࣹنّ بَارَامْ سَخْ دُ ݖِ لاَلْ؞ ‏47 وٗويْ نگَلَّ يࣹينْ، يࣹينَيْ تَبَخْ بࣵمّࣹيلِ يࣷنࣹنتّ يِ، تࣹ سࣹينِ مَامْ رࣹيࣷونْ لࣹينْ؞ ‏48 ڪࣷنْ يࣹينَ سࣹيدࣹي نࣷونُ نࣹ سࣹينْ جࣴفِ مَامْ يَ، اࣵندّ نگࣹينْ ݖِ، ندَخْ ݧࣷومْ ݧࣷو لࣹينْ رࣹيْ، يࣹينْ نگࣹينْ تَبَخْ سࣹينِ بࣵمّࣹيلْ؞ ‏49 لࣷولࣷو تَخْ اِتَمْ ڪࣵدُّگْ يࣵلَّ گِيْ خࣹلْ، بِرَلْ نࣹ:‏ ‹مَايْ يࣹبَلْ فَ ݧࣷومْ اَيْ يࣷنࣹنتّ اَڪِ ندَوْ، ݧِي ݧُ بٗومْ، ݧࣹي، ݧُ بُندّخَتَلْ؞› ‏50 سُ ڪࣷ دࣹفࣹي يࣷونْ تࣷݒّ نِتِ تࣹيْ جِي، دࣹرࣹتُ مبࣷولࣹمْ يࣷنࣹنتّ يِ تُورُ، دَلࣹ ڪࣷ ݖَ ݖࣷسَانُ اࣵدِّنَ؞ ‏51 لَ دَلࣹ ݖَ دࣹرࣹتُ اَبࣹلْ بَ ݖَ دࣹرࣹتُ سࣵڪَّرْيَا¤، مَ ݧُ رࣹيْ ݖَ دِگَّنْتࣹ سَرْخَلُڪَايْ بَ اَڪْ بࣺرَبْ بُ سࣹلّ بَ، مَا لࣹينْ ڪࣷ وَخْ دࣺيْ، دࣹيسْ نَ ڪࣷ تࣷݒّ نِتِ تࣹيْ جِي؞ ‏52 نگَلَّ يࣹينْ خَمْڪَتِ يࣷونْ يِ، يࣹينَ سࣵنڪّ ݖَابِيْ خَمْ-خَمْ؛ يࣹينْ دُگُّلࣹينْ، ݧِيْ دُگّسِ، نگࣹينْ تࣹرࣹ لࣹينَ دُگّ؞»
‏53 بَ عِيسَىٰ گࣺنّࣹي فࣷوفَ، ݖَ لَ فَرِسࣹنْ يَ اَڪْ فِرِڪَتْ يَ تࣵمْبَلࣹي خَبْتَلُ ݖِ ڪَوَمْ، دِ ڪࣷ سࣹيتْلࣷو اَيْ لاَجْ ݖِ مبِرْ يُ بَرࣹ، ‏54 دِ ڪࣷ يࣹيرُ، تࣹيْ وُتَ ڪࣴفْ ڪࣵدُّ گُ رࣴݖّࣹ ݖِ گࣺمِّݧَمْ، نگِرْ تُومَالْ ڪࣷ؞
