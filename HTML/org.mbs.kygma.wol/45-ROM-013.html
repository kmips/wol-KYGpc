نَ ݧࣺݒّ دࣺگَّلْ نجِيتْ يِ
13
‏1 نَ ڪُ نࣹڪّ دࣺگَّلْ نجِيتْ يِ يࣷرْ ڪِلِفْتࣹيفْ، ندࣹ اَمُلْ نجِيتْ لُ بَوࣷووُلْ فَ يࣵلَّ، تࣹ نجِيتْ يِ تࣷوگْ، يࣵلَّ مࣷو لࣹينْ تَبّ؞ ‏2 مࣷو تَخْ ڪُيْ بَݧَلْ نجِيتْ، تࣴرَلِنُ يࣵلَّ نگَ بَݧَلْ، تࣹ ݧِ بَݧَلْ نجِيتْ، ݧࣷو وࣷولْ سࣹينْ بࣷݒّ مبُگَلْ؞ ‏3 ڪُيْ دࣹفْ لُ بَاخْ نَگْ، دࣷو رَگَلْ ڪِلِفَ يِ، ڪِيْ دࣹفْ لُ بࣷنْ مࣷو لࣹينْ دِ رَگَلْ؞ سࣷو بࣴگّࣹي مُݖّ ݖِ رَگَلْ ڪِلِفَ نَگْ، دࣹفَلْ لُ بَاخْ، جࣷتْ نگࣴرࣴمَمْ؞ ‏4 ندَخْ ڪِلِفَ گِ مࣷويْ ندَوْ لِ يࣵلَّ يࣹبَلْ نگِرْ سَ جࣵمّ؞ وَايࣹ سࣷو دࣹي دࣹفْ لُ بࣷنْ، رَگَلَلْ، ندَخْ ڪِلِفَ گِ يࣷرُلْ اَبْ سَامَرْ ݖِگْ نࣹينْ؛ مࣷويْ ندَوَلْ يࣵلَّ لِيْ اَݧَلࣹ، تࣹ مࣷويْ جࣷتّلِ اَمْ سࣵنجّ مُ وࣵݖَّلْ ڪُيْ دࣹفْ لُ بࣷنْ؞ ‏5 مࣷو تَخْ نَنگُلْ نجِيتْ دِ وَرْتࣺيفْ، ندَخْ نجَرِݧَلْ نَنگُلْ نجِيتْ يࣹمُلْ رࣹڪّ ݖِ مُݖّ ݖِ اَمْ سࣵنجّ، وَايࣹ خࣹلْ مُ دَلَ نگِ ݖِ؞
‏6 لࣷولُ مࣷو وَرَلْ اِتْ نگࣹينْ دِ فࣹيْ گَلَگْ، ندَخْ نجِيتْ يِ ڪࣷيْ فࣹيࣹيڪُ ڪَتْ، ݧࣷويْ جَوْرِݧِ يࣵلَّ يِ سَسࣷو گࣷوگُ ڪِلِفْتࣹيفْ؞ ‏7 جࣷخْلࣹينْ ڪُ نࣹڪّ لِ نگࣹينْ وَرْلࣷوڪْ مࣷومْ؛ ڪُ نگࣹينْ وَرَ فࣹيْ گَلَگْ، فࣹيْلࣹينْ ڪࣷ ڪࣷ؛ ڪُ نگࣹينْ وَرَ فࣹيْ جُوتِ، فࣹيْلࣹينْ ڪࣷ ڪࣷ؛ ڪُ نگࣹينْ وَرَ وࣹگْ، وࣹگْلࣹينْ ڪࣷ؛ ڪُ نگࣹينْ وَرَ تࣹرَلْ، تࣹرَلْ-لࣹينْ ڪࣷ؞
ݖࣷفࣹيلْ مࣷو تࣴنڪّ يࣷونُ مُوسَىٰ
‏8 بُلࣹينْ اَمࣹيلْ ڪࣹنّ بࣷرْ، لُ مࣷيْ بࣷرُبْ سࣷݒَّنْتࣹ؛ ندَخْ ڪُ سࣷݒّ سَ مࣷرࣷومْ، يَا مَتَلْ يࣷونُ مُوسَىٰ؞ ‏9 ندَخْ ندِگَلْ يِي ڪَتْ:‏ «بُلࣹينْ جَالࣷو، بُلࣹينْ بٗومْ، بُلࣹينْ سࣵݖّ، بُلࣹينْ خࣹمّࣹمْ¤،» اَڪْ لࣹنࣹينْ ندِگَلْ لُ مُ مَنَ دࣷونْ، ݖِ گࣹنّ ڪࣵدُّ گِي لَ تࣴنڪࣹ:‏ «نِ نگࣹينْ سࣷݒّࣹ سࣹينْ بࣷݒّ، نَنگࣹينْ ڪࣷ سࣷݒّࣹ سࣹينْ مࣷرࣷومْ¤؞» ‏10 ݖࣷفࣹيلْ دُ وَرَلْ ڪࣹنّ دࣹفْ مࣷرࣷومَمْ لُ بࣷنْ؛ ڪࣷنْ ݖࣷفࣹيلْ مࣷو مَتَلْ سࣴڪّ يࣷونُ مُوسَىٰ؞
تࣹيوْلُ وَرْ نَ گࣴمْڪَتْ ݧِ
‏11 لِ تَخْ نگࣹينْ وَرَ گࣴنَ فَرْلُ ݖِ لࣷولُ مࣷويْ، خَمْ نگࣹينْ جَنتّ يِ نُ تࣷلُّ؛ سࣹينْ وَخْتُوْ يࣹوُّ جࣷتْ نَ بَ نࣷݒِّ، ندࣹ لࣺيگِ لَ نُ مُݖّ گࣴنَ جࣹگࣹ بَ نُيْ دࣷورَ گࣴمْ؞ ‏12 گُدِّ گِ مࣵگَّتْ نَ، بࣴتَ نگِ بࣴگَّ سࣹتْ؞ ڪࣷنْ نَگْ نَنُ بࣵيِّ جࣴفِ لࣴنْدࣴمْتُ تࣹ رࣵنگࣷو گࣵنَّايْ يِ ݧُيْ خَرࣹي ݖِگْ لࣹيرْ؞ ‏13 نَنُ جࣴفࣹ جࣴفِنْ وُ جࣹڪَّ دࣹفْ بࣴݖّࣴگْ؛ نَنُ سࣷرࣹ خَوَارࣹ اَڪْ مࣵنْدِتࣹ، سࣷرࣹ بَانࣹ بُ لࣹوُلْ، اَڪْ يࣵقُتࣹ يُ نِ مࣹلْ، تࣹ سࣷرࣹ ݝَايࣷو اَڪْ ڪِݧَانْ؞ ‏14 وَايࣹ يِيرࣷولࣹينْ سَنگّ عِيسَىٰ اَلْمَسِ، تࣹ نگࣹينْ بَݧَ يِتّࣹوࣷو سࣹينْ بَڪَّنْ، نگِرْ دِ تࣷݒّ سࣹينِ خࣹمّࣹمْتࣺيفْ؞
