نجࣷتْلاَيْ وَرْ نَ ݖِ بٗومْڪَتْ بُ اُمْݒࣹ
21
‏1 سُ دࣹي دَݧࣷو فࣹڪّ ڪُ ݧُ بٗومْ ݖِ بِيرْ رࣺيوْ مِ لࣹينْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ مࣷومَلࣹ، مُ تࣴدّ ݖِ اࣵلّ بِ، تࣹ خَمࣹيسُلْ ڪُ ڪࣷ رࣹيْ، ‏2 نَ سࣹينِ مَگْ اَڪْ سࣹينِ اࣵتّࣹڪَتْ دࣹمْ، نَتِّ دِگَّنْتࣹ فَ نࣺيوْ بَ تࣴدّ، اَڪْ دࣴڪّ يَ ڪࣷ سࣺقْ؞ ‏3 سُ ڪࣷ دࣹفࣹي دࣴڪّ بَ گࣴنَ جࣹگࣹوࣷوڪْ نࣺيوْ بَ، مَگِ دࣴڪّ بࣷوبَ تࣵنّࣹ ݖَ نَگْ وُ جِگࣺينْ وُ ݧُ مَسُلَ لِگّࣺييْلࣷو، تࣹ مَسࣹيسُ ڪࣷو رࣵنگّ؞ ‏4 نَ مَگِ دࣴڪّ بࣷوبُ وࣷمَّتْ نَگْ وِ بَ ݖِ خُرْ وُ وَلْ مَ دُلْ ݝِيسࣹ، تࣹ مَسࣹيسُ ݖࣹي بࣹيْ، مَسࣹيسُ ݖࣹي جِ؞ ݧُ دَلْدِ دَمّ لࣷوسُ نَگْ وُ جِگࣺينْ وِ فࣷوفَ ݖَ وَلْ مَ؞
‏5 سُ ڪࣷ دࣹفࣹي سَرْخَلْڪَتْ يِ، لࣹوࣹينْ ݧِ دِڪّ، ندَخْ ݧࣷومْ لَ سࣹينْ يࣵلَّ اَجِ سَخْ جِ تَامُ، نگِرْ ݧُ دِ ڪࣷ لِگّࣺييَلْ، دِ بَرْڪࣹيلࣹ ݖِوْ تُرَمْ، تࣹ ݖِ سࣹينْ ڪَوْ ڪࣵدُّ لࣹيسْ دِ اࣵتّࣹي بࣺݒّ جࣷتࣹ، اَڪْ گࣺݒّ لࣷرَانگࣹ ݖِ وࣵلُّ يَرَمْ؞ ‏6 دࣴڪّ بࣷوبُ گࣴنَ جࣹگࣹوࣷوڪْ فَ نࣺيوْ بَ تࣴدّ، نَ مبࣷولࣹمْ سࣹينِ مَگْ رَخَسْ سࣹينِ لࣷخࣷ، تِيمَلࣹ ݖَ نَگْ وُ جِگࣺينْ وَ ݧُ دَمّ لࣷوسْ وَ، ݖَ وَلْ مَ؞ ‏7 ݧُ بِرَلْ نࣹ:‏ «نُنْ دࣺيْ سُنُيْ لࣷخࣷ تُورُلْ دࣹرࣹتُ نِتْ ڪِي، تࣹ سُنُيْ گࣴتْ اِتْ گِسُلْ بَ ݧُ ڪࣷيْ تُورْ؞ ‏8 نگَلَّ اَجِ سَخْ جِ، جࣺگَّلَلْ إِسْرَاءِيلْ، سَ ݧࣷݧْ ݧِ نگَ جࣷتْ، تࣹ بَݧَ تࣷݒّࣹ اࣵقُ دࣹرࣹتُ جَامْبُرْ بُ دࣹفُلْ دَرَ، فِ دِگّ إِسْرَاءِيلْ سَ ݧࣷݧْ؞» سُ ڪࣷ دࣹفࣹي نجࣺگَّلْ ݧࣹيلْ لࣹينْ، ݖَ دࣹرࣹتُ نِتْ ڪَ؞ ‏9 يࣹينْ نَگْ، نࣷونُ نگࣹينْ دِ تࣹنقࣹي ݖِ سࣹينْ بِيرْ، اࣵقُ دࣹرࣹتُ جَامْبُرْ جُ ݧُ تُورْ، ندَخْ ڪࣷنْ دࣹفْ نگࣹينْ لِ اَجِ سَخْ جِ رَفࣹتْلُ؞
اَيْ سࣵرْتَ نگِ ݖِ وࣵلُّ نجَبࣷوتْ
‏10 بُ نگࣹينْ خَرࣹجࣹيڪْ سࣹينِ نࣷونْ، تࣹ سࣹينْ يࣵلَّ اَجِ سَخْ جِ تࣹگْ لࣹينْ ݖِ سࣹينِ لࣷخࣷ، بَ نگࣹينْ جࣵݒّ لࣹينْ نجَامْ، ‏11 سُ نگࣹينْ گِسࣹي ݖَ جَامْ يُ جِگࣺينْ يَ، ڪُ رَفࣹتْ ڪُ نگࣹينْ خࣺدّ، تࣹ بࣴگّ ڪࣷو جࣴلْ جَبَرْ، ‏12 يٗبُّلࣹينْ ڪࣷ سࣹينْ بِيرْ نࣺيگْ، مُ وَتُ نࣹلْ، دَگّ اَيْ وࣹوَمْ، ‏13 سُمِّ يࣺرࣹ يَ ݧُ ڪࣷ جࣵݒّࣹ نجَامْ، تࣹ تࣷوگْ سࣹينْ بِيرْ نࣺيگْ؞ دَيْ تِيسْلُ نَگْ ندࣹيَمْ اَڪْ بَايَمْ لُ مَتْ وࣹيرُ لࣴمّ؞ گَنَّاوْ لࣷولُ نگࣹينْ دࣴڪّࣷو ڪࣷ، دِ جࣴڪّࣴرَمْ، مُيْ سࣹينْ جَبَرْ؞ ‏14 بُ لࣷولُ اَمࣹي بَ نࣹيخَتُ لࣹينْ، بࣵيِّلࣹينْ ڪࣷ، مُ دࣹمْ فُ ڪࣷ نࣹيخْ، وَايࣹ بُلࣹينْ ڪࣷ جَايْ مُڪّ، دِ ڪࣷ وࣹݖِّڪࣷو خَالِسْ؞ بُلࣹينْ ڪࣷ جَامْلࣷو اِتْ، گَنَّاوْ بُ نگࣹينْ ڪࣷ جَبَرࣷو بَ نࣷݒِّ؞
‏15 ڪُ اَمْ ݧَارِ جَبَرْ، سُ خࣹجّࣷو ڪࣹنّ ڪِ، فࣷݧْ ڪِ ݖِ دࣹسْ، تࣹ ڪِ مُ خࣹجّࣷو، اَڪْ ڪِ مُ فࣷݧْ جُرَلْ ڪࣷيْ دࣷومْ يُ گٗورْ، تࣹ تَاوْ بُ گٗورْ بِ دِ دࣷومُ ڪِ مُ فࣷݧْ، ‏16 بࣺسْ بُيْ سࣺدَّلࣹ دࣷومَمْ يُ گٗورْ اَلَلَمْ، دُ سَݧَ جࣷخْ دࣷومُ خࣹجّ بِ، ݖࣺرْ بِ تَاوْ يࣹلّࣷو، وَݖّ دࣷومُ ڪَ مُ فࣷݧْ، تࣹ مُيْ تَاوْ؞ ‏17 تَاوْ بَيْ دࣷومُ ڪَ مُ فࣷݧْ، نَ ڪࣷ نَنگُلْ اَبْ ݖࣺرَمْ، جࣷخْ ڪࣷ ݧَارْ-ݖَيْ ݖࣺرُ رَڪّ جَ، ݖِ مبࣷولࣹمْ اَلَلَمْ، ندَخْ ڪࣷوڪَ سࣷقِڪࣷو ݖِ دِگّ دࣷولࣹيمْ، مࣷويْ ڪَ يࣹلّࣷو ݖࣺرُ تَاوْ؞
‏18 ڪُ اَمْ دࣷومْ جُ گٗورْ جُ دࣴگࣴرْ بࣷݒّ تࣹ تࣴ؛ دࣺگَّلُلْ بَايْ، دࣺگَّلُلْ ندࣹيْ، سُ ݧُ ڪࣷ يَرࣹي، تࣹ تَخُلْ مُ دࣺگَّلْ لࣹينْ، ‏19 نَ ڪࣷ بَايَمْ اَڪْ ندࣹيَمْ جࣵݒّ، يٗبُّ بَ ݖَ سࣹينْ مَگِ دࣴڪّ بَ، ݖَ سࣹينْ بُنتّ دࣴڪّ؞ ‏20 ݧُ وَخْ سࣹينْ مَگِ دࣴڪّ بَ، نࣹ لࣹينْ:‏ «سُنُ دࣷومْ جِي ڪُ دࣴگࣴرْ بࣷݒّ لَ تࣹ تࣴ؞ دࣺگَّلُ نُ، خَنَا دِ خَوَارࣹيڪَ مࣵنْدِ رࣹڪّ؞» ‏21 نَ ڪࣷ گٗورِ دࣴڪّ بࣺݒّ دٗورِ دࣷجْ بَ مُ دࣹي؞ نࣷونُ نگࣹينْ دِ تࣹنقࣹي لُ بࣷنْ ݖِ سࣹينْ بِيرْ؞ إِسْرَاءِيلْ گࣺݒَّ ڪࣷيْ دࣺگّ، دَلْدِ رَگَلْ؞
دࣹيسُلْ وࣺڪّ نِتْ، فَنَانَلْ ڪࣷ
‏22 سُ نِتْ مࣷيࣹي بَ دࣹي دِ اࣵتّࣹيمْ، ݧُ رࣹيْ ڪࣷ، وࣺڪّ ڪࣷ ݖِ گَرَبْ، ‏23 بُ نࣺيوَمْ فَنَانْ ݖَ گَرَبْ گَ، نَݧُ ڪࣷ دَلْدِ رࣷبْ ݖَ بࣺسْ بَ، ندَخْ ڪِ وࣺڪّࣹ اَلْڪࣵنْدࣹيْ يࣵلَّ لَ؞ بُلࣹينْ سࣷبࣹيلْ رࣺيوْ مِ لࣹينْ سࣹينْ يࣵلَّ اَجِ سَخْ سࣺدّ؞
