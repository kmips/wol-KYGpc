ندَوِ اَلْمَسِ يَ دࣷيْلࣷو نَݧُ ݒٗولْ
2
‏1 فُڪِّ اَتْ اَڪْ ݧࣹينتّ گَنَّاوْ گِ لاَ دࣹلُّ يࣹرُسَلࣹمْ، اࣵندّ اَڪْ بَرْنَبَ، يٗبَّالࣹ تِتْ اِتَمْ؞ ‏2 سَمَ تࣵنڪّ يࣷويُ نَگْ اَمْ ݒࣹيݧُ مࣷو ڪࣷ وَرَلْ؞ ݖَ لاَ وࣺيتࣷو اَڪْ ݧَ ݧُ جࣵݒّࣷونْ نࣹ ݧࣷويْ ڪࣵنگَمِ مبࣷولࣷومْ گࣴمْڪَتْ ݧَ، فَرَمْفࣵݖّࣹيلْ لࣹينْ خِبَارُ جࣵمّ بِ مَيْ يࣺگَلْ جَامْبُرْ ݧِ دُلْ يَوُتْ، نگِرْ رَگَلْ سَمَ لِگّࣺييْ بِ مَ جࣷتَ دࣹفْ، تࣹ دِ ڪࣷ وࣺيࣹ، دࣷونْ ݖࣷونࣷبْ نࣹينْ؞ ‏3 مࣷونْتࣹ تِتْ مِ مَ اࣵنْدَلࣷونْ سَخْ، تࣹگࣹيسُ ڪࣷ فَ وَرْتࣺيفُ خَرَفْ، تࣹ ݖِ اَسْڪَنُ گࣹرࣹگْ يِ دُلْ خَرَفْ لَ بࣷڪّ؞ ‏4 دࣷونْتࣹ بࣷوبَ، اَيْ نِتْ ݧُ تُدّࣷو بࣷڪِّ گࣴمْڪَتْ ݧࣷو يࣷخࣷوسُ بَ تࣵبِّ ݖِ سُنُ بِيرْ، دِ نُ يࣴدُّ، نگِرْ لُقَتُ نِ نُ مࣷومࣹ سُنُ بࣷݒّ ندَخْ اَلْمَسِ عِيسَىٰ مِ نُ گࣴمْ، تࣹ لِ ݧُ ݖِ جُبْلُ دِ دࣹلّࣷو نُ ݖِ بُومِ نجَامْ؞ ‏5 وَايࣹ ݧࣷوݧُ، نَنگُلُنُ لࣹينْ لُ تࣷلّࣷوڪْ سِمِلِ سَخْ، ندَخْ بࣴگّ لࣹينَ سࣵمَّلْ يࣹينْ، دࣴگُّگْ خِبَارُ جࣵمّ بِ، بَ مُ سَخْ ݖِ يࣹينْ؞
‏6 تࣹيوُلْ ݧࣷوݧُ ݧُ جࣵݒّࣹ وࣷونْ اَيْ ڪࣵنگَمْ، اَڪْ دَرَجَ جُ ݧُ مَسࣷونَ اَمْ، اَمَلُ مَ سࣷلࣷ، ندࣹ يࣵلَّ اَمُلْ خࣹجّ اَڪْ سࣹينْ، ݧࣷوݧُ يࣷڪُّݧُ لࣹنّ ݖِ ڪࣵدُّ گِ مَيْ خَمْلࣹ؞ ‏7 خَنَا ڪَيْ لِ ݧُ گِسْ مࣷو دِ نِ يࣵلَّ يࣹبَلࣹ ݒِيࣹيرْ نگِرْ مُ اࣵگّࣹ اَسْڪَنُ يَوُتْ خِبَارُ جࣵمّ بِ، نࣷونُ لَ مَ يࣹبَلࣹ نگِرْ مَ اࣵگّࣹ ڪࣷ جَامْبُرْ ݧِ خَرَفُلْ، ‏8 ندَخْ ڪِ لِگّࣺييْ ݖِ ݒِيࣹيرْ، نگِرْ مُ دࣷونْ ندَوَمْ ݖِ يَوُتْ يِ، مࣷو لِگّࣺييْ ݖِ مَنْ اِتَمْ، نگِرْ مَيْ ندَوَمْ ݖِ يࣹنࣹينْ خࣹيتْ يِ؞ ‏9 ݧَ ݧُ جࣵݒّࣷونْ نࣹ ݧࣷويْ ڪࣹنُيْ مبࣷولࣷومْ گࣴمْڪَتْ ݧَ نَگْ، تࣹ ݧُيْ يَعْقُوبَ اَڪْ سࣹفَسْ اَڪْ يࣷوَانْ، بَ ݧُ رࣵݧّࣹي يِوْ وِ ݧُ مَ جَگْلࣹيلْ، مَاڪْ بَرْنَبَ لَݧُ جࣷخْ لࣷخࣷ، نگِرْ نُنْ نُ لِگّࣺييِ ݖَ جَامْبُرْ ݧَ، تࣹ ݧࣷومْ ݧُ لِگّࣺييِ ݖِ يَوُتْ يِ؞ ‏10 لاَجُݧُ لُ مࣷيْ نُ فࣵتّلِڪُ نࣺيوْ دࣷولࣹ يِ، تࣹ لࣷولُ دِ لُ مَ ݒَسْتࣺيفࣷو جࣴفࣹ؞
ݒٗولْ يࣹدّ نَ سࣹفَسْ ݖَ اࣵنݖࣷسْ
‏11 وَايࣹ بَ سࣹفَسْ دِڪّࣹي اࣵنݖࣷسْ، جࣵڪّ لاَ ڪࣷ دࣹفْ، وࣹدِّ ڪࣷ، ندَخْ مࣷو نࣹڪُّلْ وࣷونْ ݖِ دࣴگّ؞ ‏12 ندَخْ ڪَتْ بَلاَ اَوْ نِتَ دِڪّ، بࣵيِّڪࣷو فَ يَعْقُوبَ، سࣹفَسْ دَ دَانْ بࣷڪّ لࣹڪّ اَڪْ جَامْبُرْ ݧِ دُلْ يَوُتْ؞ وَايࣹ بَ ݧࣷوݧُ دِڪّࣹي لَ رࣷݖِّڪُ، دَلْدِ بࣹدّࣹيڪُ جَامْبُرْ ݧِ، ندَخْ رَگَلْ وَا لࣵنگّ، گَ جࣵݒّ نࣹ آدَيْ خَرَفْ وَرْتࣺيفْ لَ؞ ‏13 يࣹنࣹينْ يَوُتْ يَ اِتْ دِ جِنِگَلَنْدࣷو، بَ بَرْنَبَ سَخْ لَجّ تࣷݒّ لࣹينْ ݖِ سࣹينُگْ جِنِگَلْ؞
‏14 وَايࣹ بَ مَ گِسࣹي نࣹ دࣷخࣹوُݧُ نجُبْ گِ دࣴگُّگْ خِبَارُ جࣵمّ بِ لاَجْ، مَا وَخْ سࣹفَسْ ݖِ ڪَنَمْ ݧࣺݒّ، نࣹ ڪࣷ:‏ «ندࣹگَمْ يَوْ مِيْ اَبْ يَوُتْ، يَا جࣴفࣹوُلْ نِ اَبْ يَوُتْ، خَنَا دِ جࣴفࣹ نِ خࣹيتِ جَامْبُرْ ݧِ، اَنَ لُ تَخْ نگَيْ اࣴرْتَلْ جَامْبُرْ ݧِ دُلْ اَيْ يَوُتْ، نگِرْ ݧُ يَوُتَلْ؟»
نگࣴمَ تَڪُّ
‏15 نُنْ دَنࣷو جُدُّ دِيْ يَوُتْ، تࣹ بࣷڪُّنُ ݖَ خࣹيتْ يࣹي دِيْ بࣵڪَّارْڪَتْ؞ ‏16 خَمْ نَنُ نَگْ نࣹ نِتْ ڪِ، دُ ݖِ جࣴفِ سࣵمّ ندِگَلِ يࣷونْ لَ ڪࣷ يࣵلَّ دِ جࣷخࣹ اࣵتّࣹبْ ڪُ جُبْ، خَنَا ݖِ نگࣴمْ گُ مُ گࣴمْ عِيسَىٰ اَلْمَسِ؞ تࣹ نُنْ اِتَمْ، اَلْمَسِ عِيسَىٰ لَنُ گࣴمْ، نگِرْ يࣵلَّ جࣷخْ نُ اࣵتّࣹبْ ڪُ جُبْ ݖِ ڪَوْ گࣴمْ گُ نُ گࣴمْ اَلْمَسِ، تࣹ دُ ݖِ ڪَوْ جࣴفْ جُ نُ سࣵمّࣹ ندِگَلِ يࣷونْ؞ ندَخْ ڪَتْ دُ ݖِ جࣴفِ سࣵمّ يࣷونْ لࣹيسْ دِ جࣷخࣹ ڪࣹنّ اࣵتّࣹبْ ڪُ جُبْ؞
‏17 بَ تࣹيْ سُ نُ دࣹي سࣵڪُّ اࣵتّࣹبْ ڪُ جُبْ، ندَخْ لِ نُ گࣴمْ اَلْمَسِ، تࣹ تࣹيوُلْ نِتْ ݧِ گِسْ نࣹ نُ نگِ بࣵڪَّارْ، نُنْ اِتَمْ، ندَخْ لࣷولُ دَيْ وُندّ نࣹ اَلْمَسِ مࣷويْ خِيرْتَلࣹ ݖِ بࣵڪَّارْ؟ مُڪّ ڪَيْ!‏ ‏18 ندَخَمْ سُ دࣹي تَبَخُبْ ندِگَلُ يࣷونْ بِ مَ تࣷجࣷونْ، مࣷومْ لاَ تَبَخَاتْ، مَا دࣹفْ سَمَ بࣷݒّ مࣷيْڪَتُبْ يࣷونْ؞ ‏19 سُ دࣹي وࣵلُّ يࣷونُ مُوسَىٰ نَگْ، مَنْ نِتْ ڪُ دࣹي لاَ، تࣹ يࣷونُ مُوسَىٰ مࣷو ڪࣷ وَرَلْ، نگِرْ مَ مَنَ دُندّ دُندّ گُ ݧࣹيلْ يࣵلَّ؞ دࣹيسْ مَا بࣷولࣹ اَڪْ اَلْمَسِ رࣹيْ ݖَ بَنتّ بَ؞ ‏20 مࣷو تَخْ دࣷوتُ مَنْ مَايْ دُندّ، وَايࣹ اَلْمَسِ مࣷويْ دُندّ لࣺيگِ ݖِ مَنْ؞ سَمَ دُنْدُ يَرَمُ سُوخْ وِي تࣹيْ نَگْ، ݖِ نگࣴمْ گِ مَ گࣴمْ دࣷومُ يࣵلَّ لاَ ڪࣷيْ دُنْدࣹ، مࣷومْ مِ مَ سࣷݒّ، بَ جࣺبَّلࣹ بࣷݒَّمْ نگِرْ مَنْ؞ ‏21 دُمَ نࣹينَلْ يِوُ يࣵلَّ، ندَخْ ڪَتْ سُ دࣷونْ ݖِ سࣵمّ ندِگَلُ يࣷونْ لࣹيسْ دِ اَمࣹ اࣵتّࣹبْ ڪُ جُبْ، ڪࣷنࣷونْ اَلْمَسِ مࣷو دࣹي ݖِگْ نࣹينْ؞
