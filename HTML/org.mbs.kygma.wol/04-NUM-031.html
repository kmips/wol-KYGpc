إِسْرَاءِيلْ دُمَ نَ مَجَنْ
31
‏1 اَجِ سَخْ جِ وَخَتِ مُوسَىٰ، نࣹ ڪࣷ:‏ ‏2 «فࣹخࣹيلْ بَ فࣹيُلْ بࣵنِّ إِسْرَاءِيلْ ݖِ وَا مَجَنْ¤؞ گَنَّاوْ لࣷولُ دࣹيسْ نَ لَ يٗبُّ، تࣴرَلْ لَ فَ سَيْ بࣷڪّ؞»
‏3 مُوسَىٰ دَلْدِ وَخْ اَڪْ مبࣷولࣷو مَ، نࣹ لࣹينْ:‏ «نَ ݧࣹنّ ݖِ يࣹينْ گࣵنَّايُ، وَاجَلْ خَرࣹ، تࣹ دَلْ ݖِ ڪَوْ وَا مَجَنْ، بَ فࣹيُلْ اَجِ سَخْ جِ؞ ‏4 گِي گِيرْ، جُنِّ، گࣹي گِيرْ جُنِّ، بَ گِيرِ إِسْرَاءِيلْ يࣺݒّ دَجْ، نگࣹينْ يࣹبَلْ لࣹينْ ݧُ خَرࣹجِ؞»
‏5 بَ مُ ڪࣷ دࣹفࣹي ݧُ تࣵنّࣹ ݖَ گࣵنگࣷورِ إِسْرَاءِيلْ، جُنِّيْ گٗورْ جٗگࣹ ݖِ گِيرْ گُ نࣹڪّ، ݧُيْ فُڪِّ جُنّࣹيڪْ ݧَارِ (12000) خَرࣹڪَتْ يُ گࣵنَّايُ؞ ‏6 مُوسَىٰ نَگْ يࣹبَلْ لࣹينْ، جُنِّ جٗگࣹ ݖِ گِيرْ گُ نࣹڪّ، ݧُ خَرࣹجِ؞ ݧࣷومْ اَڪْ سَرْخَلْڪَتْ بِ فِنࣹيَسْ دࣷومُ اࣹلَسَرْ، ݧࣷو خَرࣹجِ، يٗبَّالࣹ جُمْتُڪَايْ يُ سࣹلّ يَاڪْ لِيتِ خَرࣹ يَ؞ ‏7 ݧُ خَرࣹيڪْ مَجَنْ، نَ ڪࣷ اَجِ سَخْ جِ سَنْتࣹ مُوسَىٰ، بَ رࣹيْ گٗورْ ݧࣺݒّ، ‏8 رࣹيَالࣹ بُورِ مَجَنْ، اࣹوِ اَڪْ رࣹڪࣹمْ اَڪْ سُرْ اَڪْ اُرْ اَڪْ رࣹبَ، جُرٗومِ بُورِ مَجَنْ يَ؞ بَلاَمْ دࣷومُ بࣹوࣷرْ اِتْ، ݧُ رࣹيࣹ ڪࣷ سَامَرْ؞
‏9 بَ لࣷولُ اَمࣹي بࣵنِّ إِسْرَاءِيلْ جࣵݒّ جِگࣺينِ مَجَنْ، اَڪْ سࣹينِ تُوتْ-تࣵنڪّ، اَڪْ سࣹينْ جُرْ گࣺݒّ، گُ گُدّ اَڪْ گُ گࣵتّ، سࣹينْ اَلَلْ جࣺݒّ، ݧُ سࣴخࣴتࣷو ڪࣷ؞ ‏10 سࣹينْ دࣴڪّ يࣺݒّ اَڪْ سࣹينِ دࣴڪُّوَايْ، اَڪْ سࣹينْ دَلْ يࣺݒّ، ݧُ لَڪّ؞ ‏11 بَ مُ ڪࣷ دࣹفࣹي ݧُ جࣴلْ مبࣷولࣹمْ لَ ݧُ سࣴخࣴتࣷو، مبࣷولࣹمْ لَ ݧُ ݖَ جࣵݒّࣹ، نِتْ ݧَاڪْ جُرْ گَ؞ ‏12 ݧُ دَلْدِ يٗبُّ ݖَ مُوسَىٰ اَڪْ اࣹلَسَرْ سَرْخَلْڪَتْ بَ، اَڪْ مبࣷولࣷومْ بࣵنِّ إِسْرَاءِيلْ، ݧَ ݧُ جࣵݒّࣹ ݖَ خَرࣹ بَ، اَڪْ لَ ݧُ ݖَ جࣵݒّࣹ، اَڪْ لَ ݧُ سࣴخࣴتࣷو؞ ݧُ اِنْدِ لࣺݒّ ݖَ سࣹينْ دَلْ بَ ݖَ جࣷورِ مࣷوَبْ، فَ جَنࣷوڪْ يࣹرِڪࣷ، تࣹ دࣹنْدࣹيڪْ دࣹخُ يُرْدَنْ؞
‏13 ݖِ ڪَوْ لࣷولُ مُوسَىٰ اَڪْ اࣹلَسَرْ سَرْخَلْڪَتْ بَ اࣵندّ اَڪْ مبࣷولࣹمْ نجِيتِ مبࣷولࣷو مَ، ݧُ گࣺنّ دَلْ بَ، گَتَنْدُ لࣹينْ ݖَ بِتِ؞ ‏14 مُوسَىٰ نَگْ سࣵنجُ ݖَ ڪَوْ ڪِلِفَيْ خَرࣹ يَ، مُيْ ڪِلِفَيْ ڪُرࣺلِ جُنِّ يَاڪْ ڪِلِفَيْ ڪُرࣺلِ تࣺيمࣺيرْ، يَ جٗگࣹ خَرࣹ؞ ‏15 مُوسَىٰ نࣹ لࣹينْ:‏ «ڪࣷنْ جِگࣺينِ مَجَنْ ݧࣺݒّ نگࣹينْ بࣵيِّ، ݧُ دُندّ!‏ ‏16 خَنَا دُ ݧࣷومْ ݧࣷو دࣴدُّلࣷو وࣷونْ بࣵنِّ إِسْرَاءِيلْ اَجِ سَخْ جِ، ݖِ ندِگَلَلْ بَلاَمْ، ݖَ مبِرْ مَ وࣷونْ ݖَ ݒࣹوࣷرْ، بَ تَخْ مبَسْ مَ دِڪَّلْ مبࣷولࣷومْ اَجِ سَخْ جِ؟ ‏17 ڪࣷنْ نَگْ رࣹيْلࣹينْ لࣺݒّ لُيْ گٗورْ ݖِ تُوتْ-تࣵنڪّ يِ، مبࣷولࣹمْ لُ ݖِيْ جِگࣺينْ جُ خَمْ لَلُ گٗورْ اِتْ، رࣹيْ لࣹينْ، ‏18 مبࣷولࣹمْ لُ ݖِيْ جِگࣺينْ جُ ندَوْ، تࣹ خَمُلْ لَلُ گٗورْ، نگࣹينْ بࣵيِّ مُ دُندّ، ݧࣹيلْ لࣹينْ؞
‏19 «تࣹ يࣹينْ نَگْ، دَلْ-لࣹينْ ݖِ بِتِ، بُلࣹينْ دُگّ ݖِ بِيرْ دَلْ بِ دِيرُ جُرٗومْ ݧَارِ فَنْ، مبࣷولࣹمْ ݧِ رࣹيْ اَڪْ مبࣷولࣹمْ ݧِ لاَلْ ڪُ ݧُ رࣹيْ؞ نَنگࣹينْ سࣹتْلُ بࣺسُبْ ݧࣹتّࣹيلْ بَ، اَڪْ بࣺسُبْ جُرٗومْ ݧَارࣹيلْ بَ، يࣹينْ اَڪْ ݧِ نگࣹينْ جࣵݒّ؞ ‏20 تࣹ اِتَمْ لࣺݒّ لُيْ يࣺرࣹ، اَڪْ بࣺݒّ جُمْتُڪَايُ دࣹرْ، اَڪْ لࣺݒّ لࣹيسْ دࣹفَرࣹ ڪَوَرِ بࣴيْ، اَڪْ بࣺݒّ جُمْتُڪَايُ بَنتّ، نَنگࣹينْ ڪࣷ سࣹتَلْ؞»
‏21 بَ مُ ڪࣷ دࣹفࣹي اࣹلَسَرْ سَرْخَلْڪَتْ بَ وَخْ اَڪْ ݧَ خَرࣹجِ وࣷونْ، نࣹ لࣹينْ:‏ «لِي مࣷو دِ دࣷگَلُ يࣷونْ، بِ اَجِ سَخْ جِ دࣺنڪّ مُوسَىٰ:‏ ‏22 لُيْ وُرُسْ اَڪْ خَالِسْ اَڪْ خࣵنجَرْ، اَڪْ وࣹݧْ گُ ݧُولْ، اَڪْ وࣹݧْ گُ وࣹيخْ، اَڪْ بࣹتࣹيخْ، ‏23 مبࣷولࣹمْ لُ دُلْ لَڪّ، رࣹڪّ، نَنگࣹينْ ڪࣷ لاَلَلࣹ سَوَرَ، مُ سࣹتْ؞ ندࣷخُمْ سࣹتْلُ مِ اِتْ، وَرࣹيسْ نَ ڪࣷو دࣹلُّ سࣹتَلࣹ يࣷويِي؞ وَايࣹ مبࣷولࣹمْ لُيْ لَڪّ، ݖِ ندࣷخُمْ سࣹتَلْ مِ دࣷݝّ لࣹيسْ ڪࣷيْ دُگَلْ، نگِرْ مُ سࣹتْ؞ ‏24 بࣺسُبْ جُرٗومْ ݧَارࣹيلْ بَ، فٗوتْلࣹينْ سࣹينِ يࣺرࣹ، دَلْدِ سࣹتْ؞ گَنَّاوْ لࣷولُ مَنْ نگࣹينَ دُگّسِ ݖِ دَلْ بِ؞»
لُ جࣴمْ ݖِ سࣺدَّلࣹبْ اَلَلُ وَا مَجَنْ
‏25 اَجِ سَخْ جِ وَخَتِ مُوسَىٰ نࣹ ڪࣷ:‏ ‏26 «يَوْ اَڪْ اࣹلَسَرْ سَرْخَلْڪَتْ بِ، اَڪْ ݧِ جِيتࣹ سࣹينِ ڪࣴرْ مَامْ ݖِ مبࣷولࣷو مِ، وَݧّلࣹينْ اَلَلْ جِ ݧُ نَنگࣷو خَرࣹ، نِتْ اَڪُ جُرْ؞ ‏27 خَاجْلࣹينْ سࣴخࣴتࣷو مِ ݧَارْ، تࣹ نگࣹينْ سࣺدَّلࣹ ڪࣷ دِگَّنْتࣹ خَرࣹڪَتْ يَ خَرࣹجِ وࣷونْ، اَڪْ مبࣷولࣷو مِ ݖِ دࣹسْ يࣺݒّ؞ ‏28 تࣹ نگࣹينْ سࣵڪَّلْ اَجِ سَخْ جِ، ݖِ ݖࣺرُ خَرࣹڪَتْ يَ خَرࣹجِ وࣷونْ، اَبْ گَلَگْ بُ يࣹمࣷوڪْ بࣹنّ بࣷݒّ ݖِ جُرٗومِ تࣺيمࣺيرْ (500) يُ نࣹڪّ، ݖِ اَوْ نِتْ، مبَا اَيْ نَگْ، مبَا اَيْ مبَامْ، مبَا اَيْ گࣵتّ؞ ‏29 ݖِ ݖࣺرُ خَرࣹڪَتْ يِ نگࣹينْ ڪࣷيْ سࣵڪّࣹ، جࣷخْ ڪࣷ اࣹلَسَرْ سَرْخَلْڪَتْ بِ، مُيْ جࣷوخࣹبْ اَجِ سَخْ جِ؞ ‏30 ݖࣺرُ بࣵنِّ إِسْرَاءِيلْ بِ ݖِ دࣹسْ، نَنگࣹينْ ݖِ سࣵڪّ بࣹنّ ݖِ جُرٗومْ فُڪّ يُ نࣹڪّ (50)، ݖِ اَوْ نِتْ، مبَا اَيْ نَگْ، مبَا اَيْ مبَامْ، مبَا اَيْ گࣵتّ، مبَا گࣺݒّ جُرْ، تࣹ نگࣹينْ جࣷخْ ڪࣷ لࣹوࣹينْ ݧِ دࣺنڪࣷو دࣺنڪَانࣹيْ مࣵڪَّانُ اَجِ سَخْ جِ؞»
‏31 بَ لࣷولُ اَمࣹي مُوسَىٰ اَڪْ اࣹلَسَرْ سَرْخَلْڪَتْ بِ دࣹفࣹ ڪࣷ نࣷونࣹي ڪࣷ اَجِ سَخْ جِ سَنْتࣹ مُوسَىٰ؞ ‏32 لَ دࣹسْ ݖَ اَلَلُ سࣴخࣴتࣷو، جَ ݧَ خَرࣹجِ وࣷونْ اِنْدَالࣹ، گَنَّاوْ بَ ݧُ ݖَ گࣺنّࣹي يࣷويُ، نِي لَ تࣷلُّ:‏ گࣵتّ يِ، جُرٗومْ بࣹنِّ تࣺيمࣺيرِ جُنّࣹيڪْ جُرٗومْ ݧَارْ فُڪّ اَڪْ جُرٗومْ (675000)؛ ‏33 نَگْ يِ، جُرٗومْ ݧَارْ فُڪِّ جُنّࣹيڪْ ݧَارْ (72000)؛ ‏34 مبَامْ يِ، جُرٗومْ بࣹنّ فُڪِّ جُنّࣹيڪْ بࣹنّ (61000)، ‏35 اَوْ نِتْ، فَنْوࣹيرِ جُنِّيْ جِگࣺينْ ݧُ خَمُلْ لَلُ گٗورْ (30000)؞
‏36 گࣹنّ-وࣵلّ گِيْ ݖࣺرُ نِتْ ݧَ خَرࣹجِ وࣷونْ، نِي لَ تࣴدّࣹ:‏ گࣵتّ يِ، ݧࣹتِّ تࣺيمࣺيرِ جُنّࣹيڪْ فَنْوࣹيرْ اَڪْ جُرٗومْ ݧَارْ اَڪْ جُرٗومِ تࣺيمࣺيرْ (337500)، ‏37 گَلَگُ اَجِ سَخْ جِ گࣺنّࣹ ݖِ گࣵتّ يِ، دِ جُرٗومْ بࣹنِّ تࣺيمࣺيرْ اَڪْ جُرٗومْ ݧَارْ فُڪّ اَڪْ جُرٗومْ (675)؛ ‏38 نَگْ يِ، فَنْوࣹيرِ جُنّࣹيڪْ جُرٗومْ بࣹنّ (36000)، گَلَگُ اَجِ سَخْ جِ گࣺنّࣹ ݖِ، دِ جُرٗومْ ݧَارْ فُڪّ اَڪْ ݧَارْ (72)؛ ‏39 مبَامْ يِ، فَنْوࣹيرِ جُنّࣹيڪْ جُرٗومِ تࣺيمࣺيرْ (30500)، گَلَگُ اَجِ سَخْ جِ گࣺنّࣹ ݖِ، دِ جُرٗومْ بࣹنّ فُڪّ اَڪْ بࣹنّ (61)؛ ‏40 اَوْ نِتْ، فُڪِّ جُنّࣹيڪْ جُرٗومْ بࣹنّ (16000)، گَلَگُ اَجِ سَخْ جِ گࣺنّࣹ ݖِ، دِ فَنْوࣹيرِ بࣷݒّ اَڪْ ݧَارْ (32)؞ ‏41 بَ مُ ڪࣷ دࣹفࣹي مُوسَىٰ جࣷخْ گَلَگْ بَ اࣹلَسَرْ سَرْخَلْڪَتْ بَ، مُيْ جࣷوخࣹبْ اَجِ سَخْ جِ، نَ ڪࣷ اَجِ سَخْ جِ سَنْتࣹ مُوسَىٰ؞
‏42 گࣹنّ-وࣵلّ گِيْ ݖࣺرُ بࣵنِّ إِسْرَاءِيلْ، بَ مُوسَىٰ تࣹقَلࣹ اَڪْ ݖࣺرُبْ ݧَ خَرࣹجِ وࣷونْ، ‏43 تࣹ مُيْ ݖࣺرُ مبࣷولࣷو مَ، ݖِ وࣵلُّ گࣵتّ يِ، ݧࣹتِّ تࣺيمࣺيرِ جُنّࣹيڪْ فَنْوࣹيرْ اَڪْ جُرٗومْ ݧَارْ اَڪْ جُرٗومِ تࣺيمࣺيرْ (337500) لَ؛ ‏44 نَگْ يِ دِ فَنْوࣹيرِ جُنّࣹيڪْ جُرٗومْ بࣹنّ (36000)؛ ‏45 مبَامْ يِ، فَنْوࣹيرِ جُنّࣹيڪْ جُرٗومِ تࣺيمࣺيرْ (30500)؛ ‏46 اَوْ نِتْ، فُڪِّ جُنّࣹيڪْ جُرٗومْ بࣹنّ (16000)؞ ‏47 مُوسَىٰ دَلْدِ سࣵڪّ ݖِ ݖࣺرُ بࣵنِّ إِسْرَاءِيلْ بࣷوبُ، بࣹنّ ݖِ جُرٗومْ فُڪّ يُ نࣹڪّ؞ مُوسَىٰ جࣷخْ لࣷولَ لࣹوࣹينْ ݧِ دࣺنڪࣷو دࣺنڪَانࣹيْ مࣵڪَّانُ اَجِ سَخْ جِ، نࣷونࣹي ڪࣷ ڪࣷ اَجِ سَخْ جِ سَنْتࣹ وࣷونْ؞
ڪِلِفَيْ خَرࣹ يَ جࣴفࣹ نَݧُ نگࣷرْ
‏48 بَ لࣷولُ اَمࣹي نجِيتِ گࣵنگࣷورُ خَرࣹڪَتْ يَ، ڪِلِفَيْ ڪُرࣺلِ جُنِّ يَاڪْ ڪِلِفَيْ ڪُرࣺلِ تࣺيمࣺيرْ يَ، ݧࣷومْ ݧࣺݒّ دࣹمْ ݖَ مُوسَىٰ؞ ‏49 ݧُ نࣹ مُوسَىٰ:‏ «سَنگّ بِ، نُنْ دࣹ، وَݧّ نَنُ خَرࣹڪَتْ يِ ݖِ سُنُ ڪِلِفْتࣹيفْ، وَايࣹ ڪࣹنّ وُوتࣹوُ ݖِ؞ ‏50 مࣷو تَخْ نُ اِنْدِ جࣷوخࣹبْ اَجِ سَخْ جِ، ڪُ نࣹڪّ اَڪْ لَ نگَ مُورْلࣷو، ݖِ گࣵنجَرِ وُرُسْ؛ لَمِ ݒࣴرࣴگْ، اَڪْ تَڪُّبْ جَرَ، اَڪْ جَارࣷيْ تࣷرْلُڪَايْ، اَڪْ جَارࣷيْ نࣷݒّ، اَڪِ ݖَقْ؞ نࣷو ڪࣷ دࣹفْ سُنُ نجࣷتْلاَيَلْ بَڪَّنْ، فِ ڪَنَمْ اَجِ سَخْ جِ؞»
‏51 مُوسَىٰ اَڪْ اࣹلَسَرْ سَرْخَلْڪَتْ بَ نَنگࣷو وُرُسْ وَ ݖَ ݧࣷومْ، لࣺݒّ دِ تَڪَّايْ يُ ݧُ تࣴگّ؞ ‏52 مبࣷولࣹمْ وُرُسُ جࣷوخࣹ، وِ ݧُ جࣷوخࣹيلْ اَجِ سَخْ جِ، دِ فُڪّ اَڪْ جُرٗومْ بࣹنِّ جُنِّيْ سِيڪَلْ اَڪْ جُرٗومْ ݧَارْ تࣺيمࣺيرْ اَڪْ جُرٗومْ فُڪّ (16750)، يࣹمࣷوڪْ بَرِگࣷ اَڪْ جُرٗومْ ݧَارْ فُڪِّ ڪِلࣷ (170)، تࣹ بَوࣷو ݖِ ڪِلِفَيْ ڪُرࣺلِ جُنِّ يَاڪْ ڪِلِفَيْ ڪُرࣺلِ تࣺيمࣺيرْ يَ؞ ‏53 خَرࣹڪَتِ گࣵنگࣷورْ گَ ݧࣷومْ، دَݧࣷو فࣺيتࣹوࣷو لَ ݧُ سࣴخࣴتࣷو؞ ‏54 مُوسَىٰ اَڪْ اࣹلَسَرْ سَرْخَلْڪَتْ بَ نَگْ جࣴلࣹ وُرُسْ وَ ݖَ ڪِلِفَيْ ڪُرࣺلِ جُنِّ يَاڪْ ڪُرࣺلِ تࣺيمࣺيرْ يَ؞ ݧُ يٗبُّ ڪࣷ ݖَ بِيرْ خَيْمَبْ ندَجࣹ مَ، مُيْ سࣹيدࣹيلْ بࣵنِّ إِسْرَاءِيلْ فِ ڪَنَمْ اَجِ سَخْ جِ؞
