بَامْتُگْ يࣷونْ وِ
دِ جُرٗومࣹيلُ خَاجُ تَوْرࣹيتُ مُوسَىٰ
1
‏1 لِي مࣷويْ ڪࣵدُّ يِ مُوسَىٰ وَخْ بࣵنِّ إِسْرَاءِيلْ گࣺݒّ، ݖَ وࣵلاَّ دࣹخُ يُرْدَنْ، ݖَ مࣵنْدِݝْ مَ، فَ جࣷورْ گَ جࣵڪَّارْلࣷوڪْ سُفْ، دِگَّنْتࣹ ݒَرَنْ اَڪْ تࣷفࣹلْ اَڪْ لَبَنْ اَڪْ اَسࣹرࣷتْ اَڪْ دِسَيَبْ؞ ‏2 فُڪِّ فَنْ اَڪْ بࣹنّ مࣷويْ دࣷخْ دِگَّنْتࣹ تُنْدُ اࣷرࣹبْ بَ ڪَدࣹسْ بَرْنࣹيَ، بࣷو ڪࣷ جَارࣹي يࣷونُ تُنْدُ سࣹيِرْ؞
‏3 وَايࣹ ݧࣹينتّ فُڪّࣹيلُ اَتَمْ، گَنَّاوْ بَ بࣵنِّ إِسْرَاءِيلْ گࣺنّࣹي مِسْرَ، ڪࣹرࣷوگْ بࣹنّ فَنْ ݖَ فُڪّࣹيلُ وࣹيرْ وَاڪْ بࣹنّ، ݖَ لَ مُوسَىٰ دࣷورَ وَخْ بࣵنِّ إِسْرَاءِيلْ مبࣷولࣹمْ لُ ڪࣷ اَجِ سَخْ جِ سَنْتࣷونْ ݖَ ݧࣷومْ؞ ‏4 فࣹڪّ نَ مُوسَىٰ دُمَ سِوࣷنْ، مَ تࣷوگࣷونْ اࣹسْبࣷنْ، دِ بُورُبْ اَمࣷرࣹينْ ݧَ، دُمَ اِتْ ݖَ اࣹدّرࣹيْ، اࣷگْ بُورُ بَسَنْ بَ تࣷوگࣷونْ اَسْتَرࣷتْ؞
‏5 ݖَ وࣵلاَّ دࣹخُ يُرْدَنْ، ݖَ رࣺيوُمْ مࣷوَبْ لَ مُوسَىٰ تࣵمْبَلࣹي لࣹيرَلْ يࣷونُ يࣵلَّ وِي؞ مُ نࣹ:‏
 
مُوسَىٰ فࣵتّلِ نَ جَارْ-جَارِ إِسْرَاءِيلْ
(سَارْ 1‏.6—4‏.43)
‏6 سُنُ يࣵلَّ اَجِ سَخْ جِ دَ نࣷو وَخࣷونْ ݖَ اࣷرࣹبْ نࣹ نُ:‏ «سࣹينْ تࣷوگَايُبْ تُندّ وِي دࣷيْ نَ؞ ‏7 وَلْبَتِڪُلࣹينْ دࣹمْ، جࣵلّ-لࣹينْ ݖَ بِيرْ دِيوَانُ تُنْدِ اَمࣷرࣹينْ ݧَ، اَڪْ مبࣷولࣹمْ رࣺيوْ يَ مُ فࣹگّࣷولْ، ݖَ جࣷورْ گَ، اَڪْ تُندّ وَ، اَڪْ سُوفُ تُندّ يَ، اَڪْ نࣹگࣹوْ ݖَ بࣴجْ-سَالُمْ، بَ ݖَ تࣹفࣹسْ گَ، ݖَ رࣺيوُمْ ڪَنَانࣹينْ ݧَ، اَڪْ لِبَݝْ، بَ ݖَ اࣹفْرَاتْ، دࣹخْ گُ مَگْ گَ؞ ‏8 مُ نگࣷوگْ، مَا لࣹينْ جࣷخْ رࣺيوْ مࣷومُ؞ دࣹمْلࣹينْ نَنگُ رࣺيوْ مِ اَجِ سَخْ جِ گِݧَلࣷونْ سࣹينِ مَامْ إِبْرَاهِيمَ اَڪْ إِسْحَاقَ اَڪْ يَعْقُوبَ، ݧࣹيلْ لࣹينْ ݧࣷوڪْ سࣹينْ اَسْڪَنْ وِ لࣹينْ دِ وُوتُ؞»
‏9 مُوسَىٰ تࣹگْ ݖَ نࣹ لࣹينْ:‏ جَنتّ يࣷويُ لاَ لࣹينْ وَخْ، نࣹ لࣹينْ:‏ «دُمَ لࣹينْ اࣵتَّنْ يࣹينْ ݧࣺݒّ، مَنْ دࣷݝّ؞ ‏10 سࣹينْ يࣵلَّ اَجِ سَخْ جِ مࣷو لࣹينْ فُلْ، بَ يࣹينَ نگِي تࣹيْ، بَرࣹ بَ تࣷلُّ نِ بِدِّيوِ اَسَمَانْ؞ ‏11 يَلْ نَ لࣹينْ سࣹينْ يࣵلَّيْ مَامْ اَجِ سَخْ جِ يࣷڪَّتِ جُنِّيْ يࣷونْ، تࣹ بَرْڪࣹيلْ لࣹينْ نِ مُ لࣹينْ ڪࣷ وَخࣹ!‏ ‏12 وَايࣹ نُ مَ لࣹينْ دِ اࣵتَّنࣹ مَنْ دࣷݝّ، اَڪْ سࣹينْ ݖࣷونࣷ، اَڪْ سࣹينِ لࣴجْ-لࣴجْ، اَڪْ سࣹينِ جࣷتࣹ؟ ‏13 بࣹرَلْ-لࣹينْ سࣹينْ بࣷݒّ اَيْ نِتْ ݧُ رَفࣹتْ خࣹلْ، اَمُگْ دࣺگّ، تࣹ دِ ݧُ رࣵݧِّڪُ ݖِ سࣹينِ گِيرْ؞ سُ ڪࣷ دࣹفࣹي مَ تَبّ لࣹينْ، ݧُيْ سࣹينِ نجِيتْ؞» ‏14 بَ لࣷولُ اَمࣹي نگࣹينْ وَخْ مَ نࣹ مَ:‏ «لࣷولُ نگَ وَخْ دࣹ، خࣹلْ لَ؞» ‏15 ݖِ ڪَوْ لࣷولُ مَ جࣴلْ سࣹينْ نجِيتْ يُ رَفࣹتْ خࣹلْ يُ رࣵݧِّڪُ، يِ نگࣹينْ جُڪّࣹي ݖِ سࣹينِ گِيرْ، دَلْدِ لࣹينْ دࣹفْ سࣹينِ نجِيتْ، ݧِي يِلِفْ جُنِّيْ نِتْ، ݧِي تࣺيمࣺيرْ، ݧِي جُرٗومْ فُڪّ، ݧِي فُڪّ، ݧِي دِيْ جَوْرِݧْ ݖِ سࣹينِ گِيرْ؞ ‏16 جَنتّ يࣷويُ لاَ سَنتّ سࣹينِ اࣵتّࣹڪَتْ نࣹ لࣹينْ:‏ «دࣹيلࣹينْ دࣺگْلُ سࣹينِ بࣷڪّ تࣹيْ اࣵتّࣹ دࣴگّ، دِگَّنْتࣹ بࣵنِّ إِسْرَاءِيلْ اَڪْ مبࣷڪَّمْ، اَڪُبْ دࣷخَنْدࣺيمْ؞ ‏17 بُلࣹينْ فَرْ-فَرْلࣷو ݖِبْ اࣵتّࣹ، دࣺگْلُلࣹينْ ڪُ تُوتْ، نِ نگࣹينْ دِ دࣺگْلࣷو ڪُ مَگْ، تࣹ بُلࣹينْ رَگَلْ ڪࣹنّ، ندَخْ اࣵتّࣹ بِ يࣵلاَّيْ بࣷرࣷومْ؞ بُ لࣹينْ مبِرْ مِ اࣴݒّࣹي لࣷخࣷ نَگْ، نگࣹينْ اِنْدِلْ مَ ڪࣷ، مَ سࣹيتْ ڪࣷ؞» ‏18 جَنتّ يࣷويُ لاَ لࣹينْ سَنتّ مبࣷولࣹمْ لُ نگࣹينْ وَرَ دࣹفْ؞
ندِگَلُ يࣴرِ ڪَنَانْ جِبْ نَ
‏19 گَنَّاوْ گِ نُ بࣵيِّڪࣷو تُنْدُ اࣷرࣹبْ، جَارࣹ مبࣷولࣹمْ مࣵنْدِݝْ مُ يَا، مُ رَگْلُ مَ نگࣹينْ گِسْ؞ نُ تࣷݒّ يࣷونُ دِيوَانُ تُنْدِ اَمࣷرࣹينْ ݧِ، نَ نُ ڪࣷ سُنُ يࣵلَّ اَجِ سَخْ جِ سَنْتࣹ وࣷونْ، بَ اࣵگّ ڪَدࣹسْ بَرْنࣹيَ؞ ‏20 ݖِ ڪَوْ لࣷولُ مَ نࣹ لࣹينْ:‏ «اَگْسِ نگࣹينْ فِي ݖِ تُنْدُ اَمࣷرࣹينْ ݧِ نُ سُنُ يࣵلَّ اَجِ سَخْ جِ دِ جࣷخْ؞ ‏21 مُ نگࣷوگْ، سࣹينْ يࣵلَّ اَجِ سَخْ جِ مࣷو تࣹگْ رࣺيوْ مِ فِ سࣹينْ ڪَنَمْ؞ دࣹمْلࣹينْ نَنگُ ڪࣷ، نِ لࣹينْ ڪࣷ سࣹينْ يࣵلَّيْ مَامْ اَجِ سَخْ جِ، وَخࣹ؞ بُلࣹينْ رَگَلْ، بُلࣹينْ يࣷقِ؞»
‏22 ݖِ ڪَوْ لࣷولُ نگࣹينْ دِڪّ يࣹينْ ݧࣺݒّ فِ مَنْ، نࣹ مَ:‏ «نَنْ يࣹبَلْ اَوْ نِتْ، ݧُ جِيتُ نُ، يࣴرَلْ نُ رࣺيوْ مَ، تࣹ دࣺلْسِ وَخْ نُ يࣷونْ وُ نُيْ جَارࣹ، اَڪْ دࣴڪّ يُ مَگْ يَ نُيْ دُگّࣹجِ؞» ‏23 مَ رَفࣹتْلُ خࣹلْ مَ، تࣵنّ ݖِ يࣹينْ فُڪِّ گٗورْ اَڪْ ݧَارْ، گِيرْ گُ نࣹڪّ، گࣹنّ گٗورْ؞ ‏24 ݧُ وَلْبَتِڪُ جُبَلْ يࣷونُ دِيوَانُ تُندّ يَ، بَ اࣵگّ خُرُ اࣹسْڪࣷلْ، دَلْدِ سࣹيتْ فࣷوفَ، ‏25 بَ نࣷݒِّ سࣵڪّ ݖَ مࣹݧّࣹيفُمْ رࣺيوْ مَ، اِنْدَالࣹيلْ نُ؞ ݧُ وَخْ نُ نَگْ نࣹ نُ:‏ «رࣺيوْ مُ بَاخْ لَ نُ سُنُ يࣵلَّ اَجِ سَخْ جِ جࣷخْ؞» ‏26 تࣹيوُلْ بُگُّلࣹينْ وࣷونَ دࣹمْ، خَنَا دِ گࣵنْتَلْ سࣹينْ ندِگَلَلْ يࣵلَّ اَجِ سَخْ جِ؞ ‏27 نگࣹينْ دِ ݧُرُمْتࣷو سࣹينْ بِيرِ خَيْمَ، نَانْ:‏ «بَݧْ نُ ڪَيَ تَخْ اَجِ سَخْ جِ جࣴلࣹ نُ رࣺيوُمْ مِسْرَ، تࣹگْ نُ ݖِ لࣷخࣷيْ اَمࣷرࣹينْ ݧِ، نگِرْ ݧُ فَاگَاگَلْ نُ؞ ‏28 اَنَ فُ نُ جࣴمْ؟ تࣹ سُنُ بࣷڪّ يَ دࣹمࣷونْ يࣷقِلࣷو نُ، نࣹ نُ وَا رࣺيوْ مَا نُ اࣴݒّ، سُتْ نُ، دࣴڪّ يَ رࣺيْ تࣹ دࣵبّلِڪࣷو اَيْ تَتَ يُ اࣵڪِّ اَسَمَانْ؞ رَخْ ݖِ دࣷلِّ، اَيْ ݒࣷنڪَلْ يُ بࣷڪّ ݖِ اَسْڪَنُ اَنَگࣹينْ ݧِ لَنُ فَ گِسْ!‏»
‏29 مَ وَخْ لࣹينْ يࣹينْ نࣹ لࣹينْ:‏ «بُلࣹينْ يࣷخْ-يࣷخِ، بُلࣹينْ لࣹينْ رَگَلْ؞ ‏30 سࣹينْ يࣵلَّ اَجِ سَخْ جِ لࣹينْ جِيتُ، مࣷومْ مࣷو لࣹينْ دِ خࣹيخَلْ، نࣷونࣹي مُ دࣹفࣷونْ اَڪْ يࣹينْ ݖَ مِسْرَ، نگࣹينْ گِسْ، ‏31 ݖَ مࣵنْدِݝْ مَ اِتْ، گِسْ نگࣹينْ نَ لࣹينْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ فَبࣹ، نِ بَايْ دِ فَبࣹ دࣷومَمْ، ݖِ يࣷونْ وِ نگࣹينْ دࣷخْ وࣺݒّ، بَ اَگْسِ بࣺرَبْ بِي؞» ‏32 لࣷولُ لࣺݒّ تَخُلْ نگࣹينْ اَمْ ڪٗولُتࣹ ݖِ سࣹينْ يࣵلَّ اَجِ سَخْ جِ، ‏33 ڪِ لࣹينْ دَانْ جِيتُ ݖِ يࣷونْ وِ، نگِرْ سࣵڪَّلْ لࣹينْ فُ نگࣹينْ دِ دَلࣹ؛ گُدِّ، مُ نࣹڪّ ݖِ بِيرْ سَوَرَ، دِ لࣹينْ نِيتَلْ يࣷونْ وُ نگࣹينْ جَارࣹ؛ بࣴݖّࣴگْ، مُ نࣹڪَّلْ لࣹينْ ݖِ بِيرْ اَوْ نِيرْ؞
‏34 بَ اَجِ سَخْ جِ دࣺگّࣹي سࣹينِ ڪࣵدُّ، دَفَ مࣹرْ بَ گِݧْ نࣹ:‏ ‏35 «دُ ڪࣹنّ ݖِ نِتْ ݧِي، مَاسْ گُ بࣷنْ گِي، ڪُيْ گِسْ رࣺيوْ مُ بَاخْ مِ مَ گِݧَلࣷونْ سࣹينِ مَامْ؞ ‏36 خَنَا ڪَلࣹبْ دࣷومُ يࣹفُنࣹ مࣷو ڪࣷيْ گِسْ، تࣹ رࣺيوْ مَ مُ تࣹگْ تࣵنڪَمْ، مࣷومْ لاَ ڪࣷيْ جࣷخْ، مࣷوڪْ اَسْڪَنَمْ، ندَخْ مࣷو مَتَلْ ڪٗولُتࣹ ݖِ اَجِ سَخْ جِ؞» ‏37 مَنْ اِتَمْ اَجِ سَخْ جِ مࣹرࣹ نَ مَ ݖِ ندَخْ يࣹينْ، بَ نࣹ مَ:‏ «يَوْ اِتَمْ دࣷو ݖَ دُگّ؞ ‏38 يࣷسُوࣹ دࣷومُ نُونْ مِ ݖِ سَ ڪِلِفْتࣹيفْ مࣷو ݖَيْ دُگّ؞ نَنگَ ڪࣷ ݧَاخْ، ندَخْ مࣷومْ مࣷويْ جࣷخْ إِسْرَاءِيلْ رࣺيوْ مَ؞ ‏39 سࣹينْ تُوتْ-تࣵنڪّ يِ نگࣹينْ نࣷونْ دࣹيسْ نَ لࣹينْ سࣴخࣴتࣷو، سࣹينْ دࣷومْ يِي خَمَگُلْ لُ بَاخْ اَڪْ لُ بࣷنْ، ݧࣷويْ دُگِّ رࣺيوْ مَ؞ ݧࣷومْ لاَ ڪࣷيْ جࣷخْ، تࣹ ݧࣷو ڪࣷيْ نَنگُ؞ ‏40 يࣹينْ نَگْ وَلْبَتِڪُلࣹينْ دࣹلُّ مࣵنْدِݝْ مَ، تࣷݒّ يࣷونُ گࣺيجُ بَرَخْ يَ؞»
‏41 ݖِ ڪَوْ لࣷولُ نگࣹينْ وَخْ مَ، نࣹ مَ:‏ «نࣷو تࣷوݧْ اَجِ سَخْ جِ؞ وَايࣹ دِنَنُ دࣹمْ خَرࣹجِ نࣷونࣹي نُ ڪࣷ سُنُ يࣵلَّ اَجِ سَخْ جِ سَنْتࣹ وࣷونْ؞» ڪُ نࣹڪّ ݖِ يࣹينْ دَلْدِ رࣵنگࣷو نگࣵنَّايَمْ، نگࣹينْ دࣹفࣹ نࣹ يࣺيگِ دِيوَانُ تُندّ وَ دِنَ لࣹينْ يࣷمبّ؞ ‏42 اَجِ سَخْ جِ نࣹ مَ:‏ «وَخْ لࣹينْ نࣹ لࣹينْ بُݧُ دࣹمْ، بُݧُ خࣹيخِ ندَخْ مَنْ اَجِ سَخْ جِ اࣵنْدُمَاڪْ ݧࣷومْ؞ لُ ڪࣷ مࣷيْ دِنَݧُ دَانُ ݖِ سࣹينْ ڪَنَمِ نࣷونْ؞» ‏43 مَ وَخْ لࣹينْ ڪࣷ، دࣺگْلُوْلࣹينْ، خَنَا گࣵنْتَلْ ندِگَلَلْ اَجِ سَخْ جِ، فࣺتّࣹيرَلْ، بَ يࣺيگِ دِيوَانُ تُندّ وَ؞ ‏44 بَ لࣷولُ اَمࣹي اَمࣷرࣹينْ ݧَ دࣴڪّࣹ تُندّ وࣷووَ دِڪّ، دَجࣹيڪْ يࣹينْ، نࣹ لࣹينْ دَنّ نِيْ يَمبّ، يِتّࣹ لࣹينْ فَ سࣹيِرْ، بَ خࣷرْمَ؞ ‏45 نگࣹينْ وَلْبَتِڪُ، جࣷويْ ڪࣷ اَجِ سَخْ جِ، اَجِ سَخْ جِ دࣺگْلُوُ لࣹينْ، فَالࣹوُ لࣹينْ؞ ‏46 گَنَّاوْ گِ نگࣹينْ تࣷوگْ ڪَدࣹسْ، لُ يࣵگّ لَ نگࣹينْ فَ تࣷوگْ لࣺݒّ؞
