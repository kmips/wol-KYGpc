دِيݝَتْ نَݧُ سَݧْ-سَݧُ عِيسَىٰ
20
‏1 مُ اَمْ بࣺسْ بُ عِيسَىٰ دࣷونْ جࣵنگَلْ اَسْڪَنْ وِ ݖَ ڪࣴرْ يࣵلَّ گَ، دِ خَمْلࣹ خِبَارُ جࣵمّ بِ؞ سَرْخَلْڪَتْ يُ مَگْ يَ اَڪْ فِرِڪَتِ يࣷونْ يَ نࣹ جَلَݧْ، ݧࣷوڪْ نجِيتْ يَ؞ ‏2 ݧُ نࣹ ڪࣷ:‏ «وَخْ نُ، لِي نگَيْ دࣹفْ، ݖِ بَنْ سَݧْ-سَݧْ نگَ ڪࣷيْ دࣹفࣹ، مبَا ڪَنْ مࣷو لَ جࣷخْ بࣷوبُ سَݧْ-سَݧْ؟» ‏3 عِيسَىٰ نࣹ لࣹينْ:‏ «مَ لاَجْ لࣹينْ مَنْ اِتَمْ لࣹنّ، وَخْلࣹينْ مَ، ‏4 سٗوبࣹبْ يَحْيَىٰ ݖِ ندࣷخْ، اَسَمَانْ لَ ندِگَلْ لَ بَوࣷو، اَمْ ݖِ نِتْ ݧِ؟» ‏5 ݧُ دِيسࣷو ݖِ سࣹينْ بِيرْ، دَلْدِ نࣹ:‏ «سُ نُ نࣹي:‏ اَسَمَانْ لَ ندِگَلْ لَ بَوࣷو دَ نَانْ لُ تَخْ ڪࣷنْ گࣴمُنُ يَحْيَىٰ؟ ‏6 تࣹ سُ نُ نࣹي:‏ ݖِ نِتْ ݧِ لَ ندِگَلْ لِ بَوࣷو، مبࣷولࣷو مࣺݒَّ نُيْ سࣵنِّ اَيْ خࣹيرْ، ندَخْ وٗورْ نَ لࣹينْ نࣹ يَحْيَىٰ اَبْ يࣷنࣹنتّ لَ وࣷونْ؞» ‏7 ݧُ دَلْدِ نࣹ ڪࣷ:‏ «خَمُنُ فُ مُ ڪࣷ جࣴلࣹ؞» ‏8 عِيسَىٰ نࣹ لࣹينْ:‏ «ڪࣷنْ مَنْ اِتَمْ دُمَ لࣹينْ وَخْ ݖِ بَنْ سَݧْ-سَݧْ لاَيْ دࣹفࣹ لِي؞»
بࣹيْڪَتْ يَ دࣹفْ نَݧُ لُ اࣴݒّ
‏9 ݖِ ڪَوْ لࣷولُ عِيسَىٰ لࣺيبْ اَسْڪَنْ وَ، لِي، نࣹ لࣹينْ:‏ «جࣹنّ وَايْ مࣷو جࣴمْبَتࣷونْ اَبْ تٗوڪࣴرُ رࣹسࣹݧْ، بَتَلࣹ ڪࣷ اَيْ بࣹيْڪَتْ، دَلْدِ دࣹمْ يࣷونْ وُ يࣵگّ؞ ‏10 بَ تٗوڪࣴرْ بَ ݧࣷرࣹي، مُ يࣹبَلْ ݖَ بࣹيْڪَتْ يَ اَبْ سُرْگَ، نگِرْ ݧُ جࣷخْ ڪࣷ وࣵلَّمْ ݖَ مࣹݧّࣹيفْ مَ؞ بࣹيْڪَتْ يَ دٗورْ سُرْگَ بَ، بَ نࣷݒِّ دࣵقْ ڪࣷ، مُ دࣹلُّ لࣷخࣷيْ نࣹينْ؞ ‏11 مُ دࣹلُّ يٗنِّ بࣹنࣹينْ سُرْگَ، ݧُ دٗورْ ڪࣷ مࣷومْ اِتَمْ، تࣷرْخَلْ ڪࣷ، دࣵقْ ڪࣷ، مُ دࣹلُّ لࣷخࣷيْ نࣹينْ؞ ‏12 مُ دࣹلّࣷوتِ يٗنِّ ݧࣹتّࣹيلُ سُرْگَ، ڪࣷوڪُ اِتْ، ݧُ گَاݧْ ڪࣷ، سࣵنِّ ڪࣷ ݖَ بِتِ؞
‏13 «بࣷرࣷومْ تٗوڪࣴرْ بَ نࣹ:‏ ‹نَنْ لاَيْ دࣹفْ نَگْ؟ نَا يٗنِّ سَمَ دࣷومُ بࣷݒّ جِ مَ سࣷݒّ بࣷوگْ، خَنَا مࣷومْ، دِنَݧُ ڪࣷ رُسْ!‏› ‏14 تࣹيوُلْ بَ بࣹيْڪَتْ يَ گِسࣹي دࣷومْ جِ، دَݧُ نࣹ ݖِ سࣹينْ بِيرْ:‏ ‹ڪِي مࣷويْ دࣷنّ؛ نَنْلࣹينْ ڪࣷ رࣹيْ، سُ ڪࣷ دࣹفࣹي ندࣷنࣷ لِ فࣺيتࣹيڪْ نُنْ؞› ‏15 بَ لࣷولُ اَمࣹي ݧُ گࣺنّࣹ ڪࣷ تٗوڪࣴرْ بَ، رࣹيْ ڪࣷ؞»
«اَنَ نُ بࣷرࣷومْ تٗوڪࣴرْ بِيْ دࣹفْ بࣹيْڪَتْ يِ نَگْ؟ ‏16 خَنَا دِڪّ، رࣹيْ بࣹيْڪَتْ يࣷويُ، دࣺنڪّ تٗوڪࣴرْ بَ ݧࣹنࣹينْ؞» نِتْ ݧَ دࣺگّ لࣷولُ دَلْدِ نࣹ:‏ «يࣵلَّ تࣹرࣹ!‏» ‏17 عِيسَىٰ نࣹ لࣹينْ جࣵڪّ، دَلْدِ نࣹ:‏ «ڪࣷنْ نَگْ لِي ݧُ بِندّ، لُ مُيْ تࣹڪِّ؟
‹دࣷجْ وِ تَبَخْڪَتْ يِ بࣹدِّ وࣷونْ،
مࣷو مُجّ دِ دࣷجُ ݖࣷلّ وِ؞›
‏18 وࣷووُ دࣷجْ، ڪࣺݒّ ڪُ ݖِ دَلْ، دَمّتࣷو، تࣹ ڪُ مُ دَلْ، رَجَخࣹ لَ؞»
‏19 فِرِڪَتِ يࣷونْ يَاڪْ سَرْخَلْڪَتْ يُ مَگْ يَ نَگْ، ݖَ سَا سَ لَݧُ دࣷونْ فࣹخࣹي تࣹگْ عِيسَىٰ لࣷخࣷ، ندَخْ خَمࣷونْ نَݧُ نࣹ ݧࣷو مࣷومْ لࣺيبْ وࣷووُ، وَايࣹ نِتْ ݧَ لَݧُ رَگَلْ؞
بُورْ اَڪْ جَگْلࣹيمْ، يࣵلَّ اَڪْ جَگْلࣹيمْ
‏20 بَ مُ ڪࣷ دࣹفࣹي ݧُ بࣵيِّ عِيسَىٰ خࣹلْ، دَلْدِ يࣹبَلْ اَيْ يࣹيرُڪَتْ، سَسْ لࣹينْ ݧُ يِوْ-يِوْلُ تࣹ دِ ڪࣷ فِيرْ ݖِ ڪَوْ ڪࣵدُّ بَ جࣵݒّ ڪࣷ، نگِرْ جࣺبَّلْ ڪࣷ بࣷرࣷومْ دࣴڪّ بَيْ بࣷرࣷومْ ڪِلِفْتࣹيفْ گَ اَڪْ سَݧْ-سَݧْ بَ؞ ‏21 يࣹيرُڪَتْ يَ دِڪّ نࣹ عِيسَىٰ:‏ «سࣴرِݧْ بِ، خَمْ نَنُ نࣹ نجُبْ نگَيْ وَخْ، دِ ڪࣷ جࣵنگّلࣹ؞ اَمُلࣷو ݒَرْلࣵقُ اِتَمْ، وَايࣹ ݖِ ڪَوْ دࣴگّ نگَيْ جࣵنگّلࣹي يࣷونُ يࣵلَّ؞ ‏22 وَخْ نُ نَگْ ندَخْ يࣷونْ مَيࣹ نَ، نُ فࣹيْ بُورْ سࣹسَارْ گَلَگْ، اَمْ دࣺيتْ؞»
‏23 تࣹيوُلْ عِيسَىٰ رࣵݧّࣹي سࣹينْ ݒࣹخࣹ؞ مُ نࣹ لࣹينْ:‏ ‏24 «جࣷخْلࣹينْ مَ اَبْ ݒࣷسࣹتُبْ دِنَارْ، مَ گِسْ؞ نَتَالْ بِي اَڪْ تُرْ وِي ݖِ بِنْدࣹ، ڪُيْ بࣷرࣷومْ؟» ݧُ نࣹ ڪࣷ:‏ «سࣹسَارْ؞» ‏25 مُ نࣹ لࣹينْ:‏ «ڪࣷنْ ڪَيْ يࣴفِ سࣹسَارْ، جࣷخْلࣹينْ ڪࣷ سࣹسَارْ؛ تࣹ يࣴفِ يࣵلَّ، نگࣹينْ جࣷخْ ڪࣷ يࣵلَّ؞»
‏26 مَنُݧُ ڪࣷو فِيرْ بَ جࣵݒّࣹ ڪࣷ مࣹنّ مبِرْ ݖَ ڪَنَمْ مبࣷولࣷو مَ، خَنَا وَارُ ݖَ تࣷنْتَمْ، بَ وࣹدَمْ؞
سَدُسࣹنْ يَ فِيرَتِ نَݧُ عِيسَىٰ
‏27 ݧࣹنّ ݖِ سَدُسࣹنْ، يِ وࣹدِّ نࣹ ندࣹي-ندࣹڪِّ اَمْ نَ، ݧࣷو دِڪّࣷونْ ݖَ عِيسَىٰ؞ ݧُ نࣹ ڪࣷ:‏ ‏28 «سࣴرِݧْ بِ، مُوسَىٰ دَ نࣷو بِنْدَلْ نࣹ ڪُ گٗورْ، سُ ڪُ مُ بࣷڪَّلْ وَاجُرْ فَاتࣷو تࣹ اَمُلْ اَڪْ جَبَرَمْ دࣷومْ، نَ جࣴلْ جَبَرَمْ، نگِرْ اَمَلْ مبࣷڪَّمْ مَ دࣹي، ڪُوتَايْ؞ ‏29 لࣺيگِ جُرٗومْ ݧَارِ گٗورْ ݧُ بࣷڪّ ندࣹيْ اَڪْ بَايْ لَ وࣷونْ؞ تَاوْ بَ تَڪّ جَبَرْ، گَنَّاوْ گِ مُ فَاتُ، تࣹ بࣵيِّوُلْ دࣷومْ؞ ‏30 تࣷفࣷ بَ، ‏31 اَڪْ ݧَارࣹيلُ تࣷفࣷ بَ، ڪُ ݖِ نࣹڪّ دࣷنّ نَ ندَوْ سَ ݖَ مَگَمْ، ݧُ تࣷݒّلَنْتࣹي ڪࣷ نࣷونُ، بَ جُرٗومْ ݧَارْ ݧࣺݒّ دࣹي تࣹ اَمُݧُ ݖَ دࣷومْ؞ ‏32 گَنَّاوْ گِ ندَوْ سَ اِتْ دࣹي؞ ‏33 ندَوْ سِ نَگْ، ڪࣹرࣷوگْ ندࣹڪِّتࣹ لَ، ڪَنْ لَ ݖِيْ دࣷونْ جَبَرَمْ، گَنَّاوْ جُرٗومْ ݧَارْ ݧࣺݒَّ ڪࣷ جࣴلࣷونْ جَبَرْ؟»
‏34 عِيسَىٰ نࣹ لࣹينْ:‏ «ݖِ اࣵدِّنَ سِي لَ گٗورْ ݧِ اَڪْ جِگࣺينْ ݧِ دِ سࣺيْ؞ ‏35 وَايࣹ ݧَ ݧُ يࣹيَلࣹ ندࣹڪِّتࣹلْ اࣴلّࣴگْ، يࣹيَلࣹ لࣹينْ بَڪَّنْ ݖِ اࣵدِّنَ سِيْ ݧࣴوْ، ݧࣷوݧُ، سُ بࣷوبَا، ڪࣹنّ دُ ݖِ اَمْ جَبَرْ مبَا جࣴڪّࣴرْ؞ ‏36 دࣷوتُݧُ مَنَ دࣹي اِتْ، ندَخْ نِيْ مَلاَڪَ لَݧُيْ مࣹلْ؞ اَيْ دࣷومِ يࣵلَّ لَݧُيْ دࣷونْ، ندَخْ دَ لࣹينْ دِ فࣹڪّ دࣹڪِّ؞ ‏37 تࣹ مُوسَىٰ ݖِ بࣷݒَّمْ خَمْلࣹ نَ نࣹ ݧِ دࣹي دَݧُيْ دࣹڪِّ، نَ ڪࣷ تࣺيرࣹ اِنْدࣹي ݖَ مبِرُمْ نگَرَبْ سَ¤؞ مࣷويْ بَ مُ وࣷويࣹي بࣷرࣷومْ بِ:‏ ‹يࣵلَّيْ إِبْرَاهِيمَ، يࣵلَّيْ إِسْحَاقَ، يࣵلَّيْ يَعْقُوبَ؞› ‏38 ڪࣷنْ مࣷومْ دُ ݧِ دࣹي لَيْ سࣹينْ يࣵلَّ، خَنَا ݧِيْ دُندّ، ندَخْ فَ مࣷومْ، ݧࣺݒَّ نگِ دُندّ؞»
‏39 ݖِ ڪَوْ لࣷولُ ݧࣹنّ ݖَ فِرِڪَتِ يࣷونْ يَ اࣵدُّ نࣹ ڪࣷ:‏ «دࣴگّ نگَ وَخْ، سࣴرِݧْ بِ!‏»
‏40 گَنَّاوْ لࣷولُ ݧࣹمࣹيتُݧُ ڪࣷو لاَجْ دَرَ؞
اَلْمَسِ سࣴتْ لَ، سَنگّ لَ
‏41 عِيسَىٰ نࣹيتِ لࣹينْ:‏ «نࣹيسْ مَنَ وَخࣹ نࣹ اَلْمَسِ، سࣴتُبْ دَاوُدَ لَ؟ ‏42 دَاوُدَ ݖِ بࣷݒَّمْ مࣷو نࣹ ݖِ تࣺيرࣹبْ سَبٗورْ:‏
‹بࣷرࣷومْ بِ مࣷو نࣹ سَمَ سَنگّ:‏
«تࣷوگࣹيلْ سَمَ ندِجࣷورْ،
‏43 بَ مَ دࣹفْ سَيْ نࣷونْ، سَ ندࣴگَّسْتَلْ¤؞» ›
‏44 گَنَّاوْ دَاوُدَا وࣷويࣹ اَلْمَسِ ‹سَنگّ بِ،› نُ مُ نࣹڪّࣹ اَبْ سࣴتَمْ؟»
‏45 عِيسَىٰ نَگْ وَخْ اَڪْ اَيْ تَالِبࣹيمْ، اَسْڪَنْ وࣺݒّ دِ دࣺگّ؞ مُ نࣹ:‏ ‏46 «مࣷيْتُلࣹينْ فِرِڪَتِ يࣷونْ يِ تَامࣷو جَاگَرْ-جَاگَرࣹيڪْ سࣹينْ مبُبّ يُ رࣺيْ، دِ سࣵڪُّ نُيࣷوبْ گࣴدَّالْ ݖَ ݒࣺنݖّ مَ، اَڪْ تࣷوگُ يُ ڪَنَمࣹ ݖَ جࣵنگُ يَ، اَڪْ ݒࣺيتࣹيْ تࣹرَانگَ ݖَ بࣹرْندࣹ يَ؞ ‏47 ݧࣷويْ سَاخْ-سَاخࣹي اَلَلُ جࣴتُنْ ݧِ تࣹيْ جُلِّ جُلِّيْ نگِسْتَلْ يُ دُلْ جࣹيخْ؞ ݧࣷوݧُ لَ سࣹينْ مبُگَلْ دِ گࣴنَ تَرْ؞»
