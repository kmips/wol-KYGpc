يَعْقُوبَ دࣹمْ نَ مِسْرَ
46
‏1 إِسْرَاءِيلْ نَگْ دࣹمْ، يٗبَّالࣹ مبࣷولࣹمْ لُ بࣷڪّ ݖِ مࣷومْ؞ بَ مُ اَگْسࣹي بࣹيرْسࣹبَ، دَفَ سَرْخَلْ يࣵلَّيْ بَايَمْ إِسْحَاقَ؞ ‏2 يࣵلَّ وَخْ اَڪْ إِسْرَاءِيلْ فࣷوفَ ݖَ گُدِّ ݖِ بِيرْ گࣺنتّ نࣹ ڪࣷ:‏ «يَعْقُوبَ، يَعْقُوبَ!‏» إِسْرَاءِيلْ نࣹ ڪࣷ:‏ «نَامْ؞» ‏3 يࣵلَّ نࣹ ڪࣷ:‏ «مَنْ مَايْ يࣵلَّ، دِ سَ يࣵلَّيْ بَايْ؞ بُلْ رَگَلَ دࣹمْ مِسْرَ، ندَخْتࣹ اَسْڪَنْ وُ رࣺيْ لاَ فَيْ سࣷسࣹ ݖِ يَوْ؞ ‏4 مَنْ ݖِ سَمَ بࣷݒّ مَايْ اࣵندّ اَڪْ يَوْ مِسْرَ، تࣹ نَ لَ وٗورْ نࣹ دِنَا لَ دࣺلّࣷوسِ، تࣹ اِتْ يُوسُفَ مࣷو لَيْ دࣷگَّلِ؞»
‏5 بَ لࣷولُ اَمࣹي يَعْقُوبَ جٗگࣹ بࣹيرْسࣹبَ؛ دࣷومِ إِسْرَاءِيلْ بࣷولࣹ سࣹينْ بَايْ اَڪْ سࣹينِ دࣷومْ اَڪْ سࣹينِ سࣷخْنَ، يࣹبْ لࣹينْ ݖِ وَتِيرْ، يِ فِرْعَوْنَ يٗنّࣹي وࣷونْ، نگِرْ يٗبُّ لࣹينْ؞ ‏6 ݧَ نگَ جِيتَلْ سࣹينِ جُرْ، دَلْدِ يٗبَّالࣹ سࣹينْ اَلَلْ، جَ ݧُ اَمࣹ رࣺيوُ ڪَنَانْ؞ نࣷونُ لَ يَعْقُوبَ دࣹمࣹ مِسْرَ اࣵندّ اَڪْ نجَبࣷوتَمْ گࣺݒّ، ‏7 يٗبَّالࣹ دࣷومَمْ يُ گٗورْ يَ اَڪْ سࣴتَمْ يُ گٗورْ يَ اَڪْ دࣷومَمْ يُ جِگࣺينْ يَ اِتْ، ݧࣷوڪْ سࣴتَمْ يُ جِگࣺينْ يَ، اَڪْ مبࣷولࣹمْ ݧَ اَسْڪَنࣷو ݖَ مࣷومْ؞
نجَبࣷوتُ يَعْقُوبَ، گِ ݧࣴوْ مِسْرَ
‏8 وَا ڪࣴرْ إِسْرَاءِيلْ، يِ ݧࣴوْ مِسْرَ تࣹ دِ يَعْقُوبَ اَڪِ دࣷومَمْ، سࣹينِ تُرَ دِ:‏
رُبࣹنْ، تَاوُبْ يَعْقُوبَ؞ ‏9 دࣷومِ رُبࣹنْ يُ گٗورْ دِ اَنࣷگْ اَڪْ ݒَلُ اَڪْ اࣹسْرࣷنْ اَڪْ ڪَرْمِ؞ ‏10 دࣷومِ سِمࣹيࣷنْ يُ گٗورْ دِ يࣹمُوࣹلْ اَڪْ يَمِنْ اَڪْ اࣷوࣵدّ اَڪْ يَڪِنْ اَڪْ سࣷوَرْ اَڪْ سَوُلْ، مِ ندࣹيَمْ دِ ڪَنَانࣹينْ؞ ‏11 دࣷومِ لࣹوِ يُ گٗورْ دِ گࣹرْسࣷنْ، ڪࣹيَتْ اَڪْ مࣹرَرِ؞ ‏12 دࣷومِ يُدَ يُ گٗورْ دِ اࣹرْ اَڪْ اࣷنَنْ اَڪْ سࣹلَ اَڪْ ݒࣹرࣹسْ اَڪْ سࣹرَ، تࣹ اࣹرْ اَڪْ اࣷنَنْ فَاتُ ݖَ رࣺيوُ ڪَنَانْ؞ دࣷومِ ݒࣹرࣹسْ يُ گٗورْ دِ اࣹسْرࣷنْ اَڪْ اَمُلْ؞ ‏13 دࣷومِ اِسَاڪَرْ يُ گٗورْ دِ تࣷلَ اَڪْ ݒُوَ اَڪْ يَسُبْ اَڪْ سِمْرࣷنْ؞ ‏14 دࣷومِ سَبُلࣷنْ يُ گٗورْ دِ سࣹرࣹدّ، اࣹلࣷنْ اَڪْ يَلࣹيلْ؞ ‏15 گٗورْ ݧࣷوݧࣷو خࣹيتࣷو ݖِ لࣹيَ، تࣹ اَسْڪَنࣷو ݖِ يَعْقُوبَ ݖَ ݒَدَنْ اَرَمْ، گَنَّاوْ دِينَ مِيْ دࣷومَمْ جُ جِگࣺينْ؞ بࣷو بࣷولࣹي دࣷومَمْ يࣺݒّ، گٗورْ اَڪْ جِگࣺينْ، ݧُيْ فَنْوࣹيرْ اَڪْ ݧࣹتّ؞
‏16 دࣷومِ گࣵدّ يُ گٗورْ ݧࣷو دِ ݖࣹفࣷنْ اَڪْ اَگِ اَڪْ سُنِ اَڪْ اࣹسْبࣷنْ اَڪْ اࣹرِ اَڪْ اَرࣷدِ اَڪْ اَرࣹلِ؞ ‏17 دࣷومِ اَسࣹرْ يُ گٗورْ دِ اِمْنَ اَڪْ يِسْوَ اَڪْ اِسْوِ اَڪْ بࣹرْيَ، سࣹرَ دِ سࣹينْ جِگࣺينْ؞ دࣷومِ بࣹرْيَ يُ گٗورْ دِ اࣹبࣹرْ اَڪْ مَلْڪِيࣹلْ؞ ‏18 گٗورْ ݧࣷوݧࣷو خࣹيتࣷو ݖِ سِلْݒَ، مِ لَبَنْ مَيْ لࣹيَ، دࣷومَمْ جُ جِگࣺينْ، تࣹ ݧُ اَسْڪَنࣷو ݖِ يَعْقُوبَ، ݧُيْ فُڪِّ بَڪَّنْ اَڪْ جُرٗومْ بࣹنّ؞
‏19 دࣷومْ يُ گٗورْ يِ رَسࣹلْ، سࣷخْنَسْ يَعْقُوبَ، اَمْ نَگْ ݧࣷو دِ يُوسُفَ اَڪْ بࣹݧَمِنْ؞ ‏20 يُوسُفَ اِتَمْ اَمْ مَنَسࣹ اَڪْ اࣹفْرَيِمْ ݖَ رࣺيوُمْ مِسْرَ، سࣹينْ يَايْ دِ اَسْنَتْ دࣷومُ ݒࣷتِفࣹرَ، مَيْ نجِيتُ دِينࣹ ݖَ دࣴڪّ بَ ݧُ نَانْ اࣷنْ؞ ‏21 دࣷومِ بࣹݧَمِنْ يُ گٗورْ ݧࣷو دِ بࣹلَ اَڪْ بࣹڪࣹرْ اَڪْ اَسْبࣹلْ اَڪْ گࣹرَ اَڪْ نَامَنْ اَڪْ اࣹيِ اَڪْ رࣷسْ اَڪْ مُݒِّمْ اَڪْ اُݒِّمْ اَڪْ اࣵردّ؞ ‏22 گٗورْ ݧࣷوݧࣷو خࣹيتࣷو ݖِ رَسࣹلْ، تࣹ اَسْڪَنࣷو ݖِ يَعْقُوبَ، ݧࣺݒّ دِ فُڪِّ بَڪَّنْ اَڪْ ݧࣹينتّ؞
‏23 دࣷومُ دَنْ جُ گٗورْ دِ اُسِمْ؞ ‏24 دࣷومِ نࣹفْتَلِ يُ گٗورْ دِ يࣵݖّࣹيلْ اَڪْ گُنِّ اَڪْ يࣹݖّࣹرْ اَڪْ سِلࣹمْ؞ ‏25 گٗورْ ݧࣷوݧࣷو خࣹيتࣷو ݖِ بِلاَ، مِ لَبَنْ مَيْ دࣷومَمْ رَسࣹلْ، تࣹ ݧُ اَسْڪَنࣷو ݖِ يَعْقُوبَ، ݧࣷومْ ݧࣺݒّ دِ جُرٗومْ ݧَارِ بَڪَّنْ؞
‏26 مبࣷولࣹمْ ݧِ اࣵندّ اَڪْ يَعْقُوبَ، ݧࣴوْ مِسْرَ، مُيْ ݧِ مُ گِيرْ، تࣹ لِمَالࣹيسُلْ سࣷخْنَيْ دࣷومِ يَعْقُوبَ يَ، تࣷلُّ نَݧُ ݖِ جُرٗومْ بࣹنّ فُڪِّ بَڪَّنْ اَڪْ جُرٗومْ بࣹنّ؞ ‏27 دࣷومِ يُوسُفَ يُ گٗورْ، يَ مُ اَمْ مِسْرَ، تࣷلُّ ݖِ ݧَارْ؞ ڪࣷنْ مبࣷولࣹمْ ݧَ دࣹمْ مِسْرَ تࣹ دِ وَا ڪࣴرْ يَعْقُوبَ، تࣷلُّ نَݧُ ݖِ جُرٗومْ ݧَارْ فُڪّ؞
يَعْقُوبَ گِسْ نَ يُوسُفَ
‏28 يَعْقُوبَ دَفَ يࣹبَلْ يُدَ، مُ اࣵندّ اَڪْ يُوسُفَ، جِيتُجِ ڪࣷ دِيوَانُ گࣷسࣹنْ، يَعْقُوبَ فࣹڪِّ لࣹينْ فَ؞ ݧُ دَلْدِ اَگْسِ گࣷسࣹنْ؞ ‏29 يُوسُفَ نَگْ وَاجَلْ وَرُوَايَمْ، دَجࣹجِ اَڪْ بَايَمْ إِسْرَاءِيلْ ݖَ گࣷسࣹنْ؞ بَ يُوسُفَ اࣵگّࣹي بَ ݖَ مࣷومْ، دَ ڪࣷو لَخَسُ، نࣹ ݧࣵݒّ ݖِ ڪَوَمْ، جࣷويْ لُ يࣵگّ؞ ‏30 إِسْرَاءِيلْ نࣹ يُوسُفَ:‏ «لࣺيگِ دࣹ، بُ مَ يࣹبࣷو دࣹي، گَنَّاوْ گِسَاتْ نَا لَ، نگَيْ دُندّ؞»
‏31 بَ مُ ڪࣷ دࣹفࣹي يُوسُفَ وَخْ اَڪْ دࣷومِ بَايَمْ يَ اَڪْ وَا ڪࣴرْ بَايَمْ نࣹ لࣹينْ:‏ «مَا نگِ دࣹلُّ يࣺگَلِ فِرْعَوْنَ نࣹ ڪࣷ:‏ ‹سَمَيْ بࣷڪّ اَڪْ سَمَ وَا ڪࣴرْ بَايْ، ݧَ نࣹڪّࣷونْ ݖَ رࣺيوُ ڪَنَانْ، ݧࣴوْ نَݧُ فِ مَنْ؞ ‏32 دَفَ دِ سࣹينِ گٗورْ اَيْ سࣵمّ لَݧُ، اَگْ جُرْ لَݧُ مَسَ يَرْ، تࣹ اِنْدَالࣹ نَݧُ سࣹينْ جُرْ گُ گࣵتّ اَڪْ گُ گُدّ اَڪْ مبࣷولࣹمْ لُ ݧُ مࣷومْ؞› ‏33 ڪࣷنْ بُ لࣹينْ فِرْعَوْنَ وࣷويࣹي، نࣹ لࣹينْ:‏ ‏34 ‹لَنْ مࣷويْ سࣹينْ لِگّࣺييْ؟› نگࣹينْ نࣹ ڪࣷ:‏ ‹نُنْ دࣹ سَنگّ بِ، اَگْ جُرْ لَنُ مَسَ يَرْ، بَ نُيْ گࣷنࣹ بَ تࣹيْ، تࣹ نࣷونُ لَ ڪࣷ سُنُيْ مَامْ دَانْ دࣹفࣹ؞› سُ بࣷوبَا، نگࣹينْ مَنَ فࣺيتࣹوࣷو دِيوَانُ گࣷسࣹنْ، ندَخْتࣹ وَا مِسْرَ دَݧُيْ خࣷفْ سࣵمّ؞»
