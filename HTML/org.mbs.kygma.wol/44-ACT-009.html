سٗولْ تُوبْ نَ
9
‏1 ݖِ بِيرْ لࣷولُ سٗولْ مࣷومْ، نࣷيّࣹيوُلْ لُ مࣷيْ دِ تࣴڪّࣷوڪَ جࣺيمَ بٗومْ تَالِبࣹيْ سَنگّ بِ؞ مُ دࣹمْ نَگْ ݖَ سَرْخَلْڪَتْ بُ مَگْ بَ، ‏2 دِ سࣵڪُّ اَيْ بَتَاخَلْ يُ ݧࣹيلْ جࣵنگُيْ دࣴڪّ بَ ݧُيْ وَخْ دَمَاسْ، بَتَاخَلْ يُ يٗبَّنْتࣹ ندِگَلْ لُ يࣹيوْلُ ڪࣺݒّ ڪُ ݧُ فَ گِسْ ڪُ بࣷڪّ ݖِ يࣷونْ وࣷووُ، گٗورْ اَڪْ جِگࣺينْ، نگِرْ مُ جࣵݒّ لࣹينْ، اِنْدِ يࣹرُسَلࣹمْ؞
‏3 نَڪَ لَ سٗولْ دࣹمْ بَ جُبْ دَمَاسْ، اَگْ لࣹيرْ بَوࣷو اَسَمَانْ، جࣹڪِّ نࣹ رࣵيّ ݖِ ڪَوَمْ؞ ‏4 مُ دَانُ ݖِ سُوفْ، دَلْدِ دࣺگّ بَاتْ بُ ڪࣷ نࣹ:‏ «سٗولْ، سٗولْ، لُ تَخْ نگَ دِ مَ بُندّخَتَلْ؟» ‏5 مُ نࣹ:‏ «يَايْ ڪَنْ سَنگّ بِ؟» مُ نࣹ:‏ «مَنْ مَايْ عِيسَىٰ، مِ نگَيْ بُندّخَتَلْ؞ ‏6 لࣺيگِ نَگْ، جٗگَلْ، دُگّ ݖِ دࣴڪّ بِ، دࣹيسْ نَ لَ وَخْ لِ نگَ وَرَ دࣹفْ؞» ‏7 نِتْ ݧَ اࣵندّ اَڪْ مࣷومَ نگَ تَخَوْ، جࣷمِّ بَ نࣹ ݖࣹلّ؛ دࣺگّ نَݧُ بَاتْ بِ تࣹ گِسُݧُ ڪࣹنّ؞ ‏8 سٗولْ جٗگْ، خِݒِّ، گِسُلْ دَرَ؛ ݧُ دَلْدِ ڪࣷيْ وࣷمَّتْ، دُگَلْ ڪࣷ بِيرْ دَمَاسْ؞ ‏9 دِيرُ ݧࣹتِّ فَنْ گِسُلْ دَرَ، لࣹڪُّلْ، نَانُلْ؞
‏10 اَبْ تَالِبࣹ بُ ݧُيْ وَخْ اَنَݧَسْ نَگْ مَ نگَ وࣷونْ ݖَ دَمَاسْ؞ سَنگّ بِ فࣹيݧُ ڪࣷ، نࣹ ڪࣷ:‏ «اَنَݧَسْ!‏» مُ نࣹ ڪࣷ:‏ «مَا نگِ، سَنگّ بِ؞» ‏11 سَنگّ بِ نࣹ ڪࣷ:‏ «دࣹمَلْ بَ ݖَ مبࣹدّ، مَ ݧُيْ وَخْ مبࣹدّ مُ جُبْ، تࣹ نگَ سࣹيتْ ݖَ ڪࣴرْ يُدَ ڪُ ݧُيْ وَخْ سٗولْ تࣹ دࣴڪّࣹ تࣵرسّ؞ مَ نگَيْ ݧَانْ؞ ‏12 دَفَ گِسْ ݖِمْ ݒࣹيݧُ ڪُ ݧُيْ وَخْ اَنَݧَسْ دُگّسِ، تࣹگْ ڪࣷ لࣷخࣷ، نگِرْ مُ دࣹلُّ دِ گِسْ؞» ‏13 اَنَݧَسْ نَگْ نࣹ ڪࣷ:‏ «سَنگّ بِ، مَا دࣺگّࣹ ݖِ ݧُ بَرࣹ مبِرُمْ ڪࣷوڪُ، ݧُ سࣹيدࣹ مبࣷولࣹمْ جࣴفْ جُ بࣷنْ جِ مُ دࣹفْ سَيْ نِتْ ݧُ سࣹلّ ݖِ بِيرْ يࣹرُسَلࣹمْ؞ ‏14 تࣹ مࣷو يࣷرْ ندِگَلْ لُ تُڪّࣹي ݖِ سَرْخَلْڪَتْ يُ مَگْ يِ، نگِرْ يࣹيوْلُ ڪࣺݒّ ڪُيْ تُدّ سَوْ تُرْ؞» ‏15 سَنگّ بِ نࣹ ڪࣷ:‏ «دࣹمَلْ، ندَخْ ڪࣷوڪُ جُمْتُڪَايْ لاَ ڪࣷ تࣵنّࣹ، نگِرْ مُ خَمَلِ سَمَ تُرْ، خࣹيتِ جَامْبُرْ يِ اَڪْ سࣹينِ بُورْ، خَمَلْ ڪࣷ اِتَمْ بࣵنِّ إِسْرَاءِيلْ؞ ‏16 مَا ڪࣷيْ وࣷنْ مبࣷولࣹمْ ݖࣷونࣷ بِ مُ وَرَ دࣺڪُّ نگِرْ سَمَ تُرْ؞»
‏17 ݖِ ڪَوْ لࣷولُ اَنَݧَسْ دࣹمْ، دُگّ ݖَ ڪࣴرْ گَ، تࣹگْ سٗولْ لࣷخࣷ، نࣹ ڪࣷ:‏ «مبࣷڪّ مِ سٗولْ، سَنگّ عِيسَىٰ، مِ لَ فࣹيݧُ وࣷونْ ݖِ يࣷونْ وِ نگَ دِڪّࣹ، مࣷو مَ يࣹبَلْ ݖِ يَوْ، نگِرْ نگَ دࣹلُّ دِ گِسْ تࣹ فࣹيسࣹ نࣷو گُ سࣹلّ گِ؞» ‏18 ݖَ سَا سَ لُ مࣹلْ نِيْ وَاسِنْتٗورْ وَدّࣹ ݖَ بࣴتْ يَ، مُ دࣹلُّ دِ گِسْ؞ مُ دَلْدِ جٗگْ، ݧُ سٗوبْ ڪࣷ ݖِ ندࣷخْ؞ ‏19 بَ لࣷولُ وࣺيࣹي، مُ لࣹڪّ، دࣷورَ اَمْ دࣷولࣹ؞
سٗولْ تࣷوگْ نَ اَڪْ تَالِبࣹ يَ ݖَ دَمَاسْ اَيْ فَنْ؞ ‏20 ݖَ سَا سَ لَ تࣵمْبَلࣹي خَمْلࣹ عِيسَىٰ ݖَ جࣵنگُ يَ، نَانْ عِيسَىٰ مࣷويْ دࣷومُ يࣵلَّ؞ ‏21 مبࣷولࣹمْ ݧَ ڪࣷ دࣺگّ، دَلْدِ وَارُ، نَانْ:‏ «خَنَا دُ ڪِي مࣷو دࣷونْ بٗومْ ݧِيْ تُدّ وࣷووُ تُرْ ݖِ يࣹرُسَلࣹمْ؟ دُ مࣷو فِ دِڪّࣷونْ، نگِرْ يࣹيوْلُ لࣹينْ، يٗبُّ ݖِ سَرْخَلْڪَتْ يُ مَگْ يِ؟» ‏22 تࣹيوُلْ سٗولْ گࣴنَ اَمْ مَنࣷورࣹ، بَ فَرْ جَاخَلْ يَوُتْ ݧِ دࣴڪّࣹ دَمَاسْ، دِ فِرْندࣹيلْ نࣹ عِيسَىٰ مࣷويْ اَلْمَسِ؞
‏23 بَ ݧُ ݖَ تࣹگࣹي اَبْ دِيرْ بُ خَوَ يࣵگّ، يَوُتْ يِ فࣺنݖࣷو، نگِرْ رࣹيْلُ ڪࣷ؞ ‏24 سٗولْ نَگْ يࣺگْ سࣹينْ مࣺبࣺتْ؞ گُدِّ اَڪْ بࣴݖّࣴگْ لَݧُ دࣷونْ وَتُّ بُنْتِ دࣴڪّ بَ، نگِرْ بٗومْ ڪࣷ؞ ‏25 تࣹيوُلْ تَالِبࣹ يِ يࣹبْ ڪࣷ اَگْ گُدِّ ݖِ اَگْ دࣵمْبَ، جࣵلّࣹ ڪࣷ ݖَ وࣵلاَّ مِيرْ بَ، يࣷورْ ڪࣷ؞
‏26 بَ سٗولْ اَگْسࣹي يࣹرُسَلࣹمْ، جࣺيمْ نَا جَخَسࣷوڪْ تَالِبࣹ يِ، وَايࣹ ݧࣺݒَّ ڪࣷ رَگَلْ، ندَخْ ݧࣵڪَّ گࣴمْ اَگْ تَالِبࣹيمْ؞ ‏27 ݖِ ڪَوْ لࣷولُ بَرْنَبَ اࣵندّ اَڪْ مࣷومْ، يٗبُّ ڪࣷ ݖَ ندَوْ يَ؞ بَرْنَبَ نࣹتَّلِ لࣹينْ، نِ سٗولْ گِسࣹ سَنگّ بِ ݖَ يࣷونْ وَ، بَ مُ وَخْ اَڪْ سٗولْ، اَڪْ فِتْ وَ سٗولْ وَارࣹي ݖَ دَمَاسْ ݖِ تُرُ عِيسَىٰ؞ ‏28 بَ لࣷولُ اَمࣹي سٗولْ اࣵندّ اَڪْ ندَوْ يِ دِ دࣹمْ اَڪَ دِڪّ ݖِ بِيرْ يࣹرُسَلࣹمْ، دِ وَارࣹي اَوْ فِتْ ݖِ تُرُ سَنگّ بِ؞ ‏29 مُ بࣷولࣹ ݖِ دِ وَخْ اَڪَ وࣹرَنْتࣹ اَڪْ يَوُتْ ݧِيْ لࣵڪّ گࣹرࣹگْ، تࣹ ݧࣷومْ ݧُ دِ ڪࣷ فࣹخࣹي رࣹيْ؞ ‏30 بࣷڪِّ گࣴمْڪَتْ ݧَ نَگْ يࣺگْ ڪࣷ، دَلْدِ ڪࣷيْ يٗبُّ سࣹسَرࣹ، يࣹبَلࣹ ڪࣷ فࣷوفَ، مُ دࣹمْ تࣵرسّ؞
‏31 بَ مُ ڪࣷ دࣹفࣹي مبࣷولࣷومْ گࣴمْڪَتْ ݧِ ݖِ مبࣷولࣹمْ دِيوَانِ يُدࣹ اَڪْ گَلِلࣹ اَڪْ سَمَرِ دَلْدِ اَمْ جࣵمّ؞ ݧُيْ گࣴنَ دࣴگࣴرْ، دِ وࣺيࣹ رَگَلْ يࣵلَّ، تࣹيْ يࣷڪُّ ݖِ ندِمْبَلَلْ نࣷو گُ سࣹلّ گِ؞
ݒِيࣹيرْ فَجْ نَ نِتْ
‏32 ݒِيࣹيرْ مِ دࣷونْ وࣴرْ فُ نࣹڪّ نَگْ، اَمْ نَ بࣺسْ مُ جَارࣹ ݖَ ݧُ سࣹلّ ݧَ دࣴڪّࣹ لِدّ؞ ‏33 مُ گِسْ فࣷوفَ وَا جُ ݧُيْ وَخْ اࣹنࣹ؞ فࣹڪّ نَ دِيرُ جُرٗومْ ݧࣹتِّ اَتْ، ݖِبْ لَلْ رࣹڪّ لَيْ تࣴدّ ندَخْ يَرَمْ وُ لࣵگِّ؞ ‏34 ݒِيࣹيرْ نࣹ ڪࣷ:‏ «اࣹنࣹ، عِيسَىٰ اَلْمَسِ وࣺرَلْ نَ لَ؛ جٗگَلْ، لَلَلْ سَ بࣷݒّ؞» مُ جٗگْ ݖَ سَا سَ؞ ‏35 مبࣷولࣹمْ وَا دِيوَانِ لِدّ اَڪْ سَرࣷنْ نَگْ گِسْ ڪࣷ، ݧࣷومْ ݧࣺݒّ وَݧِّڪُ ݖِ سَنگّ بِ؞
ݒِيࣹيرْ دࣹڪَّلْ نَ نِتْ
‏36 دࣴڪّ بَ ݧُيْ وَخْ يࣷݒࣹ، اَبْ تَالِبࣹ بُ جِگࣺينَ نگَ فَ وࣷونْ، ݧُ دِ ڪࣷ وَخْ تَبِتَ مبَا دࣷرْڪَسْ ݖِ لࣵڪُّ گࣹرࣹگْ؞ دࣹفْ لُ بَاخْ اَڪْ سَرْخࣹ لَ سَخࣷو وࣷونْ؞ ‏37 ݖِ فَنْ يࣷويُ، مُ دَانُ وࣷݒّ، دَلْدِ فَاتُ؞ ݧُ سَنگّ ڪࣷ، دُگَلْ ڪࣷ ݖَ نࣺيگُ ڪَوْ تَاخْ مَ؞ ‏38 ݖِ بِيرْ لࣷولُ تَالِبࣹ يِ دࣺگّ نࣹ ݒِيࣹيرَ نگَ لِدّ، تࣹ لِدّ سࣷرࣹوُلْ يࣷݒࣹ؞ ݧُ يࣹبَلْ ݧَارِ نِتْ ݖَ مࣷومْ، نگِرْ ݧَانْ ڪࣷ مُ ݧࣴوْ تࣹ بَݧَ يࣹيخْ؞
‏39 ݒِيࣹيرْ جٗگْ، اࣵندّ اَڪْ ݧࣷومْ، بَ اَگْسِ؞ ݧُ يࣺيگࣹ ڪࣷ ݖَ نࣺيگُ ڪَوْ تَاخْ مَ؞ اَيْ جࣴتُنَ نگَ فَ وࣷونْ؞ ݧࣷومْ ݧࣺݒّ دِ جࣷويࣷو، يࣺيوْ ݒِيࣹيرْ، وࣷنْ ڪࣷ تُرْڪِ يَاڪْ مبُبّ يَ دࣷرْڪَسْ دࣹفَرࣷونْ بَ مُيْ نࣹڪّ اَڪْ ݧࣷومْ؞
‏40 ݖِ ڪَوْ لࣷولُ ݒِيࣹيرْ دࣵقْ ݧࣺݒّ ݖَ بِتِ؞ مُ سُڪّ، ݧَانْ، دَلْدِ وَلْبَتِڪُ جُبْلࣷوڪْ نࣺيوْ بِ، نࣹ:‏ «تَبِتَ، جٗگَلْ!‏» مُ نࣹ خِفࣺيتْ، گِسْ ݒِيࣹيرْ، جٗگْ تࣷوگْ؞ ‏41 ݒِيࣹيرْ مَيْ ڪࣷ لࣷخࣷ، يࣺڪَّتِ ڪࣷ؞ بَ لࣷولُ اَمࣹي ݒِيࣹيرْ وࣷو ݧُ سࣹلّ ݧَ، اَڪْ جࣴتُنْ ݧَ، وࣷنْ لࣹينْ ندَوْ سَ، مُيْ دُندّ؞ ‏42 مبِرْ مَ نَگْ سِيوْ ݖَ يࣷݒࣹ گࣺݒّ، ݧُ بَرࣹ گࣴمْ سَنگّ بِ؞ ‏43 گَنَّاوْ لࣷولُ ݒِيࣹيرْ دَلْ نَ يࣷݒࣹ اَيْ فَنْ، ڪࣴرْ وُلِّڪَتْ بُ ݧُيْ وَخْ سِمࣷݝْ؞
