نجُبْ اَڪْ يࣴرْمَنْدࣹ وَرْتࣺيفْ لَ
25
‏1 سُ اَبْ جࣷتࣹ اَمࣹي دِگَّنْتࣹ اَيْ نِتْ، نَݧُ دࣹمْ ݖَ اࣴتُّبْ اࣵتّࣹڪَايْ بَ، ݧُ اࣵتّࣹ لࣹينْ، بَ ڪِ اَمْ دࣴگّ، ݧُ جࣷخْ ڪࣷ دࣴگّ، ڪِ تࣷوݧْ، ݧُ تࣹگْ ڪࣷ تࣷوݧْ؞ ‏2 سُ دࣹي اَيْ دٗورْ لَ ڪِ تࣷوݧْ يࣹلّࣷو، نَ ڪࣷ اࣵتّࣹڪَتْ بِ تࣴرَلْ، ݧُ دٗورْ ڪࣷ ݖِ ڪَنَمَمْ، لِمُ يَرْ بَ يࣹمࣷوڪْ تࣷوݧَمْ؞ ‏3 وَايࣹ ݧࣹينتّ فُڪِّ يَرْ، بُ ݧُ ڪࣷ دٗورْ لُ ڪࣷ اࣴݒّ، ندَخْ دٗورْ ڪࣷ لُ اࣴݒّ لࣷولُ، تࣷرْخَلْ سࣹينْ مبࣷڪّ ݖِ سࣹينْ ڪَنَمْ لَيْ دࣷونْ؞
‏4 بُلࣹينْ سُنجُݧْ گࣺمِّݧُ نَگْ، تࣹ فࣹڪّ ڪࣷيْ بࣷجّ اَمْ ݒࣹݒّ؞
دࣷنّ سَ جَبَرُ مبࣷڪّ وَرْتࣺيفْ لَ
‏5 بُ اَيْ گٗورْ يُ بࣷڪّ وَاجُرْ دࣴڪّࣹي، ڪࣹنّ ݖِ ݧࣷومْ دࣹي تࣹ بࣵيِّوُلْ دࣷومْ جُ گٗورْ، جَبَرُ ڪِ دࣹي دُ گࣺنّ، دِ سࣺيْ اَڪُبْ جَامْبُرْ؞ جࣴڪّࣴرُ دࣷنَلࣹيمْ جُ بࣷڪّ اَڪْ جࣴڪّࣴرَمْ مࣷو ڪࣷ وَرَ جࣴلْ، دࣷنْلࣷو ڪࣷ؞ ‏6 سُ ڪࣷ دࣹفࣹي بُ ݖَ ندَوْ سَ تَاوْلࣷو گٗورْ، تَاوْ بَ سَخَلْ تُرُ مبࣷڪَّمْ مَ دࣹي، بَ تُرَمْ دُ فࣹيْ ݖِ بِيرْ إِسْرَاءِيلْ؞ ‏7 سُ وَا جِ بُگُّلَ جࣴلْ جَبَرُ دࣷنَلࣹيمْ جࣷوجُ، نَ ندَوْ سَ دࣹمْ بَ ݖَ مَگْ ݧَ ݖَ بُنتّ دࣴڪّ بَ، نࣹ لࣹينْ:‏ «سَمَ جࣴڪّࣴرُ دࣷنَلࣹ جِي مࣷو بَݧَ سَخَلْ تُرُ مبࣷڪَّمْ ݖِ إِسْرَاءِيلْ؞ مࣷو مَ بُگُّلَ دࣷنْلࣷو؞» ‏8 بُ لࣷولُ اَمࣹي نَ ڪࣷ مَگِ دࣴڪَّمْ وࣷولُ، وَخْ اَڪْ مࣷومْ؞ سُ وَا جِ دࣴگࣴرࣹي، نࣹ بُگُّ ڪࣷو جࣴلْ، ‏9 نَ ڪࣷ جَبَرُ دࣷنَلࣹيمْ جَ فࣹڪِّ ݖَ ڪَنَمْ مَگْ ݧَ، دَلْدِ سُمِّ بࣹنّ ݖَرَخُ وَا جِ بَ نࣷݒِّ، تِفْلِ وَا جِ ݖِ ڪَنَمْ، نࣹ ڪࣷ:‏ «نِي لࣹيسْ دِ دࣹفْ ڪُ دُلْ سَمْݒَلْ مبࣷڪَّمْ ڪࣴرْ؞» ‏10 سُ بࣷوبَا تُرْ وَ ݧُيْ دِݒّࣹي وَا ڪࣴرْ وَا جَ ݖِ دِگّ إِسْرَاءِيلْ، دِ «ڪࣴرُگْ سُمِّ ݖَرَخْ؞»
اَمْ نَ جࣴفْ جُ آيْ ݖِ سࣷتْ
‏11 سُ اَيْ گٗورْ دࣹي جࣵݒَّنْتࣹ، بَ جَبَرُ ڪࣹنّ ڪِ دِڪّ، دِ سࣷتْلࣹ جࣴڪّࣴرَمْ؛ بُ جࣵݒّࣹي ݖَ نگٗورَيْ ڪَيْ جࣵݒَّنْتࣹيڪْ جࣴڪّࣴرْ جَ، ‏12 دࣷگْلࣹينْ لࣷخࣷلْ ندَوْ سَ تࣹ بُلࣹينْ ڪࣷ ݧࣺيبْلُ؞
اَبْ نَتُّڪَايْ دَيْ مَتْ
‏13 بُلࣹينْ دࣹنݖّ ݖِ سࣹينْ مبُوسْ دࣷجُ مࣵنْدَخࣹڪَايْ اَڪْ مࣷرࣷومْ مَ، تࣹ بِي دِيسْ، بِ ݖِ دࣹسْ وࣷيࣷفْ؞ ‏14 بُلࣹينْ دࣹنݖّ سࣹينْ ڪࣴرْ نَتُّڪَايُ سࣷلْ-سࣷتِّ اَڪْ مࣷرࣷومْ مَ، بِي رࣺيْ، بِ ݖِ دࣹسْ تُوتْ؞ ‏15 دࣷجُ مࣵنْدَخࣹڪَايْ بُ مَتْ تࣹ دࣴگُّ نگࣹينْ وَرَ اَمْ، تࣹ نَتُّڪَايُ سࣷلْ-سࣷتِّ بُ مَتْ تࣹ دࣴگُّ نگࣹينْ وَرَ اَمْ، ندَخْ نگࣹينْ گُدّ فَنْ ݖَ رࣺيوْ مَ لࣹينْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ دِ جࣷخْ؞ ‏16 سࣹينْ يࣵلَّ اَجِ سَخْ جِ ڪَتْ سࣹيخْلُ نَ ڪࣺݒّ ڪُيْ دࣹفْ يࣷويِي، ڪࣺݒّ ڪُيْ دࣹفُگْ نجُبْلَݝْ؞
مبُگَلْ اَمَلࣹگْ ندِگَلْ لَ
‏17 فࣵتّلِڪُلࣹينْ لَ لࣹينْ اَمَلࣹگࣹينْ ݧَ دࣹفࣷونْ ݖَ يࣷونْ وَ، بَ نگࣹينْ گࣺنّࣹي مِسْرَ:‏ ‏18 ݧࣷو لࣹينْ دَبْ ݖِ يࣷونْ وِ، سࣷنگࣹ لࣹينْ گَنَّاوْ، فࣵدّ مبࣷولࣹمْ ݧَ مُجّࣹ وࣷونْ، فࣹڪّ نگࣹينْ سࣷنّ، لࣷوفْ، تࣹ رَگَلُݧُ ݖِ سَخْ يࣵلَّ؞ ‏19 بُ نگࣹينْ دࣹمࣹي بَ سࣹينْ يࣵلَّ اَجِ سَخْ جِ نࣷݒَّلْ لࣹينْ ݖِ مبࣷولࣹمْ سࣹينْ نࣷونْ يِ سࣺقْ رࣺيوْ، مِ مُ لࣹينْ سࣺدّ، نگِرْ نگࣹينْ نَنگُ ڪࣷ، نَنگࣹينْ فَرْ مبࣷولࣹمْ لُ ݧُ دࣷونْ فࣵتّلِڪࣷو اَمَلࣹگࣹينْ ݧِ فُ اَسَمَانْ تِيمْ؞ بُلࣹينْ ڪࣷ فࣵتّࣹ؞
