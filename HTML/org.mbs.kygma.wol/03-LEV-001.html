سَسُ سَرْخَلْڪَتْ يِ
دِ ݧࣹتّࣹيلُ خَاجُ تَوْرࣹيتُ مُوسَىٰ
اَيْ دࣷگَلْ ݖِ مبِرُمْ سَرَخْ يِ
(سَارْ 1‏.1—7‏.38)
لُ جࣴمْ ݖِ سَرَخُ رࣹنْدِ-دٗومَلْ
1
‏1 اَجِ سَخْ جِ وࣷو نَ مُوسَىٰ، وَخْ اَڪْ مࣷومْ ݖَ خَيْمَبْ ندَجࣹ مَ نࣹ ڪࣷ:‏ ‏2 «وَخَلْ بࣵنِّ إِسْرَاءِيلْ نࣹ لࣹينْ:‏ بُ ڪࣹنّ ݖِ يࣹينْ دࣹي سࣵڪَّلْ اَجِ سَخْ جِ سَرَخْ، نَ اِنْدِ اَگْ جُرْ، مُيْ گُ گُدّ مبَا گُ گࣵتّ؞
‏3 «بُ سَرَخَمْ دࣹي سَرَخُ جُرْ گُ گُدّ گُ ݧُيْ دࣹفْ رࣹنْدِ-دٗومَلْ، نَ سَرْخَلْ اَوْ يࣴڪّ وُ اَمُلْ سِڪّ؞ نَ ڪࣷ يٗبُّ ݖِ بُنتّ خَيْمَبْ ندَجࣹ مِ ݖِ ڪَنَمْ اَجِ سَخْ جِ، نگِرْ مُ نَنگُلْ ڪࣷ ڪࣷ؞ ‏4 نَ تࣹگْ لࣷخࣷومْ ݖِ ڪَوْ بࣷݒُّ جُرُ سَرَخُ رࣹنْدِ-دٗومَلْ بِ، ندَخْ ݧُ نَنگُلْ ڪࣷ ڪࣷ، مُيْ نجࣷتْلاَيَمْ؞ ‏5 نَ رࣹنْدِ يࣴڪّ وِ ݖِ ڪَنَمْ اَجِ سَخْ جِ، تࣹ دࣷومِ هَـٰرُونَ يُ گٗورْ يِيْ سَرْخَلْڪَتْ يِ ݧࣷويْ اِنْدِ دࣹرࣹتْ جِ، خࣴݒّ ڪࣷ ݖِ مبࣷولࣹمْ وࣹتِ سَرْخَلُڪَايْ بِ ݖِ بُنتّ خَيْمَبْ ندَجࣹ مِ؞ ‏6 جُرُ سَرَخُ رࣹنْدِ-دٗومَلْ بِ، نَ ڪࣷ بࣷرࣷومْ سَرَخْ بِ فࣹيسْ، دَگَّتْ ڪࣷ؞ ‏7 دࣷومِ هَـٰرُونَ، سَرْخَلْڪَتْ بِ، دَݧُيْ جَفَلْ اَبْ تَالْ ݖِ ڪَوْ سَرْخَلُڪَايْ بِ، تࣹگْ مَتّ مِ ݖِ تَالْ بِ؞ ‏8 سُ ڪࣷ دࣹفࣹي دࣷومِ هَـٰرُونَ يُ گٗورْ، يِيْ سَرْخَلْڪَتْ يِ، جࣴلْ دࣷگْ يِ، بࣷولࣹيڪْ بࣷݒّ بِ اَڪْ نࣹبّࣷنْ بِ، تࣹگْ ڪࣷ ݖِ ڪَوْ مَتُّمْ تَالْ بِ ݖِ سَرْخَلُڪَايْ بِ؞ ‏9 نَݧُ رَخَسْ يࣺرࣹيْ بِيرْ يِ اَڪْ يࣹيلْ يِ، سَرْخَلْڪَتْ بِ بࣷولࣹ لࣺݒّ لَڪّ ݖِ ڪَوْ سَرْخَلُڪَايْ بِ؞ سَرَخُ رࣹنْدِ-دٗومَلْ لَ، سَرَخُ سَوَرَ نگِرْ خࣹتُگْ جࣵمّ، ݧࣹيلْ اَجِ سَخْ جِ؞
‏10 «بُ سَرَخَمْ دࣹي اَبْ گࣵتّ بُ ݧُيْ دࣹفْ رࣹنْدِ-دٗومَلْ، مُيْ خَرْ مبَا بࣴيْ، نَ سَرْخَلْ بُ گٗورْ بُ اَمُلْ سِڪّ؞ ‏11 نَ رࣹنْدِ سَرَخْ سِ ݖِ ڪَنَمْ اَجِ سَخْ جِ، فَ فࣺيتࣹ سَرْخَلُڪَايْ بِ بࣴجْ-گࣵنَّارْ؞ نَ دࣷومِ هَـٰرُونَ يُ گٗورْ يِيْ سَرْخَلْڪَتْ يِ خࣴݒّ دࣹرࣹتْ جِ ݖِ مبࣷولࣹمْ وࣹتِ سَرْخَلُڪَايْ بِ؞ ‏12 بُ ڪࣷ بࣷرࣷومْ سَرَخْ بِ دَگَّتࣹي بَ نࣷݒِّ، سَرْخَلْڪَتْ بِ دَيْ بࣷولࣹ دࣷگْ يࣹيڪْ بࣷݒّ بِ اَڪْ نࣹبّࣷنْ بِ، تࣹگْ ڪࣷ ݖِ ڪَوْ مَتُّمْ تَالْ بِ ݖِ ڪَوْ سَرْخَلُڪَايْ بِ؞ ‏13 نَ بࣷرࣷومْ سَرَخْ بِ رَخَسْ يࣺرࣹيْ بِيرْ يِ اَڪْ يࣹيلْ يِ، سَرْخَلْڪَتْ بِ بࣷولࣹ لࣺݒّ لَڪّ ݖِ ڪَوْ سَرْخَلُڪَايْ بِ؞ سَرَخُ رࣹنْدِ-دٗومَلْ لَ، سَرَخُ سَوَرَ نگِرْ خࣹتُگْ جࣵمّ، ݧࣹيلْ اَجِ سَخْ جِ؞
‏14 «بُ نِتْ ڪِ دࣹي سَرْخَلَلْ اَجِ سَخْ جِ نجَنَاوْ لُيْ دࣷونْ رࣹنْدِ-دٗومَلْ، نَ سَرْخࣹ ݒِتَخْ مبَا خَتِ مُ ندَوْ؞ ‏15 نَ ڪࣷ سَرْخَلْڪَتْ بِ يٗبُّ ݖَ سَرْخَلُڪَايْ بَ، ڪُتّ بَاتْ بِ، لَڪّ ڪࣷ ݖِ ڪَوْ سَرْخَلُڪَايْ بِ؞ دࣹرࣹتْ جِ دࣹيسْ ڪࣷيْ سِيتَلْ ݖِ وࣹتُ سَرْخَلُڪَايْ بِ، بَ مُ جࣹيخْ؞ ‏16 نَ اِنْدِ گَگْ بِ اَڪْ لِ ݖِ نࣹڪّ تࣹ وࣹتِّ ڪࣷ فَ ݧُيْ دࣹفْ دٗومُ-تَالْ بَ، فَ فࣺيتࣹ سَرْخَلُڪَايْ بِ ݒࣹنڪُ؞ ‏17 سَرْخَلْڪَتْ بِ دَيْ جࣵݒّ ݖِ لاَفْ يِ، خࣷتِّ تࣹ بَݧْ ڪࣷو تࣹقَلࣹ، دَلْدِ ڪࣷيْ لَڪّ ݖِ سَرْخَلُڪَايْ بِ، ݖِ ڪَوْ مَتُّمْ تَالْ بِ؞ سَرَخُ رࣹنْدِ-دٗومَلْ لَ، سَرَخُ سَوَرَ نگِرْ خࣹتُگْ جࣵمّ، ݧࣹيلْ اَجِ سَخْ جِ؞
