اࣵلُّوَ جُ وࣺرْ لِي لَيْ خَمْلࣹ
2
‏1 يَوْ نَگْ وَارࣹيلْ لِ دࣴݒّࣷوڪْ اࣵلُّوَ جِ وࣺرْ، ‏2 تࣹ مࣷويْ نَ مَگْ ݧِ مࣷومْ سࣹينْ بَڪَّنْ، دِ ݧُ تࣹدّ، جِيتَلْ سࣹينُمْ خࣹلْ، تࣹ ݒَسْتࣺيفُ ݖِ وࣵلُّ نگࣴمْ، اَڪْ ݖࣷفࣹيلْ، اَڪُگْ مُݧْ؞
‏3 نَڪَ نࣷونُ، جِگࣺينْ ݧُ مَگْ ݧِ اِتَمْ، نَݧُ جࣴفࣹ نِ مُ جࣹڪّࣹ ݖِ اَيْ نِتْ ݧُ سࣹلّ؛ بُݧُ يࣵقْ دࣹرْ، بُ لࣹينْ سࣵنگَرَ جِيتَلْ، تࣹ نَݧُيْ جࣵنگّلࣹ لُ بَاخْ، ‏4 نگِرْ تࣵگَّتْ جࣹيگْ جُ ندَوْ جِ ݖِ سࣷݒّ سࣹينْ جࣴڪّࣴرْ، سࣷݒّ سࣹينِ دࣷومْ، ‏5 دِ ݧُ جِيتَلْ سࣹينُمْ خࣹلْ، سࣹتْ جِڪّࣷ، دِ لِگّࣺييْ ݖَ سࣹينِ ڪࣴرْ، بَاخْ، نَنگُلْ سࣹينْ جࣴڪّࣴرْ، بَ ڪࣹنّ دُ بَاتَلْ ڪࣵدُّگْ يࣵلَّ؞
‏6 نَڪَ نࣷونُ، خَلࣹ يُ گٗورْ يِ اِتَمْ، ݧَاخْلࣹينْ ݖِ جِيتَلْ سࣹينُمْ خࣹلْ؞ ‏7 تࣹ نَنگَ دࣷونْ، يَوْ ݖِ سَ بࣷݒّ اَبْ رࣷيُڪَايْ، ݖِ لࣺݒّ ندَخْ سَ جࣴفْ جُ بَاخْ، تࣹ نگَ بࣷولࣹ ڪࣷ اَڪْ اࣵلُّوَ جُ دࣴگُّ تࣹ تࣹدّ، ‏8 اَڪْ ڪࣵدُّ گُ وࣺرْ گُ ڪࣹنّ دُلْ مَنَ وࣹدِّ، نگِرْ رُسْلࣷو اَبْ نࣷونْ بُ اَمُلْ لࣹنّ لُ بࣷنْ لُ مُ نُ بَاتَلࣹ؞
‏9 جَامْ يِ، نَݧُ نَنگُلْ سࣹينِ سَنگّ ݖِ لࣺݒّ، دِ سࣵڪُّ سࣹينْ بࣵنّࣹيخُ سَنگّ، تࣹ ݧُ بَݧْ دِ تࣷنْتࣷونْتࣹ اَڪْ مࣷومْ، ‏10 مبَا دِ رَنْدَلْ؞ وَايࣹ نَݧُ فِرْندࣹيلْ نࣹ ݧࣷو يࣹلّࣷو ڪٗولُتࣹ گُ مَتْ سࣴڪّ، سُ ڪࣷ دࣹفࣹي دِنَݧُ سَگَلْ ݖِ لࣺݒّ، اࣵلُّوَ جِ ݧُ جࣵنگࣹ، تࣹ مُيْ جُ يࣵلَّ، سُنُ مُسَلْڪَتْ؞
‏11 ندَخْ يِوُ يࣵلَّ ڪَتْ فࣹيݧْ نَ، تࣹ لࣷولࣷو اِنْدِلْ ݧࣺݒّ اَگْ مُݖّ؞ ‏12 يِوْ وࣷووُ مࣷو نُ دِگَلْ نُ دࣴدُّ ݧࣹمࣹ يࣵلَّ، اَڪْ خࣹمّࣹمْتࣺيفِ اࣵدِّنَ، تࣹ جِيتَلْ سُنُمْ خࣹلْ، دِ جࣴفࣹ نجُبْ، تࣹ نࣹڪّࣹ اَگْ نجُلِّتࣹ ݖِ اࣵدِّنَيْ تࣹيْ جِي؞ ‏13 لࣷولُ لࣹيسْ نُ دِگَلْ، لِ فࣹيڪْ نُ نگِ سࣺينْتُ يَاڪَارُ مبࣺگْ مَ، ڪࣹرࣷوگْ بࣺسُ ݒࣹيݧُمْ سُنُ مࣵگَّايُ يࣵلَّ، مِيْ عِيسَىٰ اَلْمَسِ، مُسَلْڪَتْ بِ؞ ‏14 مࣷو مَيࣹ بَڪَّنَمْ نگِرْ نُنْ، مࣷو نُ جࣷتْ، بَ نُ مُݖّ ݖِ گࣺݒّ نجُبَدِ، مࣷو نُ دࣹفْ اَسْڪَنْ وُ مُ سࣹتَلَلْ بࣷݒَّمْ، سࣺدّࣷو نُ، نُيْ اَسْڪَنْ وُ فَرْلُ ݖِ جࣴفْ جُ بَاخْ؞
‏15 گࣷوگُ ڪࣵدُّ نگَيْ وَارࣹي، دِ دࣺنڪࣷو اَڪْ دِ فࣹمّࣷو، تࣹ بࣷولࣹ ݖِ فُلَّ جُ مَتْ سࣴڪّ؞ بُ لَ ڪࣹنّ يَبْ؞
