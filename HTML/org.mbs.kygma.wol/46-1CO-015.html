لُ جࣴمْ ݖِ ندࣹڪِّتࣹلْ اَلْمَسِ
15
‏1 لࣺيگِ نَگْ بࣷڪّ يِ، مَ فࣵتّلِ لࣹينْ خِبَارُ جࣵمّ بِ مَ لࣹينْ خِبَارࣷونْ، مُيْ لُ نگࣹينْ گࣴمْ، سَخْ ݖِ، ‏2 تࣹ ݖِ نگࣹينْ تࣹخࣹي، ندࣹگَمْ جࣵݒّ نگࣹينْ بُ دࣴگࣴرْ ݖِ ڪࣵدُّگْ خِبَارُ جࣵمّ بِ، نِ مَ لࣹينْ ڪࣷ خَمَلࣹ؞ لُ ڪࣷ مࣷيْ ݖِ نࣹينْ نگࣹينْ گࣴمْ؞
‏3 لِ مَ جࣷتࣷونْ لاَ لࣹينْ جࣴڪَّ جࣷتّلِ، تࣹ مࣷو دِ:‏ اَلْمَسِ مࣷو دࣹي نگِرْ نجࣷتُگْ سُنُيْ بࣵڪَّارْ، تࣹ نࣷونُ لَ ڪࣷ مبِندّ مِ اِنْدࣹي؞ ‏4 دࣹنݖّ نَݧُ ڪࣷ، مُ دࣹڪِّ ݖَ ݧࣹتّࣹيلُ فَنَمْ، تࣹ نࣷونُ لَ ڪࣷ مبِندّ مِ اِنْدࣹي؞ ‏5 مُ فࣹيݧُ سࣹفَسْ، تࣹگْ ݖَ فࣹيݧُ فُڪِّ تَالِبࣹ يَ اَڪْ ݧَارْ؞ ‏6 گَنَّاوْ گِ مُ فࣹيݧُ لُ اࣴݒّ جُرٗومِ تࣺيمࣺيرِ بࣷڪِّ تَالِبࣹ يُ ڪࣷ گِسَنْدࣷو، تࣹ لِ اࣴݒّ ݖِ ݧࣷوݧُ، ݧُ نگِ دُندّ بَ لࣺيگِ، دࣷونْتࣹ اَمْ نَ ݧُ ݖِ نࣹلَوْ؞ ‏7 گَنَّاوْ گِ مُ فࣹيݧُ يَعْقُوبَ، تࣹگْ ݖَ فࣹيݧُ ندَوِ اَلْمَسِ يࣺݒّ؞ ‏8 مَنْ نَگْ لَ مُجَّ فࣹيݧُ، تࣹ مَ دِبْ لِيرْ بُ اَبْ جُدّࣷومْ مࣷيْ بࣺسَمْ، ‏9 ندَخْ ڪَتْ مَا گࣴنَ ݧࣵڪّ سࣷلࣷ ݖِ ندَوِ اَلْمَسِ، تࣹ يࣹيࣷووُمَ سَخْ تُرُ ندَوْ، گَنَّاوْ مَا دࣷونْ بُندّخَتَلْ مبࣷولࣷومْ يࣵلَّ مِ؞ ‏10 ݖِ يِوُ يࣵلَّ دَالْ لاَ دࣷونࣹ ڪِ مَ دࣷونْ، تࣹ يِوْ وࣷووُ مَ ݧࣹيلْ نࣹينُلْ؞ خَنَا يࣹنࣹينْ ندَوْ يࣺݒّ سَخْ، مَا لࣹينْ گࣴنَ لِگّࣺييْ؞ دُ ݖِ مَنْ نَگْ، وَايࣹ يِوُ يࣵلَّ، وِ اࣵندّ اَڪْ مَنْ لَ؞ ‏11 ڪࣷنْ مُيْ مَنْ، دِ ݧࣷومْ، نࣷونُ لَنُ وَارࣹي، تࣹ نࣷونُ اِتْ نگࣹينْ گࣴمࣹ؞
دࣹڪَّلْ تࣴوُلْ يࣵلَّ
‏12 ندࣹگَمْ بِرَلْ نَݧُ نࣹ اَلْمَسِ دࣹڪِّ نَ، اَنَ نُ ݧࣹنّ ݖِ يࣹينْ مَنࣹ نࣹ ݧِ دࣹي دُݧُ دࣹڪِّ؟ ‏13 سُ فࣹڪّࣹي نࣹ ݧِ دࣹي دُݧُ دࣹڪِّ، ڪࣷنْ اَلْمَسِ اِتْ دࣹڪِّوُلْ؞ ‏14 سُ اَلْمَسِ دࣹڪِّوُلْ، سُنُ وَارࣹ نࣹينْ نَ، تࣹ سࣹينْ نگࣴمْ اِتْ نࣹينْ نَ؞ ‏15 تࣹ سُ بࣷوبَا نَگْ ݧُ وَرْ نࣷو جࣵݒّࣹ اَيْ سࣹيدࣹيْ فࣹنْ-ڪَتِ يࣵلَّ، ندَخْ نࣷو سࣹيدࣹيلْ يࣵلَّ، نࣹ مࣷو دࣹڪَّلْ اَلْمَسِ تࣹ دࣹڪَّلُ ڪࣷ، گَنَّاوْ ݧِ دࣹي دُݧُ دࣹڪِّ؞ ‏16 ندَخْتࣹ ݧِ دࣹي، سُ ݧُ دُلْ دࣹڪِّ ڪَيْ، اَلْمَسِ اِتْ دࣹڪِّوُلْ؞ ‏17 سُ اَلْمَسِ دࣹڪِّوُلْ نَگْ، سࣹينْ نگࣴمْ اَمُلْ نجَرِݧْ، تࣹ يࣹينَ نگِ ݖِ سࣹينِ بࣵڪَّارْ بَ تࣹيْ؞ ‏18 تࣹ ڪࣷنْ اِتْ ݧِ دࣹي گَنَّاوْ بَ ݧُ گࣴمࣹي اَلْمَسِ، سࣵنڪُ نَݧُ؞ ‏19 سُ دࣹي ݖِ گِيرُ دُندّ گِي دࣷݝّ لَنُ اَمࣹ يَاڪَارْ ݖِ اَلْمَسِ، ڪࣷنْ دࣹ نࣷو گࣴنَ تࣷسْڪَرࣹ ݧࣺݒّ؞
‏20 وَايࣹ نَگْ اَلْمَسِ دَفَ دࣹڪِّ مࣷوسْ، تࣹ مࣷويْ سَرَخُ ندࣷورْتࣹلْ مࣹݧّࣹيفْ، بِ ݧُ سࣵڪّࣹ ݖِ نگٗوبُمْ بَڪَّنِ نِتْ ݧِ دࣹي؞ ‏21 ݖِ نِتْ لَ ندࣹي دِڪّࣹ ݖِ اࣵدِّنَ؞ مࣷو تَخْ ݖِ نِتْ اِتَمْ لَ ندࣹڪِّتࣹ دِڪّࣹ ݖِ اࣵدِّنَ؞ ‏22 نࣷونُ نِتْ بࣷڪّࣹ ݖِ آدَمَ، بَ تَخْ ݧࣺݒّ دِ دࣹي، نࣷونُ لَ ݧࣺݒّ مَنَ بࣷڪّࣹ ݖِ اَلْمَسِ، بَ تَخْ ݧُ دُنْدَلاَتْ لࣹينْ، ݧࣷومْ ݧࣺݒّ؞ ‏23 وَايࣹ ڪُ نࣹڪّ اَڪْ اَيَمْ؞ جࣷوخࣹبْ ندࣷورْتࣹلْ مࣹݧّࣹيفْ مِ، اَلْمَسِ لَ؞ گَنَّاوْ لࣷولُ ݧࣷݧِ اَلْمَسِ ݧࣷويْ اَيࣹ، ڪࣹرࣷوگْ بُ دࣺلْسࣹي؞ ‏24 گَنَّاوْ لࣷولُ لَ مُجْ گَ، تࣹ ݖَ لَيْ دࣹلّࣷو نگُورْ گِ ڪِيْ يࣵلَّ، دِ بَايْ، فࣹڪّ مُ نࣹينَلْ مبࣷولࣹمْ لُيْ ڪِلِفْتࣹيفْ، اَڪْ مبࣷولࣹمْ سَݧْ-سَݧْ اَڪْ دࣷولࣹ؞ ‏25 ندَخْتࣹ فࣵوّ اَلْمَسِ نگُورُ، بَ ڪࣹرَ يࣵلَّ دِ سُڪَّلْ مبࣷولࣹمْ اَيْ نࣷونَمْ ݖِيْ تࣵنڪَمْ؞ ‏26 نࣷونْ بُ مُجّ، بِ ݧُيْ نࣹينَلْ مࣷويْ ندࣹي، ‏27 ندَخْ «لࣺݒّ لَ يࣵلَّ دِ جࣺبَّلْ اَلْمَسِ، تࣹگْ ڪࣷ فِ رࣷنْ ندࣴگُّيْ تࣵنڪَمْ¤؞» وَايࣹ بُ يࣵلَّ نࣹي، «لࣺݒّ لَ ڪࣷيْ جࣺبَّلْ،» بِرْ نَ نࣹ مࣷومْ مِ ڪࣷ جࣺبَّلْ لࣺݒّ، بࣷڪُّ ݖِ؞ ‏28 بُ ݧُ ڪࣷ جࣺبَّلࣹي لࣺݒّ نَگْ، سُ بࣷوبَا مࣷومْ دࣷومْ جِ ݖِ بࣷݒَّمْ مࣷويْ جࣺبَّلْ بࣷݒَّمْ، ڪِ ڪࣷ جࣺبَّلْ لࣺݒّ، نگِرْ يࣵلَّ دِ لࣺݒّ ݖِ لࣺݒّ؞
‏29 سُ فࣹڪّࣹي نࣹ ݧِ دࣹي دُݧُ دࣹڪِّ، ڪࣷنْ ݧِ ݧُ سٗوبْ ݖِ ندࣷخْ نگِرْ ݧِ دࣹي، لُ ݧُ ڪࣷيْ دࣹفْ، دࣷيࣹ؟ گَنَّاوْ ݧِ دࣹي دُݧُ دࣹڪِّ، اَنَ لࣹيسْ لࣹينْ دِ سٗوبࣹ نگِرْ ݧِ دࣹي؟ ‏30 تࣹ ڪࣷنْ نُنْ اِتْ، اَنَ لُ جَرْ نُيْ جَايْ سُنُ بَڪَّنْ وَخْتُ وُ نࣹڪّ؟ ‏31 بࣷڪّ يِ، مَنْ دࣹ بࣺسْ بُ نࣹڪّ مَ جࣵڪَّارْلࣷوڪْ ندࣹي، اَڪْ سَگْ بِ مَ لࣹينْ دِ سَگࣷو ندَخْ عِيسَىٰ اَلْمَسِ سُنُ بࣷرࣷومْ، مِ نگࣹينْ گࣴمْ؞ ‏32 سُ يَاڪَارُ نِتْ ݖِ اࣵدِّنَ دࣷݝّ تَخࣹي مَيْ خࣹيخْ اَڪْ اَيْ رَبِ اࣵلّ ݖَ اࣹفࣹسْ، اَنَ لُ مُ مَيْ جَرِݧْ؞ گَنَّاوْ ݧِ دࣹي دُݧُ دࣹڪِّ، ڪࣷنْ ڪَيْ مࣷويْ لِ ݧُ نࣹ:‏
«نَنُ لࣹڪّ تࣹ نَانْ، ندࣹ اࣴلّࣴگْ نُ دࣹي¤؞»
‏33 بُ لࣹينْ ڪࣹنّ نَخْ:‏ «اࣵندّ بُ بࣷنْ يࣵقْ نَ جِڪّࣷ جُ بَاخْ؞» ‏34 دࣺلّࣷوسِلࣹينْ خࣹلْ يِ ݖِ نجُبْ تࣹ بࣵيِّ بࣵڪَّارْ، ندَخْ ݧࣵڪَّ خَمْ لࣹنّ ݖِ يࣵلَّ، مࣷومْ لَ ݧࣹنّ ݖِ يࣹينْ نࣹڪّࣹ، تࣹ لࣷولُ گࣵݖّࣹ لَ ݖِ يࣹينْ، لاَ وَخْ!‏
يَرَمْ بَرࣹ نَ تࣹ وُوتࣹ
‏35 تࣹيوُلْ نِتَ نگِ نَانْ:‏ «نَنْ لَ ݧِ دࣹي دِ دࣹڪّࣹي؟ وَنْ يَرَمْ لَݧُيْ دࣹڪَّالࣹ؟» ‏36 گࣵتّ خࣹلْ!‏ لِ نگَ جِ، يَوْ ݖِ سَ بࣷݒّ، دُ دُنْدَاتْ لِ فࣹيڪْ دࣹيوُلْ بَ نࣷݒِّ؞ ‏37 تࣹ لِ نگَ جِ دُ يَرَمُ گࣵنݖَخْ گِيْ ݧࣴوْ، اَوْ فࣹݒّ ڪࣹسࣹ لَ، مَنْ نَا دࣷونْ بࣹلࣹ، مبَا وࣹنࣹينْ جِوُ؞ ‏38 وَايࣹ يࣵلاَّ ڪࣷيْ يَرَمَلْ نِ مُ ڪࣷ سࣷوبࣹ، تࣹ جِوُ وُ نࣹڪّ اَڪْ يَرَمُ گࣵنݖَخَمْ؞ ‏39 سُوخْ يࣺݒّ نَگْ دُ وࣹنّ سُوخْ وِ؞ وِي سُوخْ وُ نِتْ لَ، وࣹي وُ مَلَ، وَلࣹ وُ ݒِݖّ، وَ ݖَ دࣹسْ وُ جࣴنْ؞ ‏40 اَمْ نَ اِتَمْ اَيْ يَرَمْ يُ فَرࣹوࣷو اَسَمَانْ، اَڪْ اَيْ يَرَمْ يُ فَرࣹوࣷو سُوفْ؞ يَرَمْ وُ فَرࣹوࣷو اَسَمَانْ نَگْ، اَگْ لࣹيرَمْ وُوتࣹ نَ اَگْ لࣹيرُ يَرَمْ وُ فَرࣹوࣷو سُوفْ؞ ‏41 جَنتّ بِ اَڪْ لࣹيرُ بࣷݒَّمْ، وࣹيرْ وِ اَڪْ لࣹيرُ بࣷݒَّمْ، بِدِّيوْ يِ اَڪْ سࣹينْ لࣹيرُ بࣷݒّ، تࣹ بِدِّيوْ سَخْ وُوتࣹ نَاڪْ مࣷرࣷومْ مَ، اَگْ لࣹيرْ؞
‏42 نࣷونُ اِتَمْ لَ ندࣹڪِّتࣹلْ ݧِ دࣹي دِ دࣹمࣹ؞ لِ ݧُ سُولْ مࣷويْ جِوُ وِيْ يࣵقُ، تࣹ بُ دࣹڪّࣹي دࣷوتُ يࣵقُ؞ ‏43 ݖِ اَگْ تࣹدَّدِ لَݧُ جِ نࣺيوْ بِ ݧُ سُولْ، تࣹ تࣹرَانگَ لَيْ دࣹڪَّالࣹ؞ ݖِ نࣺيوْ دࣷولࣹ لࣹيسْ ڪࣷ جِ، مُ دࣹڪَّالࣹ دࣷولࣹ؞ ‏44 لِ ݧُ سُولْ مࣷويْ يَرَمُ جُدُّوَالࣹ وِ ݧُ جِ، مُ دࣹڪِّ دِ يَرَمْ وُيْ دُنْدࣹ نࣷو گِ؞ ندَخْتࣹ نِ يَرَمُ جُدُّوَالࣹ اَمࣹ، نِ لَ يَرَمْ وُيْ دُنْدࣹ نࣷو گِ اَمࣹ؞
‏45 نࣷونُ لَݧُ بِنْدࣹ نࣹ:‏ «آدَمَ، نِتْ ڪُ جࣴڪّ ڪَ، لَݧُ دࣹفْ جࣴمّ جُيْ دُندّ¤،» وَايࣹ آدَمَ مُ مُجّ مِ مࣷو دِ نࣷو گِيْ دُندّلࣷو؞ ‏46 دُ يَرَمْ وِ بِنْدࣷو اَگْ نࣷو نَگْ مࣷويْ جِيتُ، وَايࣹ يَرَمُ جُدُّوَالࣹ وِ مࣷويْ جِيتُ، يَرَمْ وِ بِنْدࣷو اَگْ نࣷو تࣷݒّ ݖِ؞ ‏47 آدَمَ مُ جࣴڪّ مَ، ݖِ سُوفْ لَ جٗگࣹ وࣷونْ، دِ سُوفْ؛ آدَمَمْ ݧَارࣹيلْ مِ جٗگࣹ فَ اَسَمَانْ؞ ‏48 ڪࣷوڪَ دِ سُوفْ نَگْ، نَ مُ مࣹلْ، نِ لَ نِتِ سُوفْ مࣹلْ؞ ڪُ اَسَمَانࣹ ڪَ اِتْ، نَ مُ مࣹلْ، نِ لَ ݧُ اَسَمَانࣹ ݧِ مࣹلْ؞ ‏49 تࣹ نِ نُ دࣷونࣹ تࣹيْ، تَڪّندࣹيرُ نِتُ سُوفْ ڪَ، نࣷونُ لَنُيْ دࣷونࣹجِ تَڪّندࣹيرُ ڪُ اَسَمَانࣹ ڪَ؞
‏50 لِ مَيْ وَخْ بࣷڪّ يِ، مࣷو دِ سُوخْ اَڪْ دࣹرࣹتْ دُ اَمْ ݖࣺرْ ݖِ نگُورُگْ يࣵلَّ، تࣹ لُيْ يࣵقُ مَنُلَ اَمْ ݖࣺرْ ݖِ لُ دُلْ يࣵقُ؞ ‏51 ڪَايْلࣹينْ مَ خَمَلْ لࣹينْ مبِرْ مُ اُمْݒࣹ؞ دُنُ دࣹي نُنْ ݧࣺݒّ، وَايࣹ نُنْ ݧࣺݒّ اَيْ سࣷݒِّڪُ؞ ‏52 لُ گَاوَ گَاوْ لَيْ دࣷونْ، ݖِ خࣹفْ اَڪْ خِݒِّ، بُ لِيتْ گُ مُجّ گَ جِبࣹي؛ ندࣹ لِيتْ گَ مࣷويْ جِبْ، ݧِ دࣹي، دࣹڪِّ، دِ ݧُ دُلْ يࣵقࣷوتِ، نُنْ نَگْ، نُ سࣷݒِّڪُ؞ ‏53 سُنُ يَرَمُ جُدُّوَالࣹ وِي نَرَ يࣵقُ، فࣵوّ مُ سࣷلْ وࣹنࣹينْ وُ دُلْ يࣵقُ؛ سُنُ يَرَمُ ندࣹي وِي، فࣵوّ مُ سࣷلْ يَرَمُ دࣹيتِلْ؞ ‏54 بُ يَرَمُ جُدُّوَالࣹ وِي نَرَ يࣵقُ سࣷلࣹي وࣹنࣹينْ وُ دُلْ يࣵقُ، بَ يَرَمُ ندࣹي وِي سࣷلْ يَرَمُ ندࣹيتِلْ، سُ بࣷوبَا لَ ڪࣵدُّ گِي ݧُ بِندّ دِ مَتْ:‏
«مࣴدّࣹيسْ نَ ندࣹي ݖِ گࣺيجُ ندَمْ¤؞
‏55 مࣷو يَوْ ندࣹي، اَنَ سَ ندَمْ؟
مࣷو يَوْ ندࣹي، اَنَ سَگْ فِتّ¤؟»
‏56 ندࣹي، اَگْ فِتَّمْ مࣷويْ بࣵڪَّارْ، تࣹ بࣵڪَّارْ، دࣷولࣹيمْ مࣷويْ يࣷونْ وِ؞ ‏57 وَايࣹ ݖَنتّ ݧࣹيلْ نَ يࣵلَّ، مِ نُ مَيْ ندَمْ ݖِ سُنُ سَنگّ عِيسَىٰ اَلْمَسِ!‏
‏58 ڪࣷنْ نَگْ يࣹينْ سَمَ بࣷڪّ يِ مَ سࣷݒَّلْ سَمَ بࣷݒّ، فَرْلُلࣹينْ، بُ لࣹينْ دَرَ يࣴنگَلْ، تࣹ سَخࣷولࣹينَ جࣴمْ ڪَنَمْ ݖِ لِگّࣺييُ سَنگّ بِ، تࣹ نگࣹينْ خَمْ نࣹ سࣹينُ ݧَقْ، ݖِ سࣹينُبْ اࣵندّ اَڪْ سَنگّ بِ، دُ نࣹينْ؞
