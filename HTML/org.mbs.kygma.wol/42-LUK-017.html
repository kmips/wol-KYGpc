بࣵڪَّارْلࣷو نِتْ آيْ نَ
17
‏1 عِيسَىٰ نࣹيتِ اَيْ تَالِبࣹيمْ:‏ «اَگْ فِيرْ گُيْ بࣵڪَّارْلࣷو مَنُلَ ݧࣵڪّ، وَايࣹ وٗويْ نگَلَّ ڪَ ڪࣷيْ لَلْ؞ ‏2 ݧُ تَڪّࣷونْ دࣷجْ وُ رࣺيْ ݖِ بَاتَمْ، سࣵنِّ ڪࣷ بِيرْ گࣺيجْ مࣷو گࣴنْ ݖِ مࣷومْ مُيْ يٗبّࣹ بࣵڪَّارْ ڪࣹنّ ݖِ خَلࣹ يِي؞ ‏3 وَتُّلࣹينْ سࣹينْ بࣷݒّ؞
جࣺگَّلْ-لࣹينْ ڪُ لࣹينْ تُوبَلْ
«بُ لَ سَ مبࣷڪّ تࣷوݧࣹي، يࣹدّ ڪࣷ؞ بُ تُوبࣹي، بَالْ ڪࣷ؞ ‏4 سُ لَ تࣷوݧࣷونْ جُرٗومْ ݧَارِ يࣷونْ ݖِ بࣺسْ بِ، تࣹ وَلْبَتِڪُ جُرٗومْ ݧَارِ يࣷونْ، نࣹ لَ:‏ ‹تُوبْ نَا ڪࣷ،› بَالْ ڪࣷ رࣹڪّ؞»
نگࣴمْ مَيࣹ نَ ڪࣺيمْتَانْ
‏5 ݖِ ڪَوْ لࣷولُ ندَوْ يَ نࣹ سَنگّ بِ:‏ «يࣷڪّ نُ نگࣴمْ؞» ‏6 سَنگّ بِ نࣹ لࣹينْ:‏ «سُ نگࣹينْ اَمࣷونْ ݖِ نگࣴمْ، لُ تࣷلُّ نِ ݒࣹݒُّ فُدّࣴنْ سَخْ، بَ نࣹ گَرَبْ گِي:‏ ‹بُدّࣹيڪُلْ، دࣹمْ جࣴمْبَتࣹ ݖَ گࣺيجْ گَ،› سُ بࣷوبَا گَرَبْ گِ دࣺگَّلْ لࣹينْ؞
اَبْ جَامْ دُ سࣵڪُّ جَاجࣴفْ
‏7 «ڪَنْ ݖِ يࣹينْ لَ اَبْ جَامَمْ دِ بࣹيِ مبَا مُ سࣵمِّ بَ جٗگࣹ تࣷولْ، دِڪّ، مُ نࣹ ڪࣷ:‏ ‹ڪَايْ، نُ لࣹڪّ›؟ ‏8 خَنَا ڪَيْ دَ ڪࣷيْ وَخْ نࣹ ڪࣷ:‏ ‹دࣹفَرَلْ سَمَ رࣹيرْ، تࣹ تَڪُّ، تَاجَلْ مَ، بَ مَ لࣹڪّ، نَانْ، نگَ دࣷورَ لࣹڪّ، نَانْ؞› ‏9 لَ جَامْ بَ جࣴفࣹ ندِگَلْ تَخْ نَ سَنگَمْ سَنتّ ڪࣷ؟ دࣺيدࣺيتْ ڪَيْ؞ ‏10 يࣹينْ اِتَمْ نࣷونُ؞ بُ نگࣹينْ دࣹفࣹي لِ ݧُ لࣹينْ سَنتّ لࣺݒّ بَ نࣷݒِّ، دَنگࣹينَ وَرْ نࣹ:‏ ‹اَيْ جَامْ رࣹڪّ لَنُ؞ سُنُ وَرُگَرْ لَنُ دࣹفْ؞›»
عِيسَىٰ فَجْ نَ فُڪِّ گَانَ
‏11 ݖِ بِيرْ يࣷونْ وࣷووُ عِيسَىٰ جࣴمࣷونْ يࣹرُسَلࣹمْ، مَ نگَ جَارࣹ فَ سَمَرِ اَڪْ گَلِلࣹ دِگَلࣷو؞ ‏12 مُ دࣹمْ بَيْ دُگّسِ ݖِبْ دࣴڪّ، فُڪِّ گَانَ دِڪّ، دَجࣹيڪْ مࣷومْ، بَ دࣵندّ ڪࣷ، دَلْدِ تَخَوْ؞ ݧُ خَاݖُ نࣹ:‏ ‏13 «عِيسَىٰ، نجَاتِگࣹ، يࣴرࣴمْ نُ!‏» ‏14 عِيسَىٰ نَگْ گِسْ لࣹينْ، نࣹ لࣹينْ:‏ «دࣹمْلࣹينْ سࣹيتْلُجِ سࣹينْ يَرَمْ ݖَ سَرْخَلْڪَتْ يَ؞»
بَ ݧُيْ دࣹمْ ݖَ سَرْخَلْڪَتْ يَ لَݧُ وࣺرْ، سࣹينْ يَرَمْ سࣹتْ؞ ‏15 ڪࣹنّ ݖِ ݧࣷومْ گِسْ نࣹ وࣺرْ نَ، دَلْدِ وَلْبَتِڪُ، دِ خَاݖُيْ سࣵبَّالْ يࣵلَّ؞ ‏16 مُ دِڪّ، نࣹ گُرُبْ، دࣴݒّ جࣴعࣴمْ فَ تࣵنڪِ عِيسَىٰ، سَنتّ ڪࣷ؞ وَا جَ نَگْ نِتُ سَمَرِ لَ وࣷونْ؞ ‏17 بَ لࣷولُ اَمࣹي عِيسَىٰ نࣹ:‏ «خَنَا دُ فُڪَّ وࣺرْ؟ جُرٗومْ ݧࣹينتّ ݧَ نَگْ، اَنَ ݧُ؟ ‏18 اَمُ ݖِ ڪࣹنّ ڪُ ݧࣴوْ، دࣹلّࣷو يࣵلَّ نجُڪَّلْ ڪُ دُلْ دࣷخَنْدࣺيمْ بِي!‏» ‏19 عِيسَىٰ نَگْ نࣹ ڪࣷ:‏ «جٗگَلْ دࣹمْ؛ سَ نگࣴمْ فَجْ نَ لَ؞»
نگُورُگْ يࣵلَّ دࣷورْ نَ بَ نࣷݒِّ
‏20 فَرِسࣹنْ يِ ݧࣷو لاَجࣷونْ عِيسَىٰ، نࣹ ڪࣷ:‏ «ڪَݧْ لَ نگُورُگْ يࣵلَّ دِ دِڪّ؟» مُ نࣹ لࣹينْ:‏ «دِڪُّگْ نگُورُگْ يࣵلَّ دُ اࣵندّ اَڪُمْ ݖࣹيتَانْ؞ ‏21 دࣹيسُلْ نࣹ اِتْ:‏ ‹خࣷولْ-لࣹينْ، مُ نگِ› مبَا:‏ ‹مَ نگࣹي،› ندَخْ ڪَتْ نگُورُگْ يࣵلَّ مُ نگِ ݖِ سࣹينْ بِيرْ؞»
‏22 مُ نࣹ تَالِبࣹ يَ:‏ «اَيْ جَنْتَ نگِ ݧࣴوْ يُ نگࣹينْ نَا نگࣺيجࣷو فࣹڪّࣹ بࣹنّ بࣺسْ ݖِ بࣺسِ دࣷومُ نِتْ ڪِ، تࣹ دُنگࣹينْ ڪࣷ فࣹڪّࣹ؞ ‏23 تࣹ اِتْ، دࣹيسْ نَ لࣹينْ نࣹ:‏ ‹مَ نگࣹي!‏› مبَا:‏ ‹مُ نگِي!‏› بُلࣹينْ دࣹمْ، بُلࣹينْ ݖَ تࣷݒّ؞ ‏24 ندَخْ نِ مࣹلَخْ دِ خُيّࣹ گِي وࣹتُ اَسَمَانْ، بَ نࣹ رࣵݧّ ݖَ گࣹي، نࣷونُ لَ دࣷومُ نِتْ ڪِيْ دࣹفْ ڪࣹرࣷوگْ بࣺسَمْ؞ ‏25 وَايࣹ بَلاَ بࣷوبَ فࣵوّ دࣷومُ نِتْ ڪِ اَمْ ݖࣷونࣷ يُ بَرࣹ، جࣵنڪࣷونْتࣹيڪْ دِيݝَتُ نِتِ تࣹيْ جِي؞
‏26 «نࣷونࣹي مُ دࣹمࣹ وࣷونْ ݖَ جَنْتِ نُوحٍ، نِ لَيْ دࣹمࣹ ݖَ جَنْتِ دࣷومُ نِتْ ڪِ؞ ‏27 نِتْ ݧَا نگَ دࣷونْ لࣹڪّ اَڪَ نَانْ، دِ جࣴلْ جَبَرْ اَڪَ سࣺيِ، بَ ڪࣹرࣷوگْ بࣺسْ بَ نُوحٍ دُگّࣹي ݖَ گَالْ گَ، مبࣴنّ مَ دِڪّ، سࣵنڪّ ݧࣺݒّ؞
‏28 «نࣷونࣹي اِتَمْ لَ دࣹمࣹ وࣷونْ ݖَ جَنْتِ لٗوتْ؞ نِتْ ݧَا نگَ دࣷونْ لࣹڪّ اَڪَ نَانْ، دِ جࣴندّ اَڪَ جَايْ، دِ جࣴمْبَتْ اَڪَ تَبَخْ؞ ‏29 وَايࣹ بࣺسْ بَ لٗوتْ گࣺنّࣹي سࣷدࣷمْ، لَ تَوُبْ سَوَرَ اَڪْ تَمَرَخْ وࣵݖّࣹ اَسَمَانْ، سࣵنڪّ ݧࣺݒّ؞ ‏30 نࣷونُ دࣷݝّ لَيْ دࣹمࣹ، بࣺسْ بُ دࣷومُ نِتْ ڪِيْ فࣹيݧْ؞
‏31 «بࣺسُبْ ڪࣹرࣷوگْ بࣷوبࣹي، ڪُ مُ فࣹڪّ ݖِ ڪَوْ اَمْ تَاخْ، تࣹ سَيْ يࣴفْ نࣹ ݖَ بِيرْ ڪࣴرْ گَ، بُلْ وࣵݖّ، دِ ڪࣷ جࣴلِ؞ تࣹ ڪُ مُ فࣹڪّ تࣷولْ بَ اِتَمْ، بُلْ دࣹلُّ؞ ‏32 فࣵتّلِڪُلࣹينْ جَبَرُ لٗوتْ جَ!‏ ‏33 ڪࣺݒّ ڪُيْ وُتَ رَوَلࣹ سَ بَڪَّنْ، يَايْ ݧࣵڪّ سَ بَڪَّنْ، وَايࣹ ڪࣺݒّ ڪُ ݧࣵڪّ سَ بَڪَّنْ، يَايْ دُندّ؞ ‏34 مَا لࣹينْ ڪࣷ وَخْ:‏ گُدِّگْ ڪࣹرࣷوگْ ݧَارْ اَيْ بࣷڪّ بࣹنّ لَلْ، ڪࣹنّ ڪِ، ݧُ يٗبُّ ڪࣷ، ڪَ ݖَ دࣹسْ، ݧُ بَ ڪࣷ؞ ‏35‏-36 ݧَارْ دِ وࣷلَنْدࣷو، ڪࣹنّ ڪِ، ݧُ يٗبُّ ڪࣷ، ڪَ ݖَ دࣹسْ، ݧُ بَ ڪࣷ؞»
‏37 تَالِبࣹ يِ نࣹ عِيسَىٰ:‏ «لࣷولُ فَنْ لَ، سَنگّ بِ؟» مُ نࣹ لࣹينْ:‏ «فُ مࣺدّ نࣹڪّ، فَ لَ تَنْ يَيْ دَجࣹ؞»
