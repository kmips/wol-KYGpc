فُڪِّ سَنْتَانࣹي نگِ
5
‏1 مُوسَىٰ مࣷو وࣷو بࣵنِّ إِسْرَاءِيلْ گࣺݒّ، وَخْ لࣹينْ نࣹ لࣹينْ:‏ يࣹينْ، بࣵنِّ إِسْرَاءِيلْ، دࣺگْلُلࣹينْ دࣷگَلِ يࣷونْ يِ، اَڪْ اࣵتّࣹيْ يࣷونْ يِ مَيْ وَخْ، نگࣹينْ دࣺگّ، تࣹيْ؞ جࣵنگّلࣹينْ ڪࣷ تࣹ فَرْلُ ڪࣷو جࣴفࣹ؞ ‏2 سُنُ يࣵلَّ اَجِ سَخْ جِ مࣷو فَسࣷونْ ڪٗلّࣴرࣹ اَڪْ نُنْ ݖَ تُنْدُ اࣷرࣹبْ؞ ‏3 دُ سُنُيْ مَامْ رࣹڪّ لَ اَجِ سَخْ جِ فَسࣷونْ ڪٗلّࣴرࣹ گِي اَڪْ ݧࣷومْ؛ وَايࣹ ݧࣷوڪْ نُنْ ݧِ فِي تࣹيْ، مبࣷولࣹمْ نُنْ ݧِيْ دُندّ، ݖِ لَنُ؞ ‏4 ݖِ بِيرْ جࣵڪَّارْلࣷو لَ اَجِ سَخْ جِ وَخࣹ وࣷونْ اَڪْ يࣹينْ ݖَ ڪَوْ تُندّ وَ، اࣵدّࣷو ݖَ دِگّ سَوَرَ وَ؞ ‏5 مَا تَخَوْ دِگَّنْتࣹ اَجِ سَخْ جࣹيڪْ يࣹينْ، جَنتّ يࣷويُ، دِ لࣹينْ جࣷتّلِ ڪࣵدُّگْ اَجِ سَخْ جِ، ندَخْ يࣹينَ رَگَلࣷونْ سَوَرَ وَ، بَ يࣺيگُلࣹينْ ݖَ ڪَوْ تُندّ وَ؞ مُ نࣹ:‏
‏6 «مَنْ اَجِ سَخْ جِ مَايْ سࣹينْ يࣵلَّ، جِ لࣹينْ گࣺنّࣹ رࣺيوُمْ مِسْرَ، ڪࣴرُگْ نجَامْ گَ؞
‏7 «بُلࣹينْ اَمْ يࣹنࣹينْ يࣵلَّ ݖِ سَمَ ڪَنَمْ؞
‏8 «بُلࣹينْ سࣵڪَّلْ سࣹينْ بࣷݒّ تُورْ مُيْ جࣴمَّلْ لࣹنّ لُ نࣹڪّ ݖِ ڪَوْ اَسَمَانْ، مبَا ݖِ ڪَوْ سُوفْ، مبَا لُ نࣹڪّ ݖِ ندࣷخْ مُ سُوفْ تِيمْ؞ ‏9 بُلࣹينْ ݖِ سُجّٗوتَلْ لࣹنّ، بُلࣹينْ ݖِ جَامُ لࣹنّ، ندَخْ مَنْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ، يࣵلَّ جُ فِيرْ لاَ؞ مَايْ تࣷݒّ دࣷومْ تࣷوݧُ وَاجُرْ، بَ ݖَ سࣹينْ مَاسِ سࣴتْ، بَ ݖَ مَاسُگْ سࣴتَاتِ نِتْ ݧِ مَ بَݧْ؞ ‏10 وَايࣹ مَايْ بَاخࣹ ݧِ مَ سࣷݒّ، دِ سࣵمّ سَمَيْ سَنْتَانࣹ، بَ ݖِ جُنِّيْ مَاسْ يَ لࣹينْ دِ وُوتُجِ؞
‏11 «بُلࣹينْ تُدّ سࣹينْ تُرُ يࣵلَّ اَجِ سَخْ جِ ݖِ نࣹينْ، ندَخْ اَجِ سَخْ جِ دُ ݧࣵڪَّ تࣷݒّ ڪُيْ تُدّ اَوْ تُرَمْ ݖِ نࣹينْ؞
‏12 «سࣵمّلࣹينْ بࣺسُبْ نࣷفْلاَيْ، نگِرْ سࣹلَّلْ ڪࣷ، نِ لࣹينْ ڪࣷ سࣹينْ يࣵلَّ اَجِ سَخْ جِ سَنْتࣹ؞ ‏13 جُرٗومْ بࣹنِّ فَنْ نگࣹينْ وَرَ لِگّࣺييْ، دࣹفْ ݖِ سࣹينْ سࣷخْلَ يࣺݒّ؞ ‏14 وَايࣹ بࣺسُبْ جُرٗومْ ݧَارࣹيلْ بَ، نࣷفْلاَيْ لَيْ دࣷونْ، ݧࣹيلْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ؞ بُلࣹينْ ݖِ لِگّࣺييْ لࣹنّ؛ دُ يࣹينْ، دُ سࣹينْ دࣷومْ جُ گٗورْ، اَڪْ جُ جِگࣺينْ، دُ سࣹينْ جَامْ بُ گٗورْ، اَڪْ بُ جِگࣺينْ، دُ سࣹينْ نَگْ، دُ سࣹينْ مبَامْ، دُ مبࣷولࣹمْ سࣹينُگْ جُرْ، دُ سࣹينْ گَنْ گُ نࣹڪّ ݖِ سࣹينْ بِيرْ دࣴڪّ؞ سُ ڪࣷ دࣹفࣹي سࣹينْ جَامْ بُ گٗورْ، اَڪْ سࣹينْ جَامْ بُ جِگࣺينْ مَنَ دَلْ-لُ نِ يࣹينْ؞ ‏15 فࣵتّلِڪُلࣹينْ نࣹ اَيْ جَامْ نگࣹينْ وࣷونْ ݖَ رࣺيوُمْ مِسْرَ؞ سࣹينْ يࣵلَّ اَجِ سَخْ جِ مࣷو لࣹينْ فَ گࣺنّࣹي دࣷولࣹيْ لࣷخࣷومْ، اَڪْ ڪࣵتَّنُ ݒࣴرࣴگَمْ؞ مࣷو تَخْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ سَنتّ لࣹينْ نگࣹينْ وࣵݖّࣷو اَڪْ بࣺسُبْ نࣷفْلاَيْ؞
‏16 «تࣹرَلْ-لࣹينْ سࣹينْ بَايْ، اَڪْ سࣹينْ ندࣹيْ، نِ لࣹينْ ڪࣷ سࣹينْ يࣵلَّ اَجِ سَخْ جِ سَنْتࣹ، ندَخْ نگࣹينْ گُدّ فَنْ، ندَخْ اِتْ نگࣹينْ بَاخْلࣹ ݖَ رࣺيوْ مَ لࣹينْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ جࣷخْ؞
‏17 «بُلࣹينْ بٗومْ؞
‏18 «بُلࣹينْ جَالࣷو؞
‏19 «بُلࣹينْ سࣵݖّ؞
‏20 «بُلࣹينْ سࣹيدࣹيلْ سࣹينْ مࣷرࣷومْ لُ دُلْ دࣴگّ؞
‏21 «بُلࣹينْ خࣺدّ سࣹينْ جَبَرُ دࣴڪَّنْدࣷو، بُلࣹينْ خࣹمّࣹمْ ڪࣴرَمْ، مبَا تࣷولَمْ، مبَا جَامَمْ بُ گٗورْ، مبَا جَامَمْ بُ جِگࣺينْ، مبَا اَوْ نَگَمْ، مبَا مبَامَمْ، مبَا مبࣷولࣹمْ سࣹينْ اَلَلُ دࣴڪَّنْدࣷو؞»
‏22 ڪࣵدُّ يِي لَ اَجِ سَخْ جِ وَخْ سࣹينْ ندَجࣹ مَ مࣺݒّ، ݖَ تُندّ وَ، بَ مُ اࣵدّࣷو ݖَ دِگّ سَوَرَ وَ، ڪࣵدُّ گُ خُمبّ، ݖِ بِيرْ خࣵمْبَارْ گُ لࣴنْدࣴمْ ڪࣴرُوسْ؞ تࣹگَاتُ ݖَ دَرَ؞ مُ دَلْدِ بِندّ ڪࣵدُّ يِ ݖِ ݧَارِ اࣵلُّوَيْ دࣷجْ، جࣷخْ مَ؞ ‏23 بَ نگࣹينْ دࣺگّࣹي ڪࣵدُّ گَيْ جِبࣹ ݖَ لࣴنْدࣴمْ گَ، تُندّ وَ جࣵݒّ، دِ تࣵڪّ، يࣹينَ دِڪّ، سࣹينْ مبࣷولࣹمْ ڪِلِفَيْ گِيرْ اَڪْ سࣹينِ مَگْ، بَ ݖِ مَنْ؞ ‏24 نگࣹينْ نࣹ مَ:‏ «سُنُ يࣵلَّ اَجِ سَخْ جِ ڪَتْ، مࣷو نُ وࣷنْ دَرَجَامْ اَڪْ مࣵگَّايَمْ؞ ڪࣵدّࣷومْ لَنُ دࣺگّ، مُ جِبࣹ ݖَ دِگّ سَوَرَ وَ؞ بࣺسُبْ تࣹيْ نࣷو گِسْ يࣵلَّ وَخْ اَڪْ دࣷومْ آدَمَ، بَ نࣷݒِّ، مُيْ وࣺيࣹ بَڪَّنَمْ!‏ ‏25 وَايࣹ لُ نُيْ جَايَتِ نَگْ سُنُ بَڪَّنْ، دࣷيࣹ؟ ندَخْ ڪَتْ، سَوَرَ وُ رࣺيْ وِي، سُ نُ دࣺگَّتࣹي سُنُ ڪࣵدُّگْ يࣵلَّ اَجِ سَخْ جِ، دَ نُيْ خࣷيࣷمْ؞ ‏26 اَنَ ݖِ مبࣷولࣹمْ بࣷرࣷومُ سُوخْ، ڪُ ݖِ مَسَ دࣺگّ ڪࣵدُّگْ يࣵلَّ جِيْ دُندّ، مُ جِبࣹ ݖِ سَوَرَ، نِي نُ ڪࣷ دࣺگّࣹ، بَ نࣷݒِّ دِ دُندّ؟ ‏27 جࣹگࣹجِلْ يَوْ تࣹ دࣺگْلُ مبࣷولࣹمْ لِ سُنُ يࣵلَّ اَجِ سَخْ جِ وَخْ؞ يَوْ يَا نُيْ اࣵگّࣹ مبࣷولࣹمْ لِ لَ سُنُ يࣵلَّ اَجِ سَخْ جِ وَخْ، نُ دࣺگّ، جࣴفࣹ؞»
‏28 ݖِ بِيرْ لࣷولُ اَجِ سَخْ جِ دࣺگّ لَ نگࣹينْ مَ وَخْ؞ مُ نࣹ مَ:‏ «دࣺگّ نَا لِ لَ مبࣷولࣷو مِي وَخْ؞ ݧࣷو يࣹيْ وَخْ يࣷويُ ڪࣵدُّ يࣺݒّ؞ ‏29 وَايࣹ ݖࣺيْ بُ ݧُ اَمࣷونْ بࣷوبُ خࣷلْ نگِرْ رَگَلْ مَ، دِ سࣵمّ سَمَ سَنْتَانࣹ يࣺݒّ بࣺسْ بُ نࣹڪّ، بَ تَخْ ݧُ بَاخْلࣹ، ݧࣷومْ اَڪْ سࣹينْ اَسْڪَنْ، بَ فࣵوّ!‏ ‏30 دࣹمَلْ، نگَ نࣹ لࣹينْ ݧُ دࣹلُّ ݖَ سࣹينِ خَيْمَ؞ ‏31 يَوْ نَگْ، تَخَوَلْ فِ مَنْ، بَ مَ جࣷخْ لَ مبࣷولࣹمْ سَنْتَانࣹ بِ، دࣷگَلِ يࣷونْ يِ، اَڪْ اࣵتّࣹيْ يࣷونْ يِ نگَ لࣹينْ دِ جࣵنگَلْ، تࣹ ݧُ وَرْ ڪࣷو جࣴفࣹ ݖَ رࣺيوْ مَ مَ لࣹينْ جࣷخْ، نگِرْ ݧُ نَنگُ ڪࣷ؞»
‏32 ڪࣷنْ نَگْ سࣵمّࣷولࣹينْ ڪࣷو جࣴفࣹ نِ لࣹينْ ڪࣷ سࣹينْ يࣵلَّ اَجِ سَخْ جِ سَنْتࣹ؞ بُلࣹينْ لَجّࣹي ندِجࣷورْ مبَا ݖࣵمّࣷݧْ؞ ‏33 مبࣷولࣹمْ يࣷونْ وِ لࣹينْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ تࣹگْ، تࣷݒّلࣹينْ ڪࣷ، ندَخْ نگࣹينْ دُندّ، بَاخْلࣹ تࣹ گُدّ فَنْ ݖَ رࣺيوْ مَ نگࣹينْ دِ مࣷومْ؞
