ݒِيࣹيرْ دࣷخْلࣷو نَ اَبْ لَفَݧْ
3
‏1 ݒِيࣹيرْ اَڪْ يࣷوَانْ ݧࣷو دࣷونْ دُگّسِ ڪࣴرْ يࣵلَّ گَ، نگِرْ جُلِّگْ دِگّ نجࣷلّࣷورْ؞ ‏2 فࣹڪّ ݧُ اِنْدِ فَ وَا جُ جُدُّوَالࣹ لَفَݧْ، ݧُ دَانْ ڪࣷ تࣹگْ بࣺسْ بُ نࣹڪّ ݖَ بُنتّ، بَ ݧُيْ وࣷويࣹ بُنتّ بُ رَفࣹتْ، ݖَ ڪࣴرْ يࣵلَّ گَ، نگِرْ مُيْ سَرْخُ ݧَيْ دُگّسِ؞ ‏3 وَا جَ گِسْ ݒِيࣹيرْ اَڪْ يࣷوَانْ، ݧُيْ دُگّ ݖَ ڪࣴرْ يࣵلَّ گَ، مُ دِ لࣹينْ سَرْخُ؞ ‏4 ݖِ ڪَوْ لࣷولُ ݒِيࣹيرْ نࣹ ڪࣷ جࣵڪّ، مࣷوڪْ يࣷوَانْ، نࣹ ڪࣷ:‏ «خࣷولْ نُ؞» ‏5 لَفَݧْ بِ تࣹيوْلُ لࣹينْ، دِ لࣹينْ سࣺينْتࣷو اَبْ سَرَخْ؞
‏6 ݒِيࣹيرْ نَگْ نࣹ ڪࣷ:‏ «خَالِسْ اَڪْ وُرُسْ دࣹ، يࣷرُمَ ڪࣷ؞ وَايࣹ لِ مَ اَمْ، جࣷخْ نَا لَ ڪࣷ:‏ ݖِ تُرُ عِيسَىٰ اَلْمَسِ وَا نَسَرࣹتْ بِ، دࣷخَلْ!‏» ‏7 مُ جࣵݒّ ݖِ لࣷخࣷلْ ندِجࣷورَمْ، يࣺڪَّتِ ڪࣷ؞ ݖِ سَا سِ اَيْ تࣵنڪَمْ اَڪِ ڪࣷسْتَنَمْ دࣴگࣴرْ؞ ‏8 مُ نࣹ لَيّ، تَخَوْ، تࣵمْبَلࣹي دࣷخْ، اࣵندّ اَڪْ ݧࣷومْ دُگّ ݖَ ڪࣴرْ يࣵلَّ گَ، دِ دࣷخْ اَڪَ تࣴبَنْتُ، تࣹيْ سَنتّ يࣵلَّ؞ ‏9 مبࣷولࣷو مࣺݒّ گِسْ ڪࣷ، مُيْ دࣷخْ، دِ سَنتّ يࣵلَّ؞ ‏10 ݧُ خࣵمِّ ڪࣷ، خَمْ نࣹ مࣷو دَانْ تࣷوگْ، دِ يَلْوَانْ ݖَ بُنتّ بُ رَفࣹتْ بَ ݖَ ڪࣴرْ يࣵلَّ گَ؞ ݧُ جࣷمِّ بَ جࣷمَّارَلْ، بَ فَرْ گࣴلࣴمْ ݖَ لَ ڪࣷ دِڪَّلْ؞
ݒِيࣹيرْ وَارࣹ نَ
‏11 بَ لࣷولُ اَمࣹي وَا جِ تَقْ ݖِ ݒِيࣹيرْ اَڪْ يࣷوَانْ، ݧࣺݒّ دَوْسِ، سَگࣷ جࣹيخْ، ݧُ يࣺيوْ لࣹينْ ݖَ مبَارُ بُنتّ بَ ݧُ دِݒّࣹي سُلࣹيْمَانْ؞ ‏12 ݒِيࣹيرْ نَگْ گِسْ لࣷولُ، اࣵدُّ، نࣹ مبࣷولࣷو مَ:‏ «يࣹينْ بࣷڪِّ إِسْرَاءِيلْ، لُ نگࣹينْ دِ وَارُ ݖِ لِي؟ لُ نگࣹينْ نُ نَا جࣵڪّ، مࣹلْ نِ سُنُ مَنࣷورࣹيْ بࣷݒّ مبَا سُنُگْ نجُلِّتࣹ لَنُ دࣷخْلࣷو ڪِي؟ ‏13 مࣷومْ يࣵلَّيْ إِبْرَاهِيمَ، دِ يࣵلَّيْ إِسْحَاقَ، دِ يࣵلَّيْ يَعْقُوبَ، تࣹ دِ سُنُ يࣵلَّيْ مَامْ ڪَيْ، مࣷو مࣵگَّلْ جَوْرِݧَمْ عِيسَىٰ؞ يࣹينْ نَگْ يࣹينَ ڪࣷ تࣹگْ ݖِيْ لࣷخࣷ، جࣵمْبُ ڪࣷ فَ ڪَنَمْ ݒِلاَتْ، تࣹ مُ نَرࣷونْ ڪࣷو بࣵيِّ؞ ‏14 يࣹينْ، يࣹينَ جࣵمْبُ ڪُ سࣹلّ ڪِ تࣹ جُبْ، تِنُلْ اَبْ بٗومْڪَتْ؞ ‏15 بٗومْ نگࣹينْ ڪِيْ دُندّلࣷو، وَايࣹ يࣵلاَّ ڪࣷ دࣹڪَّلْ، نُ سࣹيدࣹ ڪࣷ؞ ‏16 گࣴمْ اَوْ تُرَمْ مࣷو وَرَلْ تُرَمْ وࣷووُ دࣷولࣹيلْ نِتْ ڪِي نگࣹينْ گِسْ، خَمْ ڪࣷ؞ ڪࣷنْ نگࣴمْ مࣷو وَرَلْ لِي؛ نگࣴمْ گُ سَبَبࣷو ݖِ عِيسَىٰ مࣷو ڪࣷ مَيْ مُ وࣺرْ ݒࣺݝّ نِي ݖِ سࣹينْ ڪَنَمْ، يࣹينْ ݧࣺݒّ؞
‏17 «تࣹيوُلْ نَگْ بࣷڪّ يِ، خَمْ نَا نࣹ ݧࣵڪَّ خَمْ مࣷو لࣹينْ تَخَ جࣴفࣹ نࣷونُ، تࣹ لࣷولُ اِتْ مࣷو دَلْ سࣹينِ ڪِلِفَ؞ ‏18 وَايࣹ يࣵلَّ مࣷو سࣷتَّلࣹ نࣷونُ لَ مُ يࣺگْلࣹ وࣷونْ لُ جِيتُ، يࣷنࣹنتّ يࣺݒّ جࣷتّلِ، نࣹ اَلْمَسࣹيمْ دِنَ سࣷنّ؞ ‏19 ڪࣷنْ نَگْ تُوبْلࣹينْ تࣹ دࣴݒّ، ندَخْ سࣹينِ بࣵڪَّارْ فَرُ؞ ‏20 سُ بࣷوبَا جَمࣷنࣷيْ ݒࣺيخْ اَيْ بَوࣷو ݖِ بࣷرࣷومْ بِ، تࣹ مࣷويْ يࣹبَلْ اَلْمَسِ مُ لࣹينْ جَگْلࣹيلْ، تࣹ مُيْ عِيسَىٰ؞ ‏21 مࣷومْ لَ اَسَمَانْ وَرَ دَلَلْ، بَ ڪࣹرَ جَمࣷنࣷ جَ لࣺݒّ دِ دࣹلُّ جَگْ، نَ ڪࣷ يࣵلَّ وَخࣹ وࣷونْ، يࣷنࣹنتّ يُ سࣹلّ يَ وࣷونْ جࣷتّلِ؞ ‏22 مُوسَىٰ ڪَيْ نࣷونْ نَ:‏ ‹بࣷرࣷومْ بِ سࣹينْ يࣵلَّ مࣷويْ سࣹݒّࣹي ݖِ سࣹينُ خࣹيتْ اَبْ يࣷنࣹنتّ نِ مَنْ؛ نَنگࣹينْ ڪࣷ دࣺگَّلْ ݖِ لࣺݒّ لُ مُ لࣹينْ وَخْ؞ ‏23 ڪࣺݒّ ڪُ دࣺگَّلُلْ يࣷنࣹنتّ بࣷوبُ، دࣹيسْ نَ لَ خࣹتّࣹي ݖِ خࣹيتْ وِ، سࣵنڪّ لَ؞›
‏24 «تࣹ اِتْ مبࣷولࣹمْ يࣷنࣹنتّ يِ وَخْ، دَلࣹ ڪࣷ ݖِ سَمِيࣹلْ اَڪْ ݧِ ڪࣷ وُوتُ، ݧࣷومْ اِتْ يࣺگْلࣹ وࣷونْ نَݧُ بࣺسْ يِي نُيْ دُندّ؞ ‏25 يࣹينْ نَگْ يࣹينَيْ دࣷنࣷيْ يࣷنࣹنتّ يِ تࣹ يࣹينَيْ سࣺدُّ ݖِ ڪٗلّࣴرࣹ، گِ يࣵلَّ فَسࣷونْ اَڪْ سࣹينِ مَامْ، بَ مُ نࣹي إِبْرَاهِيمَ:‏ ‹ݖِ سَوْ اَسْڪَنْ لَ خࣹيتِ اࣵدِّنَ يࣺݒّ دِ بَرْڪࣹيلْ؞› ‏26 يࣹينْ لَ يࣵلَّ جࣴڪَّ وࣷولْ جَوْرِݧَمْ، دَلْدِ ڪࣷيْ يࣹبَلْ ݖِ يࣹينْ، نگِرْ بَرْڪࣹيلْ لࣹينْ، وَݧّࣹي ڪࣹنّ ڪُ نࣹڪّ ݖِ يࣹينْ، مُ مࣷيْ جِڪّࣷومْ يُ بࣷنْ؞»
