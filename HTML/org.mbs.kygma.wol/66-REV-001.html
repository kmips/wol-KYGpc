ڪࣵدُّيْ وَخْيࣷو نگِ، ݧࣹيلْ يࣷوَانْ ݖِ بِيرْ
ݒࣹيݧُمْ اَلْمَسِ عِيسَىٰ
اُبِّتࣹ گِ
1
‏1 لِي ݒࣹيݧُمْ عِيسَىٰ اَلْمَسِ لَ، مُ ڪࣷ يࣵلَّ جࣷخْ نگِرْ مُ وࣷنْ اَيْ جَامَمْ يࣹنّ خࣹوْ-خࣹوْ يُ دࣴگْمَلْ؞ مَلاَڪَامْ لَ يࣹبَلْ ݖِ يࣷوَانْ نگِرْ خَمْلࣹ ڪࣷ؞ ‏2 يࣷوَانْ مࣷو سࣹيدࣹ مبࣷولࣹمْ لِ مُ گِسْ، تࣹ مُ جࣴمْ ݖِ ڪࣵدُّگْ يࣵلَّ اَڪْ سࣹيدࣹسْ عِيسَىٰ اَلْمَسِ؞ ‏3 ندࣷڪّلࣹي ڪࣺݒّ ڪُ جࣵنگّ اَڪْ مبࣷولࣹمْ ݧِ دࣺگْلُ يِي ڪࣵدُّيْ وَخْيُ، تࣹيْ سࣵمّ لِ ݧُ ݖِ بِندّ، ندَخْتࣹ وَخْتُ وِ جࣹگࣹ نَ؞
اَبْ نُيࣷو ݧࣹيلْ نَ جُرٗومْ ݧَارِ مبࣷولࣷويْ گࣴمْڪَتْ
‏4 لِي مُ نگِ تُڪّࣹي ݖِ مَنْ يࣷوَانْ، ݧࣹيلْ جُرٗومْ ݧَارِ مبࣷولࣷويْ گࣴمْڪَتْ ݧَ ݖَ دِيوَانُ اَسِ؞ نَ لࣹينْ يِوْ اَڪْ جࣵمّ ݧࣹيلْ، تُڪّࣹي فَ ڪِ مَسَ نࣹڪّ، ڪِ نࣹڪّ، دِ ڪِيْ دِڪّ، تُڪّࣹي اِتْ فَ بࣷرࣷومْ جُرٗومْ ݧَارِ نࣷو يَ¤ فَ ڪَنَمْ نگࣵنگُنࣹيمْ¤، ‏5 تُڪّࣹي اِتَمْ فَ عِيسَىٰ اَلْمَسِ، سࣹيدࣹ بُ وٗورْ بِ، مࣷومْ تَاوْ بِ¤، ڪِ جࣴڪَّ دࣹڪِّ، جٗگࣹ ݖِ ݧِ دࣹي، دِ ڪِلِفَيْ بُورِ ڪَوْ سُوفْ؞
مࣷومْ مِ نُ سࣷݒّ، يِوࣹي نُ دࣹرࣹتَمْ، بَ نُ تࣹقْلِڪࣷوڪْ ڪِلِفْتࣹيفُ سُنُيْ بࣵڪَّارْ، ‏6 بَ نࣷݒِّ دࣹفْ نُ نُيْ اَسْڪَنُ سَرْخَلْڪَتِ بُورْ يࣵلَّ بَايَمْ¤، نَ ڪࣷ دَرَجَ اَڪْ ڪࣵتَّنْ ݧࣹيلْ تࣹيْ اَڪْ اࣴلّࣴگْ بَ فࣵوّ؞ اَمِينْ؞
‏7 مُ نگࣷوگُ دِ دِڪّ، اِنْدَالࣹ نِيرْ يِ،
تࣹ ݧࣺݒَّ ڪࣷيْ تࣹگْ بࣴتْ،
بَ ݖَ ݧَ ڪࣷ جَمࣷونْ سَخْ،
تࣹ گِيرِ ڪَوْ سُوفْ يࣺݒّ اَيْ جࣷويْ ندَخْ مࣷومْ؞
نࣷونُ لَ مࣷوسْ؞ اَمِينْ؞
‏8 «مَنْ، مَايْ اَلْفَ، دِ اࣷمࣹگَ¤»، لَ بࣷرࣷومْ بِ يࣵلَّ وَخْ، مࣷومْ ڪِ مَسَ نࣹڪّ، ڪِ نࣹڪّ، دِ ڪِيْ دِڪّ اࣴلّࣴگْ، تࣹ دِ اَجِ مَنْ جِ؞
يࣷوَانْ اَمْ نَ ݒࣹيݧُ ݖَ دُنُ ݒَتْمࣷسْ
‏9 مَنْ يࣷوَانْ سࣹينْ مبࣷڪّ، مِ سࣺدّࣷوڪْ يࣹينْ نَتُّ اَڪْ بَرْڪࣹبْ نگُورُگْ يࣵلَّ، اَڪْ مُݧْ گِ ݖِ عِيسَىٰ، مَا نگَ وࣷونْ فَ ݧُ مَ تࣷخَلࣷونْ، ݖَ دُنْ بَ ݧُيْ وَخْ ݒَتْمࣷسْ؞ لِ ڪࣷ وَرَلْ دِ سَمَ وَارࣹ ݖِ ڪࣵدُّگْ يࣵلَّ اَڪْ سࣹيدࣹيلْ عِيسَىٰ؞ ‏10 مَا نࣹڪّࣷونْ ݖِ بِيرْ نࣷووُگْ يࣵلَّ بࣹنّ بࣺسُبْ سَنگّ بِ، مَ دࣺگّ ݖِ سَمَ گَنَّاوْ بَاتْ بُ خُمبّ نِگْ لِيتْ؞ ‏11 مُ نࣹ:‏ «مبࣷولࣹمْ لࣷو گِسْ، بِندّ ڪࣷ ݖِبْ تࣺيرࣹ، نگَ يٗنّࣹي ڪࣷ جُرٗومْ ݧَارِ مبࣷولࣷويْ گࣴمْڪَتْ ݧَ ݖَ دࣴڪّ يَ ݧُيْ وَخْ اࣹفࣹسْ اَڪْ سَمِرِنْ اَڪْ ݒࣹرْگَمْ اَڪْ ݖَتِرْ اَڪْ سࣵردّ اَڪْ فِلَدࣹلْفِ اَڪْ لَوْدِسࣹ؞»
‏12 نَڪَ لاَ گࣹيسْتُ، دِ سࣹيتْ ڪُيْ وَخْ اَڪْ مَنْ، مَ دَلْدِ گِسْ جُرٗومْ ݧَارِ تࣹگُڪَايِ لࣵمݒّ يُ وُرُسْ؞ ‏13 ݖَ دِگّ تࣹگُڪَايِ لࣵمݒّ يَ، جࣴمّ جُيْ نِرࣷوڪْ نِتَ¤ فَ نࣹڪّ؞ مَ نگَ سࣷلْ مبُبّ مُ تࣷلُّ ݖِ ندࣴگُّيْ تࣵنڪّ يِ، تَڪّࣷو نگَݧَايَلْ وُرُسْ ݖِ دࣴنّ بِ؞ ‏14 ڪَوَرُ بࣷݒّ بَا نگَ وࣹيخْ تࣵلّ، گࣴتْ يَ دِ تࣵڪّ نِ سَوَرَ، ‏15 تࣵنڪّ يَ مࣹلْ نِ خࣵنجَرْ بُ يࣵنجّ بُ ݧُ خࣹلِّ، بَاتْ بَيْ رِيرْ نِ گَنَّخِ گࣺيجْ؞ ‏16 مَ نگَ يࣷرْ ݖَ لࣷخࣷلْ ندِجࣷورَمْ جُرٗومْ ݧَارِ بِدِّيوْ، اَبْ سَامَرُ ݧَارِ ݧَوْڪَ بُ ݧَوْ دِ لَنگࣹ ݖَ گࣺمِّݧْ گَ، خَرْ-ڪَنَمْ بَا نگَ لࣹيرْ نࣵݧّ نِ جَنْتُبْ نجࣷلّࣷورْ؞
‏17 نَڪَ لاَ ڪࣷ گِسْ، نࣹ دࣵلّ دَانُ ݖِ اَيْ تࣵنڪَمْ نِ ڪُ دࣹي؞ مُ تࣹگْ مَ لࣷخࣷلْ ندِجࣷورَمْ، نࣹ مَ:‏ «بُلْ تِيتْ؞ مَايْ ڪِ جࣴڪّ، دِ ڪِ مُجّ؞ ‏18 مَايْ ڪِيْ دُندّ، مَا دࣹي وࣷونْ، مَايْ دُندّ نِي تࣹيْ اَڪْ اࣴلّࣴگْ بَ فࣵوّ تࣹ مَا يࣷرْ ݖَابِيْ دࣹي اَڪْ نجَنِيوْ؞ ‏19 ڪࣷنْ نَگْ بِنْدَلْ لِ نگَ گِسْ:‏ لِ اَمْ نِي تࣹيْ اَڪْ اِتَمْ لِيْ ݧࣴوْ گَنَّاوْ لِي؞ ‏20 ڪُمْݒَ گِ ݖِ جُرٗومْ ݧَارِ بِدِّيوْ يِ نگَ گِسْ ݖِ سَمَ لࣷخࣷلْ ندِجࣷورْ، اَڪْ جُرٗومْ ݧَارِ تࣹگُڪَايِ لࣵمݒّ يُ وُرُسْ يِ، لِ مُيْ فِرِ مࣷويْ:‏ جُرٗومْ ݧَارِ بِدِّيوْ يِ ݧࣷو دِ مَلاَڪَيْ جُرٗومْ ݧَارِ مبࣷولࣷويْ گࣴمْڪَتْ ݧِ؛ جُرٗومْ ݧَارِ تࣹگُڪَايِ لࣵمݒّ يِ ݧࣷو دِ جُرٗومْ ݧَارِ مبࣷولࣷو يِ؞»
