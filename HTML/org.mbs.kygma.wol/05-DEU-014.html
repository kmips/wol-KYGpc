يࣷونْ وِ سࣵرْتَلْ نَ جࣴفِنْ يُ سࣹتْ
14
‏1 اَيْ دࣷومْ نگࣹينْ، ݧࣹيلْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ؞ بُلࣹينْ ݧَاسُ، بُلࣹينْ وَتْ سࣹينْ ڪَوَرُ بࣷݒّ فِ فࣺيتࣹيڪْ جࣴ بِ، نگِرْ دِ ڪࣷ مِتِتْلࣷو ڪُ دࣹي، ‏2 ندَخْ اَسْڪَنْ وُ سࣹلّ نگࣹينْ، ݧࣹيلْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ؞ يࣹينْ لَ اَجِ سَخْ جِ تَامُ، نگِرْ نگࣹينْ دࣷونْ اَسْڪَنْ وِ مُ سࣺدّࣷولْ بࣷݒَّمْ، ݖِ مبࣷولࣹمْ اَسْڪَنِ ڪَوْ سُوفْ؞
‏3 بُلࣹينْ لࣹڪّ لࣹنّ لُ سࣹيخْلُوُ؞ ‏4 لِي مࣷو دِ مَلَ يِ نگࣹينْ دِ لࣹڪّ سࣹينْ سُوخْ:‏ نَگْ اَڪْ خَرْ اَڪْ بࣴيْ؛ ‏5 اَڪْ ڪࣷوبَ اَڪْ ڪࣺوࣺلْ، اَڪْ لُ خࣹيتࣷوڪْ ڪࣷوبَ، اَڪْ لُ خࣹيتࣷوڪْ ڪࣺوࣺلْ، اَڪْ بࣴيُ اࣵلّ، اَڪْ بࣴيُ ڪَوْ تُندّ؞ ‏6 مبࣷولࣹمْ مَلَ مُ ݖِ وࣹ وِ خَرْ، سࣺدّلِڪࣷو يَارْ، تࣹ مُيْ دُݧّ، مࣷومْ نگࣹينْ دِ لࣹڪّ اَوْ سُوخَمْ؞ ‏7 وَايࣹ يِي، بُلࣹينْ لࣹڪّ سࣹينْ سُوخْ:‏ لُ ݖِيْ دُݧّ دࣷݝّ، تࣹ وࣹ وِ خَرُلْ، مبَا لُ وࣹ وِ خَرْ، تࣹ دُݧُّلْ؞ لُ مࣹلْ نِ گِلࣺيمْ اَڪْ نجࣷمْبࣷورْ اَڪْ دَمَنْ، ندَخْ دَݧُيْ دُݧّ وَايࣹ سࣹينْ وࣹ خَرُلْ؞ يࣷويُ دَگَنُلْ ݖِ يࣹينْ؞ ‏8 مبَامْ-خُوخْ، مࣷومْ، وࣹ وِ خَرْ نَ وَايࣹ دُ دُݧّ؞ ڪࣷنْ دَگَنُلْ ݖِ يࣹينْ؞ اَوْ سُوخَمْ، بُلࣹينْ ڪࣷ لࣹڪّ، تࣹ مࣺدَّمْ، بُلࣹينْ ڪࣷ لاَلْ؞
‏9 لِي نَگْ، سَݧْ نگࣹينْ ڪࣷو لࣹڪّ، ݖِ مبࣷولࣹمْ لُ دࣴڪّࣹ ندࣷخْ:‏ مبࣷولࣹمْ لُ ݖِ اَمْ ݖِيرِيرْ اَڪِ وَاسِنْتٗورْ، سَݧْ نگࣹينْ ڪࣷو لࣹڪّ؞ ‏10 وَايࣹ لُ ݖِ اَمُلْ ݖِيرِيرْ اَڪِ وَاسِنْتٗورْ، بُلࣹينْ ڪࣷ لࣹڪّ، ندَخْ دَگَنُلْ ݖِ يࣹينْ؞
‏11 مبࣷولࣹمْ نجَنَاوْ لُ سࣹتْ، سَݧْ نگࣹينْ ڪࣷو لࣹڪّ؞ ‏12 وَايࣹ لِي نگࣹينْ دُلْ لࣹڪّ ݖِ نجَنَاوْ يِ:‏ جَخَايْ اَڪْ تَنْ، بَ ݖَ مُ ݧُولْ مَ، ‏13 اَڪْ ݖࣺيلِ بُ خࣷنقّ، اَڪْ ݖࣺيلِ بُ ݧُولْ، اَڪْ مَفْ، اَڪْ لُ مُ خࣹيتࣷولْ، ‏14 اَڪْ لࣺݒّ لُيْ بَاخࣷݧْ اَڪْ لُ مُ خࣹيتࣷولْ، ‏15‏-18 اَڪْ بࣵنجٗولِ اَڪْ لࣷويْ، اَڪْ لُ مُ خࣹيتࣷولْ، اَڪْ ڪِرِگࣺيجْ، اَڪْ مبࣷولࣹمْ خࣹيتِ لِݖِّنْ، اَڪْ يࣺبّ، اَڪْ لُ خࣹيتࣷوڪْ خࣷدّ، اَڪْ مُسࣹجُبْرُ اَڪْ نجُگُبْ؞ ‏19 مبࣷولࣹمْ بࣷرࣷومِ لاَفْ يُ سࣹوْ-سࣹوَانْ، دَگَنُلْ ݖِ يࣹينْ؞ دࣹيسُ ݖِ لࣹڪّ لࣹنّ، ‏20 خَنَا بࣷرࣷومِ لاَفْ يِ ݖِ سࣹتْ؞ يࣷويُ، سَݧْ نگࣹينْ ڪࣷو لࣹڪّ؞
‏21 بُلࣹينْ لࣹڪّ لࣹنّ لُ مࣺدّ؞ جَامْبُرْ بُ دَلْ اَڪْ يࣹينْ، سَݧْ نگࣹينْ ڪࣷو جࣷخْ لُ مࣺدّ، مُ لࣹڪّ، مبَا نگࣹينْ جَايْ ڪࣷ اَبْ دࣷخَنْدࣺيمْ؞ وَايࣹ يࣹينْ اَسْڪَنْ وُ سࣹلّ نگࣹينْ، ݧࣹيلْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ؞
تࣹ بُلࣹينْ تࣷگّ اَبْ تࣹفْ ݖِ مࣹيوُمْ يَايَمْ؞
لُ جࣴمْ ݖِ ݖࣺرِ فُڪّࣹيلْ
‏22 فࣹخࣹلࣹينْ بَيْ گࣺنّࣹ بࣹنّ ݖࣺرُ فُڪّࣹيلْ ݖِ مبࣷولࣹمْ لُ سࣹينِ تࣷولْ مࣹݧّ اَتْ مُ نࣹڪّ؞ ‏23 سࣹينْ ݖࣺرُبْ فُڪّࣹيلُ ݒࣹݒّ، اَڪْ بِيݧْ بُ بࣹيسْ، اَڪْ دِوْ گُ بࣹيسْ، اَڪْ لُ سࣹينْ جُرْ تَاوْلࣷو، مُيْ نَگْ، دِ گࣵتّ، نَنگࣹينْ ڪࣷ خࣺيوْلࣷو فِ سࣹينْ ڪَنَمْ يࣵلَّ اَجِ سَخْ جِ، فِ بࣺرَبْ بِ مُ تَامࣷو دࣴعࣴلْ تُرَمْ اࣴلّࣴگْ؞ سُ ڪࣷ دࣹفࣹي نگࣹينْ تࣵگَّتُ ݖِ رَگَلْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ فُ نگࣹينْ تࣷلُّ؞ ‏24 سُ نگࣹينْ مَنُلَ دࣷخْ يࣷونْ وِ، ندَخْ سࣹينْ يࣵلَّ اَجِ سَخْ جࣹي لࣹينْ بَرْڪࣹيلࣹ مࣹݧّࣹيفْ مُ بَرࣹ مُ نگࣹينْ دُلْ اࣵتَّنْ، ݖِ دِگَّنْتࣹ بُ سࣷرࣹوَاتࣷو لࣷولْ اَڪْ بࣺرَبْ بِ سࣹينْ يࣵلَّ اَجِ سَخْ جِ تَامࣷو دࣴعࣴلْ تُرَمْ، ‏25 سُ بࣷوبَا دَنگࣹينْ دِ وࣹݖِّ ݖࣺرُبْ فُڪّࣹيلْ بَ خَالِسْ، جࣴلْ خَالِسْ بَ تࣹ دࣹمْ بَ فَ سࣹينْ يࣵلَّ اَجِ سَخْ جِ تَامُ؞ ‏26 سُ ڪࣷ دࣹفࣹي نگࣹينْ وࣹݖِّڪࣷو فَ خَالِسْ بِ لُ لࣹينْ نࣹيخْ، مُيْ نَگْ مبَا گࣵتّ، مبَا بِيݧْ، مبَا گࣹنࣹينْ نَانْ گُ دِيسْ، اَڪْ لُ لࣹينْ سࣹينْ بَڪَّنْ خِيرْ، نگࣹينْ خࣺيوْلࣷو فَ سࣹينْ ڪَنَمْ يࣵلَّ اَجِ سَخْ جِ، بࣵنّࣹيخࣷو ڪࣷ، يࣹينْ اَڪْ سࣹينْ نجَبࣷوتْ؞ ‏27 لࣹوࣹينْ گِ ݖِ سࣹينْ بِيرْ دࣴڪّ نَگْ، بُلࣹينْ لࣹينْ فࣵتّࣹ، ندَخْ اَمُݧُ وࣵلّ ݖِ سࣹينْ ݖࣺرُ سُوفْ؞
‏28 ݧࣹتِّ اَتْ يُ مَتْ يُ نࣹڪّ، نَنگࣹينْ گࣺنّࣹ مبࣷولࣹمْ سࣹينْ ݖࣺرِ فُڪّࣹيلِ مࣹݧّࣹيفْ ݖَ اَتْ مࣷومَ، تࣹ نگࣹينْ دࣹنݖّ ڪࣷ ݖَ سࣹينْ بِيرْ دࣴڪّ، ‏29 ندَخْ لࣹوࣹينْ، ݧِ اَمُلْ وࣵلّ ݖِ سࣹينْ ݖࣺرُ سُوفْ، دِڪّ، ݧࣷوڪْ دࣷخَنْدࣺيمْ يِ اَڪْ جِرِمْ يِ اَڪْ جࣴتُنْ، يِ نگࣹينْ دࣴڪَّلْ، نگِرْ ݧُ لࣹڪّ بَ رࣹگّ؞ سُ ڪࣷ دࣹفࣹي سࣹينْ يࣵلَّ اَجِ سَخْ جِ بَرْڪࣹيلْ مبࣷولࣹمْ سࣹينُ ݧَقْ؞
