نࣷو گُ سࣹلّ گِ دِڪّ نَ
2
‏1 بَ بࣺسُ مࣵگَّلُ ݒࣵنْتَڪࣷتْ اَگْسࣹي، ݧࣷومْ ݧࣺݒَّ بࣷڪّ دَجࣹ بࣹنّ بࣺرَبْ؞ ‏2 ݖَ سَا سَ ݖࣷووْ جࣷلّࣹي اَسَمَانْ، مࣹلْ نِ رِيرُمْ نگࣹلَوْ لُ بَرࣹ دࣷولࣹ، دَلْدِ فࣹيسْ دࣹلّ بِيرْ ڪࣴرْ گَ ݧُ تࣷوگْ؞ ‏3 ݖِ ڪَوْ لࣷولُ اَيْ لࣵمِّݧْ يُ مࣹلْ نِ تࣵڪّ-تࣵڪِّ سَوَرَ فࣹيݧُ لࣹينْ، لࣵمِّݧْ يَ سࣺدّلِڪࣷو، دَلْدِ تࣷݝْ ݖِ ڪَوْ ڪࣹنّ ڪُ نࣹڪّ؞ ‏4 ݧࣷومْ ݧࣺݒّ نَگْ فࣹيسࣹ نࣷو گُ سࣹلّ گِ، تࣵمْبَلࣹي لࣵڪّ لࣵڪّ يُ وُوتࣹ، نَ لࣹينْ ڪࣷ نࣷو گِ مَنْلࣷو؞
‏5 بࣷوبَ اَيْ تࣷݒّڪَتِ يࣷونُ يَوُتَ نگَ دࣴڪّࣹ يࣹرُسَلࣹمْ، تࣹ بَوࣷو ݖِ خࣹيتِ اࣵدِّنَ يࣺݒّ؞ ‏6 نَڪَ لَ ݖࣷووْ لِ جࣷلِّ، نِتْ ݧِ دَجࣹ، ڪُ نࣹڪّ ݖِ مبࣷولࣷو مِ دࣺگّ ݧُيْ لࣵڪّ سَ لࣵڪُّ بࣷݒّ، مبࣷولࣷو مَ جَاخْلࣹ؞ ‏7 ݧُ بࣷولࣹ وَارُ اَڪْ يࣺيمُ، نَا:‏ «ݧِي دِ وَخْ نِي ݧࣺݒّ، خَنَا دُݧُ وَا گَلِلࣹ؟ ‏8 اَنَ نُ ڪࣹنّ ڪُ نࣹڪّ ݖِ نُنْ نَگْ مَنَ دࣺگّࣹ ݧُيْ وَخْ ݖِ لࣵمِّݧْ وِ نگَ نࣵمݒّ؟ ‏9 مُيْ وَا ݒࣵرتّ، اَڪْ مࣹدّ اَڪْ اࣹلَمْ اَڪْ مࣹسࣷݒࣷتَمِ اَڪْ يُدࣹ اَڪْ ڪَݒَدࣷسْ اَڪْ ݒࣷنتّ اَڪْ اَسِ، ‏10 اَڪْ فِرِسِ اَڪْ ݒَمْفِلِ اَڪْ مِسْرَ اَڪْ وࣵلُّ لِبِ گَ دࣹنْدࣹيڪْ سِرࣹنْ، اَڪْ گَنْ ݧِ جٗگࣹ رࣷومْ، ‏11 اَڪْ يَوُتِ نجُدُّ-جِ-رࣺيوْ يࣹيڪْ تُوبࣹينْ ݧِ، اَڪْ وَا ڪࣹرࣹتْ اَڪْ اَرَبِ، نُنْ ݧࣺݒَّ لࣹينْ دࣺگّ ݖِ سُنُ لࣵمِّݧِ بࣷݒّ، ݧُيْ سِيوْتَانࣹ جَلࣷورࣹ يُ يࣵلَّ!‏» ‏12 ݧࣺݒّ وَارُ، بَ جࣵنَّخࣹ، دِ وَخْ ݖِ سࣹينْ بِيرْ نَا:‏ «لِي لُ مُيْ تࣹڪِّ نَگْ؟» ‏13 تࣹيوُلْ ݧࣹنّ ݧَيْ ݧَاوْلࣹ، نࣹ:‏ «ݧِي دَالْ بِيݧْ بُ بࣹيسْ دࣷݝّ لَݧُ نَانْ بَ مࣵنْدِ؞»
ݒِيࣹيرْ يࣹي نَ مبࣷولࣷو مَ
‏14 ݖِ ڪَوْ لࣷولُ ݒِيࣹيرْ تَخَوْ، مࣷوڪْ فُڪّ ݧَاڪْ بࣹنّ، دَلْدِ اࣵدُّ ݖَ ڪَوْ، نࣹ:‏ «يࣹينْ بࣷڪِّ يَوُتْ يِ، اَڪْ مبࣷولࣹمْ يࣹينْ ݧِ دࣴڪّࣹ يࣹرُسَلࣹمْ، دࣺگْلُلࣹينْ بُ بَاخْ سَمَ ڪࣵدُّ، بَ خَمْ لِي لُ مُ دࣷونْ؞ ‏15 ݧِي مࣵنْدِوُݧُ، نِ نگࣹينْ ڪࣷ فࣷوگࣹ، ندَخْ يࣷورْ-يࣷورْ دࣷݝَّ جࣷتْ؞ ‏16 لِي ڪَيْ مࣷويْ ڪࣵدُّ گَ يࣷنࣹنتّ يࣵلَّ يࣷوࣹلْ جࣷتّلِ وࣷونْ، نࣹ:‏
‏17 ‹يࣵلَّ نࣹي:‏
مُجُگْ جَمࣷنࣷ، مَايْ تُورْ سَمَگْ نࣷو
ݖِ ڪَوْ وࣺݒّ سُوخْ؛
سࣹينِ دࣷومْ، گٗورْ اَڪْ جِگࣺينْ دِ بِرَلْ وَخْيُ،
سࣹينِ خَلࣹ يُ گٗورْ دِ گِسْ اَيْ ݒࣹيݧُ،
مࣵگَّتْ ݧِ دِ گࣺنتّ اَيْ گࣺنتّ؞
‏18 ڪࣹرَ جَنتّ يࣷويُ، سَمَ جَامْ ݧِ، گٗورْ اَڪْ جِگࣺينْ،
مَا لࣹينْ دِ تُورْ سَمَگْ نࣷو،
ݧُيْ بِرَلْ وَخْيُ؞
‏19 تࣹ مَايْ دࣹفْ اَيْ ڪِرَامَ فِ اَسَمَانْ
اَڪِ فِرْندࣹ ݖِ دُنْ بِ فِ سُوفْ؛
مُيْ دࣹرࣹتْ اَڪْ سَوَرَ اَڪْ نِيرُ سَخَرْ؞
‏20 جَنتّ بࣹييْ سࣷݒّلِڪُ اَگْ لࣴنْدࣴمْ،
وࣹيرْ وِ دࣷونْ دࣹرࣹتْ،
بࣺسُبْ بࣷرࣷومْ بِ دࣷورَ تَخَوࣹ مࣵگَّايْ اَڪْ دَرَجَ؞
‏21 سُ بࣷوبَا، ڪُ وࣷو بࣷرࣷومْ بِ وَلّ ݖِوْ تُرَمْ،
دِنگَ رَوْ؞›
‏22 «يࣹينْ بࣷڪِّ إِسْرَاءِيلْ، دࣺگْلُلࣹينْ وَخْ جِي!‏ عِيسَىٰ وَا نَسَرࣹتْ بِ، يࣵلَّ مࣷو فِرْندࣹيلࣹ اَگْ دࣴگّࣷومْ، اَيْ ڪࣺيمَانْ اَڪِ ڪِرَامَ اَڪِ فِرْندࣹ يُ مُ سࣷتَّلࣹ جࣴمُّ عِيسَىٰ مࣷومُ فِ سࣹينْ بِيرْ، تࣹ لࣷولُ خَمْ نگࣹينْ ڪࣷ خࣺلّ؞ ‏23 نِتْ ڪࣷوڪُ ݧُ جࣺبَّلࣹ نِ ڪࣷ يࣵلَّ دࣷگَلࣹ تࣹ نَسࣷونْ ڪࣷ لُ جِيتُ، يࣹينَ ڪࣷ دَاجْلࣷو يࣺيفَرْ يِ ݖِ بَنتّ، رࣹيْلُ ڪࣷ؞ ‏24 وَايࣹ مࣷومْ لَ يࣵلَّ يِوࣹي ݖِ مِتِتُ ندࣹي، دࣹڪَّلْ ڪࣷ، ندَخْ دࣹي اَمُلْ ڪࣵتَّنُ تࣴنڪّ ڪࣷ؞ ‏25 دَاوُدَ وَخْ نَ ݖِ مبِرَمْ نࣹ:‏
‹مَا دࣷونْ گِسْ بࣷرࣷومْ بِ فِ سَمَ ڪَنَمْ سَا سُ نࣹڪّ،
ندِجࣷورْ لَ مَ فَرࣹ،
بَ دُمَ تࣴرࣴفْ؞
‏26 مࣷو تَخْ سَمَ خࣷلْ دِ نُخْ-نُخِ،
سَمَوْ لࣵمِّݧْ دِ تࣹخْتࣹخِ،
سَمَوْ سُوخْ سَخْ نَرَ نࣷݒّلُجࣹي يَاڪَارْ؞
‏27 دࣷو مَ وَݖّ نجَنِيوْ مࣷوسْ،
دࣷو سࣹيتَانْ سَ وَا جُ سࣹلّ، مُ يࣵقُ؞
‏28 يَا مَ خَمَلْ يࣷونُ دُندّ،
يَا مَيْ وࣺتَّلِ بَ مَتَلْ سَمَ مبࣺگْتࣹ؞›
‏29 «بࣷڪّ يِ نَگْ، مَامْ دَاوُدَ، مَنࣹيسْ نَ لࣹينَ وَخْ ݖِ لُ وࣺرْ نࣹ نࣹلَوْ نَ، بَ ݧُ دࣹنݖّ ڪࣷ، تࣹ بࣵمّࣹيلَمَ نگِ فِ، نُيْ گِسْ بَ تࣹيْ جِي؞ ‏30 وَايࣹ يࣷنࣹنتّ لَ وࣷونْ، تࣹ خَمࣷونْ نَ نࣹ يࣵلاَّ ڪࣷ گِݧَلࣷونْ نࣹ ڪُ بࣷڪّ ݖِ گࣹݧࣷومْ لَيْ تࣹگْ ݖِبْ جَلَمْ؞ ‏31 ڪࣷنْ لِ مُ گِسْ تࣹ دِڪَّگُلْ مࣷويْ ندࣹڪِّتࣹلْ اَلْمَسِ، تࣹ ݖِ مبِرَمْ لَ وَخْ نࣹ بࣵيّࣹيسُ ڪࣷ بِيرْ نجَنِيوْ، تࣹ يَرَمَمْ دُ يࣵقُ؞ ‏32 عِيسَىٰ مࣷومُ نَگْ، يࣵلاَّ ڪࣷ دࣹڪَّلْ، نُ سࣹيدࣹ ڪࣷ، نُنْ ݧࣺݒّ؞ ‏33 يࣵلاَّ ڪࣷ يࣺيگࣹ فَ ندِجࣷورَمْ، تࣹ جࣷتْ نَ ݖِ بَايْ بِ نࣷو گُ سࣹلّ گَ مُ دِگࣹ وࣷونْ؞ مࣷومْ لَ تُورْ نِي تࣹيْ، نگࣹينْ گِسْ ڪࣷ، دࣺگّ ڪࣷ؞ ‏34 ندَخْ ڪَتْ دُ دَاوُدَا يࣺيگْ اَسَمَانْ، وَايࣹ مࣷومْ ݖِ بࣷݒَّمْ مࣷو نࣹ:‏
‹بࣷرࣷومْ بِ دَا وَخْ سَمَ سَنگّ، نࣹ ڪࣷ:‏
«تࣷوگࣹيلْ سَمَ ندِجࣷورْ،
‏35 بَ مَ دࣹفْ سَيْ نࣷونْ، سَ ندࣴگَّسْتَلُ تࣵنڪّ؞» ›
‏36 «ڪࣷنْ نَگْ نَ وٗورْ وَا ڪࣴرْ إِسْرَاءِيلْ گࣺݒّ نࣹ عِيسَىٰ مࣷومُ نگࣹينْ دَاجࣷونْ ݖِ بَنتّ، يࣵلاَّ ڪࣷ دࣹفْ سَنگّ اَڪْ اَلْمَسِ؞»
‏37 بَ مبࣷولࣷو مَ دࣺگّࣹي لࣷولُ، نَقَرْ وُ مࣹلْ نِ جَمْ-جَمُ خࣷلْ لَ لࣹينْ دࣹفْ؞ ݧُ نࣹ ݒِيࣹيرْ اَڪْ ندَوْ يَ ݖَ دࣹسْ:‏ «بࣷڪّ يِ، نُ نُيْ دࣹفْ نَگْ؟» ‏38 ݒِيࣹيرْ نࣹ لࣹينْ:‏ «تُوبْلࣹينْ تࣹ ڪُ نࣹڪّ ݖِ يࣹينْ سٗوبُ ݖِ ندࣷخْ ݖِ تُرُ عِيسَىٰ اَلْمَسِ نگِرْ نجࣺگَّلُگْ بࣵڪَّارَمْ، نگࣹينْ دَلْدِ جࣷتْ ݖِ نࣷو گُ سࣹلّ گِيْ مَيࣹ؞ ‏39 ندَخْ دِگࣹ بِ، يࣹينْ اَڪْ سࣹينْ اَسْڪَنَ ڪࣷ مࣷومْ، اَڪْ جَامْبُرْ ݧِ فࣺيتࣹ فُ سࣷرࣹ ݧࣺݒّ، مبࣷولࣹمْ ݧِ سُنُ يࣵلَّ بࣷرࣷومْ بِ وࣷو؞» ‏40 يࣹنࣹينْ ڪࣵدُّ يُ بَرࣹ سࣷݧّࣹ نَ لࣹينْ ڪࣷ، ݧَاخࣹ لࣹينْ ڪࣷ، نࣹ لࣹينْ:‏ «مُݖّلࣹينْ ݖِ گِي مَاسْ گُ دࣴنگّ؞»
‏41 ݧَ نَنگُ ڪࣵدُّگْ ݒِيࣹيرْ نَگْ، ݧُ سٗوبْ لࣹينْ ݖِ ندࣷخْ؛ بࣺسُبْ ڪࣹرࣷوگْ لُ وَرَ مَتْ ݧࣹتِّ جُنِّ (3000) يࣷڪُّ نَݧُ ݖَ گࣴمْڪَتْ ݧَ؞
گࣴمْڪَتْ ݧِ بࣹنّࣷو، يࣵلَّ بَرْڪࣹيلْ
‏42 ݖِ ڪَوْ لࣷولُ ݧُ ݒَسْتࣺيفُ ݖِ نجࣵنگّلࣹمْ ندَوْ يِ اَڪْ ݖِ دُنْدَلْ اَگْ بࣷڪّࣷو، اَڪْ ݖِ دَگّ مبُرُمْ بࣷڪّࣷو مِ¤، اَڪْ ݖِ جَتَايِ جُلِّ؞ ‏43 بَ لࣷولُ اَمࣹي اَگْ رَگَلْ جࣵݒّ ݧࣺݒّ، اَيْ ڪِرَامَ يُ بَرࣹ اَڪِ فِرْندࣹ دَلْدِ سࣷتّࣹي ݖَ جࣴفِ ندَوْ يَ؞ ‏44 گࣴمْڪَتْ ݧࣺݒّ بࣹنّ لَݧُ وࣷونْ تࣹ ݧࣷو بࣷڪّࣷونْ لࣺݒّ؞ ‏45 ݖِ بِيرْ لࣷولُ سࣹينِ سُوفْ اَڪْ سࣹينْ اَلَلْ سَخْ، ݧُ جَايْ، سࣺدّࣷو ڪࣷ ݧࣷومْ ݧࣺݒّ نَ مُ دࣴݒّࣷوڪْ سࣹينِ سࣷخْلَ؞ ‏46 بࣺسْ بُ نࣹڪّ نَگْ ݧُ دࣷونْ بࣹنّ بࣷݒّ، سَخࣷو تࣹيوْ ݖَ ڪࣴرْ يࣵلَّ گَ، دِ دَگّ مبُرُمْ بࣷڪّࣷو ݖِ ڪࣴرࣷو ڪࣴرْ، تࣹيْ بࣷڪّ لࣹڪّ ݖِ بِيرْ مبࣺگْ اَڪْ وࣷيࣷفْلُ؞ ‏47 ݧُيْ مࣵگَّلْ يࣵلَّ، ݧࣺݒّ نَوْ لࣹينْ؞ بࣺسْ بُ نࣹڪّ نَگْ بࣷرࣷومْ بِ دِ يࣷڪّ ݖَ مبࣷولࣷو مࣷومَ، نِتْ ݧَ جࣷتَ مُݖّ؞
