يࣵلَّ فَسْ نَ يࣺينࣹ تَسْ اࣵدِّنَ ندَخْ مبࣷنُ نِتْ
6
‏1 بِ نگَ خَمࣹي نࣹ نِتْ ݧِ تࣵمْبَلِ نَݧࣷو فُلَنْدِوُ ݖِ ڪَوْ سُوفْ، تࣹ اَمْ اَيْ دࣷومْ يُ جِگࣺينْ، ‏2 اَيْ گࣷنࣹيْ يࣵلَّ ݧࣷو گِسْ جِگࣺينْ ݧࣹيڪْ سࣹينْ تَارْ، ݧُيْ جࣴلْ ݧِ ݧُ بࣴگّ ݖِ ݧࣷومْ، دࣹفْ لࣹينِ جَبَرْ؞ ‏3 اَجِ سَخْ جِ دَلْدِ وَخْ نࣹ:‏ «مَنُمَا بࣵيِّ نِتْ، مُيْ وࣺيࣹي دُنْدࣹ نࣷو، گِ مَ ڪࣷ مَيْ، ندَخْ نِتْ دُ دَرَ لُ دُلْ سُوخِ نࣹينْ، ڪࣷنْ دِنَا لࣹينْ اࣵݒَّلْ دِيرُبْ تࣺيمࣺيرِ اَتْ اَڪْ ݧَارْ فُڪّ (120)؞»
‏4 ݖَ جَمࣷنࣷ يࣷويَ اَڪْ اِتَمْ گَنَّاوْ بَ گࣷنࣹيْ يࣵلَّ يࣷويُ اࣵنْدࣹيڪْ جِگࣺينِ اࣵدِّنَ، بَ اَمْ ݖَ اَيْ دࣷومْ، اَوْ خࣹيتُ ݒࣷنڪَلْ وُ ݧُيْ وَخْ نࣹفِلِمْ مࣷو نࣹڪّࣷونْ ݖِ اࣵدِّنَ؞ ݧࣷويْ ݒࣷنڪَلْ يَ وࣷونْ تࣹ دࣷونࣷونْ گٗورْ ݧُ اَمࣷونْ تُرْ؞
‏5 اَجِ سَخْ جِ دَفَ گِسْ نࣹ مبࣷنُ نِتْ مࣵگّ نَ لࣷولْ ݖِ بِيرْ اࣵدِّنَ، تࣹ فُ مُ مَنَ تࣷلُّ، لُ بࣷنْ رࣹڪّ لَيْ خِنْتࣹ ݖِ خࣷلَمْ؞ ‏6 اَجِ سَخْ جِ نَگْ اَمْ تِيسْ ݖِ خࣷلَمْ، بَ مِتِتْلُ لِ مُ دࣷونْ سࣵڪّ نِتْ ݖِ ڪَوْ سُوفْ؞ ‏7 اَجِ سَخْ جِ دَلْدِ نࣹ:‏ «دِنَا فَرْ ݖِ ڪَوْ سُوفْ دࣷومْ آدَمَ يِ مَ سࣵڪّ، نِتْ اَڪْ جُرْ، لُيْ رَامْ اَڪْ لُيْ نَاوْ، ندَخْ مِتِتْلُ نَا لِ مَ لࣹينْ سࣵڪّ؞» ‏8 وَايࣹ فࣹڪّ نَ اَجِ سَخْ جِ خࣷولْ نُوحٍ بࣴتِ يࣴرْمَنْدࣹ؞
اَسْڪَنُ نُوحٍ
(سَارْ 6‏.9—9‏.29)
‏9 اَسْڪَنُ نُوحٍ مࣷويْ لِي؞ نُوحٍ نِتْ ڪُ جُبْ لَ وࣷونْ، دِ ڪُ مَتْ ݖِ بِيرْ نِتْ ݧِ مُ بࣷڪَّلْ جَمࣷنࣷ، تࣹ دِ ڪُ تࣷݒّ يࣵلَّ؞ ‏10 نُوحٍ نَگْ اَمࣷونْ نَ ݧࣹتِّ دࣷومْ يُ گٗورْ:‏ سࣹينِ تُرْ دِ سࣹمْ، خَمْ اَڪْ يَفࣹتْ؞
‏11 بࣷوبَ فࣹڪّ نَ اࣵدِّنَ يࣵقُ، تࣹ فࣹيسْ اَڪْ اَيْ، يࣵلَّ تࣹگْ ݖِ بࣴتَمْ؞ ‏12 يࣵلَّ گِسْ نࣹ اࣵدِّنَ يࣵقُ نَ، ندَخْتࣹ جࣴفِ ݧࣺݒَّ بࣷڪّ ݧَاوْ فِ ڪَوْ سُوفْ؞
‏13 بَ مُ ڪࣷ دࣹفࣹي يࣵلَّ نࣹ نُوحٍ:‏ «دࣷگَلْ نَا نࣹ دِنَا بࣷولࣹ فَاتْ مبࣷولࣹمْ بࣷرࣷومْ بَڪَّنْ، ندَخْتࣹ اَيُ نِتْ دَجَلْ نَ ڪَوْ سُوفْ؞ دِنَا لࣹينْ بࣷولࣹ اَڪْ لِ ݖِ اࣵدِّنَ، فَاگَاگَلْ؞
‏14 «يَوْ نَگْ وُتَلْ دࣺنڪُ سِݒَّرْ، يࣹتَّلْ ݖِ سَ بࣷݒّ گَالْ گُ مَگْ؛ نگَ سࣵڪّ ݖِ اَيْ نࣺيگْ، تࣹ دِوْ ڪࣷ ڪࣷلْتَارْ بِيرْ اَڪْ بِتِ؞ ‏15 نِ نگَ ڪࣷيْ دࣹفَرࣹ نِي لَ؞ نَ گُدَّايُ گَالْ گِ دِ ݧࣹتِّ تࣺيمࣺيرِ خَسَبْ، يَاتُوَايْ بِ دِ جُرٗومْ فُڪِّ خَسَبْ، تَخَوَايْ بِ دِ فَنْوࣹيرِ خَسَبْ؞ ‏16 يࣹتَّلْ مبَارْ، مُ ڪࣷ تِيمْ، تࣹ نگَ اࣵگّلࣹ گَالْ گِ، بࣵيِّ لُ تࣷلّࣷوڪْ خَسَبْ دِگَّنْتࣹ ڪࣺمࣺجْ يِ اَڪْ مبَارْ مِ؞ سࣵڪَّلْ ڪࣷ بُنتّ ݖِ وࣹتْ گِ، تࣹ سࣺدَّلࣹ ڪࣷ ݧࣹتِّ نࣺيگْ يُ تࣹگْلࣷو، بࣹنّ ݖِ سُوفْ، بࣹنّ ݖِ دِگّ بِ اَڪْ بࣹنّ ݖِ ڪَوْ؞
‏17 «مَا نگِ نِي دِ اِنْدِ مبࣴنُّ ندࣷخْ ݖِ ڪَوْ سُوفْ، مُ فَرْ لࣺݒّ لُيْ دُندّ لُ اَسَمَانْ تِيمْ؛ تࣹ لࣺݒّ لُ نࣹڪّ ݖِ ڪَوْ سُوفْ دِنَ دࣹي؞ ‏18 وَايࣹ دِنَا فَسْ ڪٗلّࣴرࣹ اَڪْ يَوْ، نگَ دُگّ ݖِ گَالْ گِ، يَاڪْ سَيْ دࣷومْ اَڪْ سَ سࣷخْنَ اَڪْ سَيْ سࣷخْنَيْ دࣷومْ؞ ‏19 تࣹ خࣹيتِ رَبْ يࣺݒّ نَنگَ ݖِ دُگَلْ ݧَارْ ݖِ گَالْ گِ، وُ ݖِ نࣹڪّ، نگَ دࣹفْ ڪࣷ گٗورْ اَڪْ جِگࣺينْ؛ سُ ڪࣷ دࣹفࣹي نگࣹينْ اࣵندّ مُݖّ؞ ‏20 خࣹيتُ ݒِݖّ وُ ݖِ نࣹڪّ اَڪْ خࣹيتُ بࣵيِّمَ وُ ݖِ نࣹڪّ اَڪْ وࣺݒّ خࣹيتُ رَبْ وُيْ رَامْ ݖِ سُوفْ، دِنَ دࣹفْ ݧَارْ، ݧࣴوْ ݖِ يَوْ، بَ رَوْ؞ ‏21 سࣵڪَّلْ ݖِ بࣺݒّ خࣹيتُ دُندّ، دࣹنݖّ، نگࣹينْ بࣷڪّ دُنْدࣹ؞»
‏22 نࣷونُ لَ نُوحٍ دࣹفْ؞ لࣺݒّ لُ ڪࣷ يࣵلَّ سَنتّ، دࣹفْ نَ ڪࣷ؞
