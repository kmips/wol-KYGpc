لُ جࣴمْ ݖِ سࣵرْتِ دُنْدِنْ ݖَ رࣺيوُمْ دِگࣹ بَ
(سَارْ 12—26)
اَجِ سَخْ جِ مࣷويْ تَامُ بࣺرَبَمْ
12
‏1 دࣷگَلْ يࣹيڪْ اࣵتّࣹيْ يࣷونْ، يِ نگࣹينْ دِ سࣵمّࣷو جࣴفࣹ سࣹينْ گِيرُ دُندّ ݖَ سُوفَسْ رࣺيوْ، مَ لࣹينْ سࣹينْ يࣵلَّيْ مَامْ اَجِ سَخْ جِ مࣷومَلࣹ، مࣷو دِ يِي:‏ ‏2 تَسْلࣹينْ تَسَرْ مبࣷولࣹمْ بࣺرَبْ، يَ خࣹيتْ يَ نگࣹينْ دِ دࣵقْ، دِ جَامࣷو سࣹينِ تُورْ؛ مُيْ ڪَوْ تُندّ يُ مَگْ يَ، دِ ڪَوْ تُندّ يُ ندَوْ يَ، دِ ڪࣹرْ گَرَبْ يُ نَاتْ يَ؞ ‏3 تࣷجْلࣹينْ سࣹينِ سَرْخَلُڪَايْ، رَجَخࣹ سࣹينْ تُورِ دࣷجْ؛ سࣹينْ خࣹرْ يِ ݧُيْ جَامࣷو اَسࣹرَ تُورْ مِ، نگࣹينْ لَڪّ؛ سࣹينْ جࣴمِّ تُورْ، نگࣹينْ گࣷرْ، بَ فَرْ تُڪَّلْ تُرِ تُورْ يِ ݖِ يࣷويُ بࣺرَبْ؞ ‏4 بُلࣹينْ جَامࣷو نࣷونُ سࣹينْ يࣵلَّ اَجِ سَخْ جِ نَگْ؞ ‏5 وَايࣹ سࣹيتْلࣹينْ بࣺرَبْ بُ سࣹينْ يࣵلَّ اَجِ سَخْ جِ دِ تَامُجِ، ݖِ مبࣷولࣹمْ سࣹينِ گِيرْ، نگِرْ تࣹگْ فَ تُرَمْ، دࣴعࣴلْ ڪࣷ فَ، تࣹ نگࣹينْ دࣹمْ فࣷوفَ؞ ‏6 فࣷوفَ نگࣹينْ دِ يٗبُّ سࣹينْ سَرَخِ دٗومَلْ، اَڪْ يࣹنࣹينْ سَرَخْ، اَڪْ سࣹينْ ݖࣺرِ فُڪّࣹيلْ، اَڪْ سࣹينُبْ جࣷوخࣹ، اَڪْ سࣹينْ سَرَخْ يُ نگࣹينْ خَسْ، اَڪْ سࣹينْ سَرَخِ يࣺينࣹ، اَڪْ سࣹينْ سَرَخِ تَاوُگْ جُرْ گُ گُدّ اَڪْ گُ گࣵتّ؞ ‏7 فࣷوفَ نگࣹينْ دِ لࣹڪّࣹ، يࣹينْ اَڪْ سࣹينِ نجَبࣷوتْ فَ سࣹينْ ڪَنَمْ يࣵلَّ اَجِ سَخْ جِ، نگࣹينْ بࣺگࣹ فَ سࣹينْ وࣺݒّ ݧَقْ وُ لࣹينْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ بَرْڪࣹيلَلْ؞
‏8 بُلࣹينْ جࣴفࣹ مبࣷولࣹمْ نِي نُيْ جࣴفࣹ فِي تࣹيْ، لُ نࣹيخْ وَايْ رࣹڪّ، دࣹفْ؞ ‏9 ندَخْ ڪَتْ اࣵگَّگُلࣹينْ بَ تࣹيْ ݖَ دَلْ-لُڪَايْ بَ، فَ لࣹينْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ سࣺدّࣹ؞ ‏10 وَايࣹ بُ نگࣹينْ جࣵلّࣹي دࣹخُ يُرْدَنْ، بَ دࣴڪِّ ݖَ رࣺيوْ مَ لࣹينْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ سࣺدّࣹ، بَ مُ نࣷݒَّلْ لࣹينْ ݖِ مبࣷولࣹمْ نࣷونْ يِ لࣹينْ سࣺقْ، نگࣹينْ دࣴڪّࣹ خࣹلْ مُ دَلْ، ‏11 سُ بࣷوبَا بࣺرَبْ بُ سࣹينْ يࣵلَّ اَجِ سَخْ جِ تَامُ نگِرْ دࣴعࣴلْ فَ تُرَمْ، فࣷوفَ نگࣹينْ دِ يٗبُّ مبࣷولࣹمْ لُ مَ لࣹينْ سَنتّ:‏ سࣹينْ سَرَخِ دٗومَلْ، اَڪْ يࣹنࣹينْ سَرَخْ، اَڪْ سࣹينْ ݖࣺرِ فُڪّࣹيلْ، اَڪْ سࣹينُبْ جࣷوخࣹ، اَڪْ مبࣷولࣹمْ نگࣴنࣹيلْ يُ نگࣹينْ دِ دࣹفْ سَرَخِ خَسْ يُ نگࣹينْ خَسࣷو اَڪْ اَجِ سَخْ جِ؞ ‏12 سُ ڪࣷ دࣹفࣹي نگࣹينْ بࣺگࣹ فِ سࣹينْ ڪَنَمْ يࣵلَّ اَجِ سَخْ جِ، يࣹينْ اَڪْ سࣹينْ دࣷومْ يُ گٗورْ اَڪْ يُ جِگࣺينْ، اَڪْ سࣹينْ جَامْ يُ گٗورْ اَڪْ يُ جِگࣺينْ، اَڪْ لࣹوࣹينْ ݧِ ݖِ سࣹينْ بِيرِ دࣴڪّ تࣹ اَمُݧُ وࣵلّ، اَمُݧُ ݖࣺرُ سُوفْ ݖِ سࣹينْ بِيرْ؞
‏13 وَتُّلࣹينْ سࣹينْ بࣷݒّ بَلَ نگࣹينَ جࣷخࣹ سࣹينْ سَرَخِ دٗومَلْ فࣺݒّ فُ نگࣹينْ گِسْ، ‏14 ندَخْ بࣺرَبْ بُ اَجِ سَخْ جِ تَامُ ݖِ گࣹنّ ݖِ سࣹينْ گِيرْ يِ، فࣷوفَ نگࣹينْ دِ جࣷخࣹ سࣹينْ سَرَخِ دٗومَلْ، تࣹ فࣷوفَ نگࣹينْ دِ دࣹفْ لࣺݒّ لُ مَ لࣹينْ سَنتّ؞ ‏15 تࣹيوُلْ سَا يُ لࣹينْ سࣷوبࣹي، سَݧْ نگࣹينَ رࣹنْدِ، لࣹڪّ ݖِ لࣺݒّ لُ لࣹينْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ بَرْڪࣹيلࣹ، اَڪْ فࣺݒّ فُ نگࣹينْ دࣴڪّ؞ ڪُ سࣷبࣹوُ اَڪْ ڪُ سࣹتْ يࣺݒّ سَݧْ نَ ݖَا لࣹڪّ، مࣹلْ نِ بُ دࣷونْ ڪࣺوࣺلْ مبَا ڪࣷوبَ؞ ‏16 وَايࣹ دࣹرࣹتْ جَ، بُلࣹينْ ڪࣷ لࣹڪّ؞ ݖِ سُوفْ نگࣹينْ ڪࣷيْ تُورْ نِمْ ندࣷخْ؞ ‏17 سࣹينْ ݖࣺرُ فُڪّࣹيلُ ݒࣹݒّ نَگْ، اَڪْ بُ بِيݧْ بُ بࣹيسْ، اَڪْ بُ دِوْ گُ بࣹيسْ، اَڪْ سࣹينْ تَاوُگْ جُرْ گُ گُدّ اَڪْ گُ گࣵتّ، اَڪْ سࣹينْ مبࣷولࣹمْ سَرَخِ خَسْ يُ نگࣹينْ خَسࣷو وࣷونْ، اَڪْ سࣹينْ سَرَخِ يࣺينࣹ، اَڪْ اَبْ جࣷوخࣹ بُ نگࣹينْ سࣵڪّࣹ ݖِ سࣹينُ ݧَقْ، سَݧُلࣹينْ ݖࣹي لࣹڪّ لࣹنّ ݖِ سࣹينْ بِيرِ دࣴڪّ، ‏18 ندَرࣹ نگࣹينْ لࣹڪّࣹ ڪࣷ فَ سࣹينْ ڪَنَمْ يࣵلَّ اَجِ سَخْ جِ؞ لࣹڪّࣹلࣹينْ ڪࣷ فَ سࣹينْ يࣵلَّ اَجِ سَخْ جِ دِ تَامُجِ، يࣹينْ اَڪْ سࣹينْ دࣷومْ يُ گٗورْ، اَڪْ يُ جِگࣺينْ، اَڪْ سࣹينْ جَامْ يُ گٗورْ اَڪْ يُ جِگࣺينْ، اَڪْ لࣹوࣹينْ ݧِ ݖِ سࣹينْ بِيرْ دࣴڪّ؞ سُ ڪࣷ دࣹفࣹي نگࣹينْ بࣺگࣹ سࣹينْ وࣺݒّ ݧَقْ، فِ سࣹينْ ڪَنَمْ يࣵلَّ اَجِ سَخْ جِ؞ ‏19 وَتُّلࣹينْ سࣹينْ بࣷݒّ نَگْ، نگِرْ بَݧَ سࣵگَّنࣹ لࣹوࣹينْ ݧِ، سࣹينْ گِيرُ دُندّ ݖَ سࣹينْ سُوفْ؞
‏20 بُ سࣹينْ يࣵلَّ اَجِ سَخْ جِ يࣷڪّࣹي سࣹينْ سُوفْ نࣷونࣹي مُ لࣹينْ ڪࣷ وَخࣹ وࣷونْ، تࣹ نگࣹينْ نَمَّ لࣹڪّ يࣵݒّ، سَا يُ لࣹينْ نࣹيخࣹي سَݧْ نگࣹينَ لࣹڪّ يࣵݒّ؞ ‏21 بُ فࣹڪّࣹي نࣹ فِ سࣹينْ يࣵلَّ اَجِ سَخْ جِ تَامࣷو دࣴعࣴلْ تُرَمْ دَفَ سࣷرࣹيڪْ يࣹينْ، مَنْ نگࣹينَ رࣹنْدِ ݖِ سࣹينْ جُرْ گُ گُدّ مبَا گُ گࣵتّ، گَ لࣹينْ اَجِ سَخْ جِ جࣷخْ، نَ مَ لࣹينْ ڪࣷ وَخࣹ وࣷونْ؞ سَݧْ نگࣹينْ ݖَا لࣹڪّࣹ سࣹينْ بِيرْ دࣴڪّ لُ لࣹينْ نࣹيخْ؞ ‏22 نِ ݧُيْ لࣹڪّࣹ يࣵݒُّ ڪࣺوࣺلْ مبَا ڪࣷوبَ رࣹڪّ نگࣹينْ ڪࣷيْ لࣹڪّࣹ؞ ڪُ سࣷبࣹوُ اَڪْ ڪُ سࣹتْ يࣺݒّ سَݧْ نَ ݖَا لࣹڪّ؞ ‏23 مࣷيُلࣹينْ بُ بَاخْ رࣹڪّ بَ بَݧَ لࣹڪّ دࣹرࣹتْ جِ، ندَخْ دࣹرࣹتْ مࣷويْ بَڪَّنْ تࣹ وَرُلࣹينْ بࣷولࣹ لࣹڪّ بَڪَّنُ جُرْ گِ، اَڪُ سُوخَمْ؞ ‏24 بُلࣹينْ لࣹڪّ دࣹرࣹتْ جِ؞ ݖِ سُوفْ نگࣹينْ ڪࣷيْ تُورْ نِمْ ندࣷخْ؞ ‏25 بُلࣹينْ ڪࣷ لࣹڪّ مُڪّ، سُ بࣷوبَا دِنگࣹينْ بَاخْلࣹ، يࣹينْ اَڪْ سࣹينْ دࣷومْ يِ لࣹينْ دِ وُوتُ؛ نگِرْ سُ بࣷوبَا يࣹينَ نگِ جࣴفࣹ لِ اَجِ سَخْ جِ رَفࣹتْلُ؞
‏26 وَايࣹ سࣹينْ يࣴفْ يُ ݧُ سࣹلَّلْ، اَڪْ سࣹينْ سَرَخْ يِ نگࣹينْ خَسࣷو، جࣴلْ-لࣹينْ ڪࣷ، يٗبُّ ݖَ بࣺرَبْ بَ اَجِ سَخْ جِ دِ تَامُجِ؞ ‏27 تࣹ نَنگࣹينْ سَرْخࣹ سࣹينْ سَرَخِ رࣹنْدِ-دٗومَلْ، يࣵݒّ وِ اَڪْ دࣹرࣹتْ جِ، ݖِ سࣹينْ ڪَوْ سَرْخَلُڪَايُ يࣵلَّ اَجِ سَخْ جِ؞ سࣹينْ دࣹرࣹتُ يࣹنࣹينْ سَرَخْ، تُورْلࣹينْ ڪࣷ ݖَ سࣹينْ ڪَوْ سَرْخَلُڪَايُ يࣵلَّ اَجِ سَخْ جِ، يࣵݒّ وِ نگࣹينْ لࣹڪّ ڪࣷ؞
‏28 سࣵمّࣷولࣹينَ دࣺگَّلْ مبࣷولࣹمْ ڪࣵدُّ يِي مَ لࣹينْ دِ دࣺنڪّ، نگِرْ نگࣹينْ بَاخْلࣹ بَ فࣵوّ، يࣹينْ اَڪْ سࣹينْ دࣷومْ، يِ لࣹينْ دِ وُوتُ، ندَخْ ڪࣷنْ يࣹينَ نگِ جࣴفࣹ لُ بَاخْ لِ سࣹينْ يࣵلَّ اَجِ سَخْ جِ رَفࣹتْلُ؞
‏29 بُ لࣹينْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ سࣵنڪَلࣹي خࣹيتْ يِ نگࣹينْ جࣴمْ، نگِرْ دࣵقْ لࣹينْ، گَنَّاوْ بُ نگࣹينْ لࣹينْ دࣵقࣹي بَ دࣴڪّࣹ سࣹينُمْ رࣺيوْ، ‏30 نَنگࣹينْ وَتُّ سࣹينْ بࣷݒّ، بَ گَنَّاوْ بُ خࣹيتْ يࣷويُ سࣵنڪࣷو ݖِ سࣹينْ ڪَنَمْ، دُنگࣹينْ تࣵبِّ ݖِگْ فِيرْ، دِ لࣹينْ رࣷيْ، نَانْ:‏ «اَنَ نُ خࣹيتْ يِي دِ جَامࣷو سࣹينِ تُورْ، نگِرْ نُنْ اِتْ نُ دࣹفْ نِ ݧࣷومْ؟» ‏31 بُلࣹينْ دࣹفࣹ نࣷونُ اَڪْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ، ندَخْ مبࣷولࣹمْ لُ سࣹيخْلُوُ لُ اَجِ سَخْ جِ بَݧْ، دࣹفَلْ نَݧُ ڪࣷ سࣹينِ تُورْ؞ سࣹينْ دࣷومْ يُ گٗورْ اَڪْ يُ جِگࣺينْ سَخْ، لَڪّ نَݧُ لࣹينْ، دࣹفَلْ لࣹينْ سࣹينِ تُورْ سَرَخْ؞
