لُ جࣴمْ ݖِ وَخْيُ، اَڪِ مَيْ ݖِ وࣵلُّ لࣵڪّ
14
‏1 خِنْتࣹوࣷولࣹينْ ݖࣷفࣹيلْ، تࣹ نگࣹينْ سࣵڪُّ مَيِ نࣷو گِ، رَوَتِنَ مَيُگْ وَخࣹ ڪࣵدُّگْ وَخْيُ، ‏2 ندَخْ ڪِيْ لࣵڪّ وࣹنࣹينْ لࣵڪّ، دُ نِتْ ݧِ لَيْ وَخَلْ، وَايࣹ يࣵلَّ لَيْ وَخَلْ، گَنَّاوْ ڪࣹنّ دࣺگُّلْ لِ مُيْ وَخْ؞ تࣹيوُلْ ݖِ نࣷو گِ لَيْ وَخࣹ اَيْ ڪُمْݒَ؞ ‏3 وَايࣹ ڪِيْ وَخࣹ ڪࣵدُّگْ وَخْيُ، نِتْ ݧِ لَيْ وَخَلْ، دِ فࣹدّلࣹي، دِ ݧَاخࣹ اَڪْ دِ نَابࣹ؞ ‏4 ڪِيْ لࣵڪّ يࣹنࣹينْ لࣵڪّ، بࣷݒَّمْ لَيْ فࣹدّلِ، وَايࣹ ڪِيْ وَخࣹ ڪࣵدُّگْ وَخْيُ، مبࣷولࣷومْ گࣴمْڪَتْ ݧِ لَيْ فࣹدّلِ سࣹينْ نگࣴمْ؞ ‏5 سُ مَ سَݧࣷونْ نگࣹينْ دِ لࣵڪّ يࣹينْ ݧࣺݒّ يࣹنࣹينْ لࣵڪّ، وَايࣹ نگࣹينْ دِ وَخࣹ ڪࣵدُّگْ وَخْيُ مࣷو مَ گࣴنَلْ؞ ڪِيْ وَخࣹ ڪࣵدُّگْ وَخْيُ مࣷو اࣴݒّ نجَرِݧْ ڪِيْ لࣵڪّ يࣹنࣹينْ لࣵڪّ، ندَرࣹ ݧُ تࣹڪِّ لِ مُيْ لࣵڪّ، بَ مبࣷولࣷومْ گࣴمْڪَتْ ݧِ جࣴلࣹ ݖَ لُ ݧُ فࣹدّلِڪࣷو؞ ‏6 لࣺيگِ بࣷڪّ يِ، سُ مَ لࣹينْ دِڪّࣹ وࣷونْ يࣹنࣹينْ لࣵڪّ يُ مَيْ لࣵڪّ، اَنَ لُ مَ لࣹينْ دِ جَرِݧْ؟ دَرَ، خَنَا مَ اࣵگّࣹ لࣹينْ ڪࣵدُّگْ ݒࣹيݧُ، مبَا خَمْ-خَمْ، مبَا وَخْيُ، مبَا دِگْلࣹ؞ ‏7 نࣷونُ لَ اِتَمْ ݖِ جُمْتُڪَايِ ݒࣷ يِ دُنْدُلْ تࣹيْ جِبَلْ ݖࣷووْ؛ مُيْ تࣷخࣷرࣷ مبَا خَلَمْ؞ سُ يࣷويُ جُمْتُڪَايْ جِبَلُلْ گَلَنْ يُ لࣹيرْ، اَنَ نࣹيسْ دِ خَمࣹ لِ تࣷخࣷرࣷ جِ، مبَا خَلَمْ گِيْ وَخْ؟ ‏8 اَگْ لِيتُ خَرࣹ اِتْ، سُ جِبَلُلْ وࣷوتࣹ بُ لࣹيرْ، اَنَ ڪُيْ وَاجَلْ اَبْ خَرࣹ؟ ‏9 نࣷونُ لَ ݖِ يࣹينْ اِتَمْ؞ سࣹينْ لࣵمِّݧْ، سُ نگࣹينْ ݖِ وَخُلْ ڪࣵدُّ گُ لࣹيرْ، اَنَ نࣹيسْ دِ خَمࣹ لِ نگࣹينْ دِ وَخْ؟ ݖِ جَوّ جِ رࣹڪّ نگࣹينْ دِ وَخْ!‏ ‏10 اَڪْ لِ لࣵڪِّ اࣵدِّنَ بَرࣹ تࣹ وُوتࣹ لࣺݒّ، وࣹنّ ݧࣵڪُّ ݖِ اَمْ ݒِرِ؞ ‏11 بَاتْ بُ جِبْ نَگْ، سُ مَ خَمُلْ تࣹڪِّتࣹيمْ، اَبْ جَامْبُرْ لاَيْ دࣷونْ ݖِ ڪِيْ وَخْ، تࣹ ڪِيْ وَخْ اِتْ، اَبْ جَامْبُرْ لَيْ دࣷونْ ݖِ مَنْ؞ ‏12 ڪࣷنْ يࣹينْ اِتَمْ، گَنَّاوْ ݧُ خࣺرْ نگࣹينْ ݖِ مَيِ نࣷو گِ، فࣹخࣹلࣹينْ ݖࣹي اَمْ لُ نࣹ خࣺيوْ، نگِرْ فࣹدّلِ مبࣷولࣷومْ گࣴمْڪَتْ ݧِ؞
‏13 مࣷو تَخْ ڪِيْ لࣵڪّ وࣹنࣹينْ لࣵڪّ وَرَ ݧَانْ نگِرْ ݧُ خَمْ ݒِرࣹيمْ؞ ‏14 سُ مَيْ ݧَانْ ݖِ وࣹنࣹينْ لࣵڪّ، سَمَ خࣷلْ اَيْ ݧَانْ، وَايࣹ سَمَمْ خࣹلْ لِگّࣺييُلْ؞ ‏15 ڪࣷنْ اَنَ نࣹيسْ دِ دࣹفْ؟ نَا ݧَانࣹ سَمَ خࣷلْ، تࣹيْ ݧَانࣹ سَمَمْ خࣹلْ اِتَمْ؞ نَا سࣵبَّالࣹ سَمَ خࣷلْ، تࣹيْ سࣵبَّالࣹ سَمَمْ خࣹلْ اِتَمْ؞ ‏16 سࣷو يࣺڪَّتࣹي ڪࣵدُّگْ ݖَنتّ گࣷو وَخࣹ خࣷلْ دࣷݝّ، اَنَ نُ نِتْ ڪُ خَمُلْ ڪُ ڪࣷ تࣹيوࣹ نَانْ:‏ «اَمِينْ،» تࣹ خَمُلْ لِ نگَيْ وَخْ؟ ‏17 ندَخَمْ يَوْ، رَفࣹتْ نگَ ڪࣵدُّگْ ݖَنتّ، وَايࣹ سَ ڪࣵدُّ فࣹدّلِوُلْ ڪࣹنࣹينْ ڪُ ݖِيْ دࣺگّ؞
‏18 سَنتّ نَا يࣵلَّ ندࣹگَمْ مَايْ لࣵڪّ يࣹنࣹينْ لࣵڪّ بَ رَوْ لࣹينْ، يࣹينْ ݧࣺݒّ؞ ‏19 وَايࣹ فُ مبࣷولࣷومْ گࣴمْڪَتْ ݧِ دَجࣹ، جُرٗومِ بَاتْ يُ مَ وَخࣹ سَمَمْ خࣹلْ نگِرْ جࣵنگّلࣹ، مࣷو مَ گࣴنَلْ جُنِّيْ بَاتْ يُ مَ وَخࣹ وࣹنࣹينْ لࣵڪّ؞
‏20 بࣷڪّ يِ، بُلࣹينْ دࣷونْ اَيْ خَلࣹ ݖِ وࣵلُّ دࣺگِّنْ؞ سُ دࣹي لُ بࣷنْ، نࣹڪّلࣹينْ ݖِ اَيْ تُوتْ-تࣵنڪّ، وَايࣹ مࣵگّلࣹينْ ݖِ وࣵلُّ دࣺگِّنْ؞ ‏21 بِنْدࣹيسْ نَ ݖِ تࣺيرࣹبْ يࣷونْ وِ نࣹ:‏
«اَيْ نِتِ لࣵڪّ-ڪَتْ،
اَيْ لࣵمِّݧِ دࣷخَنْدࣺيمْ،
ݖِ لاَيْ وَخࣹ اَڪْ اَسْڪَنْ وِي،
تࣹ تَخُلْ بَ تࣹيْ ݧُ دࣺگْلُ مَ؞
بࣷرࣷومْ بࣹي ڪࣷ وَخْ¤؞»
‏22 ڪࣷنْ نَگْ، يࣹنࣹينْ لࣵڪّ، ڪࣵدُّ يَ دُ ݖِ گࣴمْڪَتْ ݧِ لَ دِ فِرْندࣹ، وَايࣹ ݖِ ݧِ گࣴمُلْ لَ دِ فِرْندࣹ؞ تࣹ اِتْ وَخࣹ ڪࣵدُّگْ وَخْيُ، دُ ݖِ ݧِ گࣴمُلْ لَ دِ فِرْندࣹ، وَايࣹ ݖِ گࣴمْڪَتْ ݧِ لَ دِبْ فِرْندࣹ؞ ‏23 لࣺيگِ نَگْ سُ مبࣷولࣷومْ گࣴمْڪَتْ مࣺݒّ دَجࣹي بࣺرَبْ، ݧࣺݒّ دِ لࣵڪّ يࣹنࣹينْ لࣵڪّ، بُ اَيْ نِتْ ݧُ خَمُلْ مبَا ݧُ گࣴمُلْ دُگّسࣹي، دُݧُ نࣹ دَنگࣹينَ دࣷفْ؟ ‏24 وَايࣹ سُ ݧࣺݒّ دࣹي وَخࣹ ڪࣵدُّگْ وَخْيُ، بَ ڪُ گࣴمُلْ، مبَا ڪُ خَمُلْ دُگّسِ، ڪࣵدُّ گَ لَ ڪࣷ ݧࣺݒّ دِ تِيݧَلࣹ، اࣵگّࣹي ڪࣷ ڪࣷ اَبْ اࣵتّࣹيمْ، ‏25 بَ لَ لࣵقُ وࣷونْ ݖَ بِيرْ خࣷلَمْ، دَلْدِ لࣹيرْ؞ سُ ڪࣷ دࣹفࣹي مُ نࣹ دࣵلّ دࣴݒّ جࣴعࣴمْ ݖِ سُوفْ، سُجّٗوتَلْ يࣵلَّ، تࣹ نَانْ:‏ «اَيْلَيࣺوࣺنْ يࣵلاَّ نگِ ݖِ سࣹينْ بِيرْ!‏»
ندَجࣹمْ گࣴمْڪَتْ ݧِ جَرْ نَا رَفࣹتَلْ
‏26 ڪࣷنْ نَگْ اَنَ نࣹيسْ دِ دࣹفْ، بࣷڪّ يِ؟ بُ نگࣹينْ دَجࣹي، ڪِي اَمْ وࣷيُ ݖَنتّ، ڪِي اَمْ لُ مُيْ جࣵنگّلࣹ، ڪࣹي اَمْ اَمْ ݒࣹيݧُ، ڪَلࣹ اَمْ لُ مُيْ وَخࣹ وࣹنࣹينْ لࣵڪّ، ڪࣹنࣹينْ اَمْ اَمْ ݒِرِ، لࣷولُ لࣺݒّ نَ ڪࣷ فࣹدّلِ نگࣴمْ تَخَ جٗگْ؞ ‏27 سُ دࣹي وࣹنࣹينْ لࣵڪّ لࣹيسْ دِ وَخْ، نَ ݧَارْ بَ گࣴنْ گَا بَرࣹ، ݧࣹتّ، تࣷݒَّنْتࣹ، تࣹ ڪࣹنّ دِ فِرِ؞ ‏28 سُ اَمُلْ ڪُيْ فِرِ، بُ ڪِيْ لࣵڪّ لࣵڪّ وِ بِرَلْ لࣹنّ ݖِ جَتَايُ مبࣷولࣷو مِ؞ نَ يࣹلُ، نگِرْ بࣷݒَّمْ اَڪْ يࣵلَّ؞
‏29 ݧِيْ وَخࣹ ڪࣵدُّگْ وَخْيُ، نَ ݖِ ݧَارْ بَ ݧࣹتّ وَخْ، تࣹ ݧِ ݖِ دࣹسْ نَتّ وَخْ جِ؞ ‏30 سُ ڪࣹنࣹينْ ݖِ جَتَايْ بِ اَمࣹي اَمْ ݒࣹيݧُ، نَ ڪِ دࣷونْ وَخْ نࣷݒِّ؞ ‏31 ندَخْ ڪَتْ يࣹينْ ݧࣺݒّ مَنْ نگࣹينَ ڪࣹنّ-ڪࣹنّࣷو، دِ وَخࣹ ڪࣵدُّگْ وَخْيُ، نگِرْ ݧࣺݒّ جࣵنگࣹ ݖَ، بَ گࣴنَ فَرْلُ؞ ‏32 بِرَلْڪَتِ ڪࣵدُّگْ وَخْيُ يِ ݧࣷو وَرَ جِيتࣹ خࣹلْ مِ ݧُيْ وَخࣹ؞ ‏33 ندَخْ يࣵلَّ دُ يࣵلَّيْ سَلْفَاݧࣷو، يࣵلَّيْ جࣵمّ لَ؞
ݖِ مبࣷولࣹمْ مبࣷولࣷويْ ݧُ سࣹلّ ݧِ نَگْ، ‏34 نَ جِگࣺينْ ݧِ نࣷݒِّ ݖِ بِيرْ مبࣷولࣷو يِ، ندَخْ مَيࣹيسُ لࣹينْ ݧُ وَخْ ݖِ يࣷويُ جَتَايْ؞ نَ جِگࣺينْ ݧِ نَنگُ، نِ ڪࣷ يࣷونْ وِ ݖِ بࣷݒَّمْ دِگْلࣹي؞ ‏35 سُ اَمࣹي لُ ݧُ بࣴگَّ خَمْ، ݖَ ڪࣴرْ گَ لَݧُ ڪࣷيْ لاَجْ سࣹينْ جࣴڪّࣴرْ، ندَخْ جِگࣺينْ دِ وَخْ ݖِ بِيرْ مبࣷولࣷومْ گࣴمْڪَتْ ݧِ، گࣵݖّࣹ لَ؞
‏36 ندَخْ ݖِ سࣹينْ بِيرْ لَ ڪࣵدُّگْ يࣵلَّ بࣵيِّڪࣷو؟ اَمْ ݖِ يࣹينْ دࣷݝّ لَ اَگْسِ؟ ‏37 ڪُ ݖِ خَلاَتْ نࣹ مࣷومْ جࣷتّلِڪَتُبْ ڪࣵدُّگْ وَخْيُ لَ، مبَا بࣷرࣷومْ اَگْ مَيْ لَ ݖِ نࣷو گِ، وَرْ نَا خَمْ خࣺلّ نࣹ لِي مَ لࣹينْ بِندّ، سَنْتَانࣹبْ بࣷرࣷومْ بِ لَ؞ ‏38 ڪُ ݖِ خَمُلْ لࣷولُ نَگْ، ڪࣷوڪُ، خَمࣹيسُ ڪࣷ؞
‏39 ڪࣷنْ بࣷڪّ يِ، فَرْلُلࣹينْ ݖِ دِ وَخࣹ ڪࣵدُّگْ وَخْيُ، تࣹ بُلࣹينْ آيࣹ ڪُيْ لࣵڪّ يࣹنࣹينْ لࣵڪّ؞ ‏40 وَايࣹ نَ لࣺݒّ تࣹگِّنࣹ تࣹ اَوْ يࣷونْ؞
