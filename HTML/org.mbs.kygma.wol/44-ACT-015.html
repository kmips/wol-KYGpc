ݒࣺنݖࣷو اَمْ نَ ݖَ يࣹرُسَلࣹمْ
15
‏1 ݧࣹنّ نِتْ ݧُ بࣵيِّڪࣷو يُدࣹ دِڪّ اࣵنݖࣷسْ، ݧࣷو دࣷونْ دِگَلْ بࣷڪِّ تَالِبࣹ يِ، نࣹ لࣹينْ:‏ «سُ نگࣹينْ خَرَفُلْ¤، نِ ڪࣷ آدَيْ مُوسَىٰ سَنْتَانࣹي، دُنگࣹينْ مَنَ مُݖّ؞»
‏2 مبِرْ مَ نَگْ مُجّࣹ فࣺيوَلࣷو اَڪْ وࣹرَنْتࣹ وُ خَوَ مࣹتِّ، تࣹ ݒٗولْ اَڪْ بَرْنَبَ سࣺقْ ڪࣷ اَڪْ ݧࣷوݧُ؞ بَ لࣷولُ اَمࣹي ݧُ نࣹ، نَ ݒٗولْ اَڪْ بَرْنَبَ، اَڪْ ݧࣹنّ ݖَ گࣴمْڪَتِ اࣵنݖࣷسْ، اࣵندّ دࣹمْ يࣹرُسَلࣹمْ ݖَ ندَوِ عِيسَىٰ، اَڪْ مَگْ ݧَ، بَ لِجَّنْتِ مبِرْ مࣷومُ؞
‏3 ݖِ ڪَوْ لࣷولُ مبࣷولࣷومْ گࣴمْڪَتْ ݧِ يࣹبَلْ لࣹينْ، ݧُ جَارࣹ فࣹنِسِ اَڪْ سَمَرِ، دَلْدِ اࣵگّࣹ وَا فࣷوفَ نِ جَامْبُرْ ݧِ دُلْ يَوُتْ تُوبࣹ، بَ بࣷڪِّ گࣴمْڪَتْ ݧࣺݒّ بࣺگْ ݖَ لࣷولْ؞ ‏4 گَنَّاوْ لࣷولُ، ݧُ اَگْسِ يࣹرُسَلࣹمْ، مبࣷولࣷومْ گࣴمْڪَتْ ݧِ، اَڪْ ندَوِ عِيسَىٰ يِ، اَڪْ مَگْ ݧِ، دَلَلْ لࣹينْ، ݧُ نࣹتَّلِ لࣹينْ مبࣷولࣹمْ نِ يࣵلَّ دࣹفْ اَڪْ ݧࣷومْ؞
‏5 بَ مُ ڪࣷ دࣹفࣹي ݧࣹنّ ݖِ گࣴمْڪَتْ ݧِ بࣷڪّ ݖِ نگࣺرُمْ فَرِسࣹنْ يِ، دَلْدِ تَخَوْ تࣹمّ، نࣹ گࣴمْڪَتْ ݧِ دُلْ يَوُتْ، فࣵوّ ݧُ خَرْفَلْ لࣹينْ، سࣵمّلࣷو لࣹينْ يࣷونُ مُوسَىٰ؞
‏6 بَ لࣷولُ اَمࣹي ندَوِ عِيسَىٰ يِ اَڪْ مَگْ ݧِ بࣷڪّ دَجࣹ، نگِرْ سࣹيتْ مبِرْ مࣷومُ؞ ‏7 وࣹرَنْتࣹ وُ رࣺيْ اَمْ ݖَ؞ ݖِ ڪَوْ لࣷولُ ݒِيࣹيرْ جٗگْ نࣹ لࣹينْ:‏ «بࣷڪّ يِ، يࣹينْ خَمْ نگࣹينْ نࣹ بُ يࣵگّ لَ مَ يࣵلَّ سࣹݒّࣹي ݖِ سࣹينْ بِيرْ، نگِرْ جَامْبُرْ ݧِ دࣺگّࣹ ݖِ سَمَ لࣵمِّݧْ، ڪࣵدُّگْ خِبَارُ جࣵمّ بِ، بَ دࣷونْ اَيْ گࣴمْڪَتْ؞ ‏8 يࣵلَّ، مِ خَمْ خࣷلُ نِتْ نَگْ مࣷو سࣹيرࣹيلْ جَامْبُرْ ݧِ، بَ مُ لࣹينْ مَيࣹي نࣷو گُ سࣹلّ گِ، نِ مُ نُ ڪࣷ مَيࣹ نُنْ اِتَمْ؞ ‏9 يࣵلَّ اَمُلْ خࣹجّ اَڪْ سࣹينْ سُنُ دِگّ اَڪْ ݧࣷومْ، ندَخْ سࣹينْ نگࣴمْ لَ رَخَسࣹ سࣹينْ خࣷلْ؞ ‏10 لࣺيگِ نَگْ لُ تَخْ نگࣹينْ دِ دِيݝَتْ يࣵلَّ، دِ تࣹگْ تَالِبࣹ يِ يࣹنْ بُ نُ مَسُلَ اࣵتَّنْ، ݖَيْ مَامْ بَ ݖِ نُنْ؟ ‏11 يِوُ سَنگّ بِ عِيسَىٰ دَالْ لَنُ گࣴمْ نࣹ ݖِ لَنُيْ مُݖّࣹ، تࣹ ݧࣷومْ اِتَمْ، نࣷونُ رࣹڪّ؞»
‏12 بَ لࣷولُ وࣺيࣹي مبࣷولࣷو مࣺݒّ نࣹ خࣹرࣹمْ؞ ݧُ دࣺگْلُ نَگْ بَرْنَبَ اَڪْ ݒٗولْ، ݧُ خَمْلࣹ مبࣷولࣹمْ فِرْندࣹ يࣹيڪْ ڪࣺيمَانْ، يِ يࣵلَّ دࣹفࣹ سࣹينْ ݧَارِ جࣴمّ ݖِ بِيرْ جَامْبُرْ ݧِ؞ ‏13 بَ ݧُ نࣷݒّࣹي، يَعْقُوبَ اࣵدُّ نࣹ:‏ «بࣷڪّ يِ، دࣺگْلُلࣹينْ مَ؞ ‏14 سِمࣷݝْ¤ خَمْلࣹ نَ، نِ يࣵلَّ يࣺگࣹ جَامْبُرْ ݧِ ݖَ نجࣵلْبࣺينْ، بَ سࣵڪّࣹ ݖِ سࣹينْ بِيرْ، خࣹيتْ وُ مُ تُدّࣹ تُرَمْ؞ ‏15 تࣹ لࣷولُ دࣴݒّࣷو نَ اَڪْ ڪࣵدُّيْ يࣷنࣹنتّ يِ، ندَخْ بِنْدࣹيسْ نَ نࣹ:‏
‏16 ‹بࣷرࣷومْ بِ نࣹي:‏
گَنَّاوْ لࣷولُ دِنَا دࣺلْسِ؛
مَايْ يࣺڪَّتِ ڪࣴرْ دَاوُدَ گِ دَانُ،
مَايْ يࣺڪَّتِ گࣹنتّ يِ، جࣷيَنْتِ ڪࣷ،
‏17 نگِرْ ندࣹسُ نِتْ ݧِ سࣵڪُّ بࣷرࣷومْ بِ،
تࣹ ݧࣷوݧࣷو دِ جَامْبُرْ ݧِ ݧُ تُدّࣹ سَمَ تُرْ؞
‏18 بࣷرࣷومْ بࣹي خَمْلࣹ لࣷولُ بُ يࣵگَّ يࣵگّ؞›
‏19 «ڪࣷنْ نَگْ سَمَ خَلاَتْ مࣷويْ:‏ بُنُ گࣺتࣴنْ جَامْبُرْ ݧِ تُوبْ، بَ وَݧِّڪُ ݖِ يࣵلَّ؞ ‏20 خَنَا نُ بِندّ لࣹينْ، نࣹ لࣹينْ ݧُ مࣷيْتُ لُ اَيْ خࣴرࣴمْ سࣷبࣹيلْ، مࣷيْتُ ݒࣷوُمْ سࣺيْ، مࣷيْتࣷو لࣹڪّ يࣵݒُّ مَلَ مُ ݧُ تُورُلْ دࣹرࣹتَمْ، اَڪْ لࣹڪّ دࣹرࣹتْ ݖِ بࣷݒَّمْ؞ ‏21 ندَخْتࣹ ݖَ نجࣵلْبࣺينْ گَ بَ تࣹيْ اَيْ نِتَ نگِ وَارࣹي يࣷونُ مُوسَىٰ ݖِ دࣴڪّ بُ نࣹڪّ، ݧُ دِ ڪࣷ جࣵنگّ ݖِ جࣵنگُ يِ بࣺسُبْ نࣷفْلاَيْ بُ نࣹڪّ؞»
ندَجࣹ مَ بِندّ نَ ݧِ دُلْ يَوُتْ
‏22 بَ لࣷولُ اَمࣹي ندَوْ يِ اَڪْ مَگْ ݧِ مࣵنڪࣷو اَڪْ مبࣷولࣷومْ گࣴمْڪَتْ ݧࣺݒّ، نگِرْ تࣵنّࣹ ݖِ سࣹينْ بِيرْ اَيْ نِتْ، يࣹبَلْ لࣹينْ اࣵنݖࣷسْ، بࣷولࣹ لࣹينْ اَڪْ ݒٗولْ اَڪْ بَرْنَبَ؞ ݧࣷوݧࣷو دِ يُدَ، مِ ݧُ دِݒّࣹي بَرْسَبَسْ، اَڪْ سِلَسْ، ݧُ دِيْ مَگْ ݖَ بࣷڪّ يَ؞
‏23 ݧُ يٗبَّنْتࣹ لࣹينْ بَتَاخَلْ، بِندّ ݖِ نࣹ:‏
 
نُنْ ندَوِ عِيسَىٰ، اَڪْ مَگْ ݧِ، سࣹينْ بࣷڪِّ يࣹرُسَلࣹمْ، نࣷو لࣹينْ بِندّ، يࣹينْ سُنُ بࣷڪِّ گࣴمْڪَتْ ݧِ دُلْ يَوُتْ تࣹ دࣴڪّࣹ اࣵنݖࣷسْ اَڪْ سِرِ اَڪْ سِلِسِ؞ نُ نگِ جِيتَلْ سُنُبْ نُيࣷو؞ ‏24 دࣺگّ نَنُ نࣹ اَيْ نِتْ ݧُ بَوࣷو ݖِ نُنْ، ݧࣷو لࣹينْ لࣴجَلْ اَڪْ سࣹينِ وَخْ، دِ جَخَسࣹ سࣹينْ خࣹلْ، تࣹ نُنْ جࣷخُنُ لࣹينْ لࣹنّ ندِگَلْ؞ ‏25 مࣷو تَخْ نُ مࣵنڪࣷو ݖِ تࣵنّ اَيْ نِتْ ݧُ نُ يࣹبَلْ ݖِ يࣹينْ، بࣷولࣹ لࣹينْ اَڪْ سُنُيْ سࣷݒّࣹ بَرْنَبَ اَڪْ ݒٗولْ، ‏26 ݧࣷومْ ݧِ جَايْ سࣹينْ بَڪَّنْ نگِرْ سُنُ سَنگّ عِيسَىٰ اَلْمَسِ؞ ‏27 ڪࣷنْ نَگْ نࣷو يࣹبَلْ يُدَ اَڪْ سِلَسْ، نگِرْ ݧُ اࣵگّࣹ لࣹينْ ݖِ سࣹينْ گࣺمِّݧُ بࣷݒّ، لِي نُ لࣹينْ بِندّ؞ ‏28 ندَخْتࣹ نࣷو گُ سࣹلّ گِ مࣷو مࣵنڪࣷو اَڪْ نُنْ ݖِ بَݧْ لࣹينَ تࣹگْ بࣹنّ ݖࣷونࣷ، خَنَا لِي مَنُلَ ݧࣵڪّ:‏ ‏29 نگࣹينْ مࣷيْتُ يࣵݒّ وُ ݧُ سَرْخَلْ اَيْ تُورْ، اَڪْ دࣹرࣹتْ، اَڪْ يࣵݒُّ جُرْ گُ ݧُ تُورُلْ دࣹرࣹتَمْ، اَڪْ ݒࣷوُمْ سࣺيْ؞ يࣷويِي، بُ نگࣹينْ ڪࣷ مࣷيْتࣷو بُ بَاخْ، دࣹفْ نگࣹينْ لُ بَاخْ؞ جࣵمّ اَڪْ جࣵمّ؞
 
‏30 بَ لࣷولُ اَمࣹي ݧُ تࣵگّتࣷو، دَلْدِ دࣹمْ بَ اࣵنݖࣷسْ؞ ݖِ ڪَوْ لࣷولُ ݧُ وࣷو مبࣷولࣷو مِ، جࣺبَّلْ لࣹينْ بَتَاخَلْ بِ؞ ‏31 بَ ݧُ جࣵنگࣹي بَتَاخَلْ بِ، ڪࣵدُّ گَ دࣴفَلْ لࣹينْ، سࣹينْ خࣷلْ سࣹدّ؞ ‏32 يُدَ اَڪْ سِلَسْ اِتْ گَنَّاوْ اَيْ يࣷنࣹنتّ لَݧُ وࣷونْ، وَخْ نَݧُ اَڪْ بࣷڪّ يِ لُ يࣵگّ، نگِرْ ݧَاخْ لࣹينْ، فࣹدّلِ سࣹينْ نگࣴمْ؞ ‏33‏-34 بَ لࣷولُ اَمࣹي ݧُ تࣷوگْ فَ اَبْ دِيرْ، بࣷڪّ يِ دࣷورْ لࣹينَ ݧَانَلْ يࣷونُ جࣵمّ، يِوِ لࣹينْ، نگِرْ ݧُ دࣹلُّ ݖَ ݧَ لࣹينْ يࣹبَلࣷونْ؞ ‏35 ݒٗولْ اَڪْ بَرْنَبَ نَگْ دࣹسْ اࣵنݖࣷسْ؞ ݧُ اࣵندّ اَڪْ ݧࣹنࣹينْ ݧُ بَرࣹ، دِ جࣵنگّلࣹ اَڪَ خَمْلࣹ ڪࣵدُّگْ بࣷرࣷومْ بِ؞
ݒٗولْ اَڪْ سِلَسْ اࣵندّ نَݧُ
‏36 بَ ݧُ ݖَ تࣹگࣹي اَيْ فَنْ ݒٗولْ نࣹ بَرْنَبَ:‏ «نَنْ دࣹلُّ سࣹيتِ بࣷڪّ يِ ݖِ مبࣷولࣹمْ دࣴڪّ يِ نُ يࣺگْلࣹ وࣷونْ ڪࣵدُّگْ بࣷرࣷومْ بِ، بَ گِسْ نُ ݧُ دࣹفْ؞» ‏37 بَرْنَبَ نَگْ بࣴگّࣷونْ ݧُ يٗبَّالࣹ يࣷوَانْ، مِ ݧُيْ وَخْ مࣵرڪّ؞ ‏38 تࣹيوُلْ ݒٗولْ جࣵݒّ نࣹ گَنَّاوْ يࣷوَانْ مࣵرڪّ مࣷومُ مࣷو لࣹينْ وَݖّࣷونْ ݖَ ݒَمْفِلِ تࣹ اࣵنْدُلْ وࣷونْ اَڪْ ݧࣷومْ ݖِ لِگّࣺييْ بِ، يٗبَّالࣹ ڪࣷ وَرُلْ؞
‏39 مبِرْ مِ نَگْ مُجّࣹ دࣴنگࣷو بُ مࣹتِّ، بَ ݧُ تࣹقْلِڪࣷو؞ بَرْنَبَ اࣵندّ اَڪْ يࣷوَانْ مࣵرڪّ، دُگّ گَالْ، جࣴمْ سِݒَّرْ؞ ‏40 ݒٗولْ مࣷومْ، تَامُ سِلَسْ، دَلْدِ اࣵندّ اَڪْ مࣷومْ دࣹمْ، گَنَّاوْ بَ لࣹينْ بࣷڪّ يَ دࣺنڪࣹي بࣷرࣷومْ بِ ݖِوْ يِوَمْ؞ ‏41 ݖِ ڪَوْ لࣷولُ مُ جَارࣹ دِيوَانِ سِرِ اَڪْ سِلِسِ، دِ ݧَاخْ مبࣷولࣷويْ گࣴمْڪَتْ ݧَ؞
