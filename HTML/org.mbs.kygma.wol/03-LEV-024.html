لُ جࣴمْ ݖِ تࣹگُڪَايُ لࣵمݒّ بِ اَڪْ مبُرُ يِ
24
‏1 بَ لࣷولُ اَمࣹي اَجِ سَخْ جِ وَخْ نَ مُوسَىٰ نࣹ ڪࣷ:‏ ‏2 «سَنْتَلْ بࣵنِّ إِسْرَاءِيلْ، ݧُ اِنْدِلْ لَ دِوُ سࣹگَلُ اࣷلِوْ گُ سࣹتْ، نگِرْ لࣹيرْلُ، ݧُ دِ ڪࣷ تَالࣹ لࣵمݒّ يِ گُدِّ گُ نࣹڪّ؞ ‏3 ݖِ ڪَنَمْ رِدࣷ بِ لࣵقْ گَالُ اࣵلُّوَيْ سࣹيدࣹ سِ، ݖِ بِيرْ خَيْمَبْ ندَجࣹ مِ لَ هَـٰرُونَ دِ تࣹگْ لࣵمݒّ يِ، مُيْ فَنَانࣹي تࣵڪّ بَ بࣴتْ سࣹتْ، فِ ڪَنَمْ اَجِ سَخْ جِ؞ دࣷگَلَ نگࣷوگُ، دِ لُ لࣹينْ وَرْ فࣵوّ، يࣹينْ اَڪْ سࣹينْ اَسْڪَنْ؞ ‏4 نَ تࣹگْ لࣵمݒّ يِ ݖِ ڪَوْ تࣹگُڪَايُ لࣵمݒّ بَ ݧُيْ تࣴگّࣹ وُرُسُ نگَلَمْ، مُيْ فَنَانࣹي تࣵڪّ فِ ڪَنَمْ اَجِ سَخْ جِ؞
‏5 «سࣵڪَّلْ سُنگُفْ سُ مُݖّ اَيِبْ، لَڪّ ݖِ فُڪِّ مبُرُ اَڪْ ݧَارْ، مبُرُ مُ نࣹڪّ دِ جُرٗومْ بࣹنِّ ڪِلࣷيْ سُنگُفْ؞ ‏6 تࣹگَلْ مبُرُ يِ ݖِ ڪَوْ تَابَلُ وُرُسُ نگَلَمْ بِ، فِ ڪَنَمْ اَجِ سَخْ جِ، تࣹ دࣹفْ ڪࣷ ݧَارِ جَلْ، جَلْ بُ نࣹ جُرٗومْ بࣹنِّ مبُرُ؞ ‏7 جَلْ بُ نࣹڪّ تࣹگْ ݖَ ݖُورَايْ لُ رَخُلْ، مُ وُوتُ مبُرُ مِ، دِ سَرَخُ بَاخَنْتَلْ، تࣹ دِ سَرَخُ سَوَرَ، ݧࣹيلْ اَجِ سَخْ جِ؞ ‏8 بࣺسُبْ نࣷفْلاَيْ بُ نࣹڪّ نَݧُ سَخࣷو تَاجْ مبُرُ يࣷويُ فِ ڪَنَمْ اَجِ سَخْ جِ؞ لࣷولُ سَسْ لَ بُ بࣵنِّ إِسْرَاءِيلْ دِ تࣹگࣷو فࣵوّ؞ ‏9 مبُرُ يࣷويُ هَـٰرُونَ اَڪِ دࣷومَمْ يُ گٗورْ ݧࣷو ڪࣷيْ فࣺيتࣹوࣷو، تࣹ نَݧُ ڪࣷ لࣹڪّࣹ فُ سࣹلّ، ندَخْتࣹ لُ سࣹلَّ سࣹلّ لَ، دِ لُ ݧُ سࣺدّࣷو فࣵوّ ݖِ سَرَخِ سَوَرَ يِ ݧࣹيلْ اَجِ سَخْ جِ؞»
لِ وَرْ ݖِ يࣹنّ خࣹيتِ تࣷوݧَانگࣹ
‏10‏-11 بَ لࣷولُ وࣺيࣹي اَمࣷونْ نَ ݖِ بِيرْ بࣵنِّ إِسْرَاءِيلْ جࣹنّ وَايْ جُ يَايَمْ دِ دࣷومُ إِسْرَاءِيلْ تࣹ تُدّࣷونْ سࣹلࣷمِتْ دࣷومُ دِبّرِ، دَنࣹينْ بَ؞ بَايَمْ دِ وَا مِسْرَ؞ مُ اَمْ بࣺسْ خࣹيخْ جِبْ ݖِ بِيرْ دَلْ بِ، ݖِ دِگَّنْتࣹ وَا جࣷوجُ اَڪْ جࣹنّ وَايِ إِسْرَاءِيلْ جُ رَخُلْ؞ ݖِ بِيرْ لࣷولُ وَا جِ يَايَمْ دِ دࣷومُ إِسْرَاءِيلْ وَخْ يࣵلَّ لُ ݧَاوْ، تُدّ ڪࣷ جَارْ فُ بࣷنْ؞ ݧُ يٗبُّ ڪࣷ ݖَ مُوسَىٰ؞ ‏12 ݧُ تࣴجْ وَا جَ، بَ خَمْ لُ اَجِ سَخْ جِ نَمّ ݖِ مࣷومْ؞
‏13 ݖِ ڪَوْ لࣷولُ اَجِ سَخْ جِ وَخْ مُوسَىٰ نࣹ ڪࣷ:‏ ‏14 «گࣺنّࣹيلْ دَلْ بِ ڪِ مَيْ وَخْ لُ ݧَاوْ، مَنْ يࣵلَّ، تࣹ نَ مبࣷولࣹمْ ڪُ ݖِ دࣺگّ تࣹگْ لࣷخࣷومْ ݖِ ڪَوْ بࣷݒَّمْ، بَ نࣷݒِّ مبࣷولࣷو مࣺݒّ دٗورْ ڪࣷيْ دࣷجْ، بَ مُ دࣹي؞ ‏15 بࣵنِّ إِسْرَاءِيلْ نَگْ نَنگࣹينْ لࣹينْ وَخْ نࣹ لࣹينْ:‏ ڪࣺݒّ ڪُ وَخْ يࣵلاَّمْ لُ ݧَاوْ، جَارْ فُ بࣷنْ، نَ وࣺيتࣷو اَݧَمْ؞ ‏16 ڪُ ݧࣵڪّࣹ ڪࣹرْسَ تُرُ اَجِ سَخْ جِ، دࣹي رࣹڪّ مࣷويْ اࣵتّࣹيمْ؞ مبࣷولࣷو مِ مࣺݒَّ ڪࣷ وَرَ دٗورْ اَيْ دࣷجْ، بَ مُ دࣹي؞ مُيْ دࣷخَنْدࣺيمْ مُيْ نجُدُّ-جِ-رࣺيوْ، ڪُ ݖِ ݧࣵڪّࣹ ڪࣹرْسَ تُرُ اَجِ سَخْ جِ، دࣹي مࣷويْ اࣵتّࣹيمْ؞
‏17 «بُ لࣷولُ وࣹيسࣹي ڪࣺݒّ ڪُ فَاتْ بَڪَّنُ نِتْ، دࣹي رࣹڪّ مࣷويْ اࣵتّࣹيمْ؞ ‏18 ڪُ فَاتْ بَڪَّنُ مَلَ، نَ فࣹيْ بࣷرࣷومْ؞ بَڪَّنْ، بَڪَّنَ ڪࣷيْ فࣹيْ؞ ‏19 ڪࣺݒّ ڪُ گَاݧْ مࣷرࣷومَمْ، لَ مُ ڪࣷ دࣹفْ رࣹڪّ لَݧُ ڪࣷيْ دࣹفْ؞ ‏20 دَمّ-دَمّ، دَمّ-دَمَّ ڪࣷيْ فࣹيْ؛ بࣴتْ، بࣴتَ ڪࣷيْ فࣹيْ؛ بࣴݧْ اِتْ بࣴݧَ ڪࣷيْ فࣹيْ؞ نِ مُ گَاݧࣹ مࣷرࣷومَمْ رࣹڪّ، نِ لَݧُ ڪࣷيْ گَاݧࣹ؞
‏21 «ڪُ رࣹيْ اَگْ جُرْ نَ فࣹيْ بࣷرࣷومْ، وَايࣹ ڪُ رࣹيْ نِتْ دࣹي مࣷويْ اࣵتّࣹيمْ؞ ‏22 نَ اࣵتّࣹ بِ دِ بࣹنّ ݖِ دࣷخَنْدࣺيمْ اَڪْ نجُدُّ-جِ-رࣺيوْ، ندَخْ مَنْ اَجِ سَخْ جِ مَايْ سࣹينْ يࣵلَّ؞»
‏23 ݖِ ڪَوْ لࣷولُ مُوسَىٰ وَخْ ڪࣷ بࣵنِّ إِسْرَاءِيلْ، ݧُ گࣺنّࣹ دَلْ بَ وَا جَ ݧࣵڪّࣹ يࣵلَّ ڪࣹرْسَ، دٗورْ ڪࣷ اَيْ دࣷجْ، بَ مُ دࣹي؞ ڪࣷنْ لِ بࣵنِّ إِسْرَاءِيلْ دࣹفْ مࣷويْ لِ اَجِ سَخْ جِ سَنْتࣷونْ مُوسَىٰ رࣹڪّ؞
