يࣵلَّ مࣷويْ اࣵتّࣹ
2
‏1 يَوْ نِتْ ڪِيْ سِڪّࣹ نَگْ، اَمࣷو اَوْ لَيْ، ندَخْ بࣷو سِڪّࣹي ڪࣹنࣹينْ، سَ بࣷݒّ نگَ سِڪّ، ندࣹ يَوْ مِيْ سِڪّࣹ، جࣷوجُ جࣴفِنْ نگَيْ جࣴفࣹ؞ ‏2 وَايࣹ وٗورْ نَنُ نࣹ اࣵتّࣹبْ يࣵلَّ بِ مُيْ اَݧَلࣹ ݧِيْ جࣴفࣹ جࣷوجُ جࣴفِنْ، دࣴگّ لَيْ دࣴݒّࣷولْ؞ ‏3 يَوْ نِتْ ڪِيْ سِڪّ ݧِيْ جࣴفࣹ يࣷويِي، تࣹ دِ ڪࣷ دࣹفْ، ندَخْ دࣹفࣹ نگَ نࣹ يَوْ ݖِ سَ بࣷݒّ دِنگَ رࣴݖّ اࣵتّࣹبْ يࣵلَّ؟ ‏4 اَمْ دَنگَيْ سࣷفࣹنْتَلْ مبَاخُ يࣵلَّ گُ رࣺيْ، اَڪُگْ جࣺلَّلࣹيمْ، اَڪُگْ مُݧَمْ، ندَخْ ݧࣵڪَّ خَمْ نࣹ لِ يࣵلَّ نَمّ ݖِ بَاخࣹ لَ، مࣷو دِ نگَ تُوبْ؟
‏5 وَايࣹ سَگْ دࣴگࣴرْ بࣷݒّ، اَڪْ سَ خࣷلْ بُ تُوبَدِ، ݖِ نگَيْ جٗورࣹ سࣵنجُمْ يࣵلَّ ݖِ سَ ڪَوْ بࣷݒّ، ڪࣹرࣷوگْ بࣺسُبْ سࣵنجّ مَ، فَ اࣵتّࣹبْ يࣵلَّ بُ جُبْ دِ فࣹيݧࣹ؞ ‏6 ݖَ لَيْ فࣹيْ ڪُ نࣹڪّ اَيْ جࣴفَمْ؞ ‏7 ݧِ سَخࣷو دِ جࣺيمَ دࣹفْ جࣴفْ جُ بَاخْ، نگِرْ دِ ڪࣷ سࣵڪّࣷو ندَمْ، اَڪْ تࣹرَانگَ، اَڪْ تࣹخࣹ گُ سَخْ، ݧࣷوݧُ، تࣹخࣹ گُ سَخْ دࣵڪّ مࣷويْ دࣷونْ سࣹينُبْ اَݧْ؞ ‏8 وَايࣹ ݧِ فࣺتّࣹيرَلْ ݖِ سࣹينْ سࣵڪُّتࣹيفِ بࣷݒّ، نَنگُوُݧࣷو تࣷݒّ لِيْ دࣴگّ، خَنَا اَگْ نجُبَدِ نَگْ، ݧࣷوݧُ، اَمْ سࣵنجّ اَڪْ مࣹرْ مُ نࣹ جِݒّࣺتْ لَݧُيْ يࣷولࣷو؞ ‏9 نجࣵقَرࣹ اَڪْ مُسِبَ دَالْ، مࣷويْ دَلْ ݖِ ڪَوْ ڪࣺݒّ ڪُيْ دࣹفْ لُ بࣷنْ، تࣹ يَوُتْ بِ جِيتُ ݖِ، وَايࣹ ڪِ دُلْ يَوُتْ اِتْ تࣷݒّ ݖِ؛ ‏10 دَرَجَ، اَڪْ تࣹرَانگَ، اَڪْ جࣵمّ نَگْ مࣷو ݧࣹيلْ ڪࣺݒّ ڪُيْ دࣹفْ لُ بَاخْ، تࣹ يَوُتْ بِ جِيتُ ݖِ، وَايࣹ ڪِ دُلْ يَوُتْ اِتْ تࣷݒّ ݖِ؞ ‏11 ندَخْ ڪَتْ خࣹجّ اَڪْ سࣹينْ اَمُلْ فَ يࣵلَّ؞
‏12 مبࣷولࣹمْ ݧِ بࣵڪَّارْ نَگْ تࣹ فࣹڪُّ لࣹينْ ݖِ يࣷونُ مُوسَىٰ، يࣷونُ مُوسَىٰ تࣷݒُّ لࣹينْ، وَايࣹ دُ لࣹينْ تࣹرࣹي سࣵنڪُ؞ مبࣷولࣹمْ ݧِ بࣵڪَّارْ اِتْ، تࣹ مُ فࣹڪّ لࣹينْ ݖِ يࣷونُ مُوسَىٰ، ݖِ يࣷونُ مُوسَىٰ لࣹيسْ لࣹينْ دِ دَانࣹ؞ ‏13 ندَخْ ڪَتْ دُ دࣺگْلُڪَتِ يࣷونُ مُوسَىٰ لࣹيسْ دِ جࣷخࣹي اࣵتّࣹبْ ݧُ جُبْ فَ ڪَنَمْ يࣵلَّ، وَايࣹ جࣴفْڪَتِ يࣷونُ مُوسَىٰ لࣹيسْ دِ جࣷخࣹ نِتْ اࣵتّࣹبْ ڪُ جُبْ؞ ‏14 جَامْبُرْ ݧِ دُلْ يَوُتْ، تࣹ اَمُݧُ يࣷونُ مُوسَىٰ، ندࣹگَمْ تࣹيوُ لࣹينَ تࣷݒّ بِنْدُبْ جُدُّ، بَيْ جࣴفࣹ لِ يࣷونُ مُوسَىٰ لاَجْ، دَفَ فࣹڪّ ݧࣷومْ ݖِ سࣹينْ بࣷݒّ، ݧُيْ سࣹينْ يࣷونُ بࣷݒّ، دࣷونْتࣹ اَمُݧُ يࣷونُ مُوسَىٰ؞ ‏15 نࣷونُ لَݧُ وࣺرَلࣹ نࣹ جࣴفْ جِ يࣷونُ مُوسَىٰ سَنْتَانࣹ مࣷو بِنْدࣹ ݖِ سࣹينُبْ خࣷلْ، بَ سࣹينْ يࣺگْ-يࣺگُ خࣷلْ سࣹيدࣹ لࣷولُ؞ لِ ڪࣷيْ فِرْندࣹيلْ اِتْ دِ سࣹينْ خࣹلْ مِيْ دࣹفْ لࣹيگْ-لࣹيگْ مُ سِڪّ لࣹينْ، لࣹيگْ-لࣹيگْ مُ دࣴگَّلْ لࣹينْ؞ ‏16 تࣹ نࣷونُ لَيْ دࣹمࣹ ڪࣹرَ بࣺسْ بَ يࣵلَّ دِ اࣵتّࣹ مبِرِ ڪُمْݒَ يُ نِتْ ݧِ، ݖِ سَيْتُبْ عِيسَىٰ اَلْمَسِ، نِ ڪࣷ سَمَ خِبَارُ جࣵمّ بِ اِنْدࣹي؞
يࣷونُ مُوسَىٰ نَ جَرِݧْ ڪِ ڪࣷ تࣷݒّ
‏17 مَ دِڪّ نَگْ ݖِ يَوْ مِيْ وُيࣷو تُرُ يَوُتْ، تࣹ يَاڪَارْ يࣷونُ مُوسَىٰ، بࣷولࣹ ݖِ دِ سَگࣷو يࣵلَّ؞ ‏18 يَا خَمْ ݖࣷوبَرࣹيْ يࣵلَّ، تࣹ يَا جࣵنگࣹ ݖِ يࣷونُ مُوسَىٰ، بَ خَمْ لِ اَمْ سࣷلࣷ دࣴگّ؞ ‏19 يَوْ مِ مُ وٗورْ نࣹ يَا دِ وࣷمَّتْڪَتُبْ گُمْبَ يِ، تࣹ دِ لࣹيرُ ݧِ ݖِ لࣴنْدࣴمْ گِ، ‏20 يَا دِ جࣵنگّلࣹڪَتُبْ نَاتَخُونَ يِ، دِ سࣴرِݧُبْ تُوتْ-تࣵنڪّ يِ، گَنَّاوْ يَا جَگࣷو ݖِ يࣷونُ مُوسَىٰ، مبࣷولࣹمْ لُيْ جࣴمُّ خَمْ-خَمْ، اَڪْ لࣺݒّ لُيْ دࣴگّ؞ ‏21 ڪࣷنْ يَوْ مِيْ جࣵنگّلࣹ، دࣷو جࣵنگَلْ سَ بࣷݒّ؟ يَوْ مِيْ وَارࣹ، دِ آيࣹ سࣵݖّ، مبَا دࣷو سࣵݖّ؟ يَوْ مِيْ آيࣹ نجَالࣷو، مبَا دࣷو جَالࣷو؟ ‏22 يَوْ مِ بَݧْ بࣷڪَّالࣹ، مبَا دࣷو سࣴخࣴتࣷو جَامࣷوڪَايِ بࣷڪَّالࣹڪَتْ يِ؟ ‏23 يَا نگِ سَگࣷو يࣷونُ مُوسَىٰ تࣹ دِ ڪࣷ مࣷيْ بَيْ تࣹدَّدِلْ يࣵلَّ!‏ ‏24 يࣹينْ دࣹ يࣹينَ تَخْ ݧُيْ ݧࣵڪّࣹ تࣹگِّنْ تُرُ يࣵلَّ ݖِ بِيرْ خࣹيتِ جَامْبُرْ يِ دُلْ يَوُتْ، تࣹ نࣷونُ لَ ڪࣷ مبِندّ مِ اِنْدࣹي؞
خَرَفْ نَ جَرِݧْ بࣷرࣷومْ
‏25 آدَيْ خَرَفْ نَگْ اَمَلْ نَ لَ نجَرِݧْ، ندࣹگَمْ سࣵمّ نگَ يࣷونْ وِ ڪࣷ دِگْلࣹ؞ وَايࣹ سࣷو دࣹي مࣷيْڪَتُبْ يࣷونْ وِ، سَگْ خَرَفْ، ݧࣵڪَّ خَرَفْ لَيْ دࣷونْ؞ ‏26 ڪࣷنْ ڪَيْ ڪِ خَرَفُلْ، تࣹيْ جࣴفࣹ نجُبْ گِ يࣷونُ مُوسَىٰ سَنْتَانࣹ، ݧࣵڪَّ خَرَفَمْ، دࣹيسُ ڪࣷ ڪࣷ وَݧَّلْ اَگْ خَرَفْ؟ ‏27 ڪِ خَرَفُلْ ݖِوْ يَرَمَمْ، تࣹ تࣹيوُ ڪࣷو سࣵمّ يࣷونُ مُوسَىٰ، مࣷو لَيْ تࣹگْ اَبْ دَانْ، يَوْ مِ بࣷولࣹ مبِنْدُمْ يࣷونُ مُوسَىٰ، اَڪْ سَگْ خَرَفْ، تࣹ تࣹيوُ لَ دِ مࣷيْڪَتُبْ يࣷونْ وِ؞ ‏28 اَبْ يَوُتْ، دُ ݖِ مࣹلࣷ وُيْ فࣹيݧْ لࣹيسْ ڪࣷيْ نࣹڪّࣹ، تࣹ خَرَفْ اِتْ دُ ݖِ مࣵنْدَرْگَ مُيْ فࣹيݧْ ݖِوْ يَرَمْ؞ ‏29 وَايࣹ اَبْ يَوُتْ دࣴگّ، ڪُمْݒَگْ بِيرْ خࣷلْ لَ ڪࣷيْ نࣹڪّࣹ، خَرَفْ اِتْ مبِرُمْ خࣷلْ لَ؛ ݖِ خࣹلْ مِ لَيْ سࣷتّࣹي، وَايࣹ دُ ݖِ مبِنْدُمْ يࣷونْ؞ نگࣴرࣴمْ لَ اِتْ دُ ݖِ نِتْ لَيْ جٗگࣹ، خَنَا يࣵلَّ؞
