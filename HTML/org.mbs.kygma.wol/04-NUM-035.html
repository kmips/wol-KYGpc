لُ جࣴمْ ݖِ ݖࣺرِ سُوفِ لࣹوࣹينْ ݧِ
35
‏1 اَجِ سَخْ جِ وَخَتِ اَڪْ مُوسَىٰ ݖَ جࣷورِ مࣷوَبْ، ݖَ وࣹتُ دࣹخُ يُرْدَنْ، فَ جࣵڪَّارْلࣷوڪْ يࣹرِڪࣷ؞ مُ نࣹ ڪࣷ:‏ ‏2 «سَنْتَلْ بࣵنِّ إِسْرَاءِيلْ ݧُ سࣵڪَّلْ لࣹوࣹينْ ݧِ ݖِ سࣹينْ ݖࣺرُ سُوفْ، اَيْ دࣴڪّ يُ لࣹوࣹينْ ݧِ دࣴڪّࣹ؞ اࣵلُّ ݒَرْلُ يَ وࣴرْ دࣴڪّ يَ اِتْ، جࣷخْلࣹينْ ڪࣷ لࣹوࣹينْ ݧِ؞ ‏3 سُ ڪࣷ دࣹفࣹي دࣴڪّ يࣷويُ ݧࣹيلْ لࣹينْ، ݧُ دࣴڪّࣹ؞ سࣹينْ اࣵلِّ ݒَرْلُ اِتْ ݧࣹيلْ سࣹينُگْ جُرْ، اَڪْ سࣹينْ اَلَلْ، اَڪْ سࣹينْ مبࣷولࣹمْ مَلَ؞ ‏4 ݒَرْلُيْ دࣴڪّ يِ نگࣹينْ دِ جࣷخْ لࣹوࣹينْ ݧِ، ݖِ مِيرُ دࣴڪّ يَ لَيْ دَلࣹ، جࣴمْ بِتِ لُ دَيْ نِ جُنِّيْ خَسَبْ، يࣹمࣷوڪْ جُرٗومِ تࣺيمࣺيرِ مࣹيتَرْ، ݖِ وࣹتْ گُ نࣹڪّ؞ ‏5 سُ ڪࣷ دࣹفࣹي نگࣹينْ نَتّࣹ ݖَ بِتِ دِگْ-دِگَلْ بُ گُدّࣹ ݧَارِ جُنِّيْ خَسَبْ، يࣹمࣷوڪْ جُنِّيْ مࣹيتَرْ ݖِ وࣹتُ ݒࣹنڪُ؛ ݧَارِ جُنِّيْ خَسَبْ، يࣹمࣷوڪْ جُنِّيْ مࣹيتَرْ، وࣹتُ بࣴجْ-سَالُمْ؛ ݧَارِ جُنِّيْ خَسَبْ، يࣹمࣷوڪْ جُنِّيْ مࣹيتَرْ، وࣹتُ سࣷوُ؛ ݧَارِ جُنِّيْ خَسَبْ، يࣹمࣷوڪْ جُنِّيْ مࣹيتَرْ، وࣹتُ بࣴجْ-گࣵنَّارْ، دࣴڪّ بِ نࣹڪّ ݖِ دِگّ بِ؞ دِگْ-دِگَلْ بࣷوبُ دَيْ ݧࣹيلْ لࣹوࣹينْ ݧِ، دِ سࣹينْ ݒَرْلُيْ دࣴڪّ؞ ‏6 دࣴڪّ يِ نگࣹينْ دِ جࣷخْ لࣹوࣹينْ ݧِ، ݧࣷو دِ جُرٗومْ بࣹنِّ دࣴڪِّ رَوْتُ، يِ ڪُ رࣹيْ نِتْ مَنَ رَوࣹجِ، اَڪْ يࣹنࣹينْ ݧࣹينتّ فُڪِّ دࣴڪّ اَڪْ ݧَارْ يُ نگࣹينْ لࣹينْ ݖِيْ دࣷلِّلْ؞ ‏7 ڪࣷنْ مبࣷولࣹمْ دࣴڪّ يِ نگࣹينْ دِ جࣷخْ لࣹوࣹينْ ݧِ، ݧࣹينتّ فُڪّ اَڪْ جُرٗومْ ݧࣹتّ لَ، اَڪْ سࣹينْ اࣵلِّ ݒَرْلُ؞ ‏8 دࣴڪّ يِ نگࣹينْ لࣹينْ دِ گࣺنّࣹيلْ ݖِ ݖࣺرِ سُوفِ بࣵنِّ إِسْرَاءِيلْ نَگْ، لَ ݖَيْ گࣺنّࣹ ݖَ ݧَ اࣴݒّلࣹ سُوفْ، رࣺيَلْ لࣹينْ ڪࣷ، تࣹ لَ ݖَيْ گࣺنّࣹ ݖَ ݧَ نࣺيوْلࣹ سُوفْ، نگࣹينْ نࣺيوَلْ ڪࣷ؞ گِيرْ گُ نࣹڪّ دَالْ، دَيࣷبْ ݖࣺرُ سُوفَمْ سِ مُ سࣺدُّ، مࣷومْ لَيْ تࣷݒّ، نگِرْ سࣵڪّࣹ ݖِ اَيْ دࣴڪَّمْ، جࣷخْ لࣹوࣹينْ ݧِ؞»
لُ جࣴمْ ݖِ دࣴڪِّ رَوْتُ يِ
‏9 اَجِ سَخْ جِ وَخَتِ مُوسَىٰ، نࣹ ڪࣷ:‏ ‏10 «وَخَلْ بࣵنِّ إِسْرَاءِيلْ، نࣹ لࣹينْ بُ ݧُ جࣵلّࣹي دࣹخُ يُرْدَنْ بَ دُگّ رࣺيوُمْ ڪَنَانْ، ‏11 نَݧُ سࣵڪّ اَيْ دࣴڪّ يُيْ دࣷونْ سࣹينْ دࣴڪِّ رَوْتُ، بَ ڪُ رࣹيْ نِتْ تࣹ دُ تࣹيࣹيفَمْ، مَنْ فَا رَوࣹجِ؞ ‏12 مࣷويْ دࣷونْ سࣹينْ دࣴڪِّ رَوْتُ يࣹيسْ دِ مُݖّࣹ ݖِ مبࣷڪُّمْ جࣷتْڪَتْ بُيْ تࣷݒّࣹ¤، بَ ڪِ رࣹيْ نِتْ دُ دࣹي، تࣹ تَخَوَگُلْ فَ ڪَنَمْ مبࣷولࣷو مِ، نگِرْ ݧُ اࣵتّࣹ ڪࣷ؞
‏13 «ڪࣷنْ دࣴڪّ يِ نگࣹينْ دِ جࣷخْ لࣹوࣹينْ ݧِ، جُرٗومْ بࣹنَّ ݖِيْ دࣷونْ سࣹينْ دࣴڪِّ رَوْتُ؞ ‏14 ݧࣹتِّ دࣴڪّ يُ نگࣹينْ دِ جࣷخࣹ ݖِ وࣵلُّ دࣹخُ يُرْدَنْ وِي، اَڪْ ݧࣹتِّ دࣴڪّ يُ نگࣹينْ دِ جࣷخࣹ ݖَ رࣺيوُمْ ڪَنَانْ، لࣺݒّ دِ اَيْ دࣴڪِّ رَوْتُ؞ ‏15 وَايࣹ مُيْ بࣵنِّ إِسْرَاءِيلْ، مُيْ دࣷخَنْدࣺيمْ، مُيْ گَنْ گُ نࣹڪّ ݖِ سࣹينْ بِيرْ، نَ جُرٗومْ بࣹنِّ دࣴڪّ يࣷويُ دࣷونْ سࣹينْ رَوْتُ يࣹينْ ݧࣺݒّ، بَ ڪࣺݒّ ڪُ رࣹيْ نِتْ تࣹ دُ تࣹيࣹيفَمْ، مَنْ فَا رَوࣹجِ؞
‏16 «سُ دࣹي اَگْ وࣹݧْ لَ ڪࣷ دٗورْ، مُ دࣹي، مࣷو ڪࣷ رࣹيْ، تࣹ دࣹي رࣹڪّ مࣷويْ اࣵتّࣹبْ ڪِ رࣹيْ؞ ‏17 سُ دࣹي اَوْ دࣷجْ وُ مَنَ رࣹيْ لَ جࣴلْ، دٗورْ ڪࣷ ڪࣷ، مُ دࣹي، مࣷو ڪࣷ رࣹيْ، تࣹ دࣹي رࣹڪّ مࣷويْ اࣵتّࣹبْ ڪِ رࣹيْ؞ ‏18 سُ دࣹي بَنتّ بُ مَنَ رࣹيْ لَ جࣴلْ، دٗورْ ڪࣷ ڪࣷ، مُ دࣹي اِتْ، مࣷو ڪࣷ رࣹيْ، تࣹ دࣹي رࣹڪّ مࣷويْ اࣵتّࣹبْ ڪِ رࣹيْ؞ ‏19 جࣷتْڪَتْ بِيْ تࣷݒّࣹ بَڪَّنْ بَ نَگْ، مࣷويْ رࣹيْ ڪَ رࣹيْ؛ فُ مُ دَجࣹيڪْ مࣷومْ، مࣷو ڪࣷيْ رࣹيْ؞ ‏20 سُ دࣹي ݖِ بِيرْ مبَݧࣹيلْ لَ دَلࣹ ݖِ ڪَوَمْ، مبَا مُ سࣵنِّ ڪࣷ اَوْ سَانْ وُ مُ ڪࣷ وَاجَلْ، بَ مُ دࣹي، ‏21 مبَا ݖِ بِيرْ مبَݧࣹيلْ لَ ڪࣷ دٗورࣹ لࣷخࣷومْ بَ مُ دࣹي، سُ بࣷوبَا دࣹي رࣹڪّ مࣷويْ اࣵتّࣹبْ ڪَ دٗورࣹ؞ مࣷو ڪࣷ بٗومْ؞ جࣷتْڪَتْ بَيْ تࣷݒّࣹ بَڪَّنْ بَ نَگْ، بُ دَجࣹيڪْ ڪَ رࣹيْ، دَ ڪࣷيْ رࣹيْ؞
‏22 «ندࣹگَمْ نَگْ دُ ݖَاگࣹينُ مبَݧࣹيلْ، خَنَا مُ جࣹڪِّ دَلْ ݖِ ڪَوَمْ، مبَا مُ سࣵنِّ ڪࣷ ڪࣴفْ ڪُ مُ مَنَ دࣷونْ، تࣹ دُ لُ مُ وَاجࣷونْ، ‏23 مبَا مُيْ دࣷجْ وُ مَنَ رࣹيْ وُ مُ وَدَّلْ ݖِ ڪَوْ نِتْ ڪِ، بَ نِتْ ڪِ دࣹي، تࣹ گِسُلْ وࣷونْ نِتْ ڪِ، تࣹ نِتْ ڪَ دُ وࣷونْ اَبْ نࣷونَمْ، تࣹ فࣹڪُّلْ اِتَمْ مُيْ سࣵڪُّ لࣷرَانگࣹيْ نِتْ ڪَ، ‏24 سُ بࣷوبَا لَ مبࣷولࣷو مَيْ وࣺيَلْ دِگَّنْتࣹ ڪِ رࣹيْ، اَڪْ جࣷتْڪَتْ بَيْ تࣷݒّࣹ، مࣷو دِ اࣵتّࣹ يِي ݖِ يࣷونْ تࣴرَلْ:‏ ‏25 مبࣷولࣷو مِ دَيْ مُسَلْ ڪِ رࣹيْ ݖِ لࣷخࣷلْ جࣷتْڪَتُ بَڪَّنُ ڪَ دࣹي، تࣹ مبࣷولࣷو مَ دَ ڪࣷيْ دࣹلّࣷو ݖَ دࣴڪُّبْ رَوْتࣷومْ بَ مُ لࣵقُجِ وࣷونْ؞ فَ لَيْ تࣷوگْ، بَ ڪࣹرَ دࣹيوُگْ سَرْخَلْڪَتْ بُ مَگْ بَ ݧُ بَاخࣷو فَلࣹ دِوْ گُ سࣹلّ؞ ‏26 ندࣹگَمْ نَگْ ڪَ رࣹيْ مࣷو گࣺنَّ گࣺنّ، بَ وࣹيسْ ڪࣹمُ دࣴڪُّبْ رَوْتࣷومْ بَ مُ لࣵقُجِ وࣷونْ، ‏27 بَ جࣷتْڪَتْ بَيْ تࣷݒّࣹ بَڪَّنْ فࣹڪّ ڪࣷ فُ وࣹيسْ ڪࣹمُ دࣴڪُّبْ رَوْتࣷومْ، سُ بࣷوبَا سُ ڪࣷ جࣷتْڪَتُ بَڪَّنْ بَ رࣹيࣹي، دُ تَخْ مُ گࣵدُّ بَڪَّنْ؞ ‏28 ندَخْ ڪَتْ ݖِ بِيرْ دࣴڪُّبْ رَوْتࣷومْ لَ وَرَ تࣷوگْ، بَ ڪࣹرَ دࣹيوُگْ سَرْخَلْڪَتْ بُ مَگْ بِ؞ گَنَّاوْ بُ سَرْخَلْڪَتْ بُ مَگْ بِ دࣹييࣹي نَگْ، ڪِ رࣹيْ سَݧْ نَا دࣹلُّ فَ مُ بࣷڪّ؞
‏29 «لࣷولُ مࣷويْ دࣷگَلُ يࣷونْ بِ لࣹينْ ݧࣹيلْ، يࣹينْ اَڪْ سࣹينْ اَسْڪَنْ، ݖِ سࣹينْ مبࣷولࣹمِ دࣴڪُّوَايْ؞
‏30 «ڪࣺݒّ ڪُ فَاتْ بَڪَّنْ نَگْ، ݖِ ڪَوْ ڪࣵدُّ گُ تُڪّࣹي ݖِ اَيْ سࣹيدࣹ دࣷݝّ لࣹيسْ ڪࣷ سَݧَ رࣹيࣹ، مࣷومْ مَ رࣹيْ، وَايࣹ بࣹنُّبْ سࣹيدࣹ دُ سࣹيدࣹيلْ نِتْ، نگِرْ ݧُ تࣹگْ ڪࣷ اࣵتّࣹبْ دࣹي؞
‏31 «تࣹ اِتْ بُلࣹينْ جࣴلْ بࣹنّ خَالِسُ نجࣷتُگْ بَڪَّنُ ڪࣹنّ ڪُ رࣹيْ نِتْ ڪُ يࣹلّࣷو دࣹي؞ ڪࣷوڪُ دࣹي رࣹڪّ مࣷويْ اࣵتّࣹيمْ؞ ‏32 بُلࣹينْ جࣴلْ اِتْ خَالِسُ نجࣷتُگْ نِتْ ڪُ لࣵقُجِ ݖَ دࣴڪُّبْ رَوْتࣷومْ، نگِرْ مَيْ ڪࣷ مُ دࣹلُّ فَ مُ دࣴڪّ، تࣹ سَرْخَلْڪَتْ بُ مَگْ بِ دࣹيوُلْ؞ ‏33 بُلࣹينْ تِلِمَلْ رࣺيوْ مِ نگࣹينْ نࣹڪّ؞ دࣹرࣹتْ ڪَتْ مࣷويْ تِلِمَلْ رࣺيوْ مِ، تࣹ رࣺيوْ مِ دࣹيسُ ڪࣷ جࣷتࣹ لࣹنّ، ݖِ دࣹرࣹتْ جُ ݧُ تُورْ ݖِ بِيرَمْ، لُ مࣷيْ دࣹرࣹتُ ڪَ تُورْ دࣹرࣹتْ؞ ‏34 بُلࣹينْ سࣷبࣹيلْ رࣺيوْ مِ نگࣹينْ دࣴڪّ، تࣹ مَنْ مَ مࣵڪَّانࣷو ݖِ بِيرَمْ، ندَخْ مَنْ اَجِ سَخْ جِ مَا مࣵڪَّانࣷو ݖِ بِيرْ بࣵنِّ إِسْرَاءِيلْ؞»
