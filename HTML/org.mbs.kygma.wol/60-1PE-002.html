اَلْمَسِ مࣷويْ ڪࣹنُگْ مبࣷولࣷومْ گࣴمْڪَتْ ݧِ
2
‏1 ڪࣷنْ نَگْ بَلࣹينْ مبࣷولࣹمْ خࣹيتُ مبࣷنْ، اَڪْ مبࣷولࣹمْ نجُبْلَݝْ، اَڪْ جِنِگَلْ، اَڪْ ڪِݧَانْ، اَڪْ مبࣷولࣹمْ خࣹيتُ يࣵقْ دࣹرْ؞ ‏2 نِ لِيرْ بُيْ دࣷورَ جُدُّ يࣵڪَّمْتࣹي نࣵمْݒࣹ، نِ نگࣹينْ وَرَ يࣵڪَّمْتࣹي نࣵمݒّ سࣷووْ مُ سࣹتْ مِيْ ڪࣵدُّگْ يࣵلَّ؞ مࣷويْ تَخْ نگࣹينْ مࣵگّ ݖِ سࣹينْ دُنْدُگْ تࣹخࣹ، ‏3 گَنَّاوْ مࣷسْ نگࣹينْ ݖِ مبَاخُگْ سَنگّ بِ؞
‏4 دِڪّلࣹينْ ݖِ مࣷومْ، مࣷويْ دࣷجْ وُيْ دُندّ؞ نِتْ ݧࣹي ڪࣷ بࣹدِّ مࣷوسْ، وَايࣹ يࣵلاَّ ڪࣷ فَلْ، فࣷنڪّ ڪࣷ؞ ‏5 تࣹ يࣹينْ اِتَمْ يࣹينْ اَيْ دࣷجِ ݖࣴسْلاَيْ يُيْ دُندّ، دࣷجْ يِ نࣷو گِ تَبَخࣹ ڪࣴرُگْ يࣵلَّ، ݧࣹيلْ لِگّࣺييُ ݖَرْخَلْ گُ سࣹلّ، نگِرْ نگࣹينْ دِ ڪࣷ جࣺبَّلْ اَيْ سَرَخْ يُ نࣷو گِ رَفࣹتْلُ، سَرَخْ يُ يࣵلَّ نَنگُ ندَخْ عِيسَىٰ اَلْمَسِ؞ ‏6 مࣷو وَرَلْ لِي دِڪّ ݖِ مبِندّ مِ:‏
«مَا نگِي دِ دࣴجْ فِ سِيࣷݝْ، اَوْ دࣷجُ دࣷورُڪَايْ،
دࣷجُ تࣵنّࣺيفْ وُ گࣵنجَرࣹ؞
وِي دࣷجْ، ڪُ ڪࣷ وࣺيرࣷو، دࣷو رُسْ مُڪّ¤؞»
‏7 ڪࣷنْ نَگْ يࣹينْ ݧِ گࣴمْ، يࣹينْ لَ گࣵنجَرْ گِ ݧࣹيلْ؞ وَايࣹ ݖِ ݧِ گࣴمُلْ، نَگْ:‏
«مࣷو دِ دࣷجْ وَ تَبَخْڪَتْ يَ بࣹدِّ وࣷونْ،
تࣹ مُ مُجّ دِ دࣷجُ ݖࣷلّ وِ¤؞»
‏8 تࣹ اِتْ:‏
«مࣷويْ دࣷجْ وُيْ فَڪّتَلࣹ،
دِ خࣹيرْ وُيْ دَانࣹيلْ¤؞»
ݧࣷو ڪࣷ فَڪَّسْتَلࣷو ندࣹ ݧࣷو گࣴمُلْ ڪࣵدُّ گِ، تࣹ اِتْ لࣷولُ لَ لࣹينْ دࣷگَلْ بَ سࣺدّࣷونْ؞
‏9 وَايࣹ يࣹينْ نَگْ يࣹينْ اَيْ گِيرْ گِيْ تࣵنّࣹيفْ، يࣹينَيْ سَرْخَلْڪَتِ بُورْ، يࣹينْ اَيْ اَسْڪَنْ وُ سࣹلّ، يࣹينْ اَيْ خࣹيتْ وِ ݧُ جࣷتْ، نگِرْ نگࣹينْ بِرَلْ جَلࣷورࣹ يُ وࣺيتْ يِ ݧࣹيلْ ڪِ لࣹينْ وࣷو، گࣺنّࣹي لࣹينْ ݖِگْ لࣴنْدࣴمْ، اِنْدِ لࣹينْ ݖِ لࣹيرَمْ گُ يࣺيمࣹ؞ ‏10 يࣹينْ ݧِ دُلْ وࣷونْ سَخْ اَوْ خࣹيتْ، دࣺمبّ، تࣹيْ يࣹينْ اَيْ خࣹيتْ وِ يࣵلَّ ݧࣷݧࣷو¤؛ دَجَگُلࣹينْ وࣷونْ يࣴرْمَنْدࣹ، وَايࣹ تࣹيْ دَجْ نگࣹينْ يࣴرْمَنْدࣹ؞
لِيْ وࣵلُّ گࣴمْڪَتُ اَلْمَسِ ݖِ اࣵدِّنَ
‏11 سࣷݒّࣹ يِ، اَيْ گَنْ نگࣹينْ، اَيْ دࣷخَنْدࣺيمْ ݖِ ڪَوْ سُوفْ؞ ڪࣷنْ نَگْ لِ مَ لࣹينْ دِ دࣺنڪّ مࣷو دِ نگࣹينْ مࣷيُ خࣹمّࣹمْتࣺيفِ بَڪَّنْ يِيْ خࣹيخْ اَڪْ اَبْ خࣷلْ؞ ‏12 سࣹينْ دُنْدِنْ ݖِ بِيرْ جَامْبُرْ ݧِ گࣴمُلْ، رَفࣹتَلْ-لࣹينْ ڪࣷ، بَ سُ ݧُ لࣹينْ وَخَلࣷونْ نࣹ اَيْ دࣹفْڪَتِ لُ بࣷنْ نگࣹينْ اِتَمْ، سࣹينْ جࣴفْ جُ بَاخْ لَݧُيْ گِسْ، تࣹ مࣷومْ لَݧُيْ سࣵبَّالࣹ يࣵلَّ، ڪࣹرَ بࣺسُبْ دِڪَّمْ؞
دࣺگَّلْ-لࣹينْ نجِيتْ يِ
‏13 نَنگُلْ-لࣹينْ گࣺݒّ ڪِلِفْتࣹيفْ گُ تࣹگࣹ ݖِ اࣵدِّنَ نگِرْ بࣷرࣷومْ بِ؛ مُيْ بُورْ بُ مَگْ بِ گࣴنَ ڪَوࣹ، ‏14 مبَا بࣷرࣷومْ دࣴڪّ يِ دِيْ ندَوَمْ، تࣹ مُ تَبّ لࣹينْ، نگِرْ ݧُ مبُگَلْ ݧِيْ دࣹفْ لُ بࣷنْ، تࣹ سَرْگَلْ ݧِيْ دࣹفْ لُ بَاخْ؞ ‏15 ندَخْ ݖࣷوبَرࣹيْ يࣵلَّ ڪَتْ مࣷو دِ نگࣹينْ دِ دࣹفْ لُ بَاخْ، بَ وࣹدَمَلْ نِتْ ݧُ ݧࣵڪّ بࣷݒّ ݧِ خَمُلْ دَرَ؞ ‏16 اَيْ گࣷرْ نگࣹينْ، وَايࣹ جَامِ يࣵلَّ نگࣹينْ؞ ڪࣷنْ بُلࣹينْ تَافَنْتࣷو لِ نگࣹينْ دِيْ گࣷرْ بَ دࣹفْ لُ بࣷنْ؞ ‏17 ݧࣺݒّ نَگْ نگࣹينْ دِ تࣹرَلْ؛ مبࣷولࣷومْ گࣴمْڪَتْ ݧِ، سࣷݒّلࣹينْ لࣹينْ؛ يࣵلَّ، رَگَلْ-لࣹينْ ڪࣷ؛ تࣹ بُورْ، تࣹرَلْ لࣹينْ ڪࣷ؞
نَنُ مُݧْ ݖࣷونࣷ، رࣷيࣹ ڪࣷ اَلْمَسِ
‏18 يࣹينْ جَامْ يِ، نَنگُلْ-لࣹينْ تࣹ وࣹگْ وࣹگࣹيلْ گُ مَتْ سࣴڪّ سࣹينِ سَنگّ، تࣹ دُ سَنگّ يِ بَاخْ تࣹ لࣹوࣹتْ دࣷݝّ، وَايࣹ ݧُ وࣹخْ ݧِ اِتَمْ ݖِ لَݧُ؞ ‏19 ندَخْ نِتْ ڪِ سُ سࣺدّࣹي يࣵلَّ اَبْ ݖࣺرْ، بَ تَخْ مُ مُݧْ تِيسْ اَڪْ ݖࣷونࣷ بُ ڪࣷ يࣷونْ تࣹگُلْ، لࣷولُ اَوْ يِوْ لَ؞ ‏20 مُݧْ اَيْ دٗورْ يُ لَ فࣹڪّ تࣷوݧْ، اَنَ لُ ݖِيْ ندَمْ؟ وَايࣹ سࣷو مُݧࣹي ݖࣷونࣷ بࣷو دَجْ نگِرْ لُ بَاخْ لࣷو دࣹفْ، لࣷولُ دࣹ اَوْ يِوْ لَ فَ ڪَنَمْ يࣵلَّ؞ ‏21 ݖِ لࣷولُ سَخْ لَݧُ لࣹينْ وࣷو، ندَخْ اَلْمَسِ اِتَمْ سࣷنَّلْ نَ لࣹينْ، تࣹ مࣷو لࣹينْ بࣵيّࣹيلْ اَبْ رࣷيُڪَايْ، نگِرْ نگࣹينْ جَارْ ݖِيْ تࣵنڪَمْ؞
‏22 «مࣷو بࣵڪَّارُلْ، فࣹنُلْ¤؞»
‏23 سَاگَ نَݧُ ڪࣷ تࣹ فࣹيُوُلْ، دَجْ نَ ݖࣷونࣷ تࣹ تࣴڪّࣷووُلْ، وَايࣹ بࣷݒَّمْ لَ دࣺنڪّ ڪِيْ اࣵتّࣹ دࣴگّ؞ ‏24 «مࣷومْ ݖِ بࣷݒَّمْ مࣷو گࣵدُّ سُنُيْ بࣵڪَّارْ، مࣷو ڪࣷ تࣹگࣷو ݖِ يَرَمُ بࣷݒَّمْ ݖَ بَنتّ بَ،» نگِرْ نُ دࣹييَلْ بࣵڪَّارْ تࣹ دُنْدَلْ نجُبْ؛ «مࣷومْ مࣷويْ ڪِ نگࣹينْ وࣺرࣹ ݖِ اَيْ گٗومَمْ¤؞» ‏25 «يࣹينَ رࣺيرࣷونْ نِيْ خَرْ¤،» وَايࣹ دࣺلْسِ نگࣹينْ تࣹيْ ݖِ سࣹينُبْ سࣵمّ، سࣹينْ وَتُّڪَتُبْ خࣷلْ؞
