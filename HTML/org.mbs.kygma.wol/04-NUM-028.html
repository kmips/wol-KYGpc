بࣺسْ بُ نࣹڪّ لاَجْ نَ اَبْ سَرَخْ
28
‏1 اَجِ سَخْ جِ وَخَتِ مُوسَىٰ، نࣹ ڪࣷ:‏ ‏2 «سَنْتَلْ بࣵنِّ إِسْرَاءِيلْ نࣹ لࣹينْ سَمَ سَرَخْ، سَمَ ݧَمْ وِ جࣴمْ ݖِ سَمَ سَرَخِ سَوَرَ نگِرْ خࣹتُگْ جࣵمّ، ݧࣹيلْ مَ، نَݧُ فَرْلُ ݖِ دِ مَ ڪࣷ اِنْدِلْ سَا يُ اࣵݒَّمْ جࣷتࣹي؞ ‏3 نگَ وَخْلࣹينْ نࣹ لࣹينْ، لِي مࣷو دِ سَرَخُ سَوَرَ بِ ݧُيْ اِنْدِلْ اَجِ سَخْ جِ:‏ خَرْ يُ گٗورْ يُ مࣹنّ اَتْ يُ مُݖّ سِڪّ، ݧَارْ ݖِ بࣺسْ بِ، مُيْ سَرَخُ رࣹنْدِ-دٗومَلْ بِ وَرْ بࣺسْ بُ نࣹڪّ؞ ‏4 مࣹنّ خَرْ مِ، نگࣹينْ سَرْخَلْ ڪࣷ ݖِ سُبَ، ݧَارࣹيلْ بِ، نگࣹينْ سَرْخَلْ ڪࣷ دِگَّنْتࣹ نگࣷونْ اَڪْ جَنتّ بُ سࣷ؞ ‏5 بࣹنّ فُڪّࣹيلُ اࣹفَبْ سُنگُفْ سُ مُݖّ اَيِبْ، دِ ݧࣹتِّ ڪِلࣷ، مࣷويْ سَرَخُ ݒࣹݒّ بِ خَرْ مُ نࣹڪّ دِ اࣵنْدَلْ، ݧُ تࣷويَلࣹ ڪࣷ وَلاَتُ خِينُ دِوُ سࣹگَلُ اࣷلِوْ، دِ لِيتَرْ اَڪْ گࣹنّ-وࣵلّ؞ ‏6 لࣷولࣷو دِ رࣹنْدِ-دٗومَلْ بِ وَرْ بࣺسْ بُ نࣹڪّ، نَ ݧُ ڪࣷ دࣷگَلࣹ وࣷونْ ݖَ تُنْدُ سِنَيِ، نگِرْ خࣹتُگْ جࣵمّ، مُيْ سَرَخُ سَوَرَ، ݧࣹيلْ اَجِ سَخْ جِ؞
‏7 «سَرَخُ تُورَمْ وَلاَتُ خِينْ لَيْ دࣷونْ، دِ لِيتَرْ اَڪْ گࣹنّ-وࣵلّ نگِرْ خَرْ مُ ݖِ نࣹڪّ؞ ݖِ اࣴتّ بُ سࣹلّ بِ لࣹيسْ دِ سࣷتِّ سَرَخُ تُورُ بِ، نَانْ گُ دِيسْ گࣷوگُ، ݧࣹيلْ اَجِ سَخْ جِ؞ ‏8 ݧَارࣹيلُ خَرْ مِ، سَرْخَلْ-لࣹينْ ڪࣷ دِگَّنْتࣹ نگࣷونْ اَڪْ جَنتّ بُ سࣷ، مُ اࣵندّ اَڪْ سَرَخُ ݒࣹݒَّمْ اَڪْ سَرَخُ تُورَمْ نِڪِ ݖِ سُبَ، نگࣹينْ دࣹفْ ڪࣷ سَرَخُ سَوَرَ نگِرْ خࣹتُگْ جࣵمّ، ݧࣹيلْ اَجِ سَخْ جِ؞
بࣺسُبْ نࣷفْلاَيْ اَمْ نَ سَرَخَمْ
‏9 «بࣺسُبْ نࣷفْلاَيْ نَگْ ݧَارِ خَرِ مࣹنّ اَتْ يُ مُݖّ سِڪّ لَيْ دࣷونْ، اَڪْ سَرَخُ ݒࣹݒَّمْ، بِيْ ݧَارِ فُڪّࣹيلِ اࣹفَيْ سُنگُفْ سُ مُݖّ اَيِبْ، دِ جُرٗومْ بࣹنِّ ڪِلࣷ، ݧُ تࣷويَلࣹ ڪࣷ اَگْ دِوْ، تࣹ مُ اࣵندّ اَڪْ سَرَخُ تُورَمْ؞ ‏10 لࣷولࣷويْ سَرَخُ رࣹنْدِ-دٗومَلُ بࣺسُبْ نࣷفْلاَيْ، ݖِ بࣺسُبْ نࣷفْلاَيْ بُ نࣹڪّ، تࣹ بࣷڪّࣹوُلْ اَڪْ دٗومَلْ بِ وَرْ بࣺسْ بُ نࣹڪّ، تࣹ اࣵندّ اَڪْ سَرَخُ تُورَمْ؞
تࣹرُتࣹلْ وࣹيرْ اَمْ نَ سَرَخَمْ
‏11 «تࣹ اِتَمْ سࣹينْ بࣺسِ تࣹرُتࣹلْ وࣹيرْ، نَنگࣹينْ سَرْخࣹ اَبْ رࣹنْدِ-دٗومَلْ، ݧࣹيلْ اَجِ سَخْ جِ؞ ݧَارِ يࣴڪّ يُ ندَوْ، اَڪْ مࣹنّ ڪُويْ، اَڪْ جُرٗومْ ݧَارِ خَرْ يُ گٗورْ يُ مࣹنّ اَتْ، تࣹ لُ ݖِ نࣹڪّ مُݖّ سِڪّ، ‏12 اَڪْ ݧࣹتِّ فُڪّࣹيلِ اࣹفَيْ سُنگُفْ سُ مُݖّ اَيِبْ، دِ جُرٗومْ ݧࣹينْتِ ڪِلࣷ، مُيْ سَرَخُ ݒࣹݒّ، ݧُ تࣷويَلࣹ ڪࣷ اَگْ دِوْ، بࣷولࣹيڪْ يࣴڪّ وُ نࣹڪّ، اَڪْ ݧَارِ فُڪّࣹيلِ اࣹفَيْ سُنگُفْ سُ مُݖّ اَيِبْ، دِ جُرٗومْ بࣹنِّ ڪِلࣷ، مُيْ سَرَخُ ݒࣹݒّ، ݧُ تࣷويَلࣹ ڪࣷ اَگْ دِوْ، بࣷولࣹيڪْ ڪُويْ مِ، ‏13 اَڪْ بࣹنّ فُڪّࣹيلُ اࣹفَبْ سُنگُفْ سُ مُݖّ اَيِبْ، دِ ݧࣹتِّ ڪِلࣷ، مُيْ سَرَخُ ݒࣹݒّ، ݧُ تࣷويَلࣹ ڪࣷ اَگْ دِوْ، بࣷولࣹيڪْ خَرْ مُ ݖِ نࣹڪّ؞ سَرَخُ رࣹنْدِ-دٗومَلْ لَ، نگِرْ خࣹتُگْ جࣵمّ، دِ سَرَخُ سَوَرَ، ݧࣹيلْ اَجِ سَخْ جِ؞
‏14 «اَيْ سَرَخِ تُورَمْ گࣹنّ-وࣵلُّ خِينُ بِيݧْ لَ، دِ ݧࣹتِّ لِيتَرْ ݖِ يࣴڪّ وُ نࣹڪّ، اَڪْ بࣹنّ ݧࣹتّࣹيلُبْ خِينْ، دِ ݧَارِ لِيتَرْ ݖِ ڪُويْ مِ، اَڪْ وَلاَتُ خِينْ، دِ لِيتَرْ اَڪْ گࣹنّ-وࣵلُّ بِيݧْ ݖِ خَرْ مُ ݖِ نࣹڪّ؞ لࣷولࣷويْ دٗومَلُ وࣹيرْ وِ، وࣹيرْ وُ ݖِ نࣹڪّ، بَ وࣹيرِ اَتْ مِ دَجْ؞ ‏15 اَڪْ اِتَمْ بࣹنُّبْ سِڪّࣹتْ، مࣷوڪْ سَرَخُ تُورࣷومْ نگِرْ سَرَخُ ݒٗوتُمْ بࣵڪَّارْ، ݧࣹيلْ اَجِ سَخْ جِ ݖِ بࣺسُبْ تࣹرُتࣹلْ وࣹيرْ وِ، تࣹ بࣷڪّࣹوُلْ اَڪْ رࣹنْدِ-دٗومَلْ بِ ݧُيْ سَرْخࣹ بࣺسْ بُ نࣹڪّ؞
بࣺسُبْ مُݖّ اَمْ نَ سَرَخَمْ
‏16 «وࣹيرْ وِ جࣴڪّ ݖِ اَتْ مِ نَگْ، فُڪِّ فَنَمْ اَڪْ ݧࣹينتّ مࣷويْ بࣺسُبْ مُݖّ بِ، ݧࣹيلْ اَجِ سَخْ جِ؞ ‏17 اࣴلّࣴگْ سَ، فُڪِّ فَنِ وࣹيرْ وࣷووُ اَڪْ جُرٗومْ، دِ مࣵگَّلْ؛ دِيرُ جُرٗومْ ݧَارِ فَنْ، مبُرُ مُ اَمُلْ لَوِيرْ لࣹيسْ ݖَيْ لࣹڪّ؞ ‏18 بࣺسْ بُ جࣴڪّ بࣷوبُ، وࣷوتࣹ بُ سࣹلّ لࣹيسْ ݖَيْ اَمَلْ؞ مبࣷولࣹمْ لࣹيسْ بَاخࣷو لِگّࣺييْ، دࣹيسُ ڪࣷ ݖِ لِگّࣺييْ؞ ‏19 نَنگࣹينْ ݖِ دࣹفْ سَرَخُ سَوَرَ، مُ دِبْ رࣹنْدِ-دٗومَلْ، ݧࣹيلْ اَجِ سَخْ جِ:‏ ݧَارِ يࣴڪّ يُ ندَوْ، اَڪْ مࣹنّ ڪُويْ، اَڪْ جُرٗومْ ݧَارِ خَرِ مࣹنّ اَتْ، تࣹ نگࣹينْ دࣹفْ لࣺݒّ دِ لُ مُݖّ سِڪّ؞ ‏20 يࣷويِي اࣵندّ اَڪْ سَرَخُ ݒࣹݒَّمْ، دِ سُنگُفْ سُ مُݖّ اَيِبْ، سُ ݧُ تࣷويَلࣹ اَگْ دِوْ، ݧࣹتِّ فُڪّࣹيلِ اࣹفَيْ سُنگُفْ، دِ جُرٗومْ ݧࣹينْتِ ڪِلࣷ، ݖِوْ يࣴڪّ، ݧَارِ فُڪّࣹيلِ اࣹفَيْ سُنگُفْ، دِ جُرٗومْ بࣹنِّ ڪِلࣷ، ݖِمْ ڪُويْ، ‏21 اَڪْ بࣹنّ فُڪّࣹيلُ اࣹفَبْ سُنگُفْ، دِ ݧࣹتِّ ڪِلࣷ، ݖِ مِي خَرْ، بࣹنّ فُڪّࣹيلُ اࣹفَ، ݖَ مࣹي، بَ جُرٗومْ ݧَارِ خَرْ يِ دَجْ، ‏22 اَڪْ بࣹنّ سِڪّࣹتُ ݒٗوتُمْ بࣵڪَّارْ نگِرْ سࣹينْ نجࣷتْلاَيْ؞ ‏23 تࣹ بࣷڪّࣹوُلْ اَڪْ رࣹنْدِ-دٗومَلُ سُبَ سِ، رࣹنْدِ-دٗومَلْ بࣷوبُ وَرْ بࣺسْ بُ نࣹڪّ؞ ‏24 سَرَخْ يُ مࣹلْ نࣷونُ نگࣹينْ دِ سَرْخَلْ ݖِ بࣺسْ بِ، اَڪْ سࣹينْ سَرَخُ تُورُ، دِيرُ جُرٗومْ ݧَارِ فَنْ، مُيْ ݧَمُ سَرَخُ سَوَرَ نگِرْ خࣹتُگْ جࣵمّ، ݧࣹيلْ اَجِ سَخْ جِ؞ ݖِ ڪَوْ رࣹنْدِ-دٗومَلْ بِ وَرْ بࣺسْ بُ نࣹڪّ، ݖِ لࣹيسْ ڪࣷيْ دࣷلِّ؞ ‏25 بࣺسُبْ جُرٗومْ ݧَارࣹيلْ بَ، سࣹينْ بࣺسُبْ وࣷوتࣹ بُ سࣹلّ لَيْ دࣷونْ؞ مبࣷولࣹمْ لࣹيسْ بَاخࣷو لِگّࣺييْ، دࣹيسُ ڪࣷ ݖِ لِگّࣺييْ؞
نگٗوبْ اَمْ نَ سَرَخَمْ
‏26 «بࣺسُبْ ندࣷورْتࣹلْ مࣹݧّࣹيفْ بَ نگࣹينْ دِ سَرْخࣹسِ سَرَخُ ݒࣹݒّ مُ بࣹيسْ، ݧࣹيلْ اَجِ سَخْ جِ، ݖِ سࣹينْ بِيرْ مࣵگَّلُگْ اَيِ بࣺسْ نَگْ، سࣹينْ بࣺسُبْ وࣷوتࣹ بُ سࣹلّ لَيْ دࣷونْ؞ مبࣷولࣹمْ لࣹيسْ بَاخࣷو لِگّࣺييْ، دࣹيسُ ڪࣷ ݖِ لِگّࣺييْ؞ ‏27 نَنگࣹينْ ݖِ سَرْخَلْ اَبْ رࣹنْدِ-دٗومَلْ، نگِرْ خࣹتُگْ جࣵمّ، ݧࣹيلْ اَجِ سَخْ جِ:‏ ݧَارِ يࣴڪّ يُ ندَوْ، اَڪْ مࣹنّ ڪُويْ، اَڪْ جُرٗومْ ݧَارِ خَرِ مࣹنّ اَتْ؞ ‏28 سَرَخُ ݒࣹݒَّمْ دِ سُنگُفْ سُ مُݖّ اَيِبْ، سُ ݧُ تࣷويَلࣹ اَگْ دِوْ؞ ݧࣹتِّ فُڪّࣹيلِ اࣹفَيْ سُنگُفْ، دِ جُرٗومْ ݧࣹينْتِ ڪِلࣷ ݖِ يࣴڪّ وُ نࣹڪّ، اَڪْ ݧَارِ فُڪّࣹيلِ اࣹفَ، دِ جُرٗومْ بࣹنِّ ڪِلࣷ ݖِ ڪُويْ مِ، ‏29 اَڪْ فُڪّࣹيلُبْ اࣹفَ، دِ ݧࣹتِّ ڪِلࣷ، ݖِ مِي خَرْ، فُڪّࣹيلُبْ اࣹفَ ݖَ مࣹي، بَ جُرٗومْ ݧَارِ خَرْ يِ دَجْ، ‏30 اَڪْ بࣹنُّبْ سِڪّࣹتْ نگِرْ سࣹينْ نجࣷتْلاَيْ؞ ‏31 سَرْخࣹلࣹينْ لࣷولُ، تࣹ بُمُ بࣷڪّࣹ اَڪْ رࣹنْدِ-دٗومَلْ بِ وَرْ بࣺسْ بُ نࣹڪّ تࣹيْ اࣵندّ اَڪْ سَرَخُ ݒࣹݒَّمْ؞ لُ مُݖّ سِڪّ نَگْ نگࣹينْ دِ دࣹفْ سَرَخْ يࣷويِي، تࣹ نگࣹينْ دࣹفَالࣹ سَرَخِ تُورَمْ يِ ݖِ وَرْ؞
