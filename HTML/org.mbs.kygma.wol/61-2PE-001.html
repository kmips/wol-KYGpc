ݧَارࣹيلُ بَتَاخَلْ بِ يࣵلَّ مَيْ
ݒِيࣹيرْ
مُ بِندّ ڪࣷ
1
‏1 ݖِ مَنْ سِمࣷݝْ ݒِيࣹيرْ¤، مِيْ جَامُبْ عِيسَىٰ اَلْمَسِ، دِ ندَوَمْ، ݖِ مَنْ لَ بَتَاخَلْ بِي بَوࣷو، ݧࣹيلْ ݧِ ݧُ مَيْ نگࣴمْ گُ نجࣺگْ گَ رࣺيࣹ نِ سُنُ گࣷسْ، ݖِ سُنُ نجࣹڪُّ يࣵلَّ، مُسَلْڪَتْ بِ عِيسَىٰ اَلْمَسِ؞ ‏2 اَوْ يِوْ اَڪْ جࣵمّ بَاوَانْ نَ، ݧࣹيلْ لࣹينْ، ندَخْ خَمْ گُ نگࣹينْ خَمْ يࣵلَّ، اَڪْ سُنُ سَنگّ عِيسَىٰ؞
دࣴگَّلْ-لࣹينْ وࣷوتࣹبْ يࣵلَّ
‏3 سُ دࣹي نَگْ مبࣷولࣹمْ لُ دُندّ گُ اَگْ جُلِّتࣹ لاَجْ، يࣵلاَّ نُ ڪࣷ مَيࣹ دࣷولࣹيمْ، بَ مُ نُ خَمَلࣹي نࣹ دَرَجَيْ بࣷݒَّمْ، اَڪْ جَلࣷورࣹيمْ جُ وࣺيتْ لَنُ وࣷويࣹ؞ ‏4 ݖِ جࣷوجُ دَرَجَ، اَڪْ جࣷوجُ جَلࣷورࣹ جُ وࣺيتْ لࣹيسْ نُ بَاخࣹي اَيْ دِگࣹ يُ رࣺيْ نجࣺگْ تࣹ مࣵگّ؞ يࣷويُ دِگࣹ نَگْ، بُ نگࣹينْ ݖِ جࣷتࣹي، ݖِ نگࣹينْ دِ سࣺدُّ ݖِ جِڪّࣷيْ يࣵلَّ، گَنَّاوْ بُ نگࣹينْ مُݖّࣹي ݖِ يࣵقُتࣹيْ اࣵدِّنَ، گِ خࣹمّࣹمْتࣺيفُ بَڪَّنْ وَرَلْ؞
‏5 لࣷولُ ݖِ بࣷݒَّمْ وَرَ تَخْ نگࣹينْ دࣹفْ لُ نگࣹينْ مَنْ، بَ دࣷلِّ ݖِ سࣹينْ نگࣴمْ، نجࣵمْبَارْ؛ سࣹينْ نجࣵمْبَارْ، نگࣹينْ دࣷلِّ ݖِ خَمْ-خَمْ؛ ‏6 سࣹينْ خَمْ-خَمْ، نگࣹينْ دࣷلِّ ݖِ مࣷومْ سࣹينْ بَڪَّنْ؛ مࣷومْ سࣹينْ بَڪَّنْ، نگࣹينْ دࣷلِّ ݖِ اَگْ مُݧْ؛ سࣹينُگْ مُݧْ، نگࣹينْ دࣷلِّ ݖِ اَگْ نجُلِّتࣹ؛ ‏7 سࣹينُگْ نجُلِّتࣹ، نگࣹينْ دࣷلِّ ݖِ بࣴگّ سࣹينْ بࣷڪِّ گࣴمْڪَتْ؛ بࣴگّ سࣹينْ بࣷڪِّ گࣴمْڪَتْ، نگࣹينْ دࣷلِّ ݖِ ݖࣷفࣹيلْ؞ ‏8 ندَخْ يࣷويُ جِڪّࣷ ڪَتْ سُ نگࣹينْ ڪࣷ اَمࣹي، بَ مُ لَوْ، دُنگࣹينْ يَافُسْ، دُنگࣹينْ ݧࣵڪّ نجَرِݧْ نُ نگࣹينْ خَمࣹ سَنگّ عِيسَىٰ اَلْمَسِ؞ ‏9 وَايࣹ يِي جِڪّࣷ، ڪُ ڪࣷ ݧࣵڪّ، دَا فࣹڪّ مُ گࣵتّ گِسْ-گِسْ بَ فَرْ گُمْبَ، تࣹ فࣵتّࣹ نࣹ دࣹيسْ ڪࣷو فٗوتࣷونْ بَ مُ سࣹتْ، تࣹقْلِڪࣷوڪْ بࣵڪَّارِ دࣺمْبَمْ؞
‏10 ڪࣷنْ نَگْ بࣷڪّ يِ، نَنگࣹينْ گࣴنَتࣹي دࣹفْ لُ نگࣹينْ مَنْ بَ دࣴگَّلْ وࣷوتࣹ بِ ݧُ لࣹينْ وࣷو، اَڪْ تَامُ گِ ݧُ لࣹينْ تَامُ؞ سُ نگࣹينْ ڪࣷ دࣹفࣹي ڪَتْ، دُنگࣹينْ دَانُ مُڪّ؞ ‏11 نࣷونُ لࣹيسْ لࣹينْ دِ اُبِّلࣹ بُنتّ بُ يَا بُ نگࣹينْ دِ تࣵبّࣹي ݖَ نگُورْ گُ سَخْ گَيْ سُنُ مࣷومࣹيلُ سَنگّ، سُنُ مُسَلْڪَتْ، عِيسَىٰ اَلْمَسِ؞
يࣵلَّ وَخْيُلْ نَ يࣷنࣹنتّ يِ
‏12 مࣷو وَرَلْ مَ فَسْ يࣺينࣹ دِ لࣹينْ دࣴڪّࣹ فࣵتّلِ لࣷولُ، دࣷونْتࣹ خَسْ نگࣹينْ ڪࣷو خَمْ، بَ تَخْ نگࣹينْ تَخَوْ بُ دࣴگࣴرْ ݖِ دࣴگّ گِ لࣹينْ نُيُسِ؞ ‏13 لِ جَادُ ݖِ سَمَ گِسْ-گِسْ مࣷو دِ، لِ فࣹيڪْ مَيْ مࣵڪَّانࣷو اَوْ سُوخْ، دِنَا لࣹينْ ڪࣷ فࣵتّلِ، نگِرْ سࣷݧّ لࣹينْ؞ ‏14 ندَخْ دَمَا لࣹيرْ، نِ مَ ڪࣷ سُنُ بࣷرࣷومْ عِيسَىٰ اَلْمَسِ لࣹيرَلَلࣹ، بࣺسْ بِ مَيْ سُمِّ سَمَ مࣵڪَّانُ سُوخْ بِي دࣴگْمَلْ نَ؞ ‏15 وَايࣹ دِنَا دࣹفْ سَمَ ڪࣹمْ-ڪࣵتَّنْ، بَ فُ نگࣹينْ تࣷلُّ، دِ فࣵتّلِڪُ مبِرْ يࣷويُ، گَنَّاوْ بُ مَ دࣴدّࣷو؞
‏16 دُ اَيْ لࣺيبْ يُ خࣹلْ مُ خَرࣹݧْ فࣹنتّ لَنُ لࣹينْ خَمَلࣹ وࣷونْ سُنُ مَنࣷورࣹيْ بࣷرࣷومْ عِيسَىٰ اَلْمَسِ، اَڪْ نَ مُيْ دِڪّࣹ، وَايࣹ نࣷو تࣹيوࣹ مࣵگَّايَمْ؞ ‏17 ندَخْ ڪَتْ مࣷو نَنگࣷو ݖِ يࣵلَّ بَايْ بِ، تࣹرَانگَ اَڪْ دَرَجَ، بَ ڪࣷ بَاتْ بَ جِبَلࣹي، بَوࣷو فَ بࣷرࣷومْ دَرَجَ جُ يࣺيمࣹ جَ نࣹ:‏ «ڪِي مࣷو دِ سَمَ دࣷومْ، سَمَ سࣷݒّࣹ؛ ݖِ مࣷومْ لاَ اَمࣹ بࣵنّࣹيخْ؞¤» ‏18 بَاتْ بࣷوبَ جٗگࣹ اَسَمَانْ، مࣷومْ لَنُ دࣺگّ، بَ نُ نࣹڪّࣹي اَڪْ مࣷومْ ݖَ تُندّ وُ سࣹلّ وَ؞
‏19 رَخْ ݖِ دࣷلِّ نُ نگࣹيڪْ وَخِ يࣷنࣹنتّ يِ، مُيْ وَخْ جُ وٗورْ جُ جَرَ گࣴمْ، تࣹ يࣹينَ يࣹيْ تࣹيوْلُ وَخِ يࣷنࣹنتّ يِ، نِ ݧُيْ تࣹيوْلࣷو ݖࣹيݧࣹيرُ لࣵمݒّ بُ تࣵڪّࣹ بِيرْ لࣴنْدࣴمْ، بَ ڪࣹرَ فَجَرْ دِ خَرْ، بَ بِدِّيوُ نجࣴلْ فࣹقَلْ سࣹينْ خࣷلْ؞ ‏20 وَايࣹ لِ نگࣹينْ وَرَ جࣴڪَّ خَمْ مࣷو دِ اَمُلْ ݖِ مبِندّ مِ، بࣹنّ وَخْيُ بُ يࣷنࣹنتّ بَ ڪࣷ وَخࣹ، فِرࣹي خࣹلُمْ بࣷݒَّمْ؞ ‏21 ندَخْ ڪَتْ دُ گࣹنّ ڪࣵدُّگْ وَخْيُ گُ مَسَ سࣷقِڪࣷو ݖِ ݖࣷوبَرࣹيْ نِتْ، وَايࣹ ݖِ خِيرْتَلُبْ نࣷو گُ سࣹلّ گِ لَ اَوْ نِتْ دَانْ وَخࣹ وَخْيُ بُ بَوࣷو فَ يࣵلَّ؞
