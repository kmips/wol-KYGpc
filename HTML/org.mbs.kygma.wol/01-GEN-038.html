لُ جࣴمْ ݖِ يُدَ اَڪْ تَمَرْ اَڪْ سࣹينْ اَسْڪَنْ
38
‏1 ݖَ جَمࣷنࣷ يࣷويُ يُدَ دَفَ تࣹقْلِڪࣷو اَڪْ اَيْ دࣷومِ بَايَمْ، دَلْ ݖَ وَا جُ دࣴڪّ اَدُلَمْ تࣹ تُدّ اِرَ؞ ‏2 فࣷوفَ نَگْ يُدَ گِسْ فَ سࣹنّ اَسْ ندَوْ، سُ دࣷونْ دࣷومُ ڪَنَانࣹينْ بُ تُدّ سُوَ، مُ جࣴلْ ڪࣷ سࣷخْنَ؞ ‏3 مُ اࣴمبّ، اَمْ دࣷومْ جُ گٗورْ، تُدّࣹ ڪࣷ اࣹرْ؞ ‏4 مُ دࣹلُّ اࣴمبّ، اَمْ دࣷومْ جُ گٗورْ، تُدّࣹ ڪࣷ اࣷنَنْ؞ ‏5 مُ اَمَتِ دࣷومْ جُ گٗورْ، تُدّࣹ ڪࣷ سࣹلَ؞ بَ سࣹلَ دِ جُدُّ، ݖَ ڪࣹسِبْ لَ فࣹڪّ يُدَ؞
‏6 گَنَّاوْ گِ يُدَ جࣴلَلْ تَاوَمْ اࣹرْ جَبَرْ جُ تُدّ تَمَرْ؞ ‏7 اࣹرْ، تَاوُبْ يُدَ بࣷوبࣹي، دࣷونࣷونْ ڪُ اَجِ سَخْ جِ سࣹيدࣹ ݖࣷخࣷرَمْ؛ اَجِ سَخْ جِ نَگْ رࣹيْ ڪࣷ؞ ‏8 يُدَ نࣹ اࣷنَنْ:‏ «دࣹمَلْ ݖِ سَ جَبَرُ مَگْ، دࣷنّ ڪࣷ، ندَخْ نگَ گِيرَلْ سَ مَگْ اَوْ اَسْڪَنْ؞» ‏9 تࣹيوُلْ اࣷنَنْ خَمْ نࣹ دُ دࣷونْ اَسْڪَنْ وُ مُيْ مࣷومْ، بَ تَخْ سَا يُيْ دࣴڪّࣷو جَبَرُ مَگَمْ، بَيْ وَرَ اࣵگّ رࣹڪّ، تُورْ وَسَلْ وِ ݖِ سُوفْ، نگِرْ بَݧَ گِيرَلْ مَگَمْ¤؞ ‏10 اَجِ سَخْ جِ ݧَاوْلُ لِ مُيْ دࣹفْ، بَ تَخْ مُ فَاتْ ڪࣷ، مࣷومْ اِتَمْ؞ ‏11 يُدَ دَلْدِ وَخْ تَمَرْ جَبَرُ دࣷومَمْ نࣹ ڪࣷ:‏ «دࣹمَلْ سَ ڪࣴرْ بَايْ، تࣺنجࣹجِ فَ، بَ سَمَ دࣷومْ سࣹلَ مࣵگّ؞» بࣷوبَ يُدَ دَفَ رَگَلْ سࣹلَ فَاتُ مࣷومْ اِتَمْ نِ اَيْ مَگَمْ؞ تَمَرْ نَگْ دࣹمْ، تࣷوگْ ڪࣴرْ بَايَمْ؞
‏12 بَ ݧُ دࣹمࣹي بَ مُ يࣵگّ، دࣷومُ سُوَ، سࣷخْنَسْ يُدَ فَاتُ؞ بَ دࣴجْ بَ وࣺيࣹي، يُدَ دࣹمْ فࣹڪِّ وَتْڪَتِ خَرَمْ ݖَ تِمْنَ؞ اِرَ، خَرِتَمْ بِ دࣴڪّ اَدُلَمَ نگَ اࣵندّ اَڪْ مࣷومْ؞ ‏13 ݧُ يࣺگَلْ تَمَرْ نࣹ ڪࣷ:‏ «سَ گࣷرࣷو نگِي جࣴمْ تِمْنَ، دِ وَتِ اَيْ خَرَمْ؞» ‏14 بَ لࣷولُ اَمࣹي مُ سُمِّ يࣺرࣹيْ جࣴتُنْ يَ، مُورُ، سࣵنگُ، تࣷوگْ ݖَ سࣹلࣹبࣹ يࣷونْ وَ جࣴمْ اࣹنَيِمْ ݖَ يࣷونُ تِمْنَ، ندَخْ بࣷوبَ گِسْ نَ نࣹ سࣹلَ مࣵگّ نَ تࣹ مَيُݧُ ڪࣷ ڪࣷ، مُيْ جَبَرَمْ؞ ‏15 يُدَ گِسْ ڪࣷ، يَاڪَارْ نࣹ گࣵنݖّ لَ، ندَخْ لِ مُ مُورُ¤؞ ‏16 مُ جࣵدّ، ݧࣴوْ بَ ݖَ مࣷومْ، نࣹ ڪࣷ:‏ «مَيْ مَ، مَ اࣵندّ اَڪْ يَوْ؞» فࣹڪّ نَ خَمُلْ وࣷونْ نࣹ گࣷرࣷومْ لَ؞ مُ نࣹ ڪࣷ:‏ «لࣷو مَيْ فࣹيْ، نگِرْ اࣵندّ اَڪْ مَنْ؟» ‏17 مُ نࣹ ڪࣷ:‏ «دِنَا جࣴلْ بࣴيْ وُ ندَوْ ݖَ سَمَ گࣺتّ، يٗنّࣹي لَ؞» مُ نࣹ ڪࣷ:‏ «جࣷخْ مَ لࣹنّ لࣷويْ تَيْلࣹ، بَ ڪࣹرَ نگَيْ يٗنّࣹي؞» ‏18 مُ نࣹ ڪࣷ:‏ «لُ مَ لَيْ تَيْلࣹ؟» ندَوْ سِ نࣹ ڪࣷ:‏ «خَنَا سَ جَارࣷبْ مࣵنْدَرْگَالْ، اَڪْ بُومْ، گِ مُ اࣵنْدَلْ اَڪْ يࣹتْ، وِ ݖِ سَ لࣷخࣷ؞» مُ جࣷخْ ڪࣷ نَگْ، بَ نࣷݒِّ اࣵندّ اَڪْ مࣷومْ؞ فَ لَ دࣷخࣹ اࣴمبّ؞ ‏19 ندَوْ سَ دࣹمْ يࣷونَمْ، دِنْدِ مُورَايْ بَ، سࣷلاَتْ يࣺرࣹيْ جࣴتُنْ يَ؞
‏20 بَ لࣷولُ وࣺيࣹي يُدَ يࣹبَلْ خَرِتَمْ، بَ دࣴڪّ اَدُلَمْ، مُ يٗتِ ندَوْ سَ بࣴيْ وُ ندَوْ وَ، بَ مَنَ جࣷتَاتْ لَ مُ تَيْلࣹ وࣷونْ ݖَ ندَوْ سَ؞ خَرِتْ بَ نَگْ گِسُ ڪࣷ؞ ‏21 مُ لاَجْ وَا گࣷخْ بَ، نࣹ لࣹينْ:‏ «اَنَ گࣵنݖّ بِيْ تࣷوگْ اࣹنَيِمْ ݖِ يࣷونْ وِ؟» ݧُ نࣹ ڪࣷ:‏ «فِي دࣹ اَبْ گࣵنݖّ نࣹڪُّ فِ؞» ‏22 مُ دࣹلُّ ݖَ يُدَ نࣹ ڪࣷ:‏ «گِسُمَ ڪࣷ دࣹ، تࣹ اِتْ وَا گࣷخْ بَ نࣹي نَݧُ، فࣷوفَ گࣵنݖّ نࣹڪُّ فَ؞» ‏23 يُدَ نࣹ ڪࣷ:‏ «نَنُ ڪࣷ ڪࣷ وَݖّࣹ، مُ جࣴلْ، لُ ڪࣷ مࣷيْ ݧُ رࣹي نُ، گَنَّاوْ مَا نگِ نِي يٗنّࣹي بࣴيْ وُ ندَوْ وِي، تࣹ يَوْ گِسࣷو ڪࣷ ڪَيْ؞»
‏24 بَ ݧُ ݖَ تࣹگࣹي لُ تࣷلّࣷوڪْ ݧࣹتِّ وࣹيرْ، ݧُ خَمَلْ يُدَ نࣹ ڪࣷ، تَمَرْ گࣷرࣷومْ بَ، دَ دࣷونْ جَايْ بࣷݒَّمْ، بَ جَارࣹ ݖَ اࣴمبّ؞ يُدَ نࣹ:‏ «نَݧُ ڪࣷ يٗبُّ، لَڪّ ڪࣷ، بَ مُ دࣹي؞» ‏25 نَڪَ لَݧُ ڪࣷيْ يٗبُّ، مُ يٗبَّنْتࣹ گࣷرࣷومْ بَ اَيْ ڪࣵدُّ؞ دَ ڪࣷو وَخْ نࣹ ڪࣷ:‏ «ڪِ مࣷومْ لِي مࣷو مَ اࣴمْبَلْ؞» مُ تࣹگْ ݖَ نࣹ ڪࣷ:‏ «خࣷولَلْ بُ بَاخْ تࣹ وَخْ مَ ڪُ مࣷومْ جَارࣷبْ مࣵنْدَرْگَالْ، بِ اࣵندّ اَڪْ بُومْ گِ، اَڪْ يࣹتْ وِ؞» ‏26 يُدَ خࣵمِّ يࣴفْ يَ نَگْ، نࣹ:‏ «مࣷو مَ گࣴنَ نࣹڪّ ݖِ دࣴگّ دࣹ، گَنَّاوْ مَيُمَ ڪࣷ سَمَ دࣷومْ سࣹلَ؞» گَنَّاوْ بَ لࣷولُ وࣺيࣹي، اࣵنْدَتُلْ اَڪْ مࣷومْ؞
‏27 تَمَرْ دࣹمْ، بَ وَرَ وَسِنْ، ندࣹڪࣹتࣹ اَيْ سࣹيخْ لَ اࣴمبّ؞ ‏28 بَ مُيْ مُݖّ، ڪࣹنّ ݖَ دࣷومْ يَ دَا تࣵلَّلْ لࣷخࣷ بَ، رࣹوْلِڪَتْ بَ جࣵݒّ ݖَ، تَڪّ ݖَ وࣴݧْ گُ خࣷنقّ ݖُرّ؞ مُ نࣹ:‏ «ڪِي مࣷو جࣴڪَّ گࣺنّ؞» ‏29 نَڪَ لَ دࣹلّࣷو لࣷخࣷومْ، سࣹيخَمْ جُدُّ، مُ نࣹ ڪࣷ:‏ «يَوْ فࣷو بࣴتّࣹ، بَ گࣺنّ؟» ݧُ دَلْدِ ڪࣷ تُدّࣹ ݒࣹرࣹسْ (مُ فِرِ بࣴتّ)؞ ‏30 نࣹسْ تُوتْ سࣹيخْ، بِ وࣴݧْ گِ تَڪّࣹ ݖِ لࣷخࣷ لِ، دࣷورْ ݖَا تࣷݒّ، ݧُ تُدّࣹ ڪࣷ سࣹرَ؞
