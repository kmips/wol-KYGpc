لُ جࣴمْ ݖِ بࣺسُبْ نجࣷتْلاَيْ
16
‏1 بُ لࣷولُ وࣹيسࣹي اَجِ سَخْ جِ وَخْ نَاڪْ مُوسَىٰ گَنَّاوْ بَ ݧَارِ دࣷومِ هَـٰرُونَ يُ گٗورْ يَ دࣹييࣹي، بَ ݧُ تَخَوْسࣹي فَ ڪَنَمْ اَجِ سَخْ جِ؞ ‏2 اَجِ سَخْ جِ دَفَ وَخْ مُوسَىٰ نࣹ ڪࣷ:‏ «وَخَلْ سَ مَگْ هَـٰرُونَ نࣹ ڪࣷ دُ سَا يُ نࣹڪّ لَيْ اࣵگّ ݖِ بِيرْ بࣺرَبْ بُ سࣹلّ بࣹي سࣹلّ، ݖِ گَنَّاوْ رِدࣷ بِ، دِ جَنࣷوڪْ ڪُبࣹيرُ سَرَخُ نجࣷتْلاَيْ گِ اُبْ گَالْ گِ اࣵلُّوَيْ ڪٗلّࣴرࣹ گِ دࣹنݖࣹ، لُ ڪࣷ مࣷيْ دِنَ دࣹي، ندَخْ ݖِ نِيرْ وِ تِيمْ ڪُبࣹيرُ گَالْ گِ لاَيْ فࣹيݧࣹ¤؞
‏3 «نِ هَـٰرُونَ دِ دُگّࣹ ݖِ بِيرْ بࣺرَبْ بُ سࣹلّ بࣹي سࣹلّ نِي لَ:‏ دَيْ اِنْدَالࣹ اَوْ يࣴڪّ وُ ندَوْ نگِرْ اَبْ سَرَخُ ݒٗوتُمْ بࣵڪَّارْ، اَڪُمْ ڪُويْ نگِرْ اَبْ سَرَخُ رࣹنْدِ-دٗومَلْ؞ ‏4 مبُبّ مُ سࣹلّ لَيْ سࣷلْ، مُ ݧُ رࣵبّࣹ وࣴݧْ گُ اࣴݖُّ¤ بَ سࣹوْ اَڪْ تُبࣺيِ نجِيتْلاَيْ جُ ݧُ رࣵبّࣹ وࣴݧْ گُ اࣴݖُّ بَ سࣹوْ؞ نَ تَڪّࣷو لَخَسَايْ گُ ݧُ رࣵبّࣹ وࣴݧْ گُ اࣴݖُّ بَ سࣹوْ، تࣹ ڪَالَگْ وࣴݧْ گُ اࣴݖُّ بَ سࣹوْ لَيْ تَڪّ؞ يࣺرࣹ يُ سࣹلَّ نگࣷوگُ؛ ڪࣷنْ نَگْ نَ سَنگُ بَلَ مُ ڪࣷيْ سࣷلْ؞ ‏5 نَ جࣴلࣹ ݖِ مبࣷولࣷومْ بࣵنِّ إِسْرَاءِيلْ ݧَارِ سِڪّࣹتْ، سِڪّࣹتْ يُيْ دࣷونْ سَرَخُ ݒٗوتُمْ بࣵڪَّارْ اَڪْ مࣹنّ ڪُويُمْ رࣹنْدِ-دٗومَلْ؞
‏6 «هَـٰرُونَ دَيْ سَرْخَلْ يࣴڪّ وِ، مُيْ سَرَخُ ݒٗوتُمْ بࣵڪَّارَمْ، دِ نجࣷتْلاَيَلْ بࣷݒَّمْ، مࣷوڪْ وَا ڪࣴرَمْ؞ ‏7 گَنَّاوْ لࣷولُ نَ هَـٰرُونَ اِنْدِ ݧَارِ سِڪّࣹتْ يِ، تَخَوَلْ فِ ڪَنَمْ اَجِ سَخْ جِ، ݖِ بُنتّ خَيْمَبْ ندَجࣹ مِ؞ ‏8 نَ هَـٰرُونَ تࣹگَلْ ݧَارِ سِڪّࣹتْ يِ بَنتّ، نگِرْ خَمْ بُ ݧُيْ رࣹنْدِلْ اَجِ سَخْ جِ اَڪْ بِيْ دࣷونْ نگِرْ اَسَسࣹلْ¤؞ ‏9 سُ ڪࣷ دࣹفࣹي هَـٰرُونَ اِنْدِ سِڪّࣹتْ، بِ مُورْلࣷو، ݧُ سَرْخَلَلْ ڪࣷ اَجِ سَخْ جِ، رࣹنْدِ ڪࣷ، مُيْ سَرَخُ ݒٗوتُمْ بࣵڪَّارْ؞ ‏10 وَايࣹ سِڪّࣹتْ بِ مُورْلࣷو دࣷونْ سِڪّࣹتُ اَسَسࣹلْ، دࣹيسْ ڪࣷيْ تَخَوَلْ مُيْ دُندّ فِ ڪَنَمْ اَجِ سَخْ جِ، تࣹ نِ ݧُ ڪࣷيْ جࣷتࣷو، مࣷويْ دࣹيسْ ڪࣷيْ دࣵقْ مُ دࣹمْ ݖَ مࣵنْدِݝْ مَ، نگِرْ اَسَسࣹلْ؞
‏11 «بُ هَـٰرُونَ نࣷݒّࣹي، نَ اِنْدِ يࣴڪُّ سَرَخَسْ ݒٗوتُمْ بࣵڪَّارَمْ، دࣹفْ ڪࣷ نجࣷتْلاَيَمْ مࣷوڪْ وَا ڪࣴرَمْ؞ نَ رࣹنْدِ يࣴڪُّ سَرَخَسْ ݒٗوتُمْ بࣵڪَّارَمْ، ‏12 تࣹ نَ سࣵڪّ اَنْدُ خَلْ بُ فࣹيسْ، تِبّࣹ ڪࣷ ݖَ سَرْخَلُڪَايْ بَ، فِ ڪَنَمْ اَجِ سَخْ جِ، اَڪْ ݧَارِ بَرْݖِ-لࣷخࣷيْ ݖُورَايْ لُ مࣷڪّ تࣹ خࣹيݧْ، بࣷولࣹ ڪࣷ يٗبُّ ݖَ گَنَّاوْ رِدࣷ بَ؞ ‏13 نَ دࣹفْ ݖُورَايْ لِ ݖِ تَالْ بِ، فࣷوفَ ݖَ ڪَنَمْ اَجِ سَخْ جِ، سَخَرُ ݖُورَايْ سِ دِ لࣵقْ ڪُبࣹيرُ سَرَخُ نجࣷتْلاَيْ گِ اُبْ گَالُ اࣵلُّوَيْ سࣹيدࣹ سِ، ندَخْ مُ بَݧَ دࣹي؞ ‏14 نَ سࣵڪّ ݖِ دࣹرࣹتُ يࣴڪّ وِ، وِسْ-وِسَلࣹ ڪࣷ بَارَامَمْ ݖِ ڪَوْ ڪُبࣹيرُ گَالْ گِ، ݖِ وࣹتَمْ گِ فࣺيتࣹ ݒࣹنڪُ، تࣹ اِتْ نَ وِسْ-وِسَلࣹ بَارَامَمْ دࣹرࣹتْ جِ لُ مَتْ جُرٗومْ ݧَارِ يࣷونْ ݖِ ڪَنَمْ ڪُبࣹيرُ گَالْ گِ؞ ‏15 بُ نࣷݒّࣹي نَ رࣹنْدِ سِڪّࣹتْ بِيْ سَرَخَسْ ݒٗوتُمْ بࣵڪَّارُ مبࣷولࣷو مِ، تࣹ يٗبُّ دࣹرࣹتْ جِ ݖَ گَنَّاوْ رِدࣷ بَ؞ نَ دࣹفْ دࣹرࣹتْ جِ نَ مُ دࣹفࣷونْ اَڪْ دࣹرࣹتُ يࣴڪّ وَ، وِسْ-وِسَلْ ڪࣷ ݖَ ڪَوْ ڪُبࣹيرُ گَالْ گِ اَڪْ ݖِ ڪَنَمْ ڪُبࣹيرْ گِ؞ ‏16 مُ دَلْدِ مَتَلࣹ لࣷولُ نجࣷتْلاَيَلْ بࣺرَبْ بُ سࣹلّ بࣹي سࣹلّ، تࣵگَّلࣹ ڪࣷوڪْ سࣷبࣹيْ بࣵنِّ إِسْرَاءِيلْ اَڪْ سࣹينِ تࣷوݧْ، اَڪْ لُ سࣹينْ بࣵڪَّارْ مَنَ دࣷونْ؞ نࣷونُ لَيْ دࣹفَلْ مبࣷولࣹمْ خَيْمَبْ ندَجࣹ، مِ دࣴڪّ ݖِ سࣹينْ بِيرْ ݖِ سࣹينْ بِيرِ سࣷبࣹ؞ ‏17 دُ ڪࣹنّ ڪُ سَݧَ نࣹڪّ ݖِ بِيرْ خَيْمَبْ ندَجࣹ مِ، بُ هَـٰرُونَ دُگّࣹي ݖِ بِيرْ بࣺرَبْ بُ سࣹلّ بࣹي سࣹلّ، دِ مَتَلْ اَگْ نجࣷتْلاَيْ، بَ ڪࣹرࣷوگْ مُيْ گࣺنّ، گَنَّاوْ بُ جࣷتࣷو مࣷومْ ݖِ بࣷݒَّمْ، مࣷوڪْ وَا ڪࣴرَمْ، بَ جࣷتَالࣹ مبࣷولࣷومْ بࣵنِّ إِسْرَاءِيلْ مࣺݒّ؞
‏18 «بُ سࣷتَّلࣹي، نَ گࣺنّ جࣴمْ ݖَ سَرْخَلُڪَايْ بَ ݖَ ڪَنَمْ اَجِ سَخْ جِ، بَ نࣷݒِّ دࣹفَلْ سَرْخَلُڪَايْ بِ نجࣷتْلاَيَلْ بࣷݒَّمْ؞ دَيْ سࣵڪّ ݖِ دࣹرࣹتُ يࣴڪّ وِ اَڪْ جُ سِڪّࣹتْ بِ، تَقَلْ بࣺجّࣺنِ سَرْخَلُڪَايْ بِ يࣺݒّ؞ ‏19 تࣹ اِتَمْ دَيْ وِسْ-وِسَلࣹ بَارَامَمْ دࣹرࣹتْ جِ لُ مَتْ جُرٗومْ ݧَارِ يࣷونْ ݖِ ڪَوْ سَرْخَلُڪَايْ بِ، سࣹتَلْ ڪࣷ، سࣹلَّلْ ڪࣷ، مُ تࣵگّࣷوڪْ سࣷبࣹيْ بࣵنِّ إِسْرَاءِيلْ؞
‏20 «بُ هَـٰرُونَ مَتَلࣹي نجࣷتْلاَيَلْ بࣺرَبْ بُ سࣹلّ بࣹي سࣹلّ اَڪْ خَيْمَبْ ندَجࣹ مَ اَڪْ سَرْخَلُڪَايْ بَ، نَ اِنْدِ سِڪّࣹتْ بِيْ دُندّ؞ ‏21 هَـٰرُونَ دَيْ تࣹگْ اَيْ لࣷخࣷومْ ݖِ ڪَوْ بࣷݒُّ سِڪّࣹتْ بِيْ دُندّ تࣹ تُدّ ݖِ ڪَوَمْ مبࣷولࣹمْ ݧَاوْتࣺيفِ بࣵنِّ إِسْرَاءِيلْ اَڪْ سࣹينِ تࣷوݧْ، اَڪْ سࣹينِ بࣵڪَّارْ يࣺݒّ، بࣷولࣹ ڪࣷ يࣹنْ سِڪّࣹتْ بِ، تࣹگْ ڪࣷ ݖِ لࣷخࣷلْ نِتْ ڪُ ݧُ يٗنِّ، مُ دࣵقْ ڪࣷ ݖَ مࣵنْدِݝْ مَ؞ ‏22 سِڪّࣹتْ بِ دَيْ يࣹنُ سࣹينِ ݧَاوْتࣺيفْ يࣺݒّ، يٗبَّالࣹ ݖَ ندࣵندّ-فࣷيْفࣷيْ گَ، ݧُ وَݖّ ڪࣷ ݖَ مࣵنْدِݝْ مَ؞
‏23 «نَ هَـٰرُونَ دࣹلُّ ݖَ خَيْمَبْ ندَجࣹ مَ، سُمِّ يࣺرࣹ يَ ݧُ رࣵبّࣹ وࣴݧْ گُ اࣴݖُّ بَ سࣹوْ، دِ يَ مُ سࣷلࣷونْ بَلَ مࣷو دُگّ ݖَ بࣺرَبْ بُ سࣹلّ بَا سࣹلّ، تࣹ نَ فَ وَݖّ يࣺرࣹ يَ؞ ‏24 بُ لࣷولُ وࣹيسࣹي مُ سَنگࣷو فُ سࣹلّ، سࣷلْ يࣺرࣹيْ بࣷݒَّمْ، گࣺنّ، جࣷخࣹيلْ بࣷݒَّمْ اَبْ سَرَخُ رࣹنْدِ-دٗومَلْ، جࣷخࣹيلْ مبࣷولࣷو مِ سࣹينْ سَرَخُ رࣹنْدِ-دٗومَلْ نگِرْ نجࣷتْلاَيَلْ بࣷݒَّمْ اَڪْ نجࣷتْلاَيَلْ مبࣷولࣷو مِ؞ ‏25 نࣹبّࣷنُ جُرْ گِيْ سَرَخُ ݒٗوتُمْ بࣵڪَّارْ، نَ ڪࣷ لَڪّ ݖِ ڪَوْ سَرْخَلُڪَايْ بِ؞
‏26 «ڪَ وَݖِّ وࣷونْ سِڪّࣹتُ اَسَسࣹلْ، نَ فٗوتْ اَيْ يࣺرࣹيمْ، سَنگُ، دࣷورَ مَنَ دُگَّاتْ ݖِ دَلْ بِ؞ ‏27 مُيْ يࣴڪّ وِيْ سَرَخُ ݒٗوتُمْ بࣵڪَّارْ، مُيْ سِڪّࣹتْ بِيْ سَرَخُ ݒٗوتُمْ بࣵڪَّارْ تࣹ ݧُ يٗبُّ وࣷونْ سࣹينْ دࣹرࣹتْ ݖَ بِيرْ بࣺرَبْ بُ سࣹلّ بَا سࣹلّ نگِرْ دࣹفࣹ ڪࣷ نجࣷتْلاَيْ، لࣺݒّ لَݧُيْ گࣺنّࣹ دَلْ بِ؞ دࣹرْ يَاڪْ يࣵݒّ وَاڪْ سࣺبّرِيتْ مَ دࣹيسْ ڪࣷيْ بࣷولࣹ لَڪّ، بَ مُ دِبْ دٗومْ؞ ‏28 ڪِ ڪࣷ لَڪّ بُ نࣷݒّࣹي، نَ فٗوتْ اَيْ يࣺرࣹيمْ، سَنگُ، دࣷورَ مَنَ دُگَّاتْ ݖِ دَلْ بِ؞
‏29 «تࣹ اِتَمْ دࣷگَلَ نگِي، دِ لُ لࣹينْ وَرْ فࣵوّ؞ بࣺسُبْ فُڪِّ فَنْ ݖِ جُرٗومْ ݧَارࣹيلُ وࣹيرُ اَتْ مِ نَنگࣹينْ ݖِ تࣷرࣷخْلُ تࣹ دُ لࣹنّ لُ نگࣹينْ ݖِ مَنَ لِگّࣺييْ، دُ نجُدُّ-جِ-رࣺيوْ دُ دࣷخَنْدࣺيمْ بِ ݖِ سࣹينْ بِيرْ؛ ‏30 ندَخْتࣹ ݖِ بࣺسْ بࣷوبُ لَݧُ لࣹينْ دِ جࣷخࣹيلْ نجࣷتْلاَيْ نگِرْ فٗوتْ سࣹينْ مبࣷولࣹمْ بࣵڪَّارْ، بَ نگࣹينْ سࣹتْ فِ ڪَنَمْ اَجِ سَخْ جِ؞ ‏31 نَ دࣷونْ سࣹينْ بࣺسْ بُ نگࣹينْ دِ نࣷݒّلُ دࣷݝّ، نگࣹينْ وَرْ ݖࣹي تࣷرࣷخْلُ؞ دࣷگَلْ لَ بُ سَخْ دࣵڪّ؞
‏32 «سَرْخَلْڪَتْ بِ ݧُ فَلْ نَگْ، سࣷلَلْ ڪࣷ، جࣷخْ ڪࣷ سَݧْ-سَݧُ سَرْخَلْڪَتْ بُ مَگْ، مُ وُوتُ بَايَمْ، نَ دࣹفْ سَرَخُ نجࣷتْلاَيْ گِ؞ نَ سࣷلْ يࣺرࣹ يُ سࣹلّ يࣷويُ دِ يِ ݧُ رࣵبّࣹ وࣴݧْ گُ اࣴݖُّ بَ سࣹوْ، ‏33 تࣹ اِتْ نَ دࣹفْ نجࣷتْلاَيَلْ بࣺرَبْ بُ سࣹلّ بࣹي سࣹلّ اَڪْ خَيْمَبْ ندَجࣹ مِ اَڪْ سَرْخَلُڪَايْ بِ، بࣷولࣹ ݖِ دࣹفَلْ سَرْخَلْڪَتْ يِ سࣹينْ نجࣷتْلاَيْ، ݧࣷوڪْ مبࣷولࣷومْ إِسْرَاءِيلْ مࣺݒّ؞
‏34 «دࣷگَلَ نگࣷوگُ، دِ لُ لࣹينْ وَرْ فࣵوّ، تࣹ مࣷويْ ݧُيْ جࣷخࣹيلْ بࣵنِّ إِسْرَاءِيلْ سࣹينْ نجࣷتْلاَيْ بࣹنّ يࣷونْ ݖِمْ اَتْ نگِرْ سࣹينْ بࣵڪَّارْ يࣺݒّ؞»
ݧُ دَلْدِ دࣹفْ لِ اَجِ سَخْ جِ سَنْتࣷونْ مُوسَىٰ؞
