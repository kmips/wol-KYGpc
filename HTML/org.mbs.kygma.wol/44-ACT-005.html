مبُگَلْ دَبْ نَ اَنَݧَسْ اَڪْ سَفِرَ
5
‏1 نَڪَ نࣷونُ ڪُ ݧُيْ وَخْ اَنَݧَسْ، تࣹ ڪُ ݧُيْ وَخْ سَفِرَ دِ جَبَرَمْ، دَلْدِ جَايْ اَوْ لࣵݖُّ سُوفْ؞ ‏2 مُ يࣺگࣷوڪْ جَبَرَمْ، دَلْدِ لࣵقْ لࣹنّ ݖَ نجࣺگْ گَ، بَ نࣷݒِّ اِنْدِ لَ ݖَ دࣹسْ، تࣹگْ ڪࣷ ݖَ ڪَنَمْ ندَوْ يَ؞
‏3 بَ لࣷولُ اَمࣹي ݒِيࣹيرْ نࣹ ڪࣷ:‏ «اَنَݧَسْ، لُ وَرَلْ سࣹيْتَانࣹ دِ فࣹيسْ سَبْ خࣷلْ، بَ نگَيْ وࣷرْ نࣷو گُ سࣹلّ گِ، دِ لࣵقْ لࣹنّ ݖِ نجࣺگُ تࣷولْ بِ؟ ‏4 بَ مُ جَرَگُلْ، دُ يَا دࣷونْ بࣷرࣷومْ؟ گَنَّاوْ بَ مُ جَرࣹي اِتْ، خَنَا دُ نجࣺگْ گَا نگِ سَ لࣷخࣷ بَ تࣹيْ؟ اَنَ نࣷو مَنَ خَجَلࣹ مِي مࣺبࣺتْ ݖِ سَ خࣷلْ؟ دُ نِتْ نگَ وࣷرْ دࣹ، وَايࣹ يࣵلَّ نگَ وࣷرْ؞»
‏5 نَڪَ لَ اَنَݧَسْ دࣺگّ ڪࣵدُّ يࣷويُ، دَلْدِ دَانُ، دࣹي؞ تِيتَانگࣹ لُ رࣺيْ نَگْ دِڪَّلْ ݧَ ڪࣷ دࣺگّ ݧࣺݒّ؞ ‏6 بَ لࣷولُ اَمࣹي خَلࣹ يُ گٗورْ يَ دِڪّ، سࣵنگّ ڪࣷ، يٗبُّ سُولِ؞
‏7 ݧُ تࣹگْ ݖَ لُ وَرَ تࣷلّࣷوڪْ ݧࣹتِّ وَخْتُ، جَبَرَمْ دُگّسِ، تࣹ يࣺگُلْ لَ خࣹوْ؞ ‏8 ݒِيࣹيرْ نࣹ ڪࣷ:‏ «وَخْ مَ، نࣵنگَمْ نگࣹينْ جَايࣹ تࣷولْ بࣹي؟» مُ نࣹ ڪࣷ:‏ «وَاوْ، نࣵنگَمْ لَ؞»
‏9 ݖِ ڪَوْ لࣷولُ ݒِيࣹيرْ نࣹ ڪࣷ:‏ «اَنَ لُ نگࣹينْ دِ مࣵنڪࣷو، دِ نَتُّ نࣷووُگْ بࣷرࣷومْ بِ؟ دࣺگْلُلْ، تࣵنڪّ يِ سُولِ وࣷونْ سَ جࣴڪّࣴرَ نگࣷوگُ ݖِ بُنتّ بِ دِ لَ يٗبُّسِ يَوْ اِتْ؞» ‏10 ݖَ سَا سَ ندَوْ سَ دَانُ ݖَيْ تࣵنڪَمْ، دࣹي؞ خَلࣹ يُ گٗورْ يِ دُگّسِ، فࣹڪّ ڪࣷ مُ دࣹي؛ ݧُ يٗبُّ ڪࣷ، سُولْ ڪࣷ ݖَ وࣹتُ جࣴڪّࣴرَمْ؞ ‏11 بَ لࣷولُ اَمࣹي تِيتَانگࣹ جُ رࣺيَ جࣵݒّ مبࣷولࣷومْ گࣴمْڪَتْ ݧࣺݒّ، اَڪْ ݧَ ڪࣷ دࣺگّ ݧࣺݒّ؞
ندَوِ اَلْمَسِ فَجْ نَݧُ ݧُ بَرࣹ
‏12 ݖِ ڪَوْ لࣷولُ اَيْ فِرْندࣹ اَڪْ ڪࣺيمَانْ يُ بَرࣹ دِ سࣷتّࣹي ݖِ لࣷخࣷلْ ندَوْ يِ، ݖِ بِيرْ اَسْڪَنْ وِ؞ گࣴمْڪَتْ ݧِ نَگْ دِ بࣷڪّ دَجࣹ ݧࣷومْ ݧࣺݒّ فَ ݧُيْ وَخْ مبَارُمْ سُلࣹيْمَانْ، ݖَ بِيرْ ڪࣴرْ يࣵلَّ گَ، ‏13 وَايࣹ ڪࣹنّ ݖَ ݧَ ݖَ دࣹسْ ݧࣹمࣹوُلَ جَخَسࣷو اَڪْ ݧࣷومْ؞ تࣹيوُلْ اَسْڪَنْ وَ نَوْلࣷو لࣹينْ لࣷولْ؞ ‏14 گࣴمْڪَتِ سَنگّ بِ نَگْ دِ گࣴنَ بَرࣹ، ݧُيْ وࣵلِّسِ، بَ دࣷونْ مبࣷولࣷو مُ مَگْ، گٗورْ اَڪْ جِگࣺينْ؞ ‏15 نِتْ ݧِ دࣹمْ بَيْ گࣺنّࣹ جَرَگْ جِ ݖِ مبࣹدّ يِ، تࣹگْ لࣹينْ ݖِ اَيْ لَلْتُ اَڪِ بَسَݝْ، نگِرْ بُ فَ ݒِيࣹيرْ جَارࣹي، لُ بࣷنْ بࣷنْ تَڪّندࣹيرَمْ مَنَ دَلْ ݧࣹنّ ݖِ ݧࣷومْ؞ ‏16 ݖِ بِيرْ لࣷولُ نِتْ ݧُ بَرࣹ بࣵبّࣹيڪࣷو ݖَ دࣴڪّ يَ سࣺقْ يࣹرُسَلࣹمْ، اِنْدِ اَيْ جَرَگْ اَڪْ ݧُ رَبْ يُ بࣷنْ سࣷنَلْ؛ ݧࣷومْ ݧࣺݒّ دَلْدِ وࣺرْ؞
فِتْنَالْ نَݧُ ندَوْ يِ
‏17 بَ مُ ڪࣷ دࣹفࣹي خࣷلُ ڪَݧَانْ دُگّ سَرْخَلْڪَتْ بُ مَگْ بَ، اَڪْ مبࣷولࣹمْ ݧَ ڪࣷ دَرْ، تࣹ دِ وَا نگࣺرُمْ سَدُسࣹنْ ݧَ؞ ‏18 ݧُ تࣹگْ ندَوْ يَ لࣷخࣷ، تࣴجْ لࣹينْ ݖَ ڪَسࣷ بَ؞ ‏19 ݖَ گُدِّ گَ نَگْ مَلاَڪَمْ بࣷرࣷومْ بِ اُبِّ بُنْتِ ڪَسࣷ بَ، گࣺنّࣹ لࣹينْ، نࣹ لࣹينْ:‏ ‏20 «دࣹمْلࣹينْ تَخَوْ ݖَ دِگّ ڪࣴرْ يࣵلَّ گَ، نگࣹينْ خَمَلْ اَسْڪَنْ وِ مبࣷولࣹمْ لُ جࣴمْ ݖِ دُندّ گِي؞» ‏21 گَنَّاوْ بَ ݧُ دࣺگّࣹي لࣷولُ، بَ بࣴتْ سࣹتْ، ݧُ دُگّ ݖَ ڪࣴرْ يࣵلَّ گَ، دِ وَارࣹ؞
ݖِ بِيرْ لࣷولُ سَرْخَلْڪَتْ بُ مَگْ بَ اَڪْ ݧَ ڪࣷ دَرْ، دِڪّ، وࣷولُ ڪُرࣺلُ اࣵتّࣹڪَتْ يَ، مبࣷولࣹمْ ڪُرࣺلُ نجِيتِ بࣵنِّ إِسْرَاءِيلْ؛ ݧُ يٗنّࣹي، نگِرْ جࣴلِ ندَوْ يَ ݖَ ڪَسࣷ بَ؞ ‏22 دَگْ يَ نَگْ دِڪّ، فࣹڪُّݧُ لࣹينْ ݖَ ڪَسࣷ بَ، ݧُ دࣹلُّ يࣺگْلࣹجِ ڪࣷ، ‏23 نࣹ:‏ «دَنࣷو فࣹڪّ ڪَسࣷ بَ تࣴجࣹ رࣵݒّ، وَتُّڪَتْ يَ تَخَوْ ݖَ بُنتّ يَ؞ نُ تِجِّ، فࣹڪُّنُ ڪࣹنّ ݖَ بِيرْ!‏»
‏24 جَوْرِݧُ ڪࣴرْ يࣵلَّ گَ اَڪْ سَرْخَلْڪَتْ يُ مَگْ يَ دࣺگّ لࣷولُ، دَلْدِ جَاخْلࣹ لࣷولْ ݖِ ندَوْ يِ، دِ خَلاَتْ نُ مبِرْ مِيْ مُجّࣹ؞
‏25 مُ اَمْ ڪُ دِڪّ، نࣹ لࣹينْ:‏ «ݧَ نگࣹينْ تࣴجࣷونْ ڪَسࣷ دࣺيَ نگَ نࣷونُ ݖَ ڪࣴرْ يࣵلَّ گَ، دِ وَارࣹ؞» ‏26 ݖِ ڪَوْ لࣷولُ جَوْرِݧُ ڪࣴرْ يࣵلَّ گَ اَڪْ دَگْ يَ جࣴلِ ندَوْ يَ ندࣵنڪّ، بَݧْ لࣹينَ جَايْ دࣷولࣹ، نگِرْ رَگَلْ نِتْ ݧِ سࣵنِّ لࣹينْ اَيْ دࣷجْ؞
‏27 ݧُ اِنْدِ لࣹينْ، تَخَوَلْ لࣹينْ ݖَ ڪَنَمْ ڪُرࣺلُ اࣵتّࣹڪَتْ يَ، سَرْخَلْڪَتْ بُ مَگْ بَ لاَجْ لࣹينْ، ‏28 نࣹ:‏ «خَنَا آيࣹوُنُ لࣹينْ وࣷونْ نࣹ بُلࣹينْ وَارࣹيتِ مُڪّ ݖِ وࣷووُ تُرْ؟ يࣹينَ نگِ نِي اُبَلࣹ يࣹرُسَلࣹمْ اَڪْ سࣹينْ وَارࣹ، نگࣹينْ نَرْ نُ ݖࣹي گࣵدُّلࣷو بَڪَّنُ نِتْ ڪِي!‏»
‏29 ݒِيࣹيرْ اَڪْ ندَوْ يَ ݖَ دࣹسْ نَگْ نࣹ لࣹينْ:‏ «يࣵلَّ لࣹيسْ وَرَ دࣺگَّلْ، وَايࣹ دُ نِتْ؞ ‏30 سُنُ يࣵلَّيْ مَامْ مࣷو دࣹڪَّلْ عِيسَىٰ، مِ نگࣹينْ وࣺڪّ ݖِ بَنتّ، بٗومْ ڪࣷ؞ ‏31 مࣷومْ لَ يࣵلَّ يࣺڪَّتِ ݖِ وࣹتُ ندِجࣷورَمْ، دࣹفْ ڪࣷ مُيْ نجِيتْ، دِ مُسَلْڪَتْ، نگِرْ مَيْ بࣵنِّ إِسْرَاءِيلْ اَگْ تُوبࣹيلْ اَڪْ نجࣺگَّلُگْ بࣵڪَّارْ؞ ‏32 نُنْ نࣷو سࣹيدࣹ لࣷولُ، نࣷوڪْ نࣷو گُ سࣹلّ، گِ يࣵلَّ جَگْلࣹيلْ ݧِ ڪࣷ دࣺگَّلْ؞»
‏33 بَ ݧُ دࣺگّࣹي لࣷولُ، دَݧࣷو مࣹرْ بَ فࣹيسْ، نَرْ لࣹينَ رࣹيْ؞ ‏34 اَبْ فَرِسࣹنْ بُ ݧُيْ وَخْ گَمَلْيࣹلْ، دِ جࣵنگّلࣹڪَتُ يࣷونُ مُوسَىٰ بُ ݧࣺݒّ نَوْلࣷو، دَلْدِ تَخَوْ ݖَ دِگّ ڪُرࣺلُ اࣵتّࣹڪَتْ يَ؞ مُ جࣷخࣹ ندِگَلْ نگِرْ ݧُ گࣺنّࣹ لࣹينْ اَبْ دِيرْ؞ ‏35 بَ ݧُ گࣺنّࣹي، مُ نࣹ وَا ڪُرࣺلْ گَ:‏ «يࣹينْ بࣷڪِّ إِسْرَاءِيلْ، مࣷيْتُلࣹينْ بُ بَاخْ لِ نگࣹينْ دِ دࣹفْ اَڪْ نِتْ ݧِي؞ ‏36 ندَخْتࣹ بُ يࣵگُّلْ لَ تࣴدَسْ جٗگْ، دِ جَايْ دَرَجَ، بَ لُ وَرَ تࣷلّࣷوڪْ ݧࣹينْتِ تࣺيمࣺيرِ نِتْ فَرْ اَڪْ مࣷومْ؞ ݧُ رࣹيْ ڪࣷ، ݧَ ڪࣷ تࣷݒّࣷونْ ݧࣺݒّ تَسَارࣷو، سࣹينْ ݒࣹخࣹ مُجّࣹوُلْ فࣹنّ؞ ‏37 گَنَّاوَمْ اِتْ يُدَ مُ گَلِلࣹ جٗگْ، ݖَ جَنتّ يَ ݧُ دࣷونْ لِمْ وَا رࣺيوْ مِ؛ مُ يٗبَّالࣹ مبࣷولࣷو مُ ڪࣷ تࣷݒّ؞ مࣷومْ اِتَمْ دࣹي، ݧَ ڪࣷ تࣷݒّࣷونْ ݧࣺݒّ تَسَارࣷو؞ ‏38 لࣺيگِ نَگْ، دَمَ نࣹ، مَانْدُلࣹينْ ݖِ ݧِي، تࣹ بَ لࣹينْ، نگِرْ سُ سࣹينْ مࣺبࣺتْ مبَا سࣹينْ جࣴفْ بَوࣷو ݖِ نِتْ، دَيْ يࣵقُ رࣹڪّ؞ ‏39 وَايࣹ سُ بَوࣷو فَ يࣵلَّ نَگْ، دُنگࣹينْ ڪࣷ مَنَ فَنقّ، مࣷيْتُلࣹينَ جࣵنڪࣷونْتࣹ اَڪْ يࣵلَّ؞»
ݧُ دَلْدِ فࣹڪّسِ گَمَلْيࣹلْ، ‏40 وࣷووَاتْ ندَوْ يَ؞ بَ لࣷولُ اَمࣹي ݧُ دٗورْ لࣹينْ، آيࣹ لࣹينْ ݧُ وَخَتِ لࣹنّ ݖِ تُرُ عِيسَىٰ، دࣷورْ لࣹينَ يِوِ؞
‏41 ندَوْ يَ نَگْ بࣵيِّڪࣷو ݖَ ڪُرࣺلُ اࣵتّࣹڪَتْ يَ، دِ بࣺگْ، گَنَّاوْ ݧࣷومْ لَݧُ جࣵݒّ نࣹ ݧࣷو يࣹلّࣷو تࣷرࣷخْتࣹ گَ ݧُ لࣹينْ تࣹگْ ندَخْ تُرْ وِ¤؞ ‏42 ݖِ ڪَوْ لࣷولُ بࣺسْ بُ نࣹڪَّانْ، مُيْ ݖَ ڪࣴرْ يࣵلَّ، دِ ݖَ بِيرْ ڪࣴرْ يَ، نࣷݒِّوُݧࣷو جࣵنگّلࣹ اَڪَ خَمْلࣹ خِبَارُ جࣵمّ بِ يࣺينࣹ نࣹ عِيسَىٰ مࣷويْ اَلْمَسِ؞
