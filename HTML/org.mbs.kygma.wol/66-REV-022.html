وَلُمْ دُندّ مِ فࣹيݧْ نَ
22
‏1 ݖِ ڪَوْ لࣷولُ مَلاَڪَ مِ وࣷنْ مَ دࣹخُ ندࣷخُمْ دُندّ مُ لࣹيرْ نࣵݧّ نِ دࣷجُ ڪِرِسْتَالْ، ندࣷخْ مَ وَلࣹ ݖَ نگࣵنگُنࣹمْ يࣵلاَّڪْ گࣵتّ بَ، ‏2 دِ تُورُ ݖِ دِگّ ݒࣺنݖُ دࣴڪّ بِ؞ ݖِ ݧَارِ تࣵڪِّ دࣹخْ گِ، گَرَبُ دُنْدَ نگِ ݖِ، دِ مࣹݧّ فُڪِّ يࣷونْ اَڪْ ݧَارْ، وࣹيرْ وُ نࣹڪّ بࣹنّ يࣷونْ، خࣷبْ يِ دِ فَجْ اَسْڪَنِ اࣵدِّنَ؞ ‏3 اَلْڪࣵنْدࣹ نَگْ دُ فَ اَمَتِ، تࣹ نگࣵنگُنࣹ مُ يࣵلاَّڪْ گࣵتّ بِ ݖِ بِيرْ دࣴڪّ بِ لَيْ نࣹڪّ، جَامُ يࣵلَّ يِ دِ ڪࣷ سُجّٗوتَلْ، ‏4 دِ گِسْ ڪَنَمَمْ، تُرَمْ بِنْدࣹ ݖِ سࣹينْ جࣴ؞ ‏5 گُدِّ دُ اَمَتِ، تࣹ دࣹيسُلْ سࣷخْلاَتِ لࣵمݒّ مبَا نَاجْ، ندَخْ بࣷرࣷومْ بِ يࣵلَّ مࣷو لࣹينْ دِ لࣹيرَلْ؞ تࣹ دَݧُيْ نگُورُ بَ فࣵوّ؞
‏6 بَ لࣷولُ اَمࣹي مَلاَڪَ مِ نࣹ مَ:‏ «لِي وَخْ جُ وٗورْ لَ تࣹ دࣴگُّ؞ بࣷرࣷومْ بِ يࣵلَّ مِيْ خࣹلَلْ يࣷنࣹنتّ يِ، مࣷو يࣹبَلْ مَلاَڪَامْ، نگِرْ مُ وࣷنْ جَامَمْ ݧِ مبِرْ يِ دࣴگْمَلْ تࣹ دُ جَاسْ؞»
عِيسَىٰ اَلْمَسِ دِنَ دࣺلْسِ
‏7 «مَا نگِي دِ ݧࣴوْ لࣺيگِ؞ ندࣷڪّلࣹي يَوْ مِيْ سࣵمّ ڪࣵدُّيْ وَخْيُيْ تࣺيرࣹ بِي؞»
‏8 مَنْ يࣷوَانْ مَا دࣺگّ مبِرْ يِي، گِسْ ڪࣷ؞ بَ مَ ڪࣷ دࣺگّࣹي، گِسْ ڪࣷ، دَمَ نࣹ گُرُبْ سُڪّ ݖِ تࣵنڪِ مَلاَڪَ، مِ مَ ڪࣷ وࣷنْ، نَرْ ڪࣷو سُجّٗوتَلْ؞ ‏9 مُ نࣹ مَ:‏ «اࣴݒّ نَا دࣹفْ، سَ مبࣷڪُّمْ جَامْ دࣷݝّ لاَ، نِ يَاڪْ سَ بࣷڪِّ يࣷنࣹنتّ يࣹيڪْ ݧِيْ سࣵمّ ڪࣵدُّيْ تࣺيرࣹ بِي؞ سُجّٗوتَلَلْ يࣵلَّ؞» ‏10 مُ نࣹيتِ مَ:‏ «بُلْ لࣵقْ ڪࣵدُّيْ وَخْيُيْ تࣺيرࣹ بِي، ندَخْ وَخْتُ وِ جࣹگࣹ نَ؞ ‏11 ڪُ جُبَدِ، نَ سَخࣷو نجُبَدِ؛ ڪُ يࣵقُ، سَخࣷو يࣵقُتࣹ؛ ڪُ جُبْ، وࣺيࣹ دࣹفْ نجُبْ؛ ڪُ سࣹلّ اِتْ، وࣺيࣹ سࣹلّ؞»
‏12 «مَا نگِي دِ ݧࣴوْ لࣺيگِ، اࣵندّ اَڪْ سَمَ ݒࣹيْ، نگِرْ دࣹلّࣷو ڪُ نࣹڪّ جࣴفَمْ؞ ‏13 مَنْ مَايْ اَلْفَ دِ اࣷمࣹگَ؛ مَا جࣴڪّ، مَا مُجّ، مَايْ نجࣵلْبࣺينْ، مَايْ مُجْ؞
‏14 «ندࣷڪّلࣹي يࣹينْ ݧِ فٗوتْ سࣹينِ يࣺرࣹ، نگِرْ سَݧَ لࣹڪّ ݖِ گَرَبُ دُندّ گِ، جࣵلّࣹ بُنتّ يِ، دُگّ ݖِ بِيرْ دࣴڪّ بِ؞ ‏15 ݖِ بِتِ، خَجْ يَا فَ نࣹڪّ اَڪْ خࣴرࣴمْڪَتْ يِ اَڪْ مࣷيْڪَتْ يِ اَڪْ بٗومْڪَتْ يِ اَڪْ بࣷڪَّالࣹڪَتْ يِ، اَڪْ ڪࣺݒّ ڪُ سࣷݒّ فࣹنْ تࣹ دِ ڪࣷ جࣴفࣹ؞
‏16 «مَنْ عِيسَىٰ مَا يٗنِّ سَمَ مَلاَڪَ، نگِرْ مُ سࣹيدࣹ مبِرْ يِي ݖِ يࣹينْ، ݧࣹيلْ مبࣷولࣷويْ گࣴمْڪَتْ ݧِ؞ مَنْ مَايْ نجࣹبِّتُ دَاوُدَ، دِ ڪُوتَايَمْ، مَا دِ بِدِّيوُ نجࣴلْ بِ نࣹ رࣵݧّ؞»
‏17 نࣷو گِ اَڪْ سࣺيتْ بِ نࣹيݧُ:‏ «ݧࣴوَلْ!‏»
ڪُ ݖِ دࣺگّ اِتْ، نࣹيلْ:‏ «ݧࣴوَلْ!‏»
ڪُ مَرْ، ݧࣴوَلْ؛ ڪُ بࣴگّ ݖِ ندࣷخُمْ دُندّ مِ، جࣴلَلْ، ݧُ مَيْ لَ؞
‏18 مَنْ يࣷوَانْ، سࣹيدࣹ نَا ݧࣹيلْ ڪࣺݒّ ڪُ دࣺگّ ڪࣵدُّيْ وَخْيُ يُ تࣺيرࣹ بِي، نࣹ:‏ ڪࣺݒّ ڪُ ݖِ يࣷڪّ لࣹنّ، يࣵلَّ دِنَ لَ يࣷڪَّلْ مبُگَلْ، يِ ݧُ بِندّ ݖِ تࣺيرࣹ بِي؞ ‏19 تࣹ ڪُ وࣵݧِّ لࣹنّ ݖِ ڪࣵدُّيْ وَخْيُ يُ تࣺيرࣹ بِي، سَ وࣵلّ لَ يࣵلَّ دِ وࣵݧّࣹي ݖِ يࣷويِي ݧُ بِندّ ݖِ تࣺيرࣹ بِ، مُيْ گَرَبُ دُندّ گَاڪْ دࣴڪّ بُ سࣹلّ بَ؞
‏20 ڪِيْ سࣹيدࣹ مبِرْ يِي نࣹي:‏ «وَاوَاوْ، مَا نگِ ݧࣴوْ لࣺيگِ؞»
اَمِينْ، سَنگّ عِيسَىٰ، ݧࣴوَلْ!‏
‏21 يَلْ نَ يِوُ سَنگّ عِيسَىٰ اࣵندّ اَڪْ ݧࣺݒّ؞
