اَجِ سَخْ جِ رࣴدّلُ نَ دِگُ رࣺيوْ مِ
34
‏1 اَجِ سَخْ جِ وَخَتِ مُوسَىٰ، نࣹ ڪࣷ:‏ ‏2 «سَنْتَلْ بࣵنِّ إِسْرَاءِيلْ، نگَ نࣹ لࣹينْ بُ نگࣹينْ دُگّࣹي رࣺيوُمْ ڪَنَانْ، لِي مࣷو دِ رࣺيوْ مَ نگࣹينْ دِ مُورْلࣷو، مُيْ سࣹينُبْ ݖࣺرْ؞ رࣺيوُمْ ڪَنَانَ نگِ، اَڪْ ڪࣹمِ سُوفَمْ:‏
‏3 «سࣹينْ ڪࣹمُ بࣴجْ-سَالُمَ نگَيْ دَلࣹ مࣵنْدِݝُ ݖِينْ، لࣹرُ رࣺيوُمْ اࣹدࣷمْ، دَلࣹيتِ ݖَ ݖَتَلْ گࣺيجُ خࣷرࣷمْ گَ ݖَ ݒࣹنڪُ؞ ‏4 سࣹينْ ڪࣹمُ دَا دࣴنگّلُ، جَارࣹ فَ فࣺيتࣹ يࣺيگُبْ اَڪࣹرَبِمْ بࣴجْ-سَالُمْ، جَارࣹ ݖِينْ نَگْ، دࣹمْ بَ بࣴجْ-سَالُمُ ڪَدࣹسْ بَرْنࣹيَ، جࣴمْ اࣵݖَّرْ اَدَرْ بَ اࣵگّ اَسْمࣷنْ؞ ‏5 اَسْمࣷنْ لَ ڪࣹمْ گَيْ جࣵدَّاتࣹ جُبَلْ خُرُ مِسْرَ، دࣷورَ اࣵڪِّ گࣺيجْ؞
‏6 «گࣺيجْ گُ مَگْ گِ مࣷويْ دࣷونْ سࣹينْ ڪࣹمُ سࣷوُ؞ لࣷولࣷويْ سࣹينْ ڪࣹمُ سࣷوُ؞
‏7 «لِي نَگْ مࣷويْ سࣹينْ ڪࣹمُ بࣴجْ-گࣵنَّارْ؞ ݖَ گࣺيجْ گُ مَگْ گَ نگࣹينْ دِ رࣴدّࣹ، بَ ݖَ تُنْدُ اࣷرْ، ‏8 رࣴدّࣹ ڪࣷ فَ تُنْدُ اࣷرْ، مُ جࣴمْ لࣹبࣷ اَمَتْ، جࣵلّ بَ ݖࣹدࣵدّ، ‏9 وࣺيࣹيتِ فَ، جَارࣹ سِفْرࣷنْ، جࣴمْ اࣵݖَّرْ اࣹنَنْ؞ لࣷولࣷويْ سࣹينْ ڪࣹمُ بࣴجْ-گࣵنَّارْ؞
‏10 «رࣴدّلࣹينْ سࣹينْ ڪࣹمُ ݒࣹنڪُ، مُ دَلࣹ اࣵݖَّرْ اࣹنَنْ بَ سࣹفَمْ، ‏11 ڪࣹمُ گَ وࣵݖّࣹ سࣹفَمْ، جࣴمْ رِبْلَ، فَ فࣺيتࣹ اَيِنْ ݒࣹنڪُ، وࣵݖَّتِ بَ لࣹرُجِ دࣹخُ ڪِنࣹرࣹتْ، ݖَ وࣹتْ گَ ڪࣷ فࣺيتࣹ ݒࣹنڪُ، ‏12 ڪࣹمُ گَ دࣹلُّ وࣵݖّ بَ ݖَ دࣹخُ يُرْدَنْ، دࣷورَ اࣵڪِّ گࣺيجُ خࣷرࣷمْ گَ؞ لࣷولࣷويْ دࣷونْ سࣹينُمْ رࣺيوْ اَڪْ ڪࣹمࣷومْ يِ ڪࣷ وࣴرْ؞»
‏13 بَ مُ ڪࣷ دࣹفࣹي مُوسَىٰ سَنتّ بࣵنِّ إِسْرَاءِيلْ، نࣹ لࣹينْ:‏ «رࣺيوْ مࣷومُ نگࣹينْ دِ تࣹگࣷو بَنتّ، سࣺدّࣷو ڪࣷ، تࣹ مࣷومْ لَ اَجِ سَخْ جِ سَنْتَانࣹ ݧُ جࣷخْ ڪࣷ جُرٗومْ ݧࣹينْتِ گِيرْ يࣹيڪْ گࣹنّ-وࣵلّ، ‏14 گَنَّاوْ گِيرُگْ رُبࣹنْ، اَڪْ سࣹينِ ڪࣴرْ مَامْ، اَڪْ گِيرُ گࣵدّ اَڪْ سࣹينِ ڪࣴرْ مَامْ سࣺدُّ نَݧُ، تࣹ گࣹنّ-وࣵلُّ گِيرُگْ مَنَسࣹ اِتَمْ جࣷتْ نَݧُ سࣹينْ ݖࣺرْ بَ نࣷݒِّ؞ ‏15 ݧَارِ گِيرْ يَ، اَڪْ گࣹنّ-وࣵلُّ گِيرْ گࣷوگُ ݧࣷو سࣺدُّ سُوفْ ݖَ ݒࣹنڪُبْ دࣹخُ يُرْدَنْ، فَ جَنࣷوڪْ يࣹرِڪࣷ؞»
لِمْ نَݧُ سَيْتُڪَتِ سࣺدَّلࣹ بَ
‏16 اَجِ سَخْ جِ وَخَتِ مُوسَىٰ، نࣹ ڪࣷ:‏ ‏17 «ݧِي نَگْ ݧࣷو لࣹينْ دِ سࣺدَّلࣹ رࣺيوْ مِ:‏ اࣹلَسَرْ سَرْخَلْڪَتْ بِ، اَڪْ يࣷسُوࣹ دࣷومُ نُونْ؞ ‏18 اَيْ نجِيتْ نگَيْ جࣴلࣹ ڪࣹنّ-ڪࣹنّ، ݖِ گِيرْ گُ نࣹڪّ، نگِرْ ݧُ جࣵݒّ ݖِ سࣺدَّلࣹبْ رࣺيوْ مِ؞ ‏19 ݧࣷوݧࣷو دِ:‏ ݖِ وࣵلُّ گِيرُگْ يُدَ، ڪَلࣹبْ دࣷومُ يࣹفُنࣹ، ‏20 وࣵلُّ گِيرُگْ ݖِمْيࣷنࣹينْ ݧِ، سَمِيࣹلْ دࣷومُ اَمِيُتْ، ‏21 وࣵلُّ گِيرُگْ بࣹݧَمِنْ، اࣹلِدَتْ دࣷومُ ڪِسْلࣷنْ، ‏22 وࣵلُّ گِيرُگْ دَنࣹينْ ݧِ، لࣹنّ نجِيتْ، دِ بُڪِّ دࣷومُ يࣷگْلِ، ‏23 وࣵلُّ يُوسُفࣹينْ ݧِ؛ لِ فࣺيتࣹ اَڪْ گِيرُگْ مَنَسࣹينْ ݧِ، لࣹنّ نجِيتْ، دِ اَنِيࣹلْ دࣷومُ اࣹفࣷتْ، ‏24 لِ فࣺيتࣹ اَڪْ گِيرُگْ اࣹفْرَيْمࣹينْ ݧِ، لࣹنّ نجِيتْ، دِ ڪࣹمْوࣹلْ دࣷومُ سِفْتَنْ، ‏25 ݖِ وࣵلُّ گِيرُگْ سَبُلࣷنࣹينْ ݧِ، لࣹنّ نجِيتْ، دِ اࣹلِسَفَنْ دࣷومُ ݒَرْنَگْ، ‏26 وࣵلُّ گِيرُگْ اِسَڪَرࣹينْ ݧِ، لࣹنّ نجِيتْ، دِ ݒَلْتِيࣹلْ دࣷومُ اَسَنْ، ‏27 وࣵلُّ گِيرُگْ اَسࣹرࣹينْ ݧِ، لࣹنّ نجِيتْ، دِ اَوِيُتْ دࣷومُ سࣹلࣷمِ، ‏28 وࣵلُّ گِيرُگْ نࣹفْتَلࣹينْ ݧِ، لࣹنّ نجِيتْ، دِ ݒࣹدَيࣹلْ دࣷومُ اَمِيُتْ؞» ‏29 ݧࣷوݧُ لَ اَجِ سَخْ جِ سَنتّ ݧُ سࣺدَّلࣹ بࣵنِّ إِسْرَاءِيلْ، رࣺيوُمْ ڪَنَانْ؞
