سَرَخْ نࣺيوْ نَ تࣹ اࣴݒّ يࣷولْ
21
‏1 ݖِ ڪَوْ لࣷولُ عِيسَىٰ خࣷولاَتْ، دَلْدِ گِسْ بࣷرࣷومْ اَلَلْ يِ، ݧُيْ دُگَلْ سࣹينْ سَرَخْ ݖَ بࣷيࣹتُ ڪࣴرْ يࣵلَّ گَ؞ ‏2 مُ گِسْ اِتَمْ اَبْ جࣴتُنْ بُ نࣺيوْ دࣷولࣹ، مُ دࣹفْ ݖَ ݧَارِ ݒࣷسࣹتِ خࣵنجَرْ¤؞ ‏3 عِيسَىٰ نࣹ:‏ «مَا لࣹينْ ڪࣷ وَخْ دࣺيْ، جࣴتُنْ بِي، ݧِي ݧࣺݒّ، مࣷو لࣹينْ اࣴݒّلࣹ لُ مُ جࣷخࣹ؞ ‏4 ندَخْ ݧِي ݧࣺݒّ، لَ ݧُ وࣷومْلࣹ، ݖَ لَݧُ سࣷمْݒࣹ، دࣹفْ ݖِ بࣷيࣹتُ سَرَخْ بِي؛ وَايࣹ مࣷومْ ݖِ بِيرْ اَگْ ݧࣵڪَّمْ لَ جࣴلْ مبࣷولࣹمْ اَبْ دُنْدَمْ، دࣹفْ ݖِ؞»
يࣹرُسَلࣹمْ تَسْ، دࣷومُ نِتْ ڪِ دࣺلْسِ
‏5 مُ اَمْ نَگْ اَيْ نِتْ ݧُ دࣷونْ وَخْ ݖَ ڪࣴرْ يࣵلَّ گَ، اَڪْ تَارُ دࣷجِ تࣵنّࣺيفْ، يَ اَڪْ يࣹنࣹينْ اَلَلِ جࣷوخࣹ يَ ݧُ ڪࣷ رَفࣹتَلࣹ؞ عِيسَىٰ اِتَمْ نࣹ لࣹينْ:‏ ‏6 «لِي نگࣹينْ گِسْ نِي، اَيْ جَنْتَ نگِ ݧࣴوْ يُ فِ دࣷجْ دࣷوتُلْ دࣹسْ ݖِ ڪَوْ دࣷجْ، لࣺݒّ اَيْ جٗورُ؞»
‏7 ݧُ لاَجْ ڪࣷ نࣹ ڪࣷ:‏ «سࣴرِݧْ بِ، لࣷولُ ڪَݧْ لَيْ دࣷونْ؟ اَڪْ لُيْ دࣷونْ اُبْ تَڪّ بُيْ خَمْلࣹ نࣹ لࣷولࣷو نگِ وَاجَ دِڪّ؟» ‏8 عِيسَىٰ نࣹ:‏ «مࣷيْتُلࣹينْ، نگِرْ ݧُ بَݧْ لࣹينَ نَخْ، ندَخْ ݧُ بَرࣹييْ دِڪّ، تُدّࣷو سَمَ تُرْ، نَانْ:‏ ‹مَنَ،› تࣹ نَانْ:‏ ‹وَخْتُ وِ جࣹگࣹ نَ؞› بُلࣹينْ لࣹينْ تࣷݒّ؞ ‏9 بُ نگࣹينْ دࣺگّࣹي اَيْ خَرࣹ اَڪِ سَلْفَاݧࣷو، بُلࣹينْ تِيتْ، ندَخْ فࣵوّ لࣷولُ جࣴڪَّ اَمْ؞ وَايࣹ دُ دࣷونَگُمْ مُجْ گَ، ݖَ سَا سَ؞»
‏10 مُ نࣹيتِ لࣹينْ:‏ «اَوْ خࣹيتْ دِنَ جٗگَلْ اَوْ خࣹيتْ، اَمْ رࣺيوْ جٗگَلْ اَمْ رࣺيوْ؞ ‏11 اَيْ يࣴنگُ-يࣴنگُيْ سُوفْ يُ رࣺيْ دِنَ اَمْ، اَيْ خِيفْ اَڪِ مبَسْ دِڪّࣹ فُ بَرࣹ، تࣹ خࣹوْ-خࣹوْ يُ رَگْلُ دِنَ اَمْ، اَڪْ فِرْندࣹ يُ مَگْ يُ جٗگࣹ اَسَمَانْ؞ ‏12 وَايࣹ لاَتَ لࣷولُ لࣺݒّ، يࣹينْ، دࣹيسْ نَ لࣹينْ تࣹگْ لࣷخࣷ، فِتْنَالْ لࣹينْ، جࣺبَّلْ لࣹينْ ڪُرࣺلِ جࣵنگُ يَ، تࣴجْ لࣹينْ ڪَسࣷ؞ دࣹيسْ نَ لࣹينْ يٗبُّ ݖِ ڪَنَمِ بُورْ اَڪِ بࣷرࣷومْ دࣴڪّ ندَخْ سَمَ تُرْ، ‏13 نگِرْ نگࣹينْ دࣷخࣹ ݖَ يࣺڪَّتِ ڪࣵدُّيْ سࣹيدࣹ؞ ‏14 ڪࣷنْ نَگْ دࣹفْلࣹينْ بُ بَاخْ ݖِ سࣹينْ خࣹلْ، نࣹ سࣷخْلَوُلࣹينَ وَاجَلْ ڪࣵدُّگْ لَيࣷو؞ ‏15 ندَخْ مَنْ مَا لࣹينْ دِ جࣷخْ لࣵمِّݧْ وُ اࣵندّ اَڪْ خࣹلْ، وُ سࣹينْ بࣹنّ نࣷونْ دُلْ مَنَ دࣵقْ مبَا دِ ڪࣷ وࣹدِّ؞ ‏16 ݒࣹخࣹ مُ لࣹينْ جࣵݒّلُ دِنَ سَبَبࣷو سَخْ ݖِ سࣹينِ وَاجُرْ اَڪْ سࣹينْ دࣷومِ ندࣹيْ اَڪْ سࣹينِ جࣹگࣹݧَالࣹ اَڪْ سࣹينِ خَرِتْ، تࣹ دࣹيسْ نَ رࣹيْ ݧࣹنّ ݖِ يࣹينْ، ‏17 ݧࣺݒَّ لࣹينْ دِ بَݧْ ندَخْ سَمَ تُرْ؞ ‏18 وَايࣹ سࣹينْ گࣹنّ ڪَوَرُ بࣷݒّ سَخْ دُ سࣵنڪُ؞ ‏19 سَخࣷولࣹينْ مُݧْ نَگْ، دَلْدِ مُݖّ؞
‏20 «بُ نگࣹينْ گِسࣹي گࣵنگࣷورُ خَرࣹ گُ گَوْ يࣹرُسَلࣹمْ، سُ بࣷوبَا خَمْلࣹينْ نࣹ تَسْتࣹيمْ دࣴگْمَلْ نَ؞ ‏21 سُ بࣷوبَا ݧَ مُ فࣹڪّ ݖَ دِيوَانُ يُدࣹ، نَݧُ دَوْ جࣴمْ ڪَوْ تُندّ يَ، ݧِ ݖِ بِيرْ دࣴڪّ بِ، نَݧُ دَوْ گࣺنّ، تࣹ ݧَ مُ فࣹڪّ ݖَ تࣷولْ يَ، بُݧُ دُگّسِ ݖِ بِيرْ دࣴڪّ بِ؞ ‏22 ندَخْ جَنْتِ مبُگَلْ لَيْ دࣷونْ، نگِرْ سࣷتَّلْ مبࣷولࣹمْ لُ ݧُ بِنْدࣷونْ؞ ‏23 وٗويْ نگَلَّ جِگࣺينِ وࣺرُلْ ݧَ، اَڪْ ݧَيْ نࣵمْݒَلْ ݖَ يࣷويُ جَنتّ!‏ ندَخْ تِيسْ وُ رࣺيْ لَيْ دࣷونْ ݖِ رࣺيوْ مِ، اَڪَمْ سࣵنجّ ݖِ ڪَوْ اَسْڪَنْ وِي؞ ‏24 دِنَݧُ فࣵدّࣷو ݧَوْڪَيْ سَامَرْ، تࣹ دࣹيسْ نَ لࣹينْ جࣵݒّ نجَامْ، يٗبُّ ݖِ مبࣷولࣹمْ خࣹيتْ يِ؞ تࣹ جَامْبُرْ ݧِ دُلْ يَوُتْ ݧࣷويْ جࣷگَّتِ يࣹرُسَلࣹمْ، بَ ڪࣹرَ اࣵݒُّ ݧࣷوݧُ دُلْ يَوُتْ دِ مَتْ؞
‏25 «اَيْ فِرْندࣹ دِنَݧُ فࣹيݧْ ݖِ جَنتّ بِ اَڪْ وࣹيرْ وِ اَڪْ بِدِّيوْ يِ؞ ݖِ ڪَوْ سُوفْ نجࣵقَرࣹ اَڪْ تِيتَانگࣹ مࣷويْ ݧࣹيلْ خࣹيتْ يِ ندَخْ رِيرُمْ گࣺيجْ گِ اَڪْ گَنَّخْ يِ، ‏26 نِتْ ݧِيْ سࣺينْتُ لِ نَرَ دِڪَّلْ اࣵدِّنَ، دِ خࣴمْ ندَخْ تِيتَانگࣹ، ندࣹ دࣷولࣹيْ اَسَمَانْ دِنَ سࣵقِ؞ ‏27 سُ بࣷوبَا دࣹيسْ نَ گِسْ دࣷومُ نِتْ ڪِ دِ ݧࣴوْ ݖِوْ نِيرْ، اࣵندّ اَڪْ مَنࣷورࣹ اَڪْ دَرَجَ جُ رࣺيْ؞ ‏28 بُ خࣹوْ-خࣹوْ يࣷويُ تࣵمْبَلࣹي سࣷتِّ نَگْ، تَخَوْلࣹينْ تࣹ سِگِّ، ندَخْ سࣹينُگْ نجࣷتْ دࣴگْمَلْ نَ؞»
‏29 ݖِ ڪَوْ لࣷولُ عِيسَىٰ لࣺيبْ لࣹينْ نࣹ لࣹينْ:‏ «خࣷولْ-لࣹينْ گَرَبُ فِگّ مبَا گࣹنࣹينْ گَرَبْ؞ ‏30 بُ نگࣹينْ گِسࣹي مُ جࣹبِّ، ݖَ نگࣹينْ دِ خَمَلْ سࣹينْ بࣷݒّ نࣹ نَوࣹتْ جُبْسِ نَ بَ نࣷݒِّ؞ ‏31 بُ نگࣹينْ گِسࣹي خࣹوْ-خࣹوْ يࣷويُ سࣷتِّ، نࣷونُ اِتَمْ نگࣹينْ دِ خَمࣹ نࣹ نگُورُگْ يࣵلَّ دࣴگْمَلْ نَ؞
‏32 «مَا لࣹينْ ڪࣷ وَخْ دࣺيْ، نِتِ تࣹيْ جِي دُݧُ وࣺيْ مُڪّ، تࣹ لࣷولُ لࣺݒّ سࣷتِّوُلْ؞ ‏33 اَسَمَانْ اَڪْ سُوفْ دِنَ وࣹيسْ، وَايࣹ سَمَيْ وَخْ دُ وࣹيسْ مُڪّ؞
‏34 «مࣷيْتُلࣹينَ يَافُسَلْ سࣹينْ بࣷݒّ ݖِ نَانْ گُ اࣴݒّ اَڪْ مࣵنْدِتࣹ اَڪْ خَلاَتِ اࣵدِّنَ، لُ ڪࣷ مࣷيْ بࣺسْ بَ جࣹڪِّ دَلْ ݖِ سࣹينْ ڪَوْ، ‏35 ندَخْ نِ مبَالْ دِ بࣹتّࣹي، نࣷونُ لَ بࣺسْ بَيْ جࣹڪّࣹي دَلْ ݖِ ڪَوْ مبࣷولࣹمْ ݧِ ݖِ ڪَوْ سُوفْ؞ ‏36 تࣹيوْلُلࣹينْ بࣷوگْ، تࣹيْ ݧَانْ سَا سُ نࣹڪّ، بَ مَنَ مُݖّ ݖِ مبࣷولࣹمْ لࣷولُيْ نَرَ خࣹوْ، نگِرْ نگࣹينْ تَخَوْ فِ ڪَنَمْ دࣷومُ نِتْ ڪِ؞»
‏37 بࣺسْ بُ نࣹڪّ نَگْ عِيسَىٰ نگَ دࣷونْ يࣹنْدࣷو جࣵنگّلࣹ ݖَ ڪࣴرْ يࣵلَّ گَ، بُ گُدّࣹي مُ فَنَانِ ݖَ ڪَوْ تُندّ وَ ݧُيْ وَخْ تُنْدُ اࣷلِوْ يَ؞ ‏38 مبࣷولࣷو مࣺݒَّ دَانْ جࣴلُ، فࣹڪِّ ڪࣷ ݖَ ڪࣴرْ يࣵلَّ گَ، نگِرْ دࣺگْلُ ڪࣷ؞
