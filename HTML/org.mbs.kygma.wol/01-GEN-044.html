يُوسُفَ نَتُّ نَ اَيْ مَگَمْ
44
‏1 بَ لࣷولُ وࣺيࣹي مُ سَنتّ فَرَ بِيرْ ڪࣴرَمْ، نࣹ ڪࣷ:‏ «سࣷلَلْ سࣹينِ سَاڪُ ݒࣹݒّ لُ ݧُ مَنَ اࣵتَّنْ، تࣹ نگَ دࣹلّࣷو ڪُ نࣹڪّ خَالِسَمْ، تࣹگَلْ ڪࣷ ڪࣷ ݖَ بُنْتُ سَاڪࣷومْ؞ ‏2 وَايࣹ سَمَ ڪَاسُ خَالِسْ بࣷوبࣹي، يࣹبَلْ ڪࣷ ڪِ گࣴنْ دِ خَلࣹ ݖِ سَاڪࣷومْ، بࣷولࣹ ڪࣷ اَڪْ خَالِسُ ݒࣹݒَّمْ؞» مُ دࣹفْ لَ يُوسُفَ وَخْ؞
‏3 ݖَ اࣴلّࣴگْ سَ، بَ جَنتّ بَيْ فࣹنڪّ، يِوِ نَݧُ گَا ݧَ، ݧُ سࣴفْ سࣹينِ مبَامْ، دࣹمْ؞ ‏4 نَڪَ لَݧُ گࣺنّ دࣴڪّ بَ، تࣹ سࣷرࣹيگُݧُ سَخْ، يُوسُفَ سَنتّ فَرَ بِيرْ ڪࣴرَمْ، نࣹ ڪࣷ:‏ «گَاوَلْ دَبْ گَا ݧَ، تࣹ بࣷو لࣹينْ دَبࣹي، وَخْ لࣹينْ نࣹ لࣹينْ:‏ ‹لُ تَخْ ݧُ جِ لࣹينْ نجࣹڪّ، نگࣹينْ فࣹيࣹ نگࣷرࣹيدِ؟ ‏5 ندَخْ لِ نگࣹينْ جࣴلْ دُ سَمَ ڪَاسُ سَنگّ، بِ مُيْ نَانࣹ، رَوَتِنَ دِ ڪࣷ گِسَانࣹي؟ لُ ݧَاوْ نگࣹينْ دࣹفْ نِي!‏›»
‏6 مُ دَلْدِ لࣹينْ دَبْ، وَخْ لࣹينْ لࣷولُ؞ ‏7 ݧُ نࣹ ڪࣷ:‏ «ڪِلِفَ گِ، لࣷويْ وَخْ نِي؟ نُنْ دࣹ، سَنگّ بِ، جࣷمبّ نَنُ لࣷولُ!‏ ‏8 خَالِسْ بِ نُ فࣹڪّࣷونْ ݖِ سُنُ بُنْتِ سَاڪُ سَخْ، دَنࣷو اࣵنْدَاتْ اَڪْ مࣷومْ، جٗگࣹ بَ ڪَنَانْ، دࣺلّࣷوسِلْ لَ ڪࣷ، ڪࣷنْ نَنْ لَنُ مَنَ سࣵݖّࣹ سَ ڪࣴرْ سَنگّ خَالِسْ مبَا وُرُسْ؟ ‏9 نگَلَّ ڪِلِفَ گِ، يَايْ سُنُ سَنگّ؛ ڪࣷو ڪࣷ فࣹڪّࣹ ݖِ نُنْ رࣹڪّ، بࣷو يࣹبࣷو رࣹيَلْ، ݧِ دࣹسْ ݖِ نُنْ دࣷخࣹ فَ، دِ سَيْ جَامْ؞» ‏10 مُ نࣹ لࣹينْ:‏ «لِ نگࣹينْ وَخْ يࣷونْ لَ سَخْ، وَايࣹ نَ ڪِ مَ ڪࣷ فࣹڪّࣹ دࣷݝّ دِ سَمَ جَامْ، سُ ڪࣷ دࣹفࣹي ݧِ ݖِ دࣹسْ وࣵݖّ؞»
‏11 ݧُ گَاوْ نَگْ، وࣵݖّࣹ سࣹينِ سَاڪُ ݖِ سُوفْ، ڪُ نࣹڪّ اُبِّ سَاڪࣷومْ؞ ‏12 مَ نگَيْ سࣹيتْ، تࣵمْبَلࣹي ݖَ تَاوْ بَ، بَ ݖَ ݖَاتْ مَ، دࣷورَ فࣹڪّ ڪَاسْ بَ ݖَ سَاڪُ بࣹݧَمِنْ؞ ‏13 بَ لࣷولُ اَمࣹي دَݧࣷو تࣷرࣷخْ، بَ خࣷتِّ سࣹينِ يࣺرࣹ، ڪُ نࣹڪّ سࣴفَاتْ مبَامَمْ، ݧُ دࣹلُّ ݖَ دࣴڪّ بَ؞
يُدَ تِنُلْ نَ بࣹݧَمِنْ ݖَ يُوسُفَ
‏14 يُدَ اَڪِ دࣷومِ بَايَمْ نَگْ دࣺلْسِ ڪࣴرْ يُوسُفَ، فࣹڪّ دࣹمَگُلْ؞ ݧُ دَلْدِ نࣹ نࣴرࣴمْ، دࣴݒّ سࣹينْ جࣴ فِ سُوفْ؞ ‏15 يُوسُفَ نࣹ لࣹينْ:‏ «لِ نگࣹينْ دࣹفْ نَگْ؟ خَنَا خَمُلࣹينْ نࣹ نِتْ ڪُ مࣹلْ نِ مَنْ دَيْ گِسَانࣹ؟» ‏16 يُدَ نࣹ:‏ «سَنگّ بِ، لُ نُيْ وَخَتِ؟ لُ نُ مَنَ لَيْ، بَ سࣹتَلْ سُنُ بࣷݒّ؟ يࣵلاَّ نُ فࣹيݧَلْ، سَنگّ بِ؞ نَنُ بࣷڪّ نُنْ ݧࣺݒّ دِ سَيْ جَامْ، يࣹمْ ݖِ ڪࣹݒّ اَڪْ ڪِ ݧُ فࣹڪّ ڪَاسْ بِ ݖِ لࣷخࣷومْ؞» ‏17 مُ نࣹ لࣹينْ:‏ «جࣷمبّ نَا لࣷولُ؞ ڪِ ݧُ فࣹڪّ ڪَاسْ بِ ݖِ لࣷخࣷومْ مࣷويْ دࣷونْ سَمَ جَامْ؛ ݧِ ݖِ دࣹسْ دࣹلُّ ݖَ سࣹينْ بَايْ ݖِ جࣵمّ؞»
‏18 يُدَ جࣹگࣹسِ نࣹ ڪࣷ:‏ «نگَلَّ سَنگّ بِ، مَ ݧࣹمࣹ لاَ دِيسْ لࣹنّ رࣹڪّ، تࣹ سَ خࣷلْ بَݧَ تࣵنگّ ݖِ مَنْ، دࣷونْتࣹ يَاڪْ فِرْعَوْنَا تࣷلّࣷو؞ ‏19 سَنگّ بِ، دَنگَ نࣷو لاَجࣷونْ نࣹ نُ:‏ ‹ندَخْ اَمْ نگࣹينْ بَايْ وَلَّ رَڪّ؟› ‏20 بَ مُ ڪࣷ دࣹفࣹي نُ نࣹ لَ:‏ ‹اَمْ نَنُ بَايْ بُ مࣵگَّتْ؞ سُنُ رَڪّ، جِ مُ جُرْ تࣹ فࣹڪّ ڪࣷ دِ مࣵگَّتْ نَگْ، مࣷو ݖِ گࣴنْ دِ ندَوْ؞ مَگَمْ دَفَ دࣹي، مُ دࣹسْ مࣷومْ دࣷݝّ ݖِ ندࣹيَمْ، بَايَمْ سࣷݒّ ڪࣷ لࣷولْ؞› ‏21 نگَ وَخࣷونْ نُ نࣹ نُ:‏ ‹اِنْدِلْ-لࣹينْ مَ ڪࣷ، نگِرْ مَ گِسَلْ ڪࣷ سَمَ بࣷݒّ؞› ‏22 ݖِ ڪَوْ لࣷولُ، سَنگّ بِ، نُ وَخْ لَ نࣹ:‏ ‹خَلࣹ بُ گٗورْ بِ مَنُلَ تࣵگّࣷوڪْ بَايَمْ؞ بُ تࣵگّࣷو اَڪْ مࣷومْ، ڪࣷنْ بَايَمْ دِنَ دࣹي؞› ‏23 نگَ وَخْ نُ نࣹ نُ:‏ ‹بُ سࣹينْ رَڪّ اࣵنْدُلْ اَڪْ يࣹينْ، دُنگࣹينْ جࣵڪَّارْلࣷوڪْ مَنْ؞› ‏24 بࣷوبَ لَنُ دࣹمْ ݖَ سُنُ بَايْ، بَ لَيْ سِيَارࣹ، دَلْدِ ڪࣷيْ جࣷتّلِ سَيْ ڪࣵدُّ؞
‏25 «بَ نُ دࣹمࣹي بَ مُ يࣵگّ، سُنُ بَايْ نࣹ:‏ ‹دࣹلُّلࣹينْ، جࣴنْدَلِ نُ تُوتِ دُندّ؞› ‏26 تࣹيوُلْ نُ نࣹ:‏ ‹دُنُ مَنَ دࣹمْ تࣹ سُنُ رَڪّ اࣵنْدُلْ اَڪْ نُنْ، ندَخْتࣹ بُ نُ اࣵنْدُلْ اَڪْ مࣷومْ، دُنُ گِسْ ڪِلِفَ گَ؞› ‏27 سُنُ بَايْ، بَ لَيْ سِيَارࣹ، نࣹ نُ:‏ ‹خَمْ نگࣹينْ نࣹ دَمَا اَمُلْ اَڪْ سَمَ سࣷخْنَ لُ دُلْ ݧَارِ دࣷومْ؞ ‏28 ڪࣹنّ ڪِ دࣹمْ، وَݖّ مَ، مَ نࣹ مࣷومْ ڪَيْ، اَوْ رَبُ اࣵلَّ ڪࣷ دَگَّتࣹ!‏ گِسَاتُمَ ڪࣷ بَ تࣹيْ جِي؞ ‏29 تࣹ بُ نگࣹينْ مَ فِ جࣴلَلࣹي ڪِي اِتَمْ، بَ مُسِبَ دَبْ ڪࣷ، دِنگࣹينْ مَ يٗبّࣹ نَقَرْ وُ مَيْ جࣵݒّ، بَ ڪࣹرَ مَيْ اࣵندّ اَڪْ سَمَ بِجَّاوْ، تࣵبِّ نجَنِيوْ؞›
‏30 «لࣺيگِ نَگْ، بُ مَ ݧِبّࣹي ݖَ سَمَ بَايْ، بَ لَيْ سِيَارࣹ، تࣹ اࣵنْدُنࣷوڪْ خَلࣹ بِي مُيْ نࣷيّࣹي، ‏31 بُ گِسُلْ خَلࣹ بُ گٗورْ بِ اࣵندّ اَڪْ نُنْ ݧِبّسِ— آ!‏ ڪࣷنْ دࣹ دِنَ دࣹي، تࣹ دِنَنُ يٗبّࣹ سُنُ بَايْ نَقَرْ، وُ ڪࣷيْ تࣷݒّ، بَ ڪࣹرَ مُيْ اࣵندّ اَڪْ بِجَّاوَمْ، تࣵبِّ نجَنِيوْ؞ ‏32 گَنَّاوْ لࣷولُ اِتْ خَسࣷونْ نَا دِگࣹ سَمَ بَڪَّنْ، ندَخْ دَمَا وَخࣷونْ نࣹ:‏ ‹بُ مَ لَ ڪࣷ دࣺلّࣷوسِلُلْ، مَا ڪࣷيْ گࣵلّࣷو بَ ڪࣹرَ مَيْ دࣹي؞› ‏33 ڪࣷنْ نگَلَّ سَنگّ بِ، بࣵيِّ مَ مَنْ مَ دࣹسْ فِ دِ سَ جَامْ، وُوتُ خَلࣹ بِ، تࣹ مࣷومْ مُ دࣹلُّ اࣵندّ اَڪِ مَگَمْ ݧِبِّ؞ ‏34 ندَخْ خَوْمَ نَنْ لاَيْ دࣹلّࣷو ݖَ سَمَ بَايْ تࣹ اࣵنْدُمَ اَڪْ خَلࣹ بِ؞ يࣵلَّ بُمَ فࣹڪّࣹ نجࣹڪَّرْ، لِ ݖِيْ دَلْ سَمَ بَايْ!‏»
