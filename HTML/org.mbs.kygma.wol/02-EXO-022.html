يࣷونُ اَجِ سَخْ جِ ݧࣷݝَلْ نَ اَلَلُ نِتْ
22
‏1 «سُ فࣹڪّࣹي دَݧࣷو جࣵݒّ اَبْ سࣵݖّ گُدِّ، مُيْ تࣷجْ ڪࣴرْ جَامْبُرْ، بُ ݧُ ڪࣷ دٗورࣹي، بَ مُ دࣹي، دࣹيسُلْ تࣷݒّ ڪِ دٗورْ، بَڪَّنُ سࣵݖّ بِ، ‏2 وَايࣹ سُ جَنتّ خَسࣹي فࣹنڪّ، دࣹيسْ نَ ڪࣷ تࣷݒّ بَڪَّنَمْ؞
«اَبْ سࣵݖّ نَگْ، بُ ݧُ ڪࣷ جࣵݒّࣹي، نَ دࣹلّࣷو لِ مُ سࣵݖّ رࣹڪّ؞ سُ فࣹڪّࣹي اَمُ ڪࣷ، دࣹيسْ ڪࣷيْ جَايْ نجَامْ، ندَخْ ݖࣵݖّ گِ؞
‏3 «سُ دࣹي ڪُ سࣵݖّ مَلَ، مُيْ نَگْ مبَا مبَامْ مبَا اَبْ گࣵتّ، تࣹ ݧُ فࣹڪّ ڪࣷ ݖِيْ لࣷخࣷومْ، مُيْ دُندّ، نَ فࣹيْ ݧَارَمْ؞
‏4 «ڪُ وَݖّ سَگْ جُرْ، مُ رُورْ تࣷولُبْ جَامْبُرْ، بَ لࣹڪّ تࣷولَمْ مبَا تٗوڪࣴرُ رࣹسࣹݧَمْ، نَ تࣵنّ لِ گࣴنْ ݖِ مࣹݧّࣹيفُمْ تࣷولُ بࣷݒَّمْ مبَا تٗوڪࣴرُ رࣹسࣹݧَمْ، دࣵمْݒࣹ ڪࣷ بࣷرࣷومْ تࣷولْ بِ مُ لࣷرْ؞
‏5 «سُ فࣹڪّࣹي نࣹ دَݧࣷو تَالْ سَوَرَ، مُ جࣵݒّ ݖِبْ ݧَگْ، لَڪَّالࣹ ݒࣹݒّ مُ جٗورࣹ، مبَا ݒࣹݒّ مُ ݧࣷرْ مُ ݧُ گٗوبَگُلْ، مبَا مُ ݧࣷرَگُلْ، ڪَ تَالْ سَوَرَ وَ مࣷويْ خَلَ فࣹيْ لِ لَڪّ؞
‏6 «سُ نِتْ دࣺنڪࣹي مࣷرࣷومَمْ خَالِسْ، مبَا اَلَلْ، ݧُ سࣵݖّࣹ ڪࣷ ݖِ ڪࣴرْ ڪِ دࣹنݖّ، تࣹ ݧُ جࣵݒّ سࣵݖّ بِ، نَ سࣵݖّ بِ فࣹيْ ݧَارِ يࣷونْ لَ مُ سࣵݖّ؞ ‏7 سُ ݧُ ڪࣷ جࣵݒُّلْ نَگْ، نَ بࣷرࣷومْ ڪࣴرْ گِ سࣵڪُّ سࣹيدࣹ اࣵتّࣹڪَتْ يِ، تࣹ وَاتْ نࣹ، دُ مࣷومْ ݖِ بࣷݒَّمْ مࣷو آڪِمࣷو اَلَلُ مࣷرࣷومَمْ؞ ‏8 سَا يُ اَيْ نِتْ جࣷتࣹي ݖِ مبِرُمْ نَگْ مبَا مبَامْ مبَا گࣵتّ مبَا يࣺرࣹ، مبَا لُ رࣺيرْ لُ مُ دࣷونْ، ݧَارْ ݧِ سࣷلࣷوࣷو اَلَلْ جِ، مُيْ مَلَ مبَا لࣹنࣹينْ، نَݧُ يٗبُّ مبِرْ مِ فِ ڪَنَمْ اࣵتّࣹڪَتْ يِ؞ ڪُ ݖِ اࣵتّࣹڪَتْ يِ تࣹگْ تࣷوݧْ، نَ فࣹيْ مࣷرࣷومْ مِ ݧَارِ يࣷونْ؞
‏9 «سُ نِتْ دࣺنڪࣹي مࣷرࣷومَمْ مبَامْ مبَا نَگْ مبَا گࣵتّ مبَا مَلَ مُ مُ مَنَ دࣷونْ، سُ جُرْ گِ دࣹييࣹي مبَا مُ دَمّ مبَا ݧُ سࣵݖّࣹ ڪࣷ اࣵلّ تࣹ اَمُلْ سࣹيدࣹ، ‏10 لِ لࣹينْ دِ اࣵتّࣹ نگِݧْ لَيْ دࣷونْ ݖِ اَجِ سَخْ جِ؞ نَ ڪَ دࣹنݖࣷونْ مَلَ مَ گِݧْ نࣹ جࣴلُ ڪࣷ؞ نَ بࣷرࣷومْ مَلَ مِ اِتْ نَنگُ نگِݧْ لِ؞ ڪِ دࣹنݖࣷونْ دُ ڪࣷ فࣹيْ دَرَ؞ ‏11 وَايࣹ سُ ݧُ سࣵݖّࣹي مَلَ مِ ݖِ ڪࣴرَمْ، مࣷومْ مِ دࣹنݖࣷونْ مࣷويْ دࣹلّࣷو بࣷرࣷومْ؞ ‏12 سُ فࣹڪّࣹي نࣹ رَبُ اࣵلَّ فࣵدّ مَلَ مِ، نَ اِنْدِ لِ ݖِ دࣹسْ، مُيْ فِرْندࣹ؞ سُ بࣷوبَا دُ فࣹيْ دَرَ؞ ‏13 سُ نِتْ اࣵبّࣹي جُرُ مࣷرࣷومَمْ، جُرْ گَ دَمّ مبَا مُ دࣹي فُ بࣷرࣷومْ نࣹڪُّلْ، نَ ڪِ اࣵبّࣷونْ فࣹيْ؞ ‏14 وَايࣹ سُ ڪࣷ بࣷرࣷومَمْ فࣹڪّࣹي، دُ لاَجْ مُيْ فࣹيْ؞ سُ فࣹڪّࣹي نࣹ دَفَ بِنْدࣷونْ مَلَ مِ ݖِ بࣷرࣷومْ، دِ ڪࣷ فࣹيْ، نَ بࣷرࣷومْ فࣺيتࣹوࣷو نجࣺگْ، گَ ݧُ بِنْدࣹ وࣷونْ مَلَ مِ، مُيْ ݒࣹيَمْ؞
يࣷونُ اَجِ سَخْ جِ رَفࣹتَلْ نَ دُنْدِنْ
‏15 «سُ گٗورْ نَخࣹي اَبْ جَنقّ بُ ڪࣹنّ جࣷخَگُلْ وَرُگَرْ، بَ تࣴدّࣹ ڪࣷ، نَ فࣹيْ وَرُگَرَمْ، مُيْ جَبَرَمْ؞ ‏16 سُ فࣹڪّࣹي نࣹ بَايَمْ دَفَ لࣵنڪّ، نࣹ دُ ڪࣷ ڪࣷ مَيْ اِتْ، دُ تࣹرࣹ مُ وَرَ فࣹيْ ݖِ خَالِسْ لُ تࣷلّࣷوڪْ وَرُگَرُ جَنقّ؞
‏17 «اَبْ خࣴرࣴمْڪَتْ بُ جِگࣺينْ، بُلࣹينْ ڪࣷ بࣵيِّ، مُ دُندّ¤؞
‏18 «ڪࣺݒّ ڪُ جَخَسࣷو اَڪْ مَلَ، دࣹي رࣹڪّ مࣷويْ اࣵتّࣹيمْ؞
‏19 «ڪُيْ سَرْخࣹيلْ جࣹنࣹينْ يࣵلَّ جُ دُلْ اَجِ سَخْ جِ مࣷومْ رࣹڪّ، دࣹيسْ ڪࣷيْ فَاگَاگَلْ؞
‏20 «اَبْ دࣷخَنْدࣺيمْ، بُلࣹينْ ڪࣷ لࣷرْ، بُلࣹينْ ڪࣷ نࣷتْ، ندَخْتࣹ يࣹينْ اِتَمْ اَيْ دࣷخَنْدࣺيمْ نگࣹينْ وࣷونْ فَ رࣺيوُمْ مِسْرَ؞
‏21 «ڪࣺݒّ ڪُ دِبْ جࣴتُنْ اَڪُبْ جِرِمْ، بُلࣹينْ ڪࣷ نࣺيوَلْ؞ ‏22 بُ نگࣹينْ لࣹينْ نࣺيوَلࣹي دࣺيْ، بُ ݧُ يُوخࣷو، وࣷو مَ وَلّ، دِنَا دࣺگّ سࣹينْ يُوخْ مࣷوسْ؞ ‏23 سُ بࣷوبَا سَمَ مࣹرْ دِنَ تࣵنگّ، مَ رࣹيࣹ لࣹينْ سَامَرْ، سࣹينِ جَبَرْ دِيْ جࣴتُنْ، سࣹينِ دࣷومْ دِيْ جِرِمْ؞
‏24 «بُ نگࣹينْ دࣹي لࣹبَلْ ڪُ بࣷڪّ ݖِ سَمَ ݧࣷݧْ، دِ نࣺيوْ-جِ-دࣷولࣹ جُ نگࣹينْ دࣴڪَّلْ، بُلࣹينْ دࣹفْ نِ اَبْ لࣹبْلࣹڪَتْ، دِ ڪࣷ سَسْ اَبْ تࣹگَنْدَايْ؞
‏25 «سُ نگࣹينْ نَنگࣷو مبُبَّمْ تَيْلࣹ ݖِ سࣹينْ دࣴڪَّنْدࣷو، دࣹلّࣷولࣹينْ ڪࣷ ڪࣷ بَلَ جَنْتَ سࣷ؞ ‏26 مࣷومْ دࣷݝّ لَ اَمْ ݖِ يࣺرࣹ، دِ ڪࣷ سࣵنگࣹ يَرَمَمْ؛ سُ ڪࣷ ݧࣵڪّࣹي، لُ مُيْ سࣵنگࣷو بَ تࣴدّ؟ تࣹ سُ مَ وࣷويࣹي وَلّ ڪَتْ، دِنَا ݖِ دࣺگّ، ندَخْ مَنْ بࣷرࣷومْ يࣴرْمَنْدࣹ لاَ؞
‏27 «مَنْ يࣵلَّ، بُلࣹينْ وَخْ لُ ݧَاوْ ݖِ مَنْ، تࣹ سࣹينْ ڪِلِفَگْ اَسْڪَنْ، بُلࣹينْ وَخْ لُ بࣷنْ ݖِ مࣷومْ؞
‏28 «لِ تُورࣷو ݖِ سࣹينْ سࣵقْ يُ فࣹيسْ، اَڪْ لِ وَلࣹ ݖِ سࣹينْ نَلُڪَايْ يُ رࣹمْبَتْ، سَمَ ݖࣺرْ لَ؛ بُلࣹينْ ڪࣷ يࣹيخࣹ، تࣹ سࣹينْ تَاوْ يُ گٗورْ اِتْ، مَنْ نگࣹينْ لࣹينْ دِ سࣺدّࣹ؞ ‏29 سࣹينْ جُرْ گُ گُدّ اَڪْ گُ گࣵتّ اِتَمْ، نَڪَ نࣷونُ؞ جُرٗومْ ݧَارِ فَنْ لَيْ نࣹڪّ اَڪْ ندࣹيَمْ، بࣺسُبْ جُرٗومْ ݧࣹتّࣹيلْ بَ، نگࣹينْ جࣷخْ مَ ڪࣷ؞ ‏30 تࣹ اِتَمْ نِتْ ݧُ سࣹلّ نگࣹينْ وَرَ دࣷونْ، ݧࣹيلْ مَ، ڪࣷنْ نَگْ يࣵݒُّ مَلَ مُ ݧُ فࣵدّࣹ ݖِ اࣵلّ بِ، بُلࣹينْ ڪࣷ لࣹڪّ؛ سࣵنِّلࣹينْ ڪࣷ خَجْ يِ؞
