لُ جࣴمْ ݖِ جَامࣷوڪَايْ بِ
25
‏1 اَجِ سَخْ جِ وَخَتِ مُوسَىٰ نࣹ ڪࣷ:‏ ‏2 «وَخَلْ بࣵنِّ إِسْرَاءِيلْ نࣹ لࣹينْ، ݧُ اِنْدِلْ مَ اَبْ جࣷوخࣹ؞ ݖِ ڪࣺݒّ ڪُ اَمْ يࣺينࣹيمْ نگࣹينْ دِ نَنگࣷو سَمَ جࣷوخࣹ، اِنْدِلْ مَ ڪࣷ؞ ‏3 جࣷوخࣹ بِ نگࣹينْ دِ نَنگࣷو ݖِ ݧࣷومْ دِ وُرُسْ اَڪْ خَالِسْ اَڪْ خࣵنجَرْ، ‏4 اَڪْ وࣴݧْ گُ بَخَ تࣹ لاَلْ يࣷلࣹتْ اَڪْ وࣴݧْ گُ خࣹوَرْ اَڪْ گُ خࣷنقّ ݖُرّ اَڪْ وࣴݧُ لࣹي، اَڪْ ڪَوَرِ بࣴيْ ‏5 اَڪْ دࣹرِ ڪُويُ خَرْ يُ ݧُ وُلِّ، سٗوبْ ڪࣷ خࣷنقَلْ، اَڪْ دࣹرِ ݒِيݒِ¤ اَڪْ بَنْتِ سࣺنگّ¤، ‏6 اَڪْ دِوُ اࣷلِوْ گُ ݧُيْ تَالࣷو، اَڪْ ندࣵبّ لُيْ خࣹيݧَلْ دِوُ ݒَلْ، دِ خࣹيݧَلْ ݖُورَايْ، ‏7 اَڪْ ݒࣹرِ اࣷنِگْسࣴ، اَڪْ يࣹنࣹينْ ݒࣹرْ يُ ݧُيْ تَݒّ ݖِ خَرْ-سࣵنِّمْ سَرْخَلْڪَتْ بُ مَگْ بَ اَڪْ ڪِيرَايَلْ دࣴنَّمْ؞ ‏8 نَ مَ بࣵنِّ إِسْرَاءِيلْ دࣹفَرَلْ ڪࣴرْ گُ سࣹلّ، ندَخْ مَ دࣴڪّ ݖِ سࣹينْ بِيرْ؞ ‏9 نَ لࣺݒّ دࣴݒّࣷو اَڪْ لِ مَ لَيْ وࣷنْ؞ مُيْ مِسَالُ خَيْمَبْ جَامࣷوڪَايْ بِ، دِ مِسَالُ مبࣷولࣹمْ اَيْ جُمْتُڪَايَمْ، لࣺݒّ دࣹفࣹ لࣹينْ ڪࣷ نࣷونُ؞
لُ جࣴمْ ݖِ گَالْ گِيْ دࣹفْ اࣵلُّوَيْ سࣹيدࣹ سِ
‏10 «دࣹفَرْلࣹينْ گَالُ دࣺنڪُ سࣺنگّ، گُدَّايَمْ دِ ݧَارِ خَسَبْ¤ اَڪْ گࣹنّ-وࣵلّ، يَاتُوَايَمْ دِ خَسَبْ اَڪْ گࣹنّ-وࣵلّ؛ تَخَوَايَمْ دِ خَسَبْ اَڪْ گࣹنّ-وࣵلّ؞ ‏11 نَݧُ ڪࣷ خࣷوبْ وُرُسُ نگَلَمْ؞ بِيرْ اَڪْ بِتِ لَݧُ ڪࣷيْ خࣷوبْ، سࣵڪَّلْ ڪࣷ ڪࣺمࣺجُ وُرُسْ، گُ ڪࣷ وࣴرْ؞ ‏12 مٗولَلَلْ گَالْ گِ ݧࣹينْتِ لَمِ وُرُسْ يُ ݧُيْ تَڪّ ݖِ ݧࣹينْتِ تࣵنڪّ يِ؛ ݧَارِ لَمْ، وࣹتْ گِي؛ ݧَارْ وࣹتْ گࣹي؞ ‏13 يࣹتّلُلْ اَيْ بَنْتِ نجࣵݒُّيْ سࣺنگّ تࣹ نگَ خࣷوبْ بُ ݖِ نࣹڪّ وُرُسْ؞ ‏14 سُ ڪࣷ دࣹفࣹي نگَ رࣷوفْ نجࣵݒُّ يِ ݖِ لَمْ، يِ ݖِ وࣹتِ گَالْ گِ، نگِرْ ݧُ دِ ڪࣷ يٗبّࣷو گَالْ گِ؞ ‏15 ݖِ لَمِ گَالْ گِ لَ نجࣵݒُّ يِيْ دࣹسْ تࣹ بَݧْ ݖࣹي جٗگࣹيتِ؞ ‏16 يࣹبَلْ ݖِ گَالْ گِ اࣵلُّوَيْ سࣹيدࣹ سِ مَ لَيْ جࣷخْ؞
‏17 «دࣹفَرْلُلْ ڪُبࣹيرْ گُ ݧُيْ اَمَلࣹ نجࣷتْلاَيْ گُ وُرُسُ نگَلَمْ نگِرْ گَالْ گِ؞ نَ گُدَّايَمْ دِ ݧَارِ خَسَبْ اَڪْ گࣹنّ-وࣵلّ، يَاتُوَايَمْ دِ خَسَبْ اَڪْ گࣹنّ-وࣵلّ؞ ‏18 تࣴگّلُلْ ݧَارِ مَلاَڪَيْ سࣹرُبْ يُ وُرُسْ تࣹ نگَ فࣺيتَلࣹ لࣹينْ اَڪْ ݧَارِ ݖَتِ ڪُبࣹيرُ نجࣷتْلاَيْ گَ؞ ‏19 دࣹفَلْ ݖَتْ لِي اَبْ سࣹرُبْ، ݖَتْ لࣹي اَبْ سࣹرُبْ تࣹ نَنگَ مٗولاَلࣹ ڪُبࣹيرْ گِ اَڪْ سࣹرُبْ يِ ݖِ ݧَارِ ݖَتْ يِ، ݧُ اࣵندّ دِ بࣹنّ؞ ‏20 نَ سࣹرُبْ يِ تࣵلَّلْ سࣹينِ لاَفْ، جࣴمࣹ ڪَوْ تࣹ نَݧُ يِيرࣹ ڪُبࣹيرُ نجࣷتْلاَيْ گِ سࣹينِ لاَفْ؞ نَݧُ جَنࣷو، تࣹ نَ سࣹينِ ڪَنَمْ جُبْلࣷوڪْ ڪُبࣹيرُ نجࣷتْلاَيْ گِ؞ ‏21 تࣹگَلْ ڪُبࣹيرُ نجࣷتْلاَيْ گِ ݖِ ڪَوْ گَالْ گِ، گَنَّاوْ بࣷو ݖِ يࣹبࣹي اࣵلُّوَيْ سࣹيدࣹ سِ مَ لَيْ جࣷخْ؞ ‏22 فࣷوفَ لاَيْ دَجࣹيڪْ يَوْ؛ فَ تِيمْ ڪُبࣹيرُ نجࣷتْلاَيْ گَ، ݖَ دِگَّنْتࣹ ݧَارِ سࣹرُبْ يَ ݖَ ڪَوْ گَالُ سࣹيدࣹ سَ، فَ لاَ لَيْ وَخࣹ مبࣷولࣹمْ لُ مَ لَيْ سَنتّ، ݧࣹيلْ بࣵنِّ إِسْرَاءِيلْ؞
لُ جࣴمْ ݖِ تَابَلُ مبُرُمْ سَرَخْ مِ
‏23 «دࣹفَرْلُلْ تَابَلُ بَنْتُ سࣺنگّ، گُدَّايَمْ دِ ݧَارِ خَسَبْ، يَاتُوَايَمْ دِ خَسَبْ، تَخَوَايَمْ دِ خَسَبْ اَڪْ گࣹنّ-وࣵلّ؞ ‏24 خࣷوبْ ڪࣷ وُرُسُ نگَلَمْ تࣹ دࣹفَلْ ڪࣷ ڪࣺمࣺجُ وُرُسْ گُ ڪࣷ وࣴرْ؞ ‏25 دࣹفَلْ ڪࣷ ݒࣹگّ بُ ڪࣷ وࣴرْ بُ تࣷلُّ ݖِ ݧࣹينْتِ بَارَامْ، تࣹ سࣵڪَّلْ ݒࣹگّ بَ وࣷمْبُ وُرُسْ، وࣴرَلࣹ ڪࣷ؞ ‏26 دࣹفَرْلُلْ ڪࣷ ݧࣹينْتِ لَمِ وُرُسْ تࣹ تَڪّ لَمْ يَ ݖَ ݧࣹينْتِ وࣹتْ يَ، فَ فَرࣹيڪْ ݧࣹينْتِ تࣵنڪّ يَ؞ ‏27 نَ لَمْ يِ دࣹنْدࣹيڪْ ݒࣹگّ بِ، نگِرْ مَنَ تࣺيࣹ نجࣵݒُّ، يِ ݧُيْ يٗبّࣷو تَابَلْ جِ؞ ‏28 يࣹتّلُلْ اَيْ نجࣵݒُّ يُ بَنْتُ سࣺنگّ، نگَ خࣷوبْ نجࣵݒُّ يِ وُرُسْ، ݧُ دِ ڪࣷ يٗبّࣷو تَابَلْ جِ؞ ‏29 سࣵڪَّلْ ڪࣷ اَيْ ندَبَمْ يُ تࣹلّ اَڪِ مبࣵتّࣷومْ اَڪِ نجَقَمْ، اَڪْ ڪࣴلَّمْ يِ ݧُيْ تُورࣷو؞ تࣹ وُرُسُ نگَلَمْ نگَيْ دࣹفْ لُ ݖِ نࣹڪّ؞ ‏30 تࣹگَلْ مبُرُمْ تࣹيوَلْ مِ ݖِ ڪَوْ تَابَلْ جِ، مُ نࣹڪّ ݖِ سَمَ ڪَنَمْ بَ فࣵوّ؞
لُ جࣴمْ ݖِ تࣹگُڪَايُ لࣵمݒّ بِ
‏31 «دࣹفَرْلُلْ تࣹگُڪَايُ لࣵمݒّ بُ وُرُسُ نگَلَمْ؞ تࣹگُڪَايْ بَ، دࣹيسْ ڪࣷيْ تࣴگّ؞ نَ تَاتُ تࣹگُڪَايْ بَ اَڪْ ݒࣹرْ بَ اࣵندّ اَڪِ ڪَاسَمْ اَڪِ ڪࣵمْبٗوتَمْ اَڪِ مبَرَمْ، لࣺݒّ دِ بࣹنّ؞ ‏32 جُرٗومْ بࣹنِّ ݖَرْ ݧࣷويْ سࣷقِڪࣷو ݖَ وࣹتْ يَ، ݧࣹتِّ ݖَرْ ݖِ گِي وࣹتْ، ݧࣹتّ ݖَ گَ ݖَ دࣹسْ؞ ‏33 بࣹنّ ݖَرْ دَيْ اَمْ ݧࣹتِّ ڪَاسْ يُ بِنْدࣷو نِ تٗورْ-تٗورُ گَرَبُ اَمࣵندّ¤، ڪَاسْ بُ ݖِ نࣹڪّ اࣵندّ اَڪْ اَيْ ڪࣵمْبٗوتَمْ اَڪِ مبَرَمْ، ݖَرْ بَ ݖَ تࣷݒّ اَمْ ݧࣹتِّ ڪَاسْ يُ بِنْدࣷو نِ تٗورْ-تٗورُ گَرَبُ اَمࣵندّ، ڪَاسْ بُ ݖِ نࣹڪّ اࣵندّ اَڪِ ڪࣵمْبٗوتَمْ اَڪِ مبَرَمْ، جُرٗومْ بࣹنِّ ݖَرْ، يَ سࣷقِڪࣷو ݖَ ݒࣹرْ بَ، دَجْ؞ ‏34 گَنَّاوْ لࣷولُ ݒࣹرُ تࣹگُڪَايُ لࣵمݒّ بَ اِتَمْ دَيْ اَمْ ݧࣹينْتِ ڪَاسْ يُ بِنْدࣷو نِ تٗورْ-تٗورُ گَرَبُ اَمࣵندّ، اࣵندّ اَڪْ اَيْ ڪࣵمْبٗوتَمْ اَڪِ مبَرَمْ؞ ‏35 اَبْ ڪࣵمْبٗوتْ اَيْ رࣷنُ يِي ݧَارِ ݖَرْ، جࣵݒّ ݖَ؛ بࣹنࣹينْ ڪࣵمْبٗوتْ رࣷنُ ݧَارِ ݖَرْ يَ ݖَ تࣷݒّ، جࣵݒّ ݖَ، اَبْ ڪࣵمْبٗوتْ رࣷنُ ݧَارِ ݖَرْ يَ مُجّ، جࣵݒّ ݖَ، جُرٗومْ بࣹنِّ ݖَرْ، يَ سࣷقِڪࣷو ݖَ ݒࣹرُ لࣵمݒّ بَ، دَجْ؞ ‏36 ڪࣵمْبٗوتْ يَ اَڪْ ݖَرْ يَ دَݧُيْ اࣵندّ اَڪْ تࣹگُڪَايْ بَ، دِ بࣹنّ، لࣺݒّ دࣷونْ لُ ݧُ تࣴگّࣹ وࣹنّ دࣷگُ وُرُسُ نگَلَمْ؞ ‏37 دࣹفَرْلُلْ جُرٗومْ ݧَارِ لࣵمْݒَمْ؞ تࣹ نَ لࣵمݒّ يَ تࣹگࣹ تࣹگِنْ وُيْ لࣹيرَلْ ݖَ ڪَنَمْ تࣹگُڪَايْ بَ؞ ‏38 ݧِيمِ فࣹيُڪَايُ لࣵمݒّ يِ اَڪْ اَنْدِ لࣵمݒّ يِ نَ دࣷونْ وُرُسُ نگَلَمْ؞ ‏39 فَنْوࣹيرِ ڪِلࣷيْ وُرُسُ نگَلَمْ اَڪْ جُرٗومْ¤، ݖِ لࣹيسْ دِ سࣵڪّ تࣹگُڪَايُ لࣵمݒّ بِ، اَڪْ مبࣷولࣹمْ لِ مُ اࣵنْدَلْ؞ ‏40 نَنگَ خࣷولْ تࣹ تࣷݒّ مِسَالْ، يِ ݧُ لَ ݖِيْ وࣷنْ ݖِ ڪَوْ تُندّ وِ؞
