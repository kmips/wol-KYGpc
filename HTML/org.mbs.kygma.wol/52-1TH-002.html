نگَنُگْ ݒٗولْ نࣹينُلْ
2
‏1 يࣹينْ ݖِ سࣹينْ بࣷݒّ ڪَيْ، بࣷڪّ يِ، خَمْ نگࣹينْ نࣹ سُنُ نگَنْ اَڪْ يࣹينْ نࣹينُلْ؞ ‏2 خَمْ نگࣹينْ نَ نُ جࣴڪّࣷونَ سࣷنّࣹ، اَڪْ نَ ݧُ نُ تࣷرْخَلࣹ وࣷونْ ݖَ فِلِبْ، دࣴڪّ بَ، بَلَ نࣷو دِڪّ فَ يࣹينْ؞ تࣹيوُلْ ݖِ سُنُ ندِمْبَلَلْ يࣵلَّ، نُ ݧࣹمࣹ لࣹينَ اࣵگّࣹ، ݖِ بِيرْ خࣹيخْ بُ مࣹتِّ سَخْ، خِبَارُ جࣵمّ بُ يࣵلَّ بِ؞ ‏3 ندَخْتࣹ سُنُبْ وࣷوتࣹ لَلࣷووُلْ ݒࣹخࣹمْ سࣵنڪࣹ، اَڪْ مࣺبࣺتْ مُ سࣹتُلْ، اَڪُگْ نجُبْلَݝْ؞ ‏4 خَنَا ڪَيْ نࣷونُ نُ يࣵلَّ دࣷيْلࣷو، بَ دࣺنڪّ نُ لِگّࣺييُ خِبَارُ جࣵمّ بِ، نࣷونُ لَنُيْ وَارࣹي لِ نࣹيخْ يࣵلَّ مِيْ سࣹيتْ اَبْ خࣷلْ، تࣹ دُنُ ݖِ وُتْ بࣵنّࣹيخُ نِتْ ݧِ؞
‏5 مَسُنࣷو جَيࣹ، خَمْ نگࣹينْ ڪࣷ، اَگْ بࣴگّࣹ اِتْ مَسُنࣷو تَخَ جٗگْ، يࣵلَّ سࣹيدࣹ نَ ڪࣷ؛ ‏6 مَسُنࣷو سࣵڪُّ نگࣴرࣴمَلْ نِتْ، دُ ݖِ يࣹينْ، دُ ݖِ ݧࣹنࣹينْ؞ ‏7 مَنࣷونْ نَنࣷو تَافَنْتࣷو لِ نُ دِيْ ندَوِ اَلْمَسِ، بَ دِيسَلْ لࣹينْ، وَايࣹ دَنࣷو لࣹوࣹتࣷونْ دࣷمّ ݖِ سࣹينْ بِيرْ، مࣹلْ نِ جَبࣷوتْ جُيْ تࣷݒّتࣷو اَيْ دࣷومَمْ؞ ‏8 نࣷونُ لَنُ نَبࣹ وࣷونْ ݖِ يࣹينْ، بَ بُگُّنُ وࣷونَ يࣹمْ ݖِ خَمَلْ لࣹينْ دࣷݝّ خِبَارُ جࣵمّ بُ يࣵلَّ، وَايࣹ دَنُ لࣹينَ سࣷݒّ بَ يࣺينࣹ وࣷونْ لࣹينَ جࣷخْ سُنُ بَڪَّنْ سَخْ بࣷولࣹ ݖِ، ندَخْ فِ نگࣹينْ نُ مُجّ تࣷلُّ ݖِ سُنُ خࣷلْ؞ ‏9 خَنَا بࣷڪّ يِ، يࣹينَ نگِ فࣵتّلِڪُ سُنُ دࣷݧْ-دࣷݧْ اَڪْ سُنُ ݖࣷونࣷ؟ دِيرْ بَ نُ لࣹينْ دِ يࣺگَلْ خِبَارُ جࣵمّ بُ يࣵلَّ بِ، گُدِّ اَڪْ بࣴݖّࣴگْ لَݧُ دَانْ لِگّࣺييَالࣹ، نگِرْ بَݧَ دِيسَلْ ڪࣹنّ ݖِ يࣹينْ؞
‏10 يࣹينَ سࣹيدࣹ، يࣵلَّ اِتَمْ سࣹيدࣹ، سࣹلّنگَ اَڪْ نجُبْ، اَڪْ مُݖّ سِڪّ گَ نُ دࣷونْ جࣴفࣹ فَ سࣹينْ بِيرْ، يࣹينْ ݧِ گࣴمْ؞ ‏11 خَمْ نگࣹينْ نࣹ فِ بَايْ فࣺيتࣹ دࣷومَمْ لَنُ فࣺيتࣹ ڪࣹنّ ڪُ نࣹڪّ ݖِ يࣹينْ؞ ‏12 ݧَاخْ نَنُ لࣹينْ، دࣴفَلْ لࣹينْ، دࣺنڪُ لࣹينْ، نگِرْ نگࣹينْ دِ جࣴفࣹ جࣴفِنْ وُ يࣹلّࣷو اَڪْ يࣵلَّ، مِ لࣹينْ وࣷو ݖِ نگُورَمْ اَڪْ تࣹرَانگَامْ؞
‏13 لࣹنࣹينْ لُ نُ دُلْ نࣷݒّࣹي سَنْتࣹ يࣵلَّ، مࣷو دِ لِي:‏ بَ نگࣹينْ نَنگࣷو ݖِ سُنُ لࣵمِّݧْ ڪࣵدُّگْ يࣵلَّ گِ نُ لࣹينْ دࣺگْتَلْ، تࣹ مࣹلُلْ نِ ڪࣵدُّگْ نِتْ دࣷݝّ نگࣹينْ ڪࣷ نَنگࣷو، وَايࣹ ڪࣵدُّگْ يࣵلَّ گِ مُ دࣷونْ ݖِ لُ وࣺرْ، مࣷومْ نگࣹينْ ڪࣷ جࣴلࣹ، تࣹ ڪࣵدُّ گࣷوگُ مࣷويْ لِگّࣺييْ ݖِ سࣹينْ بِيرْ، يࣹينْ گࣴمْڪَتْ ݧِ؞ ‏14 ندَخْ ڪَتْ بࣷڪّ يِ، يࣹينَ اَوْ ݖِ تࣵنڪِ گࣴمْڪَتِ اَلْمَسِ عِيسَىٰ يَ فَ مبࣷولࣷويْ يࣵلَّ، يَ فَ رࣺيوُمْ يُدَ، ندࣹ ݖࣷونࣷ يَ ݧُ دَجْ ݖَ يَوُتْ يَ، يࣹينْ اِتْ سࣹينِ وَا رࣺيوْ تࣹگْ نَݧُ لࣹينْ لُ نِ مࣹلْ؞ ‏15 يَوُتْ ݧِ ݧࣷو رࣹيْ سَنگّ عِيسَىٰ، نَ ݧُ رࣹيࣹ وࣷونْ يࣷنࣹنتّ يِ، تࣹ ݧࣷو نُ بُندّخَتَلْ تࣹيْ؞ نࣹيخُݧُ يࣵلَّ، تࣹ نࣷونِ ݧࣺݒّ لَݧُ؞ ‏16 ݧࣷو نُيْ آيࣹ نُ اࣵگّࣹ جَامْبُرْ ݧِ دُلْ يَوُتْ، ڪࣵدُّ گِ لࣹينْ دِ مُسَلْ؞ نࣷونُ لَݧُ مَتَلࣹ سࣹينْ جٗورُبْ بࣵڪَّارْ يِ ݧُ نࣹڪّࣹ نَڪَ جࣹڪّ، وَايࣹ اَمْ سࣵنجَ لࣹينْ مُجّ دَبْ؞
ݒٗولْ نَمّ نَ وَا تࣹسَلࣷنِگْ
‏17 نُنْ نَگْ بࣷڪّ يِ، تࣹقَلࣹ نَݧُ نُ اَڪْ يࣹينْ اَبْ دِيرْ، وَايࣹ خࣷلْ يِ سࣷرࣹوَاتࣷووُلْ، خَنَا جࣴمّ يِ رࣹڪّ؞ وَايࣹ سُنُ نَمّࣹيلْ گُ رࣺيْ تَخْ نَ نࣷو دࣹفْ سُنُ ڪࣹمْ ڪࣵتَّنْ، ندَخْ يࣵڪَّمْتࣹي گِسࣹ وَاتْ اَڪْ يࣹينْ؞ ‏18 مࣷو تَخْ نُ بࣴگّࣷونْ لࣹينَ سࣹيتْسِ، وَايࣹ سࣹيْتَانࣹ مࣷو نُ گࣵلَّنڪࣷورْ؞ تࣹ مَنْ ݒٗولْ، جࣺيمْ نَا لࣹينَ سࣹيتْسِ بُ دُلْ بࣹنّ يࣷونْ دُ ݧَارْ؞ ‏19 ندَخْ ڪَتْ، اَنَ ڪَنْ مࣷويْ سُنُ يَاڪَارْ، دِ سُنُ مبࣺگْ، تࣹ دِ سُنُ ڪَالَ گُ نُيْ سَگࣷوجِ؟ خَنَا دُ يࣹينْ، ڪࣹرَ ݖَ سُنُ ڪَنَمْ بࣷرࣷومْ عِيسَىٰ، بُ فࣹيݧࣹي؟ ‏20 يࣹينْ دࣺيْ، لَنُيْ دَمࣷو، دِ لࣹينْ بࣺگࣹ؞
