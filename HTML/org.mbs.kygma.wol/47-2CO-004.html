نجَقِ بَنِ ڪࣹسࣹ يࣹبْ نَ نگࣴنࣹيلِ اَلَلْ
4
‏1 ندࣹگَمْ بࣷوبُ لِگّࣺييْ لَنُ سَسࣷو، ݖِ يࣴرْمَنْدࣹيْ يࣵلَّ، دُنُ يࣷقِ؞ ‏2 وَايࣹ دَنࣷو دࣵقْ جࣺݒّ جࣴفْ جُ گࣵݖّࣹلُ جُ نِتْ دِ جࣴفࣹ لࣵقْ-لࣵقَلْ؞ دࣷخُنُ اَگْ نجُبْلَݝْ تࣹ سࣷݒَّرْݧِوُنُ ڪࣵدُّگْ يࣵلَّ؞ لِيْ دࣴگّ ڪَيْ لَنُيْ فࣺسَلْ، بَ اَمْ ڪٗولُتࣹگْ ݧࣺݒّ، فِ ڪَنَمْ يࣵلَّ؞ ‏3 سُنُ خِبَارُ جࣵمّ نَگْ، سُ مُورࣷو، ݧِ نَرَ سࣵنڪُ لَ مُورُلْ؞ ‏4 اَجِ گࣴمَدِ ݧࣷوݧُ لَ سࣹيْتَانࣹ يࣵلَّيْ خَرْنُ بِي گࣴلࣴمَلْ سࣹينُمْ خࣹلْ، نگِرْ ݧُ بَݧَ گِسْ ݖࣹيݧࣹيرُ خِبَارُ جࣵمّ بُ لࣹيرُ اَلْمَسِ، مِيْ مࣹلࣷڪَانُ يࣵلَّ؞ ‏5 نُنْ ڪَتْ، دُ سُنُ بࣷݒّ لَنُيْ وَارࣹ، وَايࣹ لِ عِيسَىٰ اَلْمَسِ دِبْ سَنگّ مࣷومْ لَنُيْ وَارࣹ، تࣹ نُنْ ݖِ سُنُ بࣷݒّ نࣷويْ سࣹينِ جَامْ، ندَخْ عِيسَىٰ؞ ‏6 ندَخْتࣹ يࣵلَّ مِ نࣹ:‏ «نَ لࣹيرْ گِ لࣹيرࣹ ݖِ بِيرْ لࣴنْدࣴمْ گِ،» مࣷو لࣹيرَلْ سُنُ خࣷلْ، نگِرْ لࣹيرَانگࣹيْ خَمْ-خَمُ لࣹيرُ يࣵلَّ، تࣹ لࣹيرَانگࣹ گࣷوگُ مࣷويْ لࣹرْخَتْ ݖِ خَرْ-ڪَنَمُ عِيسَىٰ اَلْمَسِ؞
‏7 نُنْ نَگْ نࣷويْ نجَقِ بَنِ ڪࣹسࣹ يِ يࣹبْ جࣷوجُ اَلَلْ، نگِرْ مُ بِرْ نࣹ مَنࣷورࣹ جُ يࣺيمࣹ جِ اَلَلْ جِ لࣵمْبࣷو، مَنࣷورࣹيْ يࣵلاَّ، وَايࣹ دُ سُنُ جࣷسْ؞ ‏8 نُ نࣹڪّ لࣹيسْ نُيْ تَنݖࣹ، وَايࣹ دࣹيسُ نُ نَݒَّاجࣹ؞ دِنَنُ جَاخْلࣹ، وَايࣹ دُنُ جࣵقْ؞ ‏9 دࣹيسْ نَ نُ بُندّخَتَلْ، وَايࣹ دࣹيسُ نُ وَݖّ؞ دࣹيسْ نَ نُ دَانࣹيلْ، وَايࣹ دࣹيسُ نُ جࣹڪّلِ؞ ‏10 دࣹي گِ عِيسَىٰ دࣹي وࣷونْ، مࣷومْ لَنُ اࣵنْدَلْ ݖِ سُنُ يَرَمْ، فُ نُ تࣷلُّ، نگِرْ دُنْدُگْ عِيسَىٰ اِتَمْ مَنَ فࣺسࣹ ݖِ سُنُ يَرَمْ؞ ‏11 سَا سُ نࣹڪّ، نُنْ ݧِيْ دُندّ، ݧُ نگِ نُيْ جࣺبَّلْ ندࣹي ندَخْ عِيسَىٰ، بَ تَخْ دُنْدُ عِيسَىٰ اِتْ مَنَ فࣺسࣹ ݖِ سُنُوْ سُوخْ وِ جࣴمْ ندࣹي؞ ‏12 ڪࣷنْ نَگْ ندࣹي مࣷويْ دࣹفْ لِگّࣺييَمْ ݖِ نُنْ، ندُندّ دِ دࣹفْ لِگّࣺييَمْ ݖِ يࣹينْ؞
‏13 بِنْدࣹيسْ نَ نࣹ:‏ «دَمَا گࣴمْ، مࣷو تَخْ مَ وَخْ¤؞» نُنْ اِتْ، وࣹنّ خࣹيتُ نگࣴمْ وِ لَنُ اَمْ؞ دَنࣷو گࣴمْ، مࣷو تَخْ اِتَمْ نُيْ وَخْ؞ ‏14 ندَخْتࣹ خَمْ نَنُ نࣹ ڪِ دࣹڪَّلْ سَنگّ عِيسَىٰ، مࣷو نُيْ دࣹڪَّلْ اِتَمْ، نࣷوڪْ عِيسَىٰ، تࣹ مࣷو نُيْ يٗبُّ بَ نُ تࣹيوِ، نࣷوڪْ يࣹينْ، فَ ڪَنَمَمْ؞ ‏15 لࣷولُ لࣺݒّ يࣹينَ تَخْ، نگِرْ يِوُ يࣵلَّ لَوَ لَوْ، بَ ݖَنتّ گُ نࣹ گࣵݧّ تُڪّࣹي ݖِ ݧِ اࣴݒّ ݖِ نِتْ ݧِ، نگِرْ مࣵگُّگْ يࣵلَّ؞
يَاڪَارْ لَنُ سَخࣷو
‏16 مࣷو تَخْ دُنُ ݧࣵڪّ يَاڪَارْ؞ دࣷونْتࣹ سَخْ سُنُ بِتِبْ جࣴمَّ نگِ گࣴنْ دِ بࣷنْ، سُنُ جࣴمُّ بِيرْ مࣷومْ، مُ نگِ يࣹيسْلُ بࣺسْ اَڪْ بࣺسْ؞ ‏17 سُنُ ݖࣷونࣷ بُ وࣷيࣷفْ بِي ݖِ اࣵدِّنَ، ݖِ دِگَّنْتࣹ بُ گࣵتّ نَگْ، مࣷو نُيْ جُرَلْ تࣹرَانگَ جُ دِيسَايْ بَ جࣺگِّ دَيࣷ، تࣹرَانگَ جُ سَخْ دࣵڪّ؞ ‏18 مࣷو وَرَلْ خࣷولُنُ ݖِ لِ بࣴتْ مَنَ گِسْ، وَايࣹ لِ بࣴتْ دُلْ گِسْ، ݖِ لَنُيْ خࣷولْ، ندَخْ لِ بࣴتْ مَنَ گِسْ دَيْ وࣺيْ، وَايࣹ لِ بࣴتْ دُلْ گِسْ مࣷو سَخْ بَ فࣵوّ؞
