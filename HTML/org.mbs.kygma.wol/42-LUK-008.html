جِوُ وُوتࣹ نَ اَگْ مُجْ
8
‏1 گَنَّاوْ لࣷولُ عِيسَىٰ وࣴرْ دࣴڪّࣷو دࣴڪّ بُ مَگْ اَڪْ بُ ندَوْ، دِ وَارࣹ، دِ يࣺگْلࣹ خِبَارُ جࣵمّ بُ نگُورُگْ يࣵلَّ، اࣵندّ اَڪْ فُڪِّ تَالِبࣹ يَاڪْ ݧَارْ، ‏2 ݧࣷومْ اَڪْ جِگࣺينْ ݧِي مُ فَجࣷونْ، بَ تࣵگَّلࣹ لࣹينْ اَڪْ رَبْ يُ بࣷنْ اَڪِ جَگَدِ:‏ مَرْيَامَ، مِ ݧُيْ وَخْ مَگْدَلَ، مَ ݧُ تࣵگَّلࣹيڪْ جُرٗومْ ݧَارِ رَبْ، ‏3 اَڪْ يࣷوَانَ جَبَرُ ڪُسَ، سَيْتُڪَتُبْ اࣹرࣷدّ؛ اَڪْ سُسَانَ، اَڪْ ݧࣹنࣹينْ جِگࣺينْ ݧُ بَرࣹ ݧُ دَانْ دِمْبَلࣹي ݧࣷومْ عِيسَىٰ سࣹينْ اَلَلْ؞
‏4 مبࣷولࣷو مُ مَگَ دَجࣹ وࣷونْ، جٗگࣹ دࣴڪّࣷو دࣴڪّ، فࣹڪّسِ عِيسَىٰ؞ مُ لࣺيبْ لࣹينْ، نࣹ لࣹينْ:‏ ‏5 «بࣷرࣷومْ جِوُ مࣷو دࣷونْ جِيِ؞ نَڪَ لَيْ سَاوْ اَبْ تࣷولَمْ، ندࣹڪࣹ لࣹنّ ݖَ جِوُ مَا نگَ وَدّࣹ ݖَ ڪَوْ يࣷونْ وَ، نِتْ ݧِ جࣷگَّتِ ݖَ، ݒِݖّ يِ دِڪّ، لࣹڪّ ڪࣷ؞ ‏6 لࣹنࣹينْ وَدّࣹ فُ بَرࣹيْ دࣷجْ؞ نَڪَ لَ سَخْ رࣹڪّ، دَلْدِ وࣷوْ، ندَخْ ݧࣵڪّ ندࣷخْ؞ ‏7 لࣹنࣹينْ لَ وَدّࣹ بِيرِ خَاخَامْ، جٗگَنْدࣷو اَڪْ خَاخَامْ مَ، دَلْدِ فَتّ بَ وࣷوْ؞ ‏8 لَ دࣹسْ ݖَ جِوُ مَ وَدّࣹ سُوفْ سُ نَنگُ، دَلْدِ سَخْ، بَ مࣹݧَّلْ لُ ڪࣷ مَتْ تࣺيمࣺيرِ يࣷونْ؞»
ݖِ ڪَوْ لࣷولُ عِيسَىٰ اࣵدُّ ݖَ ڪَوْ نࣹ:‏ «ڪُ اَمْ نࣷݒّ يُ مُ دࣺگّࣹ ڪَتْ، نَ دࣺگّ؞»
‏9 اَيْ تَالِبࣹيمْ لاَجْ ڪࣷ لُ لࣺيبْ وࣷووُ دِ تࣹڪِّ؞ ‏10 مُ نࣹ لࣹينْ:‏ «يࣹينْ لَݧُ مَيْ، نگࣹينْ خَمْ مبٗوتُمْ نگُورُگْ يࣵلَّ، وَايࣹ ݧِ ݖِ دࣹسْ ݧࣷومْ، ݖِ اَيْ لࣺيبْ لࣹيسْ ڪࣷيْ دࣴخࣴݧْ، سُ ڪࣷ دࣹفࣹي:‏
‹ݧُيْ خࣷولࣹ تࣹ دُݧُ گِسْ،
دِ دࣺگّ تࣹ دُݧُ خَمْ؞›
‏11 «لِي لَ لࣺيبْ وِ دِ تࣹڪِّ؞ جِوُ وِ مࣷو دِ ڪࣵدُّگْ يࣵلَّ؞ ‏12 ݧِ فࣺيتࣹيڪْ يࣷونْ وِ مࣷويْ ݧِ دࣺگّ؞ سࣹيْتَانࣹ نَگْ مࣷويْ دِڪّ، فࣷقَتِ ڪࣵدُّ گِ ݖِ سࣹينْ خࣷلْ، نگِرْ ݧُ بَݧَ گࣴمْ بَ مُݖّ؞ ‏13 ݧِ فࣺيتࣹ فُ بَرࣹ اَيْ دࣷجْ، ݧࣷو دࣺگّ ڪࣵدُّ گِ، نَنگࣷو ڪࣷ خࣷلْ بُ سࣹدّ؞ وَايࣹ ݧࣷوݧُ، اَبْ رࣹينْ لَݧُ اَمُلْ؞ اَبْ دِيرْ رࣹڪّ لَݧُيْ گࣴمْ، بُ جَنْتِ نَتُّ دِڪّࣹي، ݧُ دࣹلُّ گَنَّاوْ؞ ‏14 لِ وَدّࣹ ݖِ اَيْ خَاخَامْ مࣷويْ ݧِ دࣺگّ ڪࣵدُّ گِ، تࣹ اَيْ خَلاَتْ اَڪْ اَلَلْ، اَڪْ بࣵنّࣹيخُ بَڪَّنْ فَتّ لࣹينْ، بَ تَخْ مࣹݧَّلُݧُ لُ ݧࣷرْ؞ ‏15 لِ ݖِ سُوفْ سُ بَاخْ سِ نَگْ، مࣷويْ ݧِ دࣺگّࣹ ڪࣵدُّ گِ خࣷلْ بُ دࣴگُّ تࣹ رَفࣹتْ، ݝࣷيْ ݖَ، تࣹ سَخࣷو مُݧْ بَ مࣹݧَّلْ؞
اَبْ لࣵمݒّ دُ لࣵقُ
‏16 «دࣹيسُلْ تَالْ اَبْ لࣵمݒّ، دࣴݒّࣹ ڪࣷ ندَبْ، مبَا نگَ دُگَلْ ڪࣷ ݖِ رࣷنْ لَلْ؞ دࣹيسْ ڪࣷيْ اَجْ ڪَيْ بَ ڪُيْ دُگّسِ، دِ گِسْ لࣹيرْ گِ؞ ‏17 مبࣷولࣹمْ لُ لࣵقُ دِنَ فࣹيݧْ، تࣹ مبࣷولࣹمْ لُ نࣴبُّ دِنَ سِيوْ بَ نࣹ فࣵݝّ؞ ‏18 ڪࣷنْ نَگْ تࣹيوْلُلࣹينْ نِ نگࣹينْ دِ دࣺگّࣹ؞ ڪُ اَمْ، دࣹيسْ نَ ڪࣷ دࣷلِّ، وَايࣹ ڪُ اَمُلْ، لَ مُ خَلاَتࣷونْ نࣹ اَمْ نَ ڪࣷ سَخْ، دࣹيسْ نَ ڪࣷ نَنگُ؞»
دࣺگَّلْ يࣵلاَّ گࣴنْ بࣷڪّ اَڪْ عِيسَىٰ
‏19 يَايُ عِيسَىٰ اَڪْ اَيْ رَڪَّمْ ݧࣷو دِڪّࣷونْ دِ سࣹيتْ عِيسَىٰ تࣹ مَنُݧࣷو اَگْسِ ݖِ مࣷومْ ندَخْ نِتْ ݧُ بَرࣹ؞ ‏20 ݧُ نࣹ ڪࣷ:‏ «سَ يَايْ اَڪْ سَيْ رَڪَّ نگِ تَخَوْ ݖِ بِتِ، بࣴگّ لاَ گِسْ؞» ‏21 مُ نࣹ لࣹينْ:‏ «سَمَ يَايْ اَڪْ سَمَيْ رَڪّ، ݧࣷو دِ ݧِيْ دࣺگّ ڪࣵدُّگْ يࣵلَّ، تࣹ دِ ڪࣷ جࣴفࣹ؞»
عِيسَىٰ دَلَلْ نَ نگࣹلَوْ لِ
‏22 بࣹنّ بࣺسْ لَ عِيسَىٰ دُگّࣷونْ گَالْ، مࣷوڪْ اَيْ تَالِبࣹيمْ؞ مُ نࣹ لࣹينْ:‏ «نَنُ جࣵلّ ݖَ وࣵلاَّ دࣹخْ گِ؞» ݧُ دَلْدِ دࣹمْ؞ ‏23 نَڪَ لَݧُيْ دࣹمْ نَگْ، عِيسَىٰ نࣹلَوْ؞ نگࣹلَوْ لُ رࣺيْ دَلْ ݖَ ڪَوْ دࣹخْ گَ، گَالْ گِ فࣹيسْ اَڪْ ندࣷخْ، بَ ݧُ نَرَ سࣵنڪُ؞ ‏24 تَالِبࣹ يِ يࣹي عِيسَىٰ، نࣹ ڪࣷ:‏ «نجَاتِگࣹ، نجَاتِگࣹ، نُ نگِ سࣵنڪُ!‏»
مُ يࣹوُّ، گࣴدّ نگࣹلَوْ لِ اَڪْ گَنَّخْ يُ رࣺيْ يِ؞ لࣺݒّ دَلْ، نࣹ تࣹڪّ؞ ‏25 عِيسَىٰ نࣹ تَالِبࣹ يَ:‏ «اَنَ سࣹينْ نگࣴمْ؟» ݧَ نگَ بࣷولࣹ تِيتَانگࣹ اَڪْ ڪࣺيمْتَانْ تࣹ نَانْ ݖَ سࣹينْ بِيرْ:‏ «ڪِي نَگْ مࣷو دِ ڪَنْ بَيْ جࣷخْ ندِگَلْ نگࣹلَوْ لࣹيڪْ ندࣷخْ مِ سَخْ، ݧُ دࣺگَّلْ ڪࣷ؟»
عِيسَىٰ فَجْ نَ ڪُ رَبْ جࣵݒّ
‏26 ݖِ ڪَوْ لࣷولُ ݧُ تࣴمبّ بَ تࣹيرࣹجِ ݖَ دِيوَانُ وَا سࣹرَسَ، فَ جࣵڪَّارْلࣷو اَڪْ دِيوَانُ گَلِلࣹ؞ ‏27 بَ عِيسَىٰ وࣵݖّࣹي ݖَ تࣹفࣹسْ گَ، جࣹنّ وَايُ دࣴڪّ بَ دِڪّ دَجࣹيڪْ مࣷومْ؞ وَا جَ، اَيْ رَبَ ڪࣷ جࣵݒّ؞ يࣵگّ نَا دࣹفْ يَرَمُ نࣹينْ تࣹ دࣴڪُّلْ ݖِ ڪࣴرْ، خَنَا سࣴگْ يَ؞ ‏28 بَ مُ گِسࣹي عِيسَىٰ، دَا يُوخُ، دَانُ فَ ڪَنَمَمْ، دَلْدِ خَاݖُ نࣹ:‏ «يَوْ عِيسَىٰ دࣷومُ يࣵلَّ اَجِ ڪَوࣹ جِ، اَنَ لُ مَ جࣷتࣹيڪْ يَوْ؟ دَمَ لَيْ تِنُ، بُ مَ مبُگَلْ!‏»
‏29 بࣷوبَ عِيسَىٰ تࣵمْبَلِ نَا سَنتّ رَبْ وِ، مُ گࣺنّ وَا جِ؞ رَبْ وِ جࣵݒّ نَ وَا جِ اَيْ يࣷونِ يࣷونْ؞ سُ ڪࣷ دࣹفࣹي ݧُ يࣹيوࣹ ڪࣷ اَيْ ݖࣵلَّلَ، جࣺنگّ ڪࣷ، نگِرْ وَتُّ ڪࣷ، مُ دِ ڪࣷ دَگّ، رَبْ وِ دِ ڪࣷ خَبْتَلْ، جࣴمࣹ بࣺرَبْ يُ وࣺيتْ؞
‏30 عِيسَىٰ نࣹ ڪࣷ:‏ «نࣷو تُدّ؟» مُ نࣹ ڪࣷ:‏ «ݖࣷگَّلْ؞» ندَخْ رَبْ يُ بَرࣹي دُگّࣷونْ وَا جِ؞ ‏31 رَبْ يِ نَگْ تِنُ عِيسَىٰ، نگِرْ مُ بَݧْ لࣹينَ يࣹبَلْ ݖَ خٗوتࣹ بَ¤؞
‏32 اَگْ ݖࣷگَّلُ مبَامْ-خُوخْ يُ بَرࣹي نگَ وࣷونْ فࣷوفَ، دِ فࣷرࣹ ݖَ تُندّ وَ؞ رَبْ يَ تِنُ عِيسَىٰ، نࣹ مُ مَيْ لࣹينْ، ݧُ دُگّ مبَامْ-خُوخْ يَ؞ مُ مَيْ لࣹينْ ڪࣷ؞ ‏33 رَبْ يَ نَگْ گࣺنّ نِتْ ڪَ، دُگّ مبَامْ-خُوخْ يَ، ݖࣷگَّلْ گَ بَرْتَلࣷونْدࣷو، تࣵبِّ ݖَ دࣹخْ گَ، دَلْدِ لَبْ؞ ‏34 سࣵمّ يَ گِسْ لَ خࣹوْ، دَلْدِ دَوْ، نࣹتَّلِجِ ڪࣷ وَا دࣴڪّ بَ، اَڪْ دࣴڪّ-دࣴڪَّانْ يَ؞ ‏35 نِتْ ݧَ رِيرࣹ ݖَ، دِ سࣹيتْ لَ خࣹوْ؞
نَڪَ لَݧُ دِڪّ بَ ݖِ عِيسَىٰ، يࣹمْ ݖِ نِتْ ڪِ رَبْ وِ تࣹقْلِڪࣷولْ، مُ اࣵندّ اَڪْ سَگࣷومْ، سࣷلُ، تࣷوگْ ݖَ وࣹتُ عِيسَىٰ؞ ݧُ دَلْدِ تِيتْ؞ ‏36 ݧَ ڪࣷ فࣹڪّࣹ وࣷونْ نَگْ، نࣹتَّلِ لࣹينْ نَ بࣷرࣷومِ رَبْ بَ جࣷتࣹ وَلّ؞ ‏37 ݖِ ڪَوْ لࣷولُ مبࣷولࣷومْ دِيوَانُ سࣹرَسَ گࣺݒّ تِنُ عِيسَىٰ نگِرْ مُ جٗگࣹ ݖَ سࣹينْ بِيرْ تࣹ دࣹمْ ندَخْ تِيتَانگࣹ لُ رࣺيْ لُ لࣹينْ جࣵݒّ؞ نَڪَ لَ عِيسَىٰ دُگّ گَالْ، دِ دࣹمْ، ‏38 نِتْ ڪَ تࣹقْلِڪࣷوڪْ رَبْ يِ دِ ڪࣷ سَرْخُ نگِرْ اࣵندّ اَڪْ مࣷومْ؞ تࣹيوُلْ عِيسَىٰ يِوِ ڪࣷ، نࣹ ڪࣷ:‏ ‏39 «ݧِبِّلْ، تࣹ نگَ خَمْلࣹ لِ لَ يࣵلَّ دࣹفَلْ لࣺݒّ؞» وَا جَ اِتْ دࣹمْ دِ سِيوَلْ ݖَ دࣴڪّ بَ بࣺݒّ، مبࣷولࣹمْ لَ ڪࣷ عِيسَىٰ دࣹفَلْ؞
عِيسَىٰ فَجْ نَ، دࣹڪَّلْ نَ
‏40 بَ عِيسَىٰ دࣹلّࣷو ݖَ وࣵلاَّ دࣹخْ، مبࣷولࣷو مَا ڪࣷ تࣹيرْتُ، ندَخْ ݧࣷومْ ݧࣺݒَّ ڪࣷ دࣷونْ خَارْ؞ ‏41 ݖِ ڪَوْ لࣷولُ رࣹڪّ جࣹنّ وَايْ دِڪّ، ڪُ ݧُيْ وَخْ يَيْرُسْ، دِ نجِيتُ جࣵنگُ بَ؞ مُ نࣹ گُرُبْ فِ ڪَنَمْ عِيسَىٰ، سَرْخُ ڪࣷ نگِرْ مُ ݧࣴوْ ڪࣴرَمْ، ‏42 ندَخْ ڪࣹنّ لَ اَمࣷونْ ݖِ دࣷومْ، جَنقّ بُ اَمْ فُڪِّ اَتْ اَڪْ ݧَارْ، تࣹ مَ نگَ دࣷونْ وَاجَ دࣹي؞
عِيسَىٰ نگَيْ دࣹمْ، نِتْ ݧَ رَجْ-رَجْلࣷو، تَنݖّ ڪࣷ؞ ‏43 سࣹنّ اَسْ ندَوَ نگَ ݖَ وࣷونْ، دِيرُ فُڪِّ اَتْ اَڪْ ݧَارْ، مَ نگَ دࣷونْ خࣴݒّ دࣹرࣹتْ، تࣹ دࣹفْ نَ ݖِ اَلَلَمْ جࣺݒّ، ڪࣹنّ مَنُ ڪࣷو فَجْ؞ ‏44 مُ دࣷخࣹ عِيسَىٰ گَنَّاوْ، تࣹگْ لࣷخࣷومْ ݖَ ݖَتَلْ مبُبَّمْ، دࣹرࣹتْ جَ دَلْ ݖَ سَا سَ؞
‏45 عِيسَىٰ نَگْ نࣹ:‏ «اَنَ ڪُ مَ لاَلْ؟» ڪُ نࣹڪّ دِ مِيمْ؞ ݒِيࣹيرْ نࣹ ڪࣷ:‏ «نجَاتِگࣹ، مبࣷولࣷو مࣺݒَّ لَ يࣺيوْ، دَرْ لَ؞» ‏46 تࣹيوُلْ عِيسَىٰ نࣹ:‏ «اَمْ نَ ڪُ مَ لاَلْ ڪَيْ، ندَخْ يࣺگْ نَا لࣹيرْ گُ بࣵيِّڪࣷو ݖِ مَنْ؞» ‏47 ندَوْ سَ خَمْ نࣹ فࣹيݧْ نَ؞ مُ دَلْدِ دِڪّ، دِ لࣷخْ، نࣹ گُرُبْ فَ ڪَنَمْ عِيسَىٰ، دَلْدِ وَخْ ݖَ ڪَنَمْ ݧࣺݒّ لَ وَرَلْ مُ تࣹگْ ڪࣷ لࣷخࣷومْ، اَڪْ نَ مُ وࣺرࣹ ݖَ سَا سَ؞ ‏48 عِيسَىٰ نࣹ ڪࣷ:‏ «جَنقّ، سَ نگࣴمْ فَجْ نَ لَ؞ دࣹمَلْ اَڪْ جࣵمّ؞»
‏49 نَڪَ لَ عِيسَىٰ دِ وَخْ، جࣹنّ وَايْ جٗگࣹ ݖَ ڪࣴرْ نجِيتُ جࣵنگُ بَ، دِڪّ نࣹ:‏ «مَتَتُلْ نگَيْ سࣷنَلْ سࣴرِݧْ بِ، سَ جَنقّ بَ فَاتُ نَ؞» ‏50 عِيسَىٰ دࣺگّ ݖَ، نࣹ نجِيتْ لَ:‏ «بُلْ تِيتْ؞ گࣴمَلْ رࣹڪّ، مُ دَلْدِ مُݖّ؞»
‏51 بِ عِيسَىٰ اَگْسࣹي ݖَ ڪࣴرْ گَ، مَيُلْ ڪࣹنّ اࣵندّ اَڪْ مࣷومْ دُگّ، ڪُ مࣷيْ ݒِيࣹيرْ، اَڪْ يࣷوَانْ اَڪْ يَعْقُوبَ اَڪْ بَايُ خَلࣹ بِ، اَڪْ ندࣹيَمْ؞ ‏52 ݧࣺݒَّ نگَيْ جࣷويْ اَڪَ تِيسْلُ جَنقّ بِ؞ عِيسَىٰ نَگْ نࣹ لࣹينْ:‏ «بُلࣹينْ جࣷويْ؞ جَنقّ بِ دࣹيوُلْ، دَيْ نࣹلَوْ رࣹڪّ؞» ‏53 ݖِ ڪَوْ لࣷولُ ݧُ دِ ڪࣷ ڪٗڪَّلِ، ندَخْ خَمࣷونْ نَݧُ خࣺلّ نࣹ دࣹي نَ؞ ‏54 عِيسَىٰ اِتَمْ جࣵݒّ ݖَ لࣷخࣷلْ جَنقّ بَ، دَلْدِ اࣵدُّ ݖَ ڪَوْ، نࣹ ڪࣷ:‏ «جَنقّ، جٗگَلْ؞» ‏55 نࣷو گَ دࣺلْسِ، مُ نࣹ ݧࣷڪّࣹتْ؞ عِيسَىٰ نࣹ ݧُ مَيْ ڪࣷ، مُ لࣹڪّ؞ ‏56 وَاجُرِ خَلࣹ بِ نَگْ وَارُ؞ تࣹيوُلْ عِيسَىٰ دࣺنڪُ لࣹينْ، نࣹ لࣹينْ بُ ݧُ وَخْ ڪࣹنّ لَ خࣹوْ؞
