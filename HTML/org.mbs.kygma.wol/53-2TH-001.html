ݧَارࣹيلُ بَتَاخَلْ بِ يࣵلَّ مَيْ ݒٗولْ، مُ بِندّ ڪࣷ وَا
تࣹسَلࣷنِگْ
1
‏1 مَنْ لَ ݒٗولْ، مِ اࣵندّ اَڪْ سِلْوَنْ اَڪْ تِمࣷتࣹ ݖِ ڪࣵدُّ گِي مَ يࣺڪَّتِ، ݧࣹيلْ مبࣷولࣷومْ گࣴمْڪَتِ وَا تࣹسَلࣷنِگْ، ݧِ بࣷڪّ ݖِ يࣵلَّ سُنُ بَايْ، اَڪْ سَنگّ عِيسَىٰ اَلْمَسِ؞ ‏2 اَوْ يِوْ اَڪْ جࣵمّ ݧࣹيلْ نَ لࣹينْ، تُڪّࣹي ݖِ يࣵلَّ سُنُ بَايْ، اَڪْ سَنگّ عِيسَىٰ اَلْمَسِ؞
ݖَنتّ وَرْ نَ ݖِ وَا تࣹسَلࣷنِگْ
‏3 بࣷڪّ يِ، مَنُنࣷو ݧࣵڪَّ سَنتّ يࣵلَّ ݖِ يࣹينْ فُ نُ تࣷلُّ، تࣹ مُيْ لُ جَادُ، گَنَّاوْ سࣹينْ نگࣴمَ نگِ يࣷڪُّ بُ بَاخَ بَاخْ، تࣹ ݖࣷفࣹيلُ ڪُ نࣹڪّ ݖِ يࣹينْ ݧࣺݒّ دِ گࣴنَتࣹي لَوْ، دِگَّنْتࣹيمْ اَڪْ مࣷرࣷومَمْ؞ ‏4 مࣷو تَخْ نُنْ، نُ دِ لࣹينْ دَمࣷو فِ بِيرْ مبࣷولࣷويْ گࣴمْڪَتِ يࣵلَّ يِ، ندَخْ سࣹينُگْ مُݧْ، اَڪْ سࣹينْ نگࣴمْ، ݖِ بِيرْ مبࣷولࣹمْ نِ ݧُ لࣹينْ دِ بُندّخَتَلࣹ، اَڪْ نجࣵقَرࣹ يِ نگࣹينْ دِ جࣵنڪࣷونْتࣹيلْ؞
نجࣹڪّ فࣹيݧْ نَ ݖِ اࣵتّࣹبْ يࣵلَّ
‏5 لࣷولُ لࣺݒّ دِ فِرْندࣹيلْ نجࣹڪُّ اࣵتّࣹبْ يࣵلَّ، بِ لࣹينْ دِ يࣹلَّلࣹ نگُورُگْ يࣵلَّ، گِ تَخْ نگࣹينْ دِ سࣷنّ؞ ‏6 تࣹ ڪَتْ نجࣹڪُّ يࣵلَّ دِنَ فࣹيݧْ ندَخْ ݖࣷونࣷ بَ مُيْ دࣹلّࣷو ݧِ لࣹينْ تࣹگْ اَبْ ݖࣷونࣷ؞ ‏7 دِنَ فࣹيݧْ اِتْ ندَخْ نࣷفْلاَيْ بَ مُ نُيْ بࣷولࣹيڪْ يࣹينْ ݧِيْ دࣺڪُّ اَبْ ݖࣷونࣷ، سࣺدّࣹ نُ ڪࣷ، ڪࣹرَ بࣺسْ بُ سَنگّ عِيسَىٰ فࣹيݧࣹي، اࣵندّ اَڪْ اَيْ مَلاَڪَامْ يُ لࣵمْبࣷو مَنࣷورࣹيمْ، وࣵݖّࣹ اَسَمَانْ؞ ‏8 ݖِ بِيرْ سَوَرَ وُيْ سࣴلْسࣴلِ لَيْ دࣷونْ؞ ݖَ لَيْ مبُگَلْ ݧِ خَمُلْ يࣵلَّ، اَڪْ ݧِ بَݧَ دࣺگَّلْ خِبَارُ جࣵمّ بِ جࣴمْ ݖِ سُنُ بࣷرࣷومْ عِيسَىٰ؞ ‏9 ݧࣷوݧُ لࣹيسْ دِ مبُگَلࣹ سࣵنڪُتࣹ گُ سَخْ دࣵڪّ، فُ سࣷرࣹ بࣷرࣷومْ بِ، سࣷرࣹ تࣹدّنگَيْ مَنࣷورࣹيمْ، ‏10 ڪࣹرَ بࣺسْ بُ دࣺلْسࣹي بَ جࣷتْ ݖِ تࣹدّنگَامْ ݖِ بِيرْ ݧࣷݧَمْ ݧُ سࣹلّ ݧِ، بَ مبࣷولࣹمْ ݧِ ڪࣷ گࣴمْ دِ ڪࣷ يࣺيمࣷو؞ تࣹ يࣹينْ ݖِ سࣹينْ بࣷݒّ، ݖِ نگࣹينْ دِ بࣷڪّ، ندَخْ يࣹينَ گࣴمْ لَ نُ لࣹينْ سࣹيدࣹيلࣷونْ؞
‏11 مࣷو تَخْ نُ سَخࣷو لࣹينَ ݧَانَلْ، نگِرْ سُنُ يࣵلَّ يࣹلَّلࣹ لࣹينْ اَبْ وࣷوتࣹيمْ، نگِرْ اِتَمْ مُ سࣷتَّلࣹ ڪࣵتَّنَمْ سࣹينْ مبࣷولࣹمْ مࣺبࣺتْ مُ بَاخْ، اَڪْ سࣹينْ جࣺݒّ جࣴفْ جُ لَلࣷو نگࣴمْ؞ ‏12 سُ ڪࣷ دࣹفࣹي سُنُ تُرُ سَنگّ عِيسَىٰ مࣵگّࣹ ݖِ يࣹينْ، يࣹينْ اِتْ، نگࣹينْ مࣵگّࣹ ݖِ مࣷومْ، ݖِ سُنُ ندِمْبَلَلْ يِوُ يࣵلَّ، اَڪْ سَنگّ عِيسَىٰ اَلْمَسِ؞
