لُ جࣴمْ ݖِ ݒࣹيُگْ بࣵڪَّارْ
20
‏1 اَجِ سَخْ جِ وَخْ نَ مُوسَىٰ بَ تࣹيْ نࣹ ڪࣷ:‏ ‏2 «وَخَلْ بࣵنِّ إِسْرَاءِيلْ نࣹ لࣹينْ:‏ مُيْ بࣵنِّ إِسْرَاءِيلْ مُيْ دࣷخَنْدࣺيمْ ݖِ إِسْرَاءِيلْ، ڪࣺݒّ ڪُ ݖِ جࣴلْ ݖِيْ دࣷومَمْ، جࣺبَّلْ تُورْ مِ ݧُيْ وَخْ مࣷلࣹگْ، دࣹي رࣹڪّ مࣷويْ اࣵتّࣹيمْ؞ نَ ڪࣷ وَا رࣺيوْ مِ دٗورْ اَيْ دࣷجْ، بَ مُ دࣹي؞ ‏3 مَنْ ݖِ سَمَ بࣷݒّ دِنَا نࣷونࣷو نِتْ ڪࣷوڪُ تࣹ دِنَا ڪࣷ دَگّࣹ ݖِ بِيرْ بࣵنِّ إِسْرَاءِيلْ، ندَخْ لِ مُ جࣴلْ ݖِيْ دࣷومَمْ جࣷخْ مࣷلࣹگْ، دِ سࣷبࣹيلْ سَمَ بࣺرَبْ بُ سࣹلّ تࣹيْ تࣹدَّدِلْ سَمَ تُرْ وُ سࣹلّ؞ ‏4 سُ فࣹڪّࣹي نࣹ وَا رࣺيوْ مِ دَݧُ ڪࣷو سࣹيتَانْ بَ سࣹيتَانْ ڪࣷ، مُيْ جࣷخْ مࣷلࣹگْ ڪُ بࣷڪّ ݖِيْ دࣷومَمْ تࣹ تَخُلْ ݧُ رࣹيْ ڪࣷ، ‏5 مَنْ ݖِ سَمَ بࣷݒّ مَايْ نࣷونࣷو نِتْ ڪࣷوڪُ مࣷوڪْ اَيْ بࣷڪَّمْ؛ دِنَا دَگّࣹ نِتْ ڪࣷوڪُ ݖِ بِيرْ بࣵنِّ إِسْرَاءِيلْ، گࣺنّࣹ ڪࣷ، مࣷوڪْ مبࣷولࣹمْ ݧِ مُ اࣵنْدَلْ تࣷݒّ مࣷلࣹگْ دِ بࣷڪَّالࣹ؞ ‏6 ڪُ جُبْلُ ݖِ ݧِيْ گِسَانࣹ نِتْ ݧُ دࣹي مبَا مُ جُبْلُ ݖِ بࣷرࣷومْ رَوَانْ يِيْ سࣹيتْ، تࣷݒّ لࣹينْ، دِ مَ لࣹينْ بࣷڪَّالࣹي، دِنَا ڪࣷ نࣷونࣷو مࣷومْ اِتَمْ تࣹ دِنَا ڪࣷ دَگّࣹ ݖِ بِيرْ بࣵنِّ إِسْرَاءِيلْ؞ ‏7 سࣹلّ-لُلࣹينْ تࣹ سࣹلّ، نگِرْ مَنْ اَجِ سَخْ جِ مَايْ سࣹينْ يࣵلَّ؞ ‏8 نَنگࣹينْ سࣵمّ سَمَيْ دࣷگَلِ يࣷونْ تࣹ دِ ڪࣷ جࣴفࣹ؞ مَنْ اَجِ سَخْ جِ مَايْ ڪِ لࣹينْ سࣹلَّلْ؞
‏9 «ڪࣺݒّ ڪُ سَاگَ بَايَمْ مبَا ندࣹيَمْ، دࣹي رࣹڪّ مࣷويْ اࣵتّࣹيمْ؞ گَنَّاوْ بَايَمْ لَ سَاگَ مبَا ندࣹيَمْ، مࣷويْ گࣵدُّ بَڪَّنُ بࣷݒَّمْ؞
‏10 «گٗورْ گُ نجَالࣷو، مࣷوڪْ جَبَرُ مࣷرࣷومَمْ جِ مُيْ نجَالࣷولْ، دࣹي رࣹڪّ مࣷويْ سࣹينْ اࣵتّࣹ؞
‏11 «ڪُ دࣴڪّࣷو سَ سࣷخْنَسْ بَايْ، سَ بَايْ نگَ تࣷرْخَلْ؞ ݧࣷومْ ݧَارْ ݧࣺݒّ دࣹي رࣹڪّ مࣷويْ سࣹينْ اࣵتّࣹ، تࣹ ݧࣷويْ گࣵدُّ سࣹينْ بَڪَّنُ بࣷݒّ؞ ‏12 ڪُ دࣴڪّࣷو سَ سࣷخْنَسْ دࣷومْ، مࣷوڪْ سࣷخْنَسْ دࣷومَمْ ݧࣷومْ ݧَارْ ݧࣺݒّ دࣹي مࣷويْ سࣹينْ اࣵتّࣹ؞ جࣴفُ يࣵقُتࣹ لَݧُ دࣹفْ، ڪࣷنْ ݧࣷويْ گࣵدُّ سࣹينْ بَڪَّنُ بࣷݒّ؞
‏13 «سُ گٗورْ تࣴدّࣹي گٗورْ نِ ݧُيْ تࣴدّࣹي جِگࣺينْ، ݧَاوْتࣺيفْ لَݧُ دࣹفْ ݧࣷومْ ݧَارْ؞ دࣹي رࣹڪّ مࣷويْ سࣹينْ اࣵتّࣹ، تࣹ ݧࣷويْ گࣵدُّ سࣹينْ بَڪَّنُ بࣷݒّ؞ ‏14 ڪُ بࣷولࣹ نِتْ اَڪْ ندࣹيَمْ جࣴلْ لࣹينْ جَبَرْ، دࣹفْ نَ لُ بࣷنَ بࣷنْ؞ دࣹيسْ لࣹينْ دِ بࣷولࣹ تَالْ، لَڪّ لࣹينْ، مࣷومْ اَڪْ جِگࣺينْ ݧَ، ندَخْ مبࣷنْ گُ نِ مࣹلْ بَݧَ اَمْ ݖِ سࣹينْ بِيرْ؞ ‏15 گٗورْ گُ تࣴدّࣹ مَلَ دࣹي رࣹڪّ مࣷويْ اࣵتّࣹيمْ، تࣹ مَلَ مَ اِتْ دࣹيسْ ڪࣷيْ رࣹيْ؞ ‏16 سُ جِگࣺينْ جࣴمࣹي ݖِ مَلَ مُ مُ دࣷونْ، بَ جَخَسࣷوڪْ مࣷومْ، بࣷولࣹلࣹينْ ڪُ جِگࣺينْ ڪَاڪْ مَلَ مَ، رࣹيْ؞ دࣹي رࣹڪّ مࣷويْ سࣹينْ اࣵتّࣹ، تࣹ ݧࣷويْ گࣵدُّ سࣹينْ بَڪَّنُ بࣷݒّ؞
‏17 «سُ نِتْ جࣴلࣹي جَبَرْ اَبْ جِگࣺينَمْ بُ مُ بࣷڪَّلْ جࣹنّ ندࣹيْ مبَا بࣹنّ بَايْ، ݧُ سࣺقْ لَلُ سࣺيْ، لࣷولُ گࣵݖّࣹ لَ؞ نَݧُ لࣹينْ دَگّࣹ ݖِ بِيرْ بࣵنِّ إِسْرَاءِيلْ، ݧࣷوݧُيْ گِسْ؞ گَنَّاوْ مࣷو تࣴدّࣹ اَبْ جِگࣺينَمْ، نَ وࣺيتࣷو اَݧُ ݧَاوْتࣺيفَمْ؞
‏18 «ڪُ تࣴدّࣹ جِگࣺينْ جُ گِسْ بَاخْ، تࣷرْخَلْ نَ ڪࣷ، خࣵوِّ نَ سُتُرَامْ ݖِ بِيرْ ݒٗوتَمْ تࣹ ڪُ جِگࣺينْ ڪِ اِتَمْ خࣵوِّ نَ سُتُرَسْ بࣷݒَّمْ؞ دࣹيسْ لࣹينْ وَرَ دَگّࣹ ݧࣷومْ ݧَارْ ݧࣺݒّ ݖِ بِيرْ بࣵنِّ إِسْرَاءِيلْ؞
‏19 «بُلࣹينْ دࣴڪّࣷو سࣹينْ يَايْ جُ ندَوْ مبَا سࣹينْ بࣵجَّنْ، ندَخْ ڪࣷنْ تࣷرْخَلْ نگࣹينْ ڪُ نگࣹينْ جࣹگࣹوࣷولْ لࣷولْ ݖِگْ بࣷڪّ، تࣹ يࣹينْ ݧَارْ ݧࣺݒّ اَيْ وࣺيتࣷو سࣹينْ اَݧُ ݧَاوْتࣺيفْ؞
‏20 «ڪُ دࣴڪّࣷو سࣷخْنَسْ بَايَمْ بُ ندَوْ، بَايَمْ بُ ندَوْ لَ تࣷرْخَلْ؞ ݧࣷويْ وࣺيتࣷو سࣹينْ اَݧُ ݧَاوْتࣺيفْ؞ دُݧُ اَمْ تَڪّندࣹيرْ بَ ڪࣹرࣷوگْ ݧُيْ دࣹي؞
‏21 «بُ نِتْ نَنگࣷو جَبَرُ ڪُ مُ بࣷڪَّلْ وَاجُرْ، لࣷولُ گࣵݖّࣹ لَ؞ تࣷرْخَلْ نَ ڪَ مُ بࣷڪَّلْ وَاجُرْ تࣹ دُݧُ اَمْ تَڪّندࣹيرْ؞
‏22 «سࣵمّلࣹينْ سَمَ دࣷگَلِ يࣷونْ يࣺݒّ اَڪْ سَمَ سَنْتَانࣹ يࣺݒّ تࣹ جࣴفࣹ ڪࣷ، نگِرْ رࣺيوْ مَ مَ لࣹينْ جࣴمࣹ، نگࣹينْ دࣴڪِّ فَ، بَݧْ لࣹينَ گࣴقْ؞ ‏23 بُلࣹينْ رࣷيْ دࣹفِنِ خࣹيتْ وِ مَ لࣹينْ دِ دࣵقَلْ، ندَخْ يࣷويُ يࣺݒّ لَݧُ دࣷونْ دࣹفْ، بَ مَ سِبْ لࣹينْ؞ ‏24 يࣹينْ نَگْ مَ نࣹ لࣹينْ يࣹينَيْ نَنگُ سࣹينْ رࣺيوْ ݧࣷومْ، ندَخْ مَا لࣹينْ ڪࣷ جࣷخْ، نگࣹينْ مࣷومْ ڪࣷ، مُيْ رࣺيوْ مُ مࣹيوْ مَاڪْ لࣹمْ جَ تُورࣷو؞ مَنْ مَايْ سࣹينْ يࣵلَّ اَجِ سَخْ، جِ لࣹينْ رࣵݧَّتْلࣹيڪْ يࣹنࣹينْ خࣹيتْ يِ؞ ‏25 ڪࣷنْ نَنگࣹينْ رࣵݧّࣹي مَلَ مُ دَگَنْ اَڪْ مُ دَگَنُلْ، اَڪْ اِتَمْ نجَنَاوْ لُ دَگَنْ اَڪْ لُ دَگَنُلْ، تࣹ بَݧَ سࣷبࣹيلْ سࣹينْ بࣷݒّ ندَخْ مَلَ مبَا نجَنَاوْ مبَا مبࣷولࣹمْ بِنْدࣹيفْ يُ سࣹوْ-سࣹوَانْ يِيْ دࣷخْ ݖِ سُوفْ تࣹ مَ لِمَلْ لࣹينْ نࣹ سࣹتُلْ؞
‏26 «نَنگࣹينْ مَ نࣹڪَّلْ ݧُ سࣹلّ، ندَخْتࣹ ڪُ سࣹلّ لاَ، مَنْ اَجِ سَخْ جِ، تࣹ دَمَ لࣹينَ رࣵݧَّتْلࣹيڪْ يࣹنࣹينْ خࣹيتْ يِ، نگࣹينْ دِ سَمَ ݧࣷݧْ؞
‏27 «گٗورْ گُ بࣷڪّ ݖِ يࣹينْ مبَا جِگࣺينْ، دِ سࣹيتࣹ نِتْ ݧُ دࣹي مبَا مُيْ بࣷرࣷومْ رَوَانْ بُيْ گِسَانࣹ، دࣹي رࣹڪّ مࣷويْ اࣵتّࣹيمْ؞ دࣹيسْ ڪࣷيْ دٗورْ اَيْ دࣷجْ، مُ دࣹي، گࣵدُّ بَڪَّنُ بࣷݒَّمْ؞»
