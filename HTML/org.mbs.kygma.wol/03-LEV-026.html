يࣷولْ يِ ݖِ دࣺگَّلْ يࣵلَّ
26
‏1 «بُلࣹينْ سࣵڪَّلْ سࣹينْ بࣷݒّ اَيْ يࣵلَّنْتُ، دُ لࣹنّ لُيْ جࣴمُّ تُورْ مبَا تُورُ دࣷجْ وُ نگࣹينْ سَمْݒَلْ سࣹينْ بࣷݒّ، تࣹ دࣷجْ وُ ݧُ ݧَاسْ اَيْ نَتَالْ اِتْ، بُلࣹينْ ڪࣷ سࣵڪَّلْ بࣺرَبْ ݖِ سࣹينْ رࣺيوْ، دِ ڪࣷ سُجّٗوتَلْ، ندَخْ مَنْ اَجِ سَخْ جِ مَايْ سࣹينْ يࣵلَّ؞ ‏2 سَمَ بࣺسِ نࣷفْلاَيْ، وࣷرْمَالْ-لࣹينْ ڪࣷ، تࣹ سَمَ بࣺرَبْ بُ سࣹلّ نگࣹينْ تࣹرَلْ ڪࣷ؞ مَايْ اَجِ سَخْ جِ؞
‏3 «سُ نگࣹينْ دࣹي تࣷݒّ سَمَيْ دࣷگَلْ، فࣷنڪّ سَمَيْ سَنْتَانࣹ تࣹ دِ ڪࣷ جࣴفࣹ، ‏4 ڪࣷنْ مَا لࣹينْ دِ تَوَلْ ݖِ وَخْتࣷومْ، سُوفْ سِ نَنگُ، گَرَبْ يِ مࣹݧّ؞ ‏5 دِنگࣹينْ بࣵݖّ بَ ڪࣹرࣷوگْ وِتُّمْ رࣹسࣹݧْ جࣷتْ تࣹ دِنگࣹينْ وِتّ رࣹسࣹݧْ بَ ڪࣹرࣷوگْ جِ جࣷتْ؞ دِنگࣹينْ لࣹڪّ سࣹينُمْ ݒࣹݒّ بَ سُورْ ڪࣴلّ تࣹ دِنگࣹينْ دࣴڪّ اَڪْ جࣵمّ ݖِ سࣹينُمْ رࣺيوْ؞
‏6 «مَايْ بَاخࣹي رࣺيوْ مِ جࣵمّ اِتْ، بَ بُ نگࣹينْ تࣴدّࣹي ڪࣹنّ دُ لࣹينْ تِيتَلْ؞ مَايْ جࣴلࣹ رَبْ وُ آيْ ݖِ رࣺيوْ مِ، تࣹ سَامَرْ دُ جٗگْ، دَلْ ݖِ سࣹينُمْ رࣺيوْ؞ ‏7 ڪࣷنْ دِنگࣹينْ دࣵقْ سࣹينِ نࣷونْ، سَامَرْ بَ دَلْ، ݧُ دَانُ فِ سࣹينْ ڪَنَمْ؞ ‏8 سࣹينْ جُرٗومِ نِتْ اَيْ دࣵقْ تࣺيمࣺيرْ، سࣹينْ تࣺيمࣺيرِ نِتْ دࣵقْ فُڪِّ جُنِّ؞ سࣹينِ نࣷونْ لَ سَامَرْ دِ دَلْ، ݧُ دَانُ فِ سࣹينْ ڪَنَمْ؞
‏9 «تࣹ اِتَمْ مَا لࣹينْ دِ بَاخࣹ، گِيرَلْ لࣹينْ، يࣷڪّ لࣹينْ؞ دِنَا سࣵمّ سَمَ ڪٗلّࣴرࣹيڪْ يࣹينْ؞ ‏10 دِنگࣹينْ لࣹڪّ نگٗوبْ مُ يࣵگّ، دࣹنݖࣹ لُ يࣵگّ، بَ فَرْ گࣺنّࣹ مُ يࣵگّ مَ، وُوتَلْ فَ مُ بࣹيسْ؞
‏11 «مَايْ سَمݒّ سَمَ ڪࣴرْ ݖِ سࣹينْ بِيرْ تࣹ دُمَ لࣹينْ سِبْ؞ ‏12 مَايْ اࣵندّ اَڪْ يࣹينْ، دِ سࣹينْ يࣵلَّ، نگࣹينْ دِ سَمَ ݧࣷݧْ؞ ‏13 مَنْ اَجِ سَخْ جِ مَايْ سࣹينْ يࣵلَّ جِ لࣹينْ گࣺنّࣹ رࣺيوُمْ مِسْرَ، نگࣹينْ نࣷݒّࣹي دࣷونْ سࣹينِ جَامْ؛ مَا دࣷگْ سࣹينْ بُومِ نجَامْ، سِگِّلْ لࣹينْ؞
مبُگَلْ يِ ݖِ دࣺگَّدِلْ يࣵلَّ
‏14 «وَايࣹ بُ نگࣹينْ مَ دࣺگَّلُلْ تࣹ بَݧَ جࣴفࣹ سَمَ سَنْتَانࣹ يࣷويُ يࣺݒّ، ‏15 سُ نگࣹينْ تࣹدَّدِلࣹي سَمَيْ دࣷگَلْ، دࣴدُّ سَمَيْ ندِگَلْ، بَ جࣴفࣹوُلࣹينْ سَمَ سَنْتَانࣹ يࣺݒّ، خَنَا دِ فࣹݖِّ سَمَ ڪٗلّࣴرࣹيڪْ يࣹينْ، ‏16 سُ بࣷوبَا لِ مَيْ دࣹفْ اَڪْ يࣹينْ مࣷويْ لِي:‏ مَايْ جࣹڪِّ وࣵݖّࣹ نجࣵقَرࣹ ݖِ سࣹينْ ڪَوْ، مُ اࣵندّ اَڪْ وࣷݒِّ رࣵگّ، يَرَمْ وُ تࣵنگّ جࣺرّ اَڪْ بࣴتْ يُ لࣴنْدࣴمْ، نگࣹينْ گࣴنْ دِ وࣷݒّ؞ دِنگࣹينْ جِ سࣹينُمْ ݒࣹݒّ ݖِ نࣹينْ، ندَخْ سࣹينِ نࣷونَ ڪࣷيْ لࣹڪّ؞ ‏17 مَا لࣹينْ دِ نࣷونࣷو بَ سࣹينِ نࣷونْ دُمَ لࣹينْ، سࣹينِ بَݧْ يِلِفْ لࣹينْ؞ دِنگࣹينْ دَوْ تࣹ ڪࣹنّ دࣵقُ لࣹينْ؞
‏18 «بُ لࣷولُ تَخُلْ نگࣹينْ دࣺگَّلْ مَ، مَا لࣹينْ دِ مبُگَلْ لُ ڪࣷ اࣴݒّ جُرٗومْ ݧَارِ يࣷونْ، فࣹيࣹ لࣹينْ ڪࣷ سࣹينِ بࣵڪَّارْ؞ ‏19 مَايْ سࣵگِّ سࣹينْ سَگْ بُ رࣺيْ؞ مَايْ دࣹفْ اَسَمَانْ سِ لࣹينْ تِيمْ نࣹ سࣹرࣹݝْ نِگْ وࣹݧْ، سُوفْ سِ وࣷوْ ڪࣷݝّ نِ خࣵنجَرْ؞ ‏20 سࣹينْ دࣷولࣹ دِنَ ڪَسَرَ، ندَخْ سُوفْ سِ نگࣹينْ دِ بࣹيْ دُ لࣹينْ نَنگُلْ تࣹ گَرَبِ رࣺيوْ مِ دُ مࣹݧّ؞
‏21 «سُ نگࣹينْ سَخࣹي ݖِ نࣷونࣷو مَ، بَݧْ مَا دࣺگَّلْ، مَايْ فُلاَتْ سࣹينْ مبُگَلْ فُڪِّ يࣷونْ اَڪْ جُرٗومْ، فࣹيࣹ لࣹينْ ڪࣷ سࣹينِ بࣵڪَّارْ؞ ‏22 مَايْ خَبْتَلْ رَبِ اࣵلّ يِ ݖِ سࣹينْ ڪَوْ، ݧُ خَݧْ لࣹينْ سࣹينِ دࣷومْ، سࣹينِ گࣺتّ ݧُ تَسْ، سࣹينِ نِتْ ݧُ نࣺيوَلْ، بَ سࣹينِ مبࣹدّ نࣹ وࣴيࣴݝْ؞
‏23 «سُ مَ لࣹينْ لࣷولُ يَرَلُلْ بَ تࣹيْ، خَنَا نگࣹينْ سَخْ ݖِ نࣷونࣷو مَ، ‏24 سُ بࣷوبَا مَنْ اِتَمْ مَا لࣹينْ دِ نࣷونࣷو، تࣹ اِتْ مَنْ ݖِ سَمَ بࣷݒّ مَا لࣹينْ دِ مبُگَلاَتْ لُ ڪࣷ اࣴݒّ فُڪِّ يࣷونْ اَڪْ جُرٗومْ، فࣹيࣹ لࣹينْ ڪࣷ سࣹينِ بࣵڪَّارْ؞ ‏25 مَايْ دَلَلْ سَامَرْ ݖِ سࣹينْ ڪَوْ، فࣹيࣹ لࣹينْ ڪࣷ ڪٗلّࣴرࣹ گِ نگࣹينْ فࣹݖِّ؞ بُ نگࣹينْ لࣵقࣷو ݖِ سࣹينِ بِيرْ دࣴڪّ يُ مَگْ سَخْ، مَايْ وࣵݖّࣹ مبَسْ ݖِ سࣹينْ بِيرْ، سࣹينِ نࣷونْ جࣹڪُّ لࣹينْ؞ ‏26 بُ مَ تࣺيࣹي لࣷخࣷ بِ مَ لࣹينْ دِ لࣹيلࣹ، فُڪِّ جِگࣺينْ دِنَݧُ بࣷڪّ لَڪّ سࣹينْ مبُرُمْ نجࣴلْ ݖِ بࣹنّ تَالْ دࣷݝّ، نَتّ ڪࣷ اَيْ سࣷمݒّ، سࣺدَّلࣹ، نگࣹينْ لࣹڪّ، تࣹ دُنگࣹينْ سُورْ؞
‏27 «وَايࣹ بُ لࣷولُ وࣹيسࣹي تࣹ دࣺگَّلُلࣹينْ مَ، خَنَا دِ مَ نࣷونࣷو بَ تࣹيْ، ‏28 سُ بࣷوبَا مَ گࣴنْ لࣹينَ مࣹرࣹيتِ، نࣷونࣷو لࣹينْ، تࣹ مَنْ ݖِ سَمَ بࣷݒّ مَا لࣹينْ دِ مبُگَلَتِ لُ ڪࣷ اࣴݒّ فُڪِّ يࣷونْ اَڪْ جُرٗومْ، فࣹيࣹ لࣹينْ ڪࣷ سࣹينْ بࣵڪَّارْ؞ ‏29 دِنگࣹينْ يࣵݒّ سࣹينِ دࣷومْ يُ گٗورْ، يࣵݒّ سࣹينْ دࣷومْ يُ جِگࣺينْ ندَخْ خِيفْ؞ ‏30 دِنَا تَسْ سࣹينْ بࣺرَبِ جَامࣷوڪَايْ¤، تࣷجْ سࣹينِ سُورُڪَايْ؞ سࣹينْ ڪَسَرَيْ تُورْ يِ نࣹ لࣵݖّ، مَ جَلْ ݖَ سࣹينِ نࣺيوْ؞ مَا لࣹينْ دِ سِبْ، ‏31 گࣹنْتَلْ سࣹينْ دࣴڪّ يُ مَگْ، تَسْ سࣹينْ بࣺرَبْ يُ سࣹلّ، تࣹ دࣷوتُمَ فَالࣹ خࣹتُگْ جࣵمّ ݖَ سࣹينِ سَرَخْ؞ ‏32 مَايْ تَسْ سࣹينُمْ رࣺيوْ، بَ سࣹينْ نࣷونْ يَ ڪࣷ نَنگُ يࣺيمُ ݖَ؞ ‏33 يࣹينْ نَگْ مَا لࣹينْ دِ تَسَارࣹ ݖِ بِيرْ خࣹيتْ يِ، ندَخْ مَايْ گࣺنّࣹ سَامَرْ مبَرَمْ، مُ دَلْ ݖِ سࣹينْ ڪَوْ؞ سࣹينُمْ رࣺيوْ دِنَ وࣺيتْ، سࣹينْ دࣴڪّ يُ مَگْ گࣹنتّ؞
‏34 «سُ بࣷوبَا مبࣷولࣹمْ دِيرْ بَ سُوفْ دِ وࣺيتْ مࣷومْ لَيْ دَبࣷو اَتِ نࣷفْلاَيَمْ، فࣹڪّ يࣹينْ نگࣹينْ نࣹڪّ ݖَ سࣹينْ رࣺيوُ نࣷونْ يَ؞ سُ ڪࣷ دࣹفࣹي سُوفْ سِ دِنَ نࣷݒّلُ مࣷوسْ، بَ دَبࣷو ڪࣷ اَتِ نࣷفْلاَيَمْ؞ ‏35 مبࣷولࣹمْ دِيرْ بِ سُوفْ سِيْ وࣺيتْ، مࣷومْ لَيْ نࣷݒّلࣷو لَ مُ نࣷݒّلُوُلࣷونْ، بَ لࣹينْ اَيْ اَتِ نࣷفْلاَيْ فࣹڪّࣹي نگࣹينْ دࣴڪّ فَ؞
‏36 «ݧِ دࣹسْ ݖِ يࣹينْ، دِنَا اِنْدِ نجࣵقَرࣹ ݖِ سࣹينْ خࣷلْ، فࣷوفَ نگࣹينْ نࣹڪّ ݖَ سࣹينْ رࣺيوُ نࣷونْ يَ، بَ خࣷبْ وُ نࣹ فࣷيْ-فࣷيْ نگࣹينْ دَوْ؞ دِنگࣹينْ خࣺلُ نِ ڪُ ݧُيْ دࣵقࣹ سَامَرْ، بَيْ دَانُ، تࣹ دࣵقࣹيسُ لࣹينْ؞ ‏37 دِنگࣹينْ فَڪَّسْتَلَنْتࣹ دِ دَوْ، نِ ݧُ ݧُيْ دࣵقࣹ سَامَرْ، تࣹ دࣵقࣹيسُ لࣹينْ، بَ دُنگࣹينْ مَنَ تَخَوْ ݖِ سࣹينْ ڪَنَمِ نࣷونْ؞ ‏38 دِنگࣹينْ دࣹييࣹ اࣵلّ، تࣹ سࣹينْ سُوفُ نࣷونْ يَا لࣹينْ دِ لࣹڪّ؞ ‏39 ݧِ دࣹسْ ݖِ يࣹينْ دِنَݧُ رࣵگّ ݖَ سࣹينْ رࣺيوِ نࣷونْ ندَخْ سࣹينِ بࣵڪَّارْ اَڪْ سࣹينْ بࣵڪَّارِ مَامْ؞
اَجِ سَخْ جِ دُ فࣹݖِّ ڪٗلّࣴرࣹيمْ
‏40 «سُ ڪࣷ دࣹفࣹي ݧࣷومْ دِنَݧُ تُدّ سࣹينْ بࣵڪَّارْ اَڪْ سࣹينْ بࣵڪَّارُ مَامْ يَ، ندَخْ دَݧُ مَا وࣷرْ تࣹ دࣴڪّࣹ مَا نࣷونࣷو، ‏41 بَ مَنْ اِتَمْ مَ نࣷونࣷو لࣹينْ، تࣷخَلْ لࣹينْ سࣹينْ رࣺيوُ نࣷونْ؞ بُ ݧُ تُوبࣹي سࣹينْ خࣷلْ بُ دࣴگࣴرْ نَگْ، بَ مَتَلْ سࣹينْ مبُگَلْ يِ سَبَبࣷو ݖِ سࣹينِ ݧَاوْتࣺيفْ، ‏42 سُ بࣷوبَا مَايْ بࣵيِّ خࣹلْ ݖِ سَمَ ڪٗلّࣴرࣹ گَاڪْ يَعْقُوبَ اَڪْ سَمَ ڪٗلّࣴرࣹ گَاڪْ إِسْحَاقَ اَڪْ سَمَ ڪٗلّࣴرࣹ گَاڪْ إِبْرَاهِيمَ، تࣹ دُمَ فࣵتّࣹ لِ مَ دِگࣹ ݖِ وࣵلُّ رࣺيوْ مِ؞
‏43 «وَايࣹ رࣺيوْ مِ دِنَݧُ ڪࣷ گࣹنْتَلْ، مُ دَبࣷو اَتِ نࣷفْلاَيَمْ دِيرْ بَ ݧُ ڪࣷ دࣴدّࣷو، وࣺيتَلْ ڪࣷ، تࣹ فࣹڪّ ݧࣷومْ ݧُيْ فࣹيْ سࣹينْ بࣷرُ تࣷوݧْ؛ ندَخْ ڪَتْ دَݧࣷو سࣷفࣹنْتَلْ سَمَيْ سَنْتَانࣹ، خَلَبْ سَمَيْ دࣷگَلْ؞ ‏44 ندَخَمْ دُ تَخْ مَ سࣷفࣹنْتَلْ لࣹينْ ݖَ سࣹينْ رࣺيوُ نࣷونْ، مبَا مَ دِ لࣹينْ خَلَبْ، دِ لࣹينْ فَاگَاگَلْ، ندࣹ ڪࣷنْ مَ فࣹݖِّ سَمَ ڪٗلّࣴرࣹيڪْ ݧࣷومْ؞ ندَخْ مَنْ اَجِ سَخْ جِ مَايْ سࣹينْ يࣵلَّ؞ ‏45 مَا لࣹينْ دِ بࣵيِّلْ خࣹلْ ݖِ سَمَ ڪٗلّࣴرࣹ گَاڪْ سࣹينْ مَامْ يَ مَ گࣺنّࣹ رࣺيوُمْ مِسْرَ، جَامْبُرْ ݧِ سࣹيدࣹ، نگِرْ دࣷونْ سࣹينْ يࣵلَّ؞ مَايْ اَجِ سَخْ جِ؞»
‏46 لࣷولࣷو دِ دࣷگَلْ يَاڪْ سَنْتَانࣹ يَاڪْ ندِگَلِ يࣷونْ يَ اَجِ سَخْ جِ تࣴرَلْ ݖَ دِگَّنْتࣹيمْ اَڪْ بࣵنِّ إِسْرَاءِيلْ ݖَ ڪَوْ تُنْدُ سِنَيِ، مُوسَىٰ جࣷتّلِ لࣹينْ ڪࣷ؞
