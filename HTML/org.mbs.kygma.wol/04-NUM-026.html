26
‏1 اَجِ سَخْ جِ دَفَ وَخْ مُوسَىٰ، اَڪْ اࣹلَسَرْ دࣷومُ هَـٰرُونَ سَرْخَلْڪَتْ بَ، نࣹ لࣹينْ:‏ ‏2 «وَݧّلࣹينْ مبࣷولࣷومْ بࣵنِّ إِسْرَاءِيلْ گࣺݒّ، نگࣹينْ لِمْ تُرِ گٗورْ ݧࣺݒّ، نَ ݧُ بࣷڪّࣹ سࣹينْ ڪࣴرِ مَامْ، دَلࣹ ڪࣷ ݖِ ݧِ اَمْ ݧَارْ فُڪِّ اَتْ، جࣴمْ ڪَوْ، مبࣷولࣹمْ ݧِ خَرࣹ وَرْ ݖِ بࣵنِّ إِسْرَاءِيلْ؞» ‏3 مُوسَىٰ اَڪْ اࣹلَسَرْ سَرْخَلْڪَتْ بَ وَخْ اَڪْ ݧࣷومْ ݖَ جࣷورِ مࣷوَبْ، فَ جَنࣷوڪْ يࣹرِڪࣷ ݖَ وࣹتُ دࣹخُ يُرْدَنْ؛ ݧُ نࣹ لࣹينْ ‏4 ݧِ دَلࣹ ݖِ ݧَارْ فُڪِّ اَتْ، جࣴمْ ڪَوْ لࣹيسْ دِ لِمْ، نَ ڪࣷ اَجِ سَخْ جِ سَنْتࣹ مُوسَىٰ؞ لِي مࣷو دِ لِمࣹيفُ نِتْ ݧَ سࣷقِڪࣷو ݖِ بࣵنِّ إِسْرَاءِيلْ، ݧَ گࣺنّࣹ وࣷونْ مِسْرَ:‏
‏5 رُبࣹنْ مࣷو دࣷونْ تَاوُبْ إِسْرَاءِيلْ؞ ݧِ اَسْڪَنࣷو ݖِ رُبࣹنْ ݧࣷو دِ اَنࣷگْ مِ سࣷسْ لࣵنگُ اَنࣷگࣹينْ ݧِ، اَڪْ ݧࣷومْ ݒَلُ مِ سࣷسْ لࣵنگُ ݒَلُوࣹينْ ݧِ، ‏6 اَڪْ ݧࣷومْ اࣹسْرࣷنْ مِ سࣷسْ لࣵنگُ اࣹسْرࣷنࣹينْ ݧِ، اَڪْ ݧࣷومْ ڪَرْمِ مِ سࣷسْ لࣵنگُ ڪَرْمࣹينْ ݧِ؞ ‏7 لࣵنگِ رُبࣹنࣹينْ ݧَا نگࣷوگُ، ݧَ ݧُ لࣹينْ لِمَلْ دِ ݧࣹينتّ فُڪِّ جُنّࣹيڪْ ݧࣹتّ اَڪْ جُرٗومْ ݧَارْ تࣺيمࣺيرْ اَڪْ فَنْوࣹيرْ (43730)؞
‏8 ݒَلُ مࣷو مࣹݧّ اࣹلْيَبْ، ‏9 اࣹلْيَبْ مࣹݧّ نࣹمْوࣹلْ اَڪْ دَتَنْ اَڪْ اَبِرَمْ؞ دَتَنْ اَڪْ اَبِرَمْ ݧࣷوݧِي دࣷونࣷونِ نجِيتْ ݖَ مبࣷولࣷو مَ، ݧࣷو دࣷونْ دِيݝَتْ مُوسَىٰ اَڪْ هَـٰرُونَ؞ ݧَ نگَ بࣷڪّࣷونْ ݖَ گࣵنگࣷورُ ڪࣷرࣹ، بَ ݧُيْ دِيݝَتْ اَجِ سَخْ جِ، ‏10 بَ سُوفْ سَ خَرْ، بࣷولࣹ لࣹينْ اَڪْ ڪࣷرࣹ، وࣷنّ، ڪࣹرࣷوگْ بَ گࣵنگࣷورْ گَ سࣵنڪࣷو، تࣹ سَوَرَ وَ لَڪّ ݧَارْ تࣺيمࣺيرِ نِتْ اَڪْ جُرٗومْ فُڪّ، ݧُ دَلْدِ دࣷونْ فِرْندࣹ لُيْ اَرْتࣷو؞ ‏11 دࣷومِ ڪࣷرࣹ يَ نَگْ دࣹيوُݧُ ݖَ؞
‏12 ݧِ اَسْڪَنࣷو ݖِ سِمࣹيࣷنْ، ݧࣷوڪْ سࣹينِ لࣵنگّ ݧࣷو دِ ݧࣷومْ نࣹمْوࣹلْ مِ سࣷسْ لࣵنگُ نࣹمْوࣹلࣹينْ ݧِ، اَڪْ ݧࣷومْ يَمِنْ مِ سࣷسْ لࣵنگُ يَمِنࣹينْ ݧِ، اَڪْ ݧࣷومْ يَڪِنْ مِ سࣷسْ لࣵنگُ يَڪِنࣹينْ ݧِ، ‏13 اَڪْ ݧࣷومْ سࣹرَ مِ سࣷسْ لࣵنگُ ݖࣹرَيࣹينْ ݧِ، اَڪْ ݧࣷومْ سَوُلْ مِ سࣷسْ لࣵنگُ ݖَوُلࣹينْ ݧِ؞ ‏14 لࣵنگِ ݖِمْيࣷنࣹينْ ݧَا نگࣷوگُ، دِ ݧَارْ فُڪِّ جُنّࣹيڪْ ݧَارْ اَڪْ ݧَارْ تࣺيمࣺيرْ (22200)؞
‏15 ݧِ اَسْڪَنࣷو ݖِ گࣵدّ، ݧࣷوڪْ سࣹينِ لࣵنگّ ݧࣷو دِ ݧࣷومْ ݖࣹفࣷنْ مِ سࣷسْ لࣵنگُ ݖࣹفࣷنࣹينْ ݧِ، اَڪْ ݧࣷومْ اَگِ مِ سࣷسْ لࣵنگُ اَگࣹينْ ݧِ، اَڪْ ݧࣷومْ سُنِ مِ سࣷسْ لࣵنگُ ݖُنࣹينْ ݧِ، ‏16 اَڪْ ݧࣷومْ اࣷسْنِ مِ سࣷسْ لࣵنگُ اࣷسْنࣹينْ ݧِ، اَڪْ ݧࣷومْ اࣹرِ مِ سࣷسْ لࣵنگُ اࣹرࣹينْ ݧِ، ‏17 اَڪْ ݧࣷومْ اَرࣷدّ مِ سࣷسْ لࣵنگُ اَرࣷدّࣹينْ ݧِ، اَڪْ ݧࣷومْ اَرࣹلِ مِ سࣷسْ لࣵنگُ اَرࣹلࣹينْ ݧِ؞ ‏18 لࣵنگِ گࣵدّࣹينْ ݧَا نگࣷوگُ، ݧَ ݧُ لࣹينْ لِمَلْ دِ ݧࣹينتّ فُڪِّ جُنّࣹيڪْ جُرٗومِ تࣺيمࣺيرْ (40500)؞
‏19 اَمْ نَ نَگْ دࣷومِ يُدَ يُ ݧُيْ وَخْ اࣹرْ اَڪْ اࣷنَنْ؞ وَايࣹ اࣹرْ اَڪْ اࣷنَنْ ݧَ ݖَ رࣺيوُمْ ڪَنَانْ لَݧُ فَاتࣷو؞ ‏20 ݧِ اَسْڪَنࣷو ݖِ يُدَ، ݧࣷوڪْ سࣹينِ لࣵنگّ ݧࣷو دِ ݧࣷومْ سࣹلَ مِ سࣷسْ لࣵنگُ ݖࣹلَنࣹينْ ݧِ، اَڪْ ݧࣷومْ ݒࣹرࣹسْ مِ سࣷسْ لࣵنگُ ݒࣹرࣹسࣹينْ ݧِ، اَڪْ ݧࣷومْ سࣹرَ مِ سࣷسْ لࣵنگُ ݖࣹرَيࣹينْ ݧِ؞ ‏21 ݧِ اَسْڪَنࣷو ݖِ ݒࣹرࣹسْ ݧࣷو دِ ݧࣷومْ اࣹسْرࣷنْ مِ سࣷسْ لࣵنگُ اࣹسْرࣷنࣹينْ ݧِ، اَڪْ ݧࣷومْ اَمُلْ مِ سࣷسْ لࣵنگُ اَمُلࣹينْ ݧِ؞ ‏22 لࣵنگِ يُدࣹينْ ݧَا نگࣷوگُ، ݧَ ݧُ لࣹينْ لِمَلْ دِ جُرٗومْ ݧَارْ فُڪِّ جُنّࣹيڪْ جُرٗومْ بࣹنّ اَڪْ جُرٗومِ تࣺيمࣺيرْ (76500)؞
‏23 ݧِ اَسْڪَنࣷو ݖِ اِسَاڪَرْ، ݧࣷوڪْ سࣹينِ لࣵنگّ ݧࣷو دِ تࣷلَ مِ سࣷسْ لࣵنگُ تࣷلࣹينْ ݧِ، اَڪْ ݧࣷومْ ݒُوَ مِ سࣷسْ لࣵنگُ ݒُوࣹينْ ݧِ، ‏24 اَڪْ ݧࣷومْ يَسُبْ مِ سࣷسْ لࣵنگُ يَسُبࣹينْ ݧِ، اَڪْ ݧࣷومْ سِمْرࣷنْ مِ سࣷسْ لࣵنگُ سِمْرࣷنࣹينْ ݧِ؞ ‏25 لࣵنگِ اِسَاڪَرَ نگࣷوگُ؞ ݧَ ݧُ لࣹينْ لِمَلْ دِ جُرٗومْ بࣹنّ فُڪِّ جُنّࣹيڪْ ݧࣹينتّ اَڪْ ݧࣹتِّ تࣺيمࣺيرْ (64300)؞
‏26 ݧِ اَسْڪَنࣷو ݖِ سَبُلࣷنْ، ݧࣷوڪْ سࣹينِ لࣵنگّ ݧࣷو دِ ݧࣷومْ سࣹرࣹدّ مِ سࣷسْ لࣵنگُ سࣹرࣹدّࣹينْ ݧِ، اَڪْ ݧࣷومْ اࣹلࣷنْ مِ سࣷسْ لࣵنگُ اࣹلࣷنࣹينْ ݧِ، اَڪْ ݧࣷومْ يَلࣹيلْ، مِ سࣷسْ لࣵنگُ يَلࣹلࣹينْ ݧِ؞ ‏27 لࣵنگِ سَبُلࣷنࣹينْ ݧَا نگࣷوگُ، ݧَ ݧُ لࣹينْ لِمَلْ دِ جُرٗومْ بࣹنّ فُڪِّ جُنِّ اَڪْ جُرٗومِ تࣺيمࣺيرْ (60500)؞
‏28 ݧِ اَسْڪَنࣷو ݖِ يُوسُفَ نَگْ، ݧࣷوڪْ سࣹينِ لࣵنگّ، ݧࣷو سࣷقِڪࣷو ݖِ مَنَسࣹ اَڪْ اࣹفْرَيِمْ؞ ‏29 ݧِ سࣷقِڪࣷو ݖِ مَنَسࣹ ݧࣷو دِ ݧࣷومْ مَڪِرْ مِ سࣷسْ لࣵنگُ مَڪِرࣹينْ ݧِ؞ مَڪِرْ مࣷو جُرْ گَلࣵدّ؞ ݖِ گَلࣵدّ لَ لࣵنگُ گَلࣵدّࣹينْ ݧِ سࣷقِڪࣷو؞ ‏30 ݧِي ݧࣷو اَسْڪَنࣷو ݖِ گَلࣵدّ:‏ يࣹسࣹرْ مِ سࣷسْ لࣵنگُ يࣹسࣹرࣹينْ ݧِ، اَڪْ ݧࣷومْ اࣹلࣹگْ مِ سࣷسْ لࣵنگُ اࣹلࣹگࣹينْ ݧِ، ‏31 اَڪْ اَسْرِيࣹلْ مِ سࣷسْ لࣵنگُ اَسْرِيࣹلࣹينْ ݧِ، اَڪْ سِڪࣹمْ مِ سࣷسْ لࣵنگُ سِڪࣹمࣹينْ ݧِ، ‏32 اَڪْ سࣹمِدَ مِ سࣷسْ لࣵنگُ سࣹمِدࣹينْ ݧِ، اَڪْ اࣹفࣹرْ مِ سࣷسْ لࣵنگُ اࣹفࣹرࣹينْ ݧِ؞ ‏33 سࣹلࣷفَتْ دࣷومُ اࣹفࣹرْ نَگْ اَمُلࣷونْ دࣷومْ جُ گٗورْ، خَنَا اَوْ جِگࣺينْ دࣷݝّ؞ دࣷومِ سࣹلࣷفَتْ يُ جِگࣺينْ دِ مَالَ اَڪْ نࣷوَ اَڪْ اࣷگْلَ اَڪْ مِلْڪَ اَڪْ تِرْݖَ؞ ‏34 لࣵنگِ مَنَسࣹي نگࣷوگُ، ݧَ ݧُ لࣹينْ لِمَلْ دِ جُرٗومْ فُڪِّ جُنّࣹيڪْ ݧَارْ اَڪْ جُرٗومْ ݧَارْ تࣺيمࣺيرْ (52700)؞
‏35 ݧِي نَگْ ݧࣷو اَسْڪَنࣷو ݖِ اࣹفْرَيِمْ، ݧࣷوڪْ سࣹينِ لࣵنگّ:‏ ݧࣷومْ سُتࣹلَ مِ سࣷسْ لࣵنگُ ݖُتࣹلࣹينْ ݧِ، اَڪْ ݧࣷومْ بࣹڪࣹرْ مِ سࣷسْ لࣵنگُ بࣹڪࣹرࣹينْ ݧِ، اَڪْ ݧࣷومْ تَخَنْ مِ سࣷسْ لࣵنگُ تَخَنࣹينْ ݧِ؞ ‏36 ݧِي نَگْ، اَسْڪَنُ سُتࣹلَ لَݧُ:‏ ݧࣷومْ اࣹرَنْ، اَڪْ لࣵنگُ اࣹرَنࣹينْ ݧِ مُ سࣷسْ؞ ‏37 لࣵنگِ اࣹفْرَيِمَ نگࣷوگُ، ݧَ ݧُ لࣹينْ لِمَلْ دِ فَنْوࣹيرِ جُنّࣹيڪْ ݧَارْ اَڪْ جُرٗومِ تࣺيمࣺيرْ (32500)؞
ݧࣷوݧࣷو اَسْڪَنࣷو ݖِ يُوسُفَ، ݧࣷوڪْ سࣹينِ لࣵنگّ؞
‏38 ݧِ اَسْڪَنࣷو ݖِ بࣹݧَمِنْ، ݧࣷوڪْ سࣹينِ لࣵنگّ ݧࣷو دِ ݧࣷومْ بࣹلَ مِ سࣷسْ لࣵنگُ بࣹلَيࣹينْ ݧِ، اَڪْ ݧࣷومْ اَسْبࣹلْ مِ سࣷسْ لࣵنگُ اَسْبࣹلࣹينْ ݧِ، اَڪْ ݧࣷومْ اَيْرَمْ مِ سࣷسْ لࣵنگُ اَيْرَمࣹينْ ݧِ، ‏39 اَڪْ ݧࣷومْ سُفَمْ مِ سࣷسْ لࣵنگُ سُفَمࣹينْ ݧِ، اَڪْ ݧࣷومْ اُفَمْ مِ سࣷسْ لࣵنگُ اُفَمࣹينْ ݧِ؞ ‏40 اَسْڪَنُ بࣹلَ وِ سࣷسࣷو ݖِ اࣵردّ اَڪْ نَامَنْ ݧࣷو دِ لࣵنگُ اࣵرْدࣹينْ ݧِ، اَڪْ لࣵنگُ نَامࣹينْ ݧِ سࣴتࣷو ݖِ نَامَنْ؞ ‏41 ݧࣷوݧࣷو اَسْڪَنࣷو ݖِ بࣹݧَمِنْ، ݧࣷوڪْ سࣹينِ لࣵنگّ، ݧَ ݧُ لࣹينْ لِمَلْ دِ ݧࣹينتّ فُڪِّ جُنّࣹيڪْ جُرٗومْ اَڪْ جُرٗومْ بࣹنِّ تࣺيمࣺيرْ (45600)؞
‏42 ݧِي نَگْ ݧࣷو اَسْڪَنࣷو ݖِ دَنْ، ݧࣷوڪْ سࣹينِ لࣵنگّ:‏ ݧࣷومْ سُوَمْ اَڪْ لࣵنگُ ݖُوَمࣹينْ ݧِ؞ دَنࣹينْ ݧَا نگࣷوگُ، نَ ݧُ لِمࣹ سࣹينِ لࣵنگّ؞ ‏43 مبࣷولࣹمْ لࣵنگِ ݖُوَمࣹينْ، ݧَ ݧُ لࣹينْ لِمَلْ جُرٗومْ بࣹنّ فُڪِّ جُنّࣹيڪْ ݧࣹينتّ لَ، اَڪْ ݧࣹينْتِ تࣺيمࣺيرْ (64400)؞
‏44 ݧِ اَسْڪَنࣷو ݖِ اَسࣹرْ، ݧࣷوڪْ سࣹينِ لࣵنگّ ݧࣷو دِ ݧࣷومْ اِمْنَ مِ سࣷسْ لࣵنگُ اِمْنࣹينْ ݧِ، اَڪْ ݧࣷومْ اِسْوِ مِ سࣷسْ لࣵنگُ اِسْوࣹينْ ݧِ، اَڪْ ݧࣷومْ بࣹرْيَ مِ سࣷسْ لࣵنگُ بࣹرْيࣹينْ ݧِ؞ ‏45 وَا اَسْڪَنُ بࣹرْيَ ݧࣷو دِ ݧࣷومْ اࣹبࣹرْ مِ سࣷسْ لࣵنگُ اࣹبࣹرࣹينْ ݧِ، اَڪْ ݧࣷومْ مَلْڪِيࣹلْ مِ سࣷسْ لࣵنگُ مَلْڪِيࣹلࣹينْ ݧِ؞ ‏46 دࣷومُ اَسࣹرْ جُ جِگࣺينْ نَگْ مࣷو دِ سࣹرَ؞ ‏47 لࣵنگِ اَسࣹرࣹينْ ݧَا نگࣷوگُ، ݧَ ݧُ لࣹينْ لِمَلْ دِ جُرٗومْ فُڪِّ جُنّࣹيڪْ ݧࣹتّ اَڪْ ݧࣹينْتِ تࣺيمࣺيرْ (53400)؞
‏48 ݧِ اَسْڪَنࣷو ݖِ نࣹفْتَلِ ݧࣷوڪْ سࣹينِ لࣵنگّ ݧࣷو دِ ݧࣷومْ يࣵݖّࣹيلْ، مِ سࣷسْ لࣵنگُ يَݖࣹلࣹينْ ݧِ، اَڪْ ݧࣷومْ گُنِّ مِ سࣷسْ لࣵنگُ گُنࣹينْ ݧِ، ‏49 اَڪْ ݧࣷومْ يࣹݖّࣹرْ مِ سࣷسْ لࣵنگُ يࣹݖّࣹرࣹينْ ݧِ، اَڪْ ݧࣷومْ سِلࣹمْ مِ سࣷسْ لࣵنگُ ݖِلْمࣹينْ ݧِ؞ ‏50 نࣹفْتَلࣹينْ ݧَا نگࣷوگُ، ݧࣷوڪْ سࣹينِ لࣵنگّ، ݧَ ݧُ لࣹينْ لِمَلْ دِ ݧࣹينتّ فُڪِّ جُنّࣹيڪْ جُرٗومْ اَڪْ ݧࣹينْتِ تࣺيمࣺيرْ (45400)؞
‏51 لِمࣹيفُ بࣵنِّ إِسْرَاءِيلَ نگࣷوگُ، دِ جُرٗومْ بࣹنِّ تࣺيمࣺيرِ جُنّࣹيڪْ بࣹنّ اَڪْ جُرٗومْ ݧَارْ تࣺيمࣺيرْ اَڪْ فَنْوࣹيرْ (601730)؞
لُ جࣴمْ ݖِ سࣺدَّلࣹبْ رࣺيوْ مَ
‏52 اَجِ سَخْ جِ نَگْ وَخْ مُوسَىٰ، نࣹ ڪࣷ:‏ ‏53 «ݧِي لࣹيسْ دِ سࣺدَّلࣹ رࣺيوْ مِ، تࣹ نَ ݖࣺرْ يِ دࣴݒّࣷوڪْ لِمُ نِتْ ݧِ:‏ ‏54 لࣵنگّ گُ گࣴنَ مَگْ، نَ اَمْ ݖࣺرْ بُ گࣴنَ مَگْ، لࣵنگّ گُ گࣴنَ تُوتْ، اَمْ ݖࣺرْ بُ گࣴنَ تُوتْ، لࣵنگّ گُ ݖِ نࣹڪّ، ݖࣺرْ بِ دࣴݒّࣷوڪْ اَبْ لِمَمْ لࣹيسْ ڪࣷيْ جࣷخْ؞ ‏55 اَبْ تࣹگࣷو بَنتّ دࣷݝّ لࣹيسْ دِ سࣺدَّلࣹي رࣺيوْ مِ، تࣹ سࣹينْ لِمُ گِيرِ مَامْ يَ ݧُ بࣷڪّ، نَ دࣴݒّࣷوڪْ سࣹينُبْ ݖࣺرْ؞ ‏56 ݖِ ڪَوْ تࣹگࣷو بَنتّ لࣹيسْ دِ دࣷگَلࣹ ݖࣺرُ گِيرْ گُ ݧُ بَرࣹ سࣺقْ، اَڪْ گُ ݧُ نࣺيوْ سࣺقْ؞»
ݧَارࣹيلُ لِمࣹيفُ لࣹوࣹينْ ݧِ تَخَوْ نَ
‏57 ݧِي نَگْ لَݧُ لِمَلْ لࣹوࣹينْ ݧِ اَڪْ سࣹينِ لࣵنگّ:‏ ݧࣷومْ گࣹرْسࣷنْ مِ سࣷسْ لࣵنگُ گࣹرْسࣷنࣹينْ ݧِ، اَڪْ ݧࣷومْ ڪࣹيَتْ مِ سࣷسْ لࣵنگُ ڪࣹيَتࣹينْ ݧِ، اَڪْ ݧࣷومْ مࣹرَرِ مِ سࣷسْ لࣵنگُ مࣹرَرࣹينْ ݧِ؞ ‏58 ݧِي اِتَمْ ݖِ لࣵنگِ لࣹوࣹينْ لَݧُ:‏ لࣵنگُ لِبْنࣹينْ ݧِ، اَڪْ لࣵنگُ اࣹبْرࣷنࣹينْ ݧِ، اَڪْ لࣵنگُ مَالࣹينْ ݧِ، اَڪْ لࣵنگُ مُسࣹينْ ݧِ، اَڪْ لࣵنگُ ڪࣷرَيࣹينْ ݧِ؞ ڪࣹيَتْ نَگْ مࣷويْ مَامُ اَمْرَمْ؛ ‏59 سࣷخْنَسْ اَمْرَمْ دِ يࣷڪࣹبࣹدّ دࣷومُ لࣹوِ، جَ جُدّࣷو مِسْرَ؞ يࣷڪࣹبࣹدّ اَمْ اَڪْ اَمْرَمْ هَـٰرُونَ اَڪْ مُوسَىٰ اَڪْ مَرْيَامَ، سࣹينُبْ جِگࣺينْ؞ ‏60 هَـٰرُونَ مࣷو جُرْ نَدَبْ اَڪْ اَبِيُ اَڪْ اࣹلَسَرْ اَڪْ اِتَمَرْ؞ ‏61 نَدَبْ اَڪْ اَبِيُ ݧَ نگَ دࣹي ڪࣹرࣷوگْ بَ ݧُ تَالَلࣹي اَجِ سَخْ جِ ݖُورَايْ لُ وَرُلْ؞ ‏62 ݧَ ݧُ لِمَلْ لࣹوࣹينْ ݧِ نَگْ ݧَارْ فُڪِّ جُنّࣹيڪْ ݧࣹتّ لَݧُ (23000)، دِ مبࣷولࣹمْ گٗورْ ݧَ اَمْ وࣹيرْ، جࣴمْ ڪَوْ، تࣹ لِمَالࣹيسُ لࣹينْ اَڪْ بࣵنِّ إِسْرَاءِيلْ گَ ݖَ دࣹسْ، ندَخْ جࣷخࣹيسُ لࣹينْ اَبْ ݖࣺرْ ݖِ بِيرْ بࣵنِّ إِسْرَاءِيلْ؞
تࣴجْ نَݧُ لِمࣹيفْ يَ
‏63 ݧࣷوݧُ لَݧُ لِمْ ݖِ نجِيتَلْ مُوسَىٰ اَڪْ اࣹلَسَرْ سَرْخَلْڪَتْ بَ، بَ ݧُ تࣴنڪࣹي لِمࣹيفُ بࣵنِّ إِسْرَاءِيلْ ݖَ جࣷورِ مࣷوَبْ، گَ جَنࣷوڪْ يࣹرِڪࣷ، ݖَ تࣵڪَّلْ دࣹخُ يُرْدَنْ؞ ‏64 ݧࣷوݧُ، ڪࣹنّ دࣹسُلْ ݖَ سࣹينْ بِيرْ، تࣹ بࣷڪّࣷونْ ݖَ لِمࣹيفُ بࣵنِّ إِسْرَاءِيلْ، بَ مُوسَىٰ اَڪْ هَـٰرُونَ سَرْخَلْڪَتْ بَ جࣴڪّࣷونَ تࣴنڪࣹ ݖَ مࣵنْدِݝُ سِنَيِ؞ ‏65 ندَخْ ݖِ ݧࣷومْ لَ اَجِ سَخْ جِ نࣷونْ دࣹي رࣹڪّ لَݧُيْ دࣹفْ بَ جࣹيخْ ݖَ مࣵنْدِݝْ مَ؞ ڪࣹنّ دࣹسُ ݖَ، ڪُ دُلْ ڪَلࣹبْ دࣷومُ يࣹفُنࣹ، اَڪْ يࣷسُوࣹ دࣷومُ نُونْ؞
