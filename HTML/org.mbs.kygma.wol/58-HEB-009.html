يࣷونُ مُوسَىٰ سࣵرْتَلࣷونْ نَ نُ ݧُيْ جَامࣷو يࣵلَّ
9
‏1 ڪٗلّࣴرࣹ گُ جࣴڪّ گَ اَمࣷونْ نَ مࣷومْ اِتَمْ، اَيْ ندِگَلَمْ يُ جࣴمْ ݖِ نِ ݧُ وَرَ جَامࣷو يࣵلَّ، اَڪْ بࣺرَبَمْ بُ سࣹلّ بِ فِ ڪَوْ سُوفْ؞ ‏2 خَيْمَبْ جَامࣷوڪَايْ بَ ݧُ سࣵڪّࣷونْ ݧَارِ نࣺيگْ لَ وࣷونْ؞ بُ جࣴڪّ بَ، ݖَ لَ تࣹگُڪَايُ لࣵمݒّ بَ نࣹڪّࣷونْ، اَڪْ تَابَلْ جَ، اَڪْ مبُرُمْ تࣹيوَلْ مَ؞ مࣷو دࣷونْ بࣺرَبْ بُ سࣹلّ بَ؞ ‏3 نࣺيگْ بَ ݖَ وࣵلاَّ ݧَارࣹيلُ رِدࣷ بَ، مࣷومْ لَݧُ دَانْ وࣷويࣹ بࣺرَبْ بُ سࣹلّ بَا سࣹلّ؞ ‏4 ݖَ لَ سَرْخَلُڪَايُ وُرُسْ بَ ݧُيْ تَالْ ݖُورَايْ بࣷڪّࣷونْ، گَالُ ڪٗلّࣴرࣹ گَ نࣹڪّ مࣷومْ دࣷݝّ ݖَ بِيرْ نࣺيگْ بُ سࣹلّ بَا سࣹلّ، ݧُ خࣷوبْ ڪࣷ وُرُسْ بَ مُ دَجْ؞ گَالْ گَ مࣷو يࣹبࣷونْ نجَقَلْ وُرُسْ لَ دࣹفࣷونْ ݧَمْ وَ ݧُيْ وَخْ مࣵنّ، اَڪْ يࣹتُ هَـٰرُونَ وَ جࣹبِّ وࣷونْ، اَڪْ اࣵلُّوَيْ ڪٗلّࣴرࣹ گَ؞ ‏5 ݖَ ڪَوْ گَالْ گَ لَ ݧَارِ جࣴمِّ مَلاَڪَيْ سࣹرُبْ نࣹڪّࣷونْ، دِ تࣹگْتَلࣹ لࣹيرُ يࣵلَّ، سࣹينِ لاَفْ ڪࣹݒَّارَلْ ڪُبࣹيرْ گَ ݧُيْ اَمَلࣹ اَگْ نجࣷتْلاَيْ؞ وَايࣹ يࣷويُ دُنُ ڪࣷ مَنَ دَلْدِ فَرَمْفࣵݖّࣹ لࣺيگِ؞
‏6 بَ ݧُ وَاجَلࣹي نࣷونُ يࣷويِي بَ مُ جࣹڪّ، سَرْخَلْڪَتْ يَا دَانْ دُگّ سَا سُ نࣹڪّ ݖَ نࣺيگْ بُ جࣴڪّ بَ، دِ دࣹفْ سࣹينْ لِگّࣺييْ؞ ‏7 وَايࣹ ݧَارࣹيلُ نࣺيگْ بَ، سَرْخَلْڪَتْ بُ مَگْ بَ دࣷݝّ مࣷو ݖَ دَانْ دُگّ، بࣹنّ يࣷونْ ݖِمْ اَتْ، تࣹ دُ ݧࣵڪَّ دُگَّالࣹ دࣹرࣹتْ جُ مُيْ جࣷوخࣹ نگِرْ بࣷݒَّمْ، اَڪْ اِتَمْ نگِرْ بࣵڪَّارْ يِ اَسْڪَنْ وِ دࣹفْ تࣹ دُ اَگْ تࣹيࣹيفْ؞ ‏8 نࣷونُ لَ نࣷو گُ سࣹلّ گِ فِرْندࣹيلࣹ نࣹ يࣷونْ وُ مَيࣹ دُگّ ݖَ بࣺرَبْ بُ سࣹلّ بَا سࣹلّ فࣹيݧَگُلْ وࣷونْ، تࣹ فࣹڪّ نࣺيگْ بُ جࣴڪّ بَ دِ تَخَوْ بَ تࣹيْ؞ ‏9 لࣷولُ مِسَالْ لَ، ݧࣹيلْ جَمࣷنࣷيْ تࣹيْ، دِ فِرْندࣹيلْ نࣹ جࣷوخࣹ يِ اَڪْ سَرَخِ جُرْ يِ ݧُ دَانْ جࣺبَّلࣹ مَنُݧࣷو دِنْدِ يَرَانگࣹگْ جَامُڪَتُبْ يࣵلَّ بِ، بَ مُ سࣹتْ وࣹݖّ؞ ‏10 سَرَخْ يࣷويُ، ݖِ اَوْ ݧَمْ اَڪْ اَگْ نَانْ، اَڪْ خࣹيتِ نجࣵݒّ دࣷݝّ لَ جࣴمْ، لࣺݒّ يِتّࣹوࣷو يَرَمُ سُوخُ نࣹينْ، تࣹ دِ ندِگَلْ يُ لࣹينْ وَرࣷونْ بَ ڪࣹرࣷوگْ بࣺسُبْ ݖࣷݒِّتࣹ تَخَوْ؞
اَلْمَسِ مࣷو فَرْ بࣵڪَّارْ، فَسْ ڪٗلّࣴرࣹ گُ بࣹيسْ؞
‏11 اَلْمَسِ نَگْ مࣷو دِڪّ تࣹيْ، دِ سَرْخَلْڪَتْ بُ مَگْ، بِ يࣷرْ خࣺيوَلْ يِ تࣹيوْ بَ نࣷݒِّ؞ مࣷو دُگّ ݖَ بِيرْ خَيْمَ بَ گࣴنَ مࣵگّ تࣹ مَتْ سࣴڪّ، خَيْمَ بُ لࣷخࣷلْ نِتْ دࣹفُلْ، تࣹ لࣷولُ مࣷويْ خَيْمَ بَ بࣷڪُّلْ ݖِ ݖࣵڪّࣺيفُگْ اࣵدِّنَ سِي؞ ‏12 دُ دࣹرࣹتُ سِڪّࣹتْ مبَا يࣴڪّ لَ دُگَّالࣹ، وَايࣹ دࣹرࣹتُ بࣷݒَّمْ لَ دُگّࣹ بࣹنّ يࣷونْ بَ فࣵوّ ݖَ بࣺرَبْ بُ سࣹلّ بَا سࣹلّ، تࣹ مࣷومْ لَ اَمَلࣹ نجࣷتْلاَيْ گُ سَخْ دࣵڪّ؞ ‏13 ندࣹگَمْ دࣹرࣹتُ سِڪّࣹتْ اَڪْ يࣴڪّ، اَڪْ نَگْ وُ جِگࣺينْ وُ ݧُ لَڪّ، نگِرْ سُيْ دٗومْ بَ نِتْ ڪُ سࣷبࣹوُ، يࣷويُ لࣹيسْ دَانْ سࣹلَّلࣹ نِتْ ݧِ، نگِرْ سࣹينْ ݖࣹتُگْ يَرَمُ سُوخُ نࣹينْ، ‏14 تࣹ اَلْمَسِ دِ ڪِ دࣹفْ بࣷݒَّمْ سَرَخْ بُ مُݖّ سِڪّ ݧࣹيلْ يࣵلَّ، ݖِ دࣷولࣹيْ نࣷو گُ سࣹلّ گِ، ڪࣷنْ نَگْ ݧَاتَ يࣷونْ لَ دࣹرࣹتُ اَلْمَسِ گࣴنَ مَنَ سࣹتَلْ سُنُ يَرَانگࣹ جِ سَبَبࣷو ݖِ سُنُ جࣴفْ جُ بࣷنْ جُ جࣴمࣹ ݖِ ندࣹي، بَ نُ مَنَ جَامُ يࣵلَّ جِيْ دُندّ!‏ ‏15 لࣷولࣷو تَخْ عِيسَىٰ دِ دࣹفَرࣹڪَتْ بِ فَسْ ڪٗلّࣴرࣹ گُ بࣹيسْ، نگِرْ ݧِ ݧُ وࣷو مَنَ جࣷتْ ݖِ مُورْ گُ سَخْ دࣵڪّ گِ ݧُ لࣹينْ دِگْ؞ دࣹييَمْ گِ مُ دࣹي، مࣷومْ لَ لࣹينْ جࣷتࣹ ݖِ بࣵڪَّارْ يِ ݧُ خࣴتّࣹ ڪٗلّࣴرࣹ گُ جࣴڪّ گَ؞
‏16 ڪَيِتُ ندࣷنࣷ نَگْ فُ مُ تࣹيوْ، فࣵوّ جࣷخࣹيسْ فَ فِرْندࣹيْ دࣹيوُگْ ڪَ دࣷنَلࣹ، ‏17 ندَخْ ڪَيِتُ ندࣷنࣷ دُ وࣺرْ لُ مࣷيْ گَنَّاوْ بُ نِتْ ڪَ دࣹييࣹي؞ دُ جࣵݒَّنْدِ مُڪّ لِ فࣹيڪْ ڪِ دࣷنَلࣹي نگِ دُندّ؞ ‏18 لࣷولࣷو وَرَلْ اِتْ ڪٗلّࣴرࣹ¤ گُ جࣴڪّ گَ ݖِ بࣷݒَّمْ، دࣷورࣹيسُ ڪࣷ وࣷونْ تࣹ دࣹرࣹتْ تُورُوُلْ؞ ‏19 ڪࣹرࣷوگْ بَ مُوسَىٰ بِرَلَلࣹي بࣵنِّ إِسْرَاءِيلْ گࣺݒّ مبࣷولࣹمْ سَنْتَانࣹ، نَ ڪࣷ يࣷونْ وِ لاَجࣹ، بَ مُ نࣷݒّࣹي، دࣹرࣹتُ وࣴلُّ لَ جࣴلْ، اَڪْ جُ سِڪّࣹتْ، اَڪُمْ ندࣷخْ، اَڪْ ݖَرُ گَرَبُ اِسࣷبْ بُ مُ لَخَسْ ڪَوَرُ خَرْ مُ خࣷنقّ؞ مُ ݖَݒّ ڪࣷ ݖَ دࣹرࣹتْ جَ، وِسْ-وِسَلْ ݖَ ڪَوْ تࣺيرࣹبْ يࣷونْ وَ، اَڪْ مبࣷولࣷو مࣺݒّ؞ ‏20 مُ نࣹ:‏ «لِي مࣷويْ دࣹرࣹتُ ڪٗلّࣴرࣹ گِ يࣵلَّ فَسْ اَڪْ يࣹينْ¤؞» ‏21 خَيْمَ بَ، اَڪْ مبࣷولࣹمْ ندَبِ ݖَرْخَلْ يَ اِتْ، نࣷونُ لَ وِسْ-وِسَلࣹ ݖَ ڪَوْ؞ ‏22 ندَخْ ݖِ يࣷونُ مُوسَىٰ سَخْ، دࣹرࣹتْ لࣹيسْ دِ سࣹتَلࣹ دَانَڪَ لࣺݒّ، تࣹ بُ دࣹرࣹتْ تُورُوُلْ، نجࣺگَّلْ اَمُلْ؞
‏23 جُمْتُڪَايْ يࣷويِ دِ اَيْ مِسَالِ يࣴفْ يَ فَ اَسَمَانْ، يࣷويُ سَرَخْ لࣹيسْ ڪࣷ وَرࣷونَ سࣹتَلࣹ؞ ڪࣷنْ يࣴفِ اَسَمَانْ يَ ݖِ بࣷݒَّمْ، سَرَخْ يُ ڪࣷ گࣴنْ لࣹيسْ ڪࣷ وَرَ سࣹتَلࣹ؞ ‏24 ندَخْ بࣺرَبْ يُ سࣹلّ يُ نِتْ تَبَخْ، تࣹ مُيْ رࣷيَالُبْ بࣺرَبْ بُ دࣴگُّ بَ دࣷݝّ، دُ ݖِ لَ اَلْمَسِ دُگّ؞ وَايࣹ بِيرْ اَسَمَانْ ݖِ بࣷݒَّمْ لَ دُگّ نگِرْ تࣹيوَلْ نُ تࣹيْ، فَ ڪَنَمْ يࣵلَّ؞ ‏25 دُگُّلْ بِيرْ اَسَمَانْ اِتْ نگِرْ دِ سَرْخࣹ اَڪَ سَرْخࣹوَاتْ بࣷݒَّمْ، نࣷونࣹي سَرْخَلْڪَتْ بُ مَگْ بَ دَانْ دُگّࣹ اَتْ مُ نࣹڪّ فُ سࣹلّ فَا سࣹلّ، يٗبَّالࣹ دࣹرࣹتْ جُ دُلْ جࣷسَمْ؞ ‏26 لُ ڪࣷ مࣷيْ ݖَ نجࣵلْبࣺينُ اࣵدِّنَ لَيْ دَلࣹ سࣷنّ اَيْ يࣷونِ يࣷونْ؞ وَايࣹ تࣹيْ ݖِ مُجُگْ جَمࣷنࣷ يِ لَ فࣹيݧْ بࣹنّ يࣷونْ دࣷݝّ، بَ سَرْخࣹ بࣷݒَّمْ، نگِرْ فَرْ بࣵڪَّارْ؞ ‏27 ڪࣷنْ نِ اَبْ دࣷگَلْ سࣺدّࣹ نِتْ ݧِ بࣹنّ يࣷونْ دࣷݝّ بُ ݧُيْ دࣹي، سࣹينُبْ اࣵتّࣹ تࣷݒّ ݖِ، ‏28 نࣷونُ اِتَمْ لَ اَلْمَسِ دࣷونࣹ اَبْ سَرَخْ بࣹنّ يࣷونْ، نگِرْ گࣵدُّ بࣵڪَّارُ نِتْ ݧُ بَرࣹ، تࣹ دِنَ تࣹيوْسِوَاتْ، ݖِ يࣷونْ وُ جࣷتࣹيتُلْ دَرَ اَڪْ بࣵڪَّارْ، نگِرْ اِنْدِلْ ݧِ ڪࣷيْ سࣺينْتُ سࣹينُگْ مُݖّ؞
