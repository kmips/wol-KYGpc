اَجِ سَخْ جِ ڪࣹنّ لَ
6
‏1 لِي مࣷويْ سَنْتَانࣹ بِ، اَڪْ دࣷگَلِ يࣷونْ يࣹيڪْ اࣵتّࣹيْ يࣷونْ يِ سࣹينْ يࣵلَّ اَجِ سَخْ جِ سَنْتَانࣹ مَ جࣵنگَلْ لࣹينْ ڪࣷ، نگِرْ نگࣹينْ دِ ڪࣷ جࣴفࣹ ݖَ رࣺيوْ مَ نگࣹينْ جࣵلّ جࣴمْ، نگِرْ نَنگُجِ ڪࣷ، ‏2 ندَخْ نگࣹينْ رَگَلْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ، دِ سࣵمّ مبࣷولࣹمْ دࣷگَلِ يࣷونَمْ، اَڪْ سَنْتَانࣹيمْ، يِ مَ لࣹينْ دِ دࣺنڪّ، يࣹينْ اَڪْ سࣹينِ دࣷومْ، اَڪْ سࣹينِ سࣴتْ، سࣹينْ گِيرُ دُندّ، سُ ڪࣷ دࣹفࣹي نگࣹينْ گُدّ فَنْ؞ ‏3 ڪࣷنْ يࣹينْ إِسْرَاءِيلْ، دࣺگْلُلࣹينْ تࣹ سࣵمّࣷو جࣴفࣹ، ندَخْ نگࣹينْ بَاخْلࣹ تࣹ يࣷڪُّ بُ بَاخْ، نِ لࣹينْ ڪࣷ اَجِ سَخْ جِ سࣹينْ يࣵلَّيْ مَامْ وَخࣹ، ݖَ رࣺيوْ مَ مࣹيوْ مَاڪْ لࣹمْ جَ تُورࣷو؞
‏4 دࣺگْلُلࣹينْ، بࣵنِّ إِسْرَاءِيلْ، اَجِ سَخْ جِ سُنُ يࣵلَّ، اَجِ سَخْ جِ ڪࣹنّ لَ؞ ‏5 سࣹينْ يࣵلَّ اَجِ سَخْ جِ، سࣷݒّࣹلࣹينْ ڪࣷ سࣹينْ لࣺݒِّ خࣷلْ، اَڪْ سࣹينْ لࣺݒِّ بَڪَّنْ، اَڪْ سࣹينْ لࣺݒِّ دࣷولࣹ؞ ‏6 ڪࣵدُّ يِي مَ لࣹينْ دِ دࣺنڪّ بࣺسُبْ تࣹيْ، نَ دࣴڪّ ݖِ سࣹينْ خࣷلْ؞ ‏7 نَنگࣹينْ ڪࣷ خَمَلْ سࣹينِ دࣷومْ، دِ ڪࣷ وَخࣹ بُ نگࣹينْ تࣷوگࣹي سࣹينْ بِيرْ نࣺيگْ، اَڪْ بُ نگࣹينْ دِ دࣷخْ ݖِوْ يࣷونْ، اَڪْ بُ نگࣹينْ تࣴدّࣹي، اَڪْ بُ نگࣹينْ جٗگࣹي؞ ‏8 تَڪّلࣹينْ ڪࣷ، مُيْ فِرْندࣹ ݖِ سࣹينْ لࣷخࣷ، تࣹ نَ دࣷونْ اَمْ مبَنْدَلْ ݖِ سࣹينْ دِگَّنْتࣹيْ گࣴتْ؞ ‏9 تࣹ اِتَمْ بِندّلࣹينْ ڪࣷ ݖِ سࣹينْ ڪَوْ جࣴنِ بُنْتُ نࣺيگْ، اَڪْ ݖِ سࣹينْ بُنْتِ دࣴڪّ؞
‏10 بُ نگࣹينْ دࣹمࣹي بَ سࣹينْ يࣵلَّ اَجِ سَخْ جِ يٗبُّ لࣹينْ ݖَ رࣺيوْ مَ مُ گِݧَلࣷونْ سࣹينِ مَامْ إِبْرَاهِيمَ اَڪْ إِسْحَاقَ اَڪْ يَعْقُوبَ، نࣹ يࣹينْ لَ ڪࣷيْ جࣷخْ، مُيْ دࣴڪّ يُ مَگْ يُ بَاخْ يُ نگࣹينْ تَبَخُلْ، ‏11 ڪࣴرْ يَ فࣹيسࣹ مبࣷولࣹمْ لُ بَاخْ لُ نگࣹينْ ڪࣷ فࣹيسَلُلࣷونْ، اَڪْ تࣹينْ يُ نگࣹينْ گَسُلْ، اَڪْ گَرَبِ رࣹسࣹݧْ اَڪْ اࣷلِوْ يُ نگࣹينْ جࣴمْبَتُلْ؛ بُ نگࣹينْ ݖَ لࣹڪّࣹي بَ رࣹگّ، ‏12 وَتُّلࣹينْ سࣹينْ بࣷݒّ بَلَ نگࣹينَ فࣵتّࣹ اَجِ سَخْ جِ لࣹينْ گࣺنّࣹي رࣺيوُمْ مِسْرَ، ڪࣴرُگْ نجَامْ گَ؞ ‏13 سࣹينْ يࣵلَّ اَجِ سَخْ جِ، رَگَلْ-لࣹينْ ڪࣷ؞ مࣷومْ نگࣹينْ دِ جَامُ، تࣹ اَوْ تُرَمْ نگࣹينْ دِ گِݧࣹ؞ ‏14 بُلࣹينْ تࣷݒّ يࣹنࣹينْ يࣵلَّ، ݖِ يࣵلَّيْ خࣹيتْ يِ لࣹينْ سࣺقْ؞ ‏15 ندَخْ ڪَتْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ ݖِ سࣹينْ بِيرْ، ڪُ فِيرْ لَ؞ لُ ڪࣷ مࣷيْ سࣵنجُمْ اَجِ سَخْ جِ دِنَ تࣵڪّ ݖِ سࣹينْ ڪَوْ، تࣹ دِنَ لࣹينْ سࣵنڪّ، بَ رَافَلْ لࣹينْ ݖِ ڪَوْ سُوفْ؞ ‏16 بُلࣹينْ سࣹيتْلُ سࣹينْ يࣵلَّ اَجِ سَخْ جِ، نَ نگࣹينْ ڪࣷ سࣹيتْلࣷو وࣷونْ ݖَ مَاسَ؞ ‏17 سࣵمّلࣹينَ سࣵمّ سࣹينْ سَنْتَانࣹيْ يࣵلَّ اَجِ سَخْ جِ، اَڪْ سࣹيدࣹيْ يࣷونَمْ، اَڪْ دࣷگَلِ يࣷونَمْ، يِ مُ لࣹينْ دࣺنڪّ؞ ‏18 جࣴفࣹلࣹينْ نجُبْ، اَڪْ مبَاخْ گِ اَجِ سَخْ جِ رَفࣹتْلُ؞ سُ ڪࣷ دࣹفࣹي دِنگࣹينْ بَاخْلࣹ، بَ دࣹمْ، نَنگُجِ رࣺيوْ مُ بَاخْ مِ اَجِ سَخْ جِ گِݧَلࣷونْ سࣹينِ مَامْ، ‏19 تࣹ دِنگࣹينْ فَ دࣵقࣹ سࣹينْ نࣷونْ يࣺݒّ، نَ ڪࣷ اَجِ سَخْ جِ وَخࣹ وࣷونْ؞
‏20 بُ لࣹينْ سࣹينْ دࣷومْ لاَجࣹي اࣴلّࣴگْ، نࣹ لࣹينْ:‏ «اَنَ لُ وَرَلْ سࣹيدࣹيْ يࣷونْ يِ، اَڪْ دࣷگَلِ يࣷونْ يࣹيڪْ اࣵتّࣹيْ يࣷونْ يِ لࣹينْ سُنُ يࣵلَّ اَجِ سَخْ جِ دࣺنڪّ؟» ‏21 سُ بࣷوبَا وَخْلࣹينْ سࣹينْ دࣷومْ، نࣹ ڪࣷ:‏ «اَيْ جَامِ فِرْعَوْنَ لَنُ وࣷونْ ݖَ مِسْرَ، وَايࣹ اَجِ سَخْ جࣹي نُ گࣺنّࣹي مِسْرَ ݖِ دࣷولࣹيْ لࣷخࣷومْ؞ ‏22 اَجِ سَخْ جࣹي نُ دࣹفَلْ اَيْ فِرْندࣹ، اَڪْ ڪࣺيمَانْ يُ مَگْ تࣹ مࣹتِّ ݖِ ڪَوْ مِسْرَ اَڪْ ݖِ ڪَوْ فِرْعَوْنَ اَڪْ ݖِ ڪَوْ وَا ڪࣴرَمْ گࣺݒّ، نُ تࣹگْ ݖِ سُنُيْ گࣴتْ؞ ‏23 نُنْ نَگْ، مࣷو نُ فَ جࣴلࣹ، نگِرْ اِنْدِ نُ ݖِ رࣺيوْ مِ مُ گِݧَلࣷونْ سُنُيْ مَامْ، نࣹ نُنْ لَ ڪࣷيْ جࣷخْ؞ ‏24 اَجِ سَخْ جࣹي نُ سَنتّ نُيْ جࣴفࣹ مبࣷولࣹمْ دࣷگَلِ يࣷونْ يِي، تࣹ رَگَلْ سُنُ يࣵلَّ اَجِ سَخْ جِ، نگِرْ نُ بَاخْلࣹ بࣺسْ بُ نࣹڪّ، نگِرْ اِتَمْ مُ سࣵمّ سُنُ بَڪَّنْ نِ بࣺسُبْ تࣹيْ جِي؞ ‏25 سُنُ نجࣹڪّ لَيْ دࣷونْ نَگْ، ندࣹگَمْ سࣵمّࣷو نَنࣷو جࣴفࣹ مبࣷولࣹمْ سَنْتَانࣹ بِي، فِ سُنُ ڪَنَمْ يࣵلَّ اَجِ سَخْ جِ، نِ مُ ڪࣷ سَنْتَانࣹي؞»
