سࣵمّ بࣺگࣹ نَ مࣹنّ خَرَمْ مُ فࣹيݧْ
15
‏1 مبࣷولࣹمْ جُوتِڪَتْ يࣹيڪْ بࣵڪَّارْڪَتْ يِ ݧࣷو دَانْ ݧࣴوْ ݖِ عِيسَىٰ، دِ ڪࣷ دࣺگْلُ؞ ‏2 ݖِ بِيرْ لࣷولُ فَرِسࣹنْ يَ اَڪْ فِرِڪَتِ يࣷونْ يَ دِ ݧُرُمْتُ نَانْ:‏ «ڪِي مࣷومْ بࣵڪَّارْڪَتْ يِ لَيْ نࣹيخࣷولْ، دِ بࣷڪّ اَڪْ ݧࣷومْ لࣹڪّ!‏» ‏3 عِيسَىٰ نَگْ لࣺيبْ لࣹينْ لِي، نࣹ لࣹينْ:‏ ‏4 «اَنَ ڪَنْ ݖِ يࣹينْ مࣷويْ اَمْ تࣺيمࣺيرِ خَرْ، رࣺيرْلࣹ ݖِ مࣹنّ، تࣹ بࣵيِّوُلْ ݖَ اࣵلّ بَ جُرٗومْ ݧࣹينتّ فُڪِّ خَرْ يَ اَڪْ جُرٗومْ ݧࣹينتّ (99)، نگِرْ تࣷݒِّ مَ رࣺيرْ، بَ گِسْ ڪࣷ؟ ‏5 بُ ڪࣷ گِسࣹي لَيْ بࣺگْ، تࣹگْ ڪࣷ ݖَ مبَگّ مَ، يٗبُّ ڪࣴرْ گَ؞ ‏6 تࣹ بُ اࣵگّࣹي ݖَ ڪࣴرْ گَ، خَرِتْ يَاڪْ دࣴڪّ يَ لَيْ وࣷو، نࣹ لࣹينْ:‏ ‹ڪَايْلࣹينْ نُ بࣺگࣹ، سَمَ خَرْ مَ رࣺيرࣷونْ، گِسْ نَا ڪࣷ!‏›
‏7 «مَا لࣹينْ وَخْ، نࣹ لࣹينْ:‏ بُ بࣹنّ بࣵڪَّارْڪَتْ دࣷݝّ تُوبࣹي، نࣷونُ رࣹڪّ لَيْ دࣷونࣹ مبࣺگْتࣹ فَ اَسَمَانْ، بَ رَوْ مبࣺگْتࣹ مَ فَيْ اَمْ، بُ جُرٗومْ ݧࣹينتّ فُڪّ اَڪْ جُرٗومْ ݧࣹينْتِ نِتْ جُبࣹي، تࣹ سࣷخْلَوُݧࣷو تُوبْ؞
جِگࣺينْ بࣺگࣹ نَ تَڪَّايَمْ لُ فࣹيݧْ
‏8 «تࣹ اِتْ اَنَ جَنْ جِگࣺينْ مࣷويْ اَمْ فُڪِّ لِبِدࣷورْ، بࣹنّ رࣺيرْ ݖَ، تࣹ تَالُلْ اَبْ لࣵمݒّ، بُوبْ نࣺيگْ بَ، سࣹيتَ سࣹيتْ، بَ گِسْ ڪࣷ؟ ‏9 بُ ڪࣷ گِسࣹي، خَرِتْ يَاڪْ دࣴڪّ يَ لَيْ وࣷو، نࣹ لࣹينْ:‏ ‹ڪَايْلࣹينْ نُ بࣺگࣹ، لِبِدࣷورْ بَ رࣺيرࣷونْ، گِسْ نَا ڪࣷ؞› ‏10 مَا لࣹينْ وَخْ نࣹ لࣹينْ:‏ نࣷونُ رࣹڪّ لَيْ دࣷونࣹ مبࣺگْتࣹ فَ مَلاَڪَيْ يࣵلَّ يَ، بُ بࣹنّ بࣵڪَّارْڪَتْ دࣷݝّ تُوبࣹي؞»
بَايْ بࣺگࣹ نَ دࣷومَمْ جُ فࣹيݧْ
‏11 عِيسَىٰ نࣹيتِ:‏ «نِتَ اَمࣷونْ ݧَارِ دࣷومْ يُ گٗورْ؞ ‏12 ڪَ گࣴنْ دِ ندَوْ نࣹ بَايْ بَ:‏ ‹بَايْ، جࣷخْ مَ ݖࣺرْ بِ مَ يࣷونْ مَيْ ݖِ سُنُ اَلَلْ؞› بَ لࣷولُ اَمࣹي بَايْ بَ سࣺدَّلࣹ لࣹينْ اَلَلَمْ؞ ‏13 اَيْ فَنْ يُ نࣺيوْ گَنَّاوْ گِ، ڪَ گࣴنْ دِ ندَوْ جَايْ لࣺݒَّمْ، دَلْدِ تُڪِّ رࣺيوْ مُ سࣷرࣹ، دِ تࣷݒّ بَڪَّنَمْ بَ يࣵقْ فَ اَلَلَمْ؞ ‏14 گَنَّاوْ بَ مُ سࣵنڪࣹي لࣺݒّ، خِيفْ بُ مࣹتِّ دِڪَّلْ رࣺيوْ مَ، مُ دَلْدِ تࣵمْبَلࣹي ݧࣵڪّ؞ ‏15 ݖِ ڪَوْ لࣷولُ مُ دࣹمْ، دِ سُرْگَوُ ڪُ ݖࣷسَانࣷو ݖَ رࣺيوْ مࣷومَ، ڪࣷوڪُ يࣹبَلْ ڪࣷ ݖَ اَيْ تࣷولَمْ، مُيْ سࣵمّ اَيْ مبَامْ-خُوخْ؞ ‏16 مُ دࣹمْ بَ نࣷونْ نگࣺيجَ لࣹڪّ ݖَ خࣷلِّتِ دࣷومِ گَرَبْ يَ مبَامْ-خُوخْ يَيْ لࣹڪّ، فَجࣹبْ خِيفَمْ، تࣹ ڪࣹنّ مَيُ ڪࣷ ݖَ؞ ‏17 خࣹلْ مَ مُجّ دࣺلْسِ، مُ نࣹ:‏ ‹ݧَاتَ سُرْگَ لَ سَمَ بَايْ اَمْ، سࣹينْ لࣹڪّ نࣹ گࣵݧّ، مَنْ، اَبْ خِيفْ دِ مَ رࣹيْ فِي!‏ ‏18 نَا جٗگْ، دࣹمْ ݖَ سَمَ بَايْ، نࣹ ڪࣷ:‏ «بَايْ، تࣷوݧْ نَا يࣵلَّ، تࣷوݧْ لَ؛ ‏19 يࣹيࣷوتُمَا تُدّࣷو سَخْ سَ دࣷومْ؛ لࣺيگِ دࣹفْ مَ نِ ڪُ بࣷڪّ ݖِ سَيْ سُرْگَ؞» ›
‏20 «مُ جٗگْ، نگِرْ دࣹلُّ ڪࣴرْ بَايَمْ؞ فُ سࣷرࣹ لَ ڪࣷ بَايَمْ سࣺينࣹ، مُيْ دِڪّ؞ مُ ݧࣹيوَنْتࣹ ڪࣷ، دَوْ، لَخَسُ ڪࣷ، فٗونْ ڪࣷ؞ ‏21 دࣷومْ جَ نࣹ ڪࣷ:‏ ‹بَايْ، تࣷوݧْ نَا يࣵلَّ، تࣷوݧْ لَ؛ يࣹيࣷوتُمَا تُدّࣷو سَخْ سَ دࣷومْ؞› ‏22 تࣹيوُلْ بَايْ بَ نࣹ اَيْ سُرْگَامْ:‏ ‹گَاوْلࣹينْ اِنْدِ مبُبّ مِ گࣴنَ رَفࣹتْ، سࣷلَلْ ڪࣷ، تࣹ نگࣹينْ تَڪَّلْ ڪࣷ اَبْ جَارࣷ ݖِ لࣷخࣷومْ، سࣷلَلْ ڪࣷ اَيْ دࣵلّ؞ ‏23 اِنْدِلࣹينْ سࣴلُّوْ يَفَلْ وَ، رࣹيْ، نُ لࣹڪّ، تࣹ بࣵنّࣹيخُ؛ ‏24 ندَخْ سَمَ دࣷومْ جِي مࣷو دࣹي وࣷونْ تࣹ دࣹڪِّ نَ، مࣷو رࣺيرࣷونْ تࣹ فࣹيݧْ نَ؞› ݧُ دَلْدِ تࣵمْبَلࣹي بࣵنّࣹيخُ؞
‏25 «لࣷولاَ نگَ فࣹڪّ مَگْ جَ ݖَ تࣷولْ يَ؞ بَ مَگْ جَ دِڪّࣹي بَ جُبْ ڪࣴرْ گَ، ݖَ لَ دࣺگّ ݒࣷ مَ اَڪْ ݒࣹݖّ مَ؞ ‏26 مُ وࣷو ڪࣹنّ ݖَ خَلࣹ يَ، لاَجْ ڪࣷ لُ خࣹوْ؞ ‏27 مُ نࣹ ڪࣷ:‏ ‹سَ رَڪّ جَا دࣺلْسِ، سَ بَايْ رࣹيْ سࣴلُّوْ يَرْ وَ، ندَخْ لِ مُ دࣺلْسࣹيڪْ جࣵمّ؞›
‏28 «مَگْ جَ نَگْ مࣹرْ، نَنگُوُلَ اَگْسِ، بَايَمْ گࣺنّ، دِ ڪࣷ سࣷݧّ؞ ‏29 مُ اࣵدُّ، نࣹ بَايَمْ:‏ ‹سࣹيتَلْ ݧَاتَ اَتْ لاَ لَ لِگّࣺييَلْ، سَ ندِگَلْ، مَسُمَ ڪࣷو مࣷيْ، تࣹ مَنْ، مَسُلࣷو مَا مَيْ سَخْ اَسْ تࣹفْ سُ نُ بࣺگْلࣷو، مَاڪْ سَمَيْ خَرِتْ؞ ‏30 سَ دࣷومْ جِي دِڪّ، گَنَّاوْ بَ مُ يࣵقࣹي سَ اَلَلْ ݖِ اَيْ گࣵنݖّ، نگَ رࣹيَلْ ڪࣷ سࣴلُّوْ يَفَلْ وِ!‏› ‏31 مُ نࣹ ڪࣷ:‏ ‹دࣷومْ، يَوْ دࣹ نَڪَ جࣹڪّ مَنْ نگَ نࣹڪَّلْ، تࣹ مبࣷولࣹمْ لُ مَ اَمْ، يَا ڪࣷ مࣷومْ؞ ‏32 وَايࣹ مبࣺگْتࣹ اَڪْ بࣵنّࣹيخْ وَرْ نَ، ندَخْ سَ رَڪّ جِي مࣷو دࣹي وࣷونْ تࣹ دࣹڪِّ نَ، مࣷو رࣺيرࣷونْ تࣹ فࣹيݧْ نَ!‏›»
