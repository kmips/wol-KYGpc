يࣵلَّ يٗنِّ نَ بَرْنَبَ اَڪْ سٗولْ
13
‏1 ݖَ بِيرْ مبࣷولࣷومْ گࣴمْڪَتْ، ݧَ وࣷونْ ݖَ دࣴڪّ بَ ݧُيْ وَخْ اࣵنݖࣷسْ، اَمࣷونْ نَ اَيْ يࣷنࣹنتّ اَڪْ اَيْ جࣵنگّلࣹڪَتْ:‏ ݧُيْ بَرْنَبَ، اَڪْ سِمࣹيࣷنْ مِ ݧُ دِݒّࣹي ݧُولࣹ، اَڪْ لُسِيُسْ مِ بَوࣷو سِرࣹنْ¤، اَڪْ مَنَيࣹنْ مِ يَرࣷونْدࣷو اَڪْ اࣹرࣷدّ بࣷرࣷومْ گَلِلࣹ، اَڪْ سٗولْ؞ ‏2 ݧࣷوݧࣷو دَجࣹ وࣷونْ دِ مࣵگَّلْ بࣷرࣷومْ بِ، بࣷولࣹ ڪࣷوڪْ ڪࣷورْ؞ ݖِ بِيرْ لࣷولُ نࣷو گُ سࣹلّ گِ نࣹ:‏ «بࣹرَلْ-لࣹينْ مَ نَگْ بَرْنَبَ اَڪْ سٗولْ نگِرْ لِگّࣺييْ، بِ مَ لࣹينْ وࣷويࣹ؞» ‏3 بَ لࣷولُ اَمࣹي ݧُ وࣷورْ، بࣷولࣹ ڪࣷوڪْ ݧَانْ، دَلْدِ لࣹينْ تࣹگْ سࣹينِ لࣷخࣷ، بَ نࣷݒِّ يِوِ لࣹينْ؞
دُنُ سِݒَّرْ جࣷتْ نَ ݖِ ڪࣵدُّگْ يࣵلَّ
‏4 ݧࣷوݧُ لَ نࣷو گُ سࣹلّ گِ يࣹبَلْ، ݧُ دࣹمْ دࣴڪّ بَ ݧُيْ وَخْ سࣹلُسِ، دُگّࣹ فَ گَالْ، جࣴمْ دُنْ بَ ݧُيْ وَخْ سِݒَّرْ؞
‏5 ݧَ نگَ تࣹيرࣹجِ دࣴڪّ بَ ݧُيْ وَخْ سَلَمِنْ، دَلْدِ فَيْ يࣺگْلࣹ ڪࣵدُّگْ يࣵلَّ ݖَ جࣵنگُيْ يَوُتْ يَ، ڪُ ݧُيْ وَخْ يࣷوَانْ مࣵرڪّ اࣵندّ اَڪْ ݧࣷومْ، دِ لࣹينْ جࣵݒّلࣹ؞
‏6 ݖِ بِيرْ لࣷولُ ݧُ وࣴرْ دُنْ بَ بࣺݒّ، بَ يࣹگّ ݒَفࣷسْ، ݒࣺييْ بَ؞ ݧُ تَسࣹ فَاڪْ اَبْ ݧࣹينگࣷڪَتُبْ يَوُتْ بُيْ يࣷنࣹنتّ-يࣷنࣹنتّلُ، ݧُ دِ ڪࣷ وَخْ بَرْ عِيسَىٰ؞ ‏7 مَ نگَ بࣷڪّࣷونْ ݖَ تࣷݒُّمْ سࣹرْگِيُسْ ݒٗولُسْ¤ بࣷرࣷومْ دࣴڪّ بَ، نِتْ ڪُ رَفࣹتْ خࣹلْ؞ ڪࣷوڪُ نَمَّ دࣺگّ ڪࣵدُّگْ يࣵلَّ، دَلْدِ وࣷولُ بَرْنَبَ اَڪْ سٗولْ؞ ‏8 اࣹلِمَسْ، مُ فِرِ ݧࣹينگࣷڪَتْ بِ، دِ لࣹينْ گࣵلَّنڪࣷورْ نَگْ، تࣹيْ جࣺيمَ فࣵبِّ بࣷرࣷومْ رࣺيوْ مَ، بَ دُ گࣴمْ؞ ‏9 ݖِ ڪَوْ لࣷولُ سٗولْ مِ ݧُيْ وَخْ ݒٗولْ فࣹيسࣹ نࣷو گُ سࣹلّ گِ؞ مُ خࣷولْ ݧࣹينگࣷڪَتْ بِ جࣵڪّ، ‏10 نࣹ ڪࣷ:‏ «يَوْ نِتْ ڪِ فࣹيسْ اَڪْ گࣺݒّ خࣹيتُ نجُبْلَݝْ اَڪْ ندࣴنگّتࣹ، دࣷومُ سࣹيْتَانࣹ جِ، نࣷونُ گࣺݒّ نجُبْ، دࣷو بࣵيّࣹي لُنڪَلْ يࣷونِ نجُبْ يُ بࣷرࣷومْ بِ؟ ‏11 لࣺيگِ نَگْ لࣷخࣷلْ بࣷرࣷومْ بَا نگِي فِ سَ ڪَوْ؛ دِنگَ گُمْبَ، بَ دࣷوتُلࣷو گِسْ جَنتّ بِ اَبْ دِيرْ؞» ݖَ سَا سَ گِسْ-گِسْ بُ نَقَرِ اَڪُگْ لࣴنْدࣴمْ نࣹ مِلِبْ ݖِ ڪَوَمْ، مُيْ وࣴنْدࣹيلُ، دِ وُتْ ڪُ ڪࣷ وࣷمَّتْ؞ ‏12 بَ لࣷولُ اَمࣹي بࣷرࣷومْ دࣴڪّ بَ گِسْ لَ خࣹوْ، دَلْدِ گࣴمْ، دِ يࣺيمُ ݖِ لِ مُ جࣵنگࣹ ݖِ سَنگّ بِ؞
وَا ݒِسِدِ جࣷتْ نَݧُ ݖِ ڪࣵدُّگْ يࣵلَّ
‏13 ݒٗولْ اَڪْ ݧِ مُ اࣵنْدَلْ نَگْ بࣵيِّڪࣷو ݒَفࣷسْ، دُگّ گَالْ، جࣴمْ ݒࣹرْگࣹ، ݖَ دِيوَانُ ݒَمْفِلِ؞ يࣷوَانْ مࣷومْ تࣵگّࣷو اَڪْ ݧࣷومْ، دࣹلُّ يࣹرُسَلࣹمْ؞
‏14 ݖِ ڪَوْ لࣷولُ، ݧࣷومْ ݧُ جٗگࣹ ݒࣹرْگࣹ، جࣵلّ بَ اࣵنݖࣷسْ ݖَ دِيوَانُ ݒِسِدِ؞ ݧُ دࣹمْ جࣵنگُ بَ، دُگّ، تࣷوگْ ݖَ بࣺسُبْ نࣷفْلاَيْ بَ؞ ‏15 گَنَّاوْ بَ تَرِبْ دࣷگْ يَ ݖَ تࣺيرࣹبْ يࣷونُ مُوسَىٰ، اَڪْ تࣺيرࣹبْ يࣷنࣹنتّ يَ سࣷتّࣹي، نجِيتِ جࣵنگُ بَ يٗبَّنْتࣹ لࣹينْ ڪࣵدُّ، نࣹ لࣹينْ:‏ «بࣷڪّ يِ، بُ نگࣹينْ اَمࣹي ڪࣵدُّ گُ نگࣹينْ ݧَاخࣹ مبࣷولࣷو مِ، وَخْلࣹينْ؞» ‏16 ݒٗولْ جٗگْ، تࣵلَّلْ لࣷخࣷومْ، نࣹي لࣹينْ:‏
«يࣹينْ بࣷڪِّ إِسْرَاءِيلْ اَڪْ يࣹينْ جَامْبُرِ رَگَلْڪَتِ يࣵلَّ يِ، دࣺگْلُلࣹينْ!‏ ‏17 يࣵلَّيْ بࣵنِّ إِسْرَاءِيلْ جِي مࣷو تࣵنّࣷونْ سُنُيْ مَامْ، مࣷو يَاتَلْ خࣹيتْ وِ، بَ ݧُيْ گَنࣹيَانْ ݖَ مِسْرَ؛ مࣷو لࣹينْ فَ گࣺنّࣹي ڪࣵتَّنُ لࣷخࣷومْ؞ ‏18 لُ تࣷلّࣷوڪْ ݧࣹينتّ فُڪِّ اَتْ لَ لࣹينْ مُݧَلْ ݖَ مࣵنْدِݝْ مَ؞ ‏19 مࣷو فَاگَاگَلْ جُرٗومْ ݧَارِ خࣹيتْ ݖَ رࣺيوُمْ ڪَنَانْ، دَلْدِ مُورَلࣹ سُنُيْ مَامْ رࣺيوُمْ ڪَنَانࣹينْ ݧَ؞ ‏20 لࣷولُ لࣺݒَّ نگِ اَمْ ݖِ دِيرُ ݧࣹينْتِ تࣺيمࣺيرِ اَتْ اَڪْ جُرٗومْ فُڪّ (450)؞
«گَنَّاوْ لࣷولُ مࣷو سࣵڪَّلْ سُنُيْ مَامْ اَيْ نجِيتْ، بَ ݖِ جَمࣷنࣷيْ يࣷنࣹنتّ يࣵلَّ سَمِيࣹلْ؞ ‏21 بَ لࣷولُ وࣺيࣹي، ݧُيْ سࣵڪُّ بُورْ، يࣵلَّ جࣷخْ لࣹينْ دࣷومُ ڪِسْ، سٗولْ، مِ سࣷقِڪࣷو ݖِ گِيرُگْ بࣹݧَمِنْ، مُ فَلُ دِيرُ ݧࣹينتّ فُڪِّ اَتْ؞ ‏22 مُ جࣴلࣹ ڪࣷ فَ، سَمْݒَلْ لࣹينْ دَاوُدَ، مُيْ بُورْ بُ مُ سࣹيدࣹيلْ نࣹ:‏ ‹مَا رࣵݧّࣹي دَاوُدَ دࣷومُ يࣹسࣹ، جِ سَمَ خࣷلْ جُبࣷولْ، تࣹ مࣷويْ جࣴفࣹ مبࣷولࣹمْ سَمَيْ ݖࣷوبَرࣹ؞›
‏23 «ݖِ اَسْڪَنُ ڪࣷوڪِي لَ يࣵلَّ اِنْدࣹي عِيسَىٰ، نَ مُ ڪࣷ دِگࣹي وࣷونْ، مُيْ مُسَلْڪَتْ ݧࣹيلْ إِسْرَاءِيلْ؞ ‏24 لاَتَ مُيْ ݧࣴوْ، يَحْيَىٰ مࣷو دࣷونْ يࣺينࣹ نگِرْ وَا بࣵنِّ إِسْرَاءِيلْ گࣺݒّ سٗوبُ ݖِمْ ندࣷخْ، جࣴمَّلࣹ ڪࣷ سࣹينُگْ تُوبࣹيلْ؞ ‏25 بَ يَحْيَىٰ دِ جࣹيخَلْ اَوْ سَسَمْ لَ نࣹ:‏ ‹اَنَ ڪُ نگࣹينْ فࣷوگْ نࣹ مࣷومْ لاَ؟ دُمَ ڪِ نگࣹينْ فࣷوگْ، وَايࣹ اَمْ نَ ڪُيْ ݧࣴوْ نِي سَمَ گَنَّاوْ، ڪُ مَ يࣹيࣷووُلَ سُمِّلْ سَخْ اَيْ دࣵلَّمْ؞›
‏26 «بࣷڪّ يِ، يࣹينْ اَسْڪَنُ إِبْرَاهِيمَ، يࣹينْ اَڪْ جَامْبُرِ رَگَلْڪَتِ يࣵلَّ يِ ݖِ سࣹينْ بِيرْ، نُنْ لَݧُ يٗنّࣹي ڪࣵدُّگْ مُݖّ گِي؞ ‏27 وَا يࣹرُسَلࣹمْ، اَڪْ سࣹينِ ڪِلِفَ نَگْ خࣵمِّوُݧُ وࣷونْ عِيسَىٰ، تࣹ بَ ݧُ دَانࣹي عِيسَىٰ لَݧُ سࣷتَّلْ وَخُ يࣷنࣹنتّ، يَ ݧُ دَانْ بِرَلْ بࣺسُبْ نࣷفْلاَيْ بُ نࣹڪّ؞ ‏28 گِسُݧُ ݖِ مࣷومْ لࣹنّ لُ ڪࣷ اࣵتّࣹبْ دࣹي دَبࣹ، تࣹ تࣹيوُلْ ݧُ سࣵڪُّ ݒِلاَتْ رࣹيْلُ ڪࣷ؞ ‏29 گَنَّاوْ بَ ݧُ مَتَلࣹي مبࣷولࣹمْ لُ ݧُ بِندّ ݖِ مࣷومْ، لَݧُ ڪࣷ وࣵݖّࣹي ݖَ بَنتّ بَ، دُگَلْ ڪࣷ بࣵمّࣹيلْ؞ ‏30 وَايࣹ يࣵلاَّ ڪࣷ دࣹڪَّلْ، ‏31 تࣹ دِيرُ فَنْ يُ بَرࣹ لَ فࣹيݧُ ݧَ اࣵنْدࣷونْ اَڪْ مࣷومْ، جٗگࣹ دِيوَانُ گَلِلࣹ، دࣹمْ يࣹرُسَلࣹمْ، تࣹ ݧࣷوݧࣷويْ اَيْ سࣹيدࣹيمْ تࣹيْ ݖِ اَسْڪَنْ وِ؞ ‏32 نُنْ نَگْ نࣷو لࣹينْ دِ اࣵگّࣹ خِبَارُ جࣵمّ بِي:‏ لِ يࣵلَّ دِگࣷونْ سُنُيْ مَامْ، ‏33 نُنْ سࣴتْ يِ لَ ڪࣷ دࣹفَلْ، بَ مُ يࣺڪَّتࣹي عِيسَىٰ، تࣹ نࣷونࣹي لَݧُ ڪࣷ بِنْدࣹ ݖِ ݧَارࣹيلُ سَارُ سَبٗورْ، نࣹ:‏
‹يَا دِ سَمَ دࣷومْ،
بࣺسْ نِڪِ تࣹيْ مَا لَ جُرْ؞›
‏34 نࣷونُ ڪࣷ يࣵلَّ دࣹڪَّلࣹ ندࣹڪِّتࣹ لُ دࣹي تࣷݒَّتُلْ، تࣹ يَرَمَمْ دُ يࣵقُ مُڪّ، نࣷونࣹي لَ ڪࣷ يࣵلَّ وَخࣹ وࣷونْ، نࣹ:‏
‹مَا لَيْ جࣷخْ بَرْڪࣹيْ دَاوُدَ يُ سࣹلّ يِ تࣹ وٗورْ؞›
‏35 مࣷو تَخْ مُ وَخَاتْ فࣹنࣹينْ نࣹ:‏
‹دࣷو بࣵيِّ سَ ڪُ سࣹلّ ڪِ، مُ يࣵقُ؞›
‏36 دَاوُدَ مࣷومْ، گَنَّاوْ بَ مُ دࣹفࣹي ݖࣷوبَرࣹيْ يࣵلَّ ݖِ جَمࣷنࣷومْ، نࣹلَوْ نَ، بَ ݧُ رࣷبْ ڪࣷ ݖَ وࣹتِ مَامَمْ يَ، تࣹ يࣵقُ نَ؞ ‏37 وَايࣹ مࣷومْ مِ يࣵلَّ دࣹڪَّلْ، يࣵقُوُلْ؞
‏38 «ڪࣷنْ نَگْ بࣷڪّ يِ، خَمْلࣹينْ نࣹ ݖِ عِيسَىٰ مࣷومُ لَݧُ لࣹينْ بِرَلࣹ نجࣺگَّلُگْ بࣵڪَّارْ؞ تࣹ مبࣷولࣹمْ لُ نگࣹينْ تࣴلࣹ وࣷونَ وࣵݖّࣷولْ، بَ سࣹينْ نجُبْ مَتْ ݖِ يࣷونُ مُوسَىٰ، ‏39 ڪࣺݒّ ڪُ گࣴمْ عِيسَىٰ، ݖِ مࣷومْ لَيْ وࣵݖّࣷوڪْ لࣷولُ لࣺݒّ، بَ اَگْ نجُبَمْ مَتْ؞ ‏40 وَتُّلࣹينْ نَگْ بَ لِ ݧُ وَخْ ݖِ تࣺيرࣹيْ يࣷنࣹنتّ يِ بَݧْ لࣹينَ دَلْ؞ ݧُ نࣹ:‏
‏41 ‹يࣹينْ خࣹيبَاتࣹڪَتْ يِ، خࣷولْ-لࣹينْ،
جࣷمِّلࣹينْ تࣹ نࣹ مࣹسْ،
ندَخْ جࣴفْ جِ مَيْ دࣹفْ ݖِ سࣹينْ جَمࣷنࣷ،
ڪُ لࣹينْ ڪࣷ وَخࣷونْ سَخْ،
دُنگࣹينْ ڪࣷ گࣴمْ؞›»
‏42 بَ ݧُ گࣺنّࣹي ݖَ جࣵنگُ بَ، دَݧُ لࣹينَ ݧَانْ ݧُ بَامْتُ ڪࣵدُّ گࣷوگُ بࣺسُبْ نࣷفْلاَيْ بَ ݖَ تࣷݒّ؞ ‏43 بَ مبࣷولࣷو مَ تَسَارࣷو، بَرࣹ نَ اَيْ يَوُتْ اَڪْ تُوبࣹينْ يُيْ جَامُ يࣵلَّ يُ تࣷݒّ ݖِ ݒٗولْ اَڪْ بَرْنَبَ؞ ݧُيْ وَخْ اَڪْ نِتْ ݧِ، دِ لࣹينْ ݧَاخْ، نگِرْ ݧُ سَخْ ݖِ دࣷيْلࣷو يِوُ يࣵلَّ؞
‏44 بࣺسُبْ نࣷفْلاَيْ بَ ݖَ تࣷݒّ، دَانَڪَ وَا دࣴڪّ بࣺݒَّ دَجࣹ، نگِرْ دࣺگْلُ ڪࣵدُّگْ بࣷرࣷومْ بِ؞ ‏45 بَ يَوُتْ يِ گِسࣹي مبࣷولࣷو مَ نَگْ، ڪِݧَانْ مَنْ لࣹينْ، ݧُيْ وࣹدِّ وَخِ ݒٗولْ، دِ ڪࣷ سَاگَ؞ ‏46 بَ لࣷولُ اَمࣹي فِتْ لَ لࣹينْ ݒٗولْ اَڪْ بَرْنَبَ تࣷنْتࣷو، نࣹ لࣹينْ:‏ «يࣹينْ يَوُتْ يِ دࣹ لَنُ وَرَ جࣴڪَّ يࣺگَلْ ڪࣵدُّگْ يࣵلَّ گِ، وَايࣹ گَنَّاوْ گࣵنْتَلْ نگࣹينْ ڪࣷ، بَ جࣷمْبَلࣹ سࣹينْ بࣷݒّ اَگْ تࣹخࣹ بَ فࣵوّ، نُ نگِي دِ وَلْبَتِڪُ ݖِ جَامْبُرْ ݧِ؞ ‏47 ندَخْ نࣷونُ لَ نُ ڪࣷ بࣷرࣷومْ بِ سَنْتࣹ بَ مُ نࣹي:‏
‹مَا لَ تَبّ نگَيْ لࣹيرُ جَامْبُرْ ݧِ،
نگِرْ نگَ جࣷتّلِجِ مُݖّ گِي، بَ ݖَ ݖَتِ اࣵدِّنَ؞›»
‏48 بَ جَامْبُرْ ݧَ دࣺگّࣹي لࣷولُ، دَݧࣷو بࣺگْ، دِ مࣵگَّلْ ڪࣵدُّگْ بࣷرࣷومْ بِ؛ ݖِ بِيرْ لࣷولُ ݧَ دࣷگَلْ بَ جَگْلࣹيلْ اَگْ تࣹخࣹ بَ فࣵوّ، ݧࣷومْ ݧࣺݒّ دَلْدِ گࣴمْ؞
‏49 ڪࣵدُّگْ بࣷرࣷومْ بِ نَگْ دِ لَوْ ݖَ دِيوَانْ بࣺݒّ؞ ‏50 تࣹيوُلْ يَوُتْ يَ خِيرْتَلْ تُوبࣹينْ يُ جِگࣺينْ يِ گࣴنَ رࣵݧِّڪُ، اَڪْ ڪࣵنگَمْ يُ گٗورْ يِ ݖِ دࣴڪّ بِ؞ ݧُ دَلْدِ بُندّخَتَلْ ݒٗولْ اَڪْ بَرْنَبَ، گࣺنّࣹ لࣹينْ رࣺيوْ مَ؞ ‏51 ݒٗولْ اَڪْ بَرْنَبَ نَگْ فࣴگّ سࣹينْ ݒࣴنْدُبْ ندࣴگُّيْ تࣵنڪّ، سࣹيدࣹيلْ لࣹينْ ڪࣷ سࣹينْ نگࣵنْتَلْ، دࣷورَ دࣹمْ دࣴڪّ بَ ݧُيْ وَخْ اِڪࣷݧُمْ؞ ‏52 ݖِ بِيرْ لࣷولُ تَالِبࣹ يَ دࣹسࣹ مبࣺگْ اَڪْ نࣷو گُ سࣹلّ گِ، بَ سࣹينْ خࣷلْ فࣹيسْ؞
