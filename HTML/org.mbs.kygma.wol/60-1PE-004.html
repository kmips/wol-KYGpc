نࣹڪَّلْ-لࣹينْ يࣵلَّ
4
‏1 گَنَّاوْ اَلْمَسِ نَگْ مࣷو دَجْ ݖࣷونࣷبْ يَرَمَمْ، يࣹينْ اِتَمْ گࣵنَّايࣷولࣹينْ بࣷوبُ يࣺينࣹ، ندَخْتࣹ ڪُ دَجْ ݖࣷونࣷبْ يَرَمْ، تࣵگّࣷو نَاڪْ بࣵڪَّارْ، ‏2 نگِرْ بَݧَتࣹي تࣷݒّ خࣹمّࣹمْتࣺيفِ بَڪَّنْ، خَنَا دِ تࣷݒّ ݖࣷوبَرࣹيْ يࣵلَّ، دِيرْ بِ ڪࣷ دࣹسࣹ ݖِ ڪَوْ سُوفْ؞ ‏3 ندَخْ دࣺمبّ ڪَتْ يࣵگّ نگࣹينَ دࣹفْ ݖࣷوبَرࣹيْ خࣹيتْ يِ خَمُلْ يࣵلَّ، ݖِ اَيْ ݧَاوْتࣺيفْ، اَڪْ تࣷݒّ سࣹينْ بَڪَّنْ؞ دَنگࣹينْ دَا مࣵنْدِ اَڪَ خَوَارࣹ، دࣴڪّࣹ نَانْ بِيݧْ، دِ جَامُ تُورْ يُ سࣹيخْلُوُ، وَايࣹ نَ فِ يࣹمْ نَگْ؞ ‏4 ݧَ نگࣹينْ ݖِ دَا اࣵنْدَلْ ݧࣷو وَارُ تࣹيْ، بَ دِ لࣹينْ سࣷسَلْ، ندَخْ لِ نگࣹينْ اࣵنْدَتُلْ اَڪْ ݧࣷومْ، دَوْ وُتِ گࣷوگُ يࣵقُتࣹ گُ اَفْ، ‏5 وَايࣹ ݧࣷو ڪࣷيْ لَيࣷو اَڪْ ڪِ نَرَ اࣵتّࣹ ݧِيْ دُندّ، اَڪْ ݧِ دࣹي؞ ‏6 لِ ݧُ ݖِ جُبْلُ مࣷو دِ، ݧِ دࣹي تࣹيْ سَخْ، ݧُ اࣵگّࣹ لࣹينْ خِبَارُ جࣵمّ بِ، نگِرْ بُ ݧُ لࣹينْ اࣵتّࣹي اࣵتّࣹبْ نِتْ ݧُ سࣹينْ يَرَمُ سُوخْ وَرَ دࣹي سَخْ، دُ تࣹي ݧُيْ دُندّ تࣹيْ، ندَخْ نࣷووُگْ يࣵلَّ؞
بࣹنّࣷو مࣷو وَرْ گࣴمْڪَتْ ݧِ
‏7 تࣹ اِتَمْ مُجُگْ لࣺݒّ دࣴگْمَلْ نَ؛ ڪࣷنْ نَگْ جِيتَلْ-لࣹينْ سࣹينُمْ خࣹلْ، تࣹ اࣵندّ اَڪْ سࣹينْ سَگࣷ، نگِرْ نگࣹينْ مَنْ دِ ݧَانْ؞ ‏8 وَايࣹ لِ ݖِ رَوْ مࣷويْ سࣷݒَّنْتࣹلࣹينْ ݖࣷفࣹيلْ گُ مَتْ سࣴڪّ، ندَخْ ݖࣷفࣹيلْ، بࣵڪَّارْ يُ بَرࣹ لَيْ فَرْ؞ ‏9 گَنَلَنْتࣹلࣹينْ تࣹ بَݧَ جࣵمْبَتْ؞ ‏10 ڪࣹنّ ڪُ ݖِ نࣹڪّ، مَيْ گِ ڪࣷ اَوْ يِوْ جَگْلࣹيلْ، نَ ڪࣷ تَخَوࣷو ݧࣺݒّ؞ نࣷونُ نگࣹينْ دِ دࣷونࣹ اَيْ سَيْتُڪَتْ يُ بَاخْ يُ يِوُ يࣵلَّ وُ يَا؞ ‏11 ڪُيْ وَخْ، نَ وَخࣹ نُ مُ جࣷتّلࣹي ڪࣵدُّگْ يࣵلَّ؞ ڪُيْ تَخَوࣷو، نَ تَخَوࣷو ݖِ دࣷولࣹيْ يࣵلَّ، نگِرْ مࣵگُّگْ يࣵلَّ فࣹيݧࣹ ݖِ لࣺݒّ، ندَخْ عِيسَىٰ اَلْمَسِ، مِ يࣹلّࣷو دَرَجَ اَڪْ ڪࣵتَّنْ، تࣹيْ اَڪْ اࣴلّࣴگْ بَ فࣵوّ؞ اَمِينْ؞
بُ نگࣹينْ دِ سࣷنّ، نَ اَلْمَسِ تَخْ
‏12 سࣷݒّࣹ يِ، تࣵنگࣷورُ نَتُّ بِ نگࣹينْ نࣹڪّࣹ نِي، بُمُ لࣹينْ جَاخَلْ بَ مࣹلْ نِ لُ جَادُوُلْ مࣷو لࣹينْ دِڪَّلْ؞ ‏13 وَايࣹ نࣷونُ نگࣹينْ سࣺدّࣷو ݖِ ݖࣷونࣷيْ اَلْمَسِ، بࣺگࣹلࣹينْ ڪࣷ، نگِرْ بُ ندَمَمْ فࣹيݧࣹي، نگࣹينْ گࣴنَتࣹي بࣺگْ؞ ‏14 سُ ݧُ لࣹينْ جࣵنّࣹي سَخْ ندَخْ تُرُ اَلْمَسِ، ندࣷڪّلࣹي يࣹينْ، ندَخْ نࣷو گُ دَرَجَوُ گِ، نࣷووُگْ يࣵلَّ مࣷو وࣵݖّ ݖِ يࣹينْ؞ ‏15 وَايࣹ رࣹيْ نِتْ، مبَا سࣵݖّ، مبَا دࣹفْ لُ بࣷنْ، مبَا دُگّ ݖِ يࣴفِ جَامْبُرْ، بُمُ وَرَلْ ݖࣷونࣷيْ ڪࣹنّ ݖِ يࣹينْ؞ ‏16 ڪُ سࣷنّ ندَخْ لِ مُيْ گࣴمْڪَتُبْ اَلْمَسِ نَگْ، بُمُ ݖِ رُسْ، نَ سࣵبَّالْ يࣵلَّ ڪَيْ، ݖِ وࣷووُ تُرْ؞ ‏17 ندَخْ ڪَتْ تࣹيْ لَ اࣵتّࣹ بِ وَرَ تࣵمْبَلࣹي ݖِ وَا ڪࣴرْ يࣵلَّ؞ ندࣹگَمْ ݖِ نُنْ لَيْ تࣵمْبَلࣹي نَگْ، ݧِ گࣴمُلْ خِبَارُ جࣵمّ بُ يࣵلَّ، اَنَ فُ ݧُ جࣴمْ؟ ‏18 تࣹ اِتْ:‏
«اَجِ جُبْ جِ، سُ ڪࣷ مُݖّ جَفࣹي،
يࣺيفَرْ بِ، بࣵڪَّارْڪَتْ بِ مࣷومْ، فُ مُ جࣴمْ¤؟»
‏19 ڪࣷنْ نَگْ ݧِيْ سࣷنّ تࣹ ݖࣷوبَرࣹيْ يࣵلَّ تَخْ، نَݧُ دࣺنڪّ سࣹينْ بࣷݒّ يࣵلَّ، سࣵڪّ-ڪَتْ بُ وٗورْ بِ، تࣹ ݧُيْ سَخࣷو دࣹفْ لُ بَاخْ؞
