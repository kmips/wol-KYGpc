خࣺيوْلُ، فࣵتّࣹ آيْ نَ
8
‏1 مبࣷولࣹمْ سَنْتَانࣹ بِ مَ لࣹينْ دࣺنڪّ بࣺسُبْ تࣹيْ نَگْ، سࣵمّࣷولࣹينْ ڪࣷو جࣴفࣹ، ندَخْ نگࣹينْ دُندّ، يࣷڪُّ، بَ دُگّ نَنگُجِ رࣺيوْ مَ اَجِ سَخْ جِ گِݧَلࣷونْ سࣹينِ مَامْ؞ ‏2 فࣵتّلِڪُلࣹينْ مبࣷولࣹمْ يࣷونْ وِ لࣹينْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ اَوَلࣹ دِيرُ ݧࣹينتّ فُڪِّ اَتْ يِي ݖِ مࣵنْدِݝْ مِ، تࣹگْ لࣹينْ ݖࣷونࣷ، نَتُّ لࣹينْ نگِرْ خَمْ لِ ݖِ سࣹينْ بِيرْ خࣷلْ، خَمْ ندَخْ دِنگࣹينْ وࣷرْمَالْ اَيْ سَنْتَانࣹيمْ، اَمْ دࣺيتْ؞ ‏3 مࣷو لࣹينْ تࣹگْ ݖࣷونࣷ، خِيفَلْ لࣹينْ، لࣹيلْ لࣹينْ مࣵنّ مُ نگࣹينْ خَمُلْ، تࣹ سࣹينِ مَامْ خَمُݧُ ڪࣷ وࣷونْ، نگِرْ خَمَلْ لࣹينْ نࣹ دُ اَوْ ݧَمْ دࣷݝّ لَ دࣷومْ آدَمَ دِ دُنْدࣹ، وَايࣹ گࣺݒّ ڪࣵدُّ گُ تُڪّࣹي ݖِ گࣺمِّݧُ اَجِ سَخْ جِ، دࣹيسْ نَ ڪࣷ دُنْدࣹ؞ ‏4 سࣹينِ يࣺرࣹ رࣵݒُّلْ، سࣹينِ تࣵنڪّ نࣹوِوُلْ دِيرُ ݧࣹينتّ فُڪِّ اَتْ يِي؞ ‏5 خَمْلࣹينْ خࣺلّ نࣹ نِ بَايْ دِ يَرࣹ دࣷومَمْ، نِ لَ لࣹينْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ يَرࣹ؞ ‏6 ڪࣷنْ سࣵمّلࣹينْ سࣹينْ سَنْتَانࣹيْ يࣵلَّ اَجِ سَخْ جِ، دِ تࣷݒّ اَيْ يࣷونَمْ تࣹ رَگَلْ ڪࣷ؞ ‏7 ندَخْ سࣹينْ يࣵلَّ اَجِ سَخْ جَا نگِ لࣹينْ دِ يٗبُّ رࣺيوْ مُ بَاخْ؛ رࣺيوْ مُ اَمِ وَلْ، اَڪْ بࣴتِ ندࣷخْ، اَڪْ ندࣷخِ خٗوتࣹ يُ وَلࣹيْ خُرْ اَڪِ تُندّ، ‏8 رࣺيوُمْ بࣹلࣹ اَڪْ لࣷرسّ اَڪْ رࣹسࣹݧْ اَڪْ فِگّ اَڪْ گࣴرࣴنَاتْ، اَڪْ دِوُ اࣷلِوْ، اَڪْ لࣹمْ؞ ‏9 رࣺيوْ مَ دُنگࣹينْ فَ نࣺيوࣹ نجࣴلْ، تࣹ دُنگࣹينْ فَ ݧࣵڪّࣹ دَرَ؞ رࣺيوْ مَ، اَيْ دࣷجَمْ، وࣹݧْ گُ ݧُولْ لَ، اَيْ تُنْدَمْ، خࣵنجَرْ نگࣹينْ ݖَيْ گَسࣹ؞ ‏10 دِنگࣹينْ لࣹڪّ بَ رࣹگّ، دِ سَنْتࣹ سࣹينْ يࣵلَّ اَجِ سَخْ جِ، رࣺيوْ مُ بَاخْ مَ مُ لࣹينْ جࣷخْ؞
‏11 وَتُّلࣹينْ سࣹينْ بࣷݒّ نَگْ، بَلَ نگࣹينَ فࣵتّࣹ سࣹينْ يࣵلَّ اَجِ سَخْ جِ، بَݧَ سࣵمّ اَيْ سَنْتَانࣹيمْ، اَڪْ اࣵتّࣹيْ يࣷونَمْ، اَڪْ دࣷگَلِ يࣷونَمْ يِ مَ لࣹينْ دِ دࣺنڪّ تࣹيْ؞ ‏12 سُ نگࣹينْ لࣹڪّࣹي بَ رࣹگّ، اَمْ ڪࣴرْ يُ بَاخْ يُ نگࣹينْ تَبَخْ دࣴڪّࣹ، ‏13 سࣹينْ جُرْ گُ گُدّ اَڪْ گُ گࣵتّ يࣷڪُّ، خَالِسْ اَڪْ وُرُسْ يࣷڪُّلْ لࣹينْ، سࣹينْ جࣺݒّ اَلَلْ يࣷڪُّ، ‏14 سُ بࣷوبَا مࣷيْتُلࣹينْ دَلْدِ خࣷگّلِڪُ، فࣵتّࣹ سࣹينْ يࣵلَّ اَجِ سَخْ، جِ لࣹينْ گࣺنّࣹي رࣺيوُمْ مِسْرَ، ڪࣴرُگْ نجَامْ گَ، ‏15 ڪِ لࣹينْ وࣷمَّتࣹ بِيرْ مࣵنْدِݝْ مُ يَا مُ رَگْلُ مَ، اَڪْ جَانِ دَݝَرْ يَاڪِ جِيتَمْ، ݖِ بِيرْ مَرَلْ اَڪْ ݧࣵڪُّمْ ندࣷخْ؞ مࣷو لࣹينْ خࣹلِّلْ ندࣷخْ مُ بَلّࣹ دࣷجْ وُ نࣹ سࣹرࣹݝْ؞ ‏16 مࣷو لࣹينْ لࣹيلࣹ مࣵنُّ مࣵنْدِݝْ، مَ سࣹينِ مَامْ خَمُلࣷونْ، نگِرْ تࣹگْ لࣹينْ ݖࣷونࣷ، نَتُّ لࣹينْ، تࣹ مُجّࣹ بَاخࣹ لࣹينْ؞ ‏17 دِنگࣹينْ وَخْ ݖِ سࣹينْ خࣹلْ نࣹ:‏ «سُنُ دࣷولࣹ اَڪْ سُنُ نجࣵمْبَارُ بࣷݒّ مࣷو نُ مَيْ جِي اَلَلْ؞» ‏18 وَايࣹ فࣵتّلِڪُلࣹينْ نࣹ سࣹينْ يࣵلَّ اَجِ سَخْ جِ مࣷويْ ڪِ لࣹينْ مَيْ دࣷولࣹيْ لِگّࣺييْ اَلَلْ جِ، نگِرْ فࣹدّلِ نِ مُ ڪࣷ دࣹفࣹ تࣹيْ، ڪٗلّࣴرࣹيمْ گَ مُ گِݧَلࣷونْ سࣹينِ مَامْ؞
‏19 بُ نگࣹينْ دࣹمࣹي بَ فࣵتّࣹ لُ جࣴمْ ݖِ سࣹينْ يࣵلَّ اَجِ سَخْ جِ، دِ تࣷݒّ يࣹنࣹينْ يࣵلَّ، دِ لࣹينْ جَامࣷوڪَ سُجّٗوتَلْ، سࣹيدࣹيلْ نَا لࣹينْ بࣺسْ نِڪِ تࣹيْ، نࣹ سُ بࣷوبَا دࣺيْ، دِنگࣹينْ سࣵنڪُ ݖࣵنڪُتࣹ گُ مَتْ سࣴڪّ؞ ‏20 نِ خࣹيتْ يِ اَجِ سَخْ جِ نَرَ سࣵنڪّ فِ سࣹينْ ڪَنَمْ، نࣷونُ نگࣹينْ دِ سࣵنڪࣷو ندَخْ ݧࣵڪَّ دࣺگَّلْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ؞
