مُوسَىٰ تࣵگّࣷو نَ
(سَارْ 31—34)
31
‏1 گَنَّاوْ لࣷولُ مُوسَىٰ دࣹلُّ نَ اࣵدُّ ڪࣵدُّ يِي، وَخْ اَڪْ بࣵنِّ إِسْرَاءِيلْ گࣺݒّ؞ ‏2 مُ نࣹ لࣹينْ:‏ «مَا نگِ ݖِ سَمَ تࣺيمࣺيرِ اَتْ اَڪْ ݧَارْ فُڪّ، تࣹيْ؞ مَنَتُمَا جِيتࣹ، تࣹ اِتْ اَجِ سَخْ جِ مࣷو مَ وَخْ نࣹ مَ:‏ ‹دࣷو جࣵلّ دࣹخُ يُرْدَنْ گِي؞› ‏3 سࣹينْ يࣵلَّ اَجِ سَخْ جِ ݖِ بࣷݒَّمْ مࣷو لࣹينْ دِ جِيتࣹ جࣵلّ؞ مࣷويْ فࣵڪَّسْ خࣹيتْ يࣷويُ فِ سࣹينْ ڪَنَمْ، نگࣹينْ نَنگُ سࣹينُمْ رࣺيوْ؞ يࣷسُوࣹ مࣷو لࣹينْ دِ جِيتࣹ جࣵلّ، نَ ڪࣷ اَجِ سَخْ جِ وَخࣹ وࣷونْ؞ ‏4 اَجِ سَخْ جِ مࣷو لࣹينْ دِ دࣹفْ نَ مُ دࣹفࣷونْ سِوࣷنْ اَڪْ اࣷگْ، بُورِ اَمࣷرࣹينْ ݧَ، اَڪْ سࣹينُمْ رࣺيوْ، فࣵڪَّسْ لࣹينْ؞ ‏5 اَجِ سَخْ جࣹي لࣹينْ دِ جࣷخࣹ فِ سࣹينْ ڪَنَمْ، نگࣹينْ دࣹفْ لࣹينْ مبࣷولࣹمْ لِ مَ لࣹينْ سَنْتࣷونْ؞ ‏6 دࣴگࣴرْلُلࣹينْ نَگْ تࣹ اَمْ فِتْ؞ بُلࣹينْ تِيتْ، بُلࣹينْ يࣷخْ-يࣷخِ ندَخْ ݧࣷومْ، نگِرْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ مࣷويْ اࣵندّ اَڪْ يࣹينْ؞ دُ لࣹينْ بَ، دُ لࣹينْ وَݖّ مُڪّ؞»
‏7 بَ لࣷولُ اَمࣹي مُوسَىٰ وࣷولُ يࣷسُوࣹ، وَخْ اَڪْ مࣷومْ، إِسْرَاءِيلْ گࣺݒّ تࣹگْ ݖِ سࣹينْ بࣴتْ؞ مُ نࣹ ڪࣷ:‏ «دࣴگࣴرْلُلْ تࣹ اَمْ فِتْ، ندَخْ يَوْ يَايْ اࣵندّ اَڪْ مبࣷولࣷو مِي ݖَ رࣺيوْ مَ اَجِ سَخْ جِ گِݧَلࣷونْ سࣹينِ مَامْ، نࣹ ݧࣷومْ لَ ڪࣷيْ جࣷخْ؞ يَوْ اِتْ يَا لࣹينْ ڪࣷيْ سࣺدّࣹ؞ ‏8 اَجِ سَخْ جِ نَگْ مࣷويْ دࣷخْ، جِيتُ لَ؞ مࣷومْ مࣷويْ نࣹڪّ اَڪْ يَوْ؞ دُ لَ بَ، دُ لَ وَݖّ مُڪّ؞ بُلْ تِيتْ، بُلْ يࣷقِ؞»
‏9 مُوسَىٰ مࣷو بِندّ ڪࣵدُّگْ يࣷونْ وِي، دࣺنڪّ ڪࣷ دࣷومِ لࣹوِ يُ گٗورْ، سَرْخَلْڪَتْ يَيْ گࣵدُّ گَالُ ڪٗلّࣴرࣹيْ اَجِ سَخْ جِ، ݧࣷوڪْ مَگِ إِسْرَاءِيلْ ݧࣺݒّ؞ ‏10 مُوسَىٰ سَنتّ لࣹينْ، نࣹ لࣹينْ:‏ «نجࣹيخِتَلْ جُرٗومْ ݧَارࣹيلُ اَتْ مُ نࣹڪّ، اࣵݒّࣷوڪْ اَتُمْ نجࣺگَّلُگْ بࣷرْ، ݖِ دِيرُ مࣵگَّلُ مبَارْ يِ، ‏11 بُ إِسْرَاءِيلْ گࣺݒّ تࣹيوْسࣹي فَ سࣹينْ ڪَنَمْ يࣵلَّ اَجِ سَخْ جِ، فَ بࣺرَبْ بَ مُيْ تَامُجِ، نَنگࣹينْ جࣵنگّ ڪࣵدُّيْ يࣷونْ وِي فِ ڪَنَمْ إِسْرَاءِيلْ گࣺݒّ، ݧُيْ دࣺگّ؞ ‏12 دَنگࣹينْ دِ دَجَلࣹ مبࣷولࣷو مِ، گٗورْ ݧِ اَڪْ جِگࣺينْ ݧِ اَڪْ تُوتْ-تࣵنڪّ يِ، بَ ݖِ دࣷخَنْدࣺيمْ بُ نࣹڪّ ݖِ سࣹينْ بِيرْ دࣴڪّ، ندَخْ ݧُ دࣺگْلُ، بَ تࣵگَّتُ ݖِ رَگَلْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ، دِ سࣵمّࣷو جࣴفࣹ مبࣷولࣹمْ ڪࣵدُّيْ يࣷونْ وِي؞ ‏13 سُ ڪࣷ دࣹفࣹي سࣹينْ دࣷومْ يِ خَمُلْ دِنَݧُ ݖِ دࣺگّ، بَ تࣵگَّتُ ݖِ رَگَلْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ، مبࣷولࣹمْ فَنْ يَ نگࣹينْ دِ دُندّ ݖَ رࣺيوْ، مَ نگࣹينْ جࣵلّ دࣹخُ يُرْدَنْ، جࣴمْ فَ، نگِرْ نَنگُجِ ڪࣷ؞»
‏14 بَ لࣷولُ وࣺيࣹي اَجِ سَخْ جِ نࣹ مُوسَىٰ:‏ «مَ نࣹ، سَ بࣺسُبْ دࣹي نگِ جࣹگࣹسِ؞ وࣷولْ يࣷسُوࣹ، تࣹ نگࣹينْ تࣹيوْسِ فِ خَيْمَبْ ندَجࣹ مِ، مَ يࣹبْ ڪࣷ؞» مُوسَىٰ دࣹمْ، اࣵندّ اَڪْ يࣷسُوࣹ، ݧُ تࣹيوِ فَ خَيْمَبْ ندَجࣹ مَ؞ ‏15 اَجِ سَخْ جِ نَگْ فࣹيݧْ ݖَ خَيْمَ بَ، ݖِ بِيرْ تَخَارُ نِيرْ، تَخَارُ نِيرْ وَ تَخَوْ فَ بُنتّ خَيْمَ بَ؞ ‏16 اَجِ سَخْ جِ نࣹ مُوسَىٰ:‏ «يَا نگِ نِي دِ تࣴدِّ فَ سَيْ مَامْ؞ مبࣷولࣷو مِي نَگْ دِنَݧُ تࣵمْبَلِ دِ گࣵنݖّتُيْ تࣷݒّ يࣹنࣹينْ يࣵلَّيْ دࣷخَنْدࣺيمْ يُ وَا رࣺيوْ، مَ ݧُ جࣴمْ ݖَ سࣹينْ بِيرْ؞ دِنَݧُ مَ وَݖّ، فࣹݖِّ سَمَ ڪٗلّࣴرࣹ گِ مَ فَسࣷونْ اَڪْ ݧࣷومْ، ‏17 تࣹ سَمَمْ سࣵنجّ دِنَ تࣵڪّ ݖِ سࣹينْ ڪَوْ بࣺسُبْ ڪࣹرࣷوگْ؞ مَا لࣹينْ دِ وَݖّ، لࣵقْ لࣹينْ سَمَ ڪَنَمْ، تࣹ ڪࣷنْ دࣹيسْ نَ لࣹينْ لࣹڪّ؞ مُسِبَ يُ بَرࣹ اَڪِ نجࣵقَرࣹ دِنَ لࣹينْ دِڪَّلْ؞ بࣺسُبْ ڪࣹرࣷوگْ ݧࣷو نَانْ:‏ ‹خَنَا سُنُ يࣵلاَّ نࣹڪُّلْ ݖِ سُنُ بِيرْ ڪَيْ، بَ تَخْ مُسِبَ يِي دِڪَّلْ نُ!‏› ‏18 مَنْ نَگْ مَا لࣹينْ دِ لࣵقَ لࣵقْ سَمَ ڪَنَمْ، بࣺسُبْ ڪࣹرࣷوگْ ندَخْ مبࣷولࣹمْ مبࣷنْ گِ ݧُ دࣹفْ، ندَخْ ݧࣷو وَلْبَتِڪُ ݖِ يࣹنࣹينْ يࣵلَّ؞ ‏19 لࣺيگِ نَگْ بِندّلࣹينْ وࣷيْ وِي مَ لࣹينْ دِ دࣺگْتَلْ، تࣹ يَوْ مُوسَىٰ نگَ جࣵنگَلْ ڪࣷ بࣵنِّ إِسْرَاءِيلْ؞ وࣷيْلࣷو لࣹينْ ڪࣷ، ندَخْ وࣷيْ وِي دِ سَمَ سࣹيدࣹ اَڪْ بࣵنِّ إِسْرَاءِيلْ؞ ‏20 مَا لࣹينْ دِ يٗبُّ دࣺيْ ݖَ رࣺيوْ مَ مَ گِݧَلࣷونْ سࣹينِ مَامْ، فَ مࣹيوْ مَاڪْ لࣹمْ جَ تُورࣷو، ݧُ لࣹڪّ بَ رࣹگّ، يَفْلُ، دَلْدِ وَلْبَتِڪُ ݖِ يࣹنࣹينْ يࣵلَّ، دِ لࣹينْ جَامُ، خَرَبْ مَ، فࣹݖِّ سَمَ ڪٗلّࣴرࣹيڪْ ݧࣷومْ؞ ‏21 سُ ڪࣷ دࣹفࣹي بُ لࣹينْ مُسِبَ يُ بَرࣹ دِڪَّلࣹي اَڪِ نجࣵقَرࣹ، وࣷيْ وِي مࣷويْ بِرَلْ ڪࣵدُّگْ سࣹيدࣹ فِ سࣹينْ ڪَنَمْ، ندَخْ سُ بࣷوبَا خࣹلْ دُ ڪࣷ فَايࣹ، بَ مُ رࣺيرْ سࣹينُ اَسْڪَنْ؞ ندَخْ ڪَتْ خَمْ نَا سࣹينْ مࣺبࣺتْ مِ ݧُيْ رَبَتْ، تࣹ يٗبّࣷوگُمَ لࣹينْ سَخْ ݖَ رࣺيوْ مَ مَ لࣹينْ گِݧَلࣷونْ؞» ‏22 بَ لࣷولُ اَمࣹي مُوسَىٰ بِندّ وࣷيْ وࣷووُ بࣺسُبْ ڪࣹرࣷوگْ، دَلْدِ ڪࣷيْ جࣵنگَلْ بࣵنِّ إِسْرَاءِيلْ؞
‏23 اَجِ سَخْ جِ نَگْ يࣹبْ يࣷسُوࣹ دࣷومُ نُونْ نࣹ ڪࣷ:‏ «دࣴگࣴرْلُلْ تࣹ اَمْ فِتْ، ندَخْ يَوْ يَايْ يٗبُّ بࣵنِّ إِسْرَاءِيلْ ݖَ رࣺيوْ مَ مَ گِݧࣷونْ، تࣹ مَنْ ݖِ سَمَ بࣷݒّ مَايْ نࣹڪّ اَڪْ يَوْ؞»
‏24 بَ مُوسَىٰ بِنْدࣹي ڪࣵدُّيْ يࣷونْ وࣷووُ ݖِبْ تࣺيرࣹ بَ مُ مَتْ، ‏25 دَفَ يࣹبْ لࣹوࣹينْ ݧِيْ گࣵدُّ گَالْ گِ اࣵلُّوَيْ ڪٗلّࣴرࣹيْ اَجِ سَخْ جِ دࣹنݖࣹ، نࣹ لࣹينْ:‏ ‏26 «جࣴلْ-لࣹينْ تࣺيرࣹبْ يࣷونْ وِي، نگࣹينْ تࣹگْ ڪࣷ ݖِ سࣹينْ وࣹتُ گَالُ ڪٗلّࣴرࣹيْ يࣵلَّ اَجِ سَخْ جِ، مُ دࣷونْ فَ اَبْ سࣹيدࣹ، نگِرْ اࣵتّࣹ لࣹينْ؞ ‏27 ندَخْ مَا خَمْ سࣹينْ جِڪّࣷيْ فِݒُّ، اَڪْ سࣹينُگْ دࣴگࣴرْ بࣷݒّ؞ سࣹيتْلࣹينْ، ندࣹگَمْ بِ مَيْ دُنْدَگُمْ تࣹيْ ݖِ سࣹينْ بِيرْ نگࣹينْ دِ فِݒُّ ݖِ ڪَوْ اَجِ سَخْ جِ، اَسْتَمَاڪْ گَنَّاوْ بُ مَ دࣹييࣹي؞ ‏28 وࣷولْ-لࣹينْ مَ سࣹينْ مَگِ گِيرْ يࣺݒّ، اَڪْ سࣹينِ ڪࣵنگَمْ، مَ وَخْ ڪࣵدُّ يِي، ݧُيْ دࣺگّ، بَ مَ سࣹيدࣹلࣷو اَسَمَانْ اَڪْ سُوفْ، ݖِ سࣹينْ ڪَوْ، ݧࣷومْ؞ ‏29 خَمْ نَا نࣹ گَنَّاوْ بُ مَ دࣹييࣹي، دِنگࣹينْ دࣹفَ دࣹفْ اَگْ يࣵقُتࣹ، دِنگࣹينْ تࣹگِّ يࣷونْ وِ مَ لࣹينْ جَارْلࣷو وࣷونْ، تࣹ مُسِبَا لࣹينْ دِ مُجّ دِڪَّلْ، ندَخْ يࣹينَيْ دࣹفْ جࣴفْ جُ بࣷنْ جُ اَجِ سَخْ جِ ݧَاوْلُ، بَ مࣹرْلࣷو ڪࣷ ندَخْ سࣹينِ جࣴفْ؞» ‏30 بَ مُ ڪࣷ دࣹفࣹي مُوسَىٰ جࣵنگّ ڪࣵدُّيْ وࣷيْ وِي بَ مُ مَتْ، ندَجࣹمْ إِسْرَاءِيلْ مَ مࣺݒّ دِ دࣺگّ؞
