ݒَرْ-ݒَرْلࣷو وَرُلْ گࣴمْڪَتْ ݧِ
3
‏1 مَنْ نَگْ بࣷڪّ يِ، مَنُمَ وࣷونَ وَخْ اَڪْ يࣹينْ نِ ݧُيْ وَخࣹ اَڪْ ݧُ اࣵندّ اَڪْ خࣹلُمْ يࣵلَّ، وَايࣹ نِ نِتِ سُوخُ نࣹينْ رࣹڪّ لاَ مَنࣷونَ وَخࣹ اَڪْ يࣹينْ، مبࣹتࣹ ݧُ دِيْ تُوتْ-تࣵنڪّ ݖِ يࣷونُ اَلْمَسِ؞ ‏2 مࣹيوْ لاَ لࣹينْ جࣷتَ نࣵنْدَلْ، وَايࣹ لࣹيلُمَ لࣹينْ دُگُبْ، ندَخْ اࣵتَّنُلࣹينْ ڪࣷ وࣷونْ، تࣹ بَ لࣺيگِ سَخْ اࣵتَّنُلࣹينْ ڪࣷ، ‏3 ندࣹ يࣹينَ نگِ بَ تࣹيْ ݖِ سࣹينُگْ نِتࣹيْ سُوخُ نࣹينْ؞ ندَخْ گَنَّاوْ ڪِݧَانْ اَڪْ ݝَايࣷو نگِ ݖِ سࣹينْ بِيرْ، دُ اَيْ نِتِ سُوخُ نࣹينْ نگࣹينْ بَ تࣹيْ، دِ دࣷخَلࣹ نِ نِتِ اࣵدِّنَيْ ڪࣹسࣹ؟ ‏4 بُ ڪࣹنّ نَانْ:‏ «مَنْ ݖِ ݒٗولْ لاَ بࣷڪّ،» ڪࣹنࣹينْ نَا:‏ «مَنْ نَگْ اَݒࣷلࣷسْ،» ندَخْ ڪࣷنْ دُنگࣹينْ نِتِ اࣵدِّنَيْ ڪࣹسࣹ؟
‏5 اَنَ ڪُيْ اَݒࣷلࣷسْ سَخْ؟ اَڪْ ڪُيْ ݒٗولْ؟ خَنَا اَيْ سُرْگَ دࣷݝّ يُ نگࣹينْ جَارࣹ، بَ گࣴمْ، نِ بࣷرࣷومْ بِ جَگْلࣹيلࣹ ڪࣹنّ ڪُ نࣹڪّ ݖِ نُنْ، اَوْ سَسَمْ؞ ‏6 مَا جࣴمْبَتْ، اَݒࣷلࣷسْ سُوخَتْ، وَايࣹ يࣵلَّ مࣷو جࣹبِّلْ؞ ‏7 ڪِ جࣴمْبَتْ نَگْ، اَڪْ ڪِ سُوخَتْ، ڪࣹنّ دُ ݖِ دَرَ، خَنَا يࣵلَّ مِيْ جࣹبِّلْ دࣷݝّ؞ ‏8 ڪِ جࣴمْبَتْ، اَڪْ ڪِ سُوخَتْ ݧࣷو يࣹمْ ڪࣹݒّ، وَايࣹ ڪُ ݖِ نࣹڪّ ڪࣹمُ لِگّࣺييُ بࣷݒَّمْ لَيْ يࣷولࣷو؞ ‏9 نࣷو بࣷڪّ دِيْ سُرْگَ ݖِ لِگّࣺييُ يࣵلَّ بِ؛ يࣹينْ نگࣹينْ دِ تࣷولُ يࣵلَّ، دِ تَبَخُبْ يࣵلَّ؞
‏10 يِوُ يࣵلَّ وِ مُ مَ مَيْ نَگْ، ݖِ لاَ لَلࣹ ڪࣹنُ گِ، نِ تَبَخْڪَتْ بُ خَرࣹݧْ، ڪࣹنࣹينْ دِ تَبَخْ ݖَ ڪَوْ؞ وَايࣹ نَ ڪُ نࣹڪّ خࣷولْ بُ بَاخْ نَ مُ ݖَيْ تَبَخࣹ؞ ‏11 گࣹنࣹينْ ڪࣹنُ مࣷومْ مَنࣹيسُ ڪࣷو لَلْ، خَنَا گِ خَسَ لَلࣹ، تࣹ مࣷويْ عِيسَىٰ اَلْمَسِ؞ ‏12 ڪُ تَبَخْ ݖِ ڪَوْ ڪࣹنُ گِ نَگْ، نگَ تَبَخࣹ وُرُسْ، مبَا خَالِسْ، مبَا ݒࣹرْ، مبَا دࣺنڪّ، مبَا گَتَّخْ، مبَا بࣷوبْ، ‏13 ڪُ ݖِ نࣹڪّ، سَبْ لِگّࣺييْ دِنَ فࣹيݧْ، ندَخْ بࣺسْ بَا ڪࣷيْ خَمْلࣹ، ندࣹ سَوَرَ لࣹيسْ ڪࣷيْ فࣹيݧَلࣹ؞ ڪُ ݖِ نࣹڪّ، اَڪْ لُ سَ خࣹيتُ لِگّࣺييْ مَنَ دࣷونْ، سَوَرَ لࣹيسْ ڪࣷيْ نَتّࣹ؞ ‏14 ڪُ لِگّࣺييࣷونْ بَ لَ نگَ تَبَخࣷونْ ݖَ ڪَوْ ڪࣹنُ گَ دࣹسْ، دِنگَ يࣷولُ؞ ‏15 وَايࣹ ڪُ سَبْ لِگّࣺييْ لَڪّ، ݧࣵڪّ نگَ؞ يَوْ ݖِ سَ بࣷݒّ نَگْ دِنگَ مُݖّ، مࣹلْ نِ ڪُ رࣴݖّࣹ ݖِگْ لَڪّ رࣹڪّ؞
‏16 خَنَا خَمُلࣹينْ نࣹ ڪࣴرْ يࣵلَّ نگࣹينْ، تࣹ نࣷووُگْ يࣵلَّ مࣷو دࣴڪّ ݖِ يࣹينْ؟ ‏17 ڪُ يࣵقْ ڪࣴرْ يࣵلَّ نَگْ، ڪࣷوڪُ يࣵلَّ دِنَ ڪࣷ يࣵقْ، ندَخْ ڪࣴرْ يࣵلَّ دَفَ سࣹلّ، تࣹ مࣷويْ لِ نگࣹينْ دࣷونْ؞
‏18 بُ ڪࣹنّ نَخْ بࣷݒَّمْ:‏ ڪُ تࣹگْ بࣷݒَّمْ ݖِ يࣹينْ، بࣷرࣷومْ خَمْ-خَمْ، نِ خَرْنُ بِي جࣵݒّࣹ خَمْ-خَمْ، ڪࣷوڪُ نَ دࣷونْ اَبْ دࣷفْ، نگِرْ اَمْ خَمْ-خَمْ بِ وٗورْ؞ ‏19 ندَخْتࣹ خَمْ-خَمُ اࣵدِّنَ سِي، ندࣷفْ لَ فَ يࣵلَّ؞ مࣷو تَخْ بِنْدࣹيسْ نَ نࣹ:‏ «مࣷويْ ڪِيْ جࣵݒّ ݧِ مُوسْ ݖِ سࣹينْ ݒࣹخࣹيْ مُوسَاتُ¤،» ‏20 تࣹگَتِ ݖَ نࣹ:‏ «بࣷرࣷومْ بِ خَمْ نَ خَلاَتِ بࣷرࣷومْ خࣹلْ يِ تࣹ خَمْ نَ نࣹ نࣹينْ لَ¤؞»
‏21 ڪࣷنْ نَگْ بُ ڪࣹنّ ݒُوڪَرࣹوࣷو اَيْ نِتْ، ندَخْ لࣺݒّ، يࣹينَ ڪࣷ مࣷومْ، ‏22 مُيْ ݒٗولْ، دِ اَݒࣷلࣷسْ، دِ سࣹفَسْ، دِ اࣵدِّنَ، دِ دُندّ، دِ ندࣹي، دِ تࣹيْ، دِ اࣴلّࣴگْ، لࣺݒّ، يࣹينَ ڪࣷ مࣷومْ؞ ‏23 يࣹينْ نَگْ، اَلْمَسِ مࣷو لࣹينْ مࣷومْ، تࣹ اَلْمَسِ، يࣵلاَّ ڪࣷ مࣷومْ؞
