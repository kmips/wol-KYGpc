ڪُ تُوبُلْ، سࣵنڪُ
13
‏1 جَمࣷنࣷ جࣷوجُ تࣹمْبࣹ لَ اَوْ نِتْ دِڪّ، اࣵگّࣹ عِيسَىٰ مبِرُمْ وَا گَلِلࣹ يُ ݒِلاَتْ بٗومࣷونْ، جَخَسࣹ سࣹينْ دࣹرࣹتْ اَڪْ سࣹينْ دࣹرࣹتِ جُرُ سَرَخْ يُ ݧُ جَگْلࣹيلْ يࣵلَّ؞ ‏2 عِيسَىٰ نࣹ لࣹينْ:‏ «فࣷوگْ نگࣹينْ نࣹ وَا گَلِلࣹ يࣷويُ دَݧࣷو گࣴنَ بࣵڪَّارْ مبࣷولࣹمْ سࣹينْ بࣷڪِّ گَلِلࣹ يَ ݖَ دࣹسْ، بَ تَخْ ݧُ دَجْ مِتِتْ وُ نِ تࣷلُّ؟ ‏3 مَا لࣹينْ نࣹ دࣺيدࣺيتْ ڪَيْ، تࣹ سُ نگࣹينْ تُوبُلْ، نࣷونُ نگࣹينْ دِ سࣵنڪࣷو يࣹينْ ݧࣺݒّ؞ ‏4 فُڪِّ نِتْ اَڪْ جُرٗومْ ݧࣹتّ ݧࣷوݧُ تَتَ جَ ݖَ سِلࣷوࣹ دَانُ ݖِ سࣹينْ ڪَوْ، بَ رࣹيْ لࣹينْ نَگْ، نگࣹينْ فࣷوگْ نࣹ ݧࣷو گࣴنْ دِيْ تࣷوݧْڪَتْ ݖِ يࣹرُسَلࣹمْ گࣺݒّ؟ ‏5 مَا لࣹينْ نࣹ دࣺيدࣺيتْ ڪَيْ، تࣹ سُ نگࣹينْ تُوبُلْ، يࣹينْ ݧࣺݒّ اَيْ سࣵنڪُ نِ ݧࣷومْ؞»
‏6 مُ لࣺيبْ لࣹينْ لِي نَگْ، نࣹ لࣹينْ:‏ «گَرَبُ فِگّ لَ نِتْ اَمࣷونْ ݖِ تٗوڪࣴرَمْ؞ مُ دِڪّ دِ ڪࣷ رَاسْ-سِ، گِسُ ݖَ دَرَ؞ ‏7 مُ نࣹ سُرْگَ بَ:‏ ‹ݧࣹتِّ اَتَ نگِي مَيْ دِڪّ، رَاسْ-سِ گَرَبْ گِي، گِسُمَ ݖِ دَرَ؞ ڪࣷنْ گࣷرْ ڪࣷ، دَ فِيْ خَتَلْ رࣹڪّ؞› ‏8 سُرْگَ بَ نࣹ ڪࣷ:‏ ‹سَنگّ بِ، بࣵيِّوَاتْ ڪࣷ اَتْ مِي رࣹڪّ، بَ مَ وࣺيخْ ڪࣷ، دࣹفْ ݖِ تࣷسْ؞ ‏9 جࣷمْبُلْ مُ مࣹݧّ دࣺوࣺنْ؛ لُ ڪࣷ مࣷيْ، نگَ گࣷرْ ڪࣷ؞›»
عِيسَىٰ فَجْ نَ نِتْ بࣺسُبْ نࣷفْلاَيْ
‏10 بࣹنّ بࣺسُبْ نࣷفْلاَيْ، عِيسَىٰ دࣷونْ جࣵنگّلࣹ ݖِ اَبْ جࣵنگُ؞ ‏11 ندࣹڪࣹ جࣹنّ جِگࣺينَ نگَ فَ وࣷونْ؛ رَبَ ڪࣷ وࣷݒَّلْ لُ وَرَ مَتْ فُڪِّ اَتْ اَڪْ جُرٗومْ ݧࣹتّ، مُ خُوگࣹ، مَنُلَ فࣹخࣹ بَ سِگِّ؞ ‏12 عِيسَىٰ گِسْ ڪࣷ، وࣷو ڪࣷ، نࣹ ڪࣷ:‏ «جࣷنگَمَ، تࣹقْلِڪࣷو نگَ اَڪْ سَ لاَگࣷ؞» ‏13 مُ تࣹگْ ڪࣷ اَيْ لࣷخࣷومْ، مُ سِگِّ ݖَ سَا سَ، دِ سࣵبَّالْ يࣵلَّ؞ ‏14 نجِيتُ جࣵنگُ بَ نَگْ يࣺڪَّتِ ڪࣵدُّگْ ݧَاوْلُ ندَخْ بࣺسُبْ نࣷفْلاَيْ بَ عِيسَىٰ فَجࣹ؞ مُ نࣹ مبࣷولࣷو مَ:‏ «جُرٗومْ بࣹنِّ فَنْ لَ نِتْ ݧِ وَرَ لِگّࣺييْ؞ ڪࣷنْ ݖِ فَنْ يࣷويُ نگࣹينْ وَرَ فَجُسِ، وَايࣹ دُ ݖِ بࣺسُبْ نࣷفْلاَيْ؞» ‏15 سَنگّ بِ نࣹ ڪࣷ:‏ «نَافࣺقْ يِ نگࣹينْ دࣷونْ!‏ خَنَا دُ ݖِ بࣺسُبْ نࣷفْلاَيْ لَ ڪࣹنّ ڪُ نࣹڪّ ݖِ يࣹينْ دِ يِوࣹي اَوْ يࣴڪَّمْ مبَا مبَامَمْ ݖَ گࣺتّ گَ، نگِرْ وࣴگِّ ڪࣷ؟ ‏16 جِگࣺينْ جِي دِ سࣴتُبْ إِبْرَاهِيمَ، تࣹ فُڪِّ اَتْ اَڪْ جُرٗومْ ݧࣹتَّ نگِي سࣹيْتَانࣹ لࣵگَّلْ ڪࣷ، وَرࣹيسُ ڪࣷو يِوِ، تࣹگِّلْ ڪࣷ لاَگࣷ جِي ݖِ بࣺسُبْ نࣷفْلاَيْ؟»
‏17 بَ مُ وَخࣹي لࣷولُ، نࣷونَمْ يࣺݒَّ رُسْ، مبࣷولࣷو مَ نَگْ دِ بࣺگࣹ مبࣷولࣹمْ جࣴفْ جُ يࣺيمࣹ، جَ مُيْ دࣹفْ؞
دࣷومُ فُدّࣴنْ مِسَالْ نَ نگُورُگْ يࣵلَّ
‏18 عِيسَىٰ نࣹيتِ:‏ «اَنَ لُ نگُورُگْ يࣵلَّ دِ نِرࣷولْ؟ اَنَ لُ مَ ڪࣷيْ نِرُلࣹيلْ؟ ‏19 مُ نگِ مࣹلْ نِ دࣷومُ فُدّࣴنْ بُ نِتْ جࣴلْ، سࣵنِّ ݖِبْ تࣷولَمْ؞ مُ سَخْ، مࣵگّ بَ دࣷونْ گَرَبْ گُ نجَنَاوْ يِ تࣵگّ ݖِيْ ݖَرَمْ؞»
لَوِيرْ مِسَالْ نَ نگُورُگْ يࣵلَّ
‏20 مُ دࣹلُّ نࣹ:‏ «اَنَ لُ مَيْ نِرُلࣹيلْ نگُورُگْ يࣵلَّ؟ ‏21 مُ نگِ مࣹلْ نِ لَوِيرْ جُ جِگࣺينْ سࣵڪّ، دࣹفْ ڪࣷ ݖِ ݧࣹتِّ نَتِّ سُنگُفْ، بَ تࣷويَلْ بࣺݒّ مُجّ فُنڪِ؞»
بُنْتُ تࣹخࣹ خَتْ نَ
‏22 عِيسَىٰ مࣷو دࣷونْ جَارࣹ اَيْ دࣴڪّ يُ مَگْ اَڪْ يُ ندَوْ، جࣴمْ يࣹرُسَلࣹمْ، دِ جࣵنگّلࣹ ݖَ يࣷونْ وَ؞ ‏23 مُ اَمْ ڪُ ڪࣷ نࣹ:‏ «سَنگّ بِ، خَنَا ݧِيْ مُݖّ ݧُ نࣺيوْ لَيْ دࣷونْ؟» مُ نࣹ لࣹينْ:‏ ‏24 «گٗورْ-گٗورْلُلࣹينْ بَ دُگّࣹ ݖِ بُنتّ بُ خَتْ بِ، ندَخْ مَا لࣹينْ ڪࣷ وَخْ، ݧُ بَرࣹييْ جࣺيمَ دُگّ، وَايࣹ دُݧُ ڪࣷ مَنْ؞ ‏25 بُ بࣷرࣷومْ ڪࣴرْ گِ خَسࣹي جٗگْ، بَ تࣴجْ بُنتّ بِ، فَ نگࣹينْ دِ دࣷخࣹ دࣹسْ بِتِ، دِ فࣴگّ نَانْ:‏ ‹سَنگّ بِ، اُبِّلْ نُ!‏› مُ نࣹ لࣹينْ:‏ ‹خَوْمَ فُ نگࣹينْ بࣷڪّ؞› ‏26 سُ بࣷوبَا نگࣹينْ نَانْ:‏ ‹نࣷو دَانْ بࣷڪّ دِ لࣹڪّ اَڪَ نَانْ، تࣹ سُنُيْ ݒࣺنݖّ نگَ دَانْ جࣵنگّلࣹي؞› ‏27 مُ نࣹيتِ لࣹينْ:‏ ‹نࣹي نَا لࣹينْ خَوْمَ فُ نگࣹينْ بࣷڪّ؞ سࣷرࣹلࣹينْ مَ يࣹينْ ݧࣺݒّ، دࣹفْڪَتِ نجُبَدِ يِ!‏› ‏28 فࣷوفَ لَ اَيْ جࣷويْ اَڪْ نَقَرْلُ بَيْ يࣺيُ دِ اَمࣹ يࣷونْ، ڪࣹرَ بُ نگࣹينْ گِسࣹي سࣹينِ مَامْ إِبْرَاهِيمَ اَڪْ إِسْحَاقَ اَڪْ يَعْقُوبَ اَڪْ يࣷنࣹنتّ يࣺݒّ ݖَ بِيرْ نگُورُگْ يࣵلَّ، تࣹ فࣹڪّ يࣹينْ ݖِ سࣹينْ بࣷݒّ، ݧُ دࣵقْ لࣹينْ ݖَ بِتِ؞ ‏29 ݒࣹنڪُ اَڪْ سࣷوُ، بࣴجْ-گࣵنَّارْ اَڪْ بࣴجْ-سَالُمْ، لَ اَوْ نِتْ دِ بَوࣷو، دِڪّ تࣷوگْ ݖَ ندَبَلْ بࣹرْندࣹ لَ ݖَ نگُورُگْ يࣵلَّ؞ ‏30 سُ ڪࣷ دࣹفࣹي نَگْ، ݧࣹنَّ نگِ مُجّࣹ تࣹيْ، وَايࣹ اࣴلّࣴگْ ݧࣷويْ جِيتُ؛ تࣹ ݧࣹنَّ نگِ جِيتُ تࣹيْ، وَايࣹ اࣴلّࣴگْ ݧࣷويْ مُجّࣹ؞»
دࣹيسْ نَ تَسْ يࣹرُسَلࣹمْ
‏31 ݖَ جَمࣷنࣷ جࣷوجُ تࣹمْبࣹ لَ اَيْ فَرِسࣹنْ دِڪّ، نࣹ عِيسَىٰ:‏ «جٗگࣹيلْ فِي، تࣹ دࣹمْ، ندَخْ بُورْ اࣹرࣷدّ دَ لاَ نَرَ رࣹيْ؞» ‏32 مُ نࣹ لࣹينْ:‏ «دࣹمْلࣹينْ وَخْ نجࣷمْبࣷورْ مࣷومُ، نࣹ ڪࣷ:‏ ‹مَا نگِي دِ دࣵقْ رَبْ يِ، دِ مَيࣹ اَگْ وࣺرْ، تࣹيْ اَڪْ اࣴلّࣴگْ؞ بࣺسُبْ ݧࣹتّࣹيلْ بَ نَگْ مَ سࣷتَّلْ؞› ‏33 وَايࣹ تࣹيْ اَڪْ اࣴلّࣴگْ اَڪْ سِبِرْ، فࣵوّ مَ دࣹمْ، ندَخْ اَبْ يࣷنࣹنتّ وَرࣹيسُ ڪࣷو بٗومࣹ فࣹنّ فُ دُلْ يࣹرُسَلࣹمْ؞
‏34 «يࣹرُسَلࣹمْ، يࣹرُسَلࣹمْ، يَوْ مِيْ بٗومْ يࣷنࣹنتّ يِ، دِ سࣵنِّ اَيْ دࣷجْ ݧِ ݧُ يࣹبَلْ ݖِ يَوْ، بَ ݧُ دࣹي، ݧَاتَ يࣷونْ لاَ نَمَّ دَجَلࣹ سَيْ دࣷومْ، اُوفْ نِ گَنَارْ دِ اُوفࣹ اَيْ ݖُوجَمْ، تࣹ نَنگُوُلࣷو؟ ‏35 مُ نگࣷوگْ، وَݖّࣹيسْ نَ لاَڪْ سَ ڪࣴرْ نَگْ، وَايࣹ مَا نࣹ لَ دࣷو مَ گِسَتِ بَ ڪࣹرَ جَنتّ بَ نگَ نَا:‏ ‹نَ بَرْڪࣹ وࣵݖَّلْ ڪِ ݧࣴوْ ݖِ تُرُ بࣷرࣷومْ بِ!‏›»
