عِيسَىٰ آيࣹ نَ جِنِگَلْ
12
‏1 مبࣷولࣷو مَ نَگْ مُجّ نَ دَجَلࣷو بَ مَتْ اَيْ جُنِّيْ جُنِّيْ نِتْ، بَيْ جࣷگّࣹينْتࣹ؞ ݖِ بِيرْ لࣷولُ عِيسَىٰ تࣵمْبَلࣹي وَخْ، جࣴڪّࣹ ڪࣷ ݖِ اَيْ تَالِبࣹيمْ، نࣹ:‏ «وَتُّلࣹينْ لَوِيرُ فَرِسࣹنْ ݧِ، تࣹ مࣷويْ جِنِگَلْ؞ ‏2 مبࣷولࣹمْ لُ نࣴبُّ دࣹيسْ نَ ڪࣷ فࣹيݧَلْ، تࣹ لُ لࣵقُ دࣹيسْ نَ ڪࣷ خَمْ؞ ‏3 مࣷو تَخْ مبࣷولࣹمْ لُ نگࣹينْ وَخࣹ بِيرْ لࣴنْدࣴمْ، دࣹيسْ نَ ڪࣷ دࣺگّࣹ بࣴݖّࣴگْ ندَرَڪࣵمّ، تࣹ مبࣷولࣹمْ لُ نگࣹينْ دࣺييࣹ ݖِ نࣺيگُ بِيرْ، دࣹيسْ نَ ڪࣷ بِرَلْ ݖِ ڪَوِ تَاخْ؞
‏4 «مَا لࣹينْ ڪࣷ وَخْ، يࣹينْ سَمَيْ سࣷݒّࣹ، بُلࣹينْ رَگَلْ ݧُ مَنُلْ لُ وࣹيسْ رࣹيْ سࣹينْ يَرَمْ، ندَخْ گَنَّاوْ لࣷولُ دࣷوتُݧُ لࣹينْ مَنَلْ لُ ڪࣷ رَوْ؞ ‏5 مَ خَمَلْ لࣹينْ نَگْ ڪِ نگࣹينْ وَرَ رَگَلْ:‏ ڪِيْ رࣹيْ بَ نࣷݒِّ، اَمْ سَݧْ-سَݧُ سࣵنِّ ݖَ سَوَرَ سَ، ڪࣷوڪُ دࣺيْ لاَ لࣹينْ نࣹ، رَگَلْ-لࣹينْ ڪࣷ؞ ‏6 جُرٗومِ ݒِݖِّ رَمَتُ، خَنَا جَرُلْ ݧَارِ دࣴرࣴمْ دࣷݝّ؟ مࣷونَ يࣵلَّ فࣵتّࣹوُلَ گࣹيسُ مࣹنّ ݖِ ݧࣷومْ؞ ‏7 يࣹينْ نَگْ سࣹينْ ڪَوَرِ بࣷݒّ سَخْ، وَݧّࣹيسْ نَ لࣺݒّ؞ بُلࣹينْ رَگَلْ دَرَ، ندࣹ ندِيرَانُ رَمَتُ، يࣹينَ ڪࣷ گࣴنْ؞
‏8 «تࣹ اِتْ مَا لࣹينْ ڪࣷ وَخْ، ڪࣺݒّ ڪُ دࣴگَّلْ فِ ڪَنَمْ نِتْ ݧِ، نࣹ مَنْ نگَ فࣺيتࣹيلْ، دࣷومُ نِتْ ڪِ اِتَمْ دِنَ دࣴگَّلْ فَ ڪَنَمْ مَلاَڪَيْ يࣵلَّ يَ، نࣹ مࣷومْ نگَ فࣺيتࣹيلْ؞ ‏9 وَايࣹ ڪُ وࣹدِّ فِ ڪَنَمْ نِتْ ݧِ، نࣹ مَنْ نگَ فࣺيتࣹيلْ، دࣹيسْ نَ وࣹدِّ فَ ڪَنَمْ مَلاَڪَيْ يࣵلَّ يَ، نࣹ مَنْ نگَ فࣺيتࣹيلْ؞ ‏10 ڪࣺݒّ ڪُ جࣴمࣹ ڪࣵدُّگْ سَفَانْ ݖِ دࣷومُ نِتْ ڪِ، دࣹيسْ نَ لَ جࣺگَّلْ، وَايࣹ ڪُ سَاگَ نࣷو گُ سࣹلّ گِ، دࣹيسُ لَ جࣺگَّلْ؞
‏11 «بُ ݧُ لࣹينْ يٗبّࣷو ݖَ جࣵنگُ يَ، ݖَ ڪَنَمْ ڪِلِفَ يَ، اَڪْ اࣵتّࣹڪَتْ يَ، بُلࣹينْ جَاخْلࣹ ݖِ نَنْ نگࣹينْ دِ يࣺڪَّتࣹي ڪࣵدُّگْ لَيࣷو، اَڪْ لُيْ دࣷونْ سࣹينُ لَيْ، مبَا لُ نگࣹينْ دِ وَخْ، ‏12 ندَخْ سُ بࣷوبَا نࣷو گُ سࣹلّ گِ مࣷو لࣹينْ دِ خَمَلْ ݖَ وَخْتُ وࣷووَ لَ نگࣹينْ وَرَ وَخْ؞»
يࣷولُ يࣵلَّ دࣷݝّ مࣷويْ اَلَلْ
‏13 ڪࣹنّ ݖَ مبࣷولࣷو مَ مࣷو نࣷونْ عِيسَىٰ:‏ «سࣴرِݧْ بِ، وَخَلْ ݖِ سَمَ مَگْ، مُ جࣷخْ مَ سَمَ وࣵلّ ݖِ ندࣷنࣷ لِ نُ بࣷڪّ؞» ‏14 عِيسَىٰ نࣹ ڪࣷ:‏ «وَا جِي، ڪَنْ مࣷو مَ فَلْ اࣵتّࣹڪَتْ مبَا مِرَاسْڪَتْ ݖِ سࣹينْ ڪَوْ؟» ‏15 ݖِ ڪَوْ لࣷولُ مُ نࣹ لࣹينْ ݧࣷومْ ݧࣺݒّ:‏ «مࣷيْتُلࣹينْ وࣺݒّ خࣹيتُ بࣴگّࣹ، ندَخْ لࣷو بَرࣹلࣹ بَرࣹلࣹ، سَ بَڪَّنْ اَجُوُلْ ݖِ سَ اَلَلْ؞»
‏16 مُ لࣺيبْ لࣹينْ نَگْ، نࣹ لࣹينْ:‏ «نِتْ لَ وࣷونْ ڪُ وࣷومْلࣹ، تࣹ اَبْ تࣷولَمْ نَنگُ؞ ‏17 مُ نَانْ ݖَ خࣹلْ مَ:‏ ‹نُ مَيْ دࣹفْ نَگْ؟ اَمَتُمَ سَخْ فُ مَ دࣹنݖّ سَمَ نگٗوبْ مِ!‏› ‏18 ݖِ ڪَوْ لࣷولُ مُ نࣹ:‏ ‹نِي لاَيْ دࣹفْ:‏ سَمَيْ سࣵقْ لاَيْ دَانࣹيلْ، دَلْدِ دࣹفَرَاتْ سࣵقْ يُ گࣴنَ مَگْ، دࣹنݖّ ݖِ سَمَ ݒࣹݒّ اَڪْ مبࣷولࣹمْ لُ مَ اَمْ؞ ‏19 سُ ڪࣷ دࣹفࣹي ݖِ سَمَمْ خࣹلْ لاَ نَانْ:‏ «ݖࣺيْ سَمَ وَا جِ، يَا نگࣹيڪْ اَلَلْ جُ بَرࣹ جُ مَنَ دࣹمْ اَيْ اَتِ اَتْ؛ نࣹيلْ فِنَاخْ، دِ لࣹڪّ، دِ نَانْ اَڪَ بࣺگࣹ؞» › ‏20 يࣵلَّ نَگْ نࣹ ڪࣷ:‏ ‹ݧࣵڪّ خࣹلْ!‏ گُدِّگْ تࣹيْ جِي، دࣹيسْ نَ جࣴلْ سَ بَڪَّنْ؞ ڪࣷنْ لِ نگَ وَاجَلْ، اَنَ ڪُيْ دࣷونْ بࣷرࣷومْ؟›
‏21 «ڪُيْ دَجَلࣹيلْ بࣷݒَّمْ اَلَلْ تࣹ وࣷومْلࣹوࣷو فَ يࣵلَّ، لࣷولُ مࣷويْ سَوْ دࣹمِنْ؞»
ڪُ گࣴمْ يࣵلَّ، وٗولُ ڪࣷ
‏22 گَنَّاوْ لࣷولُ عِيسَىٰ نࣹ اَيْ تَالِبࣹيمْ:‏ «مࣷو تَخْ مَ نࣹ لࣹينْ، بُلࣹينْ سࣹينْ بَڪَّنْ جَاخَلْ، نگِرْ لُ نگࣹينْ لࣹڪّ، تࣹ اِتْ بُلࣹينْ سࣹينْ يَرَمْ جَاخَلْ نگِرْ لُ نگࣹينْ سࣷلْ، ‏23 ندَخْ لࣹڪّ گِ، بَڪَّنْ بࣹي ڪࣷ اࣴݒّ سࣷلࣷ، تࣹ ڪࣷدَّايْ لِ اِتْ، يَرَمْ وࣹي ڪࣷ اࣴݒّ سࣷلࣷ؞ ‏24 سࣹيتْلࣹينْ ݖِ بَاخࣷݧْ:‏ دُ جِ، دُ گٗوبْ، اَمُلْ اُبْ دࣹنݖّ مبَا اَمْ سࣵقْ؞ تࣹيوُلْ يࣵلَّ دِ ڪࣷ دُنْدَلْ؞ يࣹينْ نَگْ، ݧَاتَ يࣷونْ نگࣹينْ گࣴنْ ݒِݖّ يِ!‏ ‏25 اَنَ ڪَنْ ݖِ يࣹينْ مࣷويْ جَاخْلࣹ، دِ خَلاَتَ خَلاَتْ بَڪَّنَمْ بَ مَنَ يࣷڪّ لُ تࣷلّࣷوڪْ خࣹفْ اَڪْ خِݒِّ ݖِ اࣵݒَّمْ؟ ‏26 ندࣹگَمْ لࣷولُ گࣴنَ تُوتْ سَخْ مࣷو لࣹينْ تࣴ، اَنَ لُ نگࣹينْ دِ خَلاَتْ ݖِ لࣹنࣹينْ؟
‏27 «سࣹيتْلࣹينْ نِ تٗورْ-تٗورْ يِيْ سَخࣹ؞ سࣷنُّݧُ، اࣴݖُّݧُ، وَايࣹ مَا لࣹينْ ڪࣷ وَخْ، سُلࣹيْمَانْ اَڪْ دَرَجَامْ جࣺݒّ سَخْ، سࣷلُوُلْ وࣷونْ نِ بࣹنّ ݖِ ݧࣷومْ؞ ‏28 ندࣹگَمْ ݧَخُمْ تࣷولْ مِيْ دُندّ تࣹيْ، تࣹ لࣺيگِ ݧُ سࣵنِّ ڪࣷ ݖِبْ تَالْ، مࣷومْ لَ يࣵلَّ سࣷلَلࣹ نࣷونُ، خَنَا اَسْتَمَاڪْ يࣹينْ ڪَيْ؟ يࣹينَڪَ نࣺيوْ نگࣴمْ!‏ ‏29 يࣹينْ نَگْ، بُلࣹينْ تِيسࣷو لُ نگࣹينْ لࣹڪّ، مبَا لُ نگࣹينْ نَانْ؛ تࣹ بُ يࣷويُ خَلاَتْ اُبْ سࣹينْ بࣷݒّ؞ ‏30 ندَخْ لࣷولُ لࣺݒّ، خࣹيتِ يࣺيفَرِ اࣵدِّنَ، ݧࣷو ڪࣷيْ تِيسࣷو، تࣹ يࣹينْ، بَايْ بِ خَمْ نَ نࣹ سࣷخْلَ نگࣹينْ ڪࣷ؞ ‏31 وَايࣹ تِيسࣷولࣹينْ نگُورَمْ، سُ بࣷوبَا دࣹيسْ نَ لࣹينْ ݖِ دࣷلِّلْ لࣷولُ؞
‏32 «بُلࣹينْ رَگَلْ دَرَ، يࣹينْ جُرْ گُ سࣹوْ گِ، ندَخْ سࣹينْ بَايْ لَ سࣷوبْ، مُ جَگْلࣹيلْ لࣹينْ نگُورْ گِ؞ ‏33 جَايْلࣹينْ سࣹينْ اَلَلْ، سَرْخࣹ خَالِسْ بِ، سࣵڪُّلࣹينْ مبُوسْ يُ دُلْ رࣵݒّ، مُيْ دࣹنݖُ اࣴلّࣴگْ بُ دُلْ جࣹيخْ فَ اَسَمَانْ، فَ ڪࣷ سࣵݖّ دُلْ جࣹگࣹ، مَخْ دُ ڪࣷ يࣵقْ؞ ‏34 ندَخْ فُ سَبْ دࣹنݖّ نࣹڪّ، سَبْ خࣷلْ اِتَ نگَ فَ؞
تࣹيوْلُ وَرْ نَ
‏35 «تَڪُّلࣹينْ تࣹ سࣹينِ لࣵمݒّ دِ تࣵڪّ؞ ‏36 مࣹلْ-لࣹينْ نِ نِتْ ݧُيْ سࣺينْتُ سࣹينْ نجَاتِگࣹ لُ جٗگࣹ ݖࣺيتْ، ندَخْ بُ دِڪّࣹي بَ فࣴگّ بُنتّ بَ، ݧُ دَلْدِ ڪࣷ اُبِّلْ؞ ‏37 ندࣷڪّلࣹي سُرْگَ يࣷويُ سࣹينْ نجَاتِگࣹ دِڪّ فࣹڪّ ݧُ تࣹيوْلُ؞ مَا لࣹينْ ڪࣷ وَخْ دࣺيْ، نجَاتِگࣹ لࣹييْ تَڪُّ، تࣷوگَلْ لࣹينْ بَ نࣷݒِّ دِڪّ، تَاجَلْ لࣹينْ، ݧُ لࣹڪّ؞ ‏38 ندࣹگَمْ بَ گُدِّ گَ خَاجࣹي لَ نجَاتِگࣹ لَ ݧِبّسِ، بَ فࣹڪّ سُرْگَ يَ تࣹيوْلࣷو نࣷونُ، اَمْ بَ سُوفْ سࣹدّࣹي لَ، لُ مُ ݖِ مَنَ دࣷونْ، ندࣷڪّلࣹي ݧࣷومْ؞ ‏39 وَايࣹ لِي دࣹ خَمْلࣹينْ ڪࣷ:‏ بࣷرࣷومْ ڪࣴرْ، سُ خَمࣷونْ وَنْ وَخْتُ لَ اَبْ سࣵݖّ دِ ݧࣴوْ، دُ ڪࣷ وَݖّ مࣷوسْ، مُ دࣵجِّيْ، دُگّ ڪࣴرَمْ؞ ‏40 ڪࣷنْ يࣹينْ اِتَمْ وَاجْلࣹينْ بَ جࣹڪّ، ندَخْ وَخْتُ وُ نگࣹينْ فࣷوگُلْ، ݖِ لَ دࣷومُ نِتْ ڪِ دِ ݧࣴوْ؞»
‏41 ݒِيࣹيرْ نَگْ نࣹ عِيسَىٰ:‏ «سَنگّ بِ، ندَخْ نُنْ دࣷݝّ نگَ لࣺيبْ لِي، اَمْ ݧࣺݒَّ ڪࣷ مࣷومْ؟» ‏42 سَنگّ بِ نࣹ ڪࣷ:‏ «اَنَ ڪَنْ نَگْ مࣷويْ سَيْتُڪَتْ بُ وٗورْ تࣹ مُوسْ، بَ نجَاتِگࣹيمْ بَتَلࣹ ڪࣷ ݖُرْگَگْ ڪࣴرَمْ، نگِرْ مُ دِ لࣹينْ جࣷخْ سࣹينْ نجࣴلْ ݖَ بَ مُيْ جࣷتْ؟ ‏43 ندࣷڪّلࣹي سُرْگَ بُ نجَاتِگࣹيمْ دِڪّ، فࣹڪّ ڪࣷ مُيْ دࣷخَلࣹ نࣷونُ؞ ‏44 مَا لࣹينْ ڪࣷ وَخْ دࣺيْ، اَلَلَمْ جࣺݒّ لَ ڪࣷ نجَاتِگࣹ لَ دِ فَرْ دࣺنڪّ؞ ‏45 وَايࣹ سُ جَوْرِݧْ جَ نࣹي ݖِ خࣹلَمْ:‏ ‹سَمَ نجَاتِگࣹ دِڪَّگُلْ لࣺيگِ،› بَ تَخْ مُ تࣵمْبَلࣹي دٗورْ سُرْگَ يَ، گٗورْ اَڪْ جِگࣺينْ، دِ لࣹڪّ اَڪَ نَانْ، دِ مࣵنْدِ، ‏46 سُ بࣷوبَا نجَاتِگࣹ لاَيْ دِڪّ بࣺسْ بُ مُ ڪࣷ سࣺينُوُلْ اَڪْ وَخْتُ وُ مُ فࣷوگُلْ، تࣹ مبُگَلْ مُ مࣹتِّ لَ ڪࣷيْ مبُگَلْ، سࣺدّ ڪࣷ ݖِ اَݧُبْ گࣴمَدِڪَتْ ݧِ؞
‏47 «جَوْرِݧْ جࣷوجُ خَمْ نَمّࣹيلُ نجَاتِگࣹيمْ، تࣹ تَخُلْ مُ وَاجْ ڪࣷ، بَ جࣴفࣹوُ ڪࣷ، اَيْ يࣹتَمْ دِنَ بَرࣹ؞ ‏48 وَايࣹ جَوْرِݧْ جِ خَمُلْ نَمّࣹيلُ نجَاتِگࣹيمْ، تࣹ تࣹيوُ ڪࣷو دࣹفْ لُ يࣹلّࣷو اَيْ دُمَ، يࣹتْ يُ نࣺيوْ لَيْ اَمْ؞ ڪࣺݒّ ڪُ ݧُ جࣷخْ لُ بَرࣹ، دࣹيسْ نَ لَ تࣷݒّ لُ بَرࣹ؞ ڪࣺݒّ ڪُ ݧُ دࣺنڪّ جࣷݒّ، دࣹيسْ نَ لَ فࣹيࣹيڪُ لُ ڪࣷ اࣴݒّ؞
نگࣴمْ اِنْدِ نَ فࣺيوَلࣷو
‏49 «سَوَرَ لاَ تَالْسِ ݖِ ڪَوْ سُوفْ، تࣹ مَاڪَ بࣴگّࣷونْ مُ نࣹ جِݒّࣺتْ خَاتْ!‏ ‏50 اَمْ نَ نَگْ سٗوبُ گُ مَ وَرَ سٗوبُ، وَايࣹ ندَوْ نجࣵقَرࣹ ݖِ مَنْ لِ فࣹيڪْ لࣷولُ سࣷتِّوُلْ؞ ‏51 فࣷوگْ نگࣹينْ نࣹ مَيࣹ جࣵمّ ݖِ ڪَوْ سُوفْ مࣷو مَ اِنْدِ؟ دࣺيتْ ڪَيْ؞ مَا لࣹينْ وَخْ لِي، خَنَا اَگْ فࣺيوَلࣷو؞ ‏52 گَنَّاوْ-سِ-تࣹيْ، بُ جُرٗومِ نِتْ بࣷڪّࣹي گࣹنّ ڪࣴرْ، دِنَݧُ فࣺيوَلࣷو؛ ݧࣹتّ فࣺيوَلࣷو اَڪْ ݧَارْ، ݧَارْ فࣺيوَلࣷو اَڪْ ݧࣹتّ؞ ‏53 بَايْ دِنَ فࣺيوَلࣷو اَڪْ دࣷومَمْ جُ گٗورْ، دࣷومْ جَ فࣺيوَلࣷو اَڪْ بَايَمْ؛ ندࣹيْ فࣺيوَلࣷو اَڪْ دࣷومَمْ جُ جِگࣺينْ، دࣷومْ جَ فࣺيوَلࣷو اَڪْ ندࣹيَمْ؛ ندࣹيْ فࣺيوَلࣷو اَڪْ جَبَرُ دࣷومَمْ، جَبَرْ جَ فࣺيوَلࣷو اَڪْ ندࣹيُ جࣴڪّࣴرْ جَ؞»
خَمْ ݒِرِمْ جَمࣷنࣷ وَرْ نَ
‏54 عِيسَىٰ نࣹيتِ مبࣷولࣷو مَ:‏ «بُ نگࣹينْ گِسࣹي مُ خِينࣹ سࣷوُ، دَنگࣹينْ دَلْ نَانْ:‏ ‹تَوْ بَا نگِ ݧࣴوْ،› تࣹ نࣷونُ لَيْ اَمࣹ؞ ‏55 بُ نگࣹلَوْ بَوࣷو بࣴجْ-سَالُمْ، نگࣹينْ نࣹ:‏ ‹تࣵنگَايْ وُ مࣹتِّ لَيْ دࣷونْ،› تࣹ مࣷويْ اَمْ؞ ‏56 نَافࣺقْ يِ نگࣹينْ دࣷونْ!‏ مࣹلࣷوْ اَسَمَانْ اَڪْ سُوفْ، مَنْ نگࣹينْ ڪࣷو فِرِ؞ خࣹوْ-خࣹوِ جَنْتِ تࣹيْ جِي نَگْ، نُ نگࣹينْ اُمݒّلࣹي ݒِرࣹيمْ؟
‏57 «تࣹ اِتَمْ لُ لࣹينْ تࣹي اࣵتّࣹ نجُبْ، يࣹينْ ݖِ سࣹينْ بࣷݒّ؟ ‏58 بࣷو اࣵنْدࣹيڪْ سَبْ جࣷتࣹوَالࣹ، نگࣹينْ جࣴمْ ݖَ اࣵتّࣹڪَتْ بَ، فࣹخࣹيلَ جُبࣷوڪْ مࣷومْ ݖِ يࣷونْ وِ؞ لُ ڪࣷ مࣷيْ مُ يٗبُّ لَ ݖِ يࣷونْ، يࣷونْ جࣺبَّلْ لَ ندَوَلْ بُورْ، مُ نࣹ لَ رࣵݒّ تࣴجْ؞ ‏59 مَ نࣹ لَ، دࣷو فَ گࣺنّࣹ مُڪّ تࣹ فࣹيُلࣷو لࣺݒّ، بَ ݖِ ݒࣷسࣹتْ بِ گࣴنَ تُوتْ؞»
