دࣹلُّ ݖِ يࣷونُ مُوسَىٰ دُ خࣹلْ
3
‏1 يࣹينْ وَا گَلَسِ، يࣹينَ ݧࣵڪّ خࣹلْ!‏ ڪُ لࣹينْ گࣴلࣴمَلْ، تࣹ نَ عِيسَىٰ اَلْمَسِ دࣹييࣹ ݖَ بَنتّ بَ دِ لُ ݧُ لࣹينْ لࣹيرَلَلْ بَ مُ مࣹلْ نِ لُ نگࣹينْ گِسࣹ سࣹينِ بࣴتْ؟ ‏2 لࣹنّ رࣹڪّ لاَ بࣴگّ نگࣹينْ خَمَلْ مَ ڪࣷ:‏ ندَخْ ݖِ جࣴفِ سࣵمّ ندِگَلُ يࣷونْ نگࣹينْ جࣷتࣹ نࣷووُگْ يࣵلَّ گِ، اَمْ لِ نگࣹينْ دࣺگّ خِبَارُ جࣵمّ بِ بَ گࣴمْ ڪࣷ مࣷو لࣹينْ ڪࣷ مَيْ؟ ‏3 اَنَ نُ نگࣹينْ مَنَ ݧࣵڪّࣹ نِي خࣹلْ؟ نگࣹينْ تࣵمْبَلࣹي ݖِ نࣷو گِ، بَ نࣷݒِّ دِ جࣺيمَ مࣷتّلِڪࣷو جࣴفِ نِتُ سُوخُ نࣹينْ!‏ ‏4 دَنگࣹينَ دࣺڪُّ يِي ݖࣷونࣷ يࣺݒّ ݖِ نࣹينْ؟ خَنَا دࣺيتْ ڪَيْ!‏ ‏5 ڪِ لࣹينْ دِ مَيْ نࣷو گِ، تࣹيْ سࣷتَّلْ اَيْ ڪࣺيمَانْ ݖِ سࣹينْ بِيرْ، ندَخْ لِ نگࣹينْ سࣵمّ يࣷونْ مࣷو تَخْ مُ دِ ڪࣷ دࣹفْ، اَمْ لِ نگࣹينْ دࣺگّ، بَ گࣴمْ مࣷو تَخْ؟
‏6 نࣷونُ لَ إِبْرَاهِيمَ گࣴمࣹ يࣵلَّ، «ݧُ وَݧَّلْ ڪࣷ نگࣴمَمْ اَگْ نجُبْ¤؞»
‏7 خَمْلࣹينْ بࣷوگْ نࣹ نِتْ ݧِيْ بࣷرࣷومْ نگࣴمْ، ݧࣷوݧُ ݧࣷويْ دࣷومِ إِبْرَاهِيمَ؞ ‏8 مبِندّ مِ مࣷو وَاجَلࣷونْ لُ جِيتُ نࣹ ݖِ ڪَوْ نگࣴمْ لࣹيسْ دِ جࣷخࣹ خࣹيتْ يِ دُلْ يَوُتْ، اࣵتّࣹبْ نِتْ ݧُ جُبْ؞ مࣷو تَخࣷونْ مُ يࣺگَلْ إِبْرَاهِيمَ لُ جِيتُ، بِي خِبَارُ جࣵمّ:‏ «خࣹيتِ اࣵدِّنَ يࣺݒّ ݖِ يَوْ لَݧُيْ بَرْڪࣹيلࣹ¤،» ‏9 بَ تَخْ نَ، بࣷرࣷومْ نگࣴمْ يِ لࣹيسْ دِ بࣷولࣹيڪْ إِبْرَاهِيمَ مِيْ بࣷرࣷومْ نگࣴمْ، بَرْڪࣹيلْ لࣹينْ؞
‏10 مبࣷولࣹمْ ݧِ يَاڪَارْ سࣹينْ جࣴفْ جُ ݧُ سࣵمّࣹ يࣷونُ مُوسَىٰ، رࣹݒّ نَݧࣷو اَلْڪُ، ندࣹ بِنْدࣹيسْ نَ نࣹ:‏ «اَلْڪࣵنْدࣹ ݧࣹيلْ نَ ڪࣺݒّ ڪُ سَخࣷووُلْ مبࣷولࣹمْ دِگَلْ يِ ݧُ بِندّ ݖِ تࣺيرࣹبْ يࣷونْ وِ، نگِرْ دِ ڪࣷ جࣴفࣹ¤؞» ‏11 لِ سࣵمّ يࣷونْ تَخُلْ ڪࣹنّ جࣷتْ اࣵتّࣹبْ ڪُ جُبْ فِ ڪَنَمْ يࣵلَّ سَخْ، لُ لࣹيرْ نࣵݧّ لَ، ندࣹ:‏ «ڪُ جُبْ ندَخْ نگࣴمَمْ مࣷويْ دُندّ؞» ‏12 يࣷونْ وِ نَگْ اَجُوُلْ ݖِ نگࣴمْ، ندࣹ:‏ «ڪُيْ جࣴفࣹ دِگَلْ يࣷويُ، يَايْ دُندّ ندَخْ ݧࣷومْ¤؞»
‏13 اَلْمَسِ مࣷومْ، مࣷو نُ جࣷتْ ݖِ اَلْڪࣵنْدࣹيْ يࣷونْ وِ، بَ مُ دࣷونࣹي ڪُ اَلْڪُ نگِرْ نُنْ، ندࣹ:‏ «اَلْڪُتࣹ ݧࣹيلْ نَ ڪࣺݒّ ڪُ ݧُ رࣹيْ، وࣺڪّ ڪࣷ ݖِ بَنتّ¤؞» ‏14 نࣷونُ لَ دࣹمࣹ نگِرْ بَرْڪࣹبْ إِبْرَاهِيمَ وࣵݖَّلْ خࣹيتْ يِ دُلْ يَوُتْ ندَخْ اَلْمَسِ عِيسَىٰ، نگِرْ اِتَمْ نُ جࣷتْ ݖِ دِگࣹبْ نࣷو گُ سࣹلّ گِ، ݖِ ڪَوْ نگࣴمْ؞
يࣷونُ مُوسَىٰ فࣹݖِّوُلْ ڪٗلّࣴرࣹ
‏15 بࣷڪّ يِ، مَ مِسَالࣹ ڪࣷ مبِرِ دࣷومْ آدَمَ؞ ڪَيِتُبْ دࣷنَلࣹ بُ نِتْ دࣷݝّ بِندّ، سُ دࣹي لُ وٗورْ، دࣹيسُ ڪࣷ نࣹينَلْ، مبَا دِ ݖِ يࣷڪّ دَرَ؞ ‏16 تࣹ إِبْرَاهِيمَ لَݧُ دِگࣷونْ دِگࣹ يِ مࣷوڪْ ڪِ تَخَوَلْ اَسْڪَنَمْ؞ نࣹوُݧُ:‏ «ݧِ تَخَوَلْ اَيْ اَسْڪَنَمْ»، مبࣹتࣹ ݧُ بَرࣹ لَݧُيْ وَخْ، وَايࣹ وࣹنّ رࣹڪّ لَ:‏ «ڪِ تَخَوَلْ اَسْڪَنَمْ،» تࣹ ڪِ تَخَوَلْ اَسْڪَنَمْ مࣷو دِ اَلْمَسِ؞ ‏17 لِ مَ ݖِيْ وَخْ مࣷويْ لِي:‏ ڪٗلّࣴرࣹگْ يࣵلَّ گِ تَخَوࣹ نِ ڪَيِتُ دࣷنَلࣹ گُ مُ دࣴگَّلْ، يࣷونُ مُوسَىٰ، وِ دِڪّ ݧࣹينْتِ تࣺيمࣺيرِ اَتْ اَڪْ فَنْوࣹيرْ گَنَّاوْ ڪٗلّࣴرࣹ گِ، مَنُ ڪࣷو نࣹينَلْ، بَ فࣷمّ دِگࣹبْ¤ يࣵلَّ؞ ‏18 سُ دࣷونْ ݖِ يࣷونُ مُوسَىٰ لَ ندࣷنَلࣹ گَ اَجُ، ڪࣷنْ دࣷوتُلْ دࣷونْ لُ دِگࣹبْ يࣵلَّ وَرَلْ؞ وَايࣹ ݖِ ڪَوْ اَبْ دِگࣹ لَ يࣵلَّ بَاخࣹي إِبْرَاهِيمَ اَوْ يِوَمْ؞
ڪُ گࣴمْ اَلْمَسِ، يࣷونُ مُوسَىٰ وࣵݖّ نَ لَ
‏19 وَاوْ ڪࣷنْ يࣷونُ مُوسَىٰ، اَنَ لُيْ نجَرِݧَمْ؟ دࣹيسْ ڪࣷو تࣷفْتَلْ ݖَ دِگࣹ بَ، نگِرْ مُيْ فࣺسَلْ مࣷيِ نِتْ ݧِ بَ ڪࣹرࣷوگْ بࣷرࣷومْ دِگࣹ، بِ تَخَوَلْ اَسْڪَنُ إِبْرَاهِيمَ، دِڪّ؞ اَيْ مَلاَڪَ ݧࣷو بِرَلْ يࣷونْ وِ، ݖِ ندِمْبَلَلْ اَبْ جࣷتّلِڪَتْ¤؞ ‏20 اَبْ جࣷتّلِڪَتْ نَگْ ݧَارَ ڪࣷيْ سࣺقْ، تࣹ يࣵلَّ ڪࣹنّ دࣷݝّ لَ؞
‏21 ندَخْ ڪࣷنْ يࣷونُ مُوسَىٰ دَفَ جُويࣷو اَڪْ دِگࣹيْ يࣵلَّ؟ مُڪّ ڪَيْ؞ ندَخْ ڪَتْ سُ فࣹڪّࣷونْ نࣹ دَݧࣷو مَيࣹ وࣹنّ يࣷونْ وُ مَنَ مَيْ نِتْ ڪُ ڪࣷ سࣵمّ، مُ دُندّ، ڪࣷنࣷونْ اࣵتّࣹبْ ڪُ جُبْ فِ ڪَنَمْ يࣵلَّ مَنَ جٗگࣹ دࣴگّ ݖِ سࣵمّ گُ نِتْ ڪِ سࣵمّ يࣷونْ وِ؞ ‏22 وَايࣹ مبِندّ مِ بِرَلْ نَ نࣹ مبࣷولࣹمْ اࣵدِّنَ مࣷو تࣴنڪُ ݖِ ڪِلِفْتࣹيفُ بࣵڪَّارْ، نگِرْ دِگࣹ بِ گࣴمْ عِيسَىٰ اَلْمَسِ دِ جَگْلࣹ، مَنَ دِڪَّلْ اَجِ گࣴمْ ݧِ؞
‏23 بَلاَ يࣷونُ نگࣴمْ وࣹي دِڪّ، دࣹيسْ نࣷو جࣵݒّࣷونْ تࣴنڪّ ݖِ يࣷونُ مُوسَىٰ، بَ ڪࣹرَ يࣷونُ نگࣴمْ وِ فࣹيݧࣹي؞ ‏24 نࣷونُ لَ يࣷونُ مُوسَىٰ دࣷونࣹ سُنُ نجَاتِگࣹ، تࣹتّࣹ نُ بَ ݖِ اَلْمَسِ، نگِرْ نُ جَگࣷو اࣵتّࣹبْ نِتْ ݧُ جُبْ ݖِ ڪَوْ نگࣴمْ؞ ‏25 وَايࣹ لࣺيگِ، بِ يࣷونُ نگࣴمْ وِ دِڪّࣹي، نࣹڪَّتُنُ ݖِ ڪِلِفْتࣹيفُ نجَاتِگࣹ؞
دࣷومِ يࣵلَّ نگࣹينْ
‏26 يࣹينْ ݧࣺݒّ دࣷومِ يࣵلَّ نگࣹينْ، ندَخْ لِ نگࣹينْ گࣴمْ اَلْمَسِ عِيسَىٰ؞ ‏27 ندَخْ ڪَتْ مبࣷولࣹمْ يࣹينْ ݧِ ݧُ سٗوبْ سٗوبࣹبْ اَلْمَسِ، جࣴمُّ اَلْمَسِ نگࣹينْ سࣷلࣷو؞ ‏28 ڪࣷنْ نَگْ اَمُلْ يَوُتْ اَڪْ ڪُ دُلْ يَوُتْ، اَمُلْ اَبْ جَامْ اَڪْ اَسْ گࣷرْ، اَمُلْ گٗورْ اَڪْ جِگࣺينْ؞ يࣹينْ ݧࣺݒّ بࣹنّ نگࣹينْ ݖِ اَلْمَسِ عِيسَىٰ؞ ‏29 تࣹ سُ نگࣹينْ بࣷڪّࣹي ݖِ اَلْمَسِ، يࣹينْ اَيْ اَسْڪَنُ إِبْرَاهِيمَ، تࣹ يࣹينَ جَگࣷو دِگࣹ بَ؞
