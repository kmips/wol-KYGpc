3
‏1 بَ لࣷولُ وࣺيࣹي نُ وَلْبَتِڪُ، اَوࣹ يࣷونْ وَ يࣺيگْ جࣴمْ بَسَنْ؞ اࣷگْ بُورُ بَسَنْ نَگْ دِڪّ دَجࣹيڪْ نُنْ، مࣷوڪْ گࣵنگࣷورَمْ گࣺݒّ، نگِرْ سࣷنگّ نُ خَرࣹ ݖَ اࣹدّرࣹيْ؞ ‏2 اَجِ سَخْ جِ نࣹ مَ:‏ «بُ ڪࣷ رَگَلْ، ندَخْ ݖِ سَيْ لࣷخࣷ لاَ ڪࣷ تࣹگْ، مࣷوڪْ گࣵنگࣷورَمْ گࣺݒّ، اَڪُمْ رࣺيوَمْ؞ نَنگَ ڪࣷ دࣹفْ نَ نگَ دࣹفࣷونْ سِوࣷنْ بُورُبْ اَمࣷرࣹينْ بَ وࣷونْ ݖَ اࣹسْبࣷنْ؞» ‏3 نࣷونُ لَ سُنُ يࣵلَّ اَجِ سَخْ جِ تࣹگࣹ اࣷگْ بُورُ بَسَنْ اِتَمْ ݖِ سُنُيْ لࣷخࣷ، مࣷوڪْ گࣵنگࣷورَمْ گࣺݒّ، نُ دُمَ لࣹينْ بَ دࣹسࣹوُلْ ڪُيْ دُندّ؞ ‏4 بࣷوبَ لَنُ نَنگُ مبࣷولࣹمِ دࣴڪَّمْ؞ اَمُلْ بࣹنّ دࣴڪّ بُ نُ نَنگُوُلْ ݖِ ݧࣷومْ:‏ جُرٗومْ بࣹنّ فُڪِّ دࣴڪّ، مبࣷولࣹمْ دِيوَانُ اَرْگࣷبْ، بَ بࣷڪّ ݖَ نگُورُگْ اࣷگْ ݖَ بَسَنْ؞ ‏5 يࣷويُ يࣺݒّ دِ دࣴڪّ يُ دࣵبّلِڪࣷو تَتَ يُ ڪَوࣹ، اَڪْ بُنتّ يُ تࣴجࣹ رࣵݒّ، تࣹ لِمَالࣹيسُلْ دࣴڪّ-دࣴڪَّانْ يُ بَرࣹي بَرࣹ، يُ دَرَ وࣴرُلْ؞ ‏6 نُ فَاگَاگَلْ لࣹينْ، نَ نُ دࣹفࣷونْ اَڪْ سِوࣷنْ بُورُبْ اࣹسْبࣷنْ؞ دࣴڪّ يࣺݒّ لَنُ فَاگَاگَلْ:‏ گٗورْ اَڪْ جِگࣺينْ بَ ݖَ تُوتْ-تࣵنڪّ يَ؞ ‏7 مبࣷولࣹمْ جُرْ گَاڪْ اَلَلِ دࣴڪّ يَ نَگْ، نُ سࣴخࣴتࣷو؞
‏8 جَنتّ يࣷويُ لَنُ نَنگࣷو ݖِ لࣷخࣷلْ ݧَارِ بُورِ اَمࣷرࣹينْ ݧࣷوݧَ ݖَ وࣵلاَّ دࣹخُ يُرْدَنْ، رࣺيوْ مَ دَلࣹ ݖَ خُرُ اَرْنࣷنْ، بَ ݖَ تُنْدُ اࣹرْمࣷنْ؞ ‏9 اࣹرْمࣷنْ گࣷوگُ لَ وَا سِدࣷنْ دِ وࣷويࣹ سِرْيࣷݝْ، اَمࣷرࣹينْ ݧِ دِ ڪࣷ وࣷويࣹ سࣹنِرْ؞ ‏10 مبࣷولࣹمْ دࣴڪِّ جࣷورْ گَ لَنُ نَنگُ، اَڪْ مبࣷولࣹمْ دِيوَانُ گَلࣵدّ، اَڪْ مبࣷولࣹمْ بَسَنْ، بَ ݖَ سَلْڪَ اَڪْ اࣹدّرࣹيْ، دࣴڪِّ نگُورُگْ اࣷگْ مُ بَسَنْ؞ ‏11 مࣹنُّمْ اࣷگْ بُورُ بَسَنْ دࣷݝّ مࣷو مُݖّࣷونْ ݖَ ندࣹسُ رࣹفَيࣹينْ ݧُ ݒࣷنڪَلࣹ ݧَ؞ لَلَمْ بَ، لَلُ وࣹݧْ گُ ݧُولْ لَ؞ مَ نگَ نࣷونࣹي ݖَ رَبَ، دࣴڪُّبْ اَمࣷنࣹينْ ݧَ؞ گُدَّايُ لَلْ بَ ݖِ خَسَبُ گٗورْ، جُرٗومْ ݧࣹينْتِ خَسَبْ لَ، دِ ݧࣹينْتِ مࣹيتَرْ اَڪْ گࣹنّ-وࣵلّ، يَاتُوَايْ بَ، ݧࣹينْتِ خَسَبْ، دِ ݧَارِ مࣹيتَرْ؞
سࣺدَّلࣹ نَݧُ ݒࣹنڪُبْ يُرْدَنْ
‏12 سُوفْ سࣷوسُ نُ نَنگُ ݖَ جَنتّ يࣷويَ، تࣹ دَلࣹ ݖَ اَرࣷوࣹرْ گَ ݖَ خُرُ اَرْنࣷنْ ݖَ بࣴجْ-سَالُمْ، اَڪْ ݖَ بࣴجْ-گࣵنَّارْ، گࣹنّ-وࣵلُّ دِيوَانُ تُنْدِ گَلࣵدّ، اَڪِ دࣴڪَّمْ، مَا ڪࣷ جࣷخْ رُبࣹنࣹينْ ݧِ، اَڪْ گࣵدّࣹينْ ݧِ؞ ‏13 ندࣹسُ گَلࣵدّ اَڪْ مبࣷولࣹمْ بَسَنْ گُ نگُورُگْ اࣷگْ، مَ جࣷخْ ڪࣷ گࣹنّ-وࣵلُّ گِيرُگْ مَنَسࣹ؞ دِيوَانُ اَرْگࣷبْ گࣺݒّ اَڪْ بَسَنْ گࣷوگُ گࣺݒّ، لَݧُ دَانْ وࣷويࣹ رࣺيوُمْ رࣹفَيࣹينْ ݧَ؞ ‏14 يَيِرْ مِ اَسْڪَنࣷو ݖِ مَنَسࣹ مࣷو نَنگُ مبࣷولࣹمْ دِيوَانُ اَرْگࣷبْ، بَ ݖَ ڪࣹمُ رࣺيوُمْ گࣹسُرࣹينْ ݧَ، اَڪْ مَاڪَتࣹينْ ݧَ؞ مࣷو تُدّࣹ بࣷݒَّمْ بَسَنْ گࣷوگُ، بَ تَخْ ݧُ دِ ڪࣷ وࣷويࣹ سَنݖِ يَيِرْ بَ تࣹيْ جِي؞ ‏15 اَسْڪَنُ مَڪِرْ لاَ جࣷخْ دِيوَانُ گَلࣵدّ¤؞ ‏16 رُبࣹنࣹينْ ݧࣹيڪْ گࣵدّࣹينْ ݧِ لاَ جࣷخْ لَ دَلࣹ دِيوَانُ گَلࣵدّ بَ ݖَ خُرُ اَرْنࣷنْ، دِگُّ خُرْ وَ دِ سࣹينْ ڪࣹمُ سُوفْ ݖِ گࣹنّ وࣹتْ، دࣹخُ يَبࣷگْ دِ سࣹينْ ڪࣹمُ ݧࣷوڪْ اَمࣷنࣹينْ ݧَ؞ ‏17 فَ جࣷورْ گَ فࣹگّࣷوڪْ دࣹخُ يُرْدَنْ اِتْ دِ ڪࣹمُ، دِگَّنْتࣹ دࣹخُ ڪِنࣹرࣹتْ ݖَ بࣴجْ-گࣵنَّارْ، اَڪْ ݖَ بࣴجْ-سَالُمُ گࣺيجُ خࣷرࣷمْ گَ، بَ بَ نگَيْ اࣵگّ ݖَ سُوفُ تُنْدُ ݒِسْگَ، ݖَ ݒࣹنڪُ؞
‏18 جَنتّ يࣷويُ لاَ لࣹينْ سَنتّ نࣹ لࣹينْ¤:‏ «سࣹينْ يࣵلَّ اَجِ سَخْ جِ مࣷو لࣹينْ جࣷخْ رࣺيوْ مِي، نگِرْ نگࣹينْ نَنگُ ڪࣷ؞ نَ خَرࣹڪَتْ يࣺݒّ گࣵنَّايُ، جࣵلّ، جِيتُ سࣹينْ بࣷڪِّ بࣵنِّ إِسْرَاءِيلْ؞ ‏19 سࣹينِ جَبَرْ دࣷݝّ، اَڪْ سࣹينِ دࣷومْ اَڪْ سࣹينُگْ جُرْ، ندَخْ خَمْ نَا نࣹ جُرْ گُ تَڪُّ نگࣹينْ اَمْ، ݧࣷويْ دࣹسْ ݖِ سࣹينْ دࣴڪّ يِ مَ لࣹينْ جࣷخْ، ‏20 بَ ڪࣹرࣷوگْ اَجِ سَخْ جِ نࣷݒَّلࣹي سࣹينْ بࣷڪِّ بࣵنِّ إِسْرَاءِيلْ نِ يࣹينْ، فࣹڪّ لࣹينْ نَنگُ، ݧࣷومْ اِتْ، رࣺيوْ مَ لࣹينْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ جࣷخْ، ݖَ وࣵلاَّ دࣹخُ يُرْدَنْ؞ سُ ڪࣷ دࣹفࣹي ڪُ نࣹڪّ ݖِ يࣹينْ سَݧَ دࣹلُّ ݖَ ݖࣺرُ سُوفَمْ بَ مَ ڪࣷ جࣷخْ؞»
‏21 يࣷسُوࣹ اِتَمْ، سَنتّ نَا ڪࣷ ݖِ جَنتّ يࣷويُ نࣹ ڪࣷ:‏ «سَ بࣴتْ مࣷو لَ مَيْ مبࣷولࣹمْ نِ سࣹينْ يࣵلَّ اَجِ سَخْ جِ دࣹفْ اَڪْ ݧَارِ بُورْ يِي؞ نࣷونُ رࣹڪّ لَ اَجِ سَخْ جِيْ دࣹفْ اَڪْ مبࣷولࣹمْ نگُورْ يَ نگَ جࣴمْ؞ ‏22 بُلࣹينْ لࣹينْ رَگَلْ، ندَخْ سࣹينْ يࣵلَّ اَجِ سَخْ جِ مࣷو لࣹينْ دِ خࣹيخَلْ؞»
مُوسَىٰ دُگُّلْ ڪَنَانْ
‏23 جَنتّ يࣷويُ لاَ تِنُ اَجِ سَخْ جِ، نࣹ ڪࣷ:‏ ‏24 «اࣺيْ بࣷرࣷومْ بِ اَجِ سَخْ جِ، يَا مَ وࣷنَگُمْ لࣹنّ ݖِ سَ مࣵگَّايْ اَڪْ سَ دࣷولࣹ، سَنگّ بِ؞ اَنَ يࣵلَّ جُ نࣹڪّ اَسَمَانْ، مبَا ݖِ ڪَوْ سُوفْ جُ مَنَ رࣷيْ سَيْ جࣴفْ اَڪْ سَيْ جَلࣷورࣹ؟ ‏25 مَيْ مَ رࣹڪّ مَ جࣵلّ، بَ گِسْ رࣺيوْ مُ بَاخْ، مَ ݖَ وࣵلاَّ دࣹخُ يُرْدَنْ، تُندّ وُ رَفࣹتْ وࣷووُ بَ ݖَ لِبَݝْ!‏» ‏26 وَايࣹ فࣹڪّ نَ اَجِ سَخْ جِ مࣹرࣹ مَ لࣷولْ ندَخْ يࣹينْ، بَ فَالࣹوُ مَ؞ اَجِ سَخْ جِ نࣹ مَ:‏ «نَ دࣷيْ؞ بُ مَ وَخَتِ لُ جࣴمْ ݖِ لࣷولُ؞ ‏27 يࣺيگَلْ بَ ݖَ ڪَوْ تُنْدُ ݒِسْگَ، نگَ دَوَلْ سَ بࣴتْ، تࣹ سࣺينُ رࣺيوْ مَ وࣹتُ سࣷوُ، بَ بࣴجْ-گࣵنَّارْ، بࣴجْ-سَالُمْ، بَ ݒࣹنڪُ؞ خࣷولَلْ بُ بَاخْ، ندࣹ دࣷو جࣵلّ دࣹخُ يُرْدَنْ گِي؞ ‏28 وَايࣹ نَنگَ يࣹبَلْ يࣷسُوࣹ، ݧَاخْ ڪࣷ، جࣵجّ ڪࣷ، ندَخْ مࣷومْ مࣷويْ جࣵلّ، جِيتࣹ مبࣷولࣷو مِي، تࣹ مࣷو لࣹينْ دِ جࣷخْ رࣺيوْ مࣷومُ نگَيْ سࣺينْ؞» ‏29 بَ مُ ڪࣷ دࣹفࣹي نُ تࣷوگْ ݖَ خُرْ وَ، جَنࣷوڪْ بࣹتْ ݒࣹوࣷرْ؞
