ݒِيࣹيرْ اَڪْ يࣷوَانْ دَجࣹ نَݧࣷوڪْ اࣵتّࣹڪَتْ يَ
4
‏1 نَڪَ لَ ݒِيࣹيرْ اَڪْ يࣷوَانْ دِ وَخْ اَڪْ مبࣷولࣷو مَ، سَرْخَلْڪَتْ يَاڪْ جَوْرِݧُ ڪࣴرْ يࣵلَّ گَاڪْ سَدُسࣹنْ يَ نࣹ جِمࣹيتْ ݖِ سࣹينْ ڪَوْ؞ ‏2 فࣹڪّ نَ لࣹينْ ݧࣺݧّ ݖَ لَ ݧُيْ وَارࣹ اَڪَ يࣺگْلࣹ ندࣹڪِّتࣹلْ عِيسَىٰ، تࣹ لࣷولُ دِ دࣴگَّلْ نࣹ ندࣹي-ندࣹڪِّ، لُ اَمْ لَ؞ ‏3 ݧُ دَلْدِ لࣹينْ تࣹگْ لࣷخࣷ، تࣴجْ لࣹينْ بَ ݖَ اࣴلّࣴگْ سَ، ندَخْ بࣷوبَ گُدِّ نَ؞ ‏4 فࣹڪّ نَ نَگْ، ݧُ بَرࣹ ݖَ ݧَ جࣷتࣷونَ دࣺگّ ڪࣵدُّ گَ گࣴمْ نَݧُ، بَ لِمُ گࣴمْڪَتْ ݧَ تࣷلُّ ݖِ جُرٗومِ جُنِّ (5000)؞
‏5 ݖَ اࣴلّࣴگْ سَ ڪِلِفَيْ يَوُتْ يِ اَڪْ مَگْ ݧِ اَڪْ فِرِڪَتِ يࣷونْ يِ دَجࣹ يࣹرُسَلࣹمْ؞ ‏6 اَنَسْ، سَرْخَلْڪَتْ بُ مَگْ بَ ݖَ لَ، اَڪْ ڪَيِفْ اَڪْ يࣷوَانْ اَڪْ اَلࣹگْسࣵنْدَرْ، اَڪْ مبࣷولࣹمْ بࣷڪِّ سَرْخَلْڪَتْ بُ مَگْ؞ ‏7 ݧُ تَخَوَلْ ݒِيࣹيرْ اَڪْ يࣷوَانْ ݖَ دِگّ بَ، لاَجْ لࣹينْ نࣹ:‏ «يࣹينْ، لِي نگࣹينْ دࣹفْ، ݖِ بَنْ خَمْ-خَمْ، اَڪْ ݖِ وَنْ تُرْ نگࣹينْ ڪࣷ دࣹفࣹ؟»
‏8 فَ لَ ݒِيࣹيرْ فࣹيسࣹ نࣷو گُ سࣹلّ گِ، نࣹ لࣹينْ:‏ «يࣹينْ ڪِلِفَيْ خࣹيتْ وِ اَڪْ مَگْ ݧِ، ‏9 ندࣹگَمْ يࣷونْ مࣷو نُيْ لاَجْ لُ جࣴمْ ݖِ ندِمْبَلْ لُ اَبْ لَفَݧْ جَگࣷو، اَڪْ ݖِ مَنْ ݒࣹخࣹ لَ لَفَݧْ بِ مُݖّࣹ، ‏10 خَمْلࣹينْ خࣺلّ، يࣹينْ ݧࣺݒّ اَڪْ بࣵنِّ إِسْرَاءِيلْ گࣺݒّ، نࣹ ݖِ تُرُ عِيسَىٰ اَلْمَسِ وَا نَسَرࣹتْ بِ، ڪِ نگࣹينْ دَاجࣷونْ ݖِ بَنتّ، تࣹ يࣵلَّ دࣹڪَّلْ ڪࣷ، ݖِ تُرَمْ دࣷݝّ لَ ڪِي وࣺرࣹ، بَ تَخَوْ ݖِ سࣹينْ ڪَنَمْ؞ ‏11 عِيسَىٰ مࣷومُ مࣷويْ:‏
‹دࣷجْ وِ نگࣹينْ بࣹدِّ وࣷونْ، يࣹينْ تَبَخْڪَتْ يِ،
تࣹ مُ مُجّ دِ دࣷجُ ݖࣷلّ وِ؞›
‏12 مُݖّ ڪَتْ اَمُلْ ݖِ ڪࣹنࣹينْ، ندَخْ فِ اَسَمَانْ تِيمْ، مَيࣹيسُلْ فِ نِتْ ݧِ وࣹنࣹينْ تُرْ وُ نُ وَرَ مُݖّࣹ؞»
‏13 بَ وَا ڪُرࣺلُ اࣵتّࣹڪَتْ يَ گِسࣹي فِتْ، وِ ݒِيࣹيرْ اَڪْ يࣷوَانْ اࣵنْدَلْ، تࣹ ݧُ خَمْ نࣹ نِتِ نࣹينْ ݧُ مَسُلَ جࣵنگّ لَݧُ، دَݧࣷو وَارُ؞ خَمࣷونْ نَݧُ نَگْ نࣹ ݧࣷوڪْ عِيسَىٰ نࣹڪّࣷونْ؞ ‏14 تࣹ اِتَمْ ڪَ ݧُ وࣺرَلَ نگَ تَخَوْ فَ سࣹينْ وࣹتْ، ݧُيْ گِسْ؞ مَنُݧُ ݖَا تࣹگَتِ دَرَ؞ ‏15 بَ مُ ڪࣷ دࣹفࣹي ݧُ جࣷخࣹ ندِگَلْ، گࣺنّࣹ لࣹينْ، دَلْدِ گِسࣹ ݖِ سࣹينْ بِيرْ؞ ‏16 ݧُ نࣹ:‏ «لُ نُيْ دࣹفْ اَڪْ ݧِي؟ ندَخْ ڪَتْ فِرْندࣹ لُ لࣹيرْ سࣷتّࣹي نَ فِ ݖِ ݧࣷومْ؛ لࣹيرْ نَ نࣵݧّ وَا يࣹرُسَلࣹمْ گࣺݒّ، تࣹ مَنُنُ ڪࣷو وࣹدِّ؞ ‏17 وَايࣹ ڪَتْ بُ مبِرْ مِ گࣴنَ سِيوْ ݖِ خࣹيتْ وِ؞ نَنْ لࣹينْ تࣴڪُّ، نگِرْ آيࣹ لࣹينْ ݧُ وَخَتِ ڪࣹنّ تُرْ وࣷووُ؞»
‏18 ݧُ وࣷو لࣹينْ نَگْ، آيࣹ لࣹينْ، نࣹ لࣹينْ مَيَتُݧُ لࣹينْ بࣹنّ يࣷونْ ݧُ وَخَتِ لࣹنّ، مبَا ݧُيْ جࣵنگَلࣹ ݖِ تُرُ عِيسَىٰ؞ ‏19 ݒِيࣹيرْ اَڪْ يࣷوَانْ نࣹ لࣹينْ:‏ «ندَخْ لِ جَادُ فَ يࣵلَّ مࣷويْ نُ دࣺگَّلْ لࣹينْ، يࣹينْ، بࣵيِّ يࣵلَّ؟ يࣹينَيْ اࣵتّࣹ لࣷولُ؞ ‏20 وَايࣹ نُنْ، لِ نُ گِسْ، دࣺگّ ڪࣷ، مَنُنُ ڪࣷو بَݧَ وَخْ؞» ‏21 بَ لࣷولُ اَمࣹي ݧُ تࣴڪُّوَاتْ لࣹينْ، دࣷورْ لࣹينَ يِوِ، ݧُ دࣹمْ، گَنَّاوْ مَنُݧُ لࣹينَ مبُگَلْ تࣹ اَمُݧُ بُنتّ ݖِ ݧࣷومْ ندَخْ مبࣷولࣷو مَ، تࣹ ݧࣺݒّ دِ مࣵگَّلْ يࣵلَّ ݖَ لَ خࣹوْ، ‏22 نگِرْ وَا، جَ ڪࣺيمْتَانْ گَ وࣺرْلࣷو اَمࣷونْ نَ لُ وࣹيسْ ݧࣹينتّ فُڪِّ اَتْ؞
مبࣷولࣷومْ گࣴمْڪَتْ ݧِ ݧَانْ نَ ݖِ يࣵلَّ
‏23 گَنَّاوْ بَ ݧُ يِوࣹي ݒِيࣹيرْ اَڪْ يࣷوَانْ، ݧُ دࣹمْ ݖَ سࣹينْ بࣷڪِّ گࣴمْڪَتْ، نࣹتَّلِ لࣹينْ مبࣷولࣹمْ لَ لࣹينْ سَرْخَلْڪَتْ يُ مَگْ يَاڪْ مَگْ ݧَ وَخْ؞ ‏24 بَ ڪࣷ بࣷڪّ يَ دࣺگّࣹي، ݧࣷومْ ݧࣺݒَّ مࣵنڪࣷو، دَلْدِ يࣺڪَّتِ ڪࣵدُّگْ ݧَانْ فَ يࣵلَّ، نࣹ:‏ «بُورْ يࣵلَّ، يَا سࣵڪّ اَسَمَانْ اَڪْ سُوفْ اَڪْ گࣺيجْ اَڪْ لࣺݒّ لِ ݖِ بِيرْ؞ ‏25 يَا وَخࣹ نࣷو گُ سࣹلّ گِ ݖِ سُنُ گࣺمِّݧُ مَامْ دَاوُدَ سَ جَامْ بِ، نگَ نࣹ:‏
‹لُ خࣹيتِ اࣵدِّنَ دِ رِيرْ؟
لُ خࣹيتْ يِيْ لَلْ ݒࣹخࣹيْ نࣹينْ؟
‏26 بُورِ اࣵدِّنَا جٗگَنْدࣷو،
ڪِلِفَ يࣹي لࣴڪّࣷو ݖِ ڪَوْ بࣷرࣷومْ بِ اَڪْ اَلْمَسࣹيمْ؞›
‏27 دࣴگّ لَ مࣷوسْ، ندَخْ ݖِ دࣴڪّ بِي لَ اࣹرࣷدّ اَڪْ ݒࣷݧْسࣴ ݒِلاَتْ لࣴڪّࣷو، اࣵندّ اَڪْ خࣹيتِ اࣵدِّنَاڪْ بࣵنِّ إِسْرَاءِيلْ ݖِ ڪَوْ عِيسَىٰ، سَ ندَوْ لُ سࣹلّ لِ نگَ فَلْ؞ ‏28 نࣷونُ اِتْ لَݧُ سࣷتَّلࣹ مبࣷولࣹمْ لِ نگَ دࣷگَلࣹ وࣷونْ سَ لࣷخࣷ اَڪْ سَ ݖࣷوبَرࣹ؞ ‏29 لࣺيگِ نَگْ بࣷرࣷومْ بِ، بࣵيِّلْ خࣹلْ سࣹينِ تࣴڪّࣷو تࣹ نگَ مَيْ نُ، نُنْ سَيْ جَامْ، نُيْ وَخࣹ سَ ڪࣵدُّ فِتْ وُ مَتْ سࣴڪّ؞ ‏30 نگَلَّ تࣵلَّلَلْ سَ لࣷخࣷ نگِرْ دِ وࣺرَلْ اَڪَ اَمَلْ اَيْ فِرْندࣹ اَڪِ ڪࣺيمَانْ ݖِ تُرُ عِيسَىٰ، سَ ندَوْ لُ سࣹلّ لِ؞»
‏31 بَ ݧُ ݧَانࣹي بَ نࣷݒِّ، بࣺرَبْ بَ ݧُ دَجࣹ دَا يࣴنگُ، ݧُ فࣹيسࣹ نࣷو گُ سࣹلّ گِ ݧࣷومْ ݧࣺݒّ؛ ݧُ دَلْدِ تࣵمْبَلࣹي وَخࣹ فِتْ ڪࣵدُّگْ يࣵلَّ گِ؞
گࣴمْڪَتْ ݧِ ݧࣷو بࣷڪّࣷونْ لࣺݒّ
‏32 ݖِ ڪَوْ لࣷولُ مبࣷولࣹمْ گࣴمْڪَتْ ݧِ بࣷڪّ مࣹنّ خࣹلْ، اَڪْ بࣹنّ خَلاَتْ؞ دُ ڪࣹنّ ڪُ نࣹ مࣷو مࣷومْ لࣹنّ ݖِ اَلَلَمْ، وَايࣹ ݧࣷو بࣷڪّࣷونْ لࣺݒّ ݖِ سࣹينْ بِيرْ؞ ‏33 ندَوْ يِ نَگْ دِ سࣹيدࣹي دࣷولࣹ جُ رࣺيْ نࣹ سَنگّ بِ عِيسَىٰ دࣹڪِّ نَ، يِوْ وُ يَا دِ لࣹينْ دِڪَّلْ ݧࣷومْ ݧࣺݒّ؞ ‏34 ڪࣹنّ ݖِ ݧࣷومْ ندٗولُلْ، ندَخْ ڪࣺݒّ ڪُ ݖِيْ بࣷرࣷومْ سُوفْ مبَا اَيْ ڪࣴرْ، دَ ڪࣷيْ جَايْ، اِنْدِ نجࣺگْ لِ، ‏35 تࣹگْ ڪࣷ ݖِ ڪَنَمْ ندَوْ يِ؞ سُ ڪࣷ دࣹفࣹي ڪُ نࣹڪّ سࣺدُّ ݖَ لَ دࣴݒّࣷوڪْ سࣷخْلاَمْ؞
‏36 ڪُ مࣹلْ نِ يُوسُفَ، بࣷڪّ ݖِ گِيرُگْ لࣹوِ، تࣹ دࣴڪّࣹ دُنُ سِݒَّرْ، ندَوْ يِ دَانْ ڪࣷ وَخْ بَرْنَبَ، مُ فِرِ ݧَاخࣹڪَتْ بِ، ‏37 اَبْ تࣷولْ لَ اَمࣷونْ؞ دَ ڪࣷو جَايْ، اِنْدِ خَالِسْ بِ، تࣹگْ فِ ڪَنَمْ ندَوْ يِ؞
