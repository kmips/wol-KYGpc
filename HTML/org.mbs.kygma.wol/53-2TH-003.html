ݧَانْلࣹينْ ݖِ يࣵلَّ
3
‏1 لِ ݖِ دࣹسْ نَگْ بࣷڪّ يِ، مࣷو دِ نگࣹينْ ݧَانَلْ نُ، نگِرْ ڪࣵدُّگْ بࣷرࣷومْ بِ گَاوَ لَوْ تࣹ اَمْ ندَمْ نَ مُ ڪࣷ اَمࣹ وࣷونْ فَ يࣹينْ، ‏2 نگِرْ اِتَمْ نُ مُݖّ ݖِ نِتْ ݧُ سࣷخࣷرْ ݧُ بࣷنْ ݧِ، ندَخْ دُ ݧࣺݒَّ گࣴمْ؞ ‏3 وَايࣹ بࣷرࣷومْ بِ مࣷويْ ڪِ جَرَ وٗولُ، تࣹ مࣷو لࣹينْ دِ دࣷولࣹيلْ، مُسَلْ لࣹينْ ݖِ ڪُ بࣷنْ ڪِ؞ ‏4 اَمْ نَنُ اِتْ ڪٗولُتࣹ گُ نُ بࣷرࣷومْ بِ مَيْ، نࣹ لِ نُ لࣹينْ سَنتّ، مࣷومْ نگࣹينْ نࣹڪّࣹ، تࣹ مࣷومْ نگࣹينْ دِ وࣺيࣹ؞ ‏5 يَلْ نَ بࣷرࣷومْ بِ وࣷمَّتْ سࣹينْ خࣷلْ، جࣴمࣹ ݖِ ݖࣷفࣹيلُ يࣵلَّ اَڪْ مُݧْ گِ ݖِ اَلْمَسِ؞
يَافُسْ وَرُلْ
‏6 ندِگَلْ لِ نُ لࣹينْ دِ جࣷخْ ݖِ سُنُ تُرُ بࣷرࣷومْ عِيسَىٰ اَلْمَسِ، مࣷو دِ نگࣹينْ دࣵندّ مࣺݒّ مبࣷڪّ مُ نࣹڪّࣹ اَگْ يَافُسْ، تࣹ بَݧَ دࣴݒّࣷو اَڪْ دࣺنڪَانࣹ يِ نگࣹينْ نَنگࣷو ݖِ نُنْ؞ ‏7 يࣹينْ ݖِ سࣹينْ بࣷݒّ، خَمْ نگࣹينْ نࣹ ݖِ سُنُيْ تࣵنڪّ نگࣹينْ وَرَ اَوࣹ، تࣹ دُ اَگْ يَافُسْ لَنُ نࣹڪّࣹ وࣷونْ ݖِ سࣹينْ بِيرْ؞ ‏8 لࣹڪُّنُ ݧَمُ ڪࣹنّ ݖِ نࣹينْ، خَنَا دِ يࣴنگُ گُدِّ اَڪْ بࣴݖّࣴگْ، ݖِ ݖࣷونࣷبْ دَانْ سُنُ دࣷولࣹ، نگِرْ بَݧَ دِيسَلْ ڪࣹنّ ݖِ يࣹينْ؞ ‏9 مࣷونَ دُ سَݧْ-سَݧْ بُ نُ ݖِ ݧࣵڪّ، خَنَا بࣴگّ لࣹينَ وࣷنْ لُ نگࣹينْ نُ رࣷيࣹ، بَ نگࣹينْ اَوْ ݖِ سُنُيْ تࣵنڪّ، ‏10 ندَخْ بَ نُ نࣹڪّࣹي اَڪْ يࣹينْ، ندِگَلْ لَنُ لࣹينْ جࣷخࣷونْ مࣷو دِ لِي:‏ «ڪُ نَنگُوُلَ لِگّࣺييْ، بُمُ لࣹڪّ؞»
‏11 دࣺگّ نَنُ نَگْ نࣹ ݧࣹنَّ نگِ ݖِ سࣹينْ بِيرْ، ݧُ نࣹڪّࣹ اَگْ يَافُسْ، دُݧُ يࣴنگُ ݖِ دَرَ، لُ مࣷيْ دِ تࣷݒّتࣷو لُ سࣹينْ يࣷونْ نࣹڪُّلْ؞ ‏12 ݧُ نِ مࣹلْ لَنُيْ سَنتّ اَڪَ دࣺنڪُ نگِرْ سَنگّ عِيسَىٰ اَلْمَسِ، نگِرْ ݧُ دَلْ، تࣹيْ يࣴنگُ، دِ دُنْدࣹ سࣹينْ ݧَقُ بࣷݒّ؞ ‏13 يࣹينْ نَگْ بࣷڪّ يِ، بُلࣹينْ تࣵيِّ ݖِ دࣹفْ لُ بَاخْ؞
‏14 ڪُ ݖِ دࣺگُّلْ سُنُ ڪࣵدُّ يِ ݧُ بِندّ ݖِ بَتَاخَلْ بِي نَگْ، بࣵيِّلࣹينْ ڪࣷ خࣹلْ، تࣹ بَݧَ جَخَسࣷو اَڪْ مࣷومْ، نگِرْ گࣵݖّࣹيلْ ڪࣷ؞ ‏15 بُمُ تَخْ نگࣹينْ دࣹفْ ڪࣷ اَبْ نࣷونْ، وَايࣹ يࣹدّلࣹينْ ڪࣷ، ندَخْ مبࣷڪّ لَ؞
فِي لَ ݒٗولْ تࣴجࣹ بَتَاخَلَمْ
‏16 گَنَّاوْ لࣷولُ، بࣷرࣷومْ بِ مࣷومْ جࣵمّ، يَلْ نَ لࣹينْ مَيَلْ بࣷݒَّمْ جࣵمِّ لࣺݒّ اَڪْ فࣺݒّ؞ يَلْ نَ بࣷرࣷومْ بِ اࣵندّ اَڪْ يࣹينْ ݧࣺݒّ؞
‏17 تࣵگّࣷو بِي، مَنْ ݒٗولْ مَا ڪࣷ بِندّ ݖِ سَمَ لࣷخࣷلْ بࣷݒّ، تࣹ نِي لاَيْ بِنْدࣹ ݖِ بࣺݒّ بَتَاخَلْ؞ ‏18 سُنُ يِوُ بࣷرࣷومْ عِيسَىٰ اَلْمَسِ ݧࣹيلْ نَ لࣹينْ يࣹينْ ݧࣺݒّ؞
