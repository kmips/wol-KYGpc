لِي لَݧُ سࣺينْتُ ݖِ نجِيتُ مبࣷولࣷو مِ
3
‏1 وَخْ جُ وٗورَ نگِي:‏ ڪُيْ سࣵڪُّ سَسُبْ جِيتࣹ مبࣷولࣷومْ گࣴمْڪَتْ ݧِ، لِگّࣺييْ بُ رَفࣹتْ لَيْ خࣴݖُّ؞ ‏2 نجِيتْ لِ نَگْ فࣵوّ مُ مُݖّ سِڪّ، دِ ڪُ دࣹنݖُلْ لُ وࣹيسْ جࣹنّ جَبَرْ، ڪُ مࣷومْ بَڪَّنَمْ، جِيتَلْ اَمْ خࣹلَمْ، دِ ڪُ نَوْلُ، مَنْ گَنْ، تࣹ مَنَ جࣵنگّلࣹ؞ ‏3 بُمُ دِ اَبْ مࣵنْدِڪَتْ، مبَا بࣴگّڪَتُبْ خࣹيخْ، وَايࣹ نَ لࣹوࣹتْ، بَݧُ اَيْ، تࣹ بُمُ دِ بࣴگّڪَتُبْ خَالِسْ؞ ‏4 ڪࣴرُگْ بࣷݒَّمْ نَگْ، نَ ڪࣷ يࣷرْ يࣷرِنْ وُ رَفࣹتْ، تࣹ نَ مَنَ يَرْ اَيْ دࣷومَمْ بَ ݧُ نَنگُلْ ڪࣷ تࣹ سࣺدّ ڪࣷ ݖࣺرْ بُ مَتْ سࣴڪّ؞ ‏5 ندَخْ ڪُ تࣴلࣹي يࣷرْ ڪࣴرُگْ بࣷݒَّمْ، اَنَ نُ مُ مَنَ سࣵمّࣹ مبࣷولࣷومْ گࣴمْڪَتِ يࣵلَّ يِ؟ ‏6 تࣹ اِتْ بُمُ دِبْ تُوبࣹينْ بُ بࣹيسْ، لُ ڪࣷ مࣷيْ اَگْ رࣺيْ-رࣺيْلُ مَنْ نَ ڪࣷو گࣴلࣴمَلْ، بَ مُ تࣵبِّ ݖِ مبُگَلْ نِ سࣹيْتَانࣹ؞ ‏7 تࣹ اِتْ ݧِ ݖِ بِتِ تࣹ گࣴمُݧُ، لَ ݧُ ڪࣷ سࣹيدࣹيلْ نَيْ لُ رَفࣹتْ، بَ دࣹرَمْ دُ يࣵقُ، ندࣹ ڪࣷنْ مُ تࣵبِّ ݖِ فِيرُ سࣹيْتَانࣹ؞
لِي لَݧُ سࣺينْتُ ݖِ تَخَوُڪَتُبْ مبࣷولࣷو مِ
‏8 نَڪَ نࣷونُ اَبْ تَخَوُڪَتُبْ مبࣷولࣷو مِ، نَ دِ ڪُ تࣹدّ، بُمُ دِ بࣷرࣷومْ ݧَارِ ڪࣵدُّ؞ بُمُ دِ ڪُ سࣵنگَرَ جِيتَلْ، مبَا ڪُيْ سࣵڪُّ اَمْ-اَمْ بُ لࣹوُلْ؞ ‏9 نَ ݝࣷيْ ݖِ مبٗوتُمْ نگࣴمْ مِي فࣹيݧْ تࣹ اࣵندّ ݖࣹيڪْ خࣹلْ مُ دَلْ؞ ‏10 دࣹيسْ ڪࣷيْ جࣴڪَّ سࣹيتْلُ نَگْ، بَ مُيْ ڪُ مُݖّ سِڪّ، ݧُ دࣷورْ ڪࣷو دُگَلْ ݖِ لِگّࣺييُ تَخَوُڪَتُبْ مبࣷولࣷو مِ؞ ‏11 نَڪَ نࣷونُ، نَ جِگࣺينْ ݧِيْ تَخَوُ مبࣷولࣷو مِ¤ دِ ݧُ تࣹدّ، بُݧُ دِيْ سࣷسْڪَتْ، نَݧُ مࣷومْ سࣹينْ بَڪَّنْ، تࣹ جَرَ وٗولُ ݖِ لࣺݒّ؞
‏12 اَبْ تَخَوُڪَتُ مبࣷولࣷو مِ بُمُ دࣹنݖّ لُ وࣹيسْ جࣹنّ جَبَرْ تࣹ نَ يࣷرْ اَيْ دࣷومَمْ، اَڪْ ڪࣴرَمْ، يࣷرِنْ وُ رَفࣹتْ؞ ‏13 ݧِ ݖِ وࣵݖّࣷو اَڪْ سࣹينْ سَسُبْ تَخَوُڪَتْ نَگْ، سࣹينُبْ لِگّࣺييْ لَݧُ جࣴنْدࣹ تَخَوَايْ بُ رَفࣹتْ، تࣹ ݧࣷويْ مَنَ وَخࣹ فِتْ، سࣹينْ نگࣴمْ گِ ݧُ گࣴمْ اَلْمَسِ عِيسَىٰ؞
مبٗوتُمْ يࣷونُ نگࣴمْ وِ مࣵگّ نَ
‏14 لِي لاَ لَيْ بِندّ، تࣹ دِ لَ يَاڪَارَ سࣹيتْسِ ݖِ فَنْ يُ گࣵتّ، ‏15 نگِرْ سُ مَ يࣹيخࣹي اِتَمْ، نگَ خَمْ نࣷويْ دࣷخَلࣹ ݖِ ڪࣴرْ يࣵلَّ، ݖِ بِيرْ مبࣷولࣷومْ يࣵلَّ جِيْ دُندّ؞ مبࣷولࣷومْ گࣴمْڪَتْ مِ مࣷويْ ڪࣹنُگْ دࣴگّ گِ، دِ ݖࣴسْلاَيَمْ؞ ‏16 دࣹيسُلْ نࣵتَّبْلࣹ مࣷوسْ مࣵگَّايُ مبٗوتْ مِي فࣹيݧࣹ ݖِ سُنُ يࣷونُ نگࣴمْ:‏
ڪِي مࣷو فࣹيݧࣹ ݖِ يَرَمُ سُوخْ،
نࣷووُگْ يࣵلَّ دࣴگَّلْ ڪࣷ،
مَلاَڪَ يَ نِيرْ ڪࣷ؛
مࣷومْ لَݧُ وَارࣹي ݖِ خࣹيتْ يِ،
گࣴمࣹيسْ نَ ڪࣷ ݖِ ڪَوْ سُوفْ اِتْ،
تࣹ يࣺيگࣹيسْ نَ ڪࣷ فَ لࣹيرْ گَ؞
