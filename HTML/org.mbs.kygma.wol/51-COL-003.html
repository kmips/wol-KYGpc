گࣴمْڪَتُبْ اَلْمَسِ، دُنْدِنْ وُ يࣹيسْ
3
‏1 گَنَّاوْ نَگْ يࣹينَ دࣹڪّࣹينْدࣷو اَڪْ اَلْمَسِ، سࣵڪُّلࣹينْ لَ فَ ڪَوْ، فَ اَلْمَسِ تࣷوگࣹ ݖَ ندِجࣷورُ يࣵلَّ؞ ‏2 لَ فَ ڪَوْ، خِنْتࣹوࣷولࣹينْ ڪࣷ، تࣹ نگࣹينْ بَ لِ فِ سُوفْ، ‏3 ندَخْتࣹ دࣹي نگࣹينْ، بَ سࣹينُگْ دُندّ اࣵندّ اَڪْ اَلْمَسِ، لࣵقࣹ فَ يࣵلَّ؞ ‏4 سࣹينُگْ دُندّ دࣴگّ اَلْمَسِ لَ، تࣹ ڪࣹرࣷوگْ بُ اَلْمَسِ فࣹيݧࣹي، يࣹينْ اِتَمْ اَيْ اࣵندّ اَڪْ مࣷومْ فࣹيݧْ، بَ سࣺدُّ ݖِ ندَمَمْ؞
‏5 ڪࣷنْ نَگْ رࣹيْلࣹينْ سࣹينْ ݖࣺرْ يِ بࣷڪّ فِي ݖِ ڪَوْ سُوفْ؛ لُ مࣹلْ نِ ݒࣷوُمْ سࣺيْ، اَڪْ جࣴفِ سࣷبࣹ، اَڪْ ݧࣵڪّ سَگࣷ ݖِ لُ جࣹڪَّدِ، اَڪْ خࣹمّࣹمْتࣺيفْ يُ بࣷنْ، اَڪْ بࣴگّࣹ، تࣹ لࣷولࣷويْ بࣷڪَّالࣹ بَ تࣹيْ، ‏6 يࣷويࣷويْ وࣵݖّࣹ سࣵنجُمْ يࣵلَّ ݖِ ڪَوْ ݧِ دࣺگَّدِ؞ ‏7 يࣹينْ اِتَمْ نࣷونُ نگࣹينْ دَانْ جࣴفࣹ بُ جࣴڪّ، بَ سࣹينُگْ دُندّ بࣷڪّࣹي فِي ݖِ ڪَوْ سُوفْ؞ ‏8 وَايࣹ لࣺيگِ لِي لࣺݒّ بَلࣹينْ ڪࣷ:‏ تࣵنگّ خࣷلْ، اَڪْ مࣹرْ مُ اࣴݒّ، اَڪْ ݖࣷخࣷرْ، اَڪْ سَاگَ، اَڪْ وَخِ ݧࣵڪّ ڪࣹرْسَ؞ ‏9 بُلࣹينْ دِ فࣹنَنْتࣹ، گَنَّاوْ سࣹينْ جࣴمّ جَ وࣷونْ دࣺمبّ، بࣷولࣹ نگࣹينْ ڪࣷوڪْ جࣴفِنَمْ جَ مُ اࣵنْدَلْ، سُمِّ، ‏10 بَ سࣷلْ سࣹينْ جࣴمّ جُ يࣹيسْ، جِيْ گࣴنْ دِ يࣹيسْلُ، تࣹ دِ گࣴنَ نِرُ ڪِ لࣹينْ ڪࣷ سࣵڪَّلْ، بَ ڪࣹرَ نگࣹينْ خَمࣹي، خَمْ-خَمْ بُ مَتْ سࣴڪّ؞ ‏11 دُنْدِنْ وُ يࣹيسْ وࣷووُ اَمُلْ جَامْبُرْ اَڪُبْ يَوُتْ، اَمُلْ ڪُ خَرَفْ اَڪْ ڪُ خَرَفُلْ، اَمُلْ اَبْ دࣷخَنْدࣺيمْ اَڪْ ڪُ خَيَدِ، اَمُلْ اَبْ جَامْ اَڪْ اَسْ گࣷرْ، وَايࣹ اَلْمَسِ مࣷويْ لࣺݒّ تࣹ مࣷو نࣹڪّ ݖِ ݧࣺݒّ؞
‏12 يࣹينَيْ ݧِ يࣵلَّ تَامُ، سࣹلَّلْ لࣹينْ، سࣷݒّ لࣹينْ؞ ڪࣷنْ نَگْ سࣷلࣷولࣹينْ خࣷلُ يࣴرْمَنْدࣹ، اَڪُگْ مبَاخْ، اَڪْ وࣷيࣷفْلُ، اَڪْ جِڪّࣷ جُ لࣹوࣹتْ، اَڪْ خࣷلُ مُݧْ؞ ‏13 مُݧَلَنْتࣹلࣹينْ تࣹ دِ بَلَنْتࣹ، ندࣹگَمْ ڪࣹنَّ اَمْ لࣹنّ لُ مُ سِڪّࣹ مࣷرࣷومَمْ؞ نِ لࣹينْ بࣷرࣷومْ بِ بَالࣹ، دࣹيلࣹينْ بَلَنْتࣹي نࣷونُ؞ ‏14 ݖِ ڪَوْ لࣷولُ لࣺݒّ نَگْ، تࣹگࣷولࣹينْ ݖࣷفࣹيلْ؛ مࣷويْ خࣵنݖّ گِ لࣹينْ فَسْتࣹ، بَ سࣹينُگْ بࣹنّࣷو مَتْ سࣴڪّ؞
‏15 نَ جࣵمُّ اَلْمَسِ سَيْتُ سࣹينْ خࣷلْ؞ ݖِ جࣷوجُ جࣵمّ لَݧُ لࣹينْ وࣷو، فࣹڪّ لࣹينْ نگࣹينْ دِ وࣹنّ يَرَمْ دࣷݝّ، تࣹ اِتْ سَخࣷولࣹينْ ݖَنتّ؞ ‏16 نَ لࣹينْ ڪࣵدُّگْ اَلْمَسِ دࣴڪّࣹ، بَيْ بَاوَانْ، نگࣹينْ دِ جَرِݧࣷو خࣹلْ مُ مَتْ سࣴڪّ مُ نگࣹينْ دِ جࣵنگّلَنْتࣹي، اَڪَ اَرْتࣷونْتࣹ؞ وࣷيࣹلࣹينْ خࣷلُبْ ݖَنتّ، ݧࣹيلْ يࣵلَّ، اَيْ تَالِفِ سَبٗورْ، اَڪْ وࣷيِ تَگّ يࣵلَّ، اَڪْ وࣷيْ يُ فࣹنْتࣷو ݖِ نࣷو گِ؞ ‏17 لُ نگࣹينْ مَنَ دࣹفْ، مُيْ ݖِ وَخْ، دِ ݖِ جࣴفْ، دࣹفْلࣹينْ ڪࣷ ݖِ تُرُ سَنگّ عِيسَىٰ، تࣹيْ سَنتّ يࣵلَّ بَايْ بِ ݖِ مࣷومْ؞
رَفࣹتࣷولࣹينْ
‏18 جِگࣺينْ ݧِ، نَنگُلْ-لࣹينْ سࣹينْ جࣴڪّࣴرْ، نࣷونُ مُ جࣹڪّࣹ ݖِ ڪُ بࣷڪّ ݖِ بࣷرࣷومْ بِ؞
‏19 گٗورْ ݧِ، سࣷݒّلࣹينْ سࣹينْ جَبَرْ، تࣹ بُلࣹينْ لࣹينْ سࣷخࣷرࣹ؞
‏20 خَلࣹ يِ، دࣺگَّلْ-لࣹينْ سࣹينِ وَاجُرْ ݖِ لࣺݒّ، ندࣹ لࣷولُ مࣷويْ لِ بࣷرࣷومْ بِ رَفࣹتْلُ؞
‏21 يࣹينْ ݧِيْ بَايْ، بُلࣹينْ يࣺڪَّتِ سࣹينْ خࣷلُ دࣷومْ، نگِرْ بَݧَ جࣹيخَلْ سࣹينْ خࣷلْ؞
‏22 جَامْ يِ، نَنگُلْ-لࣹينْ ݖِ لࣺݒّ سࣹينْ سَنگّ يِ ݖِ اࣵدِّنَ؞ بُلࣹينْ ڪࣷ دࣹفࣹ نگِسْتَلْ، مبَا نگࣹينْ دِ ڪࣷ سࣵڪّࣷو لُ لࣹينْ نࣹيخْ، وَايࣹ دࣹفࣹلࣹينْ ڪࣷ يࣺينࣹيْ خࣷلْ بُ دࣴگُّ، ندَخْ رَگَلْ بࣷرࣷومْ بِ؞ ‏23 لُ نگࣹينْ مَنَ دࣹفْ، لِگّࣺييࣹلࣹينْ ڪࣷ خࣷلْ بُ تࣵلِّ، نِ بُ ݧࣹيلُلْ وࣷونْ نِتْ دࣷݝّ، خَنَا بࣷرࣷومْ بِ؞ ‏24 تࣹ نگࣹينْ خَمْ نࣹ ݖِ لࣷخࣷلْ بࣷرࣷومْ بِ نگࣹينْ دِ نَنگࣷو يࣷولْ بِ نگࣹينْ مُورْلࣷو؞ سَنگّ اَلْمَسِ مࣷويْ ڪِ نگࣹينْ دِ جَامُ؞ ‏25 وَايࣹ ڪُ دࣹفْ نجُبَدِ، نجُبَدِ گَ مُ دࣹفْ لَيْ يࣷولࣷو، تࣹ خࣹجّ اَڪْ سࣹينْ دُ ݖِ اَمْ؞
